 743      {
 744          global $current_User, $blog, $pagenow;
 745  
 746          $max_buttons = 7;
 747  
 748          if( empty( $this->coll_list_permname ) )

 766          $blog_array = $BlogCache->load_user_blogs( $this->coll_list_permname, $this->coll_list_permlevel );
 767  
 768  
 769          $buttons = '';
 770          $select_options = '';
 771          $not_favorite_blogs = false;
 772          foreach( $blog_array as $l_blog_ID )
 773          { // Loop through all blogs that match the requested permission:
 774  
 775              $l_Blog = & $BlogCache->get_by_ID( $l_blog_ID );
 776  
 777              if( $l_Blog->get( 'favorite' ) || $l_blog_ID == $blog )
 778              { // If blog is favorute OR current blog, Add blog as a button:
 779                  $buttons .= $template[ $l_blog_ID == $blog ? 'beforeEachSel' : 'beforeEach' ];
 780  
 781                  $buttons .= '<a href="'.$url_params.'blog='.$l_blog_ID
 782                              .'" class="'.( $l_blog_ID == $blog ? 'CurrentBlog' : 'OtherBlog' ).'"';

 784                  if( !is_null($this->coll_list_onclick) )
 785                  {    // We want to include an onclick attribute:
 786                      $buttons .= ' onclick="'.sprintf( $this->coll_list_onclick, $l_blog_ID ).'"';
 787                  }
 788  
 789                  $buttons .= '>'.$l_Blog->dget( 'shortname', 'htmlbody' ).'</a> ';
 790  
 791                  if( $l_blog_ID == $blog )
 792                  {
 793                      $buttons .= $template['afterEachSel'];
 794                  }
 795                  else
 796                  {
 797                      $buttons .= $template['afterEach'];
 798                  }
 799              }
 800  
 801              if( !$l_Blog->get( 'favorite' ) )
 802              { // If blog is not favorute, Add it into the select list:
 803                  $not_favorite_blogs = true;
 804                  $select_options .= '<option value="'.$l_blog_ID.'"';
 805                  if( $l_blog_ID == $blog )
 806                  {
 807                      $select_options .= ' selected="selected"';
 808                  }
 809                  $select_options .= '>'.$l_Blog->dget( 'shortname', 'formvalue' ).'</option>';
 810              }
 811          }
 812  
 813          $r = $template['before'];
 814  
 815          $r .= $title;
 816  
 817          if( !empty( $this->coll_list_all_title ) )
 818          { // We want to add an "all" button
 819              $r .= $template[ $blog == 0 ? 'beforeEachSel' : 'beforeEach' ];
 820              $r .= '<a href="'.$this->coll_list_all_url
 821                          .'" class="'.( $blog == 0 ? 'CurrentBlog' : 'OtherBlog' ).'">'
 822                          .$this->coll_list_all_title.'</a> ';
 823              $r .= $template[ $blog == 0 ? 'afterEachSel' : 'afterEach' ];

 825  
 826          $r .= $template['buttons_start'];
 827          $r .= $buttons;
 828          $r .= $template['buttons_end'];
 829  
 830  
 831          $r .= $template['select_start'];
 832          if( $not_favorite_blogs )
 833          { // Display select list with not favorite blogs
 834              $r .= '<form action="'.$pagenow.'" method="get">';
 835              $r .= $form_params;
 836              $r .= '<select name="blog" onchange="';
 837              if( empty( $this->coll_list_onclick ) )
 838              { // Just submit...
 839                  $r .= 'if(this.value>0) this.form.submit();';
 840              }
 841              else
 842              {
 843                  $r .= sprintf( $this->coll_list_onclick, 'this.value' );
 844              }
 845              $r .= '">'
 846                  .'<option value="0">'.T_('Select blog').'</option>'
 847                  .$select_options.'</select>';
 848              $r .= '<noscript><input type="submit" value="Go" /></noscript></form>';
 849          }
 850          $r .= $template['select_end'];
 851  
 852  
 853          $r .= $template['after'];
 854  
 855          return $r;
 856      }

1308      /**
1309       * Set a specific path level (specific depth).
1310       *
1311       * First level is 0, then the first subpath/submenu is level 1, etc.
1312       *
1313       * E.g., if plugins.php gets called, there could be a call to
1314       * $AdminUI->set_path_level( 0, 'plugins' ), which selects this entry from the menu.
1315       * If a specific tab is called inside of plugins.php, there could be a call to
1316       * $AdminUI->set_path_level( 1, $tab )
1317       *
1318       * Though, it is recommended to call the wrapper functions:
1319       *  - {@link append_path_level()}
1320       *  - {@link set_path()}
1321       *
1322       * This also marks the parent node as selected and checks for permissions.
1323       *
1324       * @param integer Path level to set (starts at 0)
1325       * @param array Either the key of the path or an array(keyname, propsArray).
1326       * @param array Properties for this path entry.
1327       * @return boolean DEPRECATED True if perm granted, false if not (and we're not exiting).
1328       */
1329  	function set_path_level( $level, $pathKey, $pathProps = array() )
1330      {
1331          // Get the parent node (the level above this one):
1332          if( $level == 0 )
1333          { // first level in menu-path: parent node is NULL
1334              $parentNode = & $this->get_node_by_path( NULL );
1335          }
1336          else
1337          { // parent node is the trunk from root to previous level
1338              $parentNode = & $this->get_node_by_path( $this->get_path_range( 0, $level-1 ) );
1339          }
1340          if( ! $parentNode )
1341          { // parent node does not exist:
1342              return false;
1343          }
1344  
1345          // Store the selected entry name in the parent node:
1346          $parentNode['selected'] = $pathKey;
1347  
1348          $this->path[$level] = $pathKey;
1349          $this->pathProps[$level] = $pathProps;
1350  
1351          // pre_dump( 'set_path_level: ', $level, $pathKey, $this->pathProps[$level] );
1352  
1353          return true;
