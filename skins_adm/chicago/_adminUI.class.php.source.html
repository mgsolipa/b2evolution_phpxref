 428  
 429          $user_admin_skin = $UserSettings->get( 'admin_skin', $user_ID );
 430          if( $UserSettings->get( 'admin_skin', $current_User->ID ) == $user_admin_skin )
 431          {
 432              $Form->checklist( array(
 433                          array( 'show_evobar', 1, T_('Show evobar'), $UserSettings->get( 'show_evobar', $user_ID ) ),
 434                          array( 'show_breadcrumbs', 1, T_('Show breadcrumbs path'), $UserSettings->get( 'show_breadcrumbs', $user_ID ) ),
 435                          array( 'show_menu', 1, T_('Show Menu'), $UserSettings->get( 'show_menu', $user_ID ) ) ),
 436                      'chicago_settings', T_('Chicago skin settings') );
 437          }
 438          else
 439          {
 440              $Form->info( '', sprintf( T_( 'Admin skin settings for this user cannot be edited because this user is using a different admin skin (%s)' ), $user_admin_skin ) );
 441          }
 442  
 443          $Form->end_fieldset();
 444  
 445          // JavaScript code to dynamically change display settings. show_evobar or show_menu always have to be checked
 446          ?>
 447          <script type="text/javascript">
 448          jQuery( '[name = show_evobar], [name = show_menu]' ).click( function()
 449          {
 450              if( ! ( jQuery( '[name = show_evobar]' ).attr( 'checked' ) || jQuery( '[name = show_menu]' ).attr( 'checked' ) ) )
