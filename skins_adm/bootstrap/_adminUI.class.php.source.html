  39       * This function should init the templates - like adding Javascript through the {@link add_headline()} method.
  40       */
  41  	function init_templates()
  42      {
  43          global $Messages, $debug, $Hit, $check_browser_version;
  44  
  45          // This is included before controller specifc require_css() calls:
  46          require_css( 'results.css', 'rsc_url' ); // Results/tables styles
  47  
  48          require_js( '#jquery#', 'rsc_url' );
  49          require_js( 'jquery/jquery.raty.min.js', 'rsc_url' );
  50  
  51          require_js( '#bootstrap#', 'rsc_url' );
  52          require_css( '#bootstrap_css#', 'rsc_url' );
  53          // require_css( '#bootstrap_theme_css#', 'rsc_url' );
  54          require_js( '#bootstrap_typeahead#', 'rsc_url' );
  55  
  56          // JS to init Bootstrap tooltips (E.g. on badges with title "Admin"):
  57          add_js_headline( 'jQuery( function () { jQuery( \'[data-toggle="tooltip"]\' ).tooltip() } )' );
  58  
  59          if( $debug )
  60          {    // Use readable CSS:
  61              // rsc/less/bootstrap-basic_styles.less
  62              // rsc/less/bootstrap-basic.less
  63              // rsc/less/bootstrap-evoskins.less
  64              require_css( 'bootstrap-backoffice-b2evo_base.bundle.css', 'rsc_url' ); // Concatenation of the above
  65          }
  66          else
  67          {    // Use minified CSS:
  68              require_css( 'bootstrap-backoffice-b2evo_base.bmin.css', 'rsc_url' ); // Concatenation + Minifaction of the above
  69          }
  70          
  71          // Make sure standard CSS is called ahead of custom CSS generated below:
  72          if( $debug )
  73          {    // Use readable CSS:
  74              require_css( 'skins_adm/bootstrap/rsc/css/style.css', 'relative' );    // Relative to <base> tag (current skin folder)
  75          }
  76          else
  77          {    // Use minified CSS:
  78              require_css( 'skins_adm/bootstrap/rsc/css/style.min.css', 'relative' );    // Relative to <base> tag (current skin folder)
  79          }
  80  
  81          // Set bootstrap css classes for messages
  82          $Messages->set_params( array(
  83                  'class_outerdiv' => 'action_messages container-fluid',
  84                  'class_success'  => 'alert alert-dismissible alert-success fade in',
  85                  'class_warning'  => 'alert alert-dismissible alert-warning fade in',
  86                  'class_error'    => 'alert alert-dismissible alert-danger fade in',
  87                  'class_note'     => 'alert alert-dismissible alert-info fade in',
  88                  'before_message' => '<button type="button" class="close" data-dismiss="alert"><span aria-hidden="true">&times;</span></button>',
  89              ) );
  90  
  91          // Initialize font-awesome icons and use them as a priority over the glyphicons, @see get_icon()
  92          init_fontawesome_icons( 'fontawesome-glyphicons' );
  93  
  94          if( $check_browser_version && $Hit->get_browser_version() > 0 && $Hit->is_IE( 9, '<' ) )
  95          {    // Display info message if browser IE < 9 version and it is allowed by config var:
  96              $Messages->add( T_('Your web browser is too old. For this site to work correctly, we recommend you use a more recent browser.'), 'note' );
  97              if( $debug )
  98              {
  99                  $Messages->add( 'User Agent: '.$Hit->get_user_agent(), 'note' );
 100              }
 101          }
 102      }

 736      /**
 737       * Returns list of buttons for available Collections (aka Blogs) to work on.
 738       *
 739       * @param string Title
 740       * @return string HTML
 741       */
 742  	function get_bloglist_buttons( $title = '' )
 743      {
 744          global $blog, $current_User, $admin_url;
 745  
 746          $max_buttons = 7;
 747  
 748          if( empty( $this->coll_list_permname ) )
 749          {    // We have not requested a list of blogs to be displayed
 750              return;
 751          }
 752  
 753          // Prepare url params:
 754          $url_params = '?';
 755          foreach( $this->coll_list_url_params as $name => $value )
 756          {
 757              $url_params .= $name.'='.$value.'&amp;';
 758          }
 759  
 760          $template = $this->get_template( 'CollectionList' );
 761  
 762          $BlogCache = & get_BlogCache();
 763  
 764          $blog_array = $BlogCache->load_user_blogs( $this->coll_list_permname, $this->coll_list_permlevel );
 765  
 766          $buttons = '';
 767          $select_options = '';
 768          $not_favorite_blogs = false;
 769  
 770          foreach( $blog_array as $l_blog_ID )
 771          { // Loop through all blogs that match the requested permission:
 772  
 773              $l_Blog = & $BlogCache->get_by_ID( $l_blog_ID );
 774  
 775              if( $l_Blog->get( 'favorite' ) || $l_blog_ID == $blog )
 776              { // If blog is favorute OR current blog, Add blog as a button:
 777                  $buttons .= $template[ $l_blog_ID == $blog ? 'beforeEachSel' : 'beforeEach' ];
 778  
 779                  $buttons .= '<a href="'.$url_params.'blog='.$l_blog_ID
 780                              .'" class="btn btn-default'.( $l_blog_ID == $blog ? ' active' : '' ).'"';
 781  
 782                  if( !is_null($this->coll_list_onclick) )
 783                  {    // We want to include an onclick attribute:
 784                      $buttons .= ' onclick="'.sprintf( $this->coll_list_onclick, $l_blog_ID ).'"';
 785                  }
 786  
 787                  $buttons .= '>'.$l_Blog->dget( 'shortname', 'htmlbody' ).'</a> ';
 788  
 789                  if( $l_blog_ID == $blog )
 790                  {
 791                      $buttons .= $template['afterEachSel'];
 792                  }
 793                  else
 794                  {
 795                      $buttons .= $template['afterEach'];
 796                  }
 797              }
 798  
 799              if( !$l_Blog->get( 'favorite' ) )
 800              { // If blog is not favorute, Add it into the select list:
 801                  $not_favorite_blogs = true;
 802                  $select_options .= '<li>';
 803                  if( $l_blog_ID == $blog )
 804                  {
 805                      //$select_options .= ' selected="selected"';
 806                  }
 807                  $select_options .= '<a href="'.$url_params.'blog='.$l_blog_ID.'">'
 808                      .$l_Blog->dget( 'shortname', 'formvalue' ).'</a></li>';
 809              }
 810          }
 811  
 812          $r = $template['before'];
 813  
 814          $r .= $title;
 815  
 816          if( !empty( $this->coll_list_all_title ) )
 817          { // We want to add an "all" button
 818              $r .= $template[ $blog == 0 ? 'beforeEachSel' : 'beforeEach' ];
 819              $r .= '<a href="'.$this->coll_list_all_url
 820                          .'" class="btn btn-default'.( $blog == 0 ? ' active' : '' ).'">'
 821                          .$this->coll_list_all_title.'</a> ';
 822              $r .= $template[ $blog == 0 ? 'afterEachSel' : 'afterEach' ];
 823          }
 824  
 825          $r .= $template['buttons_start'];
 826          $r .= $buttons;
 827          $r .= $template['buttons_end'];
 828  
 829  
 830          if( $not_favorite_blogs )
 831          { // Display select list with not favorite blogs
 832              $r .= $template['select_start']
 833                  .'<a href="#" class="btn btn-default dropdown-toggle" data-toggle="dropdown">'.T_('Other')
 834                  .'<span class="caret"></span></a>'
 835                  .'<ul class="dropdown-menu">'
 836                  .$select_options
 837                  .'</ul>'
 838                  .$template['select_end'];
 839          }
 840  
 841          // Button to add new collection
 842          if( is_logged_in() && $current_User->check_perm( 'blogs', 'create' ) )
 843          { // Display a button to add new collection if current user has a permission
 844              $button_add_blog = '<a href="'.$admin_url.'?ctrl=collections&amp;action=new" class="btn btn-default" title="'.T_('New Collection').'"><span class="fa fa-plus"></span></a>';
 845          }
 846          else
 847          { // No permission to add new collection
 848              $button_add_blog = '';
 849          }
 850  
 851          $r .= str_replace( '$button_add_blog$', $button_add_blog, $template['after'] );
 852  
 853          return $r;
 854      }
