 127  	function RenderItemAsHtml( & $params )
 128      {
 129          $content = & $params['data'];
 130  
 131          if( !empty( $params['Item'] ) )
 132          { // Get Item from params
 133              $Item = & $params['Item'];
 134          }
 135          elseif( !empty( $params['Comment'] ) )
 136          { // Get Item from Comment
 137              $Comment = & $params['Comment'];
 138              $Item = & $Comment->get_Item();
 139          }
 140  
 141          if( ! empty( $Item ) )
 142          { // We are rendering Item or Comment now, Get the settings depending on Blog
 143              $item_Blog = & $Item->get_Blog();
 144              $text_styles_enabled = $this->get_coll_setting( 'text_styles', $item_Blog );
 145              $links_enabled = $this->get_coll_setting( 'links', $item_Blog );
 146              $images_enabled = $this->get_coll_setting( 'images', $item_Blog );
 147              $min_h_level = $this->get_coll_setting( 'min_h_level', $item_Blog );
 148          }
 149          elseif( ! empty( $params['Message'] ) )
 150          { // We are rendering Message now, Use FALSE by default because we don't have the settings
 151              $text_styles_enabled = false;
 152              $links_enabled = false;
 153              $images_enabled = false;
 154              $min_h_level = 1; // Use default value
 155          }
 156          else
 157          { // Unknown call, Don't render this case
 158              return;
 159          }
 160  
 161          // Init parser class with blog settings
 162          $Parsedown = Parsedown::instance();
 163          $Parsedown->parse_font_styles = $text_styles_enabled;
 164          $Parsedown->parse_links = $links_enabled;
 165          $Parsedown->parse_images = $images_enabled;
 166          $Parsedown->min_h_level = $min_h_level;
 167  
 168          // Parse markdown code to HTML
 169          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false )
 170          { // Call replace_content() on everything outside code/pre:
 171              $content = callback_on_non_matching_blocks( $content,
 172                  '~<(code|pre)[^>]*>.*?</\1>~is',
 173                  array( $Parsedown, 'parse' ) );
 174          }
 175          else
 176          { // No code/pre blocks, replace on the whole thing
 177              $content = $Parsedown->parse( $content );
 178          }
 179  
 180          return true;
 181      }

 554  		function markdown_insert_tag( field, i )
 555          {
 556              // we need to know if something is selected.
 557              // First, ask plugins, then try IE and Mozilla.
 558              var sel_text = b2evo_Callbacks.trigger_callback( "get_selected_text_for_" + field.id );
 559              var focus_when_finished = false; // used for IE
 560  
 561              if( sel_text == null )
 562              { // detect selection:
 563                  //IE support
 564                  if( document.selection )
 565                  {
 566                      field.focus();
 567                      var sel = document.selection.createRange();
 568                      sel_text = sel.text;
 569                      focus_when_finished = true;
 570                  }
 571                  //MOZILLA/NETSCAPE support
 572                  else if( field.selectionStart || field.selectionStart == '0' )
 573                  {
 574                      var startPos = field.selectionStart;
 575                      var endPos = field.selectionEnd;
 576                      sel_text = ( startPos != endPos );
 577                  }
 578              }
 579  
 580              if( sel_text )
 581              { // some text selected
 582                  textarea_wrap_selection( field, markdown_btns[i].tag_start, markdown_btns[i].tag_end, 0 );
 583              }
 584              else
 585              {
 586                  if( !markdown_check_open_tags(i) || markdown_btns[i].tag_end == '' )
 587                  {
 588                      textarea_wrap_selection( field, markdown_btns[i].tag_start, '', 0 );
 589                      markdown_add_tag(i);
 590                  }
 591                  else
 592                  {
 593                      textarea_wrap_selection( field, '', markdown_btns[i].tag_end, 0 );
 594                      markdown_remove_tag(i);
 595                  }
 596              }
 597              if( focus_when_finished )
 598              {
 599                  field.focus();
 600              }
 601          }
