 213      /**
 214       * Parse BB code
 215       *   ( The main purpose of this function is to parsing multilevel lists tags,
 216       *     i.e. when one [list] is contained inside other [list] )
 217       *
 218       * @param string Content
 219       * @param array Search list
 220       * @param array Replace list
 221       * @return string Content
 222       */
 223  	function parse_bbcode( $content, $search_list, $replace_list )
 224      {
 225          if( empty( $content ) )
 226          { // No content, Exit here
 227              return $content;
 228          }
 229  
 230          $complex = array();
 231          $simple = array();
 232  
 233          foreach( $search_list as $i => $search_text )
 234          { // Split the masks to separate things
 235              preg_match( '#(\\\\\[.+\])\(.+\)(\\\\\[.+\])#is', $search_text, $search_tags );
 236              preg_match( '#(<.+>)\$.+(<.+>)#is', $replace_list[ $i ], $replace_tags );
 237              if( !empty( $search_tags ) )
 238              {
 239                  $complex[] = array(
 240                          's_full'  => $search_text,
 241                          's_start' => '!^'.$search_tags[1].'!is',
 242                          's_end'   => '!^'.$search_tags[2].'!is',
 243                          'r_full'  => $replace_list[ $i ],
 244                          'r_start' => $replace_tags[1],
 245                          'r_end'   => $replace_tags[2],
 246                      );
 247              }
 248              else
 249              {
 250                  $simple[] = array(
 251                          's_full' => $search_text,
 252                          'r_full' => $replace_list[ $i ],
 253                      );
 254              }
 255          }
 256          //pre_dump($complex);
 257  
 258          foreach( $simple as $tag_info )
 259          { // Make simple replacements
 260              $content = preg_replace( $tag_info['s_full'], $tag_info['r_full'], $content );
 261          }
 262  
 263          // Complex parsing with multilevel tags structure
 264          $c_pos = 0;
 265          $content_length = strlen( $content );
 266          $opened_tags = array();
 267          $new_content = $content;
 268          while( $c_pos < $content_length - 2 )
 269          {
 270              if( $content[ $c_pos ].$content[ $c_pos + 1 ] == '[/' )
 271              { // Closed tag
 272                  if( !empty( $opened_tags ) )
 273                  {    // Get closed tag that was opened as last
 274                      $sub_content = substr( $content, $c_pos );
 275                      $closed_tag_info = $complex[ $opened_tags[ count( $opened_tags ) - 1 ] ];
 276                      if( preg_match( $closed_tag_info['s_end'], $sub_content ) )
 277                      {
 278                          $first_part = str_replace( $sub_content, '', $new_content );
 279                          $sub_content = preg_replace( $closed_tag_info['s_end'], $closed_tag_info['r_end'], $sub_content );
 280                          $new_content = $first_part.$sub_content;
 281                          //pre_dump( 'CLOSE', $opened_tags, $closed_tag_info['r_end'], $closed_tag_info['s_end'], $new_content );
 282                          // Remove this tag from opened tags array
 283                          array_pop( $opened_tags );
 284                      }
 285                  }
 286              }
 287              elseif( $content[ $c_pos ] == '[' )
 288              { // Opened tag
 289                  $sub_content = substr( $content, $c_pos );
 290                  foreach( $complex as $tag_num => $tag_info )
 291                  {
 292                      if( preg_match( $tag_info['s_start'], $sub_content ) )
 293                      {
 294                          $first_part = str_replace( $sub_content, '', $new_content );
 295                          $sub_content = preg_replace( $tag_info['s_start'], $tag_info['r_start'], $sub_content );
 296                          $new_content = $first_part.$sub_content;
 297                          // Add this tag to know what tag was opened as last
 298                          $opened_tags[] = $tag_num;
 299                          //pre_dump( 'OPEN', $opened_tags, $tag_info['s_start'], $new_content );
 300                          break;
 301                      }
 302                  }
 303              }
 304              $c_pos++;
 305          }
 306          //pre_dump($new_content);
 307  
 308          $new_content = $this->parse_anchor_links( $new_content );
 309  
 310          return $new_content;
 311      }

 314      /**
 315       * Parse anchor links, Set absolute path for each link with relative anchor like <a href="#">
 316       *
 317       * @param string Content
 318       * @return string Content
 319       */
 320  	function parse_anchor_links( $content )
 321      {
 322          if( preg_match_all( '/ href="#(c|p)([0-9]+)"/i', $content, $matches ) )
 323          {
 324              $CommentCache = & get_CommentCache();
 325              $ItemCache = & get_ItemCache();
 326              foreach( $matches[0] as $m => $full_match )
 327              {
 328                  $object_ID = $matches[2][$m];
 329                  $new_url = '';
 330                  switch( $matches[1][$m] )
 331                  { // Object type:
 332                      case 'p':
 333                          // Item
 334                          if( $Item = & $ItemCache->get_by_ID( $object_ID, false, false ) )
 335                          { // Replace anchor url with item permanent url
 336                              $new_url = $Item->get_permanent_url().'#p'.$object_ID;
 337                          }
 338                          break;
 339  
 340                      case 'c':
 341                          // Comment
 342                          if( $Comment = & $CommentCache->get_by_ID( $object_ID, false, false ) )
 343                          { // Replace anchor url with comment permanent url
 344                              $new_url = $Comment->get_permanent_url();
 345                          }
 346                          break;
 347  
 348                      default:
 349                          // Incorrect object type, Skip this url
 350                          continue;
 351                  }
 352                  if( !empty( $new_url ) )
 353                  { // Replace relative anchor url with new absolute url
 354                      $content = str_replace( $full_match, ' href="'.$new_url.'"', $content );
 355                  }
 356              }
 357          }
 358  
 359          return $content;
 360      }
