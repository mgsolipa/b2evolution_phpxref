 124      /**
 125       * Convert code blocks to html tags
 126       *
 127       * @param string Content
 128       * @return string Content
 129       */
 130  	function filter_code( $content )
 131      {
 132          // change all [codeblock]  segments before format_to_post() gets a hold of them
 133          // 1 - codeblock or codespan
 134          // 2 - attribs : lang &| line
 135          // 3 - code content
 136          $content = preg_replace_callback( '#\[(codeblock|codespan)([^\]]*?)\]([\s\S]+?)?\[/\1\]#i',
 137                                  array( $this, 'filter_code_callback' ), $content );
 138  
 139          return $content;
 140      }

 143      /**
 144       * Formats code ready for rendering
 145       *
 146       * @param array $block ( 1 - type, 2 - attributes, 3 - content )
 147       * @return string formatted code || empty
 148       */
 149  	function filter_code_callback( $block )
 150      {
 151          $content = trim( $block[3] );
 152  
 153          if( empty( $content ) )
 154          { // Don't render if no code content
 155              return '';
 156          }
 157  
 158          // Type of code: 'codeblock' OR 'codespan'
 159          $type = $block[1];
 160  
 161          // Language:
 162          $lang = strtolower( preg_replace( '/.*lang="?([a-z]+)"?.*/i', '$1', html_entity_decode( $block[2] ) ) );
 163          if( ! in_array( $lang, array( 'php', 'css', 'javascript', 'sql' ) ) )
 164          { // Use Markup for unknown language
 165              $lang = 'markup';
 166          }
 167  
 168          $r = '<code class="language-'.$lang.'">'.$block[3].'</code>';
 169  
 170          if( $type == 'codeblock' )
 171          { // Set special template and attributes only for codeblock
 172  
 173              // Detect number of start line:
 174              $line = intval( preg_replace( '/.*line="?(-?[0-9]+)"?.*/i', '$1', html_entity_decode( $block[2] ) ) );
 175              $line = $line != 1 ? ' data-start="'.$line.'"' : '';
 176  
 177              // Put <pre> around <code> to render codeblock
 178              $r = '<pre class="line-numbers"'.$line.'>'.$r.'</pre>';
 179          }
 180  
 181          return $r;
 182      }

 200      /**
 201       * Formats code ready for editing
 202       *
 203       * @param array $block ( 1 - start of <pre> tag, 4 - language, 5 - content )
 204       * @return string formatted code || empty
 205       */
 206  	function unfilter_code_callback( $block )
 207      {
 208          if( empty( $block[1] ) )
 209          { // [codespan]
 210              $code_tag = 'codespan';
 211              // codespan doesn't provide line numbers
 212              $line = '';
 213          }
 214          else
 215          { // [codeblock]
 216              $code_tag = 'codeblock';
 217              // Detect number of start line:
 218              preg_match( '/.*data-start="(-?[0-9]+)".*/i', html_entity_decode( $block[1] ), $line );
 219              $line = ' line="'.( isset( $line[1] ) ? intval( $line[1] ) : '1' ).'"';
 220          }
 221  
 222          // Build codeblock
 223          $r = '['.$code_tag.' lang="'.strtolower( $block[4] ).'"'.$line.']';
 224          $r .= $block[5];
 225          $r .= '[/'.$code_tag.']';
 226  
 227          return $r;
 228      }

 256  	function DisplayCommentToolbar( & $params )
 257      {
 258          if( !empty( $params['Comment'] ) )
 259          { // Comment is set, get Blog from comment
 260              $Comment = & $params['Comment'];
 261              if( !empty( $Comment->item_ID ) )
 262              {
 263                  $comment_Item = & $Comment->get_Item();
 264                  $Blog = & $comment_Item->get_Blog();
 265              }
 266          }
 267  
 268          if( empty( $Blog ) )
 269          { // Comment is not set, try global Blog
 270              global $Blog;
 271              if( empty( $Blog ) )
 272              { // We can't get a Blog, this way "apply_comment_rendering" plugin collection setting is not available
 273                  return false;
 274              }
 275          }
 276  
 277          $apply_rendering = $this->get_coll_setting( 'coll_apply_comment_rendering', $Blog );
 278          if( ! empty( $apply_rendering ) && $apply_rendering != 'never' )
 279          {
 280              return $this->display_toolbar();
 281          }
 282          return false;
