  88      /**
  89       * Texturize content
  90       *
  91       * @param string $content
  92       * @return string texturized content
  93       */
  94  	function texturize_block( $content )
  95      {
  96          $output = '';
  97          $textarr = preg_split("/(<.*>)/Us", $content, -1, PREG_SPLIT_DELIM_CAPTURE); // capture the tags as well as in between
  98          $stop = count($textarr); $next = true; // loop stuff
  99          for ($i = 0; $i < $stop; $i++) {
 100              $curl = $textarr[$i];
 101  
 102              if (strlen($curl) && '<' != $curl{0} && $next) { // If it's not a tag
 103                  $curl = str_replace('---', '&#8212;', $curl);
 104                  $curl = str_replace('--', '&#8211;', $curl);
 105                  $curl = str_replace("...", '&#8230;', $curl);
 106                  $curl = str_replace('``', '&#8220;', $curl);
 107  
 108                  // This is a hack, look at this more later. It works pretty well though.
 109                  $cockney = array("'tain't","'twere","'twas","'tis","'twill","'til","'bout","'nuff","'round");
 110                  $cockneyreplace = array("&#8217;tain&#8217;t","&#8217;twere","&#8217;twas","&#8217;tis","&#8217;twill","&#8217;til","&#8217;bout","&#8217;nuff","&#8217;round");
 111                  $curl = str_replace($cockney, $cockneyreplace, $curl);
 112  
 113                  $curl = preg_replace("/'s/", '&#8217;s', $curl);
 114                  $curl = preg_replace("/'(\d\d(?:&#8217;|')?s)/", "&#8217;$1", $curl);
 115                  $curl = preg_replace('/(\s|\A|")\'/', '$1&#8216;', $curl);
 116                  $curl = preg_replace('/(\d+)"/', '$1&Prime;', $curl);
 117                  $curl = preg_replace("/(\d+)'/", '$1&prime;', $curl);
 118                  $curl = preg_replace("/(\S)'([^'\s])/", "$1&#8217;$2", $curl);
 119                  $curl = preg_replace('/(\s|\A)"(?!\s)/', '$1&#8220;$2', $curl);
 120                  $curl = preg_replace('/"(\s|\Z)/', '&#8221;$1', $curl);
 121                  $curl = preg_replace("/'([\s.]|\Z)/", '&#8217;$1', $curl);
 122                  $curl = preg_replace("/\(tm\)/i", '&#8482;', $curl);
 123                  $curl = preg_replace("/\(c\)/i", '&#169;', $curl);
 124                  $curl = preg_replace("/\(r\)/i", '&#174;', $curl);
 125                  $curl = preg_replace('/&([^#])(?![a-z]{1,8};)/', '&#038;$1', $curl);
 126                  $curl = str_replace("''", '&#8221;', $curl);
 127  
 128                  $curl = preg_replace('/(d+)x(\d+)/', "$1&#215;$2", $curl);
 129  
 130              } elseif (strstr($curl, '<code') || strstr($curl, '<pre') || strstr($curl, '<kbd' || strstr($curl, '<style') || strstr($curl, '<script'))) {
 131                  // strstr is fast
 132                  $next = false;
 133              } else {
 134                  $next = true;
 135              }
 136              $output .= $curl;
 137          }
 138          $content = $output;
 139  
 140          return $content;
 141      }
