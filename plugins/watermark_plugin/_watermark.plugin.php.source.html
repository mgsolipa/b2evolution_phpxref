 196      /**
 197       * This gets called before an image thumbnail gets created.
 198       *
 199       * This is useful to post-process the thumbnail image (add a watermark or change colors).
 200       *
 201       * @param array Associative array of parameters
 202       *   - 'File': the related File (by reference)
 203       *   - 'imh': image resource (by reference)
 204       *   - 'size': size name (by reference)
 205       *   - 'mimetype': mimetype of thumbnail (by reference)
 206       *   - 'quality': JPEG image quality [0-100] (by reference)
 207       *   - 'root_type': file root type 'user', 'group', 'collection' etc. (by reference)
 208       *   - 'root_type_ID': ID of user, group or collection (by reference)
 209       */
 210  	function BeforeThumbCreate( & $params )
 211      {
 212          if( $params['root_type'] == 'collection' )
 213          {
 214              $BlogCache = & get_BlogCache();
 215              $Blog = & $BlogCache->get_by_ID( $params['root_type_ID'], false, false );
 216          }
 217  
 218          if( ! empty( $Blog ) )
 219          { // Use blog settings when possible
 220              $font = $this->get_coll_setting( 'coll_font', $Blog );
 221              $text = $this->get_coll_setting( 'coll_text', $Blog );
 222              $font_size = $this->get_coll_setting( 'coll_font_size', $Blog );
 223              $min_dimension = $this->get_coll_setting( 'coll_min_dimension', $Blog );
 224          }
 225          else
 226          { // Use global settings
 227              $font = $this->Settings->get( 'font' );
 228              $text = $this->Settings->get( 'text' );
 229              $font_size = $this->Settings->get( 'font_size' );
 230              $min_dimension = $this->Settings->get( 'min_dimension' );
 231          }
 232  
 233          if( ! function_exists( 'imagettftext' ) )
 234          { // The function imagettftext() is not available.
 235              return;
 236          }
 237  
 238          // Canvas width & height
 239          $width = imagesx( $params['imh'] ) ;
 240          $height = imagesy( $params['imh'] );
 241  
 242          if( $width < $min_dimension || $height < $min_dimension )
 243          { // Skip small thumbnails
 244              return;
 245          }
 246  
 247          if( $font )
 248          {    // Custom font
 249              $font = $this->fonts_dir.'/'.$font;
 250          }
 251          else
 252          {
 253              $font = $this->get_default_font();
 254          }
 255  
 256          if( !is_readable($font) )
 257          {    // Font file not found
 258              $this->debug_log( sprintf( 'Font file (%s) is not readable!', $font ) );
 259              return;
 260          }
 261  
 262          $text = html_entity_decode( $text );
 263  
 264          // Text margins
 265          $margin_left = 10;
 266          $margin_bottom = 10;
 267  
 268          // Create a transparent image
 269          $im = imagecreatetruecolor($width, $height);
 270          imagealphablending($im, true);
 271          imagefill($im, 0, 0, imagecolortransparent($im, imagecolorallocatealpha($im, 0, 0, 0, 127)));
 272          imagesavealpha($im, true);
 273  
 274          // Create some colors
 275          $white = imagecolorallocate($im, 255, 255, 255);
 276          $light_grey = imagecolorallocate($im, 230, 230, 230);
 277          $grey = imagecolorallocate($im, 60, 60, 60);
 278          $black = imagecolorallocate($im, 0, 0, 0);
 279  
 280          // Add text shadow
 281          imagettftext($im, $font_size, 0, $margin_left + 2, $height - $margin_bottom, $grey, $font, $text);
 282  
 283          // Add text
 284          imagettftext($im, $font_size, 0, $margin_left, $height - $margin_bottom, $light_grey, $font, $text);
 285  
 286          // Merge images
 287          imagecopyresampled($params['imh'], $im, 0, 0, 0, 0, $width, $height, $width, $height);
 288  
 289          // Low quality :(
 290          //imagecopymerge($params['imh'], $im, 0, 0, 0, 0, $width, $height, 90);
 291  
 292          return true;
 293      }

 314      /**
 315       * Get all font files from folder
 316       *
 317       * @return array Font files
 318       */
 319  	function get_font_files()
 320      {
 321          $fonts = array( '' => T_('Default') );
 322  
 323          // Scan fonts folder
 324          $files = scandir( $this->fonts_dir );
 325  
 326          if( $files !== false )
 327          {
 328              foreach( $files as $file )
 329              {
 330                  if( preg_match( '/\.ttf$/i', $file ) )
 331                  { // Allow only ttf files
 332                      $fonts[ $file ] = $file;
 333                  }
 334              }
 335          }
 336  
 337          return $fonts;
 338      }
