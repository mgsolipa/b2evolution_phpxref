 125      /**
 126       * Escape html entities inside <code> tag
 127       *
 128       * @param string Content
 129       * @param string Function name for callback
 130       * @return string Escaped content
 131       */
 132  	function escape_code( $content, $callback_function = 'escape_code_callback' )
 133      {
 134          if( strpos( $content, '[codeblock' ) !== false || strpos( $content, '<codeblock' ) !== false )
 135          { // Do escape the html entities in code blocks:
 136              $content = preg_replace_callback( '#([<\[]codeblock[^>\]]*[>\]])([\s\S]+?)([<\[]/codeblock[>\]])#is', array( $this, $callback_function ), $content );
 137          }
 138  
 139          if( strpos( $content, '[codespan' ) !== false || strpos( $content, '<codespan' ) !== false )
 140          { // Do escape the html entities in code spans:
 141              $content = preg_replace_callback( '#([<\[]codespan[>\]])([\s\S]+?)([<\[]/codespan[>\]])#is', array( $this, $callback_function ), $content );
 142          }
 143  
 144          if( strpos( $content, '<code' ) !== false )
 145          { // At least one tag <code> exists in the content, Do escape the html entities:
 146              $content = preg_replace_callback( '#(<code[^>]*>)([\s\S]+?)(</code>)#is', array( $this, $callback_function ), $content );
 147          }
 148  
 149          if( strpos( $content, '`' ) !== false )
 150          { // String of codespan from markdown, Do escape the html entities:
 151              $content = preg_replace_callback( '#(`)([^`\n]+)(`)#i', array( $this, $callback_function ), $content );
 152          }
 153  
 154          if( strpos( $content, '```' ) !== false )
 155          { // String of codeblock from markdown, Do escape the html entities:
 156              $content = preg_replace_callback( '#(```)([\s\S]+?)(```)#is', array( $this, $callback_function ), $content );
 157          }
 158  
 159          return $content;
 160      }

 163      /**
 164       * Escape html entities inside <code> tag
 165       *
 166       * @param string Code content
 167       * @return string Escaped code content
 168       */
 169  	function escape_code_callback( $code_content )
 170      {
 171          // Start tag
 172          $escaped_content = $code_content[1];
 173  
 174          // Escape two chars to escape html tags inside <code>
 175          $escaped_content .= str_replace( array( '<', '>' ), array( '&lt;', '&gt;' ), $code_content[2] );
 176  
 177          // End tag
 178          $escaped_content .= $code_content[3];
 179  
 180          return $escaped_content;
 181      }
