 110      /**
 111       * Perform rendering
 112       *
 113       * @param array Associative array of parameters
 114       *   'data': the data (by reference). You probably want to modify this.
 115       *   'format': see {@link format_to_output()}. Only 'htmlbody' and 'entityencoded' will arrive here.
 116       * @return boolean true if we can render something for the required output format
 117       */
 118  	function RenderItemAsHtml( & $params )
 119      {
 120          $content = & $params['data'];
 121  
 122          if( !empty( $params['Item'] ) )
 123          { // Get Item from params
 124              $Item = & $params['Item'];
 125          }
 126          elseif( !empty( $params['Comment'] ) )
 127          { // Get Item from Comment
 128              $Comment = & $params['Comment'];
 129              $Item = & $Comment->get_Item();
 130          }
 131  
 132          if( ! empty( $Item ) )
 133          { // We are rendering Item or Comment now, Get a setting depending on Blog
 134              $item_Blog = & $Item->get_Blog();
 135              $this->min_h_level = $this->get_coll_setting( 'min_h_level', $item_Blog );
 136          }
 137  
 138          if( $this->min_h_level > 2 && $this->min_h_level <= 6 )
 139          { // Restrict <h_> tags by minimum heading level
 140              foreach( $this->replace as $r => $replace )
 141              { // Do replace
 142                  $this->replace[ $r ] = preg_replace_callback( '#([^<]*<)(h[2-6])(>[^<]*</)\2(>)#i',
 143                      array( $this, 'restrict_min_h_level' ), $replace );
 144              }
 145          }
 146  
 147          $content = replace_content_outcode( $this->search, $this->replace, $content );
 148  
 149          // Find bullet lists
 150          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false || strstr( $content, '`' ) !== false )
 151          {    // Call replace_content() on everything outside code/pre:
 152              $content = callback_on_non_matching_blocks( $content,
 153                  '~(`|<(code|pre)[^>]*>).*?(\1|</\2>)~is',
 154                  array( $this, 'find_bullet_lists' ) );
 155          }
 156          else
 157          {    // No code/pre blocks, replace on the whole thing
 158              $content = $this->find_bullet_lists( $content );
 159          }
 160  
 161          return true;
 162      }

 186      /**
 187       * Find bullet lists
 188       *
 189       * @param string Content
 190       * @return string Content
 191       */
 192  	function find_bullet_lists( $content )
 193      {
 194          // Find and parse the code blocks to html view
 195          $content = $this->escape_codeblock( $content );
 196  
 197          $lines = explode( "\n", $content );
 198          $lines_count = count( $lines );
 199          $lists = array();
 200          $current_depth = 0;
 201          $content = '';
 202          foreach( $lines as $l => $line )
 203          {
 204              if( ! preg_match( '#^ /s $#xm', $line ) )
 205              {     // If not blank line
 206                  $matches = array();
 207  
 208                  if( preg_match( '#^((  )+)\*(.*)$#m', $line, $matches ) )
 209                  {    // We have a list item
 210                      $req_depth = strlen( $matches[1] ) / 2;
 211                      while( $current_depth < $req_depth )
 212                      {    // We must indent
 213                          $content .= "<ul>\n";
 214                          array_push( $lists, 'ul' );
 215                          $current_depth++;
 216                      }
 217  
 218                      while( $current_depth > $req_depth )
 219                      {    // We must close lists
 220                          $content .= '</'.array_pop( $lists ).">\n";
 221                          $current_depth--;
 222                      }
 223  
 224                      $content .= $matches[1].'<li>'.$matches[3]."</li>\n";
 225                      continue;
 226                  }
 227  
 228                  if( preg_match( '#^((  )+)([0-9]+)(.*)$#m', $line, $matches ) )
 229                  {    // We have an ordered list item
 230                      $req_depth = strlen( $matches[1] ) / 2;
 231                      while( $current_depth < $req_depth )
 232                      {    // We must indent
 233                          $content .= '<ol start="'.$matches[3].'">'."\n";
 234                          array_push( $lists, 'ol' );
 235                          $current_depth++;
 236                      }
 237  
 238                      while( $current_depth > $req_depth )
 239                      {    // We must close lists
 240                          $content .= '</'.array_pop( $lists ).">\n";
 241                          $current_depth--;
 242                      }
 243  
 244                      $content .= $matches[1].'<li>'.$matches[4]."</li>\n";
 245                      continue;
 246                  }
 247  
 248                  // Normal line.
 249  
 250                  if( $current_depth )
 251                  { // We must go back to 0
 252                      $content .= '</'.implode( ">\n</", $lists ).">\n";
 253                      $lists = array();
 254                      $current_depth = 0;
 255                  }
 256  
 257                  $content .= $line;
 258                  if( $l < $lines_count - 1 )
 259                  {    // Don't append a newline at the end, because it will create an unnecessary newline that didn't exist in source content
 260                      $content .= "\n";
 261                  }
 262  
 263              }
 264          }
 265  
 266          if( $current_depth )
 267          { // We must go back to 0
 268              $content .= '</'.implode( ">\n</", $lists ).">\n";
 269          }
 270  
 271          return $content;
 272      }

 275      /**
 276       * Parse code blocks to html view
 277       *
 278       * @param string Content
 279       * @param string
 280       */
 281  	function escape_codeblock( $content )
 282      {
 283          $search = '/ %%%
 284              ( \s*? \n )?                 # Eat optional blank line after %%%
 285              (.+?)
 286              ( \n \s*? )?                 # Eat optional blank line before %%%
 287              %%%
 288          /sx'; // %%%escaped codeblock%%%
 289  
 290          return preg_replace_callback( $search, array( $this, 'escape_codeblock_callback' ), $content );
 291      }
