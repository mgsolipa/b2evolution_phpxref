 208      /**
 209       * Get link to twitter oAuth
 210       *
 211       * @param string target type can be "blog" or "user", depends if we set blog or user setting
 212       * @param string current blog id or edited user id
 213       * @return string twitter oAuth link
 214       */
 215  	function get_twitter_link( $target_type, $target_id )
 216      {
 217          global $Blog;
 218  
 219          require_once  'twitteroauth/twitteroauth.php';
 220  
 221          // Uses either plugin CollSettings or UserSettings
 222          $oauth = $this->get_oauth_info( array(
 223                  'type'    => $target_type,
 224                  'ID'    => $target_id,
 225              ) );
 226  
 227          if( !empty( $oauth['token'] ) )
 228          { // already linked
 229              if( empty( $oauth['contact'] ) )
 230              {
 231                  $oauth['contact'] = $this->get_twitter_contact( $oauth['token'], $oauth['token_secret'] );
 232                  if( ! empty( $oauth['contact'] ) )
 233                  {
 234                      if( $target_type == 'blog' )
 235                      { // CollSettings
 236                          $this->set_coll_setting( 'twitter_contact', $oauth['contact'], $Blog->ID );
 237                          $Blog->dbupdate();
 238                      }
 239                      else if( $target_type == 'user' )
 240                      { // UserSettings
 241                          $this->UserSettings->set( 'twitter_contact', $oauth['contact'], $target_id );
 242                          $this->UserSettings->dbupdate();
 243                      }
 244                  }
 245              }
 246              $result = T_('Linked to').': @'.$oauth['contact'].'. ';
 247          }
 248  
 249          // create new connection
 250          $connection = new TwitterOAuth(TWITTER_CONSUMER_KEY, TWITTER_CONSUMER_SECRET);
 251  
 252          // set callback url
 253          $callback = $this->get_htsrv_url( 'twitter_callback', array(), '&', true );
 254          // Use the separate params for this request instead of using
 255          //     array $params(second param of the func above)
 256          //     because twitter cannot redirects to this complex url with serialized data
 257          $callback = url_add_param( $callback, 'target_type='.$target_type.'&target_id='.$target_id, '&' );
 258  
 259          $req_token = $connection->getRequestToken( $callback );
 260  
 261          if( $req_token == NULL )
 262          {
 263              return T_( 'Connection is not available!' );
 264          }
 265  
 266          $token = $req_token['oauth_token'];
 267  
 268          /* Save temporary credentials to session. */
 269          global $Session;
 270          $Session->delete( 'oauth_token' );
 271          $Session->delete( 'oauth_token_secret' );
 272          $Session->set( 'oauth_token', $req_token['oauth_token'] );
 273          $Session->set( 'oauth_token_secret', $req_token['oauth_token_secret'] );
 274          $Session->dbsave();
 275  
 276          if( empty( $result ) )
 277          { // wasn't linked to twitter
 278              $result = '<a href='.$connection->getAuthorizeURL( $req_token, false ).'>'.T_( 'Click here to link to your twitter account' ).'</a>';
 279          }
 280          else
 281          {
 282              $result = $result.'<a href='.$connection->getAuthorizeURL( $req_token, false ).'>'.T_( 'Link to another account' ).'</a>';
 283              $unlink_url = $this->get_htsrv_url( 'unlink_account', array( 'target_type' => $target_type, 'target_id' => $target_id ), '&' );
 284              $unlink_url = $unlink_url.'&'.url_crumb( $target_type );
 285              $result = $result.' / '.'<a href="'.$unlink_url.'">'.T_( 'Unlink this account' ).'</a>';
 286          }
 287  
 288          return $result;
 289      }

 302      {
 303          $connection = new TwitterOAuth(TWITTER_CONSUMER_KEY, TWITTER_CONSUMER_SECRET, $oauth_token, $oauth_token_secret );

 535              $r['token'] = $this->UserSettings->get( 'twitter_token', $user_ID );

 548  	function send_a_tweet( $content, & $Item, & $xmlrpcresp )
 549      {
 550          // Uses either plugin CollSettings or UserSettings
 551          $oauth = $this->get_oauth_info( array(
 552                  'user_ID'    => $Item->get_creator_User()->ID,
 553                  'blog_ID'    => $Item->get_Blog()->ID,
 554              ) );
 555  
 556          if( empty($oauth['msg_format']) || empty($oauth['token']) || empty($oauth['token_secret']) )
 557          { // Not found, fallback to Trying to get twitter account for User:
 558              $xmlrpcresp = T_('You must configure a twitter username/password before you can post to twitter.');
 559              return false;
 560          }
 561  
 562          $content = array_merge( array(
 563                      'title'        => '',
 564                      'excerpt'    => '',
 565                      'url'        => ''
 566                  ), $content );
 567  
 568          // Replace the title and exerpt, but before replacing decode the html entities
 569          $msg = str_replace(
 570                  array( '$title$', '$excerpt$' ),
 571                  array( html_entity_decode( $content['title'] ), html_entity_decode( $content['excerpt'] ) ),
 572                  $oauth['msg_format']
 573              );
 574  
 575          $msg_len = utf8_strlen($msg);
 576          $full_url_len = utf8_strlen( $content['url'] );
 577          $base_url_len = utf8_strlen( $Item->get_Blog()->get_baseurl_root() );
 578  
 579          if( (utf8_strpos($msg, '$url$') === 0) && ($base_url_len + $msg_len - 5) > $this->message_length_limit )
 580          {    // The message is too long and is starting with $url$
 581              $max_len = $this->message_length_limit + $full_url_len - $base_url_len;
 582              $msg = strmaxlen( str_replace( '$url$', $content['url'], $msg ), $max_len, '...' );
 583          }
 584          elseif( (utf8_strpos(strrev($msg), 'p2b# $lru$') === 0) && ($base_url_len + $msg_len - 10) > $this->message_length_limit )
 585          {    // The message is too long and is ending on '$url$ #b2p'
 586              // Strip $url$, crop the message, and add URL to the end
 587              $max_len = $this->message_length_limit - $base_url_len -1; // save room for space character
 588              $msg = strmaxlen( str_replace( '$url$ #b2p', '', $msg ), $max_len, '...' );
 589              $msg .= ' '.$content['url'].' #b2p';
 590          }
 591          elseif( (utf8_strpos(strrev($msg), '$lru$') === 0) && ($base_url_len + $msg_len - 5) > $this->message_length_limit )
 592          {    // Same as above, but without '#b2p' suffix
 593              $max_len = $this->message_length_limit - $base_url_len -1; // save room for space character
 594              $msg = strmaxlen( str_replace( '$url$', '', $msg ), $max_len, '...' );
 595              $msg .= ' '.$content['url'];
 596          }
 597          elseif( (utf8_strpos($msg, '$url$') !== false) && ($base_url_len + $msg_len - 5) > $this->message_length_limit )
 598          {    // Message is too long and $url$ is somewhere in the middle
