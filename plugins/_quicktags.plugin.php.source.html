 260  		function b2evoCheckOpenTags(button)
 261          {
 262              var tag = 0;
 263              for (i = 0; i < b2evoOpenTags.length; i++)
 264              {
 265                  if (b2evoOpenTags[i] == button)
 266                  {
 267                      tag++;
 268                  }
 269              }
 270  
 271              if (tag > 0)
 272              {
 273                  return true; // tag found
 274              }
 275              else
 276              {
 277                  return false; // tag not found
 278              }

 310  		function b2evoInsertTag( myField, i )
 311          {
 312              // we need to know if something is selected.
 313              // First, ask plugins, then try IE and Mozilla.
 314              var sel_text = b2evo_Callbacks.trigger_callback("get_selected_text_for_"+myField.id);
 315              var focus_when_finished = false; // used for IE
 316  
 317              if( sel_text == null )
 318              { // detect selection:
 319                  //IE support
 320                  if(document.selection)
 321                  {
 322                      myField.focus();
 323                      var sel = document.selection.createRange();
 324                      sel_text = sel.text;
 325                      focus_when_finished = true;
 326                  }
 327                  //MOZILLA/NETSCAPE support
 328                  else if(myField.selectionStart || myField.selectionStart == '0')
 329                  {
 330                      var startPos = myField.selectionStart;
 331                      var endPos = myField.selectionEnd;
 332                      sel_text = (startPos != endPos);
 333                  }
 334              }
 335  
 336              if( sel_text )
 337              { // some text selected
 338                  textarea_wrap_selection( myField, b2evoButtons[i].tagStart, b2evoButtons[i].tagEnd, 0 );
 339              }
 340              else
 341              {
 342                  if( !b2evoCheckOpenTags(i) || b2evoButtons[i].tagEnd == '')
 343                  {
 344                      textarea_wrap_selection( myField, b2evoButtons[i].tagStart, '', 0 );
 345                      b2evoAddTag(i);
 346                  }
 347                  else
 348                  {
 349                      textarea_wrap_selection( myField, '', b2evoButtons[i].tagEnd, 0 );
 350                      b2evoRemoveTag(i);
 351                  }
 352              }
 353              if(focus_when_finished)
 354              {
 355                  myField.focus();
 356              }
 357          }
