  83  	function highlight_code( $block )
  84      {
  85          // check if we've already grabbed existing shell functions
  86          if( empty( $this->shell_functions ) )
  87          {    // lets build a list of all native shell functions
  88              $this->shell_functions = get_defined_functions();
  89              // and add any missing ones
  90              // possible enhancement is to link all evo functions to it's own docs - yabs
  91              $this->shell_functions['internal'] = array_merge( $this->shell_functions['internal'], array( 'die', 'exit', 'array', 'require', 'require_once', 'include', 'include_once' ) );
  92          }
  93  
  94          // check if we've already grabbed the highlight colours
  95          if( empty( $this->highlight_colours ) )
  96          {    // get the users php_ini colours for highlight_string()
  97              $this->highlight_colours = array(
  98                      'highlight_bg' => '<span class="#FFFFFF',
  99                      'highlight_comment' => '<span class="#FF8000',
 100                      'highlight_default' => '<span class="#0000BB',
 101                      'highlight_html' => '<span class="#000000',
 102                      'highlight_keyword' => '<span class="#007700',
 103                      'highlight_string' => '<span class="#DD0000',
 104                  );
 105          }
 106          // lets sort out the code and highlighting
 107          $code = // take a deep breath and read this code upwards ;)
 108              // stitch it all back together again
 109              implode( "\n",
 110              // gets rid of the < ?php & * / that we added
 111              array_slice( explode( "\n",
 112              // clean it all up ready for numbering
 113              $this->parent->tidy_code_output(
 114              // find all potential shell functions and link relevant ones to the documentation
 115              preg_replace_callback( '#([a-z0-9\-_]+?)(\</span>\<span class="amc_keyword">)?\(#i', array( $this, 'link_to_manual' ),
 116              // change the php.ini highlight colours to our class names
 117              str_replace( $this->highlight_colours, array( '<span class="amc_background', '<span class="amc_comment', '<span class="amc_default', '<span class="amc_html', '<span class="amc_keyword', '<span class="amc_string' ),
 118              // convert php 4's <font> tags to <span> & prepare to convert php 4 & 5 to our class names
 119              preg_replace( array( '#\<font color="\#([0-9a-f]+?)">#i', '#\</font>#', '#\<span style="color: \#([0-9a-f]+?)">#i' ), array( '<span class="#$1">', '</span>','<span class="#$1">' ),
 120              // remove <code></code> tags and \n's added by highlight_string
 121              str_replace( array( '<code>', '</code>', "\n" ), '',
 122              // top code with < ?php & to ensure highlight occurs
 123              highlight_string( '<?php'."\n"
 124              // convert relevant entities back for highlighting
 125              .str_replace( array( '&lt;', '&gt;', '&amp;', '&quot;' ), array( '<', '>', '&', '"' ),
 126              // get rid of empty start/end lines, and add */ to the end to overcome highlight_string() bug with unterminated comments
 127               trim( $block ) )."\n".'*/', true)
 128              ) ) ) ), '<br />' )
 129              // get rid of the first and last array elements
 130              ), 1, -1 ) );
 131          return $code;
 132      }
