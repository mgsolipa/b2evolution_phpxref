 482      /**
 483       * Tidys up a block of code ready for numbering
 484       *
 485       * @param string $block - the code to be tidied up
 486       * @param string $line_seperator - the seperator between lines of code ( default \n )
 487       * @return string - the tidied code
 488       */
 489  	function tidy_code_output( $block, $line_seperator = "\n" )
 490      {
 491          // lets split the block into individual lines
 492          $code = explode( $line_seperator,
 493              // after removing windows garbage
 494              str_replace( "\r", '', $block ) );
 495  
 496          // time to rock and roll ;)
 497          $still_open = array(); // this holds all the spans that need closing and re-opening on the following code line
 498          for( $i = 0; $i < count( $code ); $i++ )
 499          {
 500              // we need to note all opening spans
 501              $spans =
 502                  // get rid of the first element, it's always empty
 503                  array_slice(
 504                  // split line at each opening span
 505                  explode( '<span class="',
 506                  // add any open spans back in
 507                  implode( '', $still_open )
 508                  .$code[$i] )
 509                  , 1 );
 510  //            pre_dump( $spans );
 511              // reset still_open array
 512              $still_open = array();
 513              // $spans now contains a list of opening spans
 514              for( $z = 0; $z < count( $spans ); $z++ )
 515              {
 516                  // add the span to the still_open array
 517                  $still_open[] = '<span class="'.substr( $spans[$z], 0, strpos( $spans[$z], '"' ) ).'">';
 518  //                pre_dump( $still_open );
 519                  // count all closing spans and remove them from the open spans list
 520                  if( $closed = substr_count( $spans[$z], '</span>' ) )
 521                      $still_open = array_slice( $still_open, 0, $closed * -1 );
 522              }
 523              // lets rebuild the code line and close any remaining spans
 524              $code[$i] = '<span class="'.implode( '<span class="', $spans ).str_repeat( '</span>', count( $still_open ) );
 525          }
 526          // lets stitch it all back together again
 527          $cleaned = implode( "\n", $code );
 528          // and get rid of any empty spans
 529          while( preg_match( '#\<span[^>]+?>\</span>#', $cleaned ) )
 530              $cleaned = preg_replace( '#\<span[^>]+?>\</span>#', '', $cleaned );
 531  //        pre_dump( $cleaned );
 532          // return the cleaned up code
 533          return $cleaned;
 534      }

 607      /**
 608       * Formats code ready for displaying
 609       * With "SwipeFriendly" line numbers
 610       *
 611       * @todo fp> no table should be needed. Also since we have odd/even coloring, word wrapping may be ok.
 612       * yabs > unfortunately the tables are required to make the odd/even colouring and swipeable numbers work :(
 613       *
 614       * an example :
 615       * Danny uploaded a page to demo his colours of choice ( http://brendoman.com/dev/youtube.html )
 616       * if you scroll down to RenderItemAsHtml() you'll see it has 4 problems
 617       * 1) some lines wrap - but they wrap underneath the line number
 618       * 2) some lines don't wrap so they force a browser scrollbar
 619       *   ( although a scrollable div will cure this)
 620       * 3) the lines that don't wrap lose their background colour once they become longer than the browser width
 621       *   ( a scrollable div has the same problem )
 622       * 4) If you try and copy/paste the code you also get the line numbers
 623       *   ( like every other code formatter that we've seen on the web that uses line numbers ... which is why most don't ;) )
 624       *
 625       * compare that to the same section of code using this plugin ( http://cvs.astonishme.co.uk/index.php/2007/04/01/a_demo )
 626       *
 627       * fp> almost all that can be fixed with proper CSS and will have the advantage of not adding extra space in front of the lines when copy/pasting. I'll work on it when I have time.
 628       *
 629       * yabs> The operative word in that sentance is "almost" ;) ... I've removed the trailing spaces for all but empty lines ( required to "prop them open" ) ... I'd forgotten all about them :p
 630       *
 631       * @param string the code block to be wrapped up
 632       * @param string the attributes - currently only the starting line number
 633       * @param string the code type ( code || php )
 634       *
 635       * @return string formatted code
 636       */
 637  	function do_numbering( $code, $offset = 0, $type = 'code' )
 638      {
 639          $temp = str_replace( array( '&nbsp;&nbsp;', '  ', "\t", '[x', '[/x' ),  array( '&#160;&#160;', '&#160;&#160;', '&#160;&#160;', '[', '[/' ), $code );
 640          $temp = explode( "\n", $temp );
 641          $count = 0;
 642          $output = '';
 643          $odd_line = false;
 644          foreach( $temp as $line )
 645          {
 646              $output .= '<tr class="amc_code_'.( ( $odd_line = !$odd_line ) ? 'odd' : 'even' ).'"><td class="amc_line">'
 647                                      .$this->create_number( ++$count + $offset ).'</td><td><code class="codeblock">'.$line
 648                                      // add an &nbsp; to empty lines to stop them "collapsing"
 649                                      .( empty( $line ) ? '&nbsp;' : '' )
 650                                      .'</code></td></tr>';//."\n"; yura: I commented this because Auto-P plugin creates the tags <p></p> from this symbol
 651          }
 652          // make "long" value a setting ? - yabs
 653          return '<p class="codeblock_title">'.$this->languageCache[ $type ]->language_title.'</p><div class="codeblock codeblock_with_title amc_'.$type.' '.( $count < 26 ? 'amc_short' : 'amc_long' ).'"><table>'.$output.'</table></div>';
 654      }

 658       * Creates the SwipeFriendly line numbers
 659       *
 660       * @param integer the line number to produce
 661       * @return string the html for the line number
 662       */
 663  	function create_number( $num )
 664      {    // part of Swipe 'n' Paste magic ;)
 665          $result = '';
 666          $count = 0;
 667          while ( $num )
 668          {
 669              $result .= '<div class="amc'.( $num - ( floor( $num / 10 ) * 10 ) ).'">';
 670              $num = floor( $num / 10 );
 671              $count++;
 672          }
 673          $result .= str_repeat( '</div>', $count );

 678      /**
 679       * Formats codeblock ready for displaying
 680       * Each language is stored as a classfile
 681       * This would allow new languages to be added more easily
 682       * It would also allow Geshi to be used as the highlighter with no code changes ;)
 683       *
 684       * Replaces both (current) highlighter functions
 685       *
 686       * ..... still requires some more thought though :p
 687       *
 688       * @param array $block ( 2 - attributes, 4 - the code )
 689       * @return string formatted code
 690       */
 691  	function render_codeblock_callback( $block )
 692      {
 693          // set the offset if present - default : 0
 694          preg_match( '#line=("|\'?)([0-9]+?)(["\']?)$#', $block[2], $match );
 695          $offset = ( empty( $match[2] ) ? 0 : $match[2] - 1 );
 696  
 697          // set the language if present - default : code
 698          preg_match( '#lang=("|\'?)([^\1]+?)([\s"\']+?)#', $block[2], $match );
 699          $language = strtolower( ( empty( $match[2] ) ? 'code' : $match[2] ) );
 700  
 701          if( $code = trim( $block[4] ) )
 702          { // we have a code block
 703              // is the relevant language highlighter already cached?
 704              if( empty( $this->languageCache[ $language ] ) )
 705              { // lets attempt to load the language
 706                  $language_file = dirname(__FILE__).'/highlighters/'.$language.'.highlighter.php';
 707                  if( is_file( $language_file ) )
 708                  { // language class exists, lets load and cache an instance of it
 709                      require_once $language_file;
 710                      $class = 'am_'.$language.'_highlighter';
 711                      $this->languageCache[ $language ] = new $class( $this );
 712                  }
 713                  else
 714                  { // language class doesn't exists, fallback to default highlighter
 715                      $language = 'code';
 716                      if( empty( $this->languageCache[ $language ] ) )
 717                      { // lets attempt to load the default language
 718                          $language_file = dirname(__FILE__).'/highlighters/'.$language.'.highlighter.php';
 719                          if( is_file( $language_file ) )
 720                          { // default lanugage exists
 721                              require_once $language_file;
 722                              $class = 'am_'.$language.'_highlighter';
 723                              // add the language to the cache
 724                              $this->languageCache[ $language ] = new $class( $this );
 725                          }
 726                          else
 727                          { // if we hit this we might as well go to the pub ;)
 728                              // echo '***** error ***** no language or default language file is present';
 729                              return $code;
 730                          }
 731                      }
 732                  }
 733              }
 734              $this->languageCache[ $language ]->requested_language = $language;
 735              $this->languageCache[ $language ]->strict_mode = $this->Settings->get( 'strict' );
 736              $code = $this->languageCache[ $language ]->highlight_code( $code );
 737              // add the line numbers
 738              $code = $this->do_numbering( $code, $offset, $language );
 739          }
 740          return $code;
 741      }
