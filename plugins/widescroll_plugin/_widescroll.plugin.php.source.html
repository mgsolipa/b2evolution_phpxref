 142  	function AdminDisplayToolbar( & $params )
 143      {
 144          $allow_HTML = false;
 145  
 146          if( ! empty( $params['Item'] ) )
 147          { // Item is set, get Blog from post
 148              $edited_Item = & $params['Item'];
 149              $Blog = & $edited_Item->get_Blog();
 150              // We editing an Item, Check if HTML is allowed for the post type:
 151              $allow_HTML = $edited_Item->get_type_setting( 'allow_html' );
 152          }
 153  
 154          if( empty( $Blog ) )
 155          { // Item is not set, try global Blog
 156              global $Blog;
 157              if( empty( $Blog ) )
 158              { // We can't get a Blog, this way "apply_rendering" plugin collection setting is not available
 159                  return false;
 160              }
 161          }
 162  
 163          if( ! empty( $params['Comment'] ) )
 164          {    // We editing a Comment, Check if HTML is allowed for the comments of current Blog:
 165              $allow_HTML = $Blog->get_setting( 'allow_html_comment' );
 166          }
 167  
 168          if( ! $allow_HTML )
 169          {    // Only when HTML is allowed in post/comment
 170              return false;
 171          }
 172  
 173          $coll_setting_name = ( $params['target_type'] == 'Comment' ) ? 'coll_apply_comment_rendering' : 'coll_apply_rendering';
 174          $apply_rendering = $this->get_coll_setting( $coll_setting_name, $Blog );
 175          if( empty( $apply_rendering ) || $apply_rendering == 'never' )
 176          { // Plugin is not enabled for current case, so don't display a toolbar:
 177              return false;
 178          }
 179  
 180          // Append css styles for tinymce editor area
 181          global $tinymce_content_css;
 182          if( empty( $tinymce_content_css ) )
 183          { // Initialize first time
 184              $tinymce_content_css = array();
 185          }
 186          $tinymce_content_css[] = get_require_url( $this->get_plugin_url().'tinymce_editor.css', true, 'css' );
 187  
 188          // Print toolbar on screen
 189          return $this->DisplayCodeToolbar();
 190      }

 199  	function DisplayCommentToolbar( & $params )
 200      {
 201          if( ! empty( $params['Comment'] ) )
 202          { // Comment is set, get Blog from comment
 203              $Comment = & $params['Comment'];
 204              if( !empty( $Comment->item_ID ) )
 205              {
 206                  $comment_Item = & $Comment->get_Item();
 207                  $Blog = & $comment_Item->get_Blog();
 208              }
 209          }
 210  
 211          if( empty( $Blog ) )
 212          { // Comment is not set, try global Blog
 213              global $Blog;
 214              if( empty( $Blog ) )
 215              { // We can't get a Blog, this way "apply_comment_rendering" plugin collection setting is not available
 216                  return false;
 217              }
 218          }
 219  
 220          if( ! $Blog->get_setting( 'allow_html_comment' ) )
 221          {    // Only when HTML is allowed in comment
 222              return false;
 223          }
 224  
 225          $apply_rendering = $this->get_coll_setting( 'coll_apply_comment_rendering', $Blog );
 226          if( empty( $apply_rendering ) || $apply_rendering == 'never' )
 227          { // Plugin is not enabled for current case, so don't display a toolbar:
 228              return false;
 229          }
 230  
 231          // Print toolbar on screen
 232          return $this->DisplayCodeToolbar();
