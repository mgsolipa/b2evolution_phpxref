 142      /**
 143       * Get HTML template for stars
 144       *
 145       * @param array Matches
 146       * @return string HTML stars
 147       */
 148  	function get_stars_template( $matches )
 149      {
 150          if( empty( $matches ) )
 151          { // No stars found
 152              return;
 153          }
 154  
 155          $active_stars = $matches[1];
 156  
 157          if( ! empty( $matches[2] ) )
 158          { // Get a number of stars from content
 159              $number_stars = intval( substr( $matches[2], 1 ) );
 160          }
 161          if( empty( $number_stars ) )
 162          { // Use 5 stars by default
 163              $number_stars = 5;
 164          }
 165  
 166          $active_stars_max = floor( $active_stars );
 167          $percents = round( ( $active_stars - $active_stars_max ) * 100 );
 168          $template = '<span class="star_plugin"'.( $number_stars != 5 ? ' style="width:'.( $number_stars * 16 ).'px"' : '' ).'>';
 169          for( $s = 1; $s <= $number_stars; $s++ )
 170          {
 171              $attrs = '';
 172              if( $s > $active_stars_max )
 173              { // Class for empty stars
 174                  $attrs .= ' class="empty"';
 175              }
 176              $template .= '<span'.$attrs.'>';
 177              if( $s == $active_stars_max + 1 && $percents > 0 )
 178              { // Star with a percent fill
 179                  $template .= '<span style="width:'.$percents.'%">%</span>';
 180              }
 181              else
 182              {
 183                  $template .= $s <= $active_stars_max ? '*' : '-';
 184              }
 185              $template .= '</span>';
 186          }
 187          $template .= '</span>';
 188  
 189          return $template;
 190      }
