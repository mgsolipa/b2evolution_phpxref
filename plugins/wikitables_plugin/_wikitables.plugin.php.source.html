  77       * @param array Associative array of parameters
  78       *   'data': the data (by reference). You probably want to modify this.

  82  	function RenderItemAsHtml( & $params )
  83      {
  84          $content = & $params['data'];
  85  
  86          // Parse wiki tables
  87          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false || strstr( $content, '```' ) !== false )
  88          { // Call replace_content() on everything outside code/pre:
  89              $content = callback_on_non_matching_blocks( $content,
  90                  '~(```|<(code|pre)[^>]*>).*?(\1|</\2>)~is',
  91                  array( $this, 'parse_tables' ) );
  92          }
  93          else
  94          { // No code/pre blocks, replace on the whole thing
  95              $content = $this->parse_tables( $content );
  96          }
  97  
  98          return true;
  99      }

 102      /**
 103       * Parse tables
 104       *
 105       * @param string Content
 106       * @return string Content
 107       */
 108  	function parse_tables( $content )
 109      {
 110          $lines = explode( "\n", $content );
 111          $out = '';
 112          $td_history = array(); // Is currently a td tag open?
 113          $last_tag_history = array(); // Save history of last lag activated (td, th or caption)
 114          $tr_history = array(); // Is currently a tr tag open?
 115          $tr_attributes = array(); // history of tr attributes
 116          $has_opened_tr = array(); // Did this table open a <tr> element?
 117          $indent_level = 0; // indent level of the table
 118  
 119          foreach( $lines as $outLine )
 120          {
 121              $line = trim( $outLine );
 122  
 123              if( $line === '' )
 124              { // empty line, go to next line
 125                  $out .= $outLine . "\n";
 126                  continue;
 127              }
 128  
 129              $first_character = $line[0];
 130              $matches = array();
 131  
 132              if( preg_match( '/^(:*)\{\|(.*)$/', $line, $matches ) )
 133              { // First check if we are starting a new table
 134                  $indent_level = strlen( $matches[1] );
 135  
 136                  $attributes = Sanitizer::fixTagAttributes( $matches[2], 'table' );
 137  
 138                  $outLine = str_repeat( '<dl><dd>', $indent_level ) . "<table{$attributes}>";
 139                  array_push( $td_history, false );
 140                  array_push( $last_tag_history, '' );
 141                  array_push( $tr_history, false );
 142                  array_push( $tr_attributes, '' );
 143                  array_push( $has_opened_tr, false );
 144              }
 145              elseif( count( $td_history ) == 0 )
 146              { // Don't do any of the following
 147                  $out .= $outLine . "\n";
 148                  continue;
 149              }
 150              elseif( substr( $line, 0, 2 ) === '|}' )
 151              { // We are ending a table
 152                  $line = '</table>' . substr( $line, 2 );
 153                  $last_tag = array_pop( $last_tag_history );
 154  
 155                  if( !array_pop( $has_opened_tr ) )
 156                  {
 157                      $line = "<tr><td></td></tr>{$line}";
 158                  }
 159  
 160                  if( array_pop( $tr_history ) )
 161                  {
 162                      $line = "</tr>{$line}";
 163                  }
 164  
 165                  if( array_pop( $td_history ) )
 166                  {
 167                      $line = "</{$last_tag}>{$line}";
 168                  }
 169                  array_pop( $tr_attributes );
 170                  $outLine = $line . str_repeat( '</dd></dl>', $indent_level );
 171              }
 172              elseif( substr( $line, 0, 2 ) === '|-' )
 173              { // Now we have a table row
 174                  $line = preg_replace( '#^\|-+#', '', $line );
 175  
 176                  // Whats after the tag is now only attributes
 177                  $attributes = Sanitizer::fixTagAttributes( $line, 'tr' );
 178                  array_pop( $tr_attributes );
 179                  array_push( $tr_attributes, $attributes );
 180  
 181                  $line = '';
 182                  $last_tag = array_pop( $last_tag_history );
 183                  array_pop( $has_opened_tr );
 184                  array_push( $has_opened_tr, true );
 185  
 186                  if ( array_pop( $tr_history ) )
 187                  {
 188                      $line = '</tr>';
 189                  }
 190  
 191                  if( array_pop( $td_history ) )
 192                  {
 193                      $line = "</{$last_tag}>{$line}";
 194                  }
 195  
 196                  $outLine = $line;
 197                  array_push( $tr_history, false );
 198                  array_push( $td_history, false );
 199                  array_push( $last_tag_history, '' );
 200              }
 201              elseif ( $first_character === '|' || $first_character === '!' || substr( $line, 0, 2 ) === '|+' )
 202              { // This might be cell elements, td, th or captions
 203                  if( substr( $line, 0, 2 ) === '|+' )
 204                  {
 205                      $first_character = '+';
 206                      $line = substr( $line, 1 );
 207                  }
 208  
 209                  $line = substr( $line, 1 );
 210  
 211                  if( $first_character === '!' )
 212                  {
 213                      $line = str_replace( '!!', '||', $line );
 214                  }
 215  
 216                  // Split up multiple cells on the same line.
 217                  $cells = explode( '||', $line );
 218  
 219                  $outLine = '';
 220  
 221                  // Loop through each table cell
 222                  foreach( $cells as $cell )
 223                  {
 224                      $previous = '';
 225                      if( $first_character !== '+' )
 226                      {
 227                          $tr_after = array_pop( $tr_attributes );
 228                          if( !array_pop( $tr_history ) )
 229                          {
 230                              $previous = "<tr{$tr_after}>\n";
 231                          }
 232                          array_push( $tr_history, true );
 233                          array_push( $tr_attributes, '' );
 234                          array_pop( $has_opened_tr );
 235                          array_push( $has_opened_tr, true );
 236                      }
 237  
 238                      $last_tag = array_pop( $last_tag_history );
 239  
 240                      if( array_pop( $td_history ) )
 241                      {
 242                          $previous = "</{$last_tag}>\n{$previous}";
 243                      }
 244  
 245                      if( $first_character === '|' )
 246                      {
 247                          $last_tag = 'td';
 248                      }
 249                      elseif ( $first_character === '!' )
 250                      {
 251                          $last_tag = 'th';
 252                      }
 253                      elseif ( $first_character === '+' )
 254                      {
 255                          $last_tag = 'caption';
 256                      }
 257                      else
 258                      {
 259                          $last_tag = '';
 260                      }
 261  
 262                      array_push( $last_tag_history, $last_tag );
 263  
 264                      // A cell could contain both parameters and data
 265                      $cell_data = explode( '|', $cell, 2 );
 266  
 267                      if( strpos( $cell_data[0], '[[' ) !== false )
 268                      {
 269                          $cell = "{$previous}<{$last_tag}>{$cell}";
 270                      }
 271                      elseif ( count( $cell_data ) == 1 )
 272                      {
 273                          $cell = "{$previous}<{$last_tag}>{$cell_data[0]}";
 274                      }
 275                      else
 276                      {
 277                          $attributes = Sanitizer::fixTagAttributes( $cell_data[0], $last_tag );
 278                          $cell = "{$previous}<{$last_tag}{$attributes}>{$cell_data[1]}";
 279                      }
 280  
 281                      $outLine .= $cell;
 282                      array_push( $td_history, true );
 283                  }
 284              }
 285              $out .= $outLine . "\n";
 286          }
 287  
 288          // Closing open td, tr && table
 289          while( count( $td_history ) > 0 )
 290          {
 291              if( array_pop( $td_history ) )
 292              {
 293                  $out .= "</td>\n";
 294              }
 295              if( array_pop( $tr_history ) )
 296              {
 297                  $out .= "</tr>\n";
 298              }
 299              if( !array_pop( $has_opened_tr ) )
 300              {
 301                  $out .= "<tr><td></td></tr>\n";
 302              }
 303  
 304              $out .= "</table>\n";
 305          }
 306  
 307          if( substr( $out, -1 ) === "\n" )
 308          { // Remove trailing line-ending (b/c)
 309              $out = substr( $out, 0, -1 );
 310          }
 311  
 312          if( $out === "<table>\n<tr><td></td></tr>\n</table>" )
 313          { // special case: don't return empty table
 314              $out = '';
 315          }
 316  
 317          return $out;
 318      }
