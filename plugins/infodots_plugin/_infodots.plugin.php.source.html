 217      /**
 218       * Callback function to load a dot from NOT rendered content
 219       *
 220       * @param array Matches
 221       * @param boolean TRUE is used only to load dot without returning of tooltip template
 222       * @return string Empty string to don't display the dot template in content, It is printed out before image tag
 223       */
 224  	function load_infodot_from_source( $matches, $only_load_dot = false )
 225      {
 226          $link_ID = intval( $matches[3] );
 227  
 228          if( empty( $link_ID ) || empty( $matches ) || empty( $this->object_ID ) )
 229          { // Skip this incorrect match
 230              return;
 231          }
 232  
 233          $LinkCache = & get_LinkCache();
 234          $Link = & $LinkCache->get_by_ID( $link_ID, false, false );
 235          if( ! $Link )
 236          { // Inform about invalid Link ID
 237              return '<div style="color:#F00"><b>'.T_('Invalid Link ID').' - '.$matches[0].'</b></div>';
 238          }
 239  
 240          if( $this->dot_numbers === NULL )
 241          { // Init dot numbers array first time
 242              $this->dot_numbers = array();
 243          }
 244  
 245          if( ! isset( $this->dot_numbers[ $link_ID ] ) )
 246          { // Start to calculate number of the dots for current Link object
 247              $this->dot_numbers[ $link_ID ] = 1;
 248          }
 249  
 250          if( ! isset( $this->loaded_objects[ $this->object_ID ] ) )
 251          { // Load dots only once
 252              if( $this->dots === NULL )
 253              { // Init dots array first time
 254                  $this->dots = array();
 255              }
 256  
 257              if( ! isset( $this->dots[ $link_ID ] ) )
 258              { // Init sub array for each Link
 259                  $this->dots[ $link_ID ] = array();
 260              }
 261  
 262              // Add dot
 263              $this->dots[ $link_ID ][] = array(
 264                      'x' => intval( $matches[4] ), // Left
 265                      'y' => intval( $matches[5] ), // Top
 266                  );
 267          }
 268  
 269          if( $only_load_dot )
 270          { // Exit here to don't execute a code below
 271              return;
 272          }
 273  
 274          $dot_num = $this->dot_numbers[ $link_ID ];
 275          if( empty( $matches[6] ) )
 276          { // No defined width
 277              $tooltip_width = '';
 278          }
 279          else
 280          { // Set css style for width
 281              $tooltip_width = substr( $matches[6], 1 );
 282              $tooltip_width = ( strlen( intval( $tooltip_width ) ) == strlen( $tooltip_width ) ? $tooltip_width.'px' : $tooltip_width );
 283              $tooltip_width = ' style="width:'.$tooltip_width.'"';
 284          }
 285          $dot_xy = ' xy="'.$this->dots[ $link_ID ][ $dot_num - 1 ]['x'].':'.$this->dots[ $link_ID ][ $dot_num - 1 ]['y'].'"';
 286  
 287          $this->dot_numbers[ $link_ID ]++;
 288  
 289          // Print this element that will be used for tooltip of the dot
 290          return '<div class="infodots_info" id="infodot_'.$link_ID.'_'.$dot_num.'"'.$dot_xy.$tooltip_width.'>'
 291                  .balance_tags( $matches[7] )
 292              .'</div>'."\n";
 293      }

 313      /**
 314       * Render the dots before <img> tag
 315       *
 316       * @param array Associative array of parameters. $params['File'] - attachment, $params['data'] - output
 317       * @param string Content of the Item/Comment
 318       */
 319  	function render_infodots( & $params, $content )
 320      {
 321          if( empty( $params['File'] ) || empty( $params['Link'] ) )
 322          { // Check input data
 323              return;
 324          }
 325  
 326          $File = $params['File'];
 327          $Link = $params['Link'];
 328  
 329          if( ! $File->is_image() )
 330          { // This plugin works only with image files
 331              return;
 332          }
 333  
 334          if( ( $LinkOwner = & $Link->get_LinkOwner() ) === false || ( $Blog = & $LinkOwner->get_Blog() ) === false )
 335          { // Couldn't get Blog object
 336              return;
 337          }
 338  
 339          global $thumbnail_sizes;
 340          $thumbnail_width = isset( $thumbnail_sizes[ $params['image_size'] ] ) ? $thumbnail_sizes[ $params['image_size'] ][1] : 0;
 341  
 342          if( $File->get_image_size( 'width' ) < $this->get_coll_setting( 'coll_min_width', $Blog ) ||
 343              $thumbnail_width < $this->get_coll_setting( 'coll_min_width', $Blog ) )
 344          { // Don't draw a dot on image if width is less than setting value
 345              return;
 346          }
 347  
 348          if( ! isset( $this->loaded_objects[ $this->object_ID ] ) )
 349          { // Load the info dots if they were not loaded before
 350              replace_content_outcode( '#<div class="infodots_info" id="infodot_(\d+)_(\d+)" xy="(-?\d+):(-?\d+)"[^>]*>(.+?)</div>#is', array( $this, 'load_infodot_from_rendered_content' ), $content, 'replace_content_callback' );
 351              $this->loaded_objects[ $this->object_ID ] = 1;
 352          }
 353  
 354          if( empty( $this->dots[ $Link->ID ] ) )
 355          { // No dots for this Link
 356              return;
 357          }
 358  
 359          $before_image = '<div class="infodots_image">'."\n";
 360          foreach( $this->dots[ $Link->ID ] as $d => $dot )
 361          { // Init html element for each dot
 362              $before_image .= '<div class="infodots_dot" rel="infodot_'.$Link->ID.'_'.( $d + 1 ).'" style="left:'.$dot['x'].'px;top:'.$dot['y'].'px"></div>'."\n";
 363          }
 364  
 365          // Append info dots html to current image tag
 366          $params['before_image'] = $params['before_image'].$before_image;
 367          $params['after_image'] = '</div>'.$params['after_image'];
 368      }
