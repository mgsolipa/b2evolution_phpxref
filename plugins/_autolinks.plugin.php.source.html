 227      {
 228          global $plugins_path;
 229  
 230          if( !isset($this->link_array[0]) )
 231          {    // global defs NOT already loaded
 232              $this->link_array[0] = array();
 233  
 234              if( $this->Settings->get( 'autolink_defs_default' ) )
 235              {    // Load defaults:
 236                  $this->read_csv_file( $plugins_path.'autolinks_plugin/definitions.default.txt', 0 );
 237              }
 238              if( $this->Settings->get( 'autolink_defs_local' ) )
 239              {    // Load local user defintions:
 240                  $this->read_csv_file( $plugins_path.'autolinks_plugin/definitions.local.txt', 0 );
 241              }
 242              $text = $this->Settings->get( 'autolink_defs_db', 0 );
 243              if( !empty($text) )
 244              {    // Load local user defintions:
 245                  $this->read_textfield( $text, 0 );
 246              }
 247          }
 248  
 249          // load defs for current blog:
 250          $coll_ID = !empty( $Blog ) ? $Blog->ID : 0;
 251          if( !isset($this->link_array[$coll_ID]) )
 252          {    // This blog is not loaded yet:
 253              $this->link_array[$coll_ID] = array();
 254              $text = $this->get_coll_setting( 'autolink_defs_coll_db', $Blog, $allow_null_blog );
 255              if( !empty($text) )
 256              {    // Load local user defintions:
 257                  $this->read_textfield( $text, $coll_ID );
 258              }
 259          }
 260  
 261          // Prepare working link array:
 262          $this->replacement_link_array = array_merge( $this->link_array[0], $this->link_array[$coll_ID] );
 263      }

 314  	function read_line( $data, $coll_ID )
 315      {
 316          if( empty( $data[0] ) )
 317          {    // Skip empty and comment lines
 318              return;
 319          }
 320  
 321          $word = $data[0];
 322          $url = isset( $data[3] ) ? $data[3] : NULL;
 323          if( $url == '-' || empty( $url ) )
 324          {    // Remove URL (useful to remove some defs on a specific site):
 325              unset( $this->link_array[0][$word] );
 326              unset( $this->link_array[$coll_ID][$word] );
 327          }
 328          else
 329          {
 330              $this->link_array[$coll_ID][$word] = array( $data[1], $url );
 331          }
 332      }

 397  	function render_content( & $content, $item_Blog = NULL, $allow_null_blog = false )
 398      {
 399          // Prepare existing links
 400          $content = $this->prepare_existing_links( $content, $item_Blog, $allow_null_blog );
 401  
 402          // reset already linked usernames
 403          $this->already_linked_usernames = array();
 404          if( ( !empty( $item_Blog ) || $allow_null_blog ) && $this->get_coll_setting( 'autolink_username', $item_Blog, $allow_null_blog ) )
 405          {    // Replace @usernames with user identity link
 406              $content = replace_content_outcode( '#@([A-Za-z0-9_.]+)#i', '@', $content, array( $this, 'replace_usernames' ) );
 407          }
 408  
 409          // load global defs
 410          $this->load_link_array( $item_Blog, $allow_null_blog );
 411  
 412          // reset already linked:
 413          $this->already_linked_array = array();
 414          if( preg_match_all( '|[\'"](http://[^\'"]+)|i', $content, $matches ) )
 415          {    // There are existing links:
 416              $this->already_linked_array = $matches[1];
 417          }
 418  
 419          $link_attrs = '';
 420          if( ( !empty( $item_Blog ) || $allow_null_blog ) && $this->get_coll_setting( $this->setting_nofollow_explicit, $item_Blog, $allow_null_blog ) )
 421          {    // Add attribute rel="nofollow" for auto-links
 422              $link_attrs .= ' rel="nofollow"';
 423          }
 424  
 425          if( $this->Settings->get( 'autolink_urls' ) )
 426          {    // First, make the URLs clickable:
 427              $content = make_clickable( $content, '&amp;', 'make_clickable_callback', $link_attrs );
 428          }
 429  
 430          if( !empty( $this->replacement_link_array ) )
 431          {    // Make the desired remaining terms/definitions clickable:
 432              $content = make_clickable( $content, '&amp;', array( $this, 'make_clickable_callback' ), $link_attrs );
 433          }
 434  
 435          return true;
 436      }

 453      /**
 454       * Callback function for {@link make_clickable()}.
 455       *
 456       * @param string Text
 457       * @param string Url delimeter
 458       * @return string The clickable text.
 459       */
 460  	function make_clickable_callback( $text, $moredelim = '&amp;' )
 461      {
 462          global $evo_charset;
 463  
 464          $regexp_modifier = '';
 465          if( $evo_charset == 'utf-8' )
 466          { // Add this modifier to work with UTF-8 strings correctly
 467              $regexp_modifier = 'u';
 468          }
 469  
 470          // Previous word in lower case format
 471          $this->previous_lword = null;
 472          // Previous word was already used/converted to a link
 473          $this->previous_used = false;
 474  
 475          // Optimization: Check if the text contains words from the replacement links strings, and call replace callback only if there is at least one word which needs to be replaced.
 476          $text_words = explode( ' ', utf8_strtolower( $text ) );
 477          foreach( $text_words as $text_word )
 478          { // Trim the signs [({/ from start and the signs ])}/.,:;!? from end of each word
 479              $clear_word = preg_replace( '#^[\[\({/]?([@\p{L}0-9_\-\.]{3,})[\.,:;!\?\]\)}/]?$#i', '$1', $text_word );
 480              if( $clear_word != $text_word )
 481              { // Append a clear word to array if word has the punctuation signs
 482                  $text_words[] = $clear_word;
 483              }
 484          }
 485          // Check if a content has at least one definition to make an url from word
 486          $text_contains_replacement = ( count( array_intersect( $text_words, array_keys( $this->replacement_link_array ) ) ) > 0 );
 487          if( $text_contains_replacement )
 488          { // Find word with 3 characters at least:
 489              $text = preg_replace_callback( '#(^|\s|[(),;\[{/])([@\p{L}0-9_\-\.]{3,})([\.,:;!\?\]\)}/]?)#i'.$regexp_modifier, array( & $this, 'replace_callback' ), $text );
 490          }
 491  
 492          // Cleanup words to be deleted:
 493          $text = preg_replace( '/[@\p{L}0-9_\-]+\s*==!#DEL#!==/i'.$regexp_modifier, '', $text );
 494  
 495          return $text;
 496      }

 499      /**
 500       * This is the 2nd level of callback!!
 501       *
 502       * @param array The matches of regexp:
 503       *     1 => punctuation signs before word
 504       *     2 => a clear word without punctuation signs
 505       *     3 => punctuation signs after word
 506       */
 507  	function replace_callback( $matches )
 508      {
 509          global $Blog;
 510  
 511          $link_attrs = '';
 512          if( !empty( $Blog ) && $this->get_coll_setting( $this->setting_nofollow_auto, $Blog ) )
 513          {    // Add attribute rel="nofollow" for auto-links
 514              $link_attrs .= ' rel="nofollow"';
 515          }
 516  
 517          $before_word = $matches[1];
 518          $word = $matches[2];
 519          $after_word = $matches[3];
 520          if( substr( $word, -1 ) == '.' )
 521          { // If word has a dot in the end
 522              $word = substr( $word, 0, -1 );
 523              $after_word = '.'.$after_word;
 524          }
 525          $lword = utf8_strtolower( $word );
 526          $r = $before_word.$word.$after_word;
 527  
 528          if( isset( $this->replacement_link_array[ $lword ] ) )
 529          { // There is an autolink definition with the current word
 530              // An optional previous required word (allows to create groups of 2 words)
 531              $previous = $this->replacement_link_array[ $lword ][0];
 532              // Url for current word
 533              $url = 'http://'.$this->replacement_link_array[ $lword ][1];
 534  
 535              if( in_array( $url, $this->already_linked_array ) || in_array( $lword, $this->already_linked_usernames ) )
 536              { // Do not repeat link to same destination:
 537                  // pre_dump( 'already linked:'. $url );
 538                  // save previous word in original and lower case format with the after word signs
 539                  $this->previous_word = $word.$after_word;
 540                  $this->previous_lword = $lword.$after_word;
 541                  $this->previous_used = false;
 542                  return $r;
 543              }
 544  
 545              if( !empty( $previous ) )
 546              { // This definitions is a group of two word separated with space
 547                  if( $this->previous_used || ( $this->previous_lword != $previous ) )
 548                  { // We do not have the required previous word or it was already used to another autolink definition
 549                      // pre_dump( 'previous word does not match', $this->previous_lword, $previous );
 550                      // save previous word in original and lower case format with the after word signs
 551                      $this->previous_word = $word.$after_word;
 552                      $this->previous_lword = $lword.$after_word;
 553                      $this->previous_used = false;
 554                      return $r;
 555                  }
 556                  $r = '==!#DEL#!==<a href="'.$url.'"'.$link_attrs.'>'.$this->previous_word.' '.$word.'</a>'.$after_word;
 557              }
 558              else
 559              { // Single word
 560                  $r = $before_word.'<a href="'.$url.'"'.$link_attrs.'>'.$word.'</a>'.$after_word;
 561              }
 562  
 563              // Make sure we don't link to same destination twice in the same text/post:
 564              $this->already_linked_array[] = $url;
 565              // Mark that the previous word was already converted to a link
 566              $this->previous_used = true;
 567          }
 568          else
 569          { // Mark that the previous word was NOT converted to a link
 570              $this->previous_used = false;
 571          }
 572  
 573          // save previous word in original and lower case format with the after word signs
 574          // Note: after_word signs are important to be saved because in case of autlink definitions with two words the first word must have exact matching at the end!
 575          $this->previous_word = $word.$after_word;
 576          $this->previous_lword = $lword.$after_word;
 577  
 578          return $r;
 579      }
