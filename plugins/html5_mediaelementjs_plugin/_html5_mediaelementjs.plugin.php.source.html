 203      /**
 204       * Event handler: Called when displaying item attachment.
 205       *
 206       * @param array Associative array of parameters. $params['File'] - attachment, $params['data'] - output
 207       * @param boolean TRUE - when render in comments
 208       * @return boolean true if plugin rendered this attachment
 209       */
 210  	function RenderItemAttachment( & $params, $in_comments = false )
 211      {
 212          $File = $params['File'];
 213  
 214          if( ! $this->is_flp_video( $File ) )
 215          {
 216              return false;
 217          }
 218  
 219          $Item = & $params['Item'];
 220          $item_Blog = $Item->get_Blog();
 221  
 222          if( $File->exists() )
 223          {
 224              /**
 225               * @var integer A number to assign each video player new id attribute
 226               */
 227              global $html5_mediaelementjs_number;
 228              $html5_mediaelementjs_number++;
 229  
 230              if( $placeholder_File = & $Item->get_placeholder_File( $File ) )
 231              { // Display placeholder/poster when image file is linked to the Item with same name as current video File
 232                  $video_placeholder_attr = ' poster="'.$placeholder_File->get_url().'"';
 233              }
 234              else
 235              { // No placeholder for current video File
 236                  $video_placeholder_attr = '';
 237              }
 238  
 239              if( $in_comments )
 240              {
 241                  $params['data'] .= '<div style="clear: both; height: 0px; font-size: 0px"></div>';
 242              }
 243  
 244              $params['data'] .= '<div class="mediajs_block">';
 245  
 246              $params['data'] .= '<video class="html5_mediaelementjs_video '.$this->get_skin_class().'" id="html5_mediaelementjs_'.$html5_mediaelementjs_number.'"'.$video_placeholder_attr.'>'.
 247                  '<source src="'.$File->get_url().'" type="'.$this->get_video_mimetype( $File ).'" align="center" />'.
 248              '</video>';
 249  
 250              if( $File->get( 'desc' ) != '' && $this->get_coll_setting( 'disp_caption', $item_Blog ) )
 251              { // Display caption
 252                  $params['data'] .= '<div class="mediajs_text">'.$File->get( 'desc' ).'</div>';
 253              }
 254  
 255              if( $this->get_coll_setting( 'allow_download', $item_Blog ) )
 256              { // Allow to download the video files
 257                  $params['data'] .= '<div class="mediajs_text"><a href="'.$File->get_url().'">'.T_('Download this video').'</a></div>';
 258              }
 259  
 260              $params['data'] .= '</div>';
 261  
 262              return true;
 263          }
 264  
 265          return false;
 266      }

 283      /**
 284       * Get a list of the skins
 285       *
 286       * @return array Skins
 287       */
 288  	function get_skins_list()
 289      {
 290          $skins_path = dirname( $this->classfile_path ).'/skins';
 291  
 292          $skins = array();
 293          // Set this skin permanently, because it is a default skin
 294          $skins['default'] = 'default';
 295  
 296          $files = scandir( $skins_path );
 297          foreach( $files as $file )
 298          {
 299              if( $file != '.' && $file != '..' && is_dir( $skins_path.'/'.$file ) )
 300              {    // Use folder name as skin name
 301                  $skins[ $file ] = $file;
 302              }
 303          }
 304  
 305          return $skins;
 306      }

 339              {    // Require css file only if it exists

 345      /**
 346       * Get video mimetype
 347       *
 348       * @param object File
 349       * @return string Mimetype
 350       */
 351  	function get_video_mimetype( $File )
 352      {
 353          switch( $File->get_ext() )
 354          {
 355              case 'flv':
 356              case 'f4v':
 357                  $mimetype = 'video/flv';
 358                  break;
 359  
 360              case 'm4v':
 361                  $mimetype = 'video/m4v';
 362                  break;
 363  
 364              case 'ogv':
 365                  $mimetype = 'video/ogg';
 366                  break;
 367  
 368              case 'webm':
 369                  $mimetype = 'video/webm';
 370                  break;
 371  
 372              case 'mp4':
 373              default:
 374                  $mimetype = 'video/mp4';
 375                  break;
 376          }
 377  
 378          return $mimetype;
 379      }
