 226      /**
 227       * Get country by IP address
 228       *
 229       * @param string IP in format xxx.xxx.xxx.xxx
 230       * @return object Country
 231       */
 232  	function get_country_by_IP( $IP )
 233      {
 234          if( $this->status != 'enabled' || ! file_exists( $this->geoip_file_path ) )
 235          {
 236              return false;
 237          }
 238  
 239          if( function_exists('geoip_country_code_by_name') )
 240          {    // GeoIP extension
 241  
 242              // Get country code by user IP address
 243              $country_code = @geoip_country_code_by_name($IP);
 244          }
 245          else
 246          {    // Include GeoIP API
 247              require_once( dirname( __FILE__ ).'/geoip.inc' );
 248  
 249              // Open GeoIP database
 250              $GeoIP = geoip_open( $this->geoip_file_path, GEOIP_STANDARD );
 251  
 252              // Get country code by user IP address
 253              $country_code = geoip_country_code_by_addr( $GeoIP, $IP );
 254  
 255              // Close GeoIP DB
 256              geoip_close( $GeoIP );
 257          }
 258  
 259          if( ! $country_code )
 260          {    // No found country with current IP address
 261              return false;
 262          }
 263  
 264          global $DB;
 265  
 266          // Get country ID by code
 267          $SQL = new SQL();
 268          $SQL->SELECT( 'ctry_ID' );
 269          $SQL->FROM( 'T_regional__country' );
 270          $SQL->WHERE( 'ctry_code = '.$DB->quote( strtolower( $country_code ) ) );
 271          $country_ID = $DB->get_var( $SQL->get() );
 272  
 273          if( !$country_ID )
 274          { // No found country in the b2evo DB
 275              return false;
 276          }
 277  
 278          // Load Country class (PHP4):
 279          load_class( 'regional/model/_country.class.php', 'Country' );
 280  
 281          $CountryCache = & get_CountryCache();
 282          $Country = $CountryCache->get_by_ID( $country_ID, false );
 283  
 284          return $Country;
 285      }

 430  	function DisplayRegisterFormBefore( & $params )
 431      {
 432          global $Settings;
 433  
 434          $registration_require_country = (bool)$Settings->get('registration_require_country');
 435          if( !$registration_require_country )
 436          {    // Country is not required on registration form. Exit here.
 437              return;
 438          }
 439  
 440          $detect_registration = $this->Settings->get( 'detect_registration' );
 441          if( $detect_registration == 'no' )
 442          {    // No detect country on registration
 443              return;
 444          }
 445  
 446          $Country = $this->get_country_by_IP( get_ip_list( true ) );
 447  
 448          if( !$Country )
 449          {    // No found country by IP address
 450              return;
 451          }
 452  
 453          switch( $detect_registration )
 454          {
 455              case 'auto':
 456                  // Auto select current country in list
 457                  $country = param( 'country', 'integer', 0 );
 458                  if( empty( $country ) )
 459                  {    // Set country ID if user didn't select country yet
 460                      set_param( 'country', $Country->ID );
 461                  }
 462                  break;
 463  
 464              case 'hide':
 465                  // Hide country selector if a country has been detected
 466                  if( ! isset( $params['Form'] ) )
 467                  {    // there's no Form where we add to, but we create our own form:
 468                      $Form = new Form( regenerate_url() );
 469  
 470                  }
 471                  else
 472                  {
 473                      $Form = & $params['Form'];
 474                  }
 475  
 476                  // Disable this setting temporary to hide a select list with countries
 477                  $Settings->set( 'registration_require_country', 0 );
 478  
 479                  // Append a hidden input element with autodetected country ID
 480                  $Form->hidden( 'country', $Country->ID );
 481                  break;
 482          }
 483      }

 524      /**
 525       * Event handler: Called when displaying the block in the "Tools" menu.
 526       *
 527       * @see Plugin::AdminToolPayload()
 528       */
 529  	function AdminToolPayload( $params )
 530      {
 531          $action = param_action();
 532  
 533          echo '<a name="geoip" style="position:relative;top:-60px"></a>';
 534  
 535          switch( $action )
 536          {
 537              case 'geoip_download':
 538                  // Display a process of downloading of GeoIP.dat
 539  
 540                  global $admin_url;
 541  
 542                  $this->print_tool_log( sprintf( T_('Downloading GeoIP.dat file from the url: %s ...'),
 543                      '<a href="'.$this->geoip_download_url.'" target="_blank">'.$this->geoip_download_url.'</a>' ) );
 544  
 545                  // DOWNLOAD:
 546                  $gzip_contents = fetch_remote_page( $this->geoip_download_url, $info, 1800 );
 547                  if( $gzip_contents === false || $info['status'] != 200 )
 548                  { // Downloading is Failed
 549                      if( empty( $info['error'] ) )
 550                      { // Some unknown error
 551                          $this->print_tool_log( T_( 'The URL is not available. It may correspond to an old version of the GeoIP.dat file.' ), 'error' );
 552                      }
 553                      else
 554                      { // Display an error of request
 555                          $this->print_tool_log( T_( $info['error'] ), 'error' );
 556                      }
 557                      break;
 558                  }
 559                  $this->print_tool_log( ' OK.<br />' );
 560  
 561                  $plugin_dir = dirname( __FILE__ );
 562                  if( ! is_writable( $plugin_dir ) )
 563                  { // Check the write rights
 564                      $this->print_tool_log( sprintf( T_('Plugin folder %s must be writable to receive GeoIP.dat. Please fix the write permissions and try again.'), '<b>'.$plugin_dir.'</b>' ), 'error' );
 565                      break;
 566                  }
 567  
 568                  $gzip_file_name = explode( '/', $this->geoip_download_url );
 569                  $gzip_file_name = $gzip_file_name[ count( $gzip_file_name ) - 1 ];
 570                  $gzip_file_path = $plugin_dir.'/'.$gzip_file_name;
 571  
 572                  if( ! save_to_file( $gzip_contents, $gzip_file_path, 'w' ) )
 573                  { // Impossible to save file...
 574                      $this->print_tool_log( sprintf( T_( 'Unable to create file: %s' ), '<b>'.$gzip_file_path.'</b>' ), 'error' );
 575  
 576                      if( file_exists( $gzip_file_path ) )
 577                      { // Remove file from disk
 578                          if( ! @unlink( $gzip_file_path ) )
 579                          { // File exists without the write rights
 580                              $this->print_tool_log( sprintf( T_( 'Unable to remove file: %s' ), '<b>'.$gzip_file_path.'</b>' ), 'error' );
 581                          }
 582                      }
 583                      break;
 584                  }
 585  
 586                  // UNPACK:
 587                  $this->print_tool_log( sprintf( T_('Extracting of the file %s...'), '<b>'.$gzip_file_path.'</b>' ) );
 588  
 589                  if( ! function_exists( 'gzopen' ) )
 590                  { // No extension
 591                      $this->print_tool_log( T_( 'There is no \'zip\' or \'zlib\' extension installed!' ), 'error' );
 592                      break;
 593                  }
 594  
 595                  if( ! ( $gzip_handle = @gzopen( $gzip_file_path, 'rb' ) ) )
 596                  { // Try to open gzip file
 597                      $this->print_tool_log( T_('Could not open the source file!'), 'error' );
 598                      break;
 599                  }
 600  
 601                  if( ! ( $out_handle = @fopen( $plugin_dir.'/'.str_replace( '.gz', '', $gzip_file_name ), 'w' ) ) )
 602                  {
 603                      $this->print_tool_log( sprintf( T_('The file %s cannot be written to disk. Please check the filesystem permissions.'), '<b>'.$plugin_dir.'/'.str_replace( '.gz', '', $gzip_file_name ).'</b>' ), 'error' );
 604                      break;
 605                  }
 606  
 607                  $i = 0;
 608                  while( ! gzeof( $gzip_handle ) )
 609                  { // Extract file by 4Kb
 610                      fwrite( $out_handle, gzread( $gzip_handle, 4096 ) );
 611                      if( $i == 100 )
 612                      { // Display the process dots after each 400Kb
 613                          $this->print_tool_log( ' .' );
 614                          $i = 0;
 615                      }
 616                      $i++;
 617                  }
 618                  $this->print_tool_log( ' OK.<br />' );
 619  
 620                  fclose( $out_handle );
 621                  gzclose( $gzip_handle );
 622  
 623                  $this->print_tool_log( sprintf( T_('Remove gzip file %s...'), '<b>'.$gzip_file_path.'</b>' ) );
 624                  if( @unlink( $gzip_file_path ) )
 625                  {
 626                      $this->print_tool_log( ' OK.<br />' );
 627                  }
 628                  else
 629                  { // Failed on removing
 630                      $this->print_tool_log( sprintf( T_('Impossible to remove the file %s. You can do it manually.'), $gzip_file_path ), 'warning' );
 631                  }
 632  
 633                  // Success message
 634                  $this->print_tool_log( '<br /><span class="text-success">'.sprintf( T_('%s file was downloaded successfully.'), 'GeoIP.dat' ).'</span>' );
 635  
 636                  // Try to enable plugin automatically:
 637                  global $Plugins;
 638                  $enable_return = $this->BeforeEnable();
 639                  if( $enable_return === true )
 640                  { // Success enabling
 641                      $this->print_tool_log( '<br /><span class="text-success">'.T_('The plugin has been enabled.').'</span>' );
 642  
 643                      if( $this->status != 'enabled' )
 644                      { // Enable this plugin automatically:
 645                          $Plugins->set_Plugin_status( $this, 'enabled' );
 646                      }
 647                  }

 697       * Event handler: Called when handling actions for the "Tools" menu.

 703  	function AdminToolAction()
 704      {
 705          $action = param_action();
 706  
 707          if( !empty( $action ) )
 708          {    // If form is submitted
 709              global $DB;
 710  
 711              switch( $action )
 712              {
 713                  case 'geoip_find_country':
 714                  // Find and Assign Registration Country for all Users
 715  
 716                      $SQL = new SQL( 'Find all users without registration country' );
 717                      $SQL->SELECT( 'user_ID, uset_value' );
 718                      $SQL->FROM( 'T_users' );
 719                      $SQL->FROM_add( 'LEFT JOIN T_users__usersettings
 720                           ON user_ID = uset_user_ID
 721                          AND uset_name = "created_fromIPv4"' );
 722                      $SQL->WHERE( 'user_reg_ctry_ID IS NULL' );
 723                      $users = $DB->get_assoc( $SQL->get() );
 724  
 725                      $total_users = count( $users );
 726                      if( $total_users == 0 )
 727                      {    // No users
 728                          $this->text_from_AdminTabAction = T_('No found users without registration country.');
 729                          break;
 730                      }
 731                      $count_nofound_country = 0;
 732  
 733                      $users_report = '';
 734                      foreach( $users as $user_ID => $created_fromIPv4 )
 735                      {
 736                          $users_report .= sprintf( T_('User #%s, IP:%s' ), $user_ID, int2ip( $created_fromIPv4 ) );
 737                          if( empty( $created_fromIPv4 ) )
 738                          {    // No defined IP, Skip this user
 739                              $count_nofound_country++;

 741                              continue;
 742                          }

 745                          $Country = $this->get_country_by_IP( int2ip( $created_fromIPv4 ) );

 747                          if( !$Country )

 751                              continue;
 752                          }
 753  
 754                          // Update user's registration country
 755                          $DB->query( 'UPDATE T_users
 756                                    SET user_reg_ctry_ID = '.$DB->quote( $Country->ID ).'
 757                                  WHERE user_ID = '.$DB->quote( $user_ID ) );
 758  
 759                          // Move user to suspect group by Country ID
 760                          antispam_suspect_user_by_country( $Country->ID, $user_ID );
 761  
 762                          $users_report .= ' - '.sprintf( T_('Country: <b>%s</b>'), $Country->get( 'name' ) ).'<br />';
 763                      }
 764  
 765                      $this->text_from_AdminTabAction = '<div>'.sprintf( T_('Count of users without registration country: <b>%s</b>' ), $total_users ).'</div>';
 766                      if( $count_nofound_country > 0 )
 767                      {    // If some users have IP address with unknown country
 768                          $this->text_from_AdminTabAction .= '<div>'.sprintf( T_('Count of users whose country could not be identified: <b>%s</b>' ), $count_nofound_country ).'</div>';
 769                      }
 770                      $this->text_from_AdminTabAction .= '<div style="margin-top:20px">'.$users_report.'</div>';
