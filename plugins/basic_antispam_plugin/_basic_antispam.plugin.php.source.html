 109  	function GetSpamKarmaForComment( & $params )
 110      {
 111          $max_comments = $this->Settings->get('max_number_of_links_feedback');
 112          if( $max_comments != -1 )
 113          { // not deactivated:
 114              $count = preg_match_all( '~(https?|ftp)://~i', $params['Comment']->content, $matches );
 115  
 116              if( $count > $max_comments )
 117              {
 118                  return 100;
 119              }
 120  
 121              if( $count == 0 )
 122              {
 123                  return 0;
 124              }
 125  
 126              return (100/$max_comments) * $count;
 127          }
 128      }

 260      /**
 261       * Do we want to apply rel="nofollow" tag?
 262       *
 263       * @return boolean
 264       */
 265  	function apply_nofollow( & $data, $Comment )
 266      {
 267          global $localtimenow;
 268  
 269          $hours = $this->Settings->get('nofollow_for_hours'); // 0=never, -1 always, otherwise for x hours
 270  
 271          if( $hours == 0 )
 272          { // "never"
 273              return;
 274          }
 275  
 276          if( $hours > 0 // -1 is "always"
 277              && mysql2timestamp( $Comment->date ) <= ( $localtimenow - $hours*3600 ) )
 278          {
 279              return;
 280          }
 281  
 282          $data = preg_replace_callback( '~(<a\s)([^>]+)>~i', create_function( '$m', '
 283                  if( preg_match( \'~\brel=([\\\'"])(.*?)\1~\', $m[2], $match ) )
 284                  { // there is already a rel attrib:
 285                      $rel_values = explode( " ", $match[2] );
 286  
 287                      if( ! in_array( \'nofollow\', $rel_values ) )
 288                      {
 289                          $rel_values[] = \'nofollow\';
 290                      }
 291  
 292                      return $m[1]
 293                          .preg_replace(
 294                              \'~\brel=([\\\'"]).*?\1~\',
 295                              \'rel=$1\'.implode( " ", $rel_values ).\'$1\',
 296                              $m[2] )
 297                          .">";
 298                  }
 299                  else
 300                  {
 301                      return $m[1].$m[2].\' rel="nofollow">\';
 302                  }' ), $data );
 303      }

 308          if( ($newstring = @preg_replace( '~(.)\\1{3,}~u', '$1$1$1', $str )) === NULL )

 312          return $newstring;

 335      /**
 336       * Check if our Host+URI is in the referred page, preferrably through
 337       * {@link register_shutdown_function()}.
 338       *
 339       * @return boolean true, if we handle {@link Hit::record_the_hit() recording of the Hit} ourself
 340       */
 341  	function AppendHitLog( & $params )
 342      {
 343          $Hit = & $params['Hit'];
 344  
 345          if( $Hit->referer_type != 'referer' )
 346          {
 347              return false;
 348          }
 349  
 350          if( function_exists( 'register_shutdown_function' ) )
 351          { // register it as a shutdown function, because it will be slow!
 352              $this->debug_log( 'AppendHitLog: loading referering page.. (through register_shutdown_function())' );
 353  
 354              register_shutdown_function( array( &$this, 'double_check_referer' ), $Hit->referer ); // this will also call Hit::record_the_hit()
 355          }
 356          else
 357          {
 358              // flush now, so that the meat of the page will get shown before it tries to check back against the refering URL.
 359              evo_flush();
 360  
 361              $this->debug_log( 'AppendHitLog: loading referering page..' );
 362  
 363              $this->double_check_referer($Hit->referer); // this will also call Hit::record_the_hit()
 364          }
 365  
 366          return true; // we handle recording
 367      }

 391      /**
 392       * Check the content of a given URL (referer), if the requested URI (with different hostname variations)
 393       * is present.
 394       *
 395       * @todo Use DB cache to avoid checking the same page again and again! (Plugin DB table)
 396       *
 397       * @param string
 398       * @param string URI to append to matching pattern for hostnames
 399       * @return boolean
 400       */
 401  	function is_referer_linking_us( $referer, $uri )
 402      {
 403          global $misc_inc_path, $lib_subdir, $ReqHost;
 404  
 405          if( empty($referer) )
 406          {
 407              return false;
 408          }
 409  
 410          // Load page content (max. 500kb), using fsockopen:
 411          $url_parsed = @parse_url($referer);
 412          if( ! $url_parsed )
 413          {
 414              return false;
 415          }
 416          if( empty($url_parsed['scheme']) ) {
 417              $url_parsed = parse_url('http://'.$referer);
 418          }
 419  
 420          $host = $url_parsed['host'];
 421          $port = ( empty($url_parsed['port']) ? 80 : $url_parsed['port'] );
 422          $path = empty($url_parsed['path']) ? '/' : $url_parsed['path'];
 423          if( ! empty($url_parsed['query']) )
 424          {
 425              $path .= '?'.$url_parsed['query'];
 426          }
 427  
 428          $fp = @fsockopen($host, $port, $errno, $errstr, 30);
 429          if( ! $fp )
 430          { // could not access referring page
 431              $this->debug_log( 'is_referer_linking_us(): could not access &laquo;'.$referer.'&raquo; (host: '.$host.'): '.$errstr.' (#'.$errno.')' );
 432              return false;
 433          }
 434  
 435          // Set timeout for data:
 436          if( function_exists('stream_set_timeout') )
 437              stream_set_timeout( $fp, 20 ); // PHP 4.3.0
 438          else
 439              socket_set_timeout( $fp, 20 ); // PHP 4
 440  
 441          // Send request:
 442          $out = "GET $path HTTP/1.0\r\n";
 443          $out .= "Host: $host:$port\r\n";
 444          $out .= "Connection: Close\r\n\r\n";
 445          fwrite($fp, $out);
 446  
 447          // Skip headers:
 448          $i = 0;
 449          $source_charset = 'iso-8859-1'; // default
 450          while( ($s = fgets($fp, 4096)) !== false )
 451          {
 452              $i++;
 453              if( $s == "\r\n" || $i > 100 /* max 100 head lines */ )
 454              {
 455                  break;
 456              }
 457              if( preg_match('~^Content-Type:.*?charset=([\w-]+)~i', $s, $match ) )
 458              {
 459                  $source_charset = $match[1];
 460              }
 461          }
 462  
 463          // Get the refering page's content
 464          $content_ref_page = '';
 465          $bytes_read = 0;
 466          while( ($s = fgets($fp, 4096)) !== false )
 467          {
 468              $content_ref_page .= $s;
 469              $bytes_read += strlen($s);
 470              if( $bytes_read > 512000 )
 471              { // do not pull more than 500kb of data!
 472                  break;
 473              }
 474          }
 475          fclose($fp);
 476  
 477          if( ! strlen($content_ref_page) )
 478          {
 479              $this->debug_log( 'is_referer_linking_us(): empty $content_ref_page ('.bytesreadable($bytes_read).' read)' );
 480              return false;
 481          }
 482  
 483  
 484          $have_idn_name = false;
 485  
 486          // Build the search pattern:
 487          // We match for basically for 'href="[SERVER][URI]', where [SERVER] is a list of possible hosts (especially IDNA)
 488          $search_pattern = '~\shref=["\']?https?://(';
 489          $possible_hosts = array( $_SERVER['HTTP_HOST'] );
 490          if( $_SERVER['SERVER_NAME'] != $_SERVER['HTTP_HOST'] )
 491          {
 492              $possible_hosts[] = $_SERVER['SERVER_NAME'];
 493          }
 494          $search_pattern_hosts = array();
 495          foreach( $possible_hosts as $l_host )
 496          {
 497              if( preg_match( '~^([^.]+\.)(.*?)([^.]+\.[^.]+)$~', $l_host, $match ) )
 498              { // we have subdomains in this hostname
 499                  if( stristr( $match[1], 'www' ) )
 500                  { // search also for hostname without 'www.'
 501                      $search_pattern_hosts[] = $match[2].$match[3];
 502                  }
 503              }
 504              $search_pattern_hosts[] = $l_host;
 505          }
 506          $search_pattern_hosts = array_unique($search_pattern_hosts);
 507          foreach( $search_pattern_hosts as $l_host )
 508          { // add IDN, because this could be linked:
 509              $l_idn_host = idna_decode( $l_host ); // the decoded puny-code ("xn--..") name (utf8)
 510  
 511              if( $l_idn_host != $l_host )
 512              {
 513                  $have_idn_name = true;
 514                  $search_pattern_hosts[] = $l_idn_host;
 515              }
 516          }
 517  
 518          // add hosts to pattern, preg_quoted
 519          for( $i = 0, $n = count($search_pattern_hosts); $i < $n; $i++ )
 520          {
 521              $search_pattern_hosts[$i] = preg_quote( $search_pattern_hosts[$i], '~' );
 522          }
 523          $search_pattern .= implode( '|', $search_pattern_hosts ).')';
 524          if( empty($uri) )
 525          { // host(s) should end with "/", "'", '"', "?" or whitespace
 526              $search_pattern .= '[/"\'\s?]';
 527          }
 528          else
 529          {
 530              $search_pattern .= preg_quote($uri, '~');
 531              // URI should end with "'", '"' or whitespace
 532              $search_pattern .= '["\'\s]';
 533          }
 534          $search_pattern .= '~i';
 535  
 536          if( $have_idn_name )
 537          { // Convert charset to UTF-8, because the decoded domain name is UTF-8, too:
 538              if( can_convert_charsets( 'utf-8', $source_charset ) )
 539              {
 540                  $content_ref_page = convert_charset( $content_ref_page, 'utf-8', $source_charset );
 541              }
 542              else
 543              {
 544                  $this->debug_log( 'is_referer_linking_us(): warning: cannot convert charset of referring page' );
 545              }
 546          }
 547  
 548          if( preg_match( $search_pattern, $content_ref_page ) )
 549          {
 550              $this->debug_log( 'is_referer_linking_us(): found current URL in page ('.bytesreadable($bytes_read).' read)' );
 551  
 552              return true;
 553          }
 554          else
 555          {
 556              if( strpos( $referer, $ReqHost ) === 0 && ! empty($uri) )
 557              { // Referer is the same host.. just search for $uri
 558                  if( strpos( $content_ref_page, $uri ) !== false )
 559                  {
 560                      $this->debug_log( 'is_referer_linking_us(): found current URI in page ('.bytesreadable($bytes_read).' read)' );
 561  
 562                      return true;
 563                  }
 564              }
 565              $this->debug_log( 'is_referer_linking_us(): '.sprintf('did not find &laquo;%s&raquo; in &laquo;%s&raquo; (%s bytes read).', $search_pattern, $referer, bytesreadable($bytes_read) ) );
 566  
 567              return false;
 568          }
 569      }

 572      /**
 573       * Simple check for duplicate comment/content from same author
 574       *
 575       * @param Comment
 576       */
 577  	function is_duplicate_comment( $Comment )
 578      {
 579          global $DB;
 580  
 581          if( ! $this->Settings->get('check_dupes') )
 582          {
 583              return false;
 584          }
 585  
 586          if( $Comment->content == '' )
 587          { // User may has many comments with empty content but with attachment pictures
 588              return false;
 589          }
 590  
 591          $sql = '
 592                  SELECT comment_ID
 593                    FROM T_comments
 594                   WHERE comment_item_ID = '.$Comment->item_ID;
 595  
 596          if( isset($Comment->author_user_ID) )
 597          { // registered user:
 598              $sql .= ' AND comment_author_user_ID = '.$Comment->author_user_ID;
 599          }
 600          else
 601          { // visitor (also trackback):
 602              $sql_ors = array();
 603              if( ! empty($Comment->author) )
 604              {
 605                  $sql_ors[] = 'comment_author = '.$DB->quote($Comment->author);
 606              }
 607              if( ! empty($Comment->author_email) )
 608              {
 609                  $sql_ors[] = 'comment_author_email = '.$DB->quote( $Comment->author_email );
 610              }
 611              if( ! empty($Comment->author_url) )
 612              {
 613                  $sql_ors[] = 'comment_author_url = '.$DB->quote($Comment->author_url);
 614              }
 615  
 616              if( ! empty($sql_ors) )
 617              {
 618                  $sql .= ' AND ( '.implode( ' OR ', $sql_ors ).' )';
 619              }
 620          }
 621  
 622          $sql .= ' AND comment_content = '.$DB->quote($Comment->content).' LIMIT 1';
 623  
 624          return $DB->get_var( $sql, 0, 0, 'Checking for duplicate feedback content.' );
 625      }
