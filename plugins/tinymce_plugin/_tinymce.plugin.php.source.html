 112      /**
 113       * Declare custom events that this plugin fires.
 114       *
 115       * The gallery2_plugin uses these.
 116       *
 117       * Plugins can set the "load_before_init" parameter with some javascript code
 118       * that will be executed before tinymce.init() is called. This is most useful
 119       * for inserting code to load an external tinymce plugin.
 120       *
 121       * Supported events are as follows:
 122       * tinymce_before_init: Allows other b2evo plugins to load tinymce plugins
 123       *                      before the tinymce init.
 124       * Example:
 125       * function tinymce_before_init( &$params ) {
 126       *   $mypluginurl = \$this->get_plugin_url()."myplugin/plugin.min.js";
 127       *   echo "tinymce.PluginManager.load('myplugin', '".$mypluginurl."');";
 128       * }
 129       *
 130       * tinymce_extend_plugins: Allows b2evo plugins to extend the plugin list.
 131       *                         TinyMCE often needs to be told not to load an
 132       *                         external plugin during it's load phase because it's
 133       *                         already been loaded. The plugin list is exposed
 134       *                         in the tinymce_plugins property in the params.
 135       * Example:
 136       * function tinymce_extend_plugins( &$params ) {
 137       *   array_push($params["tinymce_plugins"], "-myplugin");
 138       * }
 139       *
 140       * tinymce_extend_buttons: Allows b2evo plugins to extend the buttons in the
 141       *                         Third button panel.The buttons list is exposed
 142       *                         in the tinymce_buttons property in the params.
 143       * Example:
 144       * function tinymce_extend_buttons( &$params ) {
 145       *   array_push($params["tinymce_buttons"], "mypluginbutton");
 146       * }
 147       */
