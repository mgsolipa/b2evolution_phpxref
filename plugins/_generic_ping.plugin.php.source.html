 141      /**
 142       * Parse a given ping service URL
 143       *
 144       * @return false|array False in case of error, array with keys 'host', 'port', 'path' otherwise
 145       */
 146  	function parse_ping_url( $url )
 147      {
 148          if( ! preg_match( '~^([^/:]+)(:\d+)?(/.*)?$~', $url, $match ) )
 149          {
 150              return false;
 151          }
 152  
 153          $r = array(
 154                  'host' => $match[1],
 155                  'port' => empty($match[2]) ? 80 : $match[2],
 156                  'path' => empty($match[3]) ? '/' : $match[3],
 157              );
 158  
 159          return $r;
 160      }

 163      /**
 164       * Send a ping to the configured service.
 165       */
 166  	function ItemSendPing( & $params )
 167      {
 168          global $debug;
 169          global $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password;
 170  
 171          $url = $this->parse_ping_url( $this->Settings->get( 'ping_service_url' ) );
 172  
 173          $Item = $params['Item'];
 174          $item_Blog = $Item->get_Blog();
 175  
 176          $client = new xmlrpc_client( $url['path'], $url['host'], $url['port'] );
 177          $client->debug = ($debug && $params['display']);
 178  
 179          // Set proxy for outgoing connections:
 180          if( ! empty( $outgoing_proxy_hostname ) )
 181          {
 182              $client->setProxy( $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password );
 183          }
 184  
 185          if( $this->Settings->get('ping_service_extended') )
 186          {
 187              $message = new xmlrpcmsg("weblogUpdates.extendedPing", array(
 188                      new xmlrpcval( $item_Blog->get('name') ),
 189                      new xmlrpcval( $item_Blog->get('url') ),
 190                      new xmlrpcval( $Item->get_permanent_url() ),
 191                      new xmlrpcval( $item_Blog->get('atom_url') ),
 192                      // TODO: tags..
 193                      ));
 194          }
 195          else
 196          {
 197              $message = new xmlrpcmsg("weblogUpdates.ping", array(
 198                      new xmlrpcval( $item_Blog->get('name') ),
 199                      new xmlrpcval( $item_Blog->get('url') ) ));
 200          }
 201          $result = $client->send($message);
 202  
 203          $params['xmlrpcresp'] = $result;
 204  
 205          return true;
 206      }
