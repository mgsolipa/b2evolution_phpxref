 137  	function PluginSettingsUpdateAction()
 138      {
 139          global $DB, $Messages;
 140  
 141          // Store in this array all questions IDs which are inserted or updated on currect action
 142          $questions_IDs = array( -1 );
 143          // Store in this array only questions with correct format
 144          $questions_formated = array();
 145  
 146          $questions = $this->Settings->get( 'questions' );
 147          if( empty( $questions ) )
 148          { // No questions
 149              // Delete all questions from DB:
 150              $DB->query( 'DELETE FROM '.$this->get_sql_table( 'questions' ) );
 151              return true;
 152          }
 153  
 154          $questions = explode( "\n", str_replace( "\r\n", "\n", $questions ) );
 155  
 156          $insert_data = array();
 157          $update_data = array();
 158          foreach( $questions as $question )
 159          {
 160              if( preg_match( '/(.+\?)(.+)/i', $question, $match ) )
 161              { // Check a question for correct format
 162                  $question = trim( $match[1] );
 163                  $answers = trim( $match[2] );
 164                  if( !empty( $question ) && !empty( $answers ) )
 165                  { // Save this question in DB
 166                      $SQL = new SQL();
 167                      $SQL->SELECT( 'cptq_ID' );
 168                      $SQL->FROM( $this->get_sql_table( 'questions' ) );
 169                      $SQL->WHERE( 'cptq_question = '.$DB->quote( $question ) );
 170  
 171                      if( $question_ID = $DB->get_var( $SQL->get() ) )
 172                      { // This question already exists, we should only update the answers
 173                          $update_data[ $question_ID ] = $answers;
 174                      }
 175                      else
 176                      { // New question
 177                          $insert_data[] = '( '.$DB->quote( $question ).', '.$DB->quote( $answers ).' )';
 178                      }
 179  
 180                      $questions_formated[] = $question.' '.$answers;
 181                  }
 182              }
 183              elseif( ! empty( $question ) )
 184              { // Incorrect format of question
 185                  $Messages->add( sprintf( T_('Invalid question format in the line %s.'), '"<b>'.$question.'</b>"' ), 'error' );
 186              }
 187          }
 188  
 189          if( param_errors_detected() )
 190          { // Don't save the changes on errors
 191              return false;
 192          }
 193          else
 194          { // Save the changes on success
 195              if( count( $insert_data ) )
 196              { // Insert new questions in DB:
 197                  foreach( $insert_data as $qstn_data )
 198                  {
 199                      $DB->query( 'INSERT INTO '.$this->get_sql_table( 'questions' ).'
 200                          ( cptq_question, cptq_answers ) VALUES '.$qstn_data );
 201                      $questions_IDs[] = $DB->insert_id;
 202                  }
 203              }
 204  
 205              if( count( $update_data ) )
 206              { // Update the questions in DB:
 207                  foreach( $update_data as $qstn_ID => $qstn_answers )
 208                  {
 209                      $DB->query( 'UPDATE '.$this->get_sql_table( 'questions' ).'
 210                            SET cptq_answers = '.$DB->quote( $qstn_answers ).'
 211                          WHERE cptq_ID = '.$qstn_ID );
 212                      $questions_IDs[] = $qstn_ID;
 213                  }
 214              }
 215  
 216              // Delete the old questions from DB:
 217              $DB->query( 'DELETE FROM '.$this->get_sql_table( 'questions' ).'
 218                  WHERE cptq_ID NOT IN ( '.implode( ', ', $questions_IDs ).' )' );
 219  
 220              // Resave the questions in order to delere all whitespaces and questions with incorrect format
 221              $this->Settings->set( 'questions', implode( "\r\n", $questions_formated ) );
 222  
 223              return true;
 224          }
 225      }

 228      /**
 229       * Validate the given answer against our stored one.
 230       *
 231       * This event is provided for other plugins and gets used internally
 232       * for other events we're hooking into.
 233       *
 234       * @param array Associative array of parameters.
 235       * @param string Form type ( comment|register|message )
 236       * @return boolean|NULL
 237       */
 238  	function CaptchaValidated( & $params, $form_type )
 239      {
 240          global $DB, $localtimenow, $Session;
 241  
 242          if( ! $this->does_apply( $params, $form_type ) )
 243          {
 244              return;
 245          }
 246  
 247          $posted_answer = utf8_strtolower( param( 'captcha_qstn_'.$this->ID.'_answer', 'string', '' ) );
 248  
 249          if( empty( $posted_answer ) )
 250          {
 251              $this->debug_log( 'captcha_qstn_'.$this->ID.'_answer' );
 252              $params['validate_error'] = $this->T_('Please enter the captcha answer.');
 253              if( $form_type == 'comment' && ( $comment_Item = & $params['Comment']->get_Item() ) )
 254              {
 255                  syslog_insert( 'Comment captcha answer is not entered', 'warning', 'item', $comment_Item->ID, 'plugin', $this->ID );
 256              }
 257              return false;
 258          }
 259  
 260          $question = $this->CaptchaQuestion();
 261  
 262          $posted_answer_is_correct = false;
 263  
 264          $answers = explode( '|', utf8_strtolower( $question->cptq_answers ) );
 265          foreach( $answers as $answer )
 266          {
 267              if( $posted_answer == $answer )
 268              {    // Correct answer is found in DB
 269                  $posted_answer_is_correct = true;
 270                  break;
 271              }
 272          }
 273  
 274          if( !$posted_answer_is_correct )
 275          {
 276              $this->debug_log( 'Posted ('.$posted_answer.') and saved ('.$question->cptq_answers.') answer do not match!' );
 277              $params['validate_error'] = $this->T_('The entered answer is incorrect.');
 278              if( $form_type == 'comment' && ( $comment_Item = & $params['Comment']->get_Item() ) )
 279              {
 280                  syslog_insert( 'Comment captcha answer is incorrect', 'warning', 'item', $comment_Item->ID, 'plugin', $this->ID );
 281              }
 282              return false;
 283          }
 284  
 285          // If answer is correct:
 286          //   We should clean the question ID that was assigned for current session and IP address
 287          //   It gives to assign new question on the next captcha event
 288          $this->CaptchaQuestionCleanup();
 289  
 290          return true;
 291      }

 295       * Get question for current user
 296       *
 297       * @return object Question row from DB table "questions"
 298       */
 299  	function CaptchaQuestion()
 300      {
 301          global $DB, $Session;
 302  
 303          $question = NULL;
 304  
 305          $IP = ip2int( $_SERVER['REMOTE_ADDR'] );
 306  
 307          // Get question ID from Session
 308          $this->question_ID = $Session->get( 'captcha_qstn_'.$this->ID.'_ID' );
 309  
 310          if( empty( $this->question_ID ) )
 311          {    // Get question from DB by current IP address
 312              $SQL = new SQL();
 313              $SQL->SELECT( 'cq.*' );
 314              $SQL->FROM( $this->get_sql_table( 'ip_question' ) );
 315              $SQL->FROM_add( 'INNER JOIN '.$this->get_sql_table( 'questions' ).' AS cq ON cptip_cptq_ID = cptq_ID' );
 316              $SQL->WHERE( 'cptip_IP = '.$DB->quote( $IP ) );
 317  
 318              if( $question = $DB->get_row( $SQL->get() ) )
 319              {
 320                  $this->question_ID = $question->cptq_ID;
 321              }
 322          }
 323  
 324          if( empty( $this->question_ID ) )
 325          {    // Assign new random question for current IP address
 326              $question = $this->CaptchaQuestionNew();
 327          }
 328  
 329          if( empty( $question ) && !empty( $this->question_ID ) )
 330          {    // Get question data
 331              $SQL = new SQL();
 332              $SQL->SELECT( '*' );
 333              $SQL->FROM( $this->get_sql_table( 'questions' ) );
 334              $SQL->WHERE( 'cptq_ID = '.$DB->quote( $this->question_ID ) );
 335              $question = $DB->get_row( $SQL->get() );
 336  
 337              if( empty( $question ) )
 338              {    // Assign random question if previous question doesn't exist in DB
 339                  // This case may happens when admin changed the questions but user has the old question ID in the session or in DB table "ip_question"
 340                  $question = $this->CaptchaQuestionNew();
 341              }
 342          }
 343  
 344          return $question;
 345      }

 352  	function CaptchaQuestionNew()
 353      {
 354          global $DB, $Session;
 355  
 356          $IP = ip2int( $_SERVER['REMOTE_ADDR'] );
 357  
 358          // Get new random question from DB
 359          $SQL = new SQL();
 360          $SQL->SELECT( '*' );
 361          $SQL->FROM( $this->get_sql_table( 'questions' ) );
 362          $SQL->ORDER_BY( 'RAND()' );
 363          $SQL->LIMIT( 1 );
 364          $question = $DB->get_row( $SQL->get() );
 365  
 366          // Insert a record for current IP address with assigned question ID
 367          $DB->query( 'INSERT INTO '.$this->get_sql_table( 'ip_question' ).'
 368              ( cptip_IP, cptip_cptq_ID ) VALUES
 369              ( '.$DB->quote( $IP ).', '.$DB->quote( $question->cptq_ID ).' )' );
 370  
 371          // Save the assigned question ID in the session
 372          $Session->set( 'captcha_qstn_'.$this->ID.'_ID', $question->cptq_ID );
 373          $Session->dbsave();
 374  
 375          $this->question_ID = $question->cptq_ID;
 376  
 377          return $question;
 378      }

 399      /**
 400       * When a comment form gets displayed, we inject our captcha and an input field to
 401       * enter the answer.
 402       *
 403       * The question ID is saved into the user's Session and in the DB table "ip_question".
 404       *
 405       * @param array Associative array of parameters
 406       *   - 'Form': the form where payload should get added (by reference, OPTIONALLY!)
 407       *   - 'form_use_fieldset':
 408       *   - 'key': A key that is associated to the caller of the event (string, OPTIONALLY!)
 409       * @param string Form type ( comment|register|message )
 410       * @return boolean|NULL true, if displayed; false, if error; NULL if it does not apply
 411       */
 412  	function CaptchaPayload( & $params, $form_type )
 413      {
 414          global $DB, $Session;
 415  
 416          if( ! $this->does_apply( $params, $form_type ) )
 417          {
 418              return;
 419          }
 420  
 421          $question = $this->CaptchaQuestion();
 422  
 423          if( empty( $question ) )
 424          {    // No the defined questions
 425              return;
 426          }
 427  
 428          $this->debug_log( 'Question ID is: ('.$this->question_ID.')' );
 429  
 430          if( ! isset( $params['Form'] ) )
 431          {    // there's no Form where we add to, but we create our own form:
 432              $Form = new Form( regenerate_url() );
 433              $Form->begin_form();
 434          }
 435          else
 436          {
 437              $Form = & $params['Form'];
 438              if( ! isset( $params['form_use_fieldset'] ) || $params['form_use_fieldset'] )
 439              {
 440                  $Form->begin_fieldset();
 441              }
 442          }
 443  
 444          $Form->info( $this->T_('Captcha question'), $question->cptq_question );
 445          $Form->text_input( 'captcha_qstn_'.$this->ID.'_answer', param( 'captcha_qstn_'.$this->ID.'_answer', 'string', '' ),
 446                  10, $this->T_('Captcha answer'), ( empty( $params['use_placeholders'] ) ? $this->T_('Please answer the question above').'.' : '' ),
 447                  array(
 448                          'placeholder' => empty( $params['use_placeholders'] ) ? '' : T_('Please answer the question above'),
 449                      )
 450              );
 451  
 452          if( ! isset($params['Form']) )
 453          {    // there's no Form where we add to, but our own form:
 454              $Form->end_form( array( array( 'submit', 'submit', $this->T_('Validate me'), 'ActionButton' ) ) );
 455          }
 456          else
 457          {
 458              if( ! isset($params['form_use_fieldset']) || $params['form_use_fieldset'] )
 459              {
 460                  $Form->end_fieldset();
 461              }
 462          }
 463  
 464          return true;
 465      }

 545      /**
 546       * Checks if we should captcha the current request, according to the settings made.
 547       *
 548       * @param array Associative array of parameters.
 549       * @param string Form type ( comment|register|message )
 550       * @return boolean

 552  	function does_apply( $params, $form_type )
 553      {
 554          global $current_User;
 555  
 556          switch( $form_type )
 557          {
 558              case 'comment':
 559                  if( !is_logged_in() )
 560                  {
 561                      return $this->Settings->get( 'use_for_anonymous_comment' );
 562                  }
 563                  break;
 564  
 565              case 'register':
 566                  return $this->Settings->get( 'use_for_registration' );
 567  
 568              case 'message':
 569                  if( !is_logged_in() )
 570                  {
 571                      return $this->Settings->get( 'use_for_anonymous_message' );
 572                  }
 573                  break;
 574          }
 575  
 576          return false;
 577      }
