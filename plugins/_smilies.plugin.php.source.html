 282      /**
 283       * Perform rendering
 284       *
 285       * @see Plugin::RenderItemAsHtml()
 286       */
 287  	function RenderItemAsHtml( & $params )
 288      {
 289          $this->InitSmilies();    // check smilies are already cached
 290  
 291  
 292          if( ! isset( $this->search ) )
 293          {    // We haven't prepared the smilies yet
 294              $this->search = array();
 295  
 296              $tmpsmilies = $this->smilies;
 297              usort($tmpsmilies, array(&$this, 'smiliescmp'));
 298  
 299              foreach( $tmpsmilies as $smiley )
 300              {
 301                  // Detect html entities if smile code, to find encoded code version when HTML is disabled in content
 302                  $html_entities_exist = ( strpos( $smiley['code'], '>' ) !== false || strpos( $smiley['code'], '>' ) !== false );
 303  
 304                  $this->search[] = $smiley['code'];
 305                  $smiley_masked = '';
 306                  for( $i = 0; $i < strlen( $smiley['code'] ); $i++ )
 307                  {
 308                      $smiley_masked .= '&#'.ord( substr( $smiley['code'], $i, 1 ) ).';';
 309                  }
 310                  if( $html_entities_exist )
 311                  { // Add code version with encoded html entities
 312                      $this->search[] = str_replace( array( '<', '>' ), array( '&lt;', '&gt;' ), $smiley['code'] );
 313                  }
 314                  $smiley['code'] = $smiley_masked;
 315                  $this->replace[] = $this->get_smiley_img_tag( $smiley );
 316                  if( $html_entities_exist )
 317                  { // Add a duplicate of replacement when smile code has html entities
 318                      $this->replace[] = $this->get_smiley_img_tag( $smiley );
 319                  }
 320              }
 321          }
 322  
 323  
 324          // REPLACE:  But only in non-HTML blocks, totally excluding <CODE>..</CODE> and <PRE>..</PRE>
 325  
 326          $content = & $params['data'];
 327  
 328          // Lazy-check first, using stristr() (stripos() is only available since PHP5):
 329          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false || strstr( $content, '`' ) !== false )
 330          { // Call ReplaceTagSafe() on everything outside code/pre:
 331              $content = callback_on_non_matching_blocks( $content,
 332                      '~(`|<(code|pre)[^>]*>).*?(\1|</\2>)~is',
 333                      array( & $this, 'ReplaceTagSafe' ) );
 334          }
 335          else
 336          { // No code/pre blocks, replace on the whole thing
 337              $content = $this->ReplaceTagSafe( $content );
 338          }
 339  
 340          return true;
 341      }

 348  	function get_smiley_img_tag( $smiley, $override_fields = array() )
 349      {
 350          $attribs = array(
 351              'src' => $smiley['image'],
 352              'title' => format_to_output( $smiley['code'], 'htmlattr' ),
 353              'alt' => format_to_output( $smiley['code'], 'htmlattr' ),
 354              'class' => 'middle',
 355              );
 356  
 357          if( $smiley_wh = imgsize($smiley['path'], 'widthheight_assoc') )
 358              $attribs += $smiley_wh;
 359  
 360          if( $override_fields )
 361              $attribs = $override_fields + $attribs;
 362  
 363          return '<img'.get_field_attribs_as_string($attribs).' />';
 364      }

 397      /**
 398       * sorts the smilies' array by length
 399       * this is important if you want :)) to superseede :) for example
 400       */
 401  	function smiliescmp($a, $b)
 402      {
 403          if( ($diff = strlen( $b[ 'code' ] ) - strlen( $a[ 'code' ] ) ) == 0)
 404          {
 405              return strcmp( $a[ 'code' ], $b[ 'code' ] );
 406          }
 407          return $diff;
 408      }

 411      /**
 412       * Initiates the smiley array if not already initiated
 413       *
 414       * Attempts to use skin specific smileys where available
 415       *    - skins_adm/skin/rsc/smilies/
 416       *    - skins/skin/smilies/
 417       *
 418       * Attempts to fallback to default smilies
 419       *    - rsc/smilies/
 420       *
 421       * If no image file found the smiley is not added
 422       *
 423       * @return array of available smilies( code, image url )
 424       */
 425  	function InitSmilies()
 426      {
 427          if( isset( $this->smilies ) )
 428          { // smilies are already cached
 429              return;
 430          }
 431  
 432          global $admin_skin, $adminskins_path, $adminskins_url, $rsc_path, $rsc_url, $skin, $skins_path, $skins_url;
 433  
 434          // set the skin path/url and the default (rsc) path/url
 435          $currentskin_path = ( is_admin_page() ? $adminskins_path.$admin_skin.'/rsc' : $skins_path.$skin ).'/smilies/';
 436          $currentskin_url = ( is_admin_page() ? $adminskins_url.$admin_skin.'/rsc' : $skins_url.$skin ).'/smilies/';
 437          $default_path = $rsc_path.'smilies/';
 438          $default_url = $rsc_url.'smilies/';
 439  
 440          $skin_has_smilies = is_dir( $currentskin_path );    // check if skin has a /smilies/ folder
 441  
 442          $this->smilies = array();
 443          $temp_list = explode( "\n", str_replace( array( "\r", "\t" ), '', $this->Settings->get( 'smiley_list' ) ) );
 444  
 445          foreach( $temp_list as $temp_smiley )
 446          {
 447              $a_smiley = explode( '<->',    preg_replace_callback( '#^(\S.+?\s)(.+?)(\/\/.*?)*$#', array( $this, 'get_smiley' ),$temp_smiley ) );
 448              if( isset( $a_smiley[0] ) and isset( $a_smiley[1] ) )
 449              {
 450                  // lets see if the file exists
 451                  $temp_img = trim( $a_smiley[1] );
 452                  if( $skin_has_smilies && is_file( $currentskin_path.$temp_img ) )
 453                  {
 454                      $temp_url = $currentskin_url.$temp_img;    // skin has it's own smiley, use it
 455                      $temp_path = $currentskin_path.$temp_img;
 456                  }
 457                  elseif ( is_file( $default_path.$temp_img ) )
 458                  {
 459                      $temp_url = $default_url.$temp_img; // no skin image, but default smiley found so use it
 460                      $temp_path = $default_path.$temp_img;
 461                  }
 462                  else
 463                  {
 464                      $temp_url = ''; // no smiley image found, so don't add the smiley
 465                  }
 466  
 467                  if( $temp_url )
 468                      $this->smilies[] = array( 'code' => trim( $a_smiley[0] ), 'image' => $temp_url, 'path' => $temp_path );
 469              }
 470          }
 471      }
