 113  /**
 114   * This tracks who is online
 115   *
 116   * @todo dh> I wanted to add a MySQL INDEX on the sess_lastseen field, but this "plugin"
 117   *       is the only real user of this. So, when making this a plugin, this should
 118   *       add the index perhaps.
 119   * @package evocore
 120   */
 121  class OnlineSessions
 122  {
 123      /**
 124       * Number of guests (and users that want to be anonymous)
 125       *
 126       * Gets lazy-filled when needed, through {@link init()}.
 127       *
 128       * @access protected
 129       */
 130      var $_count_guests;
 131

 134       * List of registered users.
 135       *
 136       * Gets lazy-filled when needed, through {@link init()}.
 137       *
 138       * @access protected
 139       */
 140      var $_registered_Users;
 141  
 142      /**
 143       * Online session timeout in seconds.
 144       *
 145       * Default value: 300 (5 minutes). Set by {@link OnlineSessions::OnlineSessions()}.
 146       *
 147       * @access protected
 148       */
 149      var $_timeout_online_user;
 150  
 151      var $_initialized = false;
 152

 154      /**
 155       * Constructor.
 156       *
 157       * @param integer Online session timeout in seconds.
 158       */
 159  	function OnlineSessions( $timeout_online_user = 300 )
 160      {
 161          $this->_timeout_online_user = $timeout_online_user;
 162      }
 163  
 164  
 165      /**
 166       * Get an array of registered users and guests.
 167       *
 168       * @return array containing number of registered users and guests ('registered' and 'guests')
 169       */
 170  	function init()
 171      {
 172          if( $this->_initialized )
 173          {
 174              return true;
 175          }
 176          global $DB, $UserSettings, $localtimenow;
 177  
 178          $this->_count_guests = 0;
 179          $this->_registered_Users = array();
 180  
 181          $timeout_YMD = date( 'Y-m-d H:i:s', ($localtimenow - $this->_timeout_online_user) );
 182  
 183          $UserCache = & get_UserCache();
 184  
 185          // We get all sessions that have been seen in $timeout_YMD and that have a session key.
 186          // NOTE: we do not use DISTINCT here, because guest users are all "NULL".
 187          $online_user_ids = $DB->get_col( "
 188              SELECT SQL_NO_CACHE sess_user_ID
 189                FROM T_sessions
 190               WHERE sess_lastseen_ts > '".$timeout_YMD."'
 191                 AND sess_key IS NOT NULL
 192               GROUP BY sess_ID", 0, 'Sessions: get list of relevant users.' );
 193          $registered_online_user_ids = array_diff( $online_user_ids, array( NULL ) );
 194          // load all online users into the cache because we need information ( login, avatar ) about them
 195          $UserCache->load_list( $registered_online_user_ids );
 196          foreach( $online_user_ids as $user_ID )
 197          {
 198              if( !empty( $user_ID ) && ( $User = & $UserCache->get_by_ID( $user_ID, false ) ) )
 199              {
 200                  if( $UserSettings->get( 'show_online', $User->ID ) )
 201                  {    // Assign by ID so that each user is only counted once (he could use multiple user agents at the same time):
 202                      $this->_registered_Users[ $user_ID ] = & $User;
 203                  }
 204                  else
 205                  {    // Count this user as guest when he doesn't want to be visible:
 206                      $this->_count_guests++;
 207                  }
 208              }
 209              else
 210              {
 211                  $this->_count_guests++;
 212              }
 213          }
 214  
 215          $this->_initialized = true;
 216      }

 250      /**
 251       * Template function: Display the registered users who are online
 252       *
 253       * @todo get class="" out of here (put it into skins)
 254       *
 255       * @param array
 256       */
 257  	function display_online_users( $params )
 258      {
 259          global $DB, $Blog, $UserSettings;
 260  
 261          if( !isset($this->_registered_Users) )
 262          {
 263              $this->init();
 264          }
 265  
 266          // Note: not all users want to appear online, so we might have an empty list.
 267          $r = '';
 268  
 269          foreach( $this->_registered_Users as $User )
 270          {
 271              if( $UserSettings->get( 'show_online', $User->ID ) )
 272              {
 273                  if( empty($r) )
 274                  { // first user
 275                      $r .= $params['list_start'];
 276                  }
 277  
 278                  $r .= $params['item_start'];
 279                  $r .= $User->dget('preferredname');
 280                  if( $params['contacticons'] )
 281                  {    // We want contact icons:
 282                      $r .= $User->get_msgform_link();
 283                  }
 284                  $r .= $params['item_end'];
 285              }
 286          }
 287  
 288          if( !empty($r) )
 289          { // we need to close the list
 290              $r .= $params['list_end'];;
 291          }
 292  
 293          echo $r;
 294      }
 295  
 296  
 297      /**
 298       * Template function: Display number of online guests.
 299       */
 300  	function display_online_guests( $params )
 301      {
 302          if( !isset($this->_count_guests) )
 303          {
 304              $this->init();
 305          }
 306  
 307          $r = $params['list_start'];
 308          $r .= $params['item_start'];
 309          $r .= T_('Guest Users:').' ';
 310          $r .= $this->_count_guests;
 311          $r .= $params['item_end'];
 312          $r .= $params['list_end'];;
 313  
 314          echo $r;
 315      }
 316  }
