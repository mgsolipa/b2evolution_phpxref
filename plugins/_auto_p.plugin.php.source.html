 150      /**
 151       * - Split text into blocks, using $block_tags pattern.
 152       *
 153       * @param string Text
 154       * @param string The HTML tag where $text is in
 155       * @return string
 156       */
 157  	function handle_blocks( $text, $in_tag = '' )
 158      {
 159          #echo '<h1>HANDLE_BLOCKS</h1>'; pre_dump( $text, $in_tag );
 160  
 161          $new_text = '';
 162  
 163          if( preg_match( '~^(.*?)(<\s*('.$this->block_tags.')(\b[^>]*)?>)~is', $text, $match ) )
 164          { // there's a block tag:
 165              $tag = $match[3];
 166              $before_tag = $match[1];
 167  
 168              if( ! empty($before_tag) )
 169              { // Recurse (one pattern/callback deeper):
 170                  $new_text .= $this->handle_pre_blocks( $before_tag, $in_tag );
 171              }
 172  
 173              $text_after_tag = substr( $text, strlen($match[0]) );
 174  
 175              if( empty($match[4]) || substr(rtrim($match[4]), -1) != '/' )
 176              { // Opening tag (and not self-closing): handle text in tag:
 177                  $new_text .= $match[2];
 178  
 179                  // Find closing tag:
 180                  list( $text_in_tag, $closing_tag, $NL_before, $NL_after ) = $this->split_text_for_tag($tag, $text_after_tag);
 181  
 182                  if( ! empty($text_in_tag) )
 183                  { // Recurse (same level):
 184                      $text_in_tag = $this->handle_blocks( $text_in_tag, $tag );
 185                  }
 186  
 187                  $new_text .= $NL_before.$text_in_tag.$NL_after;
 188  
 189                  $new_text .= $closing_tag;
 190              }
 191              else
 192              { // self-closing tag:
 193                  $new_text .= $match[2];
 194              }
 195  
 196              if( ! empty($text_after_tag) )
 197              {
 198                  #echo '<h1>RECURSE: text_after_tag (block)</h1>';
 199                  // Recurse (same level):
 200                  $new_text .= $this->handle_blocks( $text_after_tag, $in_tag );
 201              }
 202          }
 203          else
 204          { // No BLOCKS in this $text:
 205              $new_text = $this->handle_pre_blocks($text, $in_tag);
 206          }
 207  
 208          #pre_dump( 'HANDLE_BLOCKS return: ', $new_text, $in_tag );
 209          return $new_text;
 210      }

 213      /**
 214       * Handle text which may contain inline tags
 215       *
 216       * - Explode by \n\n
 217       * - Merge blocks that span over multiple tags
 218       * - Apply BR to blocks
 219       * - Wrap block in P
 220       *
 221       * @param string Text
 222       * @param string Tag where $text is in
 223       * @return string
 224       */
 225  	function handle_pre_blocks( $text, $in_tag )
 226      {
 227          #echo '<h2>HANDLE_PRE_BLOCKS</h2>'; pre_dump( $text, $in_tag );
 228  
 229          if( $in_tag )
 230          {
 231              if( in_array($in_tag, $this->skip_tags) )
 232              {
 233                  return $text;
 234              }
 235  
 236              if( ! in_array($in_tag, $this->p_allowed_in) )
 237              { // we're in a tag, where no P tags are allowed, so just do the BRs:
 238                  return $this->handle_br( $text, $in_tag );
 239              }
 240          }
 241  
 242          $text_lines = preg_split( '~(\n\n+)~', $text, -1, /*PREG_SPLIT_NO_EMPTY |*/ PREG_SPLIT_DELIM_CAPTURE );
 243          $text_lines[] = ''; // dummy
 244  
 245          #echo '<strong>text_lines</strong><br />'; pre_dump( $text_lines, $in_tag );
 246  
 247          $new_blocks = array();
 248          $count_new_blocks = 0;
 249          for( $i = 0, $n = count($text_lines); $i < $n; $i = $i+2 /* every second block is a real one */ )
 250          {
 251              if( ! isset($new_blocks[$count_new_blocks]) )
 252              {
 253                  $new_blocks[$count_new_blocks] = '';
 254              }
 255              if( $text_lines[$i] == '' )
 256              {
 257                  $new_blocks[$count_new_blocks] .= $text_lines[$i+1];
 258                  $new_blocks[$count_new_blocks+1] = ''; // dummy
 259                  continue;
 260              }
 261              $new_blocks[$count_new_blocks] .= $text_lines[$i];
 262              $new_blocks[$count_new_blocks+1] = $text_lines[$i+1];
 263              $count_new_blocks = $count_new_blocks+2;
 264          }
 265  
 266          $text_lines = $new_blocks;
 267          #echo '<strong>new text_lines</strong><br />'; pre_dump( $new_blocks );
 268  
 269          if( trim($text) == '' )
 270          { // there's only whitespace
 271              return $text;
 272          }
 273  
 274  
 275          // fix it, so no (inline) tags span across multiple blocks:
 276          $new_blocks = array();
 277          $new_blocks_nowrap = array(); // blocks that should not be wrapped in P (opening without closing tag or vice versa)
 278          $count_new_blocks = 0;
 279          $looking_for_close_tag = array();
 280  
 281          for( $i = 0, $n = count($text_lines); $i < $n; $i = $i+2 /* every 2nd line is a real block */ )
 282          {
 283              $line = $text_lines[$i];
 284              if( ! isset($new_blocks[$count_new_blocks]) )
 285              {
 286                  $new_blocks[$count_new_blocks] = '';
 287  
 288                  $line_copy = $line;
 289  
 290                  while( preg_match( '~^(.*?)(<\s*/\s*(\w+)(\s+[^>]*?)?>)~is', $line_copy, $match )
 291                      && !(preg_match( '~^(.*?)(<\s*'.$match[3].'(\s+[^>]*?(\s*/\s*)?)?>)~is', $new_blocks[$count_new_blocks].$match[1] )) )
 292                  { // a closing tag:
 293                      $new_blocks[$count_new_blocks] .= $match[0];
 294                      $line_copy = substr($line_copy, strlen($match[0]));
 295                  }
 296                  if( ! empty($new_blocks[$count_new_blocks]) )
 297                  { // we've found a closing tag with no opening tag, this must not get wrapped in P:
 298                      $new_blocks_nowrap[] = $count_new_blocks;
 299                      $new_blocks[$count_new_blocks+1] = ''; // dummy
 300                      $new_blocks[$count_new_blocks+2] = ''; // init new
 301                      $line = substr( $line, strlen($new_blocks[$count_new_blocks]) );
 302                      $count_new_blocks = $count_new_blocks+2;
 303                  }
 304              }
 305              else
 306              { // we're looking for a closing tag:
 307                  // Find closing tag:
 308                  $line_copy = $line;
 309                  list( $text_in_tag, $closing_tag, $NL_before, $NL_after ) = $this->split_text_for_tag( $looking_for_close_tag['tag'], $line_copy /* by ref */ );
 310                  if( empty($closing_tag) )
 311                  { // not in this whole block:
 312                      $new_blocks[ $count_new_blocks ] .= $line.$text_lines[$i+1];
 313                      continue;
 314                  }
 315  
 316                  // Tag has been found:
 317                  $looking_for_close_tag = array();
 318              }
 319  
 320              if( preg_match( '~^(.*?)(<\s*(\w+)(\s+[^>/]*)?(\s*/\s*)?>)~is', $line, $match ) )
 321              { // a opening tag:
 322                  $tag = $match[3];
 323                  $pos_after_tag = strlen($match[0]);
 324  
 325                  while( ! empty($match[5]) /* "/" */ )
 326                  { // self-closing tag, find next:
 327                      $tag = false;
 328                      if( preg_match( '~^(.*?)(<\s*(\w+)(\s+[^>/]*)?(\s*/\s*)?>)~is', substr($line, $pos_after_tag), $match ) )
 329                      {
 330                          $tag = $match[3];
 331                          $pos_after_tag += strlen($match[0]);
 332                      }
 333                  }
 334                  $text_after_tag = substr($line, $pos_after_tag);
 335  
 336                  if( $tag )
 337                  {
 338                      // Find closing tag:
 339                      list( $text_in_tag, $closing_tag, $NL_before, $NL_after ) = $this->split_text_for_tag($tag, $text_after_tag /* by ref */ );
 340                      if( empty($closing_tag) )
 341                      {
 342                          $looking_for_close_tag = array(
 343                                  'tag' => $tag,
 344                                  'block' => $i,
 345                                  'pos' => strlen($new_blocks[ $count_new_blocks ])+strlen($match[1]), // position where the unclosed tag begins
 346                              );
 347                          $new_blocks[ $count_new_blocks ] .= $line.$text_lines[$i+1];
 348                          continue;
 349                      }
 350                  }
 351              }
 352              $new_blocks[ $count_new_blocks ] .= $line;
 353              $new_blocks[ ++$count_new_blocks ] = $text_lines[$i+1];
 354  
 355              $count_new_blocks++;
 356          }
 357          if( $looking_for_close_tag )
 358          {
 359              #echo '<h1>looking_for_close_tag</h1>'; pre_dump( $looking_for_close_tag );
 360              $new_blocks[ $count_new_blocks+1 ] = ''; // dummy
 361  
 362              if( $looking_for_close_tag['pos'] > 0 )
 363              { // move part of last block without closing tag to an own block:
 364                  $new_blocks[ $count_new_blocks+2 ] = substr($new_blocks[ $count_new_blocks ], $looking_for_close_tag['pos']);
 365                  $new_blocks[ $count_new_blocks ] = substr($new_blocks[ $count_new_blocks ], 0, $looking_for_close_tag['pos']);
 366                  $new_blocks[ $count_new_blocks+3 ] = ''; // dummy
 367  
 368                  $new_blocks_nowrap[] = $count_new_blocks+2;
 369              }
 370              else
 371              { // the whole block should not get wrapped!
 372                  $new_blocks_nowrap[] = $count_new_blocks;
 373              }
 374          }
 375  
 376          #echo '<h1>new_blocks:</h1>'; pre_dump( $new_blocks, $new_blocks_nowrap );
 377  
 378          $after_block_wp = '';
 379          $before_block_wp = '';
 380          if( empty($in_tag) )
 381          {
 382              $wrap_in_p = true;
 383          }
 384          elseif( in_array($in_tag, $this->p_allowed_in) )
 385          {
 386              if( ! $this->add_p_in_block )
 387              {
 388                  $wrap_in_p = false;
 389              }
 390              elseif( count($new_blocks) > 2 )
 391              {
 392                  $wrap_in_p = true;
 393              }
 394              else
 395              {
 396                  $wrap_in_p = false;
 397                  if( substr( $new_blocks[0], 0, 1 ) == "\n" )
 398                  {
 399                      $before_block_wp = "\n";
 400                      $new_blocks[0] = substr( $new_blocks[0], 1 );
 401                      $wrap_in_p = true;
 402                  }
 403                  if( substr( $new_blocks[0], -1 ) == "\n" )
 404                  {
 405                      $after_block_wp = "\n";
 406                      $new_blocks[0] = substr( $new_blocks[0], 0, -1 );
 407                      $wrap_in_p = true;
 408                  }
 409              }
 410          }
 411          else
 412          {
 413              $wrap_in_p = false;
 414          }
 415  
 416          if( $new_blocks[count($new_blocks)-2] == '' )
 417          {
 418              array_pop($new_blocks);
 419              array_pop($new_blocks);
 420          }
 421  
 422          $new_text = '';
 423  
 424          for( $i = 0, $n = count($new_blocks); $i < $n; $i = $i+2 )
 425          {
 426              #echo '<h2>--new_blocks['.$i.']: '; pre_dump( $new_blocks[$i] ); echo '</h2>';
 427

 431              {
 432                  if( $this_wrap_in_p )
 433                  { // not the last one

 439                  }

 455      /**
 456       * This is a helper for handling blocks from {@link handle_pre_blocks_helper()}.
 457       *
 458       * What comes here is supposed to have no block tags.
 459       *
 460       * @return array array( $text, $has_p )
 461       */
 462  	function handle_pre_blocks_helper( $block, $in_tag, $wrap_in_p, $ignore_NL = true )
 463      {
 464          #pre_dump( 'HANDLE_PRE_BLOCKS_HELPER begin', $block, $in_tag );
 465          $has_p = NULL;
 466          $r = '';
 467  
 468          if( $in_tag == 'blockquote' )
 469          { // XHTML strict: blockquote content needs to be in block tag
 470              $in_tag = 'p';
 471              $wrap_in_p = true; // at the end
 472          }
 473  
 474          // Remove newlines at start and end (will get re-applied later):
 475          $NL_start = '';
 476          $NL_end = '';
 477          if( $ignore_NL )
 478          {
 479              while( $block{0} == "\n" )
 480              {
 481                  $NL_start .= $block{0};
 482                  $block = substr($block, 1);
 483              }
 484              while( substr($block, -1) == "\n" )
 485              {
 486                  $NL_end .= substr($block, -1);
 487                  $block = substr($block, 0, -1);
 488              }
 489          }
 490  
 491          if( preg_match( '~^(.*?)(<\s*(\w+)(\s+[^>]*)?>)~is', $block, $match ) )
 492          { // a tag:
 493              $tag = $match[3];
 494              $before_tag = $match[1];
 495  
 496              if( ! empty($before_tag) )
 497              { // Delegate to handle_br:
 498                  $r .= $this->handle_br( $before_tag, $in_tag );
 499              }
 500  
 501              // Opening tag:
 502              $r .= $match[2];
 503  
 504              $text_after_tag = substr( $block, strlen($match[0]) );
 505  
 506              // Find closing tag:
 507              list( $text_in_tag, $closing_tag, $NL_before, $NL_after ) = $this->split_text_for_tag($tag, $text_after_tag);
 508  
 509              if( ! empty($text_in_tag) )
 510              { // Recurse (same level) - with the optional newlines at start and end, because in an inline tag every linebreak should become a BR:
 511                  list($text_in_tag, $sub_has_p) = $this->handle_pre_blocks_helper( $NL_before.$text_in_tag.$NL_after, $tag, false, false );
 512              }
 513              $r .= $text_in_tag;
 514  
 515              $r .= $closing_tag;
 516  
 517              if( ! empty($text_after_tag) )
 518              {
 519                  #echo '<h1>RECURSE: text_after_tag (handle_pre_blocks)</h1>';
 520                  // Recurse (same level):
 521                  list( $text_after_tag, $sub_has_p ) = $this->handle_pre_blocks_helper( $text_after_tag, $in_tag, false, false );
 522                  $r .= $text_after_tag;
 523              }
 524          }
 525          else
 526          { // No tags in this $text:
 527              $r .= $this->handle_br( $block, $in_tag );
 528          }
 529  
 530          if( ! empty($wrap_in_p) )
 531          {
 532              $r = '<p>'.$r.'</p>';
 533              $has_p = true;
 534          }
 535  
 536          // re-apply newlines from start and end:
 537          $r = $NL_start.$r.$NL_end;
 538  
 539          #pre_dump( 'HANDLE_PRE_BLOCKS_HELPER return: ', $r, $has_p );
 540          return array( $r, $has_p );
 541      }

 567      /**
 568       * Split the text for a given tag, mainly to find the closing tag.
 569       *
 570       * @return array
 571       */
 572  	function split_text_for_tag($tag, & $text_after_tag)
 573      {
 574          #echo '<strong>split_text_for_tag</strong><br />'; pre_dump( $tag, $text_after_tag );
 575          $depth = 1;
 576          $text_in_tag = '';
 577  
 578          $loop_text = $text_after_tag;
 579          while( 1 )
 580          {
 581              #echo '<hr />loop_text:'; pre_dump( $loop_text );
 582              if( preg_match( '~^(.*?)(<\s*(/)?\s*'.preg_quote( $tag, '~' ).'\s*(/\s*)?>)~is', $loop_text, $after_match ) )
 583              {
 584                  #pre_dump( 'after_match', $after_match );
 585                  $text_in_tag .= $after_match[1];
 586                  $found_tag = $after_match[2];
 587                  $is_closing = ( ! empty($after_match[3]) || ! empty($after_match[4]) /* self-closing */ );
 588  
 589                  if( $is_closing )
 590                  {
 591                      $depth--;
 592                      if( $depth == 0 )
 593                      { // found the matching closing tag:
 594                          $closing_tag = $found_tag;
 595                          break;
 596                      }
 597                      else
 598                      { // this closing tag is part of the outer:
 599                          $text_in_tag .= $found_tag;
 600                      }
 601                  }
 602                  else
 603                  { // found the same, but opening tag (nested)
 604                      $text_in_tag .= $found_tag;
 605                      $depth++;
 606                  }
 607  
 608                  // skip what we've matched:
 609                  $loop_text = substr($loop_text, strlen($after_match[0]) );
 610              }
 611              else
 612              { // did not find the closing tag.. :/
 613                  $closing_tag = '';
 614                  return array( false, false, false, false );
 615              }
 616          }
 617  
 618          // remove newline at start and end:
 619          if( substr($text_in_tag, 0, 1) == "\n" )
 620          {
 621              $NL_before = "\n";
 622              $text_in_tag = substr($text_in_tag, 1);
 623          }
 624          else
 625          {
 626              $NL_before = '';
 627          }
 628          if( substr($text_in_tag, -1) == "\n" )
 629          {
 630              $NL_after = "\n";
 631              $text_in_tag = substr($text_in_tag, 0, -1);
 632          }
 633          else
 634          {
 635              $NL_after = '';
 636          }
 637  
 638          $text_after_tag = substr( $text_after_tag, strlen($NL_before.$text_in_tag.$NL_after)+strlen($closing_tag) );
 639          $r = array( $text_in_tag, $closing_tag, $NL_before, $NL_after );
 640  
 641          #pre_dump( 'return: ', $r, $text_after_tag );
 642          return $r;
 643      }

 646      /**
 647       * Add "<br />" to the end of newlines, which do not end with "<br />" already and which aren't
 648       * the last line, if the "Auto-BR" setting is enabled.
 649       *
 650       * @return string
 651       */
 652  	function autobr( $text, $replace_last = true )
 653      {
 654          if( ! $this->use_auto_br )

 656              return $text;
 657          }
 658  
 659          return preg_replace( '~(?<!<br />)\n'.( $replace_last ? '' : '(?!\z)' ).'~i', "<br />\n", $text );
