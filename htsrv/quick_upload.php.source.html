  30  	function save( $path )
  31      {
  32          $input = fopen( 'php://input', 'r' );
  33          $temp_file_name = '';
  34          $temp = open_temp_file( $temp_file_name );
  35          if( is_string( $temp ) )
  36          { // Error on create a temp file
  37              return array(
  38                      'text'   => $temp,
  39                      'status' => 'error',
  40                  );
  41          }
  42  
  43          $realSize = stream_copy_to_stream( $input, $temp );
  44          fclose( $input );
  45  
  46          if( $realSize != $this->getSize() )
  47          {
  48              return false;
  49          }
  50  
  51          $target = fopen( $path, 'w' );
  52          fseek( $temp, 0, SEEK_SET );
  53          stream_copy_to_stream( $temp, $target );
  54          fclose( $target );
  55  
  56          fclose( $temp );
  57  
  58          if( ! empty( $temp_file_name ) )
  59          { // Unlink the temp file
  60              @unlink( $temp_file_name );
  61          }
  62  
  63          return true;
  64      }

  67       * Get a content of the uploaded file
  68       *
  69       * @return string|array Content OR array with errors

  71  	function get_content()
  72      {
  73          $input = fopen( 'php://input', 'rb' );
  74          $temp_file_name = '';
  75          $temp = open_temp_file( $temp_file_name );
  76          if( is_string( $temp ) )
  77          { // Error on create a temp file
  78              return array(
  79                      'text'   => $temp,
  80                      'status' => 'error',
  81                  );
  82          }
  83  
  84          stream_copy_to_stream( $input, $temp );
  85          fclose( $input );
  86  
  87          fseek( $temp, 0, SEEK_SET );
  88          $contents = '';
  89  
  90          load_funcs( 'tools/model/_system.funcs.php' );
  91          $memory_limit = system_check_memory_limit();
  92  
  93          while( ! feof( $temp ) )
  94          {
  95              $curr_mem_usage = memory_get_usage( true );
  96              if( ( $memory_limit - $curr_mem_usage ) < 8192 )
  97              { // Don't try to load the next portion of image into the memory because it would cause 'Allowed memory size exhausted' error
  98                  fclose( $temp );
  99                  return array(
 100                          'text'   => T_( 'The server (PHP script) has not enough available memory to receive this large file!' ),
 101                          'status' => 'error',
 102                      );
 103              }
 104              $contents .= fread( $temp, 8192 );
 105          }
 106  
 107          fclose( $temp );
 108  
 109          if( ! empty( $temp_file_name ) )
 110          { // Unlink the temp file
 111              @unlink( $temp_file_name );
