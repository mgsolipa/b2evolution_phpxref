  48  // Get rediredt_to param

  52  param( 'blog', 'integer', '' );

  54  param( 'post_id', 'integer', '' );

  59

  61  $sender_name = param( $dummy_fields[ 'name' ], 'string', '' );
  62  $sender_address = utf8_strtolower( param( $dummy_fields[ 'email' ], 'string', '' ) );
  63  $subject = param( $dummy_fields[ 'subject' ], 'string', '' );
  64  $message = param( $dummy_fields[ 'content' ], 'html', '' );    // We accept html but we will NEVER display it
  65  // save the message original content
  66  $original_content = $message;
  67  
  68  // Prevent register_globals injection!
  69  $recipient_address = '';
  70  $recipient_name = '';
  71  $recipient_User = NULL;
  72  $Comment = NULL;
  73  
  74  // Core param validation
  75  
  76  if( empty($subject) )
  77  {
  78      $Messages->add( T_('Please fill in the subject of your message.'), 'error' );
  79  }
  80  
  81  if( empty( $message ) )
  82  { // message should not be empty!
  83      $Messages->add( T_('Please do not send empty messages.'), 'error' );
  84  }
  85  elseif( $antispam_on_message_form && antispam_check( $message ) )
  86  { // a blacklisted keyword ha sbeen found in the message:
  87      $Messages->add( T_('The supplied message is invalid / appears to be spam.'), 'error' );
  88  }
  89  
  90  // Getting current blog info:
  91  $BlogCache = & get_BlogCache();
  92  if( !empty( $comment_id ) || !empty( $post_id ) )
  93  {
  94      $Blog = & $BlogCache->get_by_ID( $blog );    // Required
  95  }
  96  else
  97  {
  98      $Blog = & $BlogCache->get_by_ID( $blog, true, false );    // Optional
  99  }
 100  
 101  $allow_msgform = '';
 102  if( ! empty( $recipient_id ) )
 103  { // Get the email address for the recipient if a member:
 104      $UserCache = & get_UserCache();
 105      $recipient_User = & $UserCache->get_by_ID( $recipient_id );
 106  
 107      // Check if current User allows to be contacted by email:
 108      $allow_msgform = $recipient_User->get_msgform_possibility( NULL, 'email' );
 109      if( $allow_msgform != 'email' )
 110      { // should be prevented by UI
 111          debug_die( 'Invalid recipient or no permission to contact by email!' );
 112      }
 113  }
 114  elseif( ! empty( $comment_id ) )
 115  { // Get the email address for the recipient if a visiting commenter:
 116      $CommentCache = & get_CommentCache();
 117      $Comment = $CommentCache->get_by_ID( $comment_id );
 118  
 119      if( empty( $Comment ) )
 120      {
 121          debug_die( 'Invalid request, comment doesn\'t exists!' );
 122      }
 123  
 124      if( $recipient_User = & $Comment->get_author_User() )
 125      { // Comment is from a registered user:
 126          // Check if current User allows to be contacted by email:
 127          $allow_msgform = $recipient_User->get_msgform_possibility( NULL, 'email' );
 128          if( $allow_msgform != 'email' )
 129          { // should be prevented by UI
 130              debug_die( 'Invalid recipient or no permission to contact by email!' );
 131          }
 132      }
 133      elseif( empty($Comment->allow_msgform) )
 134      { // should be prevented by UI
 135          debug_die( 'Invalid recipient or no permission to contact by email!' );
 136      }
 137      else
 138      {
 139          $recipient_name = $Comment->get_author_name();
 140          $recipient_address = $Comment->get_author_email();
 141      }
 142  }

 144  if( empty($sender_name) )
 145  {
 146      $Messages->add( T_('Please fill in your name.'), 'error' );
 147  }
 148  if( empty($sender_address) )
 149  {
 150      $Messages->add( T_('Please fill in your email.'), 'error' );
 151  }
 152  elseif( !is_email($sender_address) || antispam_check( $sender_address ) ) // TODO: dh> using antispam_check() here might not allow valid users to contact the admin in case of problems due to the antispam list itself.. :/
 153  {
 154      $Messages->add( T_('Supplied email address is invalid.'), 'error' );
 155  }
 156  
 157  if( empty( $recipient_User ) && empty( $recipient_address ) )
 158  { // should be prevented by UI
 159      debug_die( 'No recipient specified!' );
 160  }

 162  // opt-out links:
 163  if( $recipient_User )
 164  { // Member:
 165      // Change the locale so the email is in the recipients language

 167  }
 168  else
 169  { // Visitor:
 170      // We don't know the recipient's language - Change the locale so the email is in the blog's language:

 175  $Plugins->trigger_event( 'MessageFormSent', array(

 188  if( $success_message )

 191              'sender_name'    => $sender_name,
 192              'sender_address' => $sender_address,
 193              'Blog'           => $Blog,
 194              'message'        => $message,
 195              'comment_id'     => $comment_id,
 196              'post_id'        => $post_id,
 197              'recipient_User' => $recipient_User,

 206      }
 207      else
 208      { // Send mail to registered user
 209          $success_message = send_mail_to_User( $recipient_User->ID, $subject, 'contact_message_new', $email_template_params, false, array( 'Reply-To' => $sender_address ) );
 210      }
 211  
 212      // restore the locale to the blog visitor language, before we would display an error message
 213      locale_restore_previous();
 214  
 215      if( $success_message )
 216      { // Email has been sent successfully
 217          if( ! is_logged_in() )
 218          { // We should save a counter (only for anonymous users)
 219              antispam_increase_counter( 'contact_email' );
 220          }
 221      }
 222      else
 223      { // Could not send email
 224          if( $demo_mode )
 225          {
 226              $Messages->add( 'Sorry, could not send email. Sending email in demo mode is disabled.', 'error' );
 227          }
 228          else
 229          {
 230              $Messages->add( T_('Sorry, could not send email.')
 231                  .'<br />'.T_('Possible reason: the PHP mail() function may have been disabled on the server.'), 'error' );

 233      }
 234  }
 235  else
 236  { // Restore the locale to the blog visitor language even in case of errors
 237      locale_restore_previous();

 242  $Plugins->trigger_event( 'MessageFormSentCleanup', array(
 243          'success_message' => $success_message,
 244      ) );
 245  
 246  if( empty( $redirect_to ) && empty( $Blog ) )
 247  {
 248      $redirect_to = $baseurl;
 249  }
 250  if( $success_message )
 251  {
 252      // Never say to whom we sent the email -- prevent user enumeration.
 253      $Messages->add( T_('Your message has been sent by email.'), 'success' );
 254      if( empty( $redirect_to ) )

 258          {
 259              $redirect_to = url_add_param( $redirect_to, 'disp=msgform&recipient_id='.$recipient_User->ID );
 260          }
 261      }
 262      header_redirect( $redirect_to );
 263      // exited here
 264  }
 265  
 266  // unsuccessful message send, save message params into the Session to not lose the content
 267  $unsaved_message_params = array();
 268  $unsaved_message_params[ 'sender_name' ] = $sender_name;
 269  $unsaved_message_params[ 'sender_address' ] = $sender_address;
 270  $unsaved_message_params[ 'subject' ] = $subject;
 271  $unsaved_message_params[ 'message' ] = $original_content;
 272  save_message_params_to_session( $unsaved_message_params );
 273  
 274  if( param_errors_detected() || empty( $redirect_to ) )
 275  {
 276      $redirect_to = url_add_param( $Blog->gen_blogurl(), 'disp=msgform&recipient_id='.$recipient_id, '&' );
 277  }
 278  header_redirect( $redirect_to );
 279  //exited here
 280  
 281  ?>
