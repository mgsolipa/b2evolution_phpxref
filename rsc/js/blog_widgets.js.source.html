 215  function makeDroppable( selector )
 216  {
 217      jQuery( selector ).droppable(
 218      {
 219          accept: ".draggable_widget", // classname of objects that can be dropped
 220          hoverClass: "droppable-hover", // classname when object is over this one
 221          greedy: true, // stops propogation if over more than one
 222          tolerance : "pointer", // droppable active when cursor over
 223          delay: 1000,
 224          drop: function(ev, ui)
 225          {    // function called when object dropped
 226              jQuery( ".fade_me" ).removeClass( "fade_me" ); // remove any existing fades
 227              jQuery( '.available_widgets' ).removeClass( 'available_widgets_active' ); // close any open windows
 228              if( !jQuery( this ).hasClass( "available_widgets" ) )
 229              {    // we're not deleting it
 230                  if( jQuery( ui.draggable ).hasClass( "new_widget" ) )
 231                  {    // this is a new widget, we need to treat it diffently
 232                      addNewWidget( ui.draggable, this ); // add as new widget
 233                  }
 234                  else
 235                  {    // this is an existing widget, just move it
 236                      jQuery( ui.draggable ).insertAfter( this ); // add the dragged widget before this widget
 237                      jQuery( ui.draggable ).addClass( "fade_me server_update" ); // add fade class
 238                      jQuery( ui.draggable ).droppable( "enable" );    // enable dropping if disabled
 239                  }
 240              }
 241              else
 242              { // we might be deleting the widget
 243                  if(  !jQuery( ui.draggable ).hasClass( "new_widget" ) )
 244                  { // we're deleting it
 245                      jQuery( ui.draggable ).remove();
 246                  }
 247              }
 248              doFade( ".fade_me" ); // fade the widget
 249              colourWidgets();
 250              sendWidgetOrder();// send the new order to the server
 251          }
 252      });
 253  }

 294  /**
 295   * Buffered server call
 296   *
 297   * Waits until delay period is over and then sends new order to the server
 298   * only sends if the current widget order has changed since last update
 299   */
 300  function bufferedServerCall()
 301  {
 302      var new_widget_order = getWidgetOrder();
 303      if( new_widget_order != current_widgets )
 304      {    // widget order has changed, we need to update
 305          jQuery( '#server_messages' ).html( '<div class="action_messages container-fluid"><ul><li><div class="alert alert-dismissible alert-info fade in">'+T_( 'Saving changes' )+'</div></li></ul></div>' ); // inform user
 306  
 307          current_widgets = new_widget_order; // store current order
 308          //add crumbs here
 309          new_widget_order += '&' + crumb_url;
 310          jQuery( '.pending_update' ).removeClass( 'pending_update' ).addClass( 'server_updating' ); // change class to "updating"
 311  
 312          SendAdminRequest( 'widgets', 're-order', new_widget_order, false ); // send current order to server
 313      }
 314      else
 315      {    // widget order either hasn't changed or has been changed back to original order
 316          jQuery( '#server_messages' ).html( '<div class="action_messages container-fluid"><ul><li><div class="alert alert-dismissible alert-warning fade in">'
 317                          +T_( 'Widget order unchanged' )+'</div></li></ul></div>' ); // inform user
 318          jQuery( '.pending_update' ).removeClass( 'pending_update' ); // remove "needs updating"
 319          colourWidgets(); // redo widget colours
 320      }
 321  }

 324  /**
 325   * Gets the current widget order
 326   *
 327   * @return string widget order
 328   */
 329  function getWidgetOrder()
 330  {
 331      // need to get every container, then every widget in container and send the lot to the server
 332      var containers = new Array()
 333      jQuery( '.widget_container' ).each(function()
 334      {
 335          var container_name = jQuery( this ).attr('id');
 336          containers[ container_name ] = '';
 337          jQuery( '#'+container_name+' .draggable_widget' ).each( function(){
 338              if( jQuery( this ).attr( 'id' ) && jQuery( this ).attr( 'id' ) != 'undefined' )
 339              {    // this is a widget
 340                   containers[container_name] += jQuery( this ).attr( 'id' ) + ', ';
 341              }
 342          });
 343      });
 344  
 345      var query_string = '';
 346      var containers_list = '';
 347      for( container in containers )
 348      {
 349          query_string += container+'='+containers[container]+'&';
 350          containers_list += container+',';
 351      }
 352  
 353      var r = 'blog='+blog+'&'+query_string+'container_list='+containers_list;
 354  
 355      // console.log( r );
 356  
 357      return r;
 358  }
