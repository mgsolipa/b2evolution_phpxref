   8  jQuery( document ).ready( function()
   9  {
  10      var IE_version = false;
  11      if( typeof( navigator ) !== "undefined" && typeof( navigator.appVersion ) !== "undefined" )
  12      {
  13          var app_match = navigator.appVersion.match( /msie (\d+)/i );
  14          if( app_match )
  15          {
  16              IE_version = parseInt( app_match[1] );
  17          }
  18      }
  19      if( IE_version && IE_version <= 9 )
  20      { // Dont allow this plugin on IE <= 9
  21          return;
  22      }
  23  
  24      var mentions = [];
  25      jQuery( ".user.login" ).each( function()
  26      { // Init array with all usernames from the page
  27          var login = jQuery( this ).text();
  28          if( login != "" && mentions.indexOf( login ) == -1 )
  29          {
  30              if( login[0] == "@" )
  31              {
  32                  login = login.substr( 1 );
  33              }
  34              mentions.push( login );
  35          }
  36      } );
  37  
  38      jQuery( "textarea.autocomplete_usernames" ).textcomplete(
  39      [ {
  40          match: /\B@(\w+)$/,
  41          search: function ( term, callback )
  42          {
  43              if( term.length < 4 )
  44              { // Search only on the page
  45                  callback( jQuery.map( mentions, function ( mention )
  46                  { return mention.indexOf(term) === 0 ? mention : null; } ) );
  47              }
  48              else
  49              { // Also search in DB by AJAX
  50                  jQuery.ajax(
  51                  {
  52                      type: "POST",
  53                      dataType: "JSON",
  54                      url: htsrv_url + "anon_async.php",
  55                      data: "action=autocomplete_usernames&q=" + term,
  56                      success: function( db_mentions )
  57                      {
  58                          db_mentions = db_mentions.concat( mentions );
  59                          callback( jQuery.map( db_mentions, function ( mention )
  60                          { return mention.indexOf(term) === 0 ? mention : null; } ) );
  61                      }
  62                  } );
  63              }
  64          },
  65          index: 1,
  66          replace: function ( mention ) { return "@" + mention + " "; },
  67          cache: true
  68      } ] );
  69  } );
