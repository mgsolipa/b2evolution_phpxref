  23  /**
  24   * Open or close a clickopen area (by use of CSS style).
  25   *
  26   * You have to define a div with id clickdiv_<ID> and a img with clickimg_<ID>,
  27   * where <ID> is the first param to the function.
  28   *
  29   * Used to expand/collapse in BACK-office:
  30   *  - _file.funcs.php: to toggle the subfolders in directory list
  31   *  - _backup_options.form.php: to toggle the backup options on upgrade action
  32   *  - _plugin_settings.form.php: to toggle the plugin event settings on edit plugin page
  33   *
  34   * @param string html id of the element to toggle
  35   * @param string CSS display property to use when visible ('inline', 'block')
  36   * @return false
  37   */
  38  function toggle_clickopen( id, hide, displayVisible )
  39  {
  40      if( !( clickdiv = document.getElementById( 'clickdiv_'+id ) )
  41              || !( clickimg = document.getElementById( 'clickimg_'+id ) ) )
  42      {
  43          alert( 'ID '+id+' not found!' );
  44          return false;
  45      }
  46  
  47      if( typeof(hide) == 'undefined' )
  48      {
  49          hide = clickdiv.style.display != 'none';
  50      }
  51  
  52      if( typeof(displayVisible) == 'undefined' )
  53      {
  54          displayVisible = ''; // setting it to "empty" is the default for an element's display CSS attribute
  55      }
  56  
  57      clickimg = jQuery( clickimg );
  58      if( clickimg.hasClass( 'fa' ) || clickimg.hasClass( 'glyphicon' ) )
  59      { // Fontawesome icon | Glyph bootstrap icon
  60          if( clickimg.data( 'toggle' ) != '' )
  61          { // This icon has a class name to toggle
  62              var icon_prefix = ( clickimg.hasClass( 'fa' ) ? 'fa' : 'glyphicon' );
  63              if( clickimg.data( 'toggle-orig-class' ) == undefined )
  64              { // Store original class name in data
  65                  clickimg.data( 'toggle-orig-class', clickimg.attr( 'class' ).replace( new RegExp( '^'+icon_prefix+' (.+)$', 'g' ), '$1' ) );
  66              }
  67              if( clickimg.hasClass( clickimg.data( 'toggle-orig-class' ) ) )
  68              { // Replace original class name with exnpanded
  69                  clickimg.removeClass( clickimg.data( 'toggle-orig-class' ) )
  70                      .addClass( icon_prefix + '-' + clickimg.data( 'toggle' ) );
  71              }
  72              else
  73              { // Revert back original class
  74                  clickimg.removeClass( icon_prefix + '-' + clickimg.data( 'toggle' ) )
  75                      .addClass( clickimg.data( 'toggle-orig-class' ) );
  76              }
  77          }
  78      }
  79      else
  80      { // Sprite icon
  81          var xy = clickimg.css( 'background-position' ).match( /-*\d+/g );
  82          // Shift background position to the right/left to the one icon in the sprite
  83          clickimg.css( 'background-position', ( parseInt( xy[0] ) + ( hide ? 16 : - 16 ) ) + 'px ' + parseInt( xy[1] ) + 'px' );
  84      }
  85  
  86      // Hide/Show content block
  87      clickdiv.style.display = hide ? 'none' : displayVisible;
  88  
  89      return false;
  90  }

 139  /**
 140   * Fade jQuery selector via backgrounds colors (bgs), back to original background
 141   * color and then remove any styles (from animations and others)
 142   *
 143   * Used only on BACK-office in the following files:
 144   *  - _misc_js.funcs.php
 145   *  - blog_widgets.js
 146   *  - links.js
 147   *  - _file_list.inc.php
 148   *
 149   * @param string|jQuery
 150   * @param Array
 151   * @param object Options ("speed")
 152   */
 153  function evoFadeBg( selector, bgs, options )
 154  {
 155      var origBg = jQuery(selector).css("backgroundColor");
 156      var speed = options && options.speed || '"slow"';
 157  
 158      var toEval = 'jQuery(selector).animate({ backgroundColor: ';
 159      for( e in bgs )
 160      {
 161          if( typeof( bgs[e] ) != 'string' )
 162          { // Skip wrong color value
 163              continue;
 164          }
 165          toEval += '"'+bgs[e]+'"'+'}, '+speed+' ).animate({ backgroundColor: ';
 166      }
 167      toEval += 'origBg }, '+speed+', "", function(){jQuery( this ).css( "backgroundColor", "" );});';
 168  
 169      eval(toEval);
 170  }

 173  /**
 174   * Set the action attribute on a form, including a Safari fix.
 175   *
 176   * This is so complicated, because the form also can have a (hidden) action value.
 177   *
 178   * @return boolean
 179   */
 180  function set_new_form_action( form, newaction )
 181  {
 182      // Stupid thing: having a field called action !
 183      var saved_action = form.attributes.getNamedItem('action').value;
 184      form.attributes.getNamedItem('action').value = newaction;
 185  
 186      // requested host+directory, used for Opera workaround below
 187      var reqdir = location.href.replace(/(\/)[^\/]*$/, "$1");
 188  
 189      // FIX for Safari (2.0.2, OS X 10.4.3) - (Konqueror does not fail here)
 190      if( form.attributes.getNamedItem('action').value != newaction
 191          && form.attributes.getNamedItem('action').value != reqdir+newaction /* Opera 9.25: action holds the complete URL, not just the given filename */
 192      )
 193      { // Setting form.action failed! (This is the case for Safari)
 194          // NOTE: checking "form.action == saved_action" (or through document.getElementById()) does not work - Safari uses the input element then
 195          { // _Setting_ form.action however sets the form's action attribute (not the input element) on Safari
 196              form.action = newaction;
 197          }
 198  
 199          if( form.attributes.getNamedItem('action').value != newaction )
 200          { // Still old value, did not work.
 201              alert('set_new_form_action: Cannot set new form action (Safari workaround).');
 202              return false;
 203          }
 204      }
 205      // END FIX for Safari
 206  
 207      return true;
 208  }

 242  /**
 243   * Submits the form after setting its action attribute to "newaction" and the blog value to "blog" (if given).
 244   *
 245   * This is used to switch to another blog or tab, but "keep" the input in the form.
 246   */
 247  function b2edit_reload( form, newaction, blog, params, reset )
 248  {
 249      // Set the new form action URL:
 250      if( ! set_new_form_action(form, newaction) )
 251      {
 252          return false;
 253      }
 254  
 255      var hidden_action_set = false;
 256  
 257      // Set the new form "action" HIDDEN value:
 258      if( form.elements.namedItem("actionArray[update]") )
 259      {
 260          jQuery(form).append('<input type="hidden" name="action" value="edit_switchtab" />');
 261          hidden_action_set = true;
 262      }
 263      else if( form.elements.namedItem("actionArray[create]") )
 264      {
 265          jQuery(form).append('<input type="hidden" name="action" value="new_switchtab" />');
 266          hidden_action_set = true;
 267      }
 268      else
 269      {
 270          jQuery(form).append('<input type="hidden" name="action" value="switchtab" />');
 271          hidden_action_set = true;
 272      }
 273  
 274      if( hidden_action_set && ( typeof params != 'undefined' ) )
 275      {
 276          for( param in params )
 277          {
 278              jQuery(form).append('<input type="hidden" name="' + param + '" value="' + params[param] + '" />');
 279          }
 280      }
 281  
 282      // Set the blog we are switching to:
 283      if( typeof blog != 'undefined' && blog != 'undefined' )
 284      {
 285          if( blog == null )
 286          { // Set to an empty string, otherwise POST param value will be 'null' in IE and it cause issues
 287              blog = '';
 288          }
 289          form.elements.blog.value = blog;
 290      }
 291  
 292      // form.action.value = 'reload';
 293      // form.post_title.value = 'demo';
 294      // alert( form.action.value + ' ' + form.post_title.value );
 295  
 296      // disable bozo validator if active:
 297      // TODO: dh> this seems to actually delete any events attached to beforeunload, which can cause problems if e.g. a plugin hooks this event
 298      window.onbeforeunload = null;
 299  
 300      if( typeof( reset ) != 'undefined' && reset == true )
 301      { // Reset the form:
 302          form.reset();
 303      }
 304  
 305      // Submit the form:
 306      form.submit();
 307  
 308      return false;
 309  }
