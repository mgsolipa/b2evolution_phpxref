   1  /**
   2   * "BOZO VALIDATOR 2" : Check if a form has been changed but not submitted when a bozo clicks
   3   * on a link which will result in potential data input loss
   4   * Tested on Firefox (XP & Mac OSX) , IE6 (XP), Safari (Mac OSX)
   5   * b2evolution - http://b2evolution.net/
   6   * @version $Id: bozo_validator.js 7905 2014-12-26 07:53:17Z yura $
   7   */
   8  var bozo_confirm_mess;
   9  
  10  var bozo = {
  11  
  12      // array of changes for each form we need to verify (needed to detect if another form has changed when we submit)
  13      'tab_changes' : Object(),
  14      // Total number of changes
  15      'nb_changes' : 0,
  16  
  17      // If no translated message has been provided, use this default:
  18      'confirm_mess' : bozo_confirm_mess ? bozo_confirm_mess : 'You have modified this form but you haven\'t submitted it yet.\nYou are about to lose your edits.\nAre you sure?',
  19  
  20      /**
  21       *    BOZO VALIDATOR INITIALIZATION
  22       * This is designed to track changes on forms whith an ID including '_checkchanges'
  23       */
  24      init: function ( )
  25      {
  26          // Loop through all forms:
  27          jQuery("form")
  28              // Initialize this form as having no changes yet:
  29              .each( function() { bozo.tab_changes[this.id] = 0; } )
  30              // add submit event on the form to control if there are changes on others forms:
  31              .submit(bozo.validate_submit)
  32              // Filter all "*_checkchanges" forms:
  33              .filter("[id$='_checkchanges']")
  34              // Hook "click" event for reset elements:
  35              .find("input[type=reset]:not([class$=_nocheckchanges])").click(bozo.reset_changes).end()
  36              // Hook "change" and "keypress" event for all others:
  37              .find("input[type=text], input[type=password], input[type=radio], input[type=checkbox], input[type=file], textarea")
  38                  .not("[class$=_nocheckchanges]")
  39                      .bind("change", bozo.change)
  40                      .bind("keypress", bozo.change);
  41      },
  42  
  43  
  44      /**
  45       *    caters for the differences between Internet Explorer and fully DOM-supporting browsers
  46       */
  47      findTarget: function ( e )
  48      {
  49          var target;
  50          if (window.event && window.event.srcElement)
  51              target = window.event.srcElement;
  52          else if (e && e.target)
  53              target = e.target;
  54  
  55          if (!target)
  56              return null;
  57  
  58          return target;
  59      },
  60  
  61  
  62      /*
  63       * called when there is a change event on an element
  64       */
  65      change: function( e )
  66      {    // Get the target element
  67          var target = bozo.findTarget( e );
  68          // Update changes number for his parent form
  69          bozo.tab_changes[ get_form( target ).id ]++;
  70          // Update Total changes number
  71          bozo.nb_changes++;
  72      },
  73  
  74  
  75      /*
  76       * Call when there is a click on a reset input
  77       * Reset changes
  78       */
  79      reset_changes: function ( e )
  80      {
  81          // Loop on the forms changes array
  82          for( i in bozo.tab_changes )
  83          {    // Reset changes number to 0
  84              bozo.tab_changes[i] = 0;
  85          }
  86          // Total changes number
  87          bozo.nb_changes = 0;
  88      },
  89  
  90  
  91      /*
  92       *    Called when there is a click event on a submit button
  93       *    If there are no changes on others forms, cancel onbeforeunload event
  94       */
  95      validate_submit: function( e )
  96      {
  97              var target = bozo.findTarget(e);
  98  
  99              var other_form_changes = 0;
 100  
 101              // Loop on the forms changes array
 102              for( i in bozo.tab_changes )
 103              {
 104                  if ( ( i != get_form( target ).id ) && bozo.tab_changes[i] )
 105                  {    // Another form contains input changes
 106                      other_form_changes++;
 107                  }
 108              }
 109  
 110              if( !other_form_changes )
 111              {    // There are no changes on others forms, so cancel onbeforeunload event
 112                  window.onbeforeunload = '';
 113                  return;
 114              }
 115      },
 116  
 117  
 118  
 119      /**
 120       * Called when the user close the window
 121       * Ask confirmation to close page without saving changes if there have been changes on all form inputs
 122       */
 123      validate_close: function( e )
 124      {
 125          if( bozo.nb_changes )
 126          { // there are input changes
 127              if( window.event )
 128              { // For ie:
 129                  window.event.returnValue = bozo.confirm_mess;
 130              }
 131              // For all other browsers:
 132              return bozo.confirm_mess;
 133          }
 134      }
 135  
 136  }
 137  
 138  // Init Bozo validator when the window is loaded:
 139  jQuery( document ).bind( 'ready', bozo.init );
 140  // Note: beforeunload is a "very special" event and cannot be added with addEvent:
 141  window.onbeforeunload = bozo.validate_close;
