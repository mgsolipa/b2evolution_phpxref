  14  function evo_display_position_onchange( selectInput, url, crumb )
  15  {
  16      var oThis = selectInput;
  17      var new_position = selectInput.value;
  18      jQuery.get( url + 'async.php?action=set_object_link_position&link_ID=' + selectInput.id.substr(17) + '&link_position=' + new_position + '&crumb_link=' + crumb, {
  19      }, function(r, status) {
  20          r = ajax_debug_clear( r );
  21          if( r == "OK" ) {
  22              evoFadeSuccess( jQuery(oThis.form).closest('tr') );
  23              jQuery(oThis.form).closest('td').removeClass('error');
  24              if( new_position == 'cover' )
  25              { // Position "Cover" can be used only by one link
  26                  jQuery( 'select[name=link_position][id!=' + selectInput.id + '] option[value=cover]:selected' ).each( function()
  27                  { // Replace previous position with "Inline"
  28                      jQuery( this ).parent().val( 'aftermore' );
  29                      evoFadeSuccess( jQuery( this ).closest('tr') );
  30                  } );
  31              }
  32          } else {
  33              jQuery(oThis).val(r);
  34              evoFadeFailure( jQuery(oThis.form).closest('tr') );
  35              jQuery(oThis.form).closest('td').addClass('error');
  36          }
  37      } );
  38      return false;
