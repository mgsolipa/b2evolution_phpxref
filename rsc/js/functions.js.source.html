  18  function pop_up_window( href, target, width, height, params )
  19  {
  20      if( typeof(width) == 'undefined' )
  21      {
  22          width = 750;
  23      }
  24  
  25      if( typeof(height) == 'undefined' )

  27          height = 550;
  28      }

  34      {
  35          params = 'scrollbars=yes, status=yes, resizable=yes, menubar=yes';
  36      }
  37  
  38      params = 'width=' + width + ', height=' + height + ', ' + 'left=' + left + ', top=' + top + ', ' + params;
  39  
  40      // Open window:
  41      opened = window.open( href, target, params );
  42  
  43      // Bring to front!
  44      opened.focus();
  45  
  46      if( typeof(openedWindows) == 'undefined' )
  47      {
  48          openedWindows = new Array(opened);
  49      }
  50      else
  51      {
  52          openedWindows.push(opened);
  53      }
  54  
  55      // Tell the caller there is no need to process href="" :
  56      return false;
  57  }

  81  function textarea_wrap_selection( myField, before, after, replace, target_document )
  82  {
  83      target_document = target_document || document;
  84  
  85      var hook_params = {
  86          'element': myField,
  87          'before': before,
  88          'after': after,
  89          'replace': replace,
  90          'target_document': target_document
  91      };
  92  
  93      // First try, if a JavaScript callback is registered to handle this.
  94      // E.g. the tinymce_plugin uses registers "wrap_selection_for_itemform_post_content"
  95      //      to replace the (non-)selection
  96      if( b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
  97      {
  98          return;
  99      }
 100  
 101      if( window.opener && ( typeof window.opener != "undefined" ) )
 102      {
 103          try
 104          { // Try find object 'b2evo_Callbacks' on window.opener to avoid halt error when page was opened from other domain
 105              if( window.opener.b2evo_Callbacks &&
 106             ( typeof window.opener.b2evo_Callbacks != "undefined" ) &&
 107             window.opener.b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
 108              { // callback in opener document (e.g. "Files" popup)
 109                  return;
 110              }
 111          }
 112          catch( e )
 113          { // Catch an error of the cross-domain restriction
 114              // Ignore this error because it dies when browser has no permission to access to other domain windows
 115          }
 116      }
 117  
 118      if( window.parent
 119          && ( typeof window.parent != "undefined" )
 120          && window.parent.b2evo_Callbacks
 121          && ( typeof window.parent.b2evo_Callbacks != "undefined" ) )
 122      {    // callback in parent document (e.g. "Links" iframe)
 123          if( window.parent.b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
 124          {
 125              return;
 126          }
 127      }
 128  
 129      // Basic handling:
 130      if(target_document.selection)
 131      { // IE support:
 132          myField.focus();
 133          sel = target_document.selection.createRange();
 134          if( replace )
 135          {
 136              sel.text = before + after;
 137          }
 138          else
 139          {
 140              sel.text = before + sel.text + after;
 141          }
 142          myField.focus();
 143      }
 144      else if (myField.selectionStart || myField.selectionStart == '0')
 145      { // MOZILLA/NETSCAPE support:
 146          var startPos = myField.selectionStart;
 147          var endPos = myField.selectionEnd;
 148          var cursorPos;
 149  
 150          var scrollTop, scrollLeft;
 151          if( myField.type == 'textarea' && typeof myField.scrollTop != 'undefined' )
 152          { // remember old position
 153              scrollTop = myField.scrollTop;
 154              scrollLeft = myField.scrollLeft;
 155          }
 156  
 157          if( replace )
 158          {
 159              myField.value = myField.value.substring( 0, startPos)
 160                  + before
 161                  + after
 162                  + myField.value.substring( endPos, myField.value.length);
 163              cursorPos = startPos + before.length + after.length;
 164          }
 165          else
 166          {
 167              myField.value = myField.value.substring( 0, startPos)
 168                  + before
 169                  + myField.value.substring(startPos, endPos)
 170                  + after
 171                  + myField.value.substring( endPos, myField.value.length);
 172              cursorPos = endPos + before.length + after.length;
 173          }
 174  
 175          if( typeof scrollTop != 'undefined' )
 176          { // scroll to old position
 177              myField.scrollTop = scrollTop;
 178              myField.scrollLeft = scrollLeft;
 179          }
 180  
 181          myField.focus();
 182          myField.selectionStart = cursorPos;
 183          myField.selectionEnd = cursorPos;
 184      }
 185      else
 186      { // Default browser support:
 187          myField.value += before + after;
 188          myField.focus();
 189      }
 190  }

 256  /**
 257   * Open or close a filter area (by use of CSS style).
 258   *
 259   * You have to define a div with id clickdiv_<ID> and a img with clickimg_<ID>,
 260   * where <ID> is the first param to the function.
 261   *
 262   * Used to expand/collapse a filter area of Results table on FRONT-office and BACK-office in the following files:
 263   *  - _uiwidget.class.php
 264   *
 265   * @param string html id of the element to toggle
 266   * @return false
 267   */
 268  function toggle_filter_area( filter_name )
 269  {
 270      // Find objects to toggle:
 271      var clickdiv = jQuery( '#clickdiv_'+filter_name );
 272      var clickimg = jQuery( '#clickimg_'+filter_name );
 273      if( clickdiv.length == 0 || clickimg.length == 0 )
 274      {
 275          alert( 'ID '+filter_name+' not found!' );
 276          return false;
 277      }
 278  
 279      if( clickimg.hasClass( 'fa' ) || clickimg.hasClass( 'glyphicon' ) )
 280      {    // Fontawesome icon | Glyph bootstrap icon
 281          if( clickimg.data( 'toggle' ) != '' && clickimg.data( 'toggle' ) != undefined )
 282          {    // This icon has a class name to toggle
 283              var icon_prefix = ( clickimg.hasClass( 'fa' ) ? 'fa' : 'glyphicon' );
 284              if( clickimg.data( 'toggle-orig-class' ) == undefined )
 285              {    // Store original class name in data
 286                  clickimg.data( 'toggle-orig-class', clickimg.attr( 'class' ).replace( new RegExp( '^'+icon_prefix+' (.+)$', 'g' ), '$1' ) );
 287              }
 288              if( clickimg.hasClass( clickimg.data( 'toggle-orig-class' ) ) )
 289              {    // Replace original class name with exnpanded
 290                  clickimg.removeClass( clickimg.data( 'toggle-orig-class' ) )
 291                      .addClass( icon_prefix + '-' + clickimg.data( 'toggle' ) );
 292              }
 293              else
 294              {    // Revert back original class
 295                  clickimg.removeClass( icon_prefix + '-' + clickimg.data( 'toggle' ) )
 296                      .addClass( clickimg.data( 'toggle-orig-class' ) );
 297              }
 298          }
 299      }
 300      else
 301      {    // Sprite icon
 302          var xy = clickimg.css( 'background-position' ).match( /-*\d+/g );
 303          // Shift background position to the right/left to the one icon in the sprite
 304          clickimg.css( 'background-position', ( parseInt( xy[0] ) + ( !clickdiv.is( ':hidden' ) ? 16 : - 16 ) ) + 'px ' + parseInt( xy[1] ) + 'px' );
 305      }
 306  
 307      if( !clickdiv.is( ':hidden' ) )
 308      {    // Hide/collapse filters:
 309          clickdiv.slideUp( 500 );
 310          jQuery.post( htsrv_url+'anon_async.php?action=collapse_filter&target='+filter_name );
 311      }
 312      else
 313      {    // Show/expand filters
 314          clickdiv.slideDown( 500 );
 315          jQuery.post( htsrv_url+'anon_async.php?action=expand_filter&target='+filter_name );
 316      }
 317  
 318      return false;
 319  }

 322  /*
 323   * Javascript callback handling, for helping plugins to interact in Javascript.
 324   *
 325   * This is, so one plugin (e.g. the tinymce_plugin) can say that it handles insertion of raw
 326   * content into a specific element ("itemform_post_content" in this case):
 327   *
 328   * <code>
 329   * if( typeof b2evo_Callbacks == "object" )
 330   * { // add a callback, that lets us insert the
 331   *   b2evo_Callbacks.register_callback( "wrap_selection_for_itemform_post_content", function(value) {
 332   *       tinyMCE.execCommand( "mceInsertRawHTML", false, value );
 333   *       return true;
 334   *     } );
 335   * }
 336   * </code>
 337   *
 338   * and others (e.g. the smilies_plugin or the youtube_plugin) should first try to use this
 339   * callback to insert the HTML:
 340   *
 341   * if( typeof b2evo_Callbacks == 'object' )
 342   * { // see if there's a callback registered that should handle this:
 343   *   if( b2evo_Callbacks.trigger_callback("wrap_selection_for_"+b2evoCanvas.id, tag) )
 344   *   {
 345   *     return;
 346   *   }
 347   * }
 348   */
 349  function b2evo_Callbacks() {
 350      this.eventHandlers = new Array();
 351  };
 352  
 353  b2evo_Callbacks.prototype = {
 354      register_callback : function(event, f) {
 355          if( typeof this.eventHandlers[event] == "undefined" )
 356          {
 357              this.eventHandlers[event] = new Array();
 358          }
 359          this.eventHandlers[event][this.eventHandlers[event].length] = f;
 360      },
 361  
 362      /**
 363       * @param String event name
 364       * @param mixed argument1
 365       * @param mixed argument2
 366       * ...
 367       * @return boolean true, if any callback returned true
 368       *                 null, if no callback registered
 369       */
 370      trigger_callback : function(event, args) {
 371  
 372          if( typeof this.eventHandlers[event] == "undefined" )
 373          {
 374              return null;
 375          }
 376  
 377          var r = false;
 378  
 379          // copy arguments and build function param string for eval():
 380          var cb_args = '';
 381          var cb_arguments = arguments;
 382          for( var i = 1; i < arguments.length; i++ ) {
 383              cb_args += "cb_arguments[" + i + "], ";
 384          }
 385          if( cb_args.length )
 386          { // remove last ", ":
 387              cb_args = cb_args.substring( 0, cb_args.length - 2 );
 388          }
 389  
 390          // eval() for each registered callback:
 391          for( var i = 0; i < this.eventHandlers[event].length; i++ )
 392          {
 393              var f = this.eventHandlers[event][i];
 394              r = eval( "f("+cb_args+");" ) || r;
 395          }
 396  
 397          return r;
 398      }
 399  };
