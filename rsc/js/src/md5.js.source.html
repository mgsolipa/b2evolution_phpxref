  63  function rstr2any_md5(input, encoding)
  64  {
  65    var divisor = encoding.length;
  66    var i, j, q, x, quotient;
  67  
  68    /* Convert to an array of 16-bit big-endian values, forming the dividend */
  69    var dividend = Array(Math.ceil(input.length / 2));
  70    for(i = 0; i < dividend.length; i++)
  71    {
  72      dividend[i] = (input.charCodeAt(i * 2) << 8) | input.charCodeAt(i * 2 + 1);
  73    }
  74  
  75    /*
  76     * Repeatedly perform a long division. The binary array forms the dividend,
  77     * the length of the encoding is the divisor. Once computed, the quotient
  78     * forms the dividend for the next step. All remainders are stored for later
  79     * use.
  80     */
  81    var full_length = Math.ceil(input.length * 8 /
  82                                      (Math.log(encoding.length) / Math.log(2)));
  83    var remainders = Array(full_length);
  84    for(j = 0; j < full_length; j++)
  85    {
  86      quotient = Array();
  87      x = 0;
  88      for(i = 0; i < dividend.length; i++)
  89      {
  90        x = (x << 16) + dividend[i];
  91        q = Math.floor(x / divisor);
  92        x -= q * divisor;
  93        if(quotient.length > 0 || q > 0)
  94          quotient[quotient.length] = q;
  95      }
  96      remainders[j] = x;
  97      dividend = quotient;
  98    }
  99  
 100    /* Convert the remainders to the output string */
 101    var output = "";
 102    for(i = remainders.length - 1; i >= 0; i--)
 103      output += encoding.charAt(remainders[i]);
 104  
 105    return output;
 106  }

 110   * Characters >255 have their high-byte silently ignored.

 112  function rstr2binl(input)
 113  {
 114    var output = Array(input.length >> 2);
 115    for(var i = 0; i < output.length; i++)
 116      output[i] = 0;
 117    for(var i = 0; i < input.length * 8; i += 8)
 118      output[i>>5] |= (input.charCodeAt(i / 8) & 0xFF) << (i%32);
 119    return output;
 120  }

 126  {
 127    var output = "";
