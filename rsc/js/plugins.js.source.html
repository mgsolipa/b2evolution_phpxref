   7  /** Init bubbletip for help icon of plugins **/
   8  // Note: This is not just backoffice, this is also used to display a bubbletip over help plugin icon e.g. on the edit post/comment pages.
   9  
  10  var plugin_number = 1;
  11  jQuery( document ).on( 'mouseover', 'a.help_plugin_icon', function()
  12  {
  13      if( jQuery( this ).hasClass( 'bubbleplugin' ) )
  14      { // Bubbletip is already initialized on this help icon
  15          return true;
  16      }
  17  
  18      jQuery( this ).attr( 'title', '' );
  19      jQuery( this ).find( 'span' ).removeAttr( 'title' );
  20  
  21      var tip_text = jQuery( this ).attr( 'rel' );
  22      if( jQuery( '#help_plugin_info_suffix' ).length > 0 )
  23      { // Append additional info
  24          tip_text += jQuery( '#help_plugin_info_suffix' ).html();
  25      }
  26      jQuery( 'body' ).append( '<div id="tip_plugin_' + plugin_number + '" style="display:none;max-width:200px;text-align:left">' + tip_text + '</div>' );
  27  
  28      var direction = 'right';
  29      if( jQuery( 'body' ).width() - jQuery( this ).position().left < 220 )
  30      { // Change position of bubbletip if we have no enough space at the right
  31          direction = 'left';
  32      }
  33  
  34      var tip = jQuery( '#tip_plugin_' + plugin_number );
  35      jQuery( this ).bubbletip( tip, {
  36              showOnInit: true,
  37              deltaDirection: direction,
  38              deltaShift: 0,
  39          } );
  40  
  41      // Add this class to avoid of the repeating of init bubbletip
  42      jQuery( this ).addClass( 'bubbleplugin' );
  43      plugin_number++;
  44  } );
