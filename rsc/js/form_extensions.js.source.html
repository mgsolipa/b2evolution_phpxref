   9   * Takes an object and returns the parent form
  10   *
  11   * @param object the child object
  12   * @return form the parent form of the child object
  13   */
  14  function get_form( object )
  15  {
  16       while( object.tagName != 'FORM' )
  17      {    // loop which goes from the checkbox to the form
  18          // alert( object.nodeName );
  19          if( typeof( object ) != 'undefined' ) // make sure that we have not gone too far
  20          {
  21              object = object.parentNode;
  22          }
  23          else
  24          {
  25              return false;
  26          }
  27      }
  28      // alert( 'ok:'+object.tagName );
  29      return object;
  30  }

  33  /**
  34   * Toggles all checkboxes of the current form
  35   *
  36   * @param form the form
  37   * @param integer force set/unset
  38   */
  39  function check( object, action )
  40  {
  41  
  42      form_obj = get_form( object );
  43  
  44      if( ! form_obj )
  45      {
  46          alert( 'Could not find form' );
  47          return false;
  48      }
  49  
  50      //checks or unchecks all checkboxes in the form
  51      i = 0;
  52      while( i < form_obj.length )
  53      {
  54          if( form_obj.elements[i].type == 'checkbox' )
  55          {
  56              form_obj.elements[i].checked = action;
  57          }
  58          i++;
  59      }
  60  
  61      // Cancel default action:
  62      return false;
  63  }

 102  function cancelClick( e )
 103  {
 104      if( window.event && window.event.returnValue )
 105      {    // Ie
 106          window.event.returnValue = false;
 107      }
 108      if( e && e.preventDefault )
 109      {    // Firefox
 110          e.preventDefault();
 111      }
 112      return false;
 113  }

 116  /**
 117   *    Surround or unsurrond all' surround_check' span restrict to check_val
 118   *    used to surround all check_all checkboxes
 119   */
 120  function surround_check( e, class_name, check_val )
 121  {    // Get the event target element
 122      var el = findTarget(e);
 123      // Get the parent form
 124      el_form = get_form( el );
 125      // Get all form inputs
 126      el_inputs = el_form.getElementsByTagName( 'INPUT' );
 127  
 128      // Loop on all inputs
 129      for( i = 0 ; i < el_inputs.length ; i++ )
 130      {
 131          el_input = el_inputs[i];
 132  
 133          if( el_input.type == 'checkbox' )
 134          {    // The input is a checkbox
 135              if( check_val == null || el_input.checked == check_val )
 136              {    // Change the parent (span) class
 137                  el_input.parentNode.className = class_name;
 138              }
 139          }
 140      }
 141  }

 200   * Clear the form the current object belongs to
 201   *
 202   * @param object the object whose form must be cleared
 203   */
 204  function clear_form( object )
 205  {
 206      object = check( object, false );
 207  
 208      // empties all the input fields of the form
 209      i = 0;
 210      while( i < object.length )
 211      {
 212          if( object.elements[i].type == 'text' )
 213          {
 214              object.elements[i].value = '';
 215          }
 216          i++;
 217      }
 218  
 219      return object;
 220  }

 227  {
 228      all_inputs = document.getElementsByTagName( 'input' );
 229  
 230      if( all_inputs.length )
 231      { // There is at least one input
 232          // Loop on all inputs to find the first input text
 233          for( i = 0 ; i < all_inputs.length ; i++ )
 234          {
 235        if( all_inputs[i].type == 'text'
 236                      && all_inputs[i].disabled != true
 237                      )
 238              {    // We found the first input text, so we focus on it
 239                  try
 240                  {    // Will fail in IE if element is not visible/displayed
 241                      all_inputs[i].focus();
 242                  }
 243                  catch( ex )
 244                  {
 245                  }
 246                  break;
 247              }

 290  /**
 291   * Decorate an input field with a "help value", which gets
 292   * removed onfocus and re-added onblur (if the fields real
 293   * value is still unchanged).
 294   *
 295   * @param string ID of the input field
 296   * @param string "Help value"
 297   */
 298  function input_decorated_help( id, hvalue )
 299  {
 300      var elm = document.getElementById(id);
 301  
 302      var onblur = function() {
 303              if( elm.value == '' || elm.value == hvalue )
 304              {
 305                  elm.style.color = '#666';
 306                  elm.value = hvalue;
 307              }
 308          }
 309  
 310      jQuery( elm ).bind( 'blur', onblur );
 311  
 312      jQuery( elm ).bind( 'focus', function() {
 313              elm.style.color = '';
 314  
 315              if( elm.value == hvalue )
 316                  elm.value = '';
 317          } );
 318  
 319      /* on form's submit: set to empty, if help value */
 320      jQuery( elm.form ).bind( 'submit', function() {
 321              if( elm.value == hvalue )
 322              {
 323                  elm.value = '';
 324              }
 325          } );
 326  
 327      // init:
 328      onblur();
 329  }

 332  /**
 333   * caters for the differences between Internet Explorer and fully DOM-supporting browsers
 334   */
 335  function findTarget( e )
 336  {
 337      var target;
 338  
 339      if( window.event && window.event.srcElement )
 340          target = window.event.srcElement;
 341      else if( e && e.target )
 342          target = e.target;
 343      if( ! target )
 344          return null;
 345  
 346      while( target != document.body && target.nodeName.toLowerCase() != 'a' )
 347          target = target.parentNode;
 348  
 349      if( target.nodeName.toLowerCase() != 'a' )
 350          return null;
 351  
 352      return target;
 353  }
