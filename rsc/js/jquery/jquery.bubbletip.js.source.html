 187                  var animation;
 188  
 189                  if (_isActive) { // the tip is currently showing; do nothing
 190                      return;
 191                  }
 192                  _isActive = true;
 193                  if (_isHiding) { // the tip is currently hiding; interrupt and start showing again
 194                      _wrapper.stop(true, false);
 195                  }
 196  
 197                  if (_options.calculateOnShow) {
 198                      _Calculate();
 199                  }
 200                  if (_options.positionAt.match(/^element|body$/i)) {
 201                      if (_options.deltaDirection.match(/^up|down$/i)) {
 202                          if (!_isHiding) {
 203                              _wrapper.css('top', parseInt(_calc.top + _calc.delta) + 'px');
 204                          }
 205                          animation = { 'top': _calc.top + 'px' };
 206                      } else {
 207                          if (!_isHiding) {
 208                              _wrapper.css('left', parseInt(_calc.left + _calc.delta) + 'px');
 209                          }
 210                          animation = { 'left': _calc.left + 'px' };
 211                      }
 212                  } else {
 213                      if (_options.deltaDirection.match(/^up|down$/i)) {
 214                          if (!_isHiding) {
 215                              _calc.mouseTop = e.pageY + _calc.top;
 216                              _wrapper.css({ 'top': parseInt(_calc.mouseTop + _calc.delta) + 'px', 'left': parseInt(e.pageX - (_wrapper.width() / 2)) + 'px' });
 217                          }
 218                          animation = { 'top': _calc.mouseTop + 'px' };
 219                      } else {
 220                          if (!_isHiding) {
 221                              _calc.mouseLeft = e.pageX + _calc.left;
 222                              _wrapper.css({ 'left': parseInt(_calc.mouseLeft + _calc.delta) + 'px', 'top': parseInt(e.pageY - (_wrapper.height() / 2)) + 'px' });

 260  			function _Calculate() {
 261                  // calculate values
 262                  if (_options.positionAt.match(/^element$/i)) {
 263                      var offset = _options.positionAtElement.offset();
 264                      if (_options.deltaDirection.match(/^up$/i)) {
 265                          _calc.top = offset.top + _options.offsetTop - _wrapper.outerHeight() + _options.deltaShift;
 266                          _calc.left = offset.left + _options.offsetLeft + ((_options.positionAtElement.outerWidth() - _wrapper.outerWidth()) / 2);
 267                          _calc.delta = _options.deltaPosition;
 268                      } else if (_options.deltaDirection.match(/^down$/i)) {
 269                          _calc.top = offset.top + _options.positionAtElement.outerHeight() + _options.offsetTop - _options.deltaShift;
 270                          _calc.left = offset.left + _options.offsetLeft + ((_options.positionAtElement.outerWidth() - _wrapper.outerWidth()) / 2);
 271                          _calc.delta = -_options.deltaPosition;
 272                      } else if (_options.deltaDirection.match(/^left$/i)) {
 273                          _calc.top = offset.top + _options.offsetTop + ((_options.positionAtElement.outerHeight() - _wrapper.outerHeight()) / 2);
 274                          _calc.left = offset.left + _options.offsetLeft - _wrapper.outerWidth() + _options.deltaShift;
 275                          _calc.delta = _options.deltaPosition;
 276                      } else if (_options.deltaDirection.match(/^right$/i)) {
 277                          _calc.top = offset.top + _options.offsetTop + ((_options.positionAtElement.outerHeight() - _wrapper.outerHeight()) / 2);
 278                          _calc.left = offset.left + _options.positionAtElement.outerWidth() + _options.offsetLeft - _options.deltaShift;
 279                          _calc.delta = -_options.deltaPosition;
 280                      }
 281                  } else if (_options.positionAt.match(/^body$/i)) {
 282                      if (_options.deltaDirection.match(/^up|left$/i)) {
 283                          _calc.top = _options.offsetTop;
 284                          _calc.left = _options.offsetLeft;
 285                          // up or left
 286                          _calc.delta = _options.deltaPosition;
 287                      } else {
 288                          if (_options.deltaDirection.match(/^down$/i)) {
 289                              _calc.top = parseInt(_options.offsetTop + _wrapper.outerHeight());
 290                              _calc.left = _options.offsetLeft;
 291                          } else {
 292                              _calc.top = _options.offsetTop;
 293                              _calc.left = parseInt(_options.offsetLeft + _wrapper.outerWidth());
 294                          }
 295                          // down or right
 296                          _calc.delta = -_options.deltaPosition;
 297                      }
 298                  } else if (_options.positionAt.match(/^mouse$/i)) {
 299                      if (_options.deltaDirection.match(/^up|left$/i)) {
 300                          if (_options.deltaDirection.match(/^up$/i)) {
 301                              _calc.top = -(_options.offsetTop + _wrapper.outerHeight());
 302                              _calc.left = _options.offsetLeft;
 303                          } else if (_options.deltaDirection.match(/^left$/i)) {
 304                              _calc.top = _options.offsetTop;
 305                              _calc.left = -(_options.offsetLeft + _wrapper.outerWidth());
 306                          }
 307                          // up or left
 308                          _calc.delta = _options.deltaPosition;
 309                      } else {
 310                          _calc.top = _options.offsetTop;
 311                          _calc.left = _options.offsetLeft;
 312                          // down or right
 313                          _calc.delta = -_options.deltaPosition;
 314                      }
 315                  }
 316                  // handle the wrapper (element|body) positioning
 317                  if (_options.positionAt.match(/^element|body$/i)) {
 318                      _wrapper.css({
 319                          'position': 'absolute',
 320                          'top': _calc.top + 'px',
 321                          'left': _calc.left + 'px'

 331          removeBubbletip: function(tips) {
 332              var tipsActive;
 333              var tipsToRemove = new Array();
 334              var tipsActiveAdjusted = new Array();
 335              var arr, i, ix;
 336              var elem;
 337  
 338              tipsActive = $.makeArray($(this).data('bubbletip_tips'));
 339  
 340              // convert the parameter array of tip id's or elements to id's
 341              arr = $.makeArray(tips);
 342              for (i = 0; i < arr.length; i++) {
 343                  tipsToRemove.push($(arr[i]).get(0).id);
 344              }
 345  
 346              for (i = 0; i < tipsActive.length; i++) {
 347                  ix = null;
 348                  if ((tipsToRemove.length == 0) || ((ix = $.inArray(tipsActive[i][0], tipsToRemove)) >= 0)) {
 349                      // remove all tips if there are none specified
 350                      // otherwise, remove only specified tips
 351  
 352                      // find the surrounding table.bubbletip
 353                      elem = $('#' + tipsActive[i][0]).get(0).parentNode;
 354                      while (elem.tagName.toLowerCase() != 'table') {
 355                          elem = elem.parentNode;
 356                      }
 357                      // attach the tip element to body and hide
 358                      $('#' + tipsActive[i][0]).appendTo( _options.wrapperContainer ).hide();
 359                      // remove the surrounding table.bubbletip
 360                      $(elem).remove();
 361  
 362                      // unbind show/hide events
 363                      $(this).unbind('.bubbletip' + tipsActive[i][1]);
 364  
 365                      // unbind window.resize event
 366                      $(window).unbind('.bubbletip' + tipsActive[i][1]);
 367                  } else {
 368                      // tip is not being removed, so add it to the adjusted array
 369                      tipsActiveAdjusted.push(tipsActive[i]);
 370                  }
 371              }
 372              $(this).data('bubbletip_tips', tipsActiveAdjusted);
 373  
 374              return this;
 375          }
