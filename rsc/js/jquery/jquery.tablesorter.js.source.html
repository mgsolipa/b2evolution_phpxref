 169                  if (!node) { return ""; }
 170                  var te, c = table.config,
 171                      t = c.textExtraction || '',
 172                      text = "";
 173                  if (t === "basic") {

 176                  } else {
 177                      if (typeof(t) === "function") {
 178                          text = t(node, table, cellIndex);
 179                      } else if (typeof (te = ts.getColumnData( table, t, cellIndex )) === 'function') {

 189  			function detectParserForColumn(table, rows, rowIndex, cellIndex) {
 190                  var cur,
 191                  i = ts.parsers.length,
 192                  node = false,
 193                  nodeValue = '',
 194                  keepLooking = true;
 195                  while (nodeValue === '' && keepLooking) {
 196                      rowIndex++;
 197                      if (rows[rowIndex]) {
 198                          node = rows[rowIndex].cells[cellIndex];
 199                          nodeValue = getElementText(table, node, cellIndex);
 200                          if (table.config.debug) {
 201                              log('Checking if value was empty on row ' + rowIndex + ', column: ' + cellIndex + ': "' + nodeValue + '"');
 202                          }
 203                      } else {
 204                          keepLooking = false;
 205                      }
 206                  }
 207                  while (--i >= 0) {
 208                      cur = ts.parsers[i];
 209                      // ignore the default text parser because it will always be true
 210                      if (cur && cur.id !== 'text' && cur.is && cur.is(nodeValue, table, node)) {
 211                          return cur;
 212                      }
 213                  }
 214                  // nothing found, return the generic parser (text)
 215                  return ts.getParserById('text');
 216              }

 271                  var cc, t, v, i, j, k, $row, rows, cols, cacheTime,
 272                      totalRows, rowData, colMax,
 273                      c = table.config,
 274                      $tb = c.$table.children('tbody'),
 275                  parsers = c.parsers;
 276                  c.cache = {};
 277                  // if no parsers found, return - it's an empty table.
 278                  if (!parsers) {
 279                      return c.debug ? log('Warning: *Empty table!* Not building a cache') : '';
 280                  }
 281                  if (c.debug) {
 282                      cacheTime = new Date();
 283                  }
 284                  // processing icon
 285                  if (c.showProcessing) {
 286                      ts.isProcessing(table, true);
 287                  }
 288                  for (k = 0; k < $tb.length; k++) {
 289                      colMax = []; // column max value per tbody
 290                      cc = c.cache[k] = {
 291                          normalized: [] // array of normalized row data; last entry contains "rowData" above
 292                          // colMax: #   // added at the end
 293                      };
 294  
 295                      // ignore tbodies with class name from c.cssInfoBlock
 296                      if (!$tb.eq(k).hasClass(c.cssInfoBlock)) {
 297                          totalRows = ($tb[k] && $tb[k].rows.length) || 0;
 298                          for (i = 0; i < totalRows; ++i) {
 299                              rowData = {
 300                                  // order: original row order #
 301                                  // $row : jQuery Object[]
 302                                  child: [] // child row text (filter widget)
 303                              };
 304                              /** Add the table data to main data array */
 305                              $row = $($tb[k].rows[i]);
 306                              rows = [ new Array(c.columns) ];
 307                              cols = [];
 308                              // if this is a child row, add it to the last row's children and continue to the next row
 309                              // ignore child row class, if it is the first row
 310                              if ($row.hasClass(c.cssChildRow) && i !== 0) {
 311                                  t = cc.normalized.length - 1;
 312                                  cc.normalized[t][c.columns].$row = cc.normalized[t][c.columns].$row.add($row);
 313                                  // add "hasChild" class name to parent row
 314                                  if (!$row.prev().hasClass(c.cssChildRow)) {
 315                                      $row.prev().addClass(ts.css.cssHasChild);
 316                                  }
 317                                  // save child row content (un-parsed!)
 318                                  rowData.child[t] = $.trim( $row[0].textContent || $row[0].innerText || $row.text() || "" );
 319                                  // go to the next for loop
 320                                  continue;
 321                              }
 322                              rowData.$row = $row;
 323                              rowData.order = i; // add original row position to rowCache
 324                              for (j = 0; j < c.columns; ++j) {
 325                                  if (typeof parsers[j] === 'undefined') {
 326                                      if (c.debug) {
 327                                          log('No parser found for cell:', $row[0].cells[j], 'does it have a header?');
 328                                      }
 329                                      continue;
 330                                  }
 331                                  t = getElementText(table, $row[0].cells[j], j);
 332                                  // allow parsing if the string is empty, previously parsing would change it to zero,
 333                                  // in case the parser needs to extract data from the table cell attributes
 334                                  v = parsers[j].id === 'no-parser' ? '' : parsers[j].format(t, table, $row[0].cells[j], j);
 335                                  cols.push(v);
 336                                  if ((parsers[j].type || '').toLowerCase() === "numeric") {
 337                                      // determine column max value (ignore sign)
 338                                      colMax[j] = Math.max(Math.abs(v) || 0, colMax[j] || 0);
 339                                  }
 340                              }
 341                              // ensure rowData is always in the same location (after the last column)
 342                              cols[c.columns] = rowData;
 343                              cc.normalized.push(cols);
 344                          }
 345                          cc.colMax = colMax;
 346                      }
 347                  }
 348                  if (c.showProcessing) {
 349                      ts.isProcessing(table); // remove processing icon
 350                  }
 351                  if (c.debug) {
 352                      benchmark("Building cache for " + totalRows + " rows", cacheTime);
 353                  }
 354              }

 559                      // make sure header exists
 560                      o = c.$headers.filter('[data-column="' + col + '"]:last')[0];
 561                      if (o) { // prevents error if sorton array is wrong
 562                          // o.count = o.count + 1;
 563                          t = ('' + v[1]).match(/^(1|d|s|o|n)/);
 564                          t = t ? t[0] : '';
 565                          // 0/(a)sc (default), 1/(d)esc, (s)ame, (o)pposite, (n)ext
 566                          switch(t) {
 567                              case '1': case 'd': // descending
 568                                  t = 1;
 569                                  break;
 570                              case 's': // same direction (as primary column)
 571                                  // if primary sort is set to "s", make it ascending
 572                                  t = primary || 0;
 573                                  break;
 574                              case 'o':
 575                                  s = o.order[(primary || 0) % (c.sortReset ? 3 : 2)];
 576                                  // opposite of primary column; but resets if primary resets
 577                                  t = s === 0 ? 1 : s === 1 ? 0 : 2;
 578                                  break;
 579                              case 'n':
 580                                  o.count = o.count + 1;
 581                                  t = o.order[(o.count) % (c.sortReset ? 3 : 2)];
 582                                  break;
 583                              default: // ascending
 584                                  t = 0;
 585                                  break;
 586                          }
 587                          primary = i === 0 ? t : primary;
 588                          s = [ col, parseInt(t, 10) || 0 ];
 589                          c.sortList.push(s);
 590                          t = $.inArray(s[1], o.order); // fixes issue #167
 591                          o.count = t >= 0 ? t : s[1] % (c.sortReset ? 3 : 2);
 592                      }
 593                  });
 594              }

 705  			function multisort(table) { /*jshint loopfunc:true */
 706                  var i, k, num, col, sortTime, colMax,
 707                      cache, order, sort, x, y,
 708                      dir = 0,
 709                      c = table.config,
 710                      cts = c.textSorter || '',
 711                      sortList = c.sortList,
 712                      l = sortList.length,
 713                      bl = table.tBodies.length;
 714                  if (c.serverSideSorting || isEmptyObject(c.cache)) { // empty table - fixes #206/#346
 715                      return;
 716                  }
 717                  if (c.debug) { sortTime = new Date(); }
 718                  for (k = 0; k < bl; k++) {
 719                      colMax = c.cache[k].colMax;
 720                      cache = c.cache[k].normalized;
 721  
 722                      cache.sort(function(a, b) {
 723                          // cache is undefined here in IE, so don't use it!
 724                          for (i = 0; i < l; i++) {
 725                              col = sortList[i][0];
 726                              order = sortList[i][1];
 727                              // sort direction, true = asc, false = desc
 728                              dir = order === 0;
 729  
 730                              if (c.sortStable && a[col] === b[col] && l === 1) {
 731                                  return a[c.columns].order - b[c.columns].order;
 732                              }
 733  
 734                              // fallback to natural sort since it is more robust
 735                              num = /n/i.test(getCachedSortType(c.parsers, col));
 736                              if (num && c.strings[col]) {
 737                                  // sort strings in numerical columns
 738                                  if (typeof (c.string[c.strings[col]]) === 'boolean') {
 739                                      num = (dir ? 1 : -1) * (c.string[c.strings[col]] ? -1 : 1);
 740                                  } else {
 741                                      num = (c.strings[col]) ? c.string[c.strings[col]] || 0 : 0;
 742                                  }
 743                                  // fall back to built-in numeric sort
 744                                  // var sort = $.tablesorter["sort" + s](table, a[c], b[c], c, colMax[c], dir);
 745                                  sort = c.numberSorter ? c.numberSorter(a[col], b[col], dir, colMax[col], table) :
 746                                      ts[ 'sortNumeric' + (dir ? 'Asc' : 'Desc') ](a[col], b[col], num, colMax[col], col, table);
 747                              } else {
 748                                  // set a & b depending on sort direction
 749                                  x = dir ? a : b;
 750                                  y = dir ? b : a;
 751                                  // text sort function
 752                                  if (typeof(cts) === 'function') {
 753                                      // custom OVERALL text sorter
 754                                      sort = cts(x[col], y[col], dir, col, table);
 755                                  } else if (typeof(cts) === 'object' && cts.hasOwnProperty(col)) {
 756                                      // custom text sorter for a SPECIFIC COLUMN
 757                                      sort = cts[col](x[col], y[col], dir, col, table);
 758                                  } else {
 759                                      // fall back to natural sort
 760                                      sort = ts[ 'sortNatural' + (dir ? 'Asc' : 'Desc') ](a[col], b[col], col, table, c);
 761                                  }
 762                              }
 763                              if (sort) { return sort; }
 764                          }
 765                          return a[c.columns].order - b[c.columns].order;
 766                      });
 767                  }
 768                  if (c.debug) { benchmark("Sorting on " + sortList.toString() + " and dir " + order + " time", sortTime); }
 769              }

1314              // Natural sort - https://github.com/overset/javascript-natural-sort (date sorting removed)
1315              // this function will only accept strings, or you'll see "TypeError: undefined is not a function"
1316              // I could add a = a.toString(); b = b.toString(); but it'll slow down the sort overall
1317              ts.sortNatural = function(a, b) {
1318                  if (a === b) { return 0; }
1319                  var xN, xD, yN, yD, xF, yF, i, mx,
1320                      r = ts.regex;
1321                  // first try and sort Hex codes
1322                  if (r.hex.test(b)) {
1323                      xD = parseInt(a.match(r.hex), 16);
1324                      yD = parseInt(b.match(r.hex), 16);
1325                      if ( xD < yD ) { return -1; }
1326                      if ( xD > yD ) { return 1; }
1327                  }
1328                  // chunk/tokenize
1329                  xN = a.replace(r.chunk, '\\0$1\\0').replace(r.chunks, '').split('\\0');
1330                  yN = b.replace(r.chunk, '\\0$1\\0').replace(r.chunks, '').split('\\0');
1331                  mx = Math.max(xN.length, yN.length);
1332                  // natural sorting through split numeric strings and default strings
1333                  for (i = 0; i < mx; i++) {
1334                      // find floats not starting with '0', string or 0 if not defined
1335                      xF = isNaN(xN[i]) ? xN[i] || 0 : parseFloat(xN[i]) || 0;
1336                      yF = isNaN(yN[i]) ? yN[i] || 0 : parseFloat(yN[i]) || 0;
1337                      // handle numeric vs string comparison - number < string - (Kyle Adams)
1338                      if (isNaN(xF) !== isNaN(yF)) { return (isNaN(xF)) ? 1 : -1; }
1339                      // rely on string comparison if different types - i.e. '02' < 2 != '02' < '2'
1340                      if (typeof xF !== typeof yF) {
1341                          xF += '';
1342                          yF += '';
1343                      }
1344                      if (xF < yF) { return -1; }
1345                      if (xF > yF) { return 1; }
1346                  }
1347                  return 0;
1348              };
