 147  	function bindAll(context, opt, target) {
 148          var id = context.attr('id'),
 149              score = $('input#' + id + '-score'),
 150              stars = context.children('span.' + id);
 151  
 152          context.mouseleave(function() {
 153              var current_score = $('input#' + this.id + '-score');
 154              initialize(context, current_score.val(), opt);
 155              clearTarget(target, current_score, opt);
 156          });
 157  
 158          stars.bind(((opt.half) ? 'mousemove' : 'mouseover'), function(e) {
 159              fillStar(id, $(this).attr('alt'), opt);
 160              if (opt.half) {
 161                  var percent = parseFloat(((e.pageX - $(this).offset().left) / opt.size).toFixed(1));
 162                  percent = (percent >= 0 && percent < 0.5) ? 0.5 : 1;
 163  
 164                  context.data('score', parseFloat($(this).attr('alt')) + percent - 1);
 165  
 166                  splitStar(context, context.data('score'), opt);
 167              } else {
 168                  fillStar(id, $(this).attr('alt'), opt);
 169              }
 170  
 171              setTarget(target, $(this).attr('alt'), opt);
 172          }).click(function(evt) {
 173              score.val(opt.half ? context.data('score') : $(this).attr('alt'));
 174  
 175              if (opt.click) {
 176                  opt.click.apply(context, [score.val(), evt]);

 181  	function clearTarget(target, score, opt) {
 182          if (target !== null) {
 183              var value = '';
 184  
 185              if (opt.targetKeep) {
 186                  value = score.val();
 187  
 188                  if (opt.targetType == 'hint') {
 189                      if (score.val() == '' && opt.cancel) {
 190                          value = opt.cancelHint;
 191                      } else {
 192                          value = opt.hintList[Math.ceil(score.val()) - 1];
 193                      }
 194                  }
 195              }
 196  
 197              if (isField(target)) {
 198                  target.val(value);
 199              } else {
 200                  target.html(value);
 201              }
 202          }

 205  	function fillStar(id, score, opt) {
 206          var qtyStar    = $('span.' + id).length,
 207              item    = 0,
 208              range    = 0,
 209              star,
 210              starOn = opt.starOn,
 211              starOff = opt.starOff;
 212  
 213          for (var i = 1; i <= qtyStar; i++) {
 214              star = $('span#' + id + '-' + i);
 215              if (i <= score) {
 216                  if (opt.iconRange && opt.iconRange.length > item) {
 217                      starOn = opt.iconRange[item][0];
 218                      range = opt.iconRange[item][1];
 219  
 220                      if (i <= range) {
 221                          star.removeClass(starOff).addClass(starOn);
 222                      }
 223  
 224                      if (i == range) {
 225                          item++;
 226                      }
 227                  } else {
 228                      star.removeClass(starOff).addClass(starOn);
 229                  }
 230              } else {
 231                  star.removeClass(starOn).addClass(starOff);
 232              }
 233          }

 236  	function fixHint(context, score, opt) {
 237          var hint = '';
 238  
 239          if (score != 0) {
 240              score = parseInt(score);
 241              hint = (score > 0 && opt.number <= opt.hintList.length && opt.hintList[score - 1] !== null) ? opt.hintList[score - 1] : score;
 242          } else {
 243              hint = opt.noRatedMsg;
 244          }
 245  
 246          context.attr('title', hint).children('span').attr('title', hint);

 253  	function initialize(context, score, opt) {
 254          var id = context.attr('id');
 255  
 256          if (isNaN(parseInt(score))) {
 257              context.children('span.' + id).removeClass(opt.starOn).addClass(opt.starOff);
 258              $('input#' + id + '-score').removeAttr('value');
 259              return;
 260          }
 261  
 262          if (score < 0) {
 263              score = 0;
 264          } else if (score > opt.number) {
 265              score = opt.number;
 266          }
 267  
 268          fillStar(id, score, opt);
 269  
 270          if (score > 0) {
 271              $('input#' + id + '-score').val(score);
 272  
 273              if (opt.half) {
 274                  splitStar(context, score, opt);
 275              }
 276          }
 277  
 278          if (opt.readOnly || context.css('cursor') == 'default') {
 279              fixHint(context, score, opt);
 280          }

 303  	function splitStar(context, score, opt) {
 304          var id        = context.attr('id'),
 305              rounded    = Math.ceil(score),
 306              diff    = (rounded - score).toFixed(1);
 307  
 308          if (diff > 0.25 && diff <= 0.75) {
 309              rounded = rounded - 0.5;
 310              $('span#' + id + '-' + Math.ceil(rounded)).removeClass(starOff).removeClass(starOn).addClass(starHalf);
 311          } else if (diff > 0.75) {
 312              rounded--;
 313          } else {
 314              $('span#' + id + '-' + rounded).removeClass(starOff).addClass(starOn);
 315          }

 328      $.fn.raty.click = function(score, idOrClass) {
 329          var context = getContext(score, idOrClass, 'click'),
 330              options = $(idOrClass).data('options');
 331  
 332          if (idOrClass.indexOf('.') >= 0) {
 333              return;
 334          }
 335  
 336          initialize(context, score, options);
 337  
 338          if (options.click) {
 339              options.click.apply(context, [score]);
 340          } else {
 341              debug('You must add the "click: function(score, evt) { }" callback.');
 342          }
 343  
 344          return context;
 345      };

 348          var context    = getContext(boo, idOrClass, 'readOnly'),

 350              cancel    = context.children('span.button-cancel');
 351

 353              return;
 354          }
 355

 384  	function getContext(value, idOrClass, name) {
 385          var context = undefined;
 386  
 387          if (idOrClass == undefined) {
 388              debug('Specify an ID or class to be the target of the action.');
 389              return;
 390          }
 391  
 392          if (idOrClass) {
 393              if (idOrClass.indexOf('.') >= 0) {
 394                  var idEach;
 395  
 396                  return $(idOrClass).each(function() {
 397                      idEach = '#' + $(this).attr('id');
 398  
 399                      if (name == 'start') {
 400                          $.fn.raty.start(value, idEach);
 401                      } else if (name == 'click') {
 402                          $.fn.raty.click(value, idEach);
 403                      } else if (name == 'readOnly') {
 404                          $.fn.raty.readOnly(value, idEach);
 405                      }
 406                  });
 407              }
 408  
 409              context = $(idOrClass);
 410  
 411              if (!context.length) {
 412                  debug('"' + idOrClass + '" is a invalid identifier for the public funtion $.fn.raty.' + name + '().');
 413                  return;
 414              }
 415          }
 416  
 417          return context;
 418      };
