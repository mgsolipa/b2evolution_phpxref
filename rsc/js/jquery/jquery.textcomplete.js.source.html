  22    var lock = function (func) {
  23      var free, locked;
  24      free = function () { locked = false; };
  25      return function () {
  26        var args;
  27        if (locked) return;
  28        locked = true;
  29        args = toArray(arguments);
  30        args.unshift(free);
  31        func.apply(this, args);
  32      };
  33    };

  88    var memoize = function (func) {
  89      var memo = {};
  90      return function (term, callback) {
  91        if (memo[term]) {
  92          callback(memo[term]);
  93        } else {
  94          func.call(this, term, function (data) {
  95            memo[term] = (memo[term] || []).concat(data);
  96            callback.apply(null, arguments);
  97          });
  98        }
  99      };

 169        /**
 170         * Show autocomplete list next to the caret.

 173          if (this.clearAtNext) {

 225          var pre, post, newSubStr;

 227          post = this.el.value.substring(this.el.selectionEnd);
 228

 232            newSubStr = newSubStr[0];

 234          pre = pre.replace(this.strategy.match, newSubStr);

 236          this.el.focus();

 244         * Returns caret's relative coordinates from textarea's left top corner.

 246        getCaretPosition: function () {
 247          // Browser native API does not provide the way to know the position of
 248          // caret in pixels, so that here we use a kind of hack to accomplish
 249          // the aim. First of all it puts a div element and completely copies
 250          // the textarea's style to the element, then it inserts the text and a
 251          // span element into the textarea.
 252          // Consequently, the span element's position is the thing what we want.
 253  
 254          if (this.el.selectionEnd === 0) return;
 255          var properties, css, $div, $span, position;
 256  
 257          properties = ['border-width', 'font-family', 'font-size', 'font-style',
 258            'font-variant', 'font-weight', 'height', 'letter-spacing',
 259            'word-spacing', 'line-height', 'text-decoration', 'width',
 260            'padding-top', 'padding-right', 'padding-bottom', 'padding-left',
 261            'margin-top', 'margin-right', 'margin-bottom', 'margin-left'
 262          ];
 263          css = $.extend({
 264            position: 'absolute',
 265            overflow: 'auto',
 266            'white-space': 'pre-wrap',
 267            top: 0,
 268            left: -9999
 269          }, getStyles(this.$el, properties));
 270  
 271          $div = $('<div></div>').css(css).text(this.getTextFromHeadToCaret());
 272          $span = $('<span></span>').text('&nbsp;').appendTo($div);
 273          this.$el.before($div);
 274          position = $span.position();
 275          position.top += $span.height() - this.$el.scrollTop();
 276          $div.remove();
 277          return position;
 278        },

 280        getTextFromHeadToCaret: function () {
 281          var text, selectionEnd, range;
 282          selectionEnd = this.el.selectionEnd;
 283          if (typeof selectionEnd === 'number') {
 284            text = this.el.value.substring(0, selectionEnd);
 285          } else if (document.selection) {
 286            range = this.el.createTextRange();
 287            range.moveStart('character', 0);
 288            range.moveEnd('textedit');
 289            text = range.text;
 290          }
 291          return text;

 297        extractSearchQuery: function (text) {
 298          // If a search query found, it returns used strategy and the query
 299          // term. If the caret is currently in a code block or search query does
 300          // not found, it returns an empty array.
 301  
 302          var i, l, strategy, match;
 303          for (i = 0, l = this.strategies.length; i < l; i++) {
 304            strategy = this.strategies[i];
 305            match = text.match(strategy.match);
 306            if (match) { return [strategy, match[strategy.index]]; }
 307          }
 308          return [];
 309        },

 342      $.extend(ListView.prototype, {
 343        shown: false,
 344  
 345        render: function (data) {
 346          var html, i, l, index, val;
 347  
 348          html = '';
 349          for (i = 0, l = data.length; i < l; i++) {
 350            val = data[i];
 351            if (include(this.data, val)) continue;
 352            index = this.data.length;
 353            this.data.push(val);
 354            html += '<li class="textcomplete-item" data-index="' + index + '"><a>';
 355            html +=   this.strategy.template(val);
 356            html += '</a></li>';
 357            if (this.data.length === this.strategy.maxCount) break;
 358          }
 359          this.$el.append(html)
 360          if (!this.data.length) {
 361            this.deactivate();
 362          } else {
 363            this.activateIndexedItem();
 364          }
 365        },
 366  
 367        clear: function () {
 368          this.data = [];
 369          this.$el.html('');
 370          this.index = 0;
 371          return this;
 372        },
 373  
 374        activateIndexedItem: function () {
 375          var $item;
 376          this.$el.find('.active').removeClass('active');
 377          this.getActiveItem().addClass('active');
 378        },
 379  
 380        getActiveItem: function () {
 381          return $(this.$el.children().get(this.index));
 382        },
 383  
 384        activate: function () {
 385          if (!this.shown) {
 386            this.$el.show();
 387            this.shown = true;
 388          }
 389          return this;
 390        },
 391  
 392        deactivate: function () {
 393          if (this.shown) {
 394            this.$el.hide();
 395            this.shown = false;
 396            this.data = this.index = null;
 397          }
 398          return this;
 399        },
 400  
 401        setPosition: function (position) {
 402          this.$el.css(position);
 403          return this;
 404        },
 405  
 406        select: function (index) {
 407          this.completer.onSelect(this.data[index]);
 408          this.deactivate();
 409        },
 410  
 411        onKeydown: function (e) {
 412          var $item;
 413          if (!this.shown) return;
 414          if (e.keyCode === 27) {         // ESC
 415              this.deactivate();
 416          } else if (e.keyCode === 38) {         // UP
 417            e.preventDefault();
 418            if (this.index === 0) {
 419              this.index = this.data.length-1;
 420            } else {
 421              this.index -= 1;
 422            }
 423            this.activateIndexedItem();
 424          } else if (e.keyCode === 40) {  // DOWN
 425            e.preventDefault();
 426            if (this.index === this.data.length - 1) {
 427              this.index = 0;
 428            } else {
 429              this.index += 1;
 430            }
 431            this.activateIndexedItem();
 432          } else if (e.keyCode === 13 || e.keyCode === 9) {  // ENTER or TAB
 433            e.preventDefault();
 434            this.select(parseInt(this.getActiveItem().data('index')));
 435          }
 436        },
 437  
 438        onClick: function (e) {
 439          var $e = $(e.target);
 440          e.originalEvent.keepTextCompleteDropdown = true;
 441          if (!$e.hasClass('textcomplete-item')) {
 442            $e = $e.parents('li.textcomplete-item');
 443          }
 444          this.select(parseInt($e.data('index')));
 445        }
 446      });

 448      return ListView;

 451    $.fn.textcomplete = function (strategies) {
 452      var i, l, strategy;
 453      for (i = 0, l = strategies.length; i < l; i++) {
 454        strategy = strategies[i];
 455        if (!strategy.template) {
 456          strategy.template = identity;
 457        }
 458        if (strategy.index == null) {
 459          strategy.index = 2;
 460        }
 461        if (strategy.cache) {
 462          strategy.search = memoize(strategy.search);
 463        }
 464        strategy.maxCount || (strategy.maxCount = 10);
 465      }
 466      new Completer(this, strategies);
 467  
 468      return this;
 469    };
