 501      // Add a token to the token list based on user input
 502      function add_token (item) {
 503          var callback = settings.onAdd;
 504  
 505          // See if the token already exists and select it if we don't want duplicates
 506          if(token_count > 0 && settings.preventDuplicates) {
 507              var found_existing_token = null;
 508              token_list.children().each(function () {
 509                  var existing_token = $(this);
 510                  var existing_data = $.data(existing_token.get(0), "tokeninput");
 511                  if(existing_data && item && existing_data.id === item.id) {
 512                      found_existing_token = existing_token;
 513                      return false;
 514                  }
 515              });
 516  
 517              if(found_existing_token) {
 518                  select_token(found_existing_token);
 519                  input_token.insertAfter(found_existing_token);
 520                  input_box.focus();
 521                  return;
 522              }
 523          }
 524  
 525          // Insert the new tokens
 526          if(settings.tokenLimit == null || token_count < settings.tokenLimit) {
 527              insert_token(item);
 528              checkTokenLimit();
 529          }
 530  
 531          // Clear input box
 532          input_box.val("");
 533          $(".token-input-list-facebook").removeClass("warning");
 534  
 535          // Don't show the help dropdown, they've got the idea
 536          hide_dropdown();
 537  
 538          // Execute the onAdd callback if defined
 539          if($.isFunction(callback)) {
 540              callback.call(hidden_input,item);
 541          }
 542      }

 592      function delete_token (token) {
 593          // Remove the id from the saved list
 594          var token_data = $.data(token.get(0), "tokeninput");
 595          var callback = settings.onDelete;
 596  
 597          var index = token.prevAll().length;
 598          if(index > selected_token_index) index--;
 599  
 600          // Delete the token
 601          token.remove();
 602          selected_token = null;
 603  
 604          // Show the input box and give it focus again
 605          input_box.focus();
 606  
 607          // Remove this token from the saved list
 608          saved_tokens = saved_tokens.slice(0,index).concat(saved_tokens.slice(index+1));
 609          if(index < selected_token_index) selected_token_index--;
 610  
 611          // Update the hidden input
 612          update_hidden_input(saved_tokens, hidden_input);
 613  
 614          token_count -= 1;
 615  
 616          if(settings.tokenLimit !== null) {
 617              input_box
 618                  .show()
 619                  .val("")
 620                  .focus();
 621          }
 622  
 623          // Execute the onDelete callback if defined
 624          if($.isFunction(callback)) {
 625              callback.call(hidden_input,token_data);
 626          }
 627      }

 771      // Do the actual search
 772      function run_search(query) {
 773          var cache_key = query + computeURL();
 774          var cached_results = cache.get(cache_key);
 775          if(cached_results) {
 776              populate_dropdown(query, cached_results);
 777          } else {
 778              // Are we doing an ajax search or local data search?
 779              if(settings.url) {
 780                  var url = computeURL();
 781                  // Extract exisiting get params
 782                  var ajax_params = {};
 783                  ajax_params.data = {};
 784                  if(url.indexOf("?") > -1) {
 785                      var parts = url.split("?");
 786                      ajax_params.url = parts[0];
 787  
 788                      var param_array = parts[1].split("&");
 789                      $.each(param_array, function (index, value) {
 790                          var kv = value.split("=");
 791                          ajax_params.data[kv[0]] = kv[1];
 792                      });
 793                  } else {
 794                      ajax_params.url = url;
 795                  }
 796  
 797                  // Prepare the request
 798                  ajax_params.data[settings.queryParam] = query;
 799                  ajax_params.type = settings.method;
 800                  ajax_params.dataType = settings.contentType;
 801                  if(settings.crossDomain) {
 802                      ajax_params.dataType = "jsonp";
 803                  }
 804  
 805                  // Attach the success callback
 806                  ajax_params.success = function(results) {
 807                    if($.isFunction(settings.onResult)) {
 808                        results = settings.onResult.call(hidden_input, results);
 809                    }
 810                    cache.add(cache_key, settings.jsonContainer ? results[settings.jsonContainer] : results);
 811  
 812                    // only populate the dropdown if the results are associated with the active search query
 813                    if(input_box.val() === query) {
 814                        populate_dropdown(query, settings.jsonContainer ? results[settings.jsonContainer] : results);
 815                    }
 816                  };
 817  
 818                  // Make the request
 819                  $.ajax(ajax_params);
 820              } else if(settings.local_data) {
 821                  // Do the search through local data
 822                  var results = $.grep(settings.local_data, function (row) {
 823                      return row[settings.propertyToSearch].indexOf(query) > -1;
 824                  });
 825  
 826                  if($.isFunction(settings.onResult)) {
 827                      results = settings.onResult.call(hidden_input, results);
 828                  }
 829                  cache.add(cache_key, results);
 830                  populate_dropdown(query, results);
 831              }
 832          }
 833      }

 837          var url = settings.url;
 838          if(typeof settings.url == 'function') {
 839              url = settings.url.call();
