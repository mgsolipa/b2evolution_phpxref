  54      if (options.scaleColor && options.scaleLength) {
  55          radius -= options.scaleLength + 2; // 2 is the distance between scale and bar
  56      }

  69      var drawCircle = function(color, lineWidth, percent) {
  70          percent = Math.min(Math.max(0, percent || 1), 1);
  71  
  72          ctx.beginPath();

  78          ctx.stroke();
  79      };

  86          var length;
  87          var i = 24;

  89          ctx.lineWidth = 1

 102              ctx.rotate(Math.PI / 12);
 103          }

 153              }
 154          } else {
 155              this.clear();
 156          }

 163              color = options.barColor(percent);
 164          } else {
 165              color = options.barColor;

 198  var EasyPieChart = function(el, opts) {
 199      var defaultOptions = {
 200          barColor: '#ef1e25',
 201          trackColor: '#f9f9f9',
 202          scaleColor: '#dfe0e0',
 203          scaleLength: 5,
 204          lineCap: 'round',
 205          lineWidth: 3,
 206          size: 110,
 207          rotate: 0,
 208          animate: 1000,
 209          easing: function (x, t, b, c, d) { // more can be found here: http://gsgd.co.uk/sandbox/jquery/easing/
 210              t = t / (d/2);
 211              if (t < 1) {
 212                  return c / 2 * t * t + b;
 213              }
 214              return -c/2 * ((--t)*(t-2) - 1) + b;
 215          },
 216          onStart: function(from, to) {
 217              return;
 218          },
 219          onStep: function(from, to, currentValue) {
 220              return;
 221          },
 222          onStop: function(from, to) {
 223              return;
 224          }
 225      };
 226  
 227      // detect present renderer
 228      if (typeof(CanvasRenderer) !== 'undefined') {
 229          defaultOptions.renderer = CanvasRenderer;
 230      } else if (typeof(SVGRenderer) !== 'undefined') {
 231          defaultOptions.renderer = SVGRenderer;
 232      } else {
 233          throw new Error('Please load either the SVG- or the CanvasRenderer');
 234      }
 235  
 236      var options = {};
 237      var currentValue = 0;
 238  
 239      /**
 240       * Initialize the plugin by creating the options object and initialize rendering
 241       */
 242      var init = function() {
 243          this.el = el;
 244          this.options = options;
 245  
 246          // merge user options into default options
 247          for (var i in defaultOptions) {
 248              if (defaultOptions.hasOwnProperty(i)) {
 249                  options[i] = opts && typeof(opts[i]) !== 'undefined' ? opts[i] : defaultOptions[i];
 250                  if (typeof(options[i]) === 'function') {
 251                      options[i] = options[i].bind(this);
 252                  }
 253              }
 254          }
 255  
 256          // check for jQuery easing
 257          if (typeof(options.easing) === 'string' && typeof(jQuery) !== 'undefined' && jQuery.isFunction(jQuery.easing[options.easing])) {
 258              options.easing = jQuery.easing[options.easing];
 259          } else {
 260              options.easing = defaultOptions.easing;
 261          }
 262  
 263          // create renderer
 264          this.renderer = new options.renderer(el, options);
 265  
 266          // initial draw
 267          this.renderer.draw(currentValue);
 268  
 269          // initial update
 270          if (el.dataset && el.dataset.percent) {
 271              this.update(parseFloat(el.dataset.percent));
 272          } else if (el.getAttribute && el.getAttribute('data-percent')) {
 273              this.update(parseFloat(el.getAttribute('data-percent')));
 274          }
 275      }.bind(this);
 276  
 277      /**
 278       * Update the value of the chart
 279       * @param  {number} newValue Number between 0 and 100
 280       * @return {object}          Instance of the plugin for method chaining
 281       */
 282      this.update = function(newValue) {
 283          newValue = parseFloat(newValue);
 284          if (options.animate) {
 285              this.renderer.animate(currentValue, newValue);
 286          } else {
 287              this.renderer.draw(newValue);
 288          }
 289          currentValue = newValue;
 290          return this;
 291      }.bind(this);
 292  
 293      init();
 294  };
