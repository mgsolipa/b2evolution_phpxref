 167          var

 212  	function launch(target) {
 213          if (!closing) {
 214              
 215              element = target;
 216              
 217              makeSettings();
 218              
 219              $related = $(element);
 220              
 221              index = 0;
 222              
 223              if (settings.rel !== 'nofollow') {
 224                  $related = $('.' + boxElement).filter(function () {
 225                      var relRelated = $.data(this, colorbox).rel || this.rel;
 226                      return (relRelated === settings.rel);
 227                  });
 228                  index = $related.index(element);
 229                  
 230                  // Check direct calls to ColorBox.
 231                  if (index === -1) {
 232                      $related = $related.add(element);
 233                      index = $related.length - 1;
 234                  }
 235              }
 236              
 237              if (!open) {
 238                  open = active = true; // Prevents the page-change action from queuing up if the visitor holds down the left or right keys.
 239                  
 240                  $box.show();
 241                  
 242                  if (settings.returnFocus) {
 243                      try {
 244                          element.blur();
 245                          $(element).one(event_closed, function () {
 246                              try {
 247                                  this.focus();
 248                              } catch (e) {
 249                                  // do nothing
 250                              }
 251                          });
 252                      } catch (e) {
 253                          // do nothing
 254                      }
 255                  }
 256  
 257                  $overlay.css({"cursor": settings.overlayClose ? "pointer" : "auto"}).show();
 258  
 259                  // Opens inital empty ColorBox prior to content being loaded.
 260                  settings.w = setSize(settings.initialWidth, 'x');
 261                  settings.h = setSize(settings.initialHeight, 'y');
 262                  publicMethod.position();
 263  
 264                  trigger(event_open, settings.onOpen);
 265  
 266                  $groupControls.add($title).hide();
 267  
 268                  $close.html(settings.close).show();
 269              }
 270  
 271              publicMethod.load(true);
 272          }

 282          var $this = this;
 283          
 284          options = options || {};
 285          
 286          if (!$this[0]) {
 287              if ($this.selector) { // if a selector was given and it didn't match any elements, go ahead and exit.
 288                  return $this;
 289              }
 290              // if no selector was given (ie. $.colorbox()), create a temporary element to work with
 291              $this = $('<a/>');

 293          }
 294          
 295          if (callback) {
 296              options.onComplete = callback;
 297          }

 409      publicMethod.position = function (speed, loadedCallback) {
 410          var top = 0, left = 0;
 411          
 412          $window.unbind('resize.' + prefix);
 413          
 414          // remove the modal so that it doesn't influence the document width/height        
 415          $box.hide();
 416          
 417          if (settings.fixed) {
 418              $box.css({position: 'fixed'});
 419          } else {
 420              top = $window.scrollTop();
 421              left = $window.scrollLeft();
 422              $box.css({position: 'absolute'});
 423          }
 424          
 425          // keeps the top and left positions within the browser's viewport.
 426          if (settings.right !== false) {
 427              left += Math.max($window.width() - settings.w - loadedWidth - interfaceWidth - setSize(settings.right, 'x'), 0);
 428          } else if (settings.left !== false) {
 429              left += setSize(settings.left, 'x');
 430          } else {
 431              left += Math.round(Math.max($window.width() - settings.w - loadedWidth - interfaceWidth, 0) / 2);
 432          }
 433          
 434          if (settings.bottom !== false) {
 435              top += Math.max(document.documentElement.clientHeight - settings.h - loadedHeight - interfaceHeight - setSize(settings.bottom, 'y'), 0);
 436          } else if (settings.top !== false) {
 437              top += setSize(settings.top, 'y');
 438          } else {
 439              top += Math.round(Math.max(document.documentElement.clientHeight - settings.h - loadedHeight - interfaceHeight, 0) / 2);
 440          }
 441          
 442          $box.show();
 443          
 444          // setting the speed to 0 to reduce the delay between same-sized content.
 445          speed = ($box.width() === settings.w + loadedWidth && $box.height() === settings.h + loadedHeight) ? 0 : speed || 0;
 446          
 447          // this gives the wrapper plenty of breathing room so it's floated contents can move around smoothly,
 448          // but it has to be shrank down around the size of div#colorbox when it's done.  If not,
 449          // it can invoke an obscure IE bug when using iframes.
 450          $wrap[0].style.width = $wrap[0].style.height = "9999px";

 452  		function modalDimensions(that) {
 453              // loading overlay height has to be explicitly set for IE6.
 454              $content[0].style.width = that.style.width;
 455              $loadingOverlay[0].style.height = $loadingOverlay[1].style.height = $content[0].style.height = that.style.height;
 456          }
 457          
 458          $box.dequeue().animate({width: settings.w + loadedWidth, height: settings.h + loadedHeight, top: top, left: left}, {
 459              duration: speed,
 460              complete: function () {
 461                  modalDimensions(this);
 462                  
 463                  active = false;
 464                  
 465                  // shrink the wrapper down to exactly the size of colorbox to avoid a bug in IE's iframe implementation.
 466                  $wrap[0].style.width = (settings.w + loadedWidth + interfaceWidth) + "px";
 467                  $wrap[0].style.height = (settings.h + loadedHeight + interfaceHeight) + "px";
 468                  
 469                  if (loadedCallback) {
 470                      loadedCallback();
 471                  }
 472                  
 473                  setTimeout(function(){  // small delay before binding onresize due to an IE8 bug.
 474                      $window.bind('resize.' + prefix, publicMethod.position);
 475                  }, 1);
 476  
 477                  ( $wrap.parent().width() <= 480 ) ? $current.hide() : $current.show();
 478                  ( $wrap.parent().width() < 380 ) ? $slideshow.hide() : $slideshow.show();
 479              },
 480              step: function () {
 481                  modalDimensions(this);
 482              }
 483          });
