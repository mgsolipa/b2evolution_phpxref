   8  jQuery( document ).ready( function()
   9  {
  10      jQuery( 'a[rel^="lightbox"]' ).each( function()
  11      {
  12          var type = jQuery( this ).attr( 'rel' ).match( /lightbox\[([a-z]+)/i );
  13          type = type ? type[1] : '';
  14          switch( type[1] )
  15          {
  16              case 'p': // post
  17                  jQuery( this ).colorbox( b2evo_colorbox_params_post );
  18                  break;
  19              case 'c': // comment
  20                  jQuery( this ).colorbox( b2evo_colorbox_params_cmnt );
  21                  break;
  22              case 'user': // user
  23                  jQuery( this ).colorbox( b2evo_colorbox_params_user );
  24                  break;
  25              default: // all others
  26                  jQuery( this ).colorbox( b2evo_colorbox_params );
  27          }
  28      } );

  30      jQuery( '#colorbox' ).swipe(

  32          swipeLeft: function( event, direction, distance, duration, fingerCount )
  33          {
  34              if( typeof( colorbox_is_zoomed ) == 'undefined' || ! colorbox_is_zoomed )

  41              if( typeof( colorbox_is_zoomed ) == 'undefined' || ! colorbox_is_zoomed )

  45          },
  46      } );
