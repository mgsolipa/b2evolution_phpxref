 187      /**
 188       * Display breadcrumbs
 189       *
 190       * @param integer Chapter ID
 191       * @param array Params
 192       */
 193  	function display_breadcrumbs( $chapter_ID, $params = array() )
 194      {
 195          if( $chapter_ID == 0 )
 196          {    // No selected chapter
 197              return;
 198          }
 199  
 200          $params = array_merge( array(
 201                  'before'    => '<div class="breadcrumbs">',
 202                  'after'     => '</div>',
 203                  'separator' => '',
 204              ), $params );
 205  
 206          global $Blog;
 207  
 208          $ChapterCache = & get_ChapterCache();
 209  
 210          $breadcrumbs = array();
 211          do
 212          {    // Get all parent chapters
 213              $Chapter = & $ChapterCache->get_by_ID( $chapter_ID );
 214  
 215              $breadcrumbs[] = '<a href="'.$Chapter->get_permanent_url().'">'.$Chapter->dget( 'name' ).'</a>';
 216  
 217              $chapter_ID = $Chapter->get( 'parent_ID' );
 218          }
 219          while( !empty( $chapter_ID ) );
 220  
 221          $breadcrumbs[] = '<a href="'.$Blog->get( 'blogurl' ).'">'.$Blog->get( 'name' ).'</a>';
 222          $breadcrumbs = array_reverse( $breadcrumbs );
 223  
 224          // Display
 225          echo $params['before'];
 226          echo implode( $params['separator'], $breadcrumbs );
 227          echo $params['after'];
 228      }

 243      /**
 244       * Get HTML code of button to create a new post
 245       *
 246       * @param integer Chapter ID
 247       * @param object Item
 248       * @return string
 249       */
 250  	function get_post_button( $chapter_ID, $Item = NULL )
 251      {
 252          global $Blog;
 253  
 254          $post_button = '';
 255  
 256          $chapter_is_locked = false;
 257  
 258          $write_new_post_url = $Blog->get_write_item_url( $chapter_ID );
 259          if( $write_new_post_url != '' )
 260          { // Display button to write a new post
 261              $post_button = '<a href="'.$write_new_post_url.'"><span class="ficon newTopic" title="'.T_('Post new topic').'"></span></a>';
 262          }
 263          else
 264          { // If a creating of new post is unavailable
 265              $ChapterCache = & get_ChapterCache();
 266              $current_Chapter = $ChapterCache->get_by_ID( $chapter_ID, false, false );
 267  
 268              if( $current_Chapter && $current_Chapter->lock )
 269              { // Display icon to inform that this forum is locked
 270                  $post_button = '<span class="ficon locked" title="'.T_('This forum is locked: you cannot post, reply to, or edit topics.').'"></span>';
 271                  $chapter_is_locked = true;
 272              }
 273          }
 274  
 275          if( !empty( $Item ) )
 276          {
 277              if( $Item->comment_status == 'closed' || $Item->comment_status == 'disabled' || $Item->is_locked() )
 278              { // Display icon to inform that this topic is locked for comments
 279                  if( !$chapter_is_locked )
 280                  { // Display this button only when chapter is not locked, to avoid a duplicate button
 281                      $post_button .= ' <span class="ficon locked" title="'.T_('This topic is locked: you cannot edit posts or make replies.').'"></span>';
 282                  }
 283              }
 284              else
 285              { // Display button to post a reply
 286                  $post_button .= ' <a href="'.$Item->get_feedback_url().'#form_p'.$Item->ID.'"><span class="ficon postReply" title="'.T_('Reply to topic').'"></span></a>';
 287              }
 288          }
 289  
 290          if( !empty( $post_button ) )
 291          { // Display button
 292              return '<div class="post_button">'.$post_button.'</div>';
 293          }
 294      }

 297      /**
 298       * Determine to display status banner or to don't display
 299       *
 300       * @param string Status of Item or Comment
 301       * @return boolean TRUE if we can display status banner for given status
 302       */
 303  	function enabled_status_banner( $status )
 304      {
 305          if( $status != 'published' )
 306          {    // Display status banner everytime when status is not 'published'
 307              return true;
 308          }
 309  
 310          if( is_logged_in() && $this->get_setting( 'banner_public' ) )
 311          {    // Also display status banner if status is 'published'
 312              //   AND current user is logged in
 313              //   AND this feature is enabled in skin settings
 314              return true;
 315          }
 316  
 317          // Don't display status banner
 318          return false;
 319      }
