 223      /**
 224       * Display breadcrumbs
 225       *
 226       * @param integer Chapter ID
 227       * @param array Params
 228       */
 229  	function display_breadcrumbs( $chapter_ID, $params = array() )
 230      {
 231          if( $chapter_ID == 0 )
 232          {    // No selected chapter
 233              return;
 234          }
 235  
 236          $params = array_merge( array(
 237                  'before'    => '<div class="breadcrumbs">',
 238                  'after'     => '</div>',
 239                  'separator' => ' -> ',
 240              ), $params );
 241  
 242          global $Blog;
 243  
 244          $ChapterCache = & get_ChapterCache();
 245  
 246          $breadcrumbs = array();
 247          do
 248          {    // Get all parent chapters
 249              $Chapter = & $ChapterCache->get_by_ID( $chapter_ID );
 250  
 251              $breadcrumbs[] = '<a href="'.$Chapter->get_permanent_url().'">'.$Chapter->dget( 'name' ).'</a>';
 252  
 253              $chapter_ID = $Chapter->get( 'parent_ID' );
 254          }
 255          while( !empty( $chapter_ID ) );
 256  
 257          $breadcrumbs[] = '<a href="'.$Blog->get( 'blogurl' ).'">'.$Blog->get( 'name' ).'</a>';
 258          $breadcrumbs = array_reverse( $breadcrumbs );
 259  
 260          // Display
 261          echo $params['before'];
 262          echo implode( $params['separator'], $breadcrumbs );
 263          echo $params['after'];
 264      }

 267      /**
 268       * Display button to create a new post
 269       *
 270       * @param integer Chapter ID
 271       */
 272  	function display_post_button( $chapter_ID, $Item = NULL )
 273      {
 274          global $Blog;
 275  
 276          $post_button = '';
 277  
 278          $chapter_is_locked = false;
 279  
 280          $write_new_post_url = $Blog->get_write_item_url( $chapter_ID );
 281          if( $write_new_post_url != '' )
 282          { // Display button to write a new post
 283              $post_button = '<a href="'.$write_new_post_url.'"><span class="ficon newTopic" title="'.T_('Post new topic').'"></span></a>';
 284          }
 285          else
 286          { // If a creating of new post is unavailable
 287              $ChapterCache = & get_ChapterCache();
 288              $current_Chapter = $ChapterCache->get_by_ID( $chapter_ID, false, false );
 289  
 290              if( $current_Chapter && $current_Chapter->lock )
 291              { // Display icon to inform that this forum is locked
 292                  $post_button = '<span class="ficon locked" title="'.T_('This forum is locked: you cannot post, reply to, or edit topics.').'"></span>';
 293                  $chapter_is_locked = true;
 294              }
 295          }
 296  
 297          if( !empty( $Item ) )
 298          {
 299              if( $Item->comment_status == 'closed' || $Item->comment_status == 'disabled' || $Item->is_locked() )
 300              { // Display icon to inform that this topic is locked for comments
 301                  if( !$chapter_is_locked )
 302                  { // Display this button only when chapter is not locked, to avoid a duplicate button
 303                      $post_button .= ' <span class="ficon locked" title="'.T_('This topic is locked: you cannot edit posts or make replies.').'"></span>';
 304                  }
 305              }
 306              else
 307              { // Display button to post a reply
 308                  $post_button .= ' <a href="'.$Item->get_feedback_url().'#form_p'.$Item->ID.'"><span class="ficon postReply" title="'.T_('Reply to topic').'"></span></a>';
 309              }
 310          }
 311  
 312          if( !empty( $post_button ) )
 313          { // Display button
 314              echo '<div class="post_button">';
 315              echo $post_button;
 316              echo '</div>';
 317          }
 318      }

 321      /**
 322       * Determine to display status banner or to don't display
 323       *
 324       * @param string Status of Item or Comment
 325       * @return boolean TRUE if we can display status banner for given status
 326       */
 327  	function enabled_status_banner( $status )
 328      {
 329          if( $status != 'published' )
 330          {    // Display status banner everytime when status is not 'published'
 331              return true;
 332          }
 333  
 334          if( is_logged_in() && $this->get_setting( 'banner_public' ) )
 335          {    // Also display status banner if status is 'published'
 336              //   AND current user is logged in
 337              //   AND this feature is enabled in skin settings
 338              return true;
 339          }
 340  
 341          // Don't display status banner
 342          return false;
 343      }
