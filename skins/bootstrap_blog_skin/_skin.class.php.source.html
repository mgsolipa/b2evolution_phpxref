 207          global $Messages, $debug;

 210          parent::display_init( array(
 211                  'jquery',                  // Load jQuery
 212                  'font_awesome',            // Load Font Awesome (and use its icons as a priority over the Bootstrap glyphicons)
 213                  'bootstrap',               // Load Bootstrap (without 'bootstrap_theme_css')
 214                  'bootstrap_evo_css',       // Load the b2evo_base styles for Bootstrap (instead of the old b2evo_base styles)
 215                  'bootstrap_messages',      // Initialize $Messages Class to use Bootstrap styles
 216                  'style_css',               // Load the style.css file of the current skin
 217                  'colorbox',                // Load Colorbox (a lightweight Lightbox alternative + customizations for b2evo)
 218                  'bootstrap_init_tooltips', // Inline JS to init Bootstrap tooltips (E.g. on comment form for allowed file extensions)
 219                  'disp_auto',               // Automatically include additional CSS and/or JS required by certain disps (replace with 'disp_off' to disable this)
 220              ) );
 221  
 222          // Skin specific initializations:
 223  
 224          // Limit images by max height:
 225          $max_image_height = intval( $this->get_setting( 'max_image_height' ) );
 226          if( $max_image_height > 0 )
 227          {
 228              add_css_headline( '.evo_image_block img { max-height: '.$max_image_height.'px; width: auto; }' );
 229          }

 501                  // We should use 'popover' tooltip plugin for bootstrap skins
 502                  // This tooltips appear on mouse over user logins or on plugin help icons
 503                  return 'popover';
 504                  break;
 505  
 506              case 'plugin_template':
 507                  // Template for plugins:
 508                  return array(
 509                          // This template is used to build a plugin toolbar with action buttons above edit item/comment area:
 510                          'toolbar_before'       => '<div class="btn-toolbar $toolbar_class$" role="toolbar">',
 511                          'toolbar_after'        => '</div>',
 512                          'toolbar_title_before' => '<div class="btn-toolbar-title">',
 513                          'toolbar_title_after'  => '</div>',
 514                          'toolbar_group_before' => '<div class="btn-group btn-group-xs" role="group">',
 515                          'toolbar_group_after'  => '</div>',
 516                          'toolbar_button_class' => 'btn btn-default',
 517                      );
 518  
 519              case 'modal_window_js_func':
 520                  // JavaScript function to initialize Modal windows, @see echo_user_ajaxwindow_js()
 521                  return 'echo_modalwindow_js_bootstrap';
 522                  break;
 523  
 524              default:
 525                  // Delegate to parent class:
 526                  return parent::get_template( $name );
 527          }

 531      /**
 532       * Check if we can display a widget container
 533       *
 534       * @param string Widget container key: 'header', 'page_top', 'menu', 'sidebar', 'sidebar2', 'footer'
 535       * @param string Skin setting name
 536       * @return boolean TRUE to display
 537       */
 538  	function is_visible_container( $container_key, $setting_name = 'access_login_containers' )
 539      {
 540          $access = $this->get_setting( $setting_name );
 541  
 542          return ( ! empty( $access ) && ! empty( $access[ $container_key ] ) );
 543      }
 544  
 545  
 546      /**
 547       * Check if we can display a sidebar for the current layout
 548       *
 549       * @param boolean TRUE to check if at least one sidebar container is visible
 550       * @return boolean TRUE to display a sidebar
 551       */
 552  	function is_visible_sidebar( $check_containers = false )
 553      {
 554          $layout = $this->get_setting( 'layout' );
 555  
 556          if( $layout != 'left_sidebar' && $layout != 'right_sidebar' )
 557          { // Sidebar is not displayed for selected skin layout
 558              return false;
 559          }
 560  
 561          if( $check_containers )
 562          { // Check if at least one sidebar container is visible
 563              return ( $this->is_visible_container( 'sidebar' ) ||  $this->is_visible_container( 'sidebar2' ) );
 564          }
 565          else
 566          { // We should not check the visibility of the sidebar containers for this case
 567              return true;
 568          }
 569      }

 578  	function get_column_class()

 594              case 'single_column_extra_narrow':

 596                  return 'col-xs-12 col-sm-10 col-sm-offset-1 col-md-8 col-md-offset-2 col-lg-6 col-lg-offset-3';
 597  
 598              case 'left_sidebar':
 599                  // Left Sidebar
 600                  return 'col-md-9 pull-right';
 601  
 602              case 'right_sidebar':
