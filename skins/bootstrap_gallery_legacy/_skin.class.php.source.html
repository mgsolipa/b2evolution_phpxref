 376       * Determine to display status banner or to don't display

 377       *

 378       * @param string Status of Item or Comment

 379       * @return boolean TRUE if we can display status banner for given status

 380       */
 381  	function enabled_status_banner( $status )
 382      {
 383          if( $status != 'published' )
 384          { // Display status banner everytime when status is not 'published'
 385              return true;
 386          }
 387          if( is_logged_in() && $this->get_setting( 'banner_public' ) )
 388          { // Also display status banner if status is 'published'
 389              //   AND current user is logged in

 390              //   AND this feature is enabled in skin settings

 391              return true;
 392          }
 393          // Don't display status banner

 394          return false;
 395      }
