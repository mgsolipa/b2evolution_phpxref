 195  	function display_init()
 196      {
 197          global $Messages, $disp, $debug;
 198  
 199          // Request some common features that the parent function (Skin::display_init()) knows how to provide:
 200          parent::display_init( array(
 201                  'jquery',                  // Load jQuery
 202                  'font_awesome',            // Load Font Awesome (and use its icons as a priority over the Bootstrap glyphicons)
 203                  'bootstrap',               // Load Bootstrap (without 'bootstrap_theme_css')
 204                  'bootstrap_evo_css',       // Load the b2evo_base styles for Bootstrap (instead of the old b2evo_base styles)
 205                  'bootstrap_messages',      // Initialize $Messages Class to use Bootstrap styles
 206                  'style_css',               // Load the style.css file of the current skin
 207                  'colorbox',                // Load Colorbox (a lightweight Lightbox alternative + customizations for b2evo)
 208                  'bootstrap_init_tooltips', // Inline JS to init Bootstrap tooltips (E.g. on comment form for allowed file extensions)
 209                  'disp_auto',               // Automatically include additional CSS and/or JS required by certain disps (replace with 'disp_off' to disable this)
 210              ) );
 211  
 212          // Skin specific initializations:
 213  
 214          // Limit images by max height:
 215          $max_image_height = intval( $this->get_setting( 'max_image_height' ) );
 216          if( $max_image_height > 0 )
 217          {
 218              add_css_headline( '.evo_image_block img { max-height: '.$max_image_height.'px; width: auto; }' );
 219          }
 220  
 221          // Initialize a template depending on current page
 222          switch( $disp )
 223          {
 224              case 'front':
 225                  // Init star rating for intro posts:
 226                  init_ratings_js( 'blog', true );
 227                  break;
 228  
 229              case 'posts':
 230                  global $cat, $bootstrap_manual_posts_text;
 231  
 232                  // Init star rating for intro posts:
 233                  init_ratings_js( 'blog', true );
 234  
 235                  $bootstrap_manual_posts_text = T_('Posts');
 236                  if( ! empty( $cat ) )
 237                  { // Init the <title> for categories page:
 238                      $ChapterCache = & get_ChapterCache();
 239                      if( $Chapter = & $ChapterCache->get_by_ID( $cat, false ) )
 240                      {
 241                          $bootstrap_manual_posts_text = $Chapter->get( 'name' );
 242                      }
 243                  }
 244                  break;
 245          }
 246  
 247          if( $this->is_left_navigation_visible() )
 248          { // Include JS code for left navigation panel only when it is displayed:
 249              require_js( $this->get_url().'left_navigation.js' );
 250          }
 251      }
