 198  	function display_init()
 199      {
 200          // call parent:
 201          parent::display_init();        // We pass NO params. This gives up the default Skins API v5 behavior.
 202  
 203          // Add custom CSS:
 204          $custom_css = '';
 205  
 206          // Custom menu styles:
 207          $custom_styles = array();
 208          if( $bg_color = $this->get_setting( 'menu_bg_color' ) )
 209          { // Background color:
 210              $custom_styles[] = 'background-color: '.$bg_color;
 211          }
 212          if( $text_color = $this->get_setting( 'menu_text_color' ) )
 213          { // Text color:
 214              $custom_styles[] = 'color: '.$text_color;
 215          }
 216          if( ! empty( $custom_styles ) )
 217          {
 218              $custom_css .= '    div.pageHeader { '.implode( ';', $custom_styles )." }\n";
 219          }
 220  
 221          // Custom page styles:
 222          $custom_styles = array();
 223          if( $bg_color = $this->get_setting( 'page_bg_color' ) )
 224          { // Background color:
 225              $custom_styles[] = 'background-color: '.$bg_color;
 226          }
 227          if( $text_color = $this->get_setting( 'page_text_color' ) )
 228          { // Text color:
 229              $custom_styles[] = 'color: '.$text_color;
 230          }
 231          if( ! empty( $custom_styles ) )
 232          {
 233              $custom_css .= '    body { '.implode( ';', $custom_styles )." }\n";
 234          }
 235  
 236          // Custom post area styles:
 237          $custom_styles = array();
 238          if( $bg_color = $this->get_setting( 'post_bg_color' ) )
 239          { // Background color:
 240              $custom_styles[] = 'background-color: '.$bg_color;
 241          }
 242          if( $text_color = $this->get_setting( 'post_text_color' ) )
 243          { // Text color:
 244              $custom_styles[] = 'color: '.$text_color;
 245          }
 246          if( ! empty( $custom_styles ) )
 247          {
 248              $custom_css .= '    div.bDetails { '.implode( ';', $custom_styles )." }\n";
 249          }
 250  
 251          global $thumbnail_sizes;
 252          $posts_thumb_size = $this->get_setting( 'posts_thumb_size' );
 253          if( isset( $thumbnail_sizes[ $posts_thumb_size ] ) )
 254          {
 255              // Make the width of image block as fixed to don't expand it by long post title text
 256              $custom_css .= '    .posts_list .bPost { max-width:'.$thumbnail_sizes[ $posts_thumb_size ][1]."px }\n";

 283      /**
 284       * Determine to display status banner or to don't display
 285       *
 286       * @param string Status of Item or Comment
 287       * @return boolean TRUE if we can display status banner for given status
 288       */
 289  	function enabled_status_banner( $status )
 290      {
 291          if( $status != 'published' )
 292          { // Display status banner everytime when status is not 'published'
 293              return true;
 294          }
 295  
 296          if( is_logged_in() && $this->get_setting( 'banner_public' ) )
 297          { // Also display status banner if status is 'published'
 298              //   AND current user is logged in
 299              //   AND this feature is enabled in skin settings
 300              return true;
 301          }
 302  
 303          // Don't display status banner
 304          return false;
 305      }
