  16  function db_delete()
  17  {
  18      global $DB, $db_config, $tableprefix;
  19  
  20      echo "Disabling foreign key checks...<br />\n";
  21      $DB->query( 'SET FOREIGN_KEY_CHECKS=0' );
  22  
  23      foreach( $db_config['aliases'] as $alias => $tablename )
  24      {
  25          echo "Dropping $tablename table...<br />\n";
  26          evo_flush();
  27          $DB->query( 'DROP TABLE IF EXISTS '.$tablename );
  28      }
  29  
  30      // Get remaining tables with the same prefix and delete them as well. Probably these tables are some b2evolution plugins tables.
  31      $remaining_tables = $DB->get_col( 'SHOW TABLES FROM `'.$db_config['name'].'` LIKE "'.$tableprefix.'%"' );
  32      foreach( $remaining_tables as $tablename )
  33      {
  34          echo "Dropping $tablename table...<br />\n";
  35          evo_flush();
  36          $DB->query( 'DROP TABLE IF EXISTS '.$tablename );
  37      }
  38  }

  48      /* REMOVE PAGE CACHE */
  49      load_class( '_core/model/_pagecache.class.php', 'PageCache' );

  52      $PageCache = new PageCache( NULL );

  55      // Skip if T_blogs table is already deleted. Note that db_delete() will not throw any errors on missing tables.
  56      if( $DB->query( 'SHOW TABLES LIKE "T_blogs"' ) )
  57      { // Get all blogs
  58          $blogs_SQL = new SQL();
  59          $blogs_SQL->SELECT( 'blog_ID' );
  60          $blogs_SQL->FROM( 'T_blogs' );
  61          $blogs = $DB->get_col( $blogs_SQL->get() );
  62  
  63          $BlogCache = & get_BlogCache( 'blog_ID' );
  64          foreach( $blogs as $blog_ID )
  65          {
  66              $Blog = $BlogCache->get_by_ID( $blog_ID );
  67  
  68              // Remove page cache of current blog
  69              $PageCache = new PageCache( $Blog );
  70              $PageCache->cache_delete();
  71          }
  72      }
