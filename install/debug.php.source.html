  90  function debug_get_config( $close_handle = true )
  91  {
  92      global $errors, $file_overrides_name, $file_section_start, $file_section_end, $file_overrides_handle;
  93  
  94      // Get debug config from file
  95      $file_overrides_handle = @fopen( $file_overrides_name, 'r+' );
  96      $file_overrides_content = '';
  97      if( $file_overrides_handle )
  98      {
  99          while( !feof( $file_overrides_handle ) )
 100          {
 101              $file_overrides_content .= fgets( $file_overrides_handle, 4096 );
 102          }
 103          if( $close_handle )
 104          {
 105              fclose( $file_overrides_handle );
 106          }
 107      }
 108      else
 109      {    // Bad file name or file has not the rights to read/write
 110          $errors[] = T_('Config file cannot be opened');
 111          return false;
 112      }
 113  
 114      return $file_overrides_content;
 115  }
