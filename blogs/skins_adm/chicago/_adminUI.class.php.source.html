 421  	function display_skin_settings( $Form, $user_ID )
 422      {
 423          global $UserSettings, $current_User;
 424          $Form->begin_fieldset( T_( 'Admin skin settings' ), array( 'id' => 'admin_skin_settings' ) );
 425          parent::display_skin_settings( $Form, $user_ID );
 426  
 427          $user_admin_skin = $UserSettings->get( 'admin_skin', $user_ID );
 428          if( $UserSettings->get( 'admin_skin', $current_User->ID ) == $user_admin_skin )
 429          {
 430              $Form->checklist( array(
 431                          array( 'show_evobar', 1, T_('Show evobar'), $UserSettings->get( 'show_evobar', $user_ID ) ),
 432                          array( 'show_breadcrumbs', 1, T_('Show breadcrumbs path'), $UserSettings->get( 'show_breadcrumbs', $user_ID ) ),
 433                          array( 'show_menu', 1, T_('Show Menu'), $UserSettings->get( 'show_menu', $user_ID ) ) ),
 434                      'chicago_settings', T_('Chicago skin settings') );
 435          }
 436          else
 437          {
 438              $Form->info( '', sprintf( T_( 'Admin skin settings for this user cannot be edited because this user is using a different admin skin (%s)' ), $user_admin_skin ) );
 439          }
 440  
 441          $Form->end_fieldset();
 442  
 443          // JavaScript code to dynamically change display settings. show_evobar or show_menu always have to be checked
 444          ?>
 445          <script type="text/javascript">
 446          jQuery( '[name = show_evobar], [name = show_menu]' ).click( function()
 447          {
 448              if( ! ( jQuery( '[name = show_evobar]' ).attr( 'checked' ) || jQuery( '[name = show_menu]' ).attr( 'checked' ) ) )
 449              {
 450                  jQuery( '[name = show_evobar]' ).attr( 'checked', true );
 451              }
 452          } );
 453          </script>
 454          <?php
