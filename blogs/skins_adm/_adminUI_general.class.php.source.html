 281                          .$params['separator']

 286  
 287              $r .= $params['after'];
 288          }

 733       * Returns list of buttons for available Collections (aka Blogs) to work on.
 734       *
 735       * @return string HTML
 736       */
 737  	function get_bloglist_buttons( $title = '' )
 738      {
 739          global $current_User, $blog, $pagenow;
 740  
 741          $max_buttons = 7;
 742  
 743          if( empty( $this->coll_list_permname ) )
 744          {    // We have not requested a list of blogs to be displayed
 745              return;
 746          }
 747  
 748          // Prepare url params:
 749          $url_params = '?';
 750          $form_params = '';
 751          foreach( $this->coll_list_url_params as $name => $value )
 752          {
 753              $url_params .= $name.'='.$value.'&amp;';
 754              $form_params .= '<input type="hidden" name="'.$name.'" value="'.$value.'" />';
 755          }
 756  
 757          $template = $this->get_template( 'CollectionList' );
 758  
 759          $BlogCache = & get_BlogCache();
 760  
 761          $blog_array = $BlogCache->load_user_blogs( $this->coll_list_permname, $this->coll_list_permlevel );
 762  
 763  
 764          $buttons = '';
 765          $select_options = '';
 766          $not_favorite_blogs = false;
 767  
 768          foreach( $blog_array as $l_blog_ID )
 769          { // Loop through all blogs that match the requested permission:
 770  
 771              $l_Blog = & $BlogCache->get_by_ID( $l_blog_ID );
 772  
 773              if( $l_Blog->get( 'favorite' ) || $l_blog_ID == $blog )
 774              { // If blog is favorute OR current blog, Add blog as a button:
 775                  $buttons .= $template[ $l_blog_ID == $blog ? 'beforeEachSel' : 'beforeEach' ];
 776  
 777                  $buttons .= '<a href="'.$url_params.'blog='.$l_blog_ID
 778                              .'" class="'.( $l_blog_ID == $blog ? 'CurrentBlog' : 'OtherBlog' ).'"';
 779  
 780                  if( !is_null($this->coll_list_onclick) )
 781                  {    // We want to include an onclick attribute:
 782                      $buttons .= ' onclick="'.sprintf( $this->coll_list_onclick, $l_blog_ID ).'"';
 783                  }
 784  
 785                  $buttons .= '>'.$l_Blog->dget( 'shortname', 'htmlbody' ).'</a> ';
 786  
 787                  if( $l_blog_ID == $blog )
 788                  {
 789                      $buttons .= $template['afterEachSel'];
 790                  }
 791                  else
 792                  {
 793                      $buttons .= $template['afterEach'];
 794                  }
 795              }
 796  
 797              if( !$l_Blog->get( 'favorite' ) )
 798              { // If blog is not favorute, Add it into the select list:
 799                  $not_favorite_blogs = true;
 800                  $select_options .= '<option value="'.$l_blog_ID.'"';
 801                  if( $l_blog_ID == $blog )
 802                  {
 803                      $select_options .= ' selected="selected"';
 804                  }
 805                  $select_options .= '>'.$l_Blog->dget( 'shortname', 'formvalue' ).'</option>';
 806              }
 807          }
 808  
 809          $r = $template['before'];
 810  
 811          $r .= $title;
 812  
 813          if( !empty( $this->coll_list_all_title ) )
 814          { // We want to add an "all" button
 815              $r .= $template[ $blog == 0 ? 'beforeEachSel' : 'beforeEach' ];
 816              $r .= '<a href="'.$this->coll_list_all_url
 817                          .'" class="'.( $blog == 0 ? 'CurrentBlog' : 'OtherBlog' ).'">'
 818                          .$this->coll_list_all_title.'</a> ';
 819              $r .= $template[ $blog == 0 ? 'afterEachSel' : 'afterEach' ];
 820          }
 821  
 822          $r .= $template['buttons_start'];
 823          $r .= $buttons;
 824          $r .= $template['buttons_end'];
 825  
 826  
 827          $r .= $template['select_start'];
 828          if( $not_favorite_blogs )
 829          { // Display select list with not favorite blogs
 830              $r .= '<form action="'.$pagenow.'" method="get">';
 831              $r .= $form_params;
 832              $r .= '<select name="blog" onchange="';
 833              if( empty( $this->coll_list_onclick ) )
 834              { // Just submit...
 835                  $r .= 'if(this.value>0) this.form.submit();';
 836              }
 837              else
 838              {
 839                  $r .= sprintf( $this->coll_list_onclick, 'this.value' );
 840              }
 841              $r .= '">'
 842                  .'<option value="0">'.T_('Select blog').'</option>'
 843                  .$select_options.'</select>';
 844              $r .= '<noscript><input type="submit" value="Go" /></noscript></form>';
 845          }
 846          $r .= $template['select_end'];
 847  
 848  
 849          $r .= $template['after'];
 850  
 851          return $r;
 852      }

1281      /**
1282       * Set a specific path level (specific depth).
1283       *
1284       * First level is 0, then the first subpath/submenu is level 1, etc.
1285       *
1286       * E.g., if plugins.php gets called, there could be a call to
1287       * $AdminUI->set_path_level( 0, 'plugins' ), which selects this entry from the menu.
1288       * If a specific tab is called inside of plugins.php, there could be a call to
1289       * $AdminUI->set_path_level( 1, $tab )
1290       *
1291       * Though, it is recommended to call the wrapper functions:
1292       *  - {@link append_path_level()}
1293       *  - {@link set_path()}
1294       *
1295       * This also marks the parent node as selected and checks for permissions.
1296       *
1297       * @param integer Path level to set (starts at 0)
1298       * @param array Either the key of the path or an array(keyname, propsArray).
1299       * @param array Properties for this path entry.
1300       * @return boolean DEPRECATED True if perm granted, false if not (and we're not exiting).
1301       */
1302  	function set_path_level( $level, $pathKey, $pathProps = array() )
1303      {
1304          // Get the parent node (the level above this one):
1305          if( $level == 0 )
1306          { // first level in menu-path: parent node is NULL
1307              $parentNode = & $this->get_node_by_path( NULL );
1308          }
1309          else
1310          { // parent node is the trunk from root to previous level
1311              $parentNode = & $this->get_node_by_path( $this->get_path_range( 0, $level-1 ) );
1312          }
1313          if( ! $parentNode )
1314          { // parent node does not exist:
1315              return false;
1316          }
1317  
1318          // Store the selected entry name in the parent node:
1319          $parentNode['selected'] = $pathKey;
1320  
1321          $this->path[$level] = $pathKey;
1322          $this->pathProps[$level] = $pathProps;
1323  
1324          // pre_dump( 'set_path_level: ', $level, $pathKey, $this->pathProps[$level] );
1325  
1326          return true;
