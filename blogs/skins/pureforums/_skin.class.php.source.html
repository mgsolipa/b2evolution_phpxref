 146       * @param integer Chapter ID

 147       * @param array Params

 148       */
 149  	function display_breadcrumbs( $chapter_ID, $params = array() )
 150      {
 151          if( $chapter_ID <= 0 )
 152          { // No selected chapter, or an exlcude chapter filter is set
 153              return;
 154          }
 155  
 156          $params = array_merge( array(
 157                  'before'    => '<div class="breadcrumbs">',
 158                  'after'     => '</div>',
 159                  'separator' => '',
 160              ), $params );
 161  
 162          global $Blog;
 163  
 164          $ChapterCache = & get_ChapterCache();
 165  
 166          $breadcrumbs = array();
 167          do
 168          {    // Get all parent chapters
 169              $Chapter = & $ChapterCache->get_by_ID( $chapter_ID );
 170  
 171              $breadcrumbs[] = '<a href="'.$Chapter->get_permanent_url().'">'.$Chapter->dget( 'name' ).'</a>';
 172  
 173              $chapter_ID = $Chapter->get( 'parent_ID' );
 174          }
 175          while( !empty( $chapter_ID ) );
 176  
 177          $breadcrumbs[] = '<a href="'.$Blog->get( 'blogurl' ).'">'.$Blog->get( 'name' ).'</a>';
 178          $breadcrumbs = array_reverse( $breadcrumbs );
 179  
 180          // Display

 181          echo $params['before'];
 182          echo implode( $params['separator'], $breadcrumbs );
 183          echo $params['after'];
 184      }

 199      /**

 200       * Get HTML code of button to create a new post

 201       *

 202       * @param integer Chapter ID

 203       * @param object Item

 204       * @return string

 205       */
 206  	function get_post_button( $chapter_ID, $Item = NULL )
 207      {
 208          global $Blog;
 209  
 210          $post_button = '';
 211  
 212          $chapter_is_locked = false;
 213  
 214          $write_new_post_url = $Blog->get_write_item_url( $chapter_ID );
 215          if( $write_new_post_url != '' )
 216          { // Display button to write a new post
 217              $post_button = '<a href="'.$write_new_post_url.'"><span class="ficon newTopic" title="'.T_('Post new topic').'"></span></a>';
 218          }
 219          else
 220          { // If a creating of new post is unavailable
 221              $ChapterCache = & get_ChapterCache();
 222              $current_Chapter = $ChapterCache->get_by_ID( $chapter_ID, false, false );
 223  
 224              if( $current_Chapter && $current_Chapter->lock )
 225              { // Display icon to inform that this forum is locked
 226                  $post_button = '<span class="ficon locked" title="'.T_('This forum is locked: you cannot post, reply to, or edit topics.').'"></span>';
 227                  $chapter_is_locked = true;
 228              }
 229          }
 230  
 231          if( !empty( $Item ) )
 232          {
 233              if( $Item->comment_status == 'closed' || $Item->comment_status == 'disabled' || $Item->is_locked() )
 234              { // Display icon to inform that this topic is locked for comments
 235                  if( !$chapter_is_locked )
 236                  { // Display this button only when chapter is not locked, to avoid a duplicate button
 237                      $post_button .= ' <span class="ficon locked" title="'.T_('This topic is locked: you cannot edit posts or make replies.').'"></span>';
 238                  }
 239              }
 240              else
 241              { // Display button to post a reply
 242                  $post_button .= ' <a href="'.$Item->get_feedback_url().'#form_p'.$Item->ID.'"><span class="ficon postReply" title="'.T_('Reply to topic').'"></span></a>';
 243              }
 244          }
 245  
 246          if( !empty( $post_button ) )
 247          { // Display button
 248              return '<div class="post_button">'.$post_button.'</div>';
 249          }
 250      }

 252      /**

 253       * Get chapters

 254       *

 255       * @param integer Chapter parent ID

 256       */
 257  	function get_chapters( $parent_ID = 0 )
 258      {
 259          global $Blog, $skin_chapters_cache;
 260  
 261          if( isset( $skin_chapters_cache ) )
 262          {    // Get chapters from cache
 263              return $skin_chapters_cache;
 264          }
 265  
 266          $skin_chapters_cache = array();
 267          if( $parent_ID > 0 )
 268          {    // Get children of selected chapter
 269              global $DB, $Settings;
 270  
 271              $skin_chapters_cache = array();
 272  
 273              $SQL = new SQL();
 274              $SQL->SELECT( 'cat_ID' );
 275              $SQL->FROM( 'T_categories' );
 276              $SQL->WHERE( 'cat_parent_ID = '.$DB->quote( $parent_ID ) );
 277              if( $Settings->get( 'chapter_ordering' ) == 'manual' )
 278              {    // Manual order
 279                  $SQL->ORDER_BY( 'cat_meta, cat_order' );
 280              }
 281              else
 282              {    // Alphabetic order
 283                  $SQL->ORDER_BY( 'cat_meta, cat_name' );
 284              }
 285  
 286              $ChapterCache = & get_ChapterCache();
 287  
 288              $categories = $DB->get_results( $SQL->get() );
 289              foreach( $categories as $c => $category )
 290              {
 291                  $skin_chapters_cache[$c] = $ChapterCache->get_by_ID( $category->cat_ID );
 292                  // Get children

 293                  $SQL->WHERE( 'cat_parent_ID = '.$DB->quote( $category->cat_ID ) );
 294                  $children = $DB->get_results( $SQL->get() );
 295                  foreach( $children as $child_Chapter )
 296                  {
 297                      $skin_chapters_cache[$c]->children[$child_Chapter->cat_ID] = $ChapterCache->get_by_ID( $child_Chapter->cat_ID );
 298                  }
 299              }
 300          }
 301          else
 302          {    // Get the all chapters for current blog
 303              $ChapterCache = & get_ChapterCache();
 304              $ChapterCache->load_subset( $Blog->ID );
 305  
 306              if( isset( $ChapterCache->subset_cache[ $Blog->ID ] ) )
 307              {
 308                  $skin_chapters_cache = $ChapterCache->subset_cache[ $Blog->ID ];
 309  
 310                  foreach( $skin_chapters_cache as $c => $Chapter )
 311                  { // Init children
 312                      foreach( $skin_chapters_cache as $child_Chapter )
 313                      { // Again go through all chapters to find a children for current chapter
 314                          if( $Chapter->ID == $child_Chapter->get( 'parent_ID' ) )
 315                          { // Add to array of children
 316                              $skin_chapters_cache[$c]->children[$child_Chapter->ID] = $child_Chapter;
 317                          }
 318                      }
 319                  }
 320  
 321                  foreach( $skin_chapters_cache as $c => $Chapter )
 322                  { // Unset the child chapters
 323                      if( $Chapter->get( 'parent_ID' ) )
 324                      {
 325                          unset( $skin_chapters_cache[$c] );
 326                      }
 327                  }
 328              }
 329          }
 330  
 331          return $skin_chapters_cache;
 332      }

 341  	function enabled_status_banner( $status )

 346          }

 350              //   AND current user is logged in

 353          }
