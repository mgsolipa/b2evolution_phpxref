 109  /**

 110   * Change nav position to fixed or revert to static

 111   */
 112  function change_position_nav()
 113  {
 114      if( has_touch_event )
 115      { // Don't fix the objects on touch devices
 116          return;
 117      }
 118  
 119      if( nav_size )
 120      { // Navigation bar
 121          if( !$nav.hasClass( 'fixed' ) && jQuery( window ).scrollTop() > $nav.offset().top - nav_top )
 122          { // Make nav as fixed if we scroll down
 123              $nav.before( $navSpacer );
 124              $nav.addClass( 'fixed' ).css( 'top', nav_top + 'px' );
 125          }
 126          else if( $nav.hasClass( 'fixed' ) && jQuery( window ).scrollTop() < $navSpacer.offset().top - nav_top )
 127          { // Remove 'fixed' class from nav if we scroll to the top of page
 128              $nav.removeClass( 'fixed' ).css( 'top', '' );
 129              $navSpacer.remove();
 130          }
 131      }
 132  }
 133

 135  var nav_size = $nav.size();

 137  var $navSpacer = $( '<div />', {
 138          "height": $nav.outerHeight( true ),
 139      } );

 141  jQuery( window ).resize( function()
 142  {

 144  } );
