 140  	function display_init()
 141      {
 142          // call parent:

 143          parent::display_init();
 144  
 145          // Add CSS:

 146          require_css( 'basic_styles.css', 'blog' ); // the REAL basic styles

 147          require_css( 'basic.css', 'blog' ); // Basic styles

 148          require_css( 'blog_base.css', 'blog' ); // Default styles for the blog navigation

 149          require_css( 'item_base.css', 'blog' ); // Default styles for the post CONTENT

 150  
 151          // Colorbox (a lightweight Lightbox alternative) allows to zoom on images and do slideshows with groups of images:

 152          if ($this->get_setting("colorbox")) 
 153          {
 154              require_js_helper( 'colorbox', 'blog' );
 155          }
 156  
 157          // Make sure standard CSS is called ahead of custom CSS generated below:

 158          require_css( 'style.css', 'relative' );
 159  
 160          // Add custom CSS:

 161          $custom_css = '';
 162  
 163          // Custom menu styles:

 164          $custom_styles = array();
 165          if( $bg_color = $this->get_setting( 'menu_bg_color' ) )
 166          { // Background color:
 167              $custom_styles[] = 'background-color: '.$bg_color;
 168          }
 169          if( $text_color = $this->get_setting( 'menu_text_color' ) )
 170          { // Text color:
 171              $custom_styles[] = 'color: '.$text_color;
 172          }
 173          if( ! empty( $custom_styles ) )
 174          {
 175              $custom_css .= '    div.pageHeader { '.implode( ';', $custom_styles )." }\n";
 176          }
 177  
 178          // Custom page styles:

 179          $custom_styles = array();
 180          if( $bg_color = $this->get_setting( 'page_bg_color' ) )
 181          { // Background color:
 182              $custom_styles[] = 'background-color: '.$bg_color;
 183          }
 184          if( $text_color = $this->get_setting( 'page_text_color' ) )
 185          { // Text color:
 186              $custom_styles[] = 'color: '.$text_color;
 187          }
 188          if( ! empty( $custom_styles ) )
 189          {
 190              $custom_css .= '    body { '.implode( ';', $custom_styles )." }\n";
 191          }
 192  
 193          // Custom post area styles:

 194          $custom_styles = array();
 195          if( $bg_color = $this->get_setting( 'post_bg_color' ) )
 196          { // Background color:
 197              $custom_styles[] = 'background-color: '.$bg_color;
 198          }
 199          if( $text_color = $this->get_setting( 'post_text_color' ) )
 200          { // Text color:
 201              $custom_styles[] = 'color: '.$text_color;
 202          }
 203          if( ! empty( $custom_styles ) )
 204          {
 205              $custom_css .= '    div.bDetails { '.implode( ';', $custom_styles )." }\n";
 206          }
 207  
 208          global $thumbnail_sizes;
 209          $posts_thumb_size = $this->get_setting( 'posts_thumb_size' );
 210          if( isset( $thumbnail_sizes[ $posts_thumb_size ] ) )
 211          {
 212              // Make the width of image block as fixed to don't expand it by long post title text

 213              $custom_css .= '    .posts_list .bPost { max-width:'.$thumbnail_sizes[ $posts_thumb_size ][1]."px }\n";
 214              // Set width & height for block with text "No pictures yet"
