   1  # Apache configuration for b2evolution's base folder
   2  # Lines starting with # are comments.
   3  # Updated 2013-07-16
   4  
   5  # PHP / SECURITY:
   6  <IfModule mod_php5.c>
   7      # On some overly "secured" systems, the following may need to be added in each folder:
   8      # AddHandler application/x-httpd-php5 .php
   9  
  10      # The following will try to configure PHP in the way b2evolution would like it to be:
  11      php_flag register_globals Off
  12      # You may want to turn the following to Off on production servers
  13      # Please do this only if you know what you're doing and will be able to retrieve potential errors from your error log
  14      php_flag display_errors On
  15      # The following is the same as 'E_ALL & ~ E_STRICT' (except those constants may not be available here)
  16      php_value error_reporting 30719
  17  
  18      # You may also need some of the following:
  19      # Increase memory limit
  20      # php_value memory_limit 256M
  21  </IfModule>
  22  
  23  # DISABLE LISTING FILES ON SERVER:
  24  # Don't show directory contents (not desirable, especially for media folders)
  25  Options -Indexes
  26  # Note: if you have access to the command line of your server, it would probably
  27  # be even better to type this:  a2dismod autoindex
  28  # (This disables the directory listing module completely)
  29  
  30  # DEFAULT DOCUMENT TO DISPLAY:
  31  # In case no specific file is requested, we want to fall back to index.php
  32  # In case some subdirectories are accessed, we want to display index.html
  33  <IfModule mod_dir.c>
  34      DirectoryIndex index.php index.html
  35  </IfModule>
  36  
  37  # CLEAN URLS:
  38  # The following will ensure you can pass extra params after the filename on the URL
  39  AcceptPathInfo    On
  40  
  41  # CATCH EVERYTHING INTO B2EVO: (even if no php file appears in the URL)
  42  # The following will allow you to have a blog running right off the site root,
  43  # using index.php as a stub but not showing it in the URLs.
  44  # This will add support for URLs like:  http://example.com/2006/08/29/post-title
  45  <IfModule mod_rewrite.c>
  46      RewriteEngine On
  47  
  48      # This line may be needed or not.
  49      # enabling this would prevent running in a subdir like /blog/index.php
  50      # RewriteBase /
  51  
  52      # Redirect any .htm .html or no-extension-file that's not an existing file or directory to index.php:
  53      RewriteCond %{REQUEST_FILENAME} !-d
  54      RewriteCond %{REQUEST_FILENAME} !-f
  55      RewriteRule ^[^.]+(\.(html|htm))?$ index.php [L]
  56  </IfModule>
