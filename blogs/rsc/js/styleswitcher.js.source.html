  15  function StyleSwitcher()
  16  {
  17      /**
  18       * Cross browser event handling for IE5+, NS6+ an Mozilla/Gecko
  19       * NOTE: taken/copied here from /rsc/js/functions.js
  20       * @author Scott Andrew
  21       */
  22      var addEvent = function( elm, evType, fn, useCapture )
  23      {
  24          if( elm.addEventListener )
  25          { // Standard & Mozilla way:
  26              elm.addEventListener( evType, fn, useCapture );
  27              return true;
  28          }
  29          else if( elm.attachEvent )
  30          { // IE way:
  31              var r = elm.attachEvent( 'on'+evType, fn );
  32              return r;
  33          }
  34          else
  35          { // "dirty" way (IE Mac for example):
  36              // Will overwrite any previous handler! :((
  37              elm['on'+evType] = fn;
  38              return false;
  39          }
  40      };
  41  
  42      var oThis = this;
  43  
  44      this.onload = function(e)
  45      {
  46          var cookie = oThis.readCookie("evo_style");
  47          var title = cookie ? cookie : oThis.getPreferredStyleSheet();
  48          oThis.setActiveStyleSheet(title);
  49      };
  50  
  51      addEvent( window, 'load', this.onload, false );
  52  
  53      this.onload(); // do it now already, so there's no "skin changing" after load on "larger" pages
  54  };

  57  StyleSwitcher.prototype.setActiveStyleSheet = function (title)
  58  {
  59      var i, a;
  60  
  61      var valid = false;
  62  
  63      // test if it's a valid one:
  64      for(var i=0; (a = document.getElementsByTagName("link")[i]); i++)
  65      {
  66          if(a.getAttribute("rel").indexOf("style") != -1 && a.getAttribute("title"))
  67          {
  68              if(a.getAttribute("title") == title)
  69              {
  70                  valid = true;
  71                  break;
  72              }
  73          }
  74      }
  75  
  76      if( ! valid )
  77          return false;
  78  
  79      // IE6 and IE7 need to disable all stylesheets first and then re-enable it..
  80      // ..at least when we call the onload event directly, too.
  81      // There is a bug with Konqueror (http://bugs.kde.org/135849) but it does not seem to apply here anymore?!
  82      for(i=0; (a = document.getElementsByTagName("link")[i]); i++)
  83      {
  84          if(a.getAttribute("rel").indexOf("style") != -1 && a.getAttribute("title"))
  85          {
  86              a.disabled = true;
  87              if(a.getAttribute("title") == title)
  88              {
  89                  a.disabled = false;
  90              }
  91          }
  92      }
  93  
  94      this.createCookie("evo_style", title, 365);
  95  
  96      return true;
  97  };

  99  StyleSwitcher.prototype.getActiveStyleSheet = function()
 100  {
 101      var i, a;
 102      for(i=0; (a = document.getElementsByTagName("link")[i]); i++)
 103      {
 104          if(a.getAttribute("rel").indexOf("style") != -1 && a.getAttribute("title") && !a.disabled)
 105          {
 106              return a.getAttribute("title");
 107          }
 108      }
 109      return null;
 110  };

 112  StyleSwitcher.prototype.getPreferredStyleSheet = function()
 113  {
 114      var i, a;
 115      for(i=0; (a = document.getElementsByTagName("link")[i]); i++)
 116      {
 117          if(a.getAttribute("rel").indexOf("style") != -1
 118              && a.getAttribute("rel").indexOf("alt") == -1
 119              && a.getAttribute("title") )
 120          {
 121              return a.getAttribute("title");
 122          }
 123      }
 124      return null;
 125  };

 127  StyleSwitcher.prototype.createCookie = function(name,value,days)
 128  {
 129      if (days)
 130      {
 131          var date = new Date();
 132          date.setTime(date.getTime()+(days*24*60*60*1000));
 133          var expires = "; expires="+date.toGMTString();
 134      }
 135      else expires = "";
 136      document.cookie = name+"="+value+expires+"; path=/";
 137  };

 139  StyleSwitcher.prototype.readCookie = function(name)
 140  {
 141      var nameEQ = name + "=";
 142      var ca = document.cookie.split(';');
 143      for(var i=0;i < ca.length;i++)
 144      {
 145          var c = ca[i];
 146          while (c.charAt(0)==' ')
 147          {
 148              c = c.substring(1,c.length);
 149          }
 150          if (c.indexOf(nameEQ) == 0)
 151          {
 152              return c.substring(nameEQ.length,c.length);
 153          }
 154      }
 155  
 156      return null;
 157  };
