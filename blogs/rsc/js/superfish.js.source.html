   8      $.fn.hoverIntent = function(f,g) {
   9          // default configuration options
  10          var cfg = {
  11              sensitivity: 7,
  12              interval: 100,
  13              timeout: 0
  14          };
  15          // override configuration options with user supplied object
  16          cfg = $.extend(cfg, g ? { over: f, out: g } : f );
  17  
  18          // instantiate variables
  19          // cX, cY = current X and Y position of mouse, updated by mousemove event
  20          // pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
  21          var cX, cY, pX, pY;
  22  
  23          // A private function for getting mouse position
  24          var track = function(ev) {
  25              cX = ev.pageX;
  26              cY = ev.pageY;
  27          };
  28  
  29          // A private function for comparing current and previous mouse position
  30          var compare = function(ev,ob) {
  31              ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
  32              // compare mouse positions to see if they've crossed the threshold
  33              if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) < cfg.sensitivity ) {
  34                  $(ob).unbind("mousemove",track);
  35                  // set hoverIntent state to true (so mouseOut can be called)
  36                  ob.hoverIntent_s = 1;
  37                  return cfg.over.apply(ob,[ev]);
  38              } else {
  39                  // set previous coordinates for next time
  40                  pX = cX; pY = cY;
  41                  // use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
  42                  ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
  43              }
  44          };
  45  
  46          // A private function for delaying the mouseOut function
  47          var delay = function(ev,ob) {
  48              ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
  49              ob.hoverIntent_s = 0;
  50              return cfg.out.apply(ob,[ev]);
  51          };
  52  
  53          // A private function for handling mouse 'hovering'
  54          var handleHover = function(e) {
  55              // next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
  56              var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
  57              while ( p && p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
  58              if ( p == this ) { return false; }
  59  
  60              // copy objects to be passed into t (required for event object to be passed in IE)
  61              var ev = jQuery.extend({},e);
  62              var ob = this;
  63  
  64              // cancel hoverIntent timer if it exists
  65              if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }
  66  
  67              // else e.type == "onmouseover"
  68              if (e.type == "mouseover") {
  69                  // set "previous" X and Y position based on initial entry point
  70                  pX = ev.pageX; pY = ev.pageY;
  71                  // update "current" X and Y position based on mousemove
  72                  $(ob).bind("mousemove",track);
  73                  // start polling interval (self-calling timeout) to compare mouse coordinates over time
  74                  if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}
  75  
  76              // else e.type == "onmouseout"
  77              } else {
  78                  // unbind expensive mousemove event
  79                  $(ob).unbind("mousemove",track);
  80                  // if hoverIntent state is true, then call the mouseOut function after the specified delay
  81                  if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
  82              }
  83          };
  84  
  85          // bind the function to the two event listeners
  86          return this.mouseover(handleHover).mouseout(handleHover);
  87      };
  88      
  89  })(jQuery);
