 142  	function makeSettings(i) {
 143          settings = $.extend({}, $.data(element, colorbox));
 144  
 145          for (i in settings) {
 146              if ($.isFunction(settings[i]) && i.substring(0, 2) !== 'on') { // checks to make sure the function isn't one of the callbacks, they will be handled at the appropriate time.
 147                  settings[i] = settings[i].call(element);
 148              }
 149          }

 151          settings.rel = settings.rel || element.rel || 'nofollow';
 152          settings.href = settings.href || $(element).attr('href');
 153          settings.title = settings.title || element.title;

 155          if (typeof settings.href === "string") {
 156              settings.href = $.trim(settings.href);
 157          }

 214  	function launch(target) {
 215          if (!closing) {
 216              
 217              element = target;
 218              
 219              makeSettings();
 220              
 221              $related = $(element);
 222              
 223              index = 0;
 224              
 225              if (settings.rel !== 'nofollow') {
 226                  $related = $('.' + boxElement).filter(function () {
 227                      var relRelated = $.data(this, colorbox).rel || this.rel;
 228                      return (relRelated === settings.rel);
 229                  });
 230                  index = $related.index(element);
 231                  
 232                  // Check direct calls to ColorBox.
 233                  if (index === -1) {
 234                      $related = $related.add(element);
 235                      index = $related.length - 1;
 236                  }
 237              }
 238              
 239              if (!open) {
 240                  open = active = true; // Prevents the page-change action from queuing up if the visitor holds down the left or right keys.
 241                  
 242                  $box.show();
 243                  
 244                  if (settings.returnFocus) {
 245                      try {
 246                          element.blur();
 247                          $(element).one(event_closed, function () {
 248                              try {
 249                                  this.focus();
 250                              } catch (e) {
 251                                  // do nothing
 252                              }
 253                          });
 254                      } catch (e) {
 255                          // do nothing
 256                      }
 257                  }
 258  
 259                  $overlay.css({"cursor": settings.overlayClose ? "pointer" : "auto"}).show();
 260  
 261                  // Opens inital empty ColorBox prior to content being loaded.
 262                  settings.w = setSize(settings.initialWidth, 'x');
 263                  settings.h = setSize(settings.initialHeight, 'y');
 264                  publicMethod.position();
 265                  
 266                  if (isIE6) {
 267                      $window.bind('resize.' + event_ie6 + ' scroll.' + event_ie6, function () {
 268                          $overlay.css({width: $window.width(), height: $window.height(), top: $window.scrollTop(), left: $window.scrollLeft()});
 269                      }).trigger('resize.' + event_ie6);
 270                  }
 271  
 272                  trigger(event_open, settings.onOpen);
 273  
 274                  $groupControls.add($title).hide();
 275  
 276                  $close.html(settings.close).show();
 277              }
 278  
 279              publicMethod.load(true);
 280          }

 417      publicMethod.position = function (speed, loadedCallback) {
 418          var top = 0, left = 0;
 419          
 420          $window.unbind('resize.' + prefix);
 421          
 422          // remove the modal so that it doesn't influence the document width/height        
 423          $box.hide();
 424          
 425          if (settings.fixed && !isIE6) {
 426              $box.css({position: 'fixed'});
 427          } else {
 428              top = $window.scrollTop();
 429              left = $window.scrollLeft();
 430              $box.css({position: 'absolute'});
 431          }
 432          
 433          // keeps the top and left positions within the browser's viewport.
 434          if (settings.right !== false) {
 435              left += Math.max($window.width() - settings.w - loadedWidth - interfaceWidth - setSize(settings.right, 'x'), 0);
 436          } else if (settings.left !== false) {
 437              left += setSize(settings.left, 'x');
 438          } else {
 439              left += Math.round(Math.max($window.width() - settings.w - loadedWidth - interfaceWidth, 0) / 2);
 440          }
 441          
 442          if (settings.bottom !== false) {
 443              top += Math.max(document.documentElement.clientHeight - settings.h - loadedHeight - interfaceHeight - setSize(settings.bottom, 'y'), 0);
 444          } else if (settings.top !== false) {
 445              top += setSize(settings.top, 'y');
 446          } else {
 447              top += Math.round(Math.max(document.documentElement.clientHeight - settings.h - loadedHeight - interfaceHeight, 0) / 2);
 448          }
 449          
 450          $box.show();
 451          
 452          // setting the speed to 0 to reduce the delay between same-sized content.
 453          speed = ($box.width() === settings.w + loadedWidth && $box.height() === settings.h + loadedHeight) ? 0 : speed || 0;
 454          
 455          // this gives the wrapper plenty of breathing room so it's floated contents can move around smoothly,
 456          // but it has to be shrank down around the size of div#colorbox when it's done.  If not,
 457          // it can invoke an obscure IE bug when using iframes.
 458          $wrap[0].style.width = $wrap[0].style.height = "9999px";

 460  		function modalDimensions(that) {
 461              // loading overlay height has to be explicitly set for IE6.
 462              $content[0].style.width = that.style.width;
 463              $loadingOverlay[0].style.height = $loadingOverlay[1].style.height = $content[0].style.height = that.style.height;
 464          }
 465          
 466          $box.dequeue().animate({width: settings.w + loadedWidth, height: settings.h + loadedHeight, top: top, left: left}, {
 467              duration: speed,
 468              complete: function () {
 469                  modalDimensions(this);
 470                  
 471                  active = false;
 472                  
 473                  // shrink the wrapper down to exactly the size of colorbox to avoid a bug in IE's iframe implementation.
 474                  $wrap[0].style.width = (settings.w + loadedWidth + interfaceWidth) + "px";
 475                  $wrap[0].style.height = (settings.h + loadedHeight + interfaceHeight) + "px";
 476                  
 477                  if (loadedCallback) {
 478                      loadedCallback();
 479                  }
 480                  
 481                  setTimeout(function(){  // small delay before binding onresize due to an IE8 bug.
 482                      $window.bind('resize.' + prefix, publicMethod.position);
 483                  }, 1);
 484              },
 485              step: function () {
 486                  modalDimensions(this);
 487              }
 488          });
 489  
 490          if( $loaded.width() < 380 )
 491          {
 492              $current.hide();
 493              $slideshow.hide();
 494          }
