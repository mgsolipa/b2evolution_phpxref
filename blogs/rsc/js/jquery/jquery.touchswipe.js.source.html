 535              if (SUPPORTS_TOUCH) {
 536                  // get the total number of fingers touching the screen
 537                  fingerCount = event.touches.length;
 538              }
 539              //Else this is the desktop, so stop the browser from dragging the image
 540              else {
 541                  jqEvent.preventDefault(); //call this on jq event so we are cross browser
 542              }
 543  
 544              //clear vars..
 545              distance = 0;
 546              direction = null;
 547              pinchDirection=null;
 548              duration = 0;
 549              startTouchesDistance=0;
 550              endTouchesDistance=0;
 551              pinchZoom = 1;
 552              pinchDistance = 0;
 553              fingerData=createAllFingerData();
 554              maximumsMap=createMaximumsData();
 555              cancelMultiFingerRelease();
 556  
 557              
 558              // check the number of fingers is what we are looking for, or we are capturing pinches
 559              if (!SUPPORTS_TOUCH || (fingerCount === options.fingers || options.fingers === ALL_FINGERS) || hasPinches()) {
 560                  // get the coordinates of the touch
 561                  createFingerData( 0, evt );
 562                  startTime = getTimeStamp();
 563                  
 564                  if(fingerCount==2) {
 565                      //Keep track of the initial pinch distance, so we can calculate the diff later
 566                      //Store second finger data as start
 567                      createFingerData( 1, event.touches[1] );
 568                      startTouchesDistance = endTouchesDistance = calculateTouchesDistance(fingerData[0].start, fingerData[1].start);
 569                  }
 570                  
 571                  if (options.swipeStatus || options.pinchStatus) {
 572                      ret = triggerHandler(event, phase);
 573                  }
 574              }
 575              else {
 576                  //A touch with more or less than the fingers we are looking for, so cancel
 577                  ret = false; 
 578              }
 579  
 580              //If we have a return value from the users handler, then return and cancel
 581              if (ret === false) {
 582                  phase = PHASE_CANCEL;
 583                  triggerHandler(event, phase);
 584                  return ret;
 585              }
 586              else {
 587                  setTouchInProgress(true);
 588              }
 589  
 590              return null;
 591          };

 595          /**
 596          * Event handler for a touch move event. 
 597          * If we change fingers during move, then cancel the event
 598          * @inner
 599          * @param {object} jqEvent The normalised jQuery event object.
 600          */
 601  		function touchMove(jqEvent) {
 602              
 603              //As we use Jquery bind for events, we need to target the original event object
 604              //If these events are being programmatically triggered, we don't have an original event object, so use the Jq one.
 605              var event = jqEvent.originalEvent ? jqEvent.originalEvent : jqEvent;
 606              
 607              //If we are ending, cancelling, or within the threshold of 2 fingers being released, don't track anything..
 608              if (phase === PHASE_END || phase === PHASE_CANCEL || inMultiFingerRelease())
 609                  return;
 610  
 611              var ret,
 612                  evt = SUPPORTS_TOUCH ? event.touches[0] : event;
 613              
 614  
 615              //Update the  finger data 
 616              var currentFinger = updateFingerData(evt);
 617              endTime = getTimeStamp();
 618              
 619              if (SUPPORTS_TOUCH) {
 620                  fingerCount = event.touches.length;
 621              }
 622  
 623              phase = PHASE_MOVE;
 624  
 625              //If we have 2 fingers get Touches distance as well
 626              if(fingerCount==2) {
 627                  
 628                  //Keep track of the initial pinch distance, so we can calculate the diff later
 629                  //We do this here as well as the start event, in case they start with 1 finger, and the press 2 fingers
 630                  if(startTouchesDistance==0) {
 631                      //Create second finger if this is the first time...
 632                      createFingerData( 1, event.touches[1] );
 633                      
 634                      startTouchesDistance = endTouchesDistance = calculateTouchesDistance(fingerData[0].start, fingerData[1].start);
 635                  } else {
 636                      //Else just update the second finger
 637                      updateFingerData(event.touches[1]);
 638                  
 639                      endTouchesDistance = calculateTouchesDistance(fingerData[0].end, fingerData[1].end);
 640                      pinchDirection = calculatePinchDirection(fingerData[0].end, fingerData[1].end);
 641                  }
 642                  
 643                  
 644                  pinchZoom = calculatePinchZoom(startTouchesDistance, endTouchesDistance);
 645                  pinchDistance = Math.abs(startTouchesDistance - endTouchesDistance);
 646              }
 647              
 648              
 649              if ( (fingerCount === options.fingers || options.fingers === ALL_FINGERS) || !SUPPORTS_TOUCH || hasPinches() ) {
 650                  
 651                  direction = calculateDirection(currentFinger.start, currentFinger.end);
 652                  
 653                  //Check if we need to prevent default event (page scroll / pinch zoom) or not
 654                  validateDefaultEvent(jqEvent, direction);
 655  
 656                  //Distance and duration are all off the main finger
 657                  distance = calculateDistance(currentFinger.start, currentFinger.end);
 658                  duration = calculateDuration();
 659  
 660                  //Cache the maximum distance we made in this direction
 661                  setMaxDistance(direction, distance);
 662  
 663  
 664                  if (options.swipeStatus || options.pinchStatus) {
 665                      ret = triggerHandler(event, phase);
 666                  }
 667                  
 668                  
 669                  //If we trigger end events when threshold are met, or trigger events when touch leaves element
 670                  if(!options.triggerOnTouchEnd || options.triggerOnTouchLeave) {
 671                      
 672                      var inBounds = true;
 673                      
 674                      //If checking if we leave the element, run the bounds check (we can use touchleave as its not supported on webkit)
 675                      if(options.triggerOnTouchLeave) {
 676                          var bounds = getbounds( this );
 677                          inBounds = isInBounds( currentFinger.end, bounds );
 678                      }
 679                      
 680                      //Trigger end handles as we swipe if thresholds met or if we have left the element if the user has asked to check these..
 681                      if(!options.triggerOnTouchEnd && inBounds) {
 682                          phase = getNextPhase( PHASE_MOVE );
 683                      } 
 684                      //We end if out of bounds here, so set current phase to END, and check if its modified 
 685                      else if(options.triggerOnTouchLeave && !inBounds ) {
 686                          phase = getNextPhase( PHASE_END );
 687                      }
 688                          
 689                      if(phase==PHASE_CANCEL || phase==PHASE_END)    {
 690                          triggerHandler(event, phase);
 691                      }                
 692                  }
 693              }
 694              else {
 695                  phase = PHASE_CANCEL;
 696                  triggerHandler(event, phase);
 697              }
 698  
 699              if (ret === false) {
 700                  phase = PHASE_CANCEL;
 701                  triggerHandler(event, phase);
 702              }
 703          }

 707          /**
 708          * Event handler for a touch end event. 
 709          * Calculate the direction and trigger events
 710          * @inner
 711          * @param {object} jqEvent The normalised jQuery event object.
 712          */
 713  		function touchEnd(jqEvent) {
 714              //As we use Jquery bind for events, we need to target the original event object
 715              var event = jqEvent.originalEvent;
 716                  
 717  
 718              //If we are still in a touch with another finger return
 719              //This allows us to wait a fraction and see if the other finger comes up, if it does within the threshold, then we treat it as a multi release, not a single release.
 720              if (SUPPORTS_TOUCH) {
 721                  if(event.touches.length>0) {
 722                      startMultiFingerRelease();
 723                      return true;
 724                  }
 725              }
 726              
 727              //If a previous finger has been released, check how long ago, if within the threshold, then assume it was a multifinger release.
 728              //This is used to allow 2 fingers to release fractionally after each other, whilst maintainig the event as containg 2 fingers, not 1
 729              if(inMultiFingerRelease()) {    
 730                  fingerCount=previousTouchFingerCount;
 731              }    
 732                   
 733              //call this on jq event so we are cross browser 
 734              jqEvent.preventDefault(); 
 735              
 736              //Set end of swipe
 737              endTime = getTimeStamp();
 738              
 739              //Get duration incase move was never fired
 740              duration = calculateDuration();
 741              
 742              //If we trigger handlers at end of swipe OR, we trigger during, but they didnt trigger and we are still in the move phase
 743              if(didSwipeBackToCancel()) {
 744                  phase = PHASE_CANCEL;
 745                  triggerHandler(event, phase);
 746              } else if (options.triggerOnTouchEnd || (options.triggerOnTouchEnd == false && phase === PHASE_MOVE)) {
 747                  phase = PHASE_END;
 748                  triggerHandler(event, phase);
 749              }
 750              //Special cases - A tap should always fire on touch end regardless,
 751              //So here we manually trigger the tap end handler by itself
 752              //We dont run trigger handler as it will re-trigger events that may have fired already
 753              else if (!options.triggerOnTouchEnd && hasTap()) {
 754                  //Trigger the pinch events...
 755                  phase = PHASE_END;
 756                  triggerHandlerForGesture(event, phase, TAP);
 757              }
 758              else if (phase === PHASE_MOVE) {
 759                  phase = PHASE_CANCEL;
 760                  triggerHandler(event, phase);
 761              }
 762  
 763              setTouchInProgress(false);
 764  
 765              return null;
 766          }

 827           * Checks if the time and distance thresholds have been met, and if so then the appropriate handlers are fired.
 828           */
 829  		function getNextPhase(currentPhase) {
 830              
 831              var nextPhase = currentPhase;
 832              
 833              // Ensure we have valid swipe (under time and over distance  and check if we are out of bound...)
 834              var validTime = validateSwipeTime();
 835              var validDistance = validateSwipeDistance();
 836              var didCancel = didSwipeBackToCancel();
 837                          
 838              //If we have exceeded our time, then cancel    
 839              if(!validTime || didCancel) {
 840                  nextPhase = PHASE_CANCEL;
 841              }
 842              //Else if we are moving, and have reached distance then end
 843              else if (validDistance && currentPhase == PHASE_MOVE && (!options.triggerOnTouchEnd || options.triggerOnTouchLeave) ) {
 844                  nextPhase = PHASE_END;
 845              } 
 846              //Else if we have ended by leaving and didn't reach distance, then cancel
 847              else if (!validDistance && currentPhase==PHASE_END && options.triggerOnTouchLeave) {
 848                  nextPhase = PHASE_CANCEL;
 849              }
 850              
 851              return nextPhase;
 852          }

 855          /**
 856          * Trigger the relevant event handler
 857          * The handlers are passed the original event, the element that was swiped, and in the case of the catch all handler, the direction that was swiped, "left", "right", "up", or "down"
 858          * @param {object} event the original event object
 859          * @param {string} phase the phase of the swipe (start, end cancel etc) {@link $.fn.swipe.phases}
 860          * @inner
 861          */
 862  		function triggerHandler(event, phase) {
 863              
 864              var ret = undefined;
 865              
 866              // SWIPE GESTURES
 867              if(didSwipe() || hasSwipes()) { //hasSwipes as status needs to fire even if swipe is invalid
 868                  //Trigger the swipe events...
 869                  ret = triggerHandlerForGesture(event, phase, SWIPE);
 870              } 
 871              
 872              // PINCH GESTURES (if the above didn't cancel)
 873              else if((didPinch() || hasPinches()) && ret!==false) {
 874                  //Trigger the pinch events...
 875                  ret = triggerHandlerForGesture(event, phase, PINCH);
 876              }
 877              
 878              // CLICK / TAP (if the above didn't cancel)
 879              if(didDoubleTap() && ret!==false) {
 880                  //Trigger the tap events...
 881                  ret = triggerHandlerForGesture(event, phase, DOUBLE_TAP);
 882              }
 883              
 884              // CLICK / TAP (if the above didn't cancel)
 885              else if(didLongTap() && ret!==false) {
 886                  //Trigger the tap events...
 887                  ret = triggerHandlerForGesture(event, phase, LONG_TAP);
 888              }
 889  
 890              // CLICK / TAP (if the above didn't cancel)
 891              else if(didTap() && ret!==false) {
 892                  //Trigger the tap event..
 893                  ret = triggerHandlerForGesture(event, phase, TAP);
 894              }
 895              
 896              
 897              
 898              // If we are cancelling the gesture, then manually trigger the reset handler
 899              if (phase === PHASE_CANCEL) {
 900                  touchCancel(event);
 901              }
 902              
 903              // If we are ending the gesture, then manually trigger the reset handler IF all fingers are off
 904              if(phase === PHASE_END) {
 905                  //If we support touch, then check that all fingers are off before we cancel
 906                  if (SUPPORTS_TOUCH) {
 907                      if(event.touches.length==0) {
 908                          touchCancel(event);    
 909                      }
 910                  } 
 911                  else {
 912                      touchCancel(event);
 913                  }
 914              }
 915                      
 916              return ret;
 917          }

 938                  //Trigger the event...

 948

 970                                  ret = options.swipeLeft.call($element, event, direction, distance, duration, fingerCount);
 971                              }
 972                              break;
 973      
 974                          case RIGHT:
 975                              //Trigger the event
 976                              $element.trigger('swipeRight', [direction, distance, duration, fingerCount]);
 977                      
 978                              //Fire the callback
 979                              if (options.swipeRight) {
 980                                  ret = options.swipeRight.call($element, event, direction, distance, duration, fingerCount);
 981                              }
 982                              break;
 983      
 984                          case UP:
 985                              //Trigger the event
 986                              $element.trigger('swipeUp', [direction, distance, duration, fingerCount]);
 987                      
 988                              //Fire the callback
 989                              if (options.swipeUp) {
 990                                  ret = options.swipeUp.call($element, event, direction, distance, duration, fingerCount);

 992                              break;

 996                              $element.trigger('swipeDown', [direction, distance, duration, fingerCount]);

 998                              //Fire the callback

1016                      //If the status cancels, then dont run the subsequent event handlers..

1019

1021                      
1022                      switch (pinchDirection) {
1023                          case IN:
1024                              //Trigger the event
1025                              $element.trigger('pinchIn', [pinchDirection || null, pinchDistance || 0, duration || 0, fingerCount, pinchZoom]);
1026                      
1027                              //Fire the callback
1028                              if (options.pinchIn) {
1029                                  ret = options.pinchIn.call($element, event, pinchDirection || null, pinchDistance || 0, duration || 0, fingerCount, pinchZoom);
1030                              }
1031                              break;
1032                          
1033                          case OUT:
1034                              //Trigger the event
1035                              $element.trigger('pinchOut', [pinchDirection || null, pinchDistance || 0, duration || 0, fingerCount, pinchZoom]);
1036                      
1037                              //Fire the callback
1038                              if (options.pinchOut) {

1041                              break;    
1042                      }
1043                  }
1044              }
1045              
1046  
1047  
1048                  
1049                  
1050              if(gesture==TAP) {

1065                              doubleTapStartTime=null;
1066                              //Trigger the event
1067                              $element.trigger('tap', [event.target]);
1068

1072                                  ret = options.tap.call($element, event, event.target);
1073                              }
1074                          }, this), options.doubleTapThreshold );
1075

1078                          
1079                          //Trigger the event
1080                          $element.trigger('tap', [event.target]);
1081  
1082                          
1083                          //Fire the callback
1084                          if(options.tap) {
1085                              ret = options.tap.call($element, event, event.target);
1086                          }
1087                      }
1088                  }

1091              else if (gesture==DOUBLE_TAP) {

1118                          ret = options.longTap.call($element, event, event.target);

1122

1124          }
