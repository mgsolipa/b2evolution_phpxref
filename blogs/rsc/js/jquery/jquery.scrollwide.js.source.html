   1  /**

   2   * jQuery Wide Scroll

   3   * ---------------------------------------------------------------------------------

   4   *

   5   * jQuery Wide Scroll is a plugin that generates a customizable horizontal scrolling.

   6   *

   7   * @version 1.0

   8   * @author  Yuriy Bakhtin

   9   *

  10   * Usage with default values:

  11   * ---------------------------------------------------------------------------------

  12   * $('#div').scrollWide();

  13   *

  14   * <div id="div"></div>

  15   *

  16   * Usage with custom values:

  17   * ---------------------------------------------------------------------------------

  18   * 

  19   * $('#div').scrollWide( {

  20   *   scroll_step: 80, // Step of scrolling in percents of the display width 1%-100%

  21   *   scroll_time: 500, // Time to scroll one step (ms)

  22   *   width: 20, // Width of the scroll controls

  23   *   margin: 5, // Space between wide div and the scroll controls

  24   *   class_name: 'wscontrol',

  25   *   class_name_over: 'wsover',

  26   *   class_name_click: 'wsclick',

  27   *   class_name_left: 'wsleft',

  28   *   class_name_right: 'wsright',

  29  * } );

  30   */
  31  
  32  ;(function($) {
  33  
  34      $.fn.scrollWide = function( settings )
  35      {
  36          if( this.length == 0 )
  37          {
  38              return;
  39          }
  40          else if( this.length > 1 )
  41          {
  42              return this.each( function()
  43              {
  44                  $.fn.scrollWide.apply( $( this ), [settings] );
  45              } );
  46          }
  47  
  48          var opt = $.extend( {}, $.fn.scrollWide.defaults, settings ),
  49              $this = $( this ),
  50              id    = this.attr( 'id' );
  51  
  52          if( !this.HasScrollBarHorizontal() )
  53          {    // No horizontal scrollbar
  54              return $this;
  55          }
  56  
  57          if( id === undefined )
  58          {
  59              id = 'scrollWide-' + $this.index();
  60              $this.attr( 'id', id );
  61          }
  62  
  63          $this.data( 'options', opt );
  64  
  65          initialize( $this );
  66  
  67          return $this;
  68      };
  69  
  70      $.fn.HasScrollBarHorizontal = function()
  71      {
  72          var _elm = $(this)[0];
  73          var _hasScrollBar = false;
  74          if( $(this).width() < _elm.scrollWidth ) /* ( _elm.clientHeight < _elm.scrollHeight ) - Vertical Scroll Bar */
  75          {
  76              _hasScrollBar = true;
  77          }
  78          return _hasScrollBar;
  79      }

  81  	function initialize( obj )
  82      {
  83          var options = obj.data('options'),
  84              obj_orig_width = obj.width();
  85  
  86          obj.css( {
  87              width: obj_orig_width - options.width - options.margin,
  88              marginRight: options.width + options.margin + 2
  89          } );
  90  
  91          var scroll_right = create_control( obj, 'right' );
  92          var scroll_left = create_control( obj, 'left' );
  93  
  94          obj.bind( 'scroll', function()
  95          {    // Do these actions when main div is scrolling
  96              var maxScrollLeft = this.scrollWidth - this.clientWidth;
  97              if( $( this ).scrollLeft() == maxScrollLeft )
  98              {    // Hide right scroll control if scrollbar is located in the right position
  99                  scroll_right.hide();
 100              }
 101              else
 102              {    // Show right scroll control
 103                  scroll_right.show();
 104              }
 105  
 106              if( $( this ).scrollLeft() == 0 )
 107              {    // Hide left scroll control if scrollbar is located in the left position
 108                  scroll_left.hide();
 109                  obj.css( 'marginLeft', '0' );
 110              }
 111              else
 112              {    // Show left scroll control
 113                  scroll_left.show();
 114                  obj.css( { // Move main div to clear a space for the left scroll control
 115                      position: 'relative',
 116                      marginLeft: options.width + options.margin + 2
 117                  } );
 118              }
 119  
 120              if( scroll_right.is(':visible') && scroll_left.is(':visible') )
 121              {    // When left & right scroll controls are visible
 122                  obj.css( 'width', obj_orig_width - 2 * ( options.width + options.margin ) - 2 );
 123              }
 124              else
 125              {    // Only one control is visible
 126                  obj.css( 'width', obj_orig_width - options.width - options.margin - 1 );
 127              }
 128          } );
 129  
 130          $( [ scroll_right.get(0), scroll_left.get(0) ] ).bind( 'mouseover', function()
 131          {    // Change style of the controls elements on mouseover event
 132              $( '.' + options.class_name ).removeClass( options.class_name_over );
 133              $( this ).addClass( options.class_name_over );
 134              $( '.' + options.class_name ).removeClass( options.class_name_click );
 135          } );
 136  
 137          $( [ scroll_right.get(0), scroll_left.get(0) ] ).bind( 'mouseout', function()
 138          {    // Change style of the controls elements on mouseout event
 139              $( this ).removeClass( options.class_name_over );
 140              $( '.' + options.class_name ).removeClass( options.class_name_click );
 141          } );
 142  
 143          $( [ scroll_right.get(0), scroll_left.get(0) ] ).bind( 'mousedown mouseup', function()
 144          {    // Change style of the controls elements on mousedown & mouseup events
 145              $( this ).toggleClass( options.class_name_click );
 146          } );
 147  
 148          scroll_right.bind( 'click', function()
 149          {    // Bind action to scroll to the right
 150              var scroll_value = obj.scrollLeft() + Math.floor( obj.width() * options.scroll_step / 100 );
 151              obj.animate( {scrollLeft: scroll_value + 'px'}, options.scroll_time );
 152          } );
 153  
 154          scroll_left.bind( 'click', function()
 155          {    // Bind action to scroll to the left
 156              var scroll_value = obj.scrollLeft() - Math.floor( obj.width() * options.scroll_step / 100 );
 157              obj.animate( { scrollLeft: scroll_value + 'px' }, options.scroll_time );
 158          } );
 159      }

 161  	function create_control( obj, type )
 162      {
 163          var options = obj.data('options'),
 164              control_id = obj.attr( 'id' ) + '-' + type;
 165  
 166          var control_class_name = options.class_name_left;
 167          if( type == 'right' )
 168          {
 169              control_class_name = options.class_name_right;
 170          }
 171  
 172          // Insert new control element

 173          obj.after( '<div id="' + control_id + '" class="' + options.class_name + ' ' + control_class_name + '"></div>' );
 174  
 175          var scroll_control_obj = $( '#' + control_id );
 176  
 177          var control_css_left = obj.offset().left;
 178          if( type == 'right' )
 179          {
 180              control_css_left += obj.outerWidth() + options.margin;
 181          }
 182          else
 183          {    // left control
 184              if( $(this).scrollLeft() == 0 )
 185              {    // Hide left control
 186                  scroll_control_obj.hide();
 187              }
 188          }
 189  
 190          scroll_control_obj.css( { // Set position and sizes for new created scroll control element
 191              position: 'absolute',
 192              left: control_css_left,
 193              top: obj.offset().top,
 194              width: options.width,
 195              height: obj.outerHeight() - 2
 196          } );
 197  
 198          return scroll_control_obj;
 199      }
