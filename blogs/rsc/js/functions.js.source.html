   9   * Cross browser event handling for IE5+, NS6+ an Mozilla/Gecko
  10   * @obsolete Use jQuery instead
  11   * @author Scott Andrew

  13  function addEvent( elm, evType, fn, useCapture )
  14  {
  15      if( elm.addEventListener )
  16      { // Standard & Mozilla way:
  17          elm.addEventListener( evType, fn, useCapture );
  18          return true;
  19      }
  20      else if( elm.attachEvent )
  21      { // IE way:
  22          var r = elm.attachEvent( 'on'+evType, fn );
  23          return r;
  24      }
  25      else
  26      { // "dirty" way (IE Mac for example):
  27          // Will overwrite any previous handler! :((
  28          elm['on'+evType] = fn;
  29          return false;
  30      }
  31  }

  37  function pop_up_window( href, target, width, height, params )
  38  {
  39      if( typeof(width) == 'undefined' )
  40      {
  41          width = 750;
  42      }
  43  
  44      if( typeof(height) == 'undefined' )
  45      {
  46          height = 550;
  47      }
  48  
  49      var left = (screen.width - width) / 2;
  50      var top = (screen.height - height) / 2;
  51  
  52      if( typeof(params) == 'undefined' )
  53      {
  54          params = 'scrollbars=yes, status=yes, resizable=yes, menubar=yes';
  55      }
  56  
  57      params = 'width=' + width + ', height=' + height + ', ' + 'left=' + left + ', top=' + top + ', ' + params;
  58  
  59      // Open window:
  60      opened = window.open( href, target, params );
  61  
  62      // Bring to front!
  63      opened.focus();
  64  
  65      if( typeof(openedWindows) == 'undefined' )
  66      {
  67          openedWindows = new Array(opened);
  68      }
  69      else
  70      {
  71          openedWindows.push(opened);
  72      }
  73  
  74      // Tell the caller there is no need to process href="" :
  75      return false;
  76  }

  89  function toggle_clickopen( id, hide, displayVisible )
  90  {
  91      if( !( clickdiv = document.getElementById( 'clickdiv_'+id ) )
  92              || !( clickimg = document.getElementById( 'clickimg_'+id ) ) )
  93      {
  94          alert( 'ID '+id+' not found!' );
  95          return false;
  96      }
  97  
  98      if( typeof(hide) == 'undefined' )
  99      {
 100          hide = document.getElementById( 'clickdiv_'+id ).style.display != 'none';
 101      }
 102  
 103      if( typeof(displayVisible) == 'undefined' )
 104      {
 105          displayVisible = ''; // setting it to "empty" is the default for an element's display CSS attribute
 106      }
 107  
 108      if( hide )
 109      {
 110          clickdiv.style.display = 'none';
 111          clickimg.style.backgroundPosition = bgxy_expand;
 112      }
 113      else
 114      {
 115          clickdiv.style.display = displayVisible;
 116          clickimg.style.backgroundPosition = bgxy_collapse;
 117      }
 118  
 119      return false;
 120  }

 129  /**
 130   * Textarea insertion code.
 131   *
 132   * @var element
 133   * @var text
 134   * @var text
 135   * @var boolean
 136   * @var document (needs only be passed from a popup window as window.opener.document)
 137   */
 138  function textarea_wrap_selection( myField, before, after, replace, target_document )
 139  {
 140      target_document = target_document || document;
 141  
 142      var hook_params = {
 143          'element': myField,
 144          'before': before,
 145          'after': after,
 146          'replace': replace,
 147          'target_document': target_document
 148      };
 149  
 150      // First try, if a JavaScript callback is registered to handle this.
 151      // E.g. the tinymce_plugin uses registers "wrap_selection_for_itemform_post_content"
 152      //      to replace the (non-)selection
 153      if( b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
 154      {
 155          return;
 156      }
 157  
 158      if( window.opener && ( typeof window.opener != "undefined" ) )
 159      {
 160          try
 161          { // Try find object 'b2evo_Callbacks' on window.opener to avoid halt error when page was opened from other domain
 162              if( window.opener.b2evo_Callbacks &&
 163             ( typeof window.opener.b2evo_Callbacks != "undefined" ) &&
 164             window.opener.b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
 165              { // callback in opener document (e.g. "Files" popup)
 166                  return;
 167              }
 168          }
 169          catch( e )
 170          { // Catch an error of the cross-domain restriction
 171              // Ignore this error because it dies when browser has no permission to access to other domain windows
 172          }
 173      }
 174  
 175      if( window.parent
 176          && ( typeof window.parent != "undefined" )
 177          && window.parent.b2evo_Callbacks
 178          && ( typeof window.parent.b2evo_Callbacks != "undefined" ) )
 179      {    // callback in parent document (e.g. "Links" iframe)
 180          if( window.parent.b2evo_Callbacks.trigger_callback( "wrap_selection_for_"+myField.id, hook_params ) )
 181          {
 182              return;
 183          }
 184      }
 185  
 186      // Basic handling:
 187      if(target_document.selection)
 188      { // IE support:
 189          myField.focus();
 190          sel = target_document.selection.createRange();
 191          if( replace )
 192          {
 193              sel.text = before + after;
 194          }
 195          else
 196          {
 197              sel.text = before + sel.text + after;
 198          }
 199          myField.focus();
 200      }
 201      else if (myField.selectionStart || myField.selectionStart == '0')
 202      { // MOZILLA/NETSCAPE support:
 203          var startPos = myField.selectionStart;
 204          var endPos = myField.selectionEnd;
 205          var cursorPos;
 206  
 207          var scrollTop, scrollLeft;
 208          if( myField.type == 'textarea' && typeof myField.scrollTop != 'undefined' )
 209          { // remember old position
 210              scrollTop = myField.scrollTop;
 211              scrollLeft = myField.scrollLeft;
 212          }
 213  
 214          if( replace )
 215          {
 216              myField.value = myField.value.substring( 0, startPos)
 217                  + before
 218                  + after
 219                  + myField.value.substring( endPos, myField.value.length);
 220              cursorPos = startPos + before.length + after.length;
 221          }
 222          else
 223          {
 224              myField.value = myField.value.substring( 0, startPos)
 225                  + before
 226                  + myField.value.substring(startPos, endPos)
 227                  + after
 228                  + myField.value.substring( endPos, myField.value.length);
 229              cursorPos = endPos + before.length + after.length;
 230          }
 231  
 232          if( typeof scrollTop != 'undefined' )
 233          { // scroll to old position
 234              myField.scrollTop = scrollTop;
 235              myField.scrollLeft = scrollLeft;
 236          }
 237  
 238          myField.focus();
 239          myField.selectionStart = cursorPos;
 240          myField.selectionEnd = cursorPos;
 241      }
 242      else
 243      { // Default browser support:
 244          myField.value += before + after;
 245          myField.focus();
 246      }
 247  }

 249  /**
 250   * Open or close a filter area (by use of CSS style).
 251   *
 252   * You have to define a div with id clickdiv_<ID> and a img with clickimg_<ID>,
 253   * where <ID> is the first param to the function.

 255   * @param string html id of the element to toggle

 258  function toggle_filter_area( filter_name )
 259  {
 260      // Find objects to toggle:
 261      if( !( clickdiv = document.getElementById( 'clickdiv_'+filter_name ) )
 262              || !( clickimg = document.getElementById( 'clickimg_'+filter_name ) ) )
 263      {
 264          alert( 'ID '+filter_name+' not found!' );
 265          return false;
 266      }
 267  
 268      // Determine if we want to show or to hide (based on current state).
 269      hide = document.getElementById( 'clickdiv_'+filter_name ).style.display != 'none';
 270  
 271      if( hide )
 272      {    // Hide/collapse filters:
 273          clickdiv.style.display = 'none';
 274          clickimg.style.backgroundPosition = bgxy_expand;
 275          jQuery.post( htsrv_url+'anon_async.php?action=collapse_filter&target='+filter_name );
 276      }
 277      else
 278      {    // Show/expand filters
 279          clickdiv.style.display = 'block';
 280          clickimg.style.backgroundPosition = bgxy_collapse;
 281          jQuery.post( htsrv_url+'anon_async.php?action=expand_filter&target='+filter_name );
 282      }
 283  
 284      return false;
 285  }

 288  /*
 289   * Javascript callback handling, for helping plugins to interact in Javascript.
 290   *
 291   * This is, so one plugin (e.g. the tinymce_plugin) can say that it handles insertion of raw
 292   * content into a specific element ("itemform_post_content" in this case):
 293   *
 294   * <code>
 295   * if( typeof b2evo_Callbacks == "object" )
 296   * { // add a callback, that lets us insert the
 297   *   b2evo_Callbacks.register_callback( "wrap_selection_for_itemform_post_content", function(value) {
 298   *       tinyMCE.execCommand( "mceInsertRawHTML", false, value );
 299   *       return true;
 300   *     } );
 301   * }
 302   * </code>
 303   *
 304   * and others (e.g. the smilies_plugin or the youtube_plugin) should first try to use this
 305   * callback to insert the HTML:
 306   *
 307   * if( typeof b2evo_Callbacks == 'object' )
 308   * { // see if there's a callback registered that should handle this:
 309   *   if( b2evo_Callbacks.trigger_callback("wrap_selection_for_"+b2evoCanvas.id, tag) )
 310   *   {
 311   *     return;
 312   *   }
 313   * }
 314   */
 315  function b2evo_Callbacks() {
 316      this.eventHandlers = new Array();
 317  };
 318  
 319  b2evo_Callbacks.prototype = {
 320      register_callback : function(event, f) {
 321          if( typeof this.eventHandlers[event] == "undefined" )
 322          {
 323              this.eventHandlers[event] = new Array();
 324          }
 325          this.eventHandlers[event][this.eventHandlers[event].length] = f;
 326      },
 327  
 328      /**
 329       * @param String event name
 330       * @param mixed argument1
 331       * @param mixed argument2
 332       * ...
 333       * @return boolean true, if any callback returned true
 334       *                 null, if no callback registered
 335       */
 336      trigger_callback : function(event, args) {
 337  
 338          if( typeof this.eventHandlers[event] == "undefined" )
 339          {
 340              return null;
 341          }
 342  
 343          var r = false;
 344  
 345          // copy arguments and build function param string for eval():
 346          var cb_args = '';
 347          var cb_arguments = arguments;
 348          for( var i = 1; i < arguments.length; i++ ) {
 349              cb_args += "cb_arguments[" + i + "], ";
 350          }
 351          if( cb_args.length )
 352          { // remove last ", ":
 353              cb_args = cb_args.substring( 0, cb_args.length - 2 );
 354          }
 355  
 356          // eval() for each registered callback:
 357          for( var i = 0; i < this.eventHandlers[event].length; i++ )
 358          {
 359              var f = this.eventHandlers[event][i];
 360              r = eval( "f("+cb_args+");" ) || r;
 361          }
 362  
 363          return r;
 364      }
 365  };

 402   * Fade jQuery selector via backgrounds colors (bgs), back to original background
 403   * color and then remove any styles (from animations and others)
 404   *
 405   * @param string|jQuery
 406   * @param Array
 407   * @param object Options ("speed")
 408   */
 409  function evoFadeBg( selector, bgs, options )
 410  {
 411      var origBg = jQuery(selector).css("backgroundColor");
 412      var speed = options && options.speed || "slow";
 413  
 414      var toEval = 'jQuery(selector).animate({ backgroundColor: ';
 415      for( e in bgs )
 416      {
 417          toEval += '"'+bgs[e]+'"'+'}, "'+speed+'" ).animate({ backgroundColor: ';
 418      }
 419      toEval += 'origBg },"'+speed+'", "", function(){jQuery( this ).removeAttr( "style" );});';
 420  
 421      eval(toEval);
