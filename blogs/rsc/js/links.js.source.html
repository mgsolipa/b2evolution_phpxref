  26  function evo_display_position_onchange( selectInput, url, crumb )
  27  {
  28      var oThis = selectInput;
  29      var new_position = selectInput.value;
  30      jQuery.get( url + 'async.php?action=set_object_link_position&link_ID=' + selectInput.id.substr(17) + '&link_position=' + new_position + '&crumb_link=' + crumb, {
  31      }, function(r, status) {
  32          r = ajax_debug_clear( r );
  33          if( r == "OK" ) {
  34              evoFadeSuccess( jQuery(oThis.form).closest('tr') );
  35              jQuery(oThis.form).closest('td').removeClass('error');
  36              if( new_position == 'albumart')
  37              { // Position "Album Art" can be used only by one link
  38                  jQuery( 'select[name=link_position][id!=' + selectInput.id + '] option[value=albumart]:selected' ).each( function()
  39                  { // Replace previous position with "Inline"
  40                      jQuery( this ).parent().val( 'aftermore' );
  41                      evoFadeSuccess( jQuery( this ).closest('tr') );
  42                  } );
  43              }
  44          } else {
  45              jQuery(oThis).val(r);
  46              evoFadeFailure( jQuery(oThis.form).closest('tr') );
  47              jQuery(oThis.form).closest('td').addClass('error');
  48          }
  49      } );
  50      return false;
  51  }

  86  function item_unlink( link_ID )
  87  {
  88      var b2evoCanvas = window.parent.document.getElementById( 'itemform_post_content' );
  89      if( b2evoCanvas != null )
  90      { // Canvas exists
  91          var textarea_obj = jQuery( '#itemform_post_content', window.parent.document );
  92          if( textarea_obj.length == 0 )
  93          { // No found textarea
  94              return;
  95          }
  96          var content = textarea_obj.val();
  97          var regexp = new RegExp( '\\\[image:' + link_ID + ':?[^\\\]]*\\\]', 'ig' );
  98          textarea_obj.val( content.replace( regexp, '' ) );
  99      }
