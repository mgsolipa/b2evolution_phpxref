  47  function SendServerRequest( url )
  48  {
  49      // add a & to the URL if we already have a query string, otherwise add a ?
  50      url += ( url.indexOf( '?' ) != -1 ) ? '&' : '?';
  51      url += 'display_mode=js'; // add flag for js display mode
  52      var data = url.split( '?' );
  53      url = data[0];
  54      data = data[1];
  55  
  56      jQuery.ajax(
  57      {    // Load JavaScript via AJAX request
  58          // It gives the adding a log into "JS log"
  59          // The script is executed if AJAX request is completed
  60          type: 'POST',
  61          url: url,
  62          data: data,
  63          dataType: 'script'
  64      } );
  65  }

 116      var _dispatcher; // admin url

 119          /**
 120           * Initialise the helper object
 121           * Adds any translation strings found in html

 137              _interval_default = params.interval; // change interval if required
 138              _dispatcher = params.dispatcher; // store dispatcher

 146          /**
 147           * Enables server calls to be buffered
 148           *
 149           * @param ticker_callback (function) Called each time ticker occurs
 150           * @param send_callback (function) Called when send event occurs
 151           * @param delay (int) initial delay for buffer : defaults to _delay_default
 152           * @param interval (int) initial ticker interval : defaults to _interval_default
 153           * @param buffer_name (string) name for the buffer
 154           */
 155          BufferedServerCall:function()
 156          {
 157              // set available params to defaults
 158              var params = jQuery.fn.extend({
 159                      // no comma after final entry or IE barfs
 160                      ticker_callback: function(){ return true; }, // callback for ticker checks
 161                      send_callback: function(){}, // callback for sending
 162                      delay: _delay_default, // time to buffer call for
 163                      interval: _interval_default, // interval between polls
 164                      buffer_name:'' // name for the buffer
 165                      }, ( arguments.length ? arguments[0] : '' ) );
 166  
 167              if( ticker_status = params.ticker_callback( params.delay ) )
 168              {
 169                  if( ticker_status !== true )
 170                  {
 171                      b2evoHelper.log( 'Ticker status : '+ticker_status );
 172                  }
 173                  switch( ticker_status )
 174                  {
 175                      case 'cancel' : // cancel the server call
 176                          b2evoHelper.DisplayMessage( '<div class="log_message">'+T_( 'Update cancelled' )+'</div>' );
 177                          return;
 178  
 179                      case 'pause' : // pause the server call
 180                          b2evoHelper.DisplayMessage( '<div class="log_error">'+T_( 'Update Paused' )+' : '+b2evoHelper.str_repeat( '.', params.delay / params.interval )+'</div>' );
 181                          me.BufferedServerLoop( params );
 182                          return;
 183  
 184                      case 'ignore' : // don't change current message, ask again on next ticker
 185                          me.BufferedServerLoop(params);
 186                          return;
 187  
 188                      case 'immediate' : // send call without delay
 189                          break;
 190  
 191                      default :
 192                          params.delay -= params.interval;
 193                          if( params.delay > 0 )
 194                          { // still buffered
 195                              b2evoHelper.DisplayMessage( '<div class="log_error">'+T_( 'Changes pending' )+' : '+b2evoHelper.str_repeat( '.', params.delay / params.interval )+'</div>' );
 196                              me.BufferedServerLoop(params);
 197                              return;
 198                          }
 199                          // send the call;
 200                          b2evoHelper.DisplayMessage( '<div class="log_message">'+T_( 'Saving changes' )+'</div>' );
 201                          params.send_callback();
 202                  }
 203              }
 204          },

 263           * Send a request to the server

 265          SendServerRequest:function(){
 266              // set available params to defaults
 267              var params = jQuery.fn.extend({
 268                      // no comma after final entry or IE barfs
 269                      url: '', // destination url
 270                      data: '', // associated data for the call
 271                      error:function(){ return false; }, // trigger for onerror
 272                      ok:function(){ return false; } // trigger for 200
 273                      }, ( arguments.length ? arguments[0] : '' ) );
 274              if( params.url )
 275              { // we have a url
 276                  params.url += ( params.url.indexOf( '?' ) === true ? '&' : '?' )+'mode=js';

 279                      params.url += '&'+params.data;
 280                  }

 282                  script.attr( 'src', params.url );
