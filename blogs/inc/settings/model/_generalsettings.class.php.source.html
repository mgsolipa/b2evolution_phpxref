 310      /**
 311       * Constructor.
 312       *
 313       * This loads the general settings and checks db_version.
 314       *
 315       * It will also turn off error-reporting/halting of the {@link $DB DB object}
 316       * temporarily to present a more decent error message if tables do not exist yet.
 317       *
 318       * Because the {@link $DB DB object} itself creates a connection when it gets
 319       * created "Error selecting database" occurs before we can check for it here.
 320       */
 321  	function GeneralSettings()
 322      {
 323          global $new_db_version, $DB, $demo_mode, $instance_name, $basehost;
 324  
 325          $save_DB_show_errors = $DB->show_errors;
 326          $save_DB_halt_on_error = $DB->halt_on_error;
 327          $DB->halt_on_error = false;
 328          $DB->show_errors = false;
 329  
 330          // Init through the abstract constructor. This should be the first DB connection.
 331          parent::AbstractSettings( 'T_settings', array( 'set_name' ), 'set_value', 0 );
 332  
 333          // check DB version:
 334          if( $this->get( 'db_version' ) != $new_db_version )
 335          { // Database is not up to date:
 336              if( $DB->last_error )
 337              {
 338                  $error_message = '<p>MySQL error:</p>'.$DB->last_error;
 339              }
 340              else
 341              {
 342                  $error_message = '<p>Database schema is not up to date!</p>'
 343                      .'<p>You have schema version &laquo;'.(integer)$this->get( 'db_version' ).'&raquo;, '
 344                      .'but we would need &laquo;'.(integer)$new_db_version.'&raquo;.</p>';
 345              }
 346              global $adminskins_path;
 347              require $adminskins_path.'conf_error.main.php'; // error & exit
 348          }
 349  
 350          $DB->halt_on_error = $save_DB_halt_on_error;
 351          $DB->show_errors = $save_DB_show_errors;
 352  
 353  
 354          if( $demo_mode )
 355          { // Demo mode requires to allow multiple concurrent sessions:
 356              $this->_defaults['multiple_sessions'] = 'always';
 357          }
 358  
 359          // set those defaults which needs some global variables
 360          $this->_defaults['notification_sender_email'] = $instance_name.'-noreply@'.preg_replace( '/^www\./i', '', $basehost );
 361          $this->_defaults['notification_return_path'] = $instance_name.'-return@'.preg_replace( '/^www\./i', '', $basehost );
 362      }

 365      /**
 366       * Get a 32-byte string that can be used as salt for public keys.
 367       *
 368       * @return string
 369       */
 370  	function get_public_key_salt()
 371      {
 372          $public_key_salt = $this->get( 'public_key_salt' );
 373          if( empty($public_key_salt) )
 374          {
 375              $public_key_salt = generate_random_key(32);
 376              $this->set( 'public_key_salt', $public_key_salt );
 377              $this->dbupdate();
 378          }
 379          return $public_key_salt;
 380      }

 383      /**
 384       * Get a member param by its name
 385       *
 386       * @param mixed Name of parameter
 387       * @param boolean true to return param's real value
 388       * @return mixed Value of parameter
 389       */
 390  	function get( $parname, $real_value = false )
 391      {
 392          if( $real_value )
 393          {
 394              return parent::get( $parname );
 395          }
 396  
 397          switch($parname)
 398          {
 399              case 'allow_avatars':
 400                  return ( parent::get( $parname ) && isset($GLOBALS['files_Module']) );
 401                  break;
 402  
 403              case 'upload_enabled':
 404                  return ( parent::get( $parname ) && isset($GLOBALS['files_Module']) );
 405                  break;
 406  
 407              default:
 408                  return parent::get( $parname );
 409          }
