 187  function country_regions_count( $country_ID )

 189      global $DB, $admin_url;

 191      $regions_count = $DB->get_var( '
 192          SELECT COUNT(rgn_ID)
 193            FROM T_regional__region
 194           WHERE rgn_ctry_ID = "'.$country_ID.'"' );
 195  
 196      if( $regions_count > 0 )
 197      {
 198          $regions_count = '<a href="'.$admin_url.'?ctrl=regions&amp;c='.$country_ID.'">'.$regions_count.'</a>';
