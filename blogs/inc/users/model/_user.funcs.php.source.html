 185  function get_login_url( $source, $redirect_to = NULL, $force_normal_login = false, $blog_ID = NULL )
 186  {

 190      {
 191          $redirect = $redirect_to;
 192      }
 193      else
 194      {
 195          $redirect = regenerate_url( '', '', '', '&' );
 196      }
 197  
 198      if( use_in_skin_login() )
 199      { // use in-skin login
 200          if( empty( $blog_ID ) )
 201          { // Use current blog if it is not defined
 202              global $blog;

 204          }
 205          $BlogCache = & get_BlogCache();
 206          $Blog = $BlogCache->get_by_ID( $blog_ID );
 207          if( ! empty( $redirect ) )
 208          {

 210          }
 211          $url = url_add_param( $Blog->get( 'loginurl' ), $redirect, '&' );
 212      }
 213      else
 214      { // Normal login
 215          if( ! empty( $redirect ) )
 216          {
 217              $redirect = '?redirect_to='.rawurlencode( url_rel_to_same_host( $redirect, $secure_htsrv_url ) );
 218          }
 219          $url = $secure_htsrv_url.'login.php'.$redirect;
 220      }
 221  
 222      return url_add_param( $url, 'source='.rawurlencode($source), '&' );

 271  /**
 272   * Send notification to users with edit users permission
 273   *
 274   * @param string notification email suject
 275   * @param string notificaiton email template name
 276   * @param array notification email template params
 277   */
 278  function send_admin_notification( $subject, $template_name, $template_params )
 279  {
 280      global $Session, $UserSettings, $current_User;
 281  
 282      $UserCache = & get_UserCache();
 283      $template_params = array_merge( array(
 284              'login' => '',
 285          ), $template_params );
 286  
 287      // Set default subject and permname:
 288      $subject_suffix = ': '.$template_params['login'];
 289      $perm_name = 'users';
 290  
 291      switch( $template_name )
 292      {
 293          case 'account_new':
 294              $check_setting = 'notify_new_user_registration';
 295              break;
 296  
 297          case 'account_activated':
 298              $check_setting = 'notify_activated_account';
 299              break;
 300  
 301          case 'account_closed':
 302              $check_setting = 'notify_closed_account';
 303              break;
 304  
 305          case 'account_reported':
 306              $check_setting = 'notify_reported_account';
 307              break;
 308  
 309          case 'account_changed':
 310              $check_setting = 'notify_changed_account';
 311              break;
 312  
 313          case 'scheduled_task_error_report':
 314              $subject_suffix = '';
 315              $check_setting = 'notify_cronjob_error';
 316              $perm_name = 'options';
 317              break;
 318  
 319          default:
 320              debug_die( 'Unhandled admin notification template!' );
 321      }
 322  
 323      if( empty( $current_User ) && !empty( $Session ) && $Session->has_User() )
 324      { // current_User is not set at the time of registration
 325          $current_User = & $Session->get_User();
 326      }
 327  
 328      if( empty( $UserSettings ) )
 329      { // initialize UserSettings
 330          load_class( 'users/model/_usersettings.class.php', 'UserSettings' );
 331          $UserSettings = new UserSettings();
 332      }
 333  
 334      // load users with edit all users permission
 335      $UserCache->load_where( 'user_grp_ID = 1 OR user_grp_ID IN ( SELECT gset_grp_ID FROM T_groups__groupsettings WHERE gset_name = "perm_'.$perm_name.'" AND gset_value = "edit" )' );
 336      // iterate through UserCache
 337      $UserCache->rewind();
 338      while( $User = & $UserCache->get_next() )
 339      { // Loop through Users
 340          if( is_logged_in() && $current_User->ID == $User->ID )
 341          { // Don't send a notification to current user, because he already knows about this event
 342              continue;
 343          }
 344          if( $UserSettings->get( $check_setting, $User->ID ) && $User->check_perm( $perm_name, 'edit' ) )
 345          { // this user must be notifed
 346              locale_temp_switch( $User->get( 'locale' ) );
 347              // send mail to user (using his local)
 348              $localized_subject = T_( $subject ).$subject_suffix;
 349              send_mail_to_User( $User->ID, $localized_subject, $template_name, $template_params ); // ok, if this may fail
 350              locale_restore_previous();
 351          }
 352      }
 353  }

 395   * Check a settings from user for Back office and from skin for Front office
 396   *
 397   * @param string Setting name ( gender_colored OR bubbletip)
 398   * @return bool Use colored gender
 399   */
 400  function check_setting( $setting_name )
 401  {
 402      global $Settings, $Blog, $SkinCache;
 403  
 404      if( ! isset( $Blog ) && ! is_admin_page() )
 405      {    // If we use some page without blog data
 406          return false;
 407      }
 408  
 409      if( is_admin_page() )
 410      {    // Check setting in the Back office
 411          if( $Settings->get( $setting_name ) )
 412          {    // Set TRUE if the setting is ON
 413              return true;
 414          }
 415      }
 416      else
 417      {    // Check setting in the Front office for current blog & skin
 418          global $Blog, $SkinCache;
 419          if( ! isset( $SkinCache ) )
 420          {    // Init $SkinCache if it doesn't still exist
 421              $SkinCache = & get_SkinCache();
 422          }
 423          $skin = & $SkinCache->get_by_ID( $Blog->get( 'skin_ID' ) );
 424          if( $skin->get_setting( $setting_name ) )
 425          { // If setting is ON for current Blog & Skin
 426              if( $setting_name == 'bubbletip' )
 427              {    // Check separate case for setting 'bubbletip'
 428                  if( is_logged_in() || $Settings->get( $setting_name.'_anonymous' ) )
 429                  {    // If user is logged in OR Anonymous user can see bubbletips
 430                      return true;
 431                  }
 432              }
 433              else
 434              { // Setting 'gender_colored' doesn't depend on user's logged status
 435                  return true;
 436              }
 437          }
 438      }
 439  
 440      return false;
 441  }

1203  /**
1204   * Check profile parameters and add errors through {@link param_error()}.
1205   *
1206   * @param array associative array.
1207   *     Either array( $value, $input_name ) or just $value;
1208   *     ($input_name gets used for associating it to a form fieldname)
1209   *     - 'login': check for non-empty
1210   *     - 'nickname': check for non-empty
1211   *     - 'icq': must be a number
1212   *     - 'email': mandatory, must be well formed
1213   *     - 'country': check for non-empty
1214   *     - 'url': must be well formed, in allowed scheme, not blacklisted
1215   *     - 'pass1' / 'pass2': passwords (twice), must be the same and not == login (if given)
1216   *     - 'pass_required': false/true (default is true)
1217   * @param User|NULL A user to use for additional checks (password != login/nick).
1218   */
1219  function profile_check_params( $params, $User = NULL )
1220  {
1221      global $Messages, $Settings;
1222  
1223      foreach( $params as $k => $v )
1224      {
1225          // normalize params:
1226          if( $k != 'pass_required' && ! is_array($v) )
1227          {
1228              $params[$k] = array($v, $k);
1229          }
1230      }
1231  
1232      // checking login has been typed:
1233      if( isset($params['login'][0]) )
1234      {
1235          if( empty( $params['login'][0] ) )
1236          { // login can't be empty
1237              param_error( $params['login'][1], T_('Please enter your login.') );
1238          }
1239          else
1240          {
1241              param_check_valid_login( 'login' );
1242          }
1243      }
1244  
1245      // checking e-mail address
1246      if( isset($params['email'][0]) )
1247      {
1248          if( empty($params['email'][0]) )
1249          {
1250              param_error( $params['email'][1], T_('Please enter your e-mail address.') );
1251          }
1252          elseif( !is_email($params['email'][0]) )
1253          {
1254              param_error( $params['email'][1], T_('The email address is invalid.') );
1255          }
1256      }
1257  
1258      // Checking country
1259      if( isset($params['country']) && empty($params['country'][0]) )
1260      {
1261          param_error( 'country', T_('Please select country.') );
1262      }
1263  
1264      // Checking first name
1265      if( isset($params['firstname']) && empty($params['firstname'][0]) )
1266      {
1267          param_error( 'firstname', T_('Please enter your first name.') );
1268      }
1269  
1270      // Checking gender
1271      if( isset($params['gender']) )
1272      {
1273          if( empty($params['gender'][0]) )
1274          {
1275              param_error( 'gender', T_('Please select gender.') );
1276          }
1277          elseif( ( $params['gender'][0] != 'M' ) && ( $params['gender'][0] != 'F' ) )
1278          {
1279              param_error( 'gender', 'Gender value is invalid' );
1280          }
1281      }
1282  
1283      // Checking URL:
1284      if( isset($params['url']) )
1285      {
1286          if( $error = validate_url( $params['url'][0], 'commenting' ) )
1287          {
1288              param_error( $params['url'][1], T_('Supplied URL is invalid: ').$error );
1289          }
1290      }
1291  
1292      // Check passwords:
1293  
1294      $pass_required = isset( $params['pass_required'] ) ? $params['pass_required'] : true;
1295  
1296      if( isset($params['pass1'][0]) && isset($params['pass2'][0]) )
1297      {
1298          if( $pass_required || !empty($params['pass1'][0]) || !empty($params['pass2'][0]) )
1299          { // Password is required or was given
1300              // checking the password has been typed twice
1301              if( empty($params['pass1'][0]) || empty($params['pass2'][0]) )
1302              {
1303                  param_error( $params['pass2'][1], T_('Please enter your password twice.') );
1304              }
1305  
1306              // checking the password has been typed twice the same:
1307              if( $params['pass1'][0] !== $params['pass2'][0] )
1308              {
1309                  param_error( $params['pass1'][1], T_('You typed two different passwords.') );
1310              }
1311              elseif( $Settings->get('passwd_special') && !preg_match('~[\x20-\x2f\x3a-\x40\x5b-\x60\x7b-\x7f]~', $params['pass1'][0] )  )
1312              {
1313                  param_error( $params['pass1'][1], T_('Your password should contain at least one special character (like & ! $ * - _ + etc.)') );
1314              }
1315              elseif( evo_strlen($params['pass1'][0]) < $Settings->get('user_minpwdlen') )
1316              {
1317                  param_error( $params['pass1'][1], sprintf( T_('The minimum password length is %d characters.'), $Settings->get('user_minpwdlen')) );
1318              }
1319              elseif( isset($User) && $params['pass1'][0] == $User->get('login') )
1320              {

1326              }
