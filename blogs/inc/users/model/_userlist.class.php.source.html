 259  	function load_from_Request( $use_filters = true )
 260      {
 261          $this->filters = $this->default_filters;
 262  
 263          if( $use_filters )
 264          {
 265              // Do we want to restore filters or do we want to create a new filterset

 266              $filter = param( 'filter', 'string', '' );
 267              switch( $filter )
 268              {
 269                  case 'new':
 270                      $this->refresh_query = true;
 271                      break;
 272  
 273                  case 'reset':
 274                      // We want to reset the memorized filterset:

 275                      global $Session;
 276                      $Session->delete( $this->filterset_name );
 277  
 278                      // Memorize global variables:

 279                      $this->set_filters( array() );
 280                      $this->refresh_query = true;
 281                      // We have applied no filterset:

 282                      return false;
 283                      /* BREAK */

 284  
 285                  case 'refresh':
 286                      $this->refresh_query = true;
 287                      return $this->restore_filterset();
 288  
 289                  default:
 290                      return $this->restore_filterset();
 291              }
 292  
 293              /**

 294               * Filter preset

 295               */
 296              $this->filters['filter_preset'] = param( 'filter_preset', 'string', $this->default_filters['filter_preset'], true );
 297  
 298              // Activate preset default filters if necessary:

 299              $this->activate_preset_filters();
 300          }
 301  
 302          /*

 303           * Restrict by keywords

 304           */
 305          $this->filters['keywords'] = param( 'keywords', 'string', $this->default_filters['keywords'], true );         // Search string

 306  
 307          /*

 308           * Restrict by gender

 309           */
 310          $gender_men = param( 'gender_men', 'boolean', strpos( $this->default_filters['gender'], 'M' ), true );
 311          $gender_women = param( 'gender_women', 'boolean', strpos( $this->default_filters['gender'], 'F' ), true );
 312          if( ( $gender_men && ! $gender_women ) || ( ! $gender_men && $gender_women ) )
 313          {    // Find men OR women
 314              $this->filters['gender'] = $gender_men ? 'M' : 'F';
 315          }
 316          else if( $gender_men && $gender_women )
 317          {    // Find men AND women
 318              $this->filters['gender'] = 'MF';
 319          }
 320  
 321          /*

 322           * Restrict by status

 323           */
 324          $this->filters['account_status'] = param( 'account_status', 'string', $this->default_filters['account_status'], true );
 325          if( $this->filters['account_status'] === $this->default_filters['account_status'] &&
 326              param( 'status_activated', 'boolean', $this->default_filters['status_activated'], true ) )
 327          {
 328              $this->filters['status_activated'] = 'activated';
 329          }
 330          else
 331          {
 332              $this->filters['status_activated'] = $this->default_filters['status_activated'];
 333          }
 334  
 335          /*

 336           * Restrict by reported state ( was reported or not )

 337           */
 338          $this->filters['reported'] = param( 'reported', 'integer', $this->default_filters['reported'], true );
 339  
 340          /*

 341           * Restrict by custom sender email settings

 342           */
 343          $this->filters['custom_sender_email'] = param( 'custom_sender_email', 'integer', $this->default_filters['custom_sender_email'], true );
 344          $this->filters['custom_sender_name'] = param( 'custom_sender_name', 'integer', $this->default_filters['custom_sender_name'], true );
 345  
 346          /*

 347           * Restrict by user group

 348           */
 349          $this->filters['group'] = param( 'group', 'string', $this->default_filters['group'], true );
 350  
 351          /*

 352           * Restrict by locations

 353           */
 354          $this->filters['country'] = param( 'country', 'integer', $this->default_filters['country'], true );
 355          $this->filters['region'] = param( 'region', 'integer', $this->default_filters['region'], true );
 356          $this->filters['subregion'] = param( 'subregion', 'integer', $this->default_filters['subregion'], true );
 357          $this->filters['city'] = param( 'city', 'integer', $this->default_filters['city'], true );
 358  
 359          /*

 360           * Restrict by age group

 361           */
 362          $this->filters['age_min'] = param( 'age_min', 'integer', $this->default_filters['age_min'], true );
 363          $this->filters['age_max'] = param( 'age_max', 'integer', $this->default_filters['age_max'], true );
 364  
 365          /*

 366           * Restrict by user fields

 367           */
 368          $criteria_types = param( 'criteria_type', 'array/integer', array(), true );
 369          $criteria_values = param( 'criteria_value', 'array/string', array(), true );
 370          $userfields = array();
 371          foreach( $criteria_types as $c => $type )
 372          {
 373              $userfields[] = array(
 374                      'type' => $type,
 375                      'value' => $criteria_values[$c]
 376                  );
 377          }
 378          $this->filters['userfields'] = $userfields;
 379  
 380          // 'paged'

 381          $this->page = param( $this->page_param, 'integer', 1, true );      // List page number in paged display

 382  
 383          // 'order'

 384          global $Session;
 385          $prev_filters = $Session->get( $this->filterset_name );
 386          if( !empty( $prev_filters['order'] ) )
 387          {    // Restore an order from saved session
 388              $this->order = $this->filters['order'] = $prev_filters['order'];
 389          }
 390  
 391          if( $use_filters && $filter == 'new' )
 392          {
 393              $this->save_filterset();
 394          }
 395  
 396          return ! param_errors_detected();
 397      }

 405  	function query_init()
 406      {
 407          global $current_User;
 408  
 409          if( empty( $this->filters ) )
 410          {    // Filters have not been set before, we'll use the default filterset:
 411              // If there is a preset filter, we need to activate its specific defaults:

 412              $this->filters['filter_preset'] = param( 'filter_preset', 'string', $this->default_filters['filter_preset'], true );
 413              $this->activate_preset_filters();
 414  
 415              // Use the default filters:

 416              $this->set_filters( $this->default_filters );
 417          }
 418  
 419          // GENERATE THE QUERY:

 420  
 421          // The SQL Query object:

 422          // If group == -1 we shouldn't group list by user group

 423          $this->query_params['grouped'] = ( $this->filters['group'] != -1 );
 424          $this->UserQuery = new UserQuery( $this->Cache->dbtablename, $this->Cache->dbprefix, $this->Cache->dbIDname, $this->query_params );
 425          if( isset( $this->query_params['keywords_fields'] ) )
 426          { // Change keywords_fields from query params
 427              $this->UserQuery->keywords_fields = $this->query_params['keywords_fields'];
 428          }
 429          if( isset( $this->query_params['where_status_closed'] ) )
 430          { // Limit by closed users
 431              $this->UserQuery->where_status( 'closed', $this->query_params['where_status_closed'] );
 432          }
 433  
 434          // If browse users from different countries is restricted, then always filter to the current User country

 435          if( has_cross_country_restriction() )
 436          { // Browse users from different countries is restricted, filter to current user country
 437              $ctry_filter = $current_User->ctry_ID;
 438              // if country filtering was changed the qurey must be refreshed

 439              $this->refresh_query = ( $this->refresh_query || ( $ctry_filter != $this->filters['country'] ) );
 440          }
 441          else
 442          { // Browse users from different countries is allowed
 443              $ctry_filter = $this->filters['country'];
 444          }
 445  
 446          /*

 447           * filtering stuff:

 448           */
 449          $this->UserQuery->where_keywords( $this->filters['keywords'] );
 450          $this->UserQuery->where_gender( $this->filters['gender'] );
 451          $this->UserQuery->where_status( $this->filters['status_activated'] );
 452          $this->UserQuery->where_status( $this->filters['account_status'], true, true );
 453          $this->UserQuery->where_reported( $this->filters['reported'] );
 454          $this->UserQuery->where_custom_sender( $this->filters['custom_sender_email'], $this->filters['custom_sender_name'] );
 455          $this->UserQuery->where_group( $this->filters['group'] );
 456          $this->UserQuery->where_location( 'ctry', $ctry_filter );
 457          $this->UserQuery->where_location( 'rgn', $this->filters['region'] );
 458          $this->UserQuery->where_location( 'subrg', $this->filters['subregion'] );
 459          $this->UserQuery->where_location( 'city', $this->filters['city'] );
 460          $this->UserQuery->where_age_group( $this->filters['age_min'], $this->filters['age_max'] );
 461          $this->UserQuery->where_userfields( $this->filters['userfields'] );
 462  
 463          if( $this->get_order_field_list() != '' )
 464          {
 465              $this->UserQuery->order_by( str_replace( '*', $this->get_order_field_list(), $this->UserQuery->order_by ) );
 466          }
 467          else
 468          {
 469              $this->UserQuery->order_by( $this->get_order_field_list() );
 470          }
 471      }

 474      /**

 475       * Run Query: GET DATA ROWS *** HEAVY ***

 476       */
 477  	function query()
 478      {
 479          global $DB, $Session, $localtimenow;
 480  
 481          if( !is_null( $this->rows ) )
 482          { // Query has already executed:
 483              return;
 484          }
 485  
 486          // INIT THE QUERY:

 487          $this->query_init();
 488  
 489  
 490          // We are going to proceed in two steps (we simulate a subquery)

 491          // 1) we get the IDs we need

 492          // 2) we get all the other fields matching these IDs

 493          // This is more efficient than manipulating all fields at once.

 494  
 495          // *** STEP 1 ***

 496          $user_IDs = $this->filters['users'];
 497          if( $this->refresh_query || // Some filters are changed
 498              $localtimenow - $Session->get( $this->filterset_name.'_refresh_time' ) > 7200 ) // Time has passed ( 2 hours )
 499          {    // We should create new list of user IDs
 500              global $Timer;
 501              $Timer->start( 'Users_IDs', false );
 502  
 503              $step1_SQL = new SQL();
 504              $step1_SQL->SELECT( 'T_users.user_ID, IF( user_avatar_file_ID IS NOT NULL, 1, 0 ) as has_picture, COUNT( DISTINCT blog_ID ) AS nb_blogs' );
 505              if( !empty( $this->filters['reported'] ) && $this->filters['reported'] )
 506              {    // Filter is set to 'Reported users'
 507                  $step1_SQL->SELECT_add( ', COUNT( DISTINCT urep_reporter_ID ) AS user_rep' );
 508              }
 509              $step1_SQL->FROM( $this->UserQuery->get_from( '' ) );
 510              $step1_SQL->WHERE( $this->UserQuery->get_where( '' ) );
 511              $step1_SQL->GROUP_BY( $this->UserQuery->get_group_by( '' ) );
 512              $step1_SQL->ORDER_BY( $this->UserQuery->get_order_by( '' ) );
 513              $step1_SQL->LIMIT( 0 );
 514  
 515              // Get list of the IDs we need:

 516              $user_IDs = $DB->get_col( $step1_SQL->get(), 0, 'UserList::Query() Step 1: Get ID list' );
 517              // Update filter with user IDs

 518              $this->filters['users'] = $user_IDs;
 519              $this->save_filterset();
 520  
 521              $Timer->stop( 'Users_IDs' );
 522          }
 523  
 524          // GET TOTAL ROW COUNT:

 525          parent::count_total_rows( count( $user_IDs ) );
 526  
 527          // Pagination, Get user IDs from array for current page

 528          $user_IDs_paged = array_slice( $user_IDs, ( $this->page - 1 ) * $this->limit, $this->limit );
 529  
 530          // *** STEP 2 ***

 531          $step2_SQL = $this->UserQuery;
 532  
 533          if( ! empty( $user_IDs_paged ) )
 534          {    // Init sql query to get users by IDs
 535              $step2_SQL->WHERE( $this->Cache->dbIDname.' IN ('.implode( ',', $user_IDs_paged ).') ' );
 536              $step2_SQL->ORDER_BY( 'FIND_IN_SET( user_ID, "'.implode( ',', $user_IDs_paged ).'" )' );
 537          }
 538          else
 539          {    // No users
 540              $step2_SQL->WHERE( 'user_ID IS NULL' );
 541          }
 542  
 543          $this->sql = $step2_SQL->get();
 544  
 545          // ATTENTION: we skip the parent on purpose here!! fp> refactor

 546          DataObjectList2::query( false, false, false, 'UserList::Query() Step 2' );
 547      }

 698       * Skip to previous/next User

 699       *

 700       * @param integer the currently selected user ID ( Note: it must be set only if we would like to skip some users from the list )

 701       * @param string prev | next  (relative to the current sort order)

 702       */
 703      function & get_prevnext_User( $direction = 'next', $selected_user_ID = NULL )
 704      {
 705          $users_list = $this->filters['users'];
 706  
 707          if( count( $users_list ) < 2 )
 708          {    // Short users list
 709              $r = NULL;
 710              return $r;
 711          }
 712  
 713          // ID of selected user

 714          if( $selected_user_ID === NULL )
 715          { // get currently selected user ID from param
 716              $selected_user_ID = get_param( 'user_ID' );
 717          }
 718  
 719          $user_key = array_search( $selected_user_ID, $users_list );
 720          if( is_int( $user_key ) )
 721          {    // Selected user is located in the list
 722              $prevnext_key = $direction == 'next' ? $user_key + 1 : $user_key - 1;
 723              if( isset( $users_list[$prevnext_key] ) )
 724              {    // Prev/next user is located in the list
 725                  $prevnext_ID = $users_list[$prevnext_key];
 726              }
 727          }
 728  
 729          if( empty( $prevnext_ID ) )
 730          {    // No prev/next user
 731              $r = NULL;
 732              return $r;
 733          }
 734  
 735          $UserCache = & get_UserCache();
 736          $User = & $UserCache->get_by_ID( $prevnext_ID, false, false );
 737  
 738          return $User;
 739      }

 742      /**

 743       * Set an order of a list (Use this function after when all $this->cols are already defined)

 744       *

 745       * @param string Field name

 746       * @param string Order direction (A|D)

 747       * @param boolean Save the filters from Session

 748       */
 749  	function set_order( $order_field, $direction = 'D', $save_filters = false )
 750      {
 751          global $Session;
 752  
 753          if( empty( $this->cols ) )
 754          { // The columns are not defined yet, Exit here
 755              return;
 756          }
 757  
 758          // Force filter param to reset the previous filters

 759          set_param( 'filter', 'new' );
 760          $this->refresh_query = true;
 761  
 762          foreach( $this->cols as $col_num => $col )
 763          {    // Find a column number
 764              if( $col['order'] == $order_field )
 765              {
 766                  break;
 767              }
 768          }
 769  
 770          if( $save_filters )
 771          { // Get the filters from Session
 772              $this->filters = $Session->get( $this->filterset_name );
 773              if( ! is_array( $this->filters ) )
 774              {
 775                  $this->filters = array();
 776              }
 777              $this->filters = array_merge( $this->default_filters, $this->filters );
 778          }
 779          else
 780          { // Reset the filters
 781              $this->filters = array();
 782          }
 783  
 784          // Rewrite a previous order to new value

 785          $this->filters['order'] = str_repeat( '-', $col_num ).$direction;
 786          $this->order = $this->filters['order'];
 787  
 788          // Save a new order

 789          $Session->set( $this->filterset_name, $this->filters );
 790          $this->save_filterset();
 791      }
