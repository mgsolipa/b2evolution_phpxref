 119       * Load data from Request form fields.

 123  	function load_from_Request()
 124      {
 125          global $Messages, $demo_mode;
 126  
 127          // Edited Group Name
 128          param( 'edited_grp_name', 'string' );
 129          param_check_not_empty( 'edited_grp_name', T_('You must provide a group name!') );
 130          $this->set_from_Request('name', 'edited_grp_name', true);
 131  
 132          // Edited Group Permission Blogs
 133          param( 'edited_grp_perm_blogs', 'string', true );
 134          $this->set_from_Request( 'perm_blogs', 'edited_grp_perm_blogs', true );
 135  
 136          $apply_antispam = ( param( 'apply_antispam', 'integer', 0 ) ? 0 : 1 );
 137          $perm_xhtmlvalidation = param( 'perm_xhtmlvalidation', 'string', true );
 138          $perm_xhtmlvalidation_xmlrpc = param( 'perm_xhtmlvalidation_xmlrpc', 'string', true );
 139          $prevent_css_tweaks = ( param( 'prevent_css_tweaks', 'integer', 0 ) ? 0 : 1 );
 140          $prevent_iframes = ( param( 'prevent_iframes', 'integer', 0 ) ? 0 : 1 );
 141          $prevent_javascript = ( param( 'prevent_javascript', 'integer', 0 ) ? 0 : 1 );
 142          $prevent_objects = ( param( 'prevent_objects', 'integer', 0 ) ? 0 : 1 );
 143  
 144          if( $demo_mode && ( $apply_antispam || ( $perm_xhtmlvalidation != 'always' ) && ( $perm_xhtmlvalidation_xmlrpc != 'always' )
 145               || $prevent_css_tweaks || $prevent_iframes || $prevent_javascript || $prevent_objects ) )
 146          { // Demo mode restriction: Do not allow to change these settings in demo mode, because it may lead to security problem!
 147              $Messages->add( 'Validation settings and security filters are not editable in demo mode!', 'error' );
 148          }
 149          else
 150          {
 151              // Apply Antispam
 152              $this->set( 'perm_bypass_antispam', $apply_antispam );
 153  
 154              // XHTML Validation
 155              $this->set( 'perm_xhtmlvalidation', $perm_xhtmlvalidation );
 156  
 157              // XHTML Validation XMLRPC
 158              $this->set( 'perm_xhtmlvalidation_xmlrpc', $perm_xhtmlvalidation_xmlrpc );
 159  
 160              // CSS Tweaks
 161              $this->set( 'perm_xhtml_css_tweaks', $prevent_css_tweaks );
 162  
 163              // Iframes
 164              $this->set( 'perm_xhtml_iframes', $prevent_iframes );
 165  
 166              // Javascript
 167              $this->set( 'perm_xhtml_javascript', $prevent_javascript );
 168  
 169              // Objects
 170              $this->set( 'perm_xhtml_objects', $prevent_objects );
 171          }
 172  
 173          // Stats
 174          $this->set( 'perm_stats', param( 'edited_grp_perm_stats', 'string', true ) );
 175  
 176          // Load pluggable group permissions from request
 177          $GroupSettings = & $this->get_GroupSettings();
 178          foreach( $GroupSettings->permission_values as $name => $value )
 179          {
 180              // We need to handle checkboxes and radioboxes separately , because when a checkbox isn't checked the checkbox variable is not sent
 181              if( $name == 'perm_createblog' || $name == 'perm_getblog' || $name == 'perm_templates'
 182                  || $name == 'cross_country_allow_profiles' || $name == 'cross_country_allow_contact' )
 183              { // These permissions are represented by checkboxes, all other pluggable group permissions are represented by radiobox.
 184                  $value = param( 'edited_grp_'.$name, 'string', 'denied' );
 185              }
 186              elseif( ( $name == 'perm_admin' || $name == 'perm_users' ) && ( $this->ID == 1 ) )
 187              { // Admin group has always admin perm, it can not be set or changed.
 188                  continue;
 189              }
 190              else
 191              {
 192                  $value = param( 'edited_grp_'.$name, 'string', '' );
 193              }
 194              if( ( $value != '') || ( $name == 'max_new_threads'/*allow empty*/ ) )
 195              { // if radio is not set, then doesn't change the settings
 196                  $GroupSettings->set( $name, $value, $this->ID );
 197              }
 198          }
 199  
 200          return !param_errors_detected();
 201      }

 241      /**
 242       * Check a permission for this group.
 243       *
 244       * @param string Permission name:
 245       *                - templates
 246       *                - stats
 247       *                - spamblacklist
 248       *                - options
 249       *                - users
 250       *                - blogs
 251       *                - admin (levels "visible", "hidden")
 252       *                - messaging
 253       * @param string Requested permission level
 254       * @param mixed Permission target (blog ID, array of cat IDs...)
 255       * @return boolean True on success (permission is granted), false if permission is not granted
 256       */
 257  	function check_perm( $permname, $permlevel = 'any', $perm_target = NULL )
 258      {
 259          global $Debuglog;
 260  
 261          $perm = false; // Default is false!
 262  
 263          // echo "<br>Checking group perm $permname:$permlevel against $permvalue";
 264          if( isset($this->{'perm_'.$permname}) )
 265          {
 266              $permvalue = $this->{'perm_'.$permname};
 267          }
 268          else
 269          { // Object's perm-property not set!
 270              $Debuglog->add( 'Group permission perm_'.$permname.' not defined!', 'perms' );
 271  
 272              $permvalue = false; // This will result in $perm == false always. We go on for the $Debuglog..
 273          }
 274  
 275          $pluggable_perms = array( 'admin', 'shared_root', 'import_root', 'spamblacklist', 'slugs', 'templates', 'options', 'emails', 'files', 'users' );
 276          if( in_array( $permname, $pluggable_perms ) )
 277          {
 278              $permname = 'perm_'.$permname;
 279          }
 280          // echo "<br>Checking group perm $permname:$permlevel against $permvalue";
 281  
 282          // Check group permission:
 283          switch( $permname )
 284          {
 285              case 'blogs':
 286                  switch( $permvalue )
 287                  { // Depending on current group permission:
 288  
 289                      case 'editall':
 290                          // All permissions granted
 291                          $perm = true;
 292                          break;
 293  
 294                      case 'viewall':
 295                          // User can only ask for view perm
 296                          if(( $permlevel == 'view' ) || ( $permlevel == 'any' ))
 297                          { // Permission granted
 298                              $perm = true;
 299                              break;
 300                          }
 301                  }
 302  
 303                  if( ! $perm && ( $permlevel == 'create' ) && $this->check_perm( 'perm_createblog', 'allowed' ) )
 304                  { // User is allowed to create a blog (for himself)
 305                      $perm = true;
 306                  }
 307                  break;
 308  
 309              case 'stats':
 310                  if( ! $this->check_perm( 'admin', 'restricted' ) )
 311                  {
 312                      $perm = false;
 313                      break;
 314                  }
 315                  switch( $permvalue )
 316                  { // Depending on current group permission:
 317  
 318                      case 'edit':
 319                          // All permissions granted
 320                          $perm = true;
 321                          break;
 322  
 323                      case 'view':
 324                          // User can ask for view perm...
 325                          if( $permlevel == 'view' )
 326                          {
 327                              $perm = true;
 328                              break;
 329                          }
 330                          // ... or for any lower priority perm... (no break)
 331  
 332                      case 'user':
 333                          // This is for stats. User perm can grant permissions in the User class
 334                          // Here it will just allow to list
 335                      case 'list':
 336                          // User can only ask for list perm
 337                          if( $permlevel == 'list' )
 338                          {
 339                              $perm = true;
 340                              break;
 341                          }
 342                  }
 343                  break;
 344  
 345              case 'perm_files':
 346                  if( ! $this->check_perm( 'admin', 'restricted' ) )
 347                  {
 348                      $perm = false;
 349                      break;
 350                  }
 351                  // no break, perm_files is pluggable permission
 352  
 353              default:
 354  
 355                  // Check pluggable permissions using group permission check function
 356                  $perm = Module::check_perm( $permname, $permlevel, $perm_target, 'group_func', $this );
 357                  if( $perm === NULL )
 358                  {    // Even if group permisson check function doesn't exist we should return false value
 359                      $perm = false;
 360                  }
 361  
 362                  break;
 363          }
 364  
 365          $target_ID = $perm_target;
 366          if( is_object($perm_target) ) $target_ID = $perm_target->ID;
 367  
 368          $Debuglog->add( "Group perm $permname:$permlevel:$target_ID => ".($perm?'granted':'DENIED'), 'perms' );
 369  
 370          return $perm;
 371      }

 374      /**
 375       * Check permission for this group on a specified blog
 376       *
 377       * This is not for direct use, please call {@link User::check_perm()} instead
 378       * user is checked for privileges first, group lookup only performed on a false result
 379       *
 380       * @see User::check_perm()
 381       * @param string Permission name can be any from the blog advanced perm names. A few possible permname:
 382       *                  - blog_ismember
 383       *                  - blog_can_be_assignee
 384       *                  - blog_del_post
 385       *                  - blog_edit_ts
 386       *                  - blog_post_statuses
 387       *                  - blog_edit
 388       *                  - blog_comment_statuses
 389       *                  - blog_edit_cmt
 390       *                  - blog_cats
 391       *                  - blog_properties
 392       * @param string Permission level
 393       * @param integer Permission target blog ID
 394       * @param Item post that we want to edit
 395       * @param User for who we would like to check this permission
 396       * @return boolean 0 if permission denied
 397       */
 398  	function check_perm_bloggroups( $permname, $permlevel, $perm_target_blog, $perm_target = NULL, $User = NULL )
 399      {
 400          if( !isset( $this->blog_post_statuses[$perm_target_blog] ) )
 401          {
 402              $this->blog_post_statuses[$perm_target_blog] = array();
 403              if( ! load_blog_advanced_perms( $this->blog_post_statuses[$perm_target_blog], $perm_target_blog, $this->ID, 'bloggroup' ) )
 404              { // Could not load blog advanced user perms
 405                  return false;
 406              }
 407          }
 408  
 409          $blog_perms = $this->blog_post_statuses[$perm_target_blog];
 410          if( empty( $User ) )
 411          { // User is not set
 412              $user_ID = NULL;
 413          }
 414          else
 415          { // User is set, advanced user perms must be loaded
 416              $user_ID = $User->ID;
 417              if( isset( $User->blog_post_statuses[$perm_target_blog] ) )
 418              { // Merge user advanced perms with group advanced perms
 419                  $edit_perms = array( 'no' => 0, 'own' => 1, 'anon' => 2, 'lt' => 3, 'le' => 4, 'all' => 5 );
 420                  foreach( $User->blog_post_statuses[$perm_target_blog] as $key => $value )
 421                  { // For each collection advanced permission use the higher perm value between user and group perms
 422                      if( ( $key == 'blog_edit' ) || ( $key == 'blog_edit_cmt' ) )
 423                      {
 424                          if( $edit_perms[$value] > $edit_perms[$blog_perms[$key]] )
 425                          { // Use collection user edit permission because it is greater than the collection group perm
 426                              $blog_perms[$key] = $value;
 427                          }
 428                      }
 429                      elseif( isset( $blog_perms[$key] ) )
 430                      { // Check user and group perm as well
 431                          $blog_perms[$key] = (int) $value | (int) $blog_perms[$key];
 432                      }
 433                  }
 434              }
 435          }
 436          return check_blog_advanced_perm( $blog_perms, $user_ID, $permname, $permlevel, $perm_target );
 437      }
