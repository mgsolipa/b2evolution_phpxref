 179      /**
 180       * Load data from Request form fields.
 181       *
 182       * @return boolean true if loaded data seems valid.
 183       */
 184  	function load_from_Request()
 185      {
 186          // Group
 187          $old_group_ID = $this->ufgp_ID; // Save old group ID to know if it was changed
 188          param_string_not_empty( 'ufdf_ufgp_ID', T_('Please select a group.') );
 189          $this->set_from_Request( 'ufgp_ID' );
 190  
 191          // Type
 192          param_string_not_empty( 'ufdf_type', T_('Please enter a type.') );
 193          $this->set_from_Request( 'type' );
 194  
 195          // Name
 196          param_string_not_empty( 'ufdf_name', T_('Please enter a name.') );
 197          $this->set_from_Request( 'name' );
 198  
 199          // Options
 200          if( param( 'ufdf_type', 'string' ) == 'list' )
 201          {    // Save 'Options' only for Field type == 'Option list'
 202              $ufdf_options = explode( "\n", param( 'ufdf_options', 'text' ) );
 203              if( count( $ufdf_options ) < 2 )
 204              {    // We don't want save an option list with one item
 205                  param_error( 'ufdf_options', T_('Please enter at least 2 options on 2 different lines.') );
 206              }
 207              $this->set_from_Request( 'options' );
 208          }
 209          else
 210          { // The 'options' field must be set because it doesn't have a default value
 211              $this->set( 'options', '' );
 212          }
 213  
 214          // Required
 215          param_string_not_empty( 'ufdf_required', 'Please select Hidden, Optional, Recommended or Required.' );
 216          $this->set_from_Request( 'required' );
 217  
 218          // Duplicated
 219          param_string_not_empty( 'ufdf_duplicated', 'Please select Forbidden, Allowed or List style.' );
 220          $this->set_from_Request( 'duplicated' );
 221  
 222          // Order
 223          if( $old_group_ID != $this->ufgp_ID )
 224          { // Group is changing, set order as last
 225              $this->set( 'order', $this->get_last_order( $this->ufgp_ID ) );
 226          }
 227  
 228          // Suggest
 229          if( param( 'ufdf_type', 'string' ) == 'word' )
 230          { // Save 'Suggest values' only for Field type == 'Single word'
 231              param( 'ufdf_suggest', 'integer', 0 );
 232              $this->set_from_Request( 'suggest' );
 233          }
 234  
 235          // Bubbletip
 236          param( 'ufdf_bubbletip', 'text', '' );
 237          $this->set_from_Request( 'bubbletip' );
 238  
 239          return ! param_errors_detected();
 240      }
