 151      /**
 152       * Get a setting from the DB user settings table
 153       *
 154       * @param string name of setting
 155       * @param integer User ID (by default $current_User->ID will be used)
 156       */
 157  	function get( $setting, $user_ID = NULL )
 158      {
 159          global $Settings;
 160  
 161          if( ! isset($user_ID) )
 162          {
 163              global $current_User;
 164  
 165              if( ! isset($current_User) )
 166              { // no current/logged in user:
 167                  $result = $this->get_default($setting);
 168                  if( $result == NULL )
 169                  {
 170                      $result = $Settings->get( 'def_'.$setting );
 171                  }
 172                  return $result;
 173              }
 174  
 175              $user_ID = $current_User->ID;
 176          }
 177  
 178          $result = parent::get( $user_ID, $setting );
 179          if( $result == NULL )
 180          {
 181              $result = $Settings->get( 'def_'.$setting );
 182          }
 183          return $result;
 184      }

 236      /**
 237       * Get a param from Request and save it to UserSettings, or default to previously saved user setting.
 238       *
 239       * If the user setting was not set before (and there's no default given that gets returned), $default gets used.
 240       *
 241       * @todo Move this to _abstractsettings.class.php - the other Settings object can also make use of it!
 242       *
 243       * @param string Request param name
 244       * @param string User setting name. Make sure this is unique!
 245       * @param string Force value type to one of:
 246       * - integer
 247       * - float
 248       * - string (strips (HTML-)Tags, trims whitespace)
 249       * - array
 250       * - object
 251       * - null
 252       * - html (does nothing)
 253       * - '' (does nothing)
 254       * - '/^...$/' check regexp pattern match (string)
 255       * - boolean (will force type to boolean, but you can't use 'true' as a default since it has special meaning. There is no real reason to pass booleans on a URL though. Passing 0 and 1 as integers seems to be best practice).
 256       * Value type will be forced only if resulting value (probably from default then) is !== NULL
 257       * @param mixed Default value or TRUE if user input required
 258       * @param boolean Do we need to memorize this to regenerate the URL for this page?
 259       * @param boolean Override if variable already set
 260       * @return NULL|mixed NULL, if neither a param was given nor {@link $UserSettings} knows about it.
 261       */
 262  	function param_Request( $param_name, $uset_name, $type = '', $default = '', $memorize = false, $override = false ) // we do not force setting it..
 263      {
 264          $value = param( $param_name, $type, NULL, $memorize, $override, false ); // we pass NULL here, to see if it got set at all
 265  
 266          if( $value !== false )
 267          { // we got a value
 268              $this->set( $uset_name, $value );
 269              $this->dbupdate();
 270          }
 271          else
 272          { // get the value from user settings
 273              $value = $this->get($uset_name);
 274  
 275              if( is_null($value) )
 276              { // it's not saved yet and there's not default defined ($_defaults)
 277                  $value = $default;
 278              }
 279              if( $memorize )
 280              { // Memorize param
 281                  memorize_param( $param_name, $type, $default, $value );
 282              }
 283          }
 284  
 285          set_param( $param_name, $value );
 286          return get_param($param_name);
 287      }

 291       * Reset a user settings to the default values
 292       * 
 293       * @param integer user ID
 294       * @param boolean set to true to save modifications
 295       */
 296  	function reset_to_defaults( $user_ID, $db_save = true )
 297      {
 298          // Remove all UserSettings where a default or configurable default exists:
 299          foreach( $this->_defaults as $k => $v )
 300          {
 301              $this->delete( $k, $user_ID );
 302          }
 303          foreach( $this->_configurable_defaults as $k => $v )
 304          {
 305              $this->delete( $k, $user_ID );
 306          }
