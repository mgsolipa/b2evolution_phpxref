 212    /**
 213       * Generate a single post link for the item
 214       *
 215       * @param boolean allow redir to permalink, true | false | 'auto' to prevent redit only if single isn't the current permalink type
 216        * @param string base url to use
 217       * @param string glue between url params
 218       */
 219  	function get_single_url( $allow_redir = true, $blogurl = '', $glue = '&amp;' )
 220      {
 221          $this->get_Blog();
 222  
 223          if( empty( $blogurl ) )
 224          {
 225              $blogurl = $this->Blog->gen_blogurl();
 226          }
 227  
 228          $single_links = $this->Blog->get_setting('single_links');
 229  
 230           if( !empty( $this->urltitle ) && $single_links != 'param_num' )
 231          {    // We can and we want to use the url title:
 232              $urlparam = 'title='.$this->urltitle;
 233              $urltail = $this->urltitle;
 234          }
 235          else
 236          {
 237              $urlparam = 'p='.$this->ID;
 238              $urltail = 'p'.$this->ID;
 239          }
 240  
 241          switch( $single_links )
 242          {
 243              case 'param_num':
 244              case 'param_title':
 245                  $permalink = url_add_param( $blogurl, $urlparam.$glue.'more=1'.$glue.'c=1'.$glue.'tb=1'.$glue.'pb=1', $glue );
 246                  break;
 247  
 248              case 'y':
 249                  $permalink = url_add_tail( $blogurl, mysql2date('/Y/', $this->issue_date).$urltail );
 250                  break;
 251  
 252              case 'ym':
 253                  $permalink = url_add_tail( $blogurl, mysql2date('/Y/m/', $this->issue_date).$urltail );
 254                  break;
 255  
 256              case 'ymd':
 257                  $permalink = url_add_tail( $blogurl, mysql2date('/Y/m/d/', $this->issue_date).$urltail );
 258                  break;
 259  
 260              case 'subchap':
 261                  $main_Chapter = & $this->get_main_Chapter();
 262                  $permalink = url_add_tail( $blogurl, '/'.$main_Chapter->urlname.'/'.$urltail );
 263                  break;
 264  
 265              case 'chapters':
 266                  $main_Chapter = & $this->get_main_Chapter();
 267                  $permalink = url_add_tail( $blogurl, '/'.$main_Chapter->get_url_path().$urltail );
 268                  break;
 269  
 270              case 'short':
 271              default:
 272                  $permalink = url_add_tail( $blogurl, '/'.$urltail );
 273                  break;
 274          }
 275  
 276          if( $allow_redir == 'auto' )
 277          {    // We allow redir only if the permalink is already single.
 278              // In other words: we implicitely allow redir if there is no need to redir!
 279              // and more useful: we explicitly prevent redir if we know it would take place.
 280              $allow_redir = ($this->Blog->get_setting( 'permalinks' ) == 'single');
 281          }
 282  
 283          if( ! $allow_redir )
 284          {
 285              $permalink = url_add_param( $permalink, 'redir=no', $glue );
 286          }
 287  
 288          return $permalink;
 289      }

 334      /**
 335       * Generate the permalink for the item.
 336       *
 337       * Note: Each item has an unique permalink at any given time.
 338       * Some admin settings may however change the permalinks for previous items.
 339       * Note: This actually only returns the URL, to get a real link, use {@link Item::get_permanent_link()}
 340       *
 341       * @todo archives modes in clean URL mode
 342       *
 343       * @param string single, archive, subchap
 344       * @param string base url to use
 345       * @param string glue between url params
 346       */
 347  	function get_permanent_url( $permalink_type = '', $blogurl = '', $glue = '&amp;' )
 348      {
 349          global $DB, $cacheweekly, $Settings, $posttypes_specialtypes, $posttypes_nopermanentURL, $posttypes_catpermanentURL;
 350  
 351          $this->get_Blog();
 352          if( $this->Blog->get_setting( 'front_disp' ) == 'page' &&
 353              $this->Blog->get_setting( 'front_post_ID' ) == $this->ID )
 354          { // This item is used as front specific page on the blog's home
 355              $permalink_type = 'none';
 356          }
 357          elseif( in_array( $this->ptyp_ID, $posttypes_specialtypes ) ) // page, intros, sidebar
 358          {    // This is not an "in stream" post:
 359              if( in_array( $this->ptyp_ID, $posttypes_nopermanentURL ) )
 360              {    // This type of post is not allowed to have a permalink:
 361                  $permalink_type = 'none';
 362              }
 363              elseif( in_array( $this->ptyp_ID, $posttypes_catpermanentURL ) )
 364              {    // This post has a permanent URL as url to main chapter:
 365                  $permalink_type = 'cat';
 366              }
 367              else
 368              { // allowed to have a permalink:
 369                  // force use of single url:
 370                  $permalink_type = 'single';
 371              }
 372          }
 373          elseif( empty( $permalink_type ) )
 374          {    // Normal "in stream" post:
 375              // Use default from collection settings (may be an "in stream" URL):
 376              $permalink_type = $this->Blog->get_setting( 'permalinks' );
 377          }
 378  
 379          switch( $permalink_type )
 380          {
 381              case 'archive':
 382                  return $this->get_archive_url( $blogurl, $glue );
 383  
 384              case 'subchap':
 385                  return $this->get_chapter_url( $blogurl, $glue );
 386  
 387              case 'none':
 388                  // This is a silent fallback when we try to permalink to an Item that cannot be addressed directly:
 389                  // Link to blog home:
 390                  return $this->Blog->gen_blogurl();
 391  
 392              case 'cat':
 393                  // Link to permanent url of main chapter:
 394                  $this->get_main_Chapter();
 395                  return $this->main_Chapter->get_permanent_url( NULL, $blogurl, 1, NULL, $glue );
 396  
 397              case 'single':
 398              default:
 399                  return $this->get_single_url( true, $blogurl, $glue );
 400          }
 401      }

 404      /**
 405       * Template function: list all the category names
 406       *
 407       * @param string Output format for each cat, see {@link format_to_output()}
 408       */
 409  	function categories( $params = array() )
 410      {
 411          // Make sure we are not missing any param:
 412          $params = array_merge( array(
 413                  'before'          => ' ',
 414                  'after'           => ' ',
 415                  'include_main'    => true,
 416                  'include_other'   => true,
 417                  'include_external'=> true,
 418                  'before_main'     => '',       // string fo display before the MAIN category,
 419                  'after_main'      => '',       // string fo display after the MAIN category
 420                  'before_other'    => '',       // string fo display before OTHER categories
 421                  'after_other'     => '',       // string fo display after OTHER categories
 422                  'before_external' => '<em>',   // string fo display before EXTERNAL categories
 423                  'after_external'  => '</em>',  // string fo display after EXTERNAL categories,
 424                  'separator'       => ', ',
 425                  'link_categories' => true,
 426                  'link_title'      => '#',
 427                  'format'          => 'htmlbody',
 428                  'show_locked'     => false,
 429              ), $params );
 430  
 431  
 432          if( $params['link_title'] == '#' )
 433          { /* TRANS: When the categories for a specific post are displayed, the user can click
 434                      on these cats to browse them, this is the default href title displayed there */
 435              $params['link_title'] = T_('Browse category');
 436          }
 437  
 438          $categoryNames = array();
 439          foreach( $this->get_Chapters() as $Chapter )
 440          {
 441              $cat_name = $Chapter->dget( 'name' );
 442  
 443              if( $params['link_categories'] )
 444              { // we want to display links
 445                  $lBlog = & $Chapter->get_Blog();
 446                  $cat_name = '<a href="'.$Chapter->get_permanent_url().'" title="'.evo_htmlspecialchars($params['link_title']).'">'.$cat_name.'</a>';
 447              }
 448  
 449              if( $Chapter->ID == $this->main_cat_ID )
 450              { // We are displaying the main cat!
 451                  if( !$params['include_main'] )
 452                  { // ignore main cat !!!
 453                      continue;
 454                  }
 455                  $cat_name = $params['before_main'].$cat_name.$params['after_main'];
 456              }
 457              elseif( $Chapter->blog_ID == $this->blog_ID )
 458              { // We are displaying another cat in the same blog
 459                  if( !$params['include_other'] )
 460                  { // ignore main cat !!!
 461                      continue;
 462                  }
 463                  $cat_name = $params['before_other'].$cat_name.$params['after_other'];
 464              }
 465              else
 466              { // We are displaying an external cat (in another blog)
 467                  if( !$params['include_external'] )
 468                  { // ignore main cat !!!
 469                      continue;
 470                  }
 471                  $cat_name = $params['before_external'].$cat_name.$params['after_external'];
 472              }
 473  
 474              if( $Chapter->lock && $params[ 'show_locked' ] )
 475              {
 476                  $cat_name .= '<span style="padding-left:5px;" >'.get_icon( 'file_not_allowed', 'imgtag', array( 'title' => T_('Locked')) ).'</span>';
 477              }
 478  
 479              $categoryNames[] = $cat_name;
 480          }
 481  
 482          echo $params['before'];
 483          echo format_to_output( implode( $params['separator'], $categoryNames ), $params['format'] );
 484           echo $params['after'];
 485      }

 488      /**
 489       * Add nav_target param into the end of the url, but only if it is necessary
 490       *
 491       * @param string the url
 492       * @param string the current blog or current skin post_navigation setting
 493       * @param integer the ID of the navigation target
 494       * @param string glue
 495       * @return string the received url or the received url extended with the navigation param
 496       */
 497  	function add_navigation_param( $url, $post_navigation, $nav_target, $glue = '&amp;' )
 498      {
 499          if( empty( $url ) || empty( $nav_target ) )
 500          { // the url or the navigation target is not set we can't modify anything
 501              return $url;
 502          }
 503  
 504          switch( $post_navigation )
 505          {
 506              case 'same_category': // navigate through the selected category
 507                  if( $this->main_cat_ID != $nav_target )
 508                  {
 509                      $url = url_add_param( $url, 'cat='.$nav_target, $glue );
 510                  }
 511                  break;
 512  
 513              // 'same_tag' should be added here, with 'tag' param
 514  
 515              case 'same_author': // navigate through this item author's posts ( param not needed because a post always has only one author )
 516              case 'same_blog': // by default don't add any param
 517              default:
 518                  break;
 519          }
 520  
 521          return $url;
 522      }

 562      /**
 563       * Get list of Chapter objects.
 564       *
 565       * sam2kb> TODO: Cache item cat IDs into Item::categories property instead of global $cache_postcats
 566       *
 567       * @return array of {@link Chapter chapters} (references)
 568       */
 569  	function get_Chapters()
 570      {
 571          global $DB, $preview, $postIDlist, $cache_postcats;
 572  
 573          if( $preview )
 574          {    // Preview mode
 575              global $extracats, $post_category;
 576  
 577              $extracats = param( 'post_extracats', 'array/integer', array() );
 578              if( ( ! empty( $post_category ) ) && ( ! in_array( $post_category, $extracats ) ) )
 579              {
 580                  $extracats[] = $post_category;
 581              }
 582              $categoryIDs = $extracats;
 583          }
 584          elseif( empty( $this->ID ) )
 585          { // This item doesn't exist yet
 586              $categoryIDs = NULL;
 587          }
 588          else
 589          {
 590              $search_post_ids = empty( $postIDlist ) ? NULL : explode( ',', $postIDlist );
 591              if( empty( $search_post_ids ) || !in_array( $this->ID, $search_post_ids ) )
 592              {    // Load cats for current item
 593                  $categoryIDs = postcats_get_byID( $this->ID );
 594              }
 595              else
 596              {    // Load cats for items list
 597                  if( ! isset($cache_postcats[$this->ID]) )
 598                  {    // Add to cache
 599                      $sql = "SELECT postcat_post_ID, postcat_cat_ID
 600                              FROM T_postcats
 601                              WHERE postcat_post_ID IN ($postIDlist)
 602                              ORDER BY postcat_post_ID, postcat_cat_ID";
 603  
 604                      foreach( $DB->get_results( $sql, ARRAY_A, 'Get categories for items' ) as $row )
 605                      {
 606                          $postcat_post_ID = $row['postcat_post_ID'];
 607                          if( ! isset( $cache_postcats[$postcat_post_ID] ) )
 608                          {
 609                               $cache_postcats[$postcat_post_ID] = array();
 610                          }
 611                          $cache_postcats[$postcat_post_ID][] = $row['postcat_cat_ID'];
 612                      }
 613                  }
 614                  $categoryIDs = $cache_postcats[$this->ID];
 615              }
 616          }
 617  
 618          $chapters = array();
 619          if( ! empty( $categoryIDs ) )
 620          {
 621              $ChapterCache = & get_ChapterCache();
 622              // Load all required Chapters
 623              $ChapterCache->load_list( $categoryIDs );
 624  
 625              foreach( $categoryIDs as $cat_ID )
 626              {
 627                  if( $Chapter = & $ChapterCache->get_by_ID( $cat_ID, false ) )
 628                  {
 629                      $chapters[] = $Chapter;
 630                  }
 631              }
 632          }
 633  
 634          return $chapters;
 635      }

 643      function & get_main_Chapter()
 644      {
 645          if( is_null( $this->main_Chapter ) )
 646          {
 647              $ChapterCache = & get_ChapterCache();
 648              /**
 649               * @var Chapter
 650               */
 651              $this->main_Chapter = & $ChapterCache->get_by_ID( $this->main_cat_ID, false );
 652              if( empty( $this->main_Chapter ) )
 653              {    // If main chapter is broken we should get it from one of extra chapters
 654                  $chapters = $this->get_Chapters();
 655                  foreach( $chapters as $Chapter )
 656                  {
 657                      if( !empty( $Chapter ) )
 658                      {    // We have found a valid Chapter...
 659                          $this->main_Chapter = & $Chapter;
 660                          $this->main_cat_ID = $Chapter->ID;
 661                          break;
 662                      }
 663                  }
 664              }
 665              if( empty( $this->main_Chapter ) )
 666              {    // If we still don't have a valid Chapter, display clean error and die().
 667                  global $admin_url, $Blog, $blog;
 668                  if( empty( $Blog ) )
 669                  {
 670                      if( !empty( $blog ) )
 671                      {
 672                          $BlogCache = & get_BlogCache();
 673                          $Blog = & $BlogCache->get_by_ID( $blog, false );
 674                      }
 675                  }
 676  
 677                  $url_to_edit_post = $admin_url.'?ctrl=items&amp;action=edit&amp;p='.$this->ID;
 678  
 679                  if( !empty( $Blog ) )
 680                  {
 681                      $url_to_edit_post .= '&amp;blog='.$Blog->ID;
 682                      if( is_admin_page() )
 683                      {    // Try to set a main category
 684                          $default_cat_ID = $Blog->get_setting( 'default_cat_ID' );
 685                          if( !empty( $default_cat_ID ) )
 686                          {    // If default category is set
 687                              $this->main_cat_ID = $default_cat_ID;
 688                              $this->main_Chapter = & $ChapterCache->get_by_ID( $this->main_cat_ID, false );
 689                          }
 690                          else
 691                          {    // Set from first chapter of the blog
 692                              $ChapterCache->clear();
 693                              $ChapterCache->load_subset( $Blog->ID );
 694                              if( $Chapter = & $ChapterCache->get_next() )
 695                              {
 696                                  $this->main_cat_ID = $Chapter->ID;
 697                                  $this->main_Chapter = & $Chapter;
 698                              }
 699                          }
 700                      }
 701                  }
 702  
 703                  $message = sprintf( 'Item with ID <a %s>%s</a> has an invalid main category ID %s.',
 704                          'href="'.$url_to_edit_post.'"',
 705                          $this->ID,
 706                          $this->main_cat_ID
 707                      );
 708                  if( empty( $Blog ) )
 709                  {    // No blog defined
 710                      $message .= ' In addition we cannot fallback to the default category because no valid blog ID has been specified.';
 711                  }
 712  
 713                  if( empty( $this->main_Chapter ) )
 714                  {    // Main chapter is not defined, because blog doesn't have the default cat ID and even blog doesn't have any categories
 715                      debug_die( $message );
 716                  }
 717                  else
 718                  {    // Main chapter is defined, we can show the page
 719                      global $Messages;
 720                      $Messages->add( $message );
 721                  }
 722              }
 723          }
 724  
 725          return $this->main_Chapter;
 726      }

 947       * Check if cross post navigation should stay in the current blog or not.
 948       * Also check that this item has at least one category that belongs to the given blog.
 949       * If current blog is the same as item blog then, this function will return false, because no need to check.
 950       *
 951       * @param string 'auto' value means this call needs to decide to stay in the current blog or not. Every other value will return false!
 952       * @param integer the given "current" blog ID (its usually the current blog id)

 954       */
 955  	function check_cross_post_nav( $target_blog, $blog_ID )
 956      {
 957          global $cross_post_nav_in_same_blog;
 958  
 959          if( $target_blog != 'auto' )
 960          { // target_blog is not set to auto, we have to navigate to the item's main cat's blog.
 961              return false;
 962          }
 963  
 964          $this->get_Blog();
 965          if( $this->Blog->ID == $blog_ID )
 966          { // item's blog is the same as target blog

 971          { // we have to navigate to the item's main cat's blog.
 972              return false;
 973          }

1094      /**
1095       * Get "nice" title of the Item
1096       * @return string
1097       */
1098  	function get_title( $params = array() )
1099      {
1100          global $ReqURL, $Blog, $MainList;
1101  
1102          // Set default post navigation
1103          $def_post_navigation = empty( $Blog ) ? 'same_blog' : $Blog->get_setting( 'post_navigation' );
1104  
1105          // Make sure we are not missing any param:
1106          $params = array_merge( array(
1107                  'before'          => '',
1108                  'after'           => '',
1109                  'format'          => 'htmlbody',
1110                  'link_type'       => '#',
1111                  'link_class'      => '#',
1112                  'max_length'      => '',
1113                  'target_blog'     => '',
1114                  'nav_target'      => NULL,
1115                  'post_navigation' => $def_post_navigation,
1116                  'title_field'     => 'title',
1117              ), $params );
1118  
1119          // Set post navigation target
1120          $nav_target = ( ( $params['nav_target'] === NULL ) && isset($MainList) && !empty( $MainList->nav_target ) ) ? $MainList->nav_target : $params['nav_target'];
1121  
1122          $blogurl = '';
1123          if( !empty($Blog) && $this->check_cross_post_nav( $params['target_blog'], $Blog->ID ) )
1124          {
1125              $blogurl = $Blog->gen_blogurl();
1126          }
1127  
1128          $title = format_to_output( $this->$params['title_field'], $params['format'] );
1129  
1130          if( $params['max_length'] != '' )
1131          {    // Crop long title
1132              $title = strmaxlen( $title, intval($params['max_length']) );
1133          }
1134  
1135          if( empty( $title ) )
1136          {
1137              return;
1138          }
1139  
1140          if( $params['link_type'] == '#' )
1141          {    // Use default link type from settings:
1142              if( $this->is_intro() )
1143              {    // This is an intro, do not link title by default:
1144                  $params['link_type'] = 'none';
1145              }
1146              elseif( is_same_url( $this->get_permanent_url( '', $blogurl, '&' ), $ReqURL ) )
1147              {    // We are on the single url already:
1148                  $params['link_type'] = 'none';
1149              }
1150              else if( $this->ptyp_ID == 3000 )
1151              {    // tblue> This is a sidebar link, link to its "link to" URL by default:
1152                  $params['link_type'] = 'linkto_url';
1153              }
1154              else
1155              {    // This is a normal post: use default link strategy from Blog settings:
1156                  $this->get_Blog();
1157                  $params['link_type'] = $this->Blog->get_setting( 'title_link_type' );
1158              }
1159          }
1160  
1161          switch( $params['link_type'] )
1162          {
1163              case 'auto':
1164                  $url = ( empty($this->url) ? $this->get_permanent_url() : $this->url );
1165                  break;
1166  
1167              case 'permalink':
1168                  $url = $this->get_permanent_url( '', $blogurl );
1169                  break;
1170  
1171              case 'linkto_url':
1172                  $url = $this->url;
1173                  break;
1174  
1175              case 'admin_view':
1176                  $url = '?ctrl=items&amp;blog='.$this->get_blog_ID().'&amp;p='.$this->ID;
1177                  break;
1178  
1179              case 'none':
1180              default:
1181          }
1182  
1183          if( !empty( $url ) )
1184          { // url is set, also add navigation param if it is necessary
1185              $url = $this->add_navigation_param( $url, $params['post_navigation'], $nav_target );
1186          }
1187  
1188          $link_class = '';
1189          if( $params['link_class'] != '#' )
1190          {
1191              $link_class = ' class="'.$params['link_class'].'"';
1192          }
1193  
1194          $r = $params['before'];
1195          if( !empty($url) )
1196          {
1197              $r .= '<a href="'.$url.'"'.$link_class.'>'.$title.'</a>';
1198          }
1199          else
1200          {
1201              $r .= $title;
1202          }
1203          $r .= $params['after'];
1204          return $r;
1205      }
