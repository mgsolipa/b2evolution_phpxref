 259  	function query()
 260      {
 261          global $DB;
 262  
 263          if( !is_null( $this->rows ) )
 264          { // Query has already executed:
 265              return;
 266          }
 267  
 268          // INIT THE QUERY:
 269          $this->query_init();
 270  
 271          // Check the number of totla rows after it was initialized in the query_init() function
 272          if( isset( $this->total_rows ) && ( intval( $this->total_rows ) === 0 ) )
 273          { // Count query was already executed and returned 0
 274              return;
 275          }
 276  
 277          $select_temp_order = '';
 278          if( !empty( $this->ItemQuery->order_by ) && strpos( $this->ItemQuery->order_by, 'post_order' ) !== false )
 279          {    // Move the items with NULL order to the end of the list
 280              $select_temp_order = ', IF( post_order IS NULL, 999999999, post_order ) AS temp_order';
 281              $this->ItemQuery->ORDER_BY( str_replace( 'post_order', 'temp_order', $this->ItemQuery->get_order_by( '' ) ) );
 282          }
 283  
 284          // Results style orders:
 285          // $this->ItemQuery->ORDER_BY_prepend( $this->get_order_field_list() );
 286  
 287  
 288          // We are going to proceed in two steps (we simulate a subquery)
 289          // 1) we get the IDs we need
 290          // 2) we get all the other fields matching these IDs
 291          // This is more efficient than manipulating all fields at once.
 292  
 293          // *** STEP 1 ***
 294          // walter> Accordding to the standart, to DISTINCT queries, all columns used
 295          // in ORDER BY must appear in the query. This make que query work with PostgreSQL and
 296          // other databases.
 297          // fp> That can dramatically fatten the returned data. You must handle this in the postgres class (check that order fields are in select)
 298          $step1_sql = 'SELECT DISTINCT '.$this->Cache->dbIDname // .', '.implode( ', ', $order_cols_to_select )
 299                                      .$select_temp_order
 300                                      .$this->ItemQuery->get_from()
 301                                      .$this->ItemQuery->get_where()
 302                                      .$this->ItemQuery->get_group_by()
 303                                      .$this->ItemQuery->get_order_by()
 304                                      .$this->ItemQuery->get_limit();
 305  
 306          // echo $DB->format_query( $step1_sql );
 307  
 308          // Get list of the IDs we need:
 309          $ID_list = implode( ',', $DB->get_col( $step1_sql, 0, 'ItemList2::Query() Step 1: Get ID list' ) );
 310  
 311          // *** STEP 2 ***
 312          $this->sql = 'SELECT *'.$select_temp_order.'
 313                            FROM '.$this->Cache->dbtablename;
 314          if( !empty($ID_list) )
 315          {
 316              $this->sql .= ' WHERE '.$this->Cache->dbIDname.' IN ('.$ID_list.') '
 317                                          .$this->ItemQuery->get_order_by();
 318          }
 319          else
 320          {
 321              $this->sql .= ' WHERE 0';
 322          }
 323  
 324          //echo $DB->format_query( $this->sql );
 325  
 326          // ATTENTION: we skip the parent on purpose here!! fp> refactor
 327          DataObjectList2::query( false, false, false, 'ItemList2::Query() Step 2' );
 328      }

 389      /**
 390       * Returns values needed to make sort links for a given column
 391       * Needed because the order is not handled by the result class.
 392       * Reason: Sometimes the item list needs to be ordered without having a display table, and columns. The result class order is based on columns.
 393       *
 394       * Returns an array containing the following values:
 395       *  - current_order : 'ASC', 'DESC' or ''
 396       *  - order_asc : url needed to order in ascending order
 397       *  - order_desc
 398       *  - order_toggle : url needed to toggle sort order
 399       *
 400       * @param integer column to sort
 401       * @return array
 402       */
 403  	function get_col_sort_values( $col_idx )
 404      {
 405          $col_order_fields = $this->cols[$col_idx]['order'];
 406  
 407          // Current order:
 408          if( $this->filters['orderby'] == $col_order_fields || $this->param_prefix.$this->filters['orderby'] == $col_order_fields )
 409          {
 410              $col_sort_values['current_order'] = $this->filters['order'];
 411          }
 412          else
 413          {
 414              $col_sort_values['current_order'] = '';
 415          }
 416  
 417  
 418          // Generate sort values to use for sorting on the current column:
 419          $col_sort_values['order_asc'] = regenerate_url( array($this->param_prefix.'order',$this->param_prefix.'orderby'),
 420                                                                              $this->param_prefix.'order=ASC&amp;'.$this->param_prefix.'orderby='.$col_order_fields );
 421          $col_sort_values['order_desc'] = regenerate_url(  array($this->param_prefix.'order',$this->param_prefix.'orderby'),
 422                                                                              $this->param_prefix.'order=DESC&amp;'.$this->param_prefix.'orderby='.$col_order_fields );
 423  
 424          if( !$col_sort_values['current_order'] && isset( $this->cols[$col_idx]['default_dir'] ) )
 425          {    // There is no current order on this column and a default order direction is set for it
 426              // So set a default order direction for it
 427  
 428              if( $this->cols[$col_idx]['default_dir'] == 'A' )
 429              {    // The default order direction is A, so set its toogle  order to the order_asc
 430                  $col_sort_values['order_toggle'] = $col_sort_values['order_asc'];
 431              }
 432              else
 433              { // The default order direction is A, so set its toogle order to the order_desc
 434                  $col_sort_values['order_toggle'] = $col_sort_values['order_desc'];
 435              }
 436          }
 437          elseif( $col_sort_values['current_order'] == 'ASC' )
 438          {    // There is an ASC current order on this column, so set its toogle order to the order_desc
 439              $col_sort_values['order_toggle'] = $col_sort_values['order_desc'];
 440          }
 441          else
 442          { // There is a DESC or NO current order on this column,  so set its toogle order to the order_asc
 443              $col_sort_values['order_toggle'] = $col_sort_values['order_asc'];
 444          }
 445  
 446          // pre_dump( $col_sort_values );
 447  
 448          return $col_sort_values;
 449      }

 456  	function prevnext_item_links( $params )
 457      {
 458          global $posttypes_specialtypes;
 459  
 460          $params = array_merge( array(
 461                                      'template' => '$prev$$separator$$next$',
 462                                      'prev_start' => '',
 463                                      'prev_text' => '&laquo; $title$',
 464                                      'prev_end' => '',
 465                                      'prev_no_item' => '',
 466                                      'prev_class' => '',
 467                                      'separator' => '',
 468                                      'next_start' => '',
 469                                      'next_text' => '$title$ &raquo;',
 470                                      'next_end' => '',
 471                                      'next_no_item' => '',
 472                                      'next_class' => '',
 473                                      'target_blog' => '',
 474                                      'post_navigation' => $this->Blog->get_setting( 'post_navigation' ),
 475                                      'types' => '-'.implode(',',$posttypes_specialtypes), // Exclude pages, intros & sidebar stuff
 476                                      'featured' => NULL,
 477                                  ), $params );
 478  
 479          $current_Item = & $this->get_by_idx(0);
 480          // Note: current_Item may be null when User doesn't have permission
 481          if( $current_Item )
 482          { // current Item is available, init navigation target
 483              switch( $params['post_navigation'] )
 484              {
 485                  case 'same_category': // sometimes requires the 'cat' param because a post may belong to multiple categories
 486                      if( empty( $this->nav_target ) )
 487                      {
 488                          $this->nav_target = $current_Item->main_cat_ID;
 489                      }
 490                      $this->filters['cat_array'][] = $this->nav_target;
 491                      // Note: If there will be other navigation type ( like tag ) with params, those filters must be removed.
 492                      break;
 493  
 494                  case 'same_author': // This doesn't require extra param because a post always has only one author
 495                      $this->filters['authors'] = $current_Item->creator_user_ID;
 496                      // reset cat filters because only the authors are important
 497                      $this->filters['cat_array'] = array();
 498                      break;
 499  
 500                  default:
 501                      break;
 502              }
 503          }
 504  
 505          $prev = $this->prev_item_link( $params['prev_start'], $params['prev_end'], $params[ 'prev_text' ], $params[ 'prev_no_item' ], false, $params[ 'target_blog'], $params['prev_class'], $params['types'], $params['featured'], $params['post_navigation'] );
 506          $next = $this->next_item_link( $params['next_start'], $params['next_end'], $params[ 'next_text' ], $params[ 'next_no_item' ], false, $params[ 'target_blog'], $params['next_class'], $params['types'], $params['featured'], $params['post_navigation'] );
 507  
 508          if( empty( $prev ) || empty( $next ) )
 509          {    // Use separator text only when prev & next are not empty
 510              $params['separator'] = '';
 511          }
 512  
 513          $output = str_replace( '$prev$', $prev, $params['template'] );
 514          $output = str_replace( '$next$', $next, $output );
 515          $output = str_replace( '$separator$', $params['separator'], $output );
 516  
 517          if( !empty( $output ) )
 518          {    // we have some output, lets wrap it
 519              echo( $params['block_start'] );
 520              echo $output;
 521              echo( $params['block_end'] );
 522          }
 523      }

 564              $output .= $next_Item->get_permanent_link( $text, '#', $class, $target_blog, $post_navigation, $this->nav_target );
 565              $output .= $after;
 566          }
 567          else
 568          {
 569              $output = $no_item;
 570          }
 571          if( $display ) echo $output;
 572          return $output;
 573      }

 603       * Generate the permalink for the next item in collection.
 604       *

 607       * Note: This actually only returns the URL, to get a real link, use {@link ItemList::next_item_link()}

 628      /**
 629       * Skip to previous/next Item
 630       *
 631       * If several items share the same spot (like same issue datetime) then they'll get all skipped at once.
 632       *
 633       * @param string prev | next  (relative to the current sort order)
 634       */
 635      function & get_prevnext_Item( $direction = 'next', $types = '', $featured = NULL, $post_navigation = 'same_blog' )
 636      {
 637          global $DB, $ItemCache, $posttypes_specialtypes;
 638  
 639          if( ! $this->single_post )
 640          {    // We are not on a single post:
 641              $r = NULL;
 642              return $r;
 643          }
 644  
 645          /**
 646           * @var Item
 647           */
 648          $current_Item = $this->get_by_idx(0);
 649  
 650          if( is_null($current_Item) )
 651          {    // This happens if we are on a single post that we do not actually have permission to view
 652              $r = NULL;
 653              return $r;
 654          }
 655  
 656          if( in_array( $current_Item->ptyp_ID, $posttypes_specialtypes ) ) // page, intros, ads
 657          {    // We are not on a REGULAR post -- we cannot navigate:
 658              $r = NULL;
 659              return $r;
 660          }
 661  
 662          if( !empty( $this->prevnext_Item[$direction][$post_navigation] ) )
 663          {
 664              return $this->prevnext_Item[$direction][$post_navigation];
 665          }
 666  
 667          $next_Query = new ItemQuery( $this->Cache->dbtablename, $this->Cache->dbprefix, $this->Cache->dbIDname );
 668  
 669          // GENERATE THE QUERY:
 670  
 671          /*
 672           * filtering stuff:
 673           */
 674          $next_Query->where_chapter2( $this->Blog, $this->filters['cat_array'], $this->filters['cat_modifier'],
 675                                                                   $this->filters['cat_focus'] );
 676          $next_Query->where_author( $this->filters['authors'] );
 677          $next_Query->where_author_logins( $this->filters['authors_login'] );
 678          $next_Query->where_assignees( $this->filters['assignees'] );
 679          $next_Query->where_assignees_logins( $this->filters['assignees_login'] );
 680          $next_Query->where_author_assignee( $this->filters['author_assignee'] );
 681          $next_Query->where_locale( $this->filters['lc'] );
 682          $next_Query->where_statuses( $this->filters['statuses'] );
 683          // types param is kept only for the case when some custom types should be displayed
 684          $next_Query->where_types( !empty( $types ) ? $types : $this->filters['types'] );
 685          $next_Query->where_keywords( $this->filters['keywords'], $this->filters['phrase'], $this->filters['exact'] );
 686          // $next_Query->where_ID( $this->filters['post_ID'], $this->filters['post_title'] );
 687          $next_Query->where_datestart( $this->filters['ymdhms'], $this->filters['week'],
 688                                             $this->filters['ymdhms_min'], $this->filters['ymdhms_max'],
 689                                             $this->filters['ts_min'], $this->filters['ts_max'] );
 690          $next_Query->where_visibility( $this->filters['visibility_array'] );
 691          $next_Query->where_featured( $featured );
 692  
 693          /*
 694           * ORDER BY stuff:
 695           */
 696          if( ($direction == 'next' && $this->filters['order'] == 'DESC')
 697              || ($direction == 'prev' && $this->filters['order'] == 'ASC') )
 698          {
 699              $order = 'DESC';
 700              $operator = ' < ';
 701          }
 702          else
 703          {
 704              $order = 'ASC';
 705              $operator = ' > ';
 706          }
 707  
 708          $orderby = str_replace( ' ', ',', $this->filters['orderby'] );
 709          $orderby_array = explode( ',', $orderby );
 710  
 711          // Format each order param with default column names:
 712          $orderbyorder_array = preg_replace( '#^(.+)$#', $this->Cache->dbprefix.'$1 '.$order, $orderby_array );
 713  
 714          // Add an ID parameter to make sure there is no ambiguity in ordering on similar items:
 715          $orderbyorder_array[] = $this->Cache->dbIDname.' '.$order;
 716  
 717          $order_by = implode( ', ', $orderbyorder_array );
 718  
 719          // Special case for RAND:
 720          $order_by = str_replace( $this->Cache->dbprefix.'RAND ', 'RAND() ', $order_by );
 721  
 722          $next_Query->order_by( $order_by );
 723  
 724  
 725          // LIMIT to 1 single result
 726          $next_Query->LIMIT( '1' );
 727  
 728          // fp> TODO: I think some additional limits need to come back here (for timespans)
 729  
 730  
 731          /*
 732           * Position right after the current element depending on current sorting params
 733           *
 734           * If there are several items on the same issuedatetime for example, we'll then differentiate on post ID
 735           * WARNING: you cannot combine criterias with AND here; you need stuf like a>a0 OR (a=a0 AND b>b0)
 736           */
 737          switch( $orderby_array[0] )
 738          {
 739              case 'datestart':
 740                  // special var name:
 741                  $next_Query->WHERE_and( $this->Cache->dbprefix.$orderby_array[0]
 742                                                                  .$operator
 743                                                                  .$DB->quote($current_Item->issue_date)
 744                                                                  .' OR ( '
 745                                    .$this->Cache->dbprefix.$orderby_array[0]
 746                                                                      .' = '
 747                                                                      .$DB->quote($current_Item->issue_date)
 748                                                                      .' AND '
 749                                                                      .$this->Cache->dbIDname
 750                                                                      .$operator
 751                                                                      .$current_Item->ID
 752                                                                  .')'
 753                                                           );
 754                  break;
 755  
 756              case 'title':
 757              case 'ptyp_ID':
 758              case 'datecreated':
 759              case 'datemodified':
 760              case 'last_touched_ts':
 761              case 'urltitle':
 762              case 'priority':
 763                  $next_Query->WHERE_and( $this->Cache->dbprefix.$orderby_array[0]
 764                                                                  .$operator
 765                                                                  .$DB->quote($current_Item->{$orderby_array[0]})
 766                                                                  .' OR ( '
 767                                    .$this->Cache->dbprefix.$orderby_array[0]
 768                                                                      .' = '
 769                                                                      .$DB->quote($current_Item->{$orderby_array[0]})
 770                                                                      .' AND '
 771                                                                      .$this->Cache->dbIDname
 772                                                                      .$operator
 773                                                                      .$current_Item->ID
 774                                                                  .')'
 775                                                              );
 776                  break;
 777  
 778              case 'order':
 779                  // We have to integrate a rounding error margin
 780                  $comp_order_value = $current_Item->order;
 781                  $and_clause = '';
 782  
 783                  if( is_null($comp_order_value) )
 784                  {    // current Item has NULL order
 785                      if( $operator == ' < ' )
 786                      {    // This is needed when browsing through a descending ordered list and we reach the limit where orders are not set/NULL (ex: b2evo screenshots)
 787                          $and_clause .= $this->Cache->dbprefix.$orderby_array[0].' IS NULL AND ';
 788                      }
 789                      else
 790                      { // This is needed when browsing through a descending ordered list and we want to browse back into the posts that have numbers (pb appears if first NULL posts is the highest ID)
 791                          $and_clause .= $this->Cache->dbprefix.$orderby_array[0].' IS NOT NULL OR ';
 792                      }
 793                      $and_clause .= $this->Cache->dbIDname
 794                          .$operator
 795                          .$current_Item->ID;
 796                  }
 797                  else
 798                  {
 799                      if( $operator == ' < ' )
 800                      {    // This is needed when browsing through a descending ordered list and we reach the limit where orders are not set/NULL (ex: b2evo screenshots)
 801                          $and_clause .= $this->Cache->dbprefix.$orderby_array[0].' IS NULL OR ';
 802                      }
 803                      $and_clause .= $this->Cache->dbprefix.$orderby_array[0]
 804                                                      .$operator
 805                                                      .( $operator == ' < ' ? $comp_order_value-0.000000001 : $comp_order_value+0.000000001 )
 806                                                      .' OR ( '
 807                                .$this->Cache->dbprefix.$orderby_array[0]
 808                                                          .( $operator == ' < ' ? ' <= '.($comp_order_value+0.000000001) : ' >= '.($comp_order_value-0.000000001) )
 809                                                          .' AND '
 810                                                          .$this->Cache->dbIDname
 811                                                          .$operator
 812                                                          .$current_Item->ID
 813                                                      .')';
 814                  }
 815  
 816                  $next_Query->WHERE_and( $and_clause );
 817                  break;
 818  
 819              case 'RAND':
 820                  // Random order. Don't show current item again.
 821                  $next_Query->WHERE_and( $this->Cache->dbprefix.'ID <> '.$current_Item->ID );
 822                  break;
 823  
 824              default:
 825                  echo 'WARNING: unhandled sorting: '.evo_htmlspecialchars( $orderby_array[0] );
 826          }
 827  
 828          // GET DATA ROWS:
 829  
 830  
 831          // We are going to proceed in two steps (we simulate a subquery)
 832          // 1) we get the IDs we need
 833          // 2) we get all the other fields matching these IDs
 834          // This is more efficient than manipulating all fields at once.
 835  
 836          // Step 1:
 837          $step1_sql = 'SELECT DISTINCT '.$this->Cache->dbIDname
 838                                      .$next_Query->get_from()
 839                                      .$next_Query->get_where()
 840                                      .$next_Query->get_group_by()
 841                                      .$next_Query->get_order_by()
 842                                      .$next_Query->get_limit();
 843  
 844          //echo $DB->format_query( $step1_sql );
 845  
 846          // Get list of the IDs we need:
 847          $next_ID = $DB->get_var( $step1_sql, 0, 0, 'Get ID of next item' );
 848  
 849          //pre_dump( $next_ID );
 850  
 851          // Step 2: get the item (may be NULL):
 852          $this->prevnext_Item[$direction][$post_navigation] = & $ItemCache->get_by_ID( $next_ID, true, false );
 853  
 854          return $this->prevnext_Item[$direction][$post_navigation];
 855  
 856      }
 857  }
