  83  function fcpf_categories_select( $parent_category_ID = -1, $level = 0 )
  84  {
  85      global $blog, $DB;
  86      $result_Array = array();
  87  
  88      $SQL = new SQL();
  89      $SQL->SELECT( 'cat_ID, cat_name' );
  90      $SQL->FROM( 'T_categories' );
  91      $SQL->WHERE( 'cat_blog_ID = '.$DB->quote( $blog ) );
  92      if( $parent_category_ID == -1 )
  93      {
  94          $SQL->WHERE_and( 'cat_parent_ID IS NULL' );
  95      }
  96      else
  97      {
  98          $SQL->WHERE( 'cat_parent_ID = '.$DB->quote( $parent_category_ID ) );
  99      }
 100      $SQL->ORDER_BY( 'cat_name' );
 101      $categories = $DB->get_results( $SQL->get() );
 102  
 103      if( ! empty( $categories ) )
 104      {
 105          foreach( $categories as $category )
 106          {
 107              $result_Array[] = array(
 108                      'value' => $category->cat_ID,
 109                      'label' => str_repeat( '&nbsp;&nbsp;&nbsp;', $level ).$category->cat_name
 110                  );
 111  
 112              $child_Categories_opts = fcpf_categories_select( $category->cat_ID, $level + 1 );
 113              if( $child_Categories_opts != '' )
 114              {
 115                  foreach( $child_Categories_opts as $cat )
 116                  {
 117                      $result_Array[] = $cat;
 118                  }
 119              }
 120          }
 121      }
 122      return $result_Array;
 123  }
