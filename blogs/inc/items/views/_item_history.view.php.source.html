  92      $iver_compare_selector[ $input_name ]++;
  93  
  94      $style = '';

  96          ( $iver_compare_selector[ $input_name ] > $selected && $direction == 'down' ) )
  97      {    // Hide inputs
  98          $style = ' style="display:none"';

 103      {    // Select this input
 104          $checked = ' checked="checked"';
 105      }
 106
