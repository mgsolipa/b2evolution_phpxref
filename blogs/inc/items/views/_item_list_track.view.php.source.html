  67  /**
  68   * Get title of the item/task cell by field type
  69   *
  70   * @param string Type of the field: 'priority', 'status', 'assigned'
  71   * @param object Item
  72   * @param integer Priority
  73   * @return string
  74   */
  75  function td_task_cell( $type, $Item )
  76  {
  77      global $current_User;
  78  
  79      switch( $type )
  80      {
  81          case 'priority':
  82              $value = $Item->priority;
  83              $title = item_priority_title( $Item->priority );
  84              break;
  85  
  86          case 'status':
  87              $value = $Item->pst_ID;
  88              $title = $Item->get( 't_extra_status' );
  89              if( empty( $title ) )
  90              {
  91                  $title = T_('No status');
  92              }
  93              break;
  94  
  95          case 'assigned':
  96              $value = $Item->assigned_user_ID;
  97              if( empty( $value ) )
  98              {
  99                  $title = T_('No user');
 100              }
 101              else
 102              {
 103                  $UserCache = & get_UserCache();
 104                  $User = & $UserCache->get_by_ID( $Item->assigned_user_ID );
 105                  $title = $User->get_colored_login( array( 'mask' => '$avatar$ $login$' ) );
 106              }
 107              break;
 108  
 109          default:
 110              $value = 0;
 111              $title = '';
 112      }
 113  
 114      if( $current_User->check_perm( 'item_post!CURSTATUS', 'edit', false, $Item ) )
 115      { // Current user can edit this item
 116          return '<a href="#" rel="'.$value.'">'.$title.'</a>';
 117      }
 118      else
 119      { // No perms to edit item, Display only a title
 120          return $title;
 121      }
 122  }
