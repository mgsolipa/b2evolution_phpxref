 139  	function get_default_group_permissions( $grp_ID )
 140      {
 141          switch( $grp_ID )
 142          {
 143              case 1: // Administrators group ID equals 1
 144                  global $test_install_all_features;
 145                  $perm_messaging = $test_install_all_features ? 'abuse' : 'delete';
 146                  $max_new_threads = ''; // empty = no limit
 147                  break;
 148              case 2: // Moderators group equals 2
 149              case 3: // Trusted users group ID equals 3
 150                  $perm_messaging = 'write'; // Messaging permissions
 151                  $max_new_threads = '10'; // Maximum number of new threads per day
 152                  break;
 153              case 4: // Normal users group ID equals 4
 154                  $perm_messaging = 'write';
 155                  $max_new_threads = '5';
 156                  break;
 157              case 5:        // Misbehaving/Suspect users (group ID 5) have permission by default:
 158                  $perm_messaging = 'write';
 159                  $max_new_threads = '1';
 160                  break;
 161              case 6:  // Spammers/restricted Users
 162              default: // Other groups
 163                  $perm_messaging = 'reply';
 164                  $max_new_threads = '1';
 165                  break;
 166          }
 167  
 168          // We can return as many default permissions as we want:
 169          // e.g. array ( permission_name => permission_value, ... , ... )
 170          return $permissions = array( 'perm_messaging' => $perm_messaging, 'max_new_threads' => $max_new_threads );
 171      }

 241       * Check a permission for the group. ( see 'group_func' in get_available_group_permissions() function )
 242       *
 243       * @param string Requested permission level
 244       * @param string Permission value
 245       * @param mixed Permission target (blog ID, array of cat IDs...)
 246       * @return boolean True on success (permission is granted), false if permission is not granted
 247       */
 248  	function check_messaging_group_perm( $permlevel, $permvalue, $permtarget )
 249      {
 250          $perm = false;
 251          switch ( $permvalue )
 252          {
 253              case 'abuse':
 254                  // Abuse Management & Read, Send & Delete any messages (able to send messages even for blocked contacts)
 255                  if( $permlevel == 'abuse' )
 256                  {
 257                      $perm = true;
 258                      break;
 259                  }
 260              case 'delete':
 261                  // Read, Send & Delete any messages (able to send messages even for blocked contacts)
 262                  if( $permlevel == 'delete' )
 263                  { // User can ask for delete perm...
 264                      $perm = true;
 265                      break;
 266                  }
 267              case 'write':
 268                  // Read & Send messages to anyone (except for blocked contacts)
 269                  if( $permlevel == 'write' )
 270                  {
 271                      $perm = true;
 272                      break;
 273                  }
 274              case 'reply':
 275                  //  Read & Send messages to people in contacts list only (except for blocked contacts)
 276                  if( $permlevel == 'reply' )
 277                  {
 278                      $perm = true;
 279                      break;
 280                  }
 281          }
 282  
 283          return $perm;
 284      }

 381  	function handle_htsrv_action()
 382      {
 383          global $current_User, $Blog, $Session, $Messages, $samedomain_htsrv_url;
 384  
 385          // Init objects we want to work on.
 386          $action = param_action( true, true );
 387          $disp = param( 'disp', '/^[a-z0-9\-_]+$/', 'threads' );
 388  
 389          // Check that this action request is not a CSRF hacked request:
 390          $Session->assert_received_crumb( 'messaging_'.$disp );
 391  
 392          // Load classes
 393          load_class( 'messaging/model/_thread.class.php', 'Thread' );
 394          load_class( 'messaging/model/_message.class.php', 'Message' );
 395  
 396          if( !is_logged_in() )
 397          { // user must be logged in
 398              debug_die( 'User must be logged in to proceed with messaging updates!' );
 399          }
 400  
 401          // Check permission:
 402          $current_User->check_perm( 'perm_messaging', 'reply', true );
 403  
 404          // set where to redirect
 405          $redirect_to = param( 'redirect_to', 'url', NULL );
 406          if( empty( $redirect_to ) )
 407          {
 408              if( isset( $Blog ) )
 409              {
 410                  $redirect_to = url_add_param( $Blog->gen_baseurl(), 'disp='.$disp );
 411              }
 412              else
 413              {
 414                  $redirect_to = url_add_param( $baseurl, 'disp='.$disp );
 415              }
 416          }
 417  
 418          if( ( $disp != 'contacts' ) && ( $thrd_ID = param( 'thrd_ID', 'integer', '', true ) ) )
 419          {// Load thread from cache:
 420              $ThreadCache = & get_ThreadCache();
 421              if( ($edited_Thread = & $ThreadCache->get_by_ID( $thrd_ID, false )) === false )
 422              {    unset( $edited_Thread );
 423                  forget_param( 'thrd_ID' );
 424                  $Messages->add( sprintf( T_('Requested &laquo;%s&raquo; object does not exist any longer.'), T_('Thread') ), 'error' );
 425                  $action = 'nil';
 426              }
 427          }
 428  
 429          switch( $disp )
 430          {
 431              // threads action
 432              case 'threads':
 433                  if( $action != 'create' )
 434                  { // Make sure we got a thrd_ID:
 435                      param( 'thrd_ID', 'integer', true );
 436                  }
 437  
 438                  switch( $action )
 439                  {
 440                      case 'create': // create thread
 441                          // Stop a request from the blocked IP addresses or Domains
 442                          antispam_block_request();
 443  
 444                          // check if create new thread is allowed
 445                          if( check_create_thread_limit() )
 446                          { // max new threads limit reached, don't allow to create new thread
 447                              debug_die( 'Invalid request, new conversation limit already reached!' );
 448                          }
 449  
 450                          if( !create_new_thread() )
 451                          { // unsuccessful new thread creation
 452                              global $edited_Thread, $edited_Message, $thrd_recipients, $thrd_recipients_array;
 453  
 454                              $redirect_to .= '&action=new';
 455                              // save new message and thread params into the Session to not lose the content
 456                              $unsaved_message_params = array();
 457                              $unsaved_message_params[ 'subject' ] = $edited_Thread->title;
 458                              $unsaved_message_params[ 'message' ] = $edited_Message->text;
 459                              $unsaved_message_params[ 'thrdtype' ] = param( 'thrdtype', 'string', 'individual' );  // alternative: discussion
 460                              $unsaved_message_params[ 'thrd_recipients' ] = $thrd_recipients;
 461                              $unsaved_message_params[ 'thrd_recipients_array' ] = $thrd_recipients_array;
 462                              save_message_params_to_session( $unsaved_message_params );
 463                          }
 464                          break;
 465  
 466                      case 'delete': // delete thread
 467                          // Check permission:
 468                          $current_User->check_perm( 'perm_messaging', 'delete', true );
 469  
 470                          $confirmed = param( 'confirmed', 'integer', 0 );
 471                          if( $confirmed )
 472                          {
 473                              $msg = sprintf( T_('Thread &laquo;%s&raquo; deleted.'), $edited_Thread->dget('title') );
 474                              $edited_Thread->dbdelete( true );
 475                              unset( $edited_Thread );
 476                              forget_param( 'thrd_ID' );
 477                              $Messages->add( $msg, 'success' );
 478                          }
 479                          else
 480                          {
 481                              $delete_url = $samedomain_htsrv_url.'action.php?mname=messaging&thrd_ID='.$edited_Thread->ID.'&action=delete&confirmed=1&redirect_to='.$redirect_to.'&'.url_crumb( 'messaging_threads' );
 482                              $ok_button = '<span class="linkbutton"><a href="'.$delete_url.'">'.T_( 'I am sure!' ).'!</a></span>';
 483                              $cancel_button = '<span class="linkbutton"><a href="'.$redirect_to.'">CANCEL</a></span>';
 484                              $msg = sprintf( T_( 'You are about to delete all messages in the conversation &laquo;%s&raquo;.' ), $edited_Thread->dget('title') );
 485                              $msg .= '<br />'.T_( 'This CANNOT be undone!').'<br />'.T_( 'Are you sure?' ).'<br /><br />'.$ok_button."\t".$cancel_button;
 486                              $Messages->add( $msg, 'error' );
 487                          }
 488                          break;
 489  
 490                      case 'leave': // user wants to leave the thread
 491                          leave_thread( $edited_Thread->ID, $current_User->ID, false );
 492  
 493                          $Messages->add( sprintf( T_( 'You have successfuly left the &laquo;%s&raquo; conversation!' ), $edited_Thread->get( 'title' ) ), 'success' );
 494                          break;
 495  
 496                      case 'close': // close the thread
 497                      case 'close_and_block': // close the thread and block contact
 498                          leave_thread( $edited_Thread->ID, $current_User->ID, true );
 499  
 500                          // user has closed this conversation because there was only one other user involved
 501                          $Messages->add( sprintf( T_( 'You have successfuly closed the &laquo;%s&raquo; conversation!' ), $edited_Thread->get( 'title' ) ), 'success' );
 502                          if( $action == 'close_and_block' )
 503                          { // user also wants to block contact with the other user involved in this thread
 504                              $block_user_ID = param( 'block_ID', 'integer', true );
 505                              $UserCache = & get_UserCache();
 506                              $blocked_User = $UserCache->get_by_ID( $block_user_ID );
 507  
 508                              set_contact_blocked( $block_user_ID, true );
 509                              $Messages->add( sprintf( T_( '&laquo;%s&raquo; was blocked.' ), $blocked_User->get( 'login' ) ), 'success' );
 510                          }
 511                          break;
 512                  }
 513                  break; // break from threads action switch
 514  
 515              // contacts action
 516              case 'contacts':
 517                  $user_ID = param( 'user_ID', 'string', true );
 518  
 519                  if( ( $action != 'block' ) && ( $action != 'unblock' ) )
 520                  { // only block or unblock is valid
 521                      debug_die( "Invalid action param" );
 522                  }
 523                  set_contact_blocked( $user_ID, ( ( $action == 'block' ) ? 1 : 0 ) );
 524                  $redirect_to = str_replace( '&amp;', '&', $redirect_to );
 525                  break;
 526  
 527              // messages action
 528              case 'messages':
 529                  // Stop a request from the blocked IP addresses or Domains
 530                  antispam_block_request();
 531  
 532                  if( $action == 'create' )
 533                  { // create new message
 534                      create_new_message( $thrd_ID );
 535                  }
 536                  elseif( $action == 'delete' )
 537                  {
 538                      // Check permission:
 539                      $current_User->check_perm( 'perm_messaging', 'delete', true );
 540  
 541                      $msg_ID = param( 'msg_ID', 'integer', true );
 542                      $MessageCache = & get_MessageCache();
 543                      if( ($edited_Message = & $MessageCache->get_by_ID( $msg_ID, false )) === false )
 544                      {
 545                          $Messages->add( sprintf( T_('Requested &laquo;%s&raquo; object does not exist any longer.'), T_('Message') ), 'error' );
 546                          break;
 547                      }
 548  
 549                      $confirmed = param( 'confirmed', 'integer', 0 );
 550                      if( $confirmed )
 551                      { // delete message
 552                          $edited_Message->dbdelete();
 553                          unset( $edited_Message );
 554                          $Messages->add( T_('Message deleted.'), 'success' );
 555                      }
 556                      else
 557                      {
 558                          $delete_url = $samedomain_htsrv_url.'action.php?mname=messaging&disp=messages&thrd_ID='.$thrd_ID.'&msg_ID='.$msg_ID.'&action=delete&confirmed=1';
 559                          $delete_url = url_add_param( $delete_url, 'redirect_to='.rawurlencode( $redirect_to ), '&' ).'&'.url_crumb( 'messaging_messages' );
 560                          $ok_button = '<span class="linkbutton"><a href="'.$delete_url.'">'.T_( 'I am sure!' ).'!</a></span>';
 561                          $cancel_button = '<span class="linkbutton"><a href="'.$redirect_to.'">CANCEL</a></span>';
 562                          $msg = T_('You are about to delete this message. ').'<br /> '.T_('This CANNOT be undone!').'<br />'.T_( 'Are you sure?' ).'<br /><br />'.$ok_button.$cancel_button;
 563                          $Messages->add( $msg, 'error' );
 564                      }
 565                  }
 566                  break;
 567          }
 568  
 569          header_redirect( $redirect_to ); // Will save $Messages into Session
 570      }
