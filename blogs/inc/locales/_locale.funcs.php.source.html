  48      /**
  49       * TRANSLATE!
  50       *
  51       * Translate a text to the desired locale or to the current locale.
  52       *
  53       * @param string String to translate, '' to get language file info (as in gettext spec)
  54       * @param string locale to translate to, '' to use current locale
  55       * @param array Array containing the following keys (all optional):
  56       *              - 'ext_transarray': A reference to an alternate array
  57       *                                  to use for the caching of the
  58       *                                  translated strings or NULL to use
  59       *                                  the internal array.
  60       *              - 'alt_basedir': Alternate base directory to search
  61       *                               for translations, e. g. a plugin or
  62       *                               skin directory.
  63       *              - 'for_helper': (boolean) Is the translation for the b2evoHelper object?
  64       * @return string The translated string or the original string on error.
  65       *
  66       * @internal The last parameter/its 'alt_basedir' key is used by
  67       *           Plugin::T_() and Skin::T_().
  68       */
  69      function T_( $string, $req_locale = '', $params = array() )
  70      {
  71          /**
  72           * The translations keyed by locale.
  73           *
  74           * This array is only used if $params['ext_transarray'] === NULL.
  75           *
  76           * @var array
  77           * @static
  78           */
  79          static $_trans = array();
  80  
  81          global $current_locale, $locales, $locales_path, $plugins_path;
  82          global $evo_charset, $Debuglog;
  83  
  84          $params = array_merge( array(
  85                                  'ext_transarray' => NULL,
  86                                  'alt_basedir'    => '',
  87                                  'for_helper'     => false,
  88                                  ), $params );
  89  
  90          if( empty( $req_locale ) )
  91          { // By default we use the current locale
  92              if( empty( $current_locale ) )
  93              { // don't translate if we have no locale
  94                  return $string;
  95              }
  96  
  97              $req_locale = $current_locale;
  98          }
  99  
 100          if( ! isset( $locales[$req_locale]['messages'] ) )
 101          {
 102              $Debuglog->add( 'No messages file path for locale. $locales["'
 103                      .$req_locale.'"] is '.var_export( @$locales[$req_locale], true ), 'locale' );
 104  
 105              if( ! empty( $evo_charset ) ) // this extra check is needed, because $evo_charset may not yet be determined.. :/
 106              {
 107                  $string = convert_charset( $string, $evo_charset, 'iso-8859-1' );
 108              }
 109              return $string;
 110          }
 111  
 112          $messages = $locales[$req_locale]['messages'];
 113  
 114          if ( is_null( $params['ext_transarray'] ) )
 115          {    // use our array
 116              //$Debuglog->add( 'Using internal array', 'locale' );
 117              $trans = & $_trans;
 118          }
 119          else
 120          {    // use external array:
 121              //$Debuglog->add( 'Using external array', 'locale' );
 122              $trans = & $params['ext_transarray'];
 123          }
 124  
 125          if( ! isset( $trans[ $messages ] ) )
 126          { // Translations for current locale have not yet been loaded:
 127              if ( $params['alt_basedir'] != '' )
 128              {    // Load the translation file from the alternative base dir:
 129                  //$Debuglog->add( 'Using alternative basedir ['.$params['alt_basedir'].']', 'locale' );
 130                  $path = $params['alt_basedir'].'/locales/'.$messages.'/_global.php';
 131              }
 132              else
 133              {    // Load our global translation file.
 134                  $path = $locales_path.$messages.'/_global.php';
 135              }
 136  
 137              if( file_exists($path) && is_readable($path) )
 138              {
 139                  $Debuglog->add( 'T_: Loading file: '.$path, 'locale' );
 140                  include_once $path;
 141              }
 142              else
 143              {
 144                  $Debuglog->add( 'T_: Messages file does not exist or is not readable: '.$path, 'locale' );
 145              }
 146              if( ! isset($trans[ $messages ] ) )
 147              { // Still not loaded... file doesn't exist, memorize that no translations are available
 148                  // echo 'file not found!';
 149                  $trans[ $messages ] = array();
 150              }
 151              else
 152              {
 153                  if( ! isset($trans[$messages]['__meta__']) )
 154                  { // Unknown/old messages format (< version 1):
 155                      $Debuglog->add( 'Found deprecated messages format (no __meta__ info).', 'locale' );
 156                      // Translate keys (e.g. 'foo\nbar') to real strings ("foo\nbar")
 157                      // Doing this here for all strings, is actually faster than doing it on key lookup (like it has been done before always)
 158                      foreach($trans[$messages] as $k => $v)
 159                      {
 160                          if( ($pos = strpos($k, '\\')) === false )
 161                          { // fast-path-skip
 162                              continue;
 163                          }
 164                          // Replace string as done in the good old days:
 165                          $new_k = str_replace( array('\n', '\r', '\t'), array("\n", "\r", "\t"), $k );
 166                          if( $new_k != $k )
 167                          {
 168                              $trans[$messages][$new_k] = $v;
 169                              unset($trans[$messages][$k]);
 170                          }
 171                      }
 172                  }
 173              }
 174          }
 175  
 176          // sam2kb> b2evolution creates _global.php files with "\n" line breaks, and we must normalize newlines
 177          // in supplied string before trying to translate it. Otherwise strings won't match.
 178          // fp> TODO: this is not really satisfying in the long term. We need our own
 179          // parser that will extract T_() TS_() NT_() etc string and create a normalized potfile.
 180          // Actually it sgould create several potfiles. One for general use, one for admin, one for install, etc.
 181          // That way translators can concentrate on the most essential stuff first.
 182          $search_string = str_replace( array("\r\n", "\r"), "\n", $string );
 183  
 184          if( isset( $trans[ $messages ][ $search_string ] ) )
 185          { // If the string has been translated:
 186              //$Debuglog->add( 'String ['.$string.'] found', 'locale' );
 187              $r = $trans[ $messages ][ $search_string ];
 188              if( isset($trans[$messages]['__meta__']['charset']) )
 189              { // new format: charset in meta data:
 190                  $messages_charset = $trans[$messages]['__meta__']['charset'];
 191              }
 192              else
 193              { // old format.. extract charset from content type or fall back to setting from global locale definition:
 194                  $meta = $trans[$messages][''];
 195                  if( preg_match( '~^Content-Type: text/plain; charset=(.*);?$~m', $meta, $match ) )
 196                  {
 197                      $messages_charset = $match[1];
 198                  }
 199                  else
 200                  {
 201                      $messages_charset = $locales[$req_locale]['charset'];
 202                  }
 203                  // Set it accordingly to new format.
 204                  $trans[$messages]['__meta__']['charset'] = $messages_charset;
 205              }
 206          }
 207          else
 208          {
 209              //$Debuglog->add( 'String ['.$string.'] not found', 'locale' );
 210              // Return the English string:
 211              $r = $string;
 212              $messages_charset = 'iso-8859-1'; // our .php file encoding
 213          }
 214  
 215          if( ! empty($evo_charset) ) // this extra check is needed, because $evo_charset may not yet be determined.. :/
 216          {
 217              $r = convert_charset( $r, $evo_charset, $messages_charset );
 218          }
 219          else
 220          {
 221              $Debuglog->add(sprintf('Warning: evo_charset not set to translate "%s"', evo_htmlspecialchars($string)), 'locale');
 222          }
 223  
 224          if( $params['for_helper'] )
 225          { // translation is for the b2evoHelper object
 226              add_js_translation( $string, $r );
 227          }
 228  
 229          //$Debuglog->add( 'Result: ['.$r.']', 'locale' );
 230          return $r;
 231      }

 265   * Temporarily switch to another locale
 266   *
 267   * Calls can be nested, see {@link locale_restore_previous()}.
 268   *
 269   * @param string locale to activate
 270   * @return boolean true on success, false on failure

 272  function locale_temp_switch( $locale )
 273  {
 274      global $saved_locales, $current_locale, $Timer;
 275  
 276      // $Timer->resume( 'locale_temp_switch' );
 277  
 278      if( !isset( $saved_locales ) || ! is_array( $saved_locales ) )
 279      {
 280          $saved_locales = array();
 281      }
 282  
 283      $prev_locale = $current_locale;
 284      if( locale_activate( $locale ) )
 285      {
 286          array_push( $saved_locales, $prev_locale );
 287          return true;
 288      }
 289  
 290      // $Timer->stop( 'locale_temp_switch' );
 291      return false;
 292  }

 314  /**
 315   * Activate a locale.
 316   *
 317   * @todo dh> this should make sure, that e.g. "charset" is set for the locale in {@link $locales}. See http://forums.b2evolution.net/viewtopic.php?p=43980#43980
 318   *
 319   * @param string locale to activate
 320   * @param boolean True on success/change, false on failure (if already set or not existant)
 321   */
 322  function locale_activate( $locale )
 323  {
 324      global $locales, $current_locale, $current_charset;
 325  
 326      if( $locale == $current_locale
 327              || empty( $locale )
 328              || ! isset( $locales[$locale] ) )
 329      {
 330          return false;
 331      }
 332  
 333      // Memorize new locale:
 334      $current_locale = $locale;
 335      // Memorize new charset:
 336      $current_charset = $locales[ $locale ][ 'charset' ];
 337      return true;
 338  }

 342   * locale_by_lang(-)
 343   *
 344   * Find first locale matching lang
 345   */
 346  function locale_by_lang( $lang, $fallback_to_default = true )
 347  {
 348      global $locales, $default_locale;
 349  
 350      foreach( $locales as $locale => $locale_params )
 351      {
 352          if( substr( $locale, 0 ,2 ) == $lang )
 353          { // found first matching locale
 354              return $locale;
 355          }
 356      }
 357  
 358      // Not found...
 359      if( $fallback_to_default )
 360          return $default_locale;
 361      else
 362          return $lang;
 363  }

 535  /**
 536   * [callback function] Outputs an <option> set with default locale selected.
 537   * Optionally returns an array with a locale key and name if there's only one enabled locale.
 538   *
 539   * @param string default value
 540   * @param boolean echo output?
 541   * @param boolean Return array (locale key + name) if there's only one enabled locale?
 542   * @return string|array The options string or an array (locale key + name) if there's only one enabled locale and $array_if_onelocale == true.
 543   */
 544  function locale_options( $default = '', $disp = true, $array_if_onelocale = false )
 545  {
 546      global $locales, $default_locale;
 547  
 548      if( empty( $default ) ) $default = $default_locale;
 549  
 550      $r = '';
 551      $enabled_count = 0;
 552      $enabled_lastkey = '';
 553  
 554      foreach( $locales as $this_localekey => $this_locale )
 555      {
 556          if( $this_locale['enabled'] || $this_localekey == $default )
 557          {
 558              $r .= '<option value="'. $this_localekey. '"';
 559              if( $this_localekey == $default )
 560                  $r .= ' selected="selected"';
 561              $r .= '>'. T_($this_locale['name']). '</option>';
 562  
 563              ++$enabled_count;
 564              $enabled_lastkey = $this_localekey;
 565          }
 566      }
 567  
 568      if( $disp )
 569      {    // the result must be displayed
 570          echo $r;
 571      }
 572  
 573      if ( $array_if_onelocale && $enabled_count == 1 )
 574      {    // We've only one enabled locale:
 575          return array( $enabled_lastkey, $locales[$enabled_lastkey]['name'] );
 576      }
 577      else
 578      {    // Return the string.
 579          return $r;
 580      }
 581  }

 596  /**
 597   * Detect language from HTTP_ACCEPT_LANGUAGE
 598   *
 599   * HTTP_ACCEPT_LANGUAGE is sorted by prio and then the best match is used
 600   * (either full locale ("en-US") or best fitting locale for a short one ("en").
 601   *
 602   * This gets tested in {@link test_locale_from_httpaccept()}.
 603   *
 604   * @author Rewritten by blueyed in Revision 1.42
 605   *
 606   * @return string Locale made out of HTTP_ACCEPT_LANGUAGE or $default_locale, if no match
 607   */
 608  function locale_from_httpaccept()
 609  {
 610      global $locales, $default_locale;
 611      if( isset($_SERVER['HTTP_ACCEPT_LANGUAGE']) )
 612      {
 613          $accept = strtolower( $_SERVER['HTTP_ACCEPT_LANGUAGE'] );
 614  
 615          // Create list of accepted locales.
 616          if( ! preg_match_all('/([a-z]{1,8}(?:-[a-z]{1,8})?)\s*(?:;\s*q\s*=\s*(1|0(?:\.[0-9]+)?))?/i', $accept, $accept_list) )
 617          {
 618              return $default_locale;
 619          }
 620          // Create list of enabled locales.
 621          $enabled_locales = array();
 622          foreach($locales as $k => $v)
 623          {
 624              if( empty($v['enabled']) )
 625              {
 626                  continue;
 627              }
 628              $enabled_locales[strtolower($k)] = $k;
 629          }
 630          if( empty($enabled_locales) )
 631          {
 632              return $default_locale;
 633          }
 634          // Build mapping of short code to long code(s)
 635          $short_locales = array();
 636          foreach($enabled_locales as $v)
 637          {
 638              if( $pos = strpos($v, '-') )
 639              {
 640                  $short = substr($v, 0, $pos);
 641                  $short_locales[$short][] = $v;
 642              }
 643          }
 644          // Create "locale" => "prio" list
 645          $accept_list = array_combine($accept_list[1], $accept_list[2]);
 646          $maxq = count($accept_list)+1;
 647          foreach( $accept_list as $k => $v )
 648          {
 649              if( $v === '' )
 650              { // should be kept in order
 651                  $accept_list[$k] = $maxq--;
 652              }
 653              elseif( $v == 0 )
 654              { // not acceptable (RFC 2616)
 655                  unset($accept_list[$k]);
 656              }
 657          }
 658          arsort($accept_list);
 659          $accept_list = array_values(array_keys($accept_list));
 660  
 661          // Go through the list of accepted locales and find best match.
 662          for( $i = 0, $n = count($accept_list); $i<$n; $i++ )
 663          {
 664              $test = $accept_list[$i];
 665  
 666              if( isset($enabled_locales[$test]) )
 667              { // the accepted locale is enabled and a full match
 668                  return $enabled_locales[$test];
 669              }
 670              if( isset($short_locales[$test]) )
 671              { // this is a short locale: find the best/first match in accepted locales
 672                  $first = NULL;
 673                  foreach($short_locales[$test] as $v)
 674                  {
 675                      $pos = array_search(strtolower($v), $accept_list);
 676                      if( $pos !== false && ( ! isset($first) || $pos < $first ) )
 677                      {
 678                          $first = $pos;
 679                      }
 680                  }
 681                  if( isset($first) )
 682                  { // found exact match, via short locale.
 683                      return $enabled_locales[$accept_list[$first]];
 684                  }
 685                  if( isset($enabled_locales[$test.'-'.$test]) )
 686                  { // test for e.g. "de-DE" when only "de" is accepted
 687                      return $enabled_locales[$test.'-'.$test];
 688                  }
 689                  // Fallback: use first enabled locale matching the short accepted
 690                  return $short_locales[$test][0];
 691              }
 692          }
 693      }
 694      return $default_locale;
 695  }

 710  /**
 711   * load locales from DB into $locales array. Also sets $default_locale.
 712   *
 713   * @return mixed new default locale on succes, false on failure
 714   */
 715  function locale_overwritefromDB()
 716  {
 717      global $DB, $locales, $default_locale, $Settings, $Debuglog;
 718  
 719      $usedprios = array();  // remember which priorities are used already.
 720      $priocounter = 0;
 721      $query = 'SELECT
 722                          loc_locale, loc_charset, loc_datefmt, loc_timefmt, loc_startofweek,
 723                          loc_name, loc_messages, loc_priority, loc_transliteration_map, loc_enabled
 724                          FROM T_locales ORDER BY loc_priority';
 725  
 726      foreach( $DB->get_results( $query, ARRAY_A ) as $row )
 727      { // Loop through loaded locales:
 728  
 729          if( $row['loc_priority'] == $priocounter )
 730          { // priority conflict (the same)
 731              $priocounter++;
 732          }
 733          else
 734          {
 735              $priocounter = $row['loc_priority'];
 736          }
 737  
 738          //remember that we used this
 739          $usedprios[] = $priocounter;
 740  
 741          $transliteration_map = '';
 742          // Try to unserialize the value
 743          if( ($r = @unserialize(@base64_decode($row[ 'loc_transliteration_map' ]))) !== false )
 744          {
 745              $transliteration_map = $r;
 746          }
 747  
 748          $locales[ $row['loc_locale'] ] = array(
 749                  'charset'     => $row[ 'loc_charset' ],
 750                  'datefmt'     => $row[ 'loc_datefmt' ],
 751                  'timefmt'     => $row[ 'loc_timefmt' ],
 752                  'startofweek' => $row[ 'loc_startofweek' ],
 753                  'name'        => $row[ 'loc_name' ],
 754                  'messages'    => $row[ 'loc_messages' ],
 755                  'transliteration_map' => $transliteration_map,
 756                  'priority'    => $priocounter,
 757                  'enabled'     => $row[ 'loc_enabled' ],
 758                  'fromdb'      => 1
 759              );
 760      }
 761  
 762      // set default priorities, if nothing was set in DB.
 763      // Missing "priority gaps" will get filled here.
 764      if( $DB->num_rows != count($locales) )
 765      { // we have locales from conf file that need a priority
 766          $priocounter = 1;
 767          foreach( $locales as $lkey => $lval )
 768          { // Loop through memory locales:
 769              if( !isset($lval['priority']) )
 770              { // Found one that has no assigned priority
 771                  while( in_array( $priocounter, $usedprios ) )
 772                  {
 773                      $priocounter++;
 774                  }
 775                  // Priocounter has max value
 776                  $locales[$lkey]['priority'] = $priocounter;
 777                  $usedprios[] = $priocounter;
 778              }
 779          }
 780      }
 781  
 782      // sort by priority
 783      uasort( $locales, 'locale_priosort' );
 784  
 785      // overwrite default_locale from DB settings - if enabled.
 786      // Checks also if previous $default_locale is enabled. Defaults to en-EU, even if not enabled.
 787      $locale_fromdb = $Settings->get('default_locale');
 788  
 789      if( $locale_fromdb )
 790      {
 791          if( !isset( $locales[$locale_fromdb] ) )
 792          {
 793              $Debuglog->add( 'Default locale ['.$locale_fromdb.'] from general settings is not available.', 'locale' );
 794              return false;
 795          }
 796          else
 797          {
 798              $default_locale = $locale_fromdb;
 799              return $default_locale;
 800          }
 801      }
 802  }

 805  /**
 806   * Write $locales array to DB table
 807   */
 808  function locale_updateDB()
 809  {
 810      global $locales, $DB, $Settings, $Messages, $action;
 811      global $saved_params;
 812  
 813      $templocales = $locales;
 814      $disabled_locales = array();
 815      $saved_params = array();
 816  
 817      $lnr = 0;
 818      // Loop through list of all HTTP POSTed params:
 819      foreach( $_POST as $pkey => $pval )
 820      {
 821          if( ! preg_match('/loc_(\d+)_(.*)/', $pkey, $matches) )
 822          {
 823              continue;
 824          }
 825  
 826          // This is a locale related parameter, get it now:
 827          $pval = param( $pkey, 'string', '' );
 828          $saved_params[$pkey] = $pval;
 829  
 830          $lfield = $matches[2];
 831  
 832          if( $matches[1] != $lnr )
 833          { // we have a new locale
 834              $lnr = $matches[1];
 835              $plocale = $pval;
 836  
 837              if( $templocales[ $plocale ]['enabled'] )
 838              { // Collect previously enabled locales
 839                  $disabled_locales[$plocale] = $plocale;
 840              }
 841              // checkboxes default to 0
 842              $templocales[ $plocale ]['enabled'] = 0;
 843          }
 844          elseif( $lnr != 0 )  // be sure to have catched a locale before
 845          {
 846              if( $lfield == 'startofweek' && ( $lfield < 0 || $lfield > 6 ) )
 847              { // startofweek must be between 0 and 6
 848                  continue;
 849              }
 850              if( $lfield == 'enabled' )
 851              { // This locale is enabled, remove from the list of the disabled locales
 852                  unset( $disabled_locales[$plocale] );
 853              }
 854              $templocales[$plocale][$lfield] = $pval;
 855          }
 856      }
 857  
 858      $locales = $templocales;
 859  
 860      // Make sure the default locale is enabled
 861      $current_default_locale = $Settings->get('default_locale');
 862      if( isset( $current_default_locale ) && ( ! $locales[$current_default_locale]['enabled'] ) )
 863      { // The default locale is not enabled, we do not allow to update locales in this case
 864          $Messages->add( T_('Updating locales are not enabled until default locale is not valid!'), 'error' );
 865          return false;
 866      }
 867  
 868      // Check the usage of the disabled locales
 869      $disabled_locales = ( count( $disabled_locales ) > 0 ) ? $DB->quote( $disabled_locales ) : false;
 870      if( $disabled_locales && ( $action != 'confirm_update' ) )
 871      { // Some locales were disabled, create warning message if required
 872          $user_disabled_locales = $DB->get_assoc( '
 873              SELECT user_locale, count( user_ID ) as number_of_users
 874              FROM T_users
 875              WHERE user_locale IN ( '.$disabled_locales.' )
 876              GROUP BY user_locale
 877              HAVING number_of_users > 0
 878              ORDER BY user_locale'
 879          );
 880          $coll_disabled_locales = $DB->get_assoc( '
 881              SELECT blog_locale, count( blog_ID ) as number_of_blogs
 882              FROM T_blogs
 883              WHERE blog_locale IN ( '.$disabled_locales.' )
 884              GROUP BY blog_locale
 885              HAVING number_of_blogs > 0
 886              ORDER BY blog_locale'
 887          );
 888  
 889          global $warning_message;
 890          $warning_message = array();
 891          $users_message = T_('%d users with invalid locale %s');
 892          $colls_message = T_('%d collections with invalid locale %s');
 893          foreach( $user_disabled_locales as $disabled_locale => $numbe_of_users )
 894          { // Disabled locale is used by users, add warning
 895              $warning_message[] = sprintf( $users_message, $numbe_of_users, $disabled_locale );
 896              if( isset( $coll_disabled_locales[$disabled_locale] ) )
 897              { // Disabled locale is used by blogs, add warning
 898                  $warning_message[] = sprintf( $colls_message, $coll_disabled_locales[$disabled_locale], $disabled_locale );
 899                  unset( $coll_disabled_locales[$disabled_locale] );
 900              }
 901          }
 902          foreach( $coll_disabled_locales as $disabled_locale => $numbe_of_colls )
 903          { // Disabled locale is used by blogs, add warning
 904              $warning_message[] = sprintf( $colls_message, $numbe_of_colls, $disabled_locale );
 905          }
 906  
 907          if( !empty( $warning_message ) )
 908          { // There are disabled locales which are used, create the final warning message
 909              $warning_message = T_('You have disabled some locales. This results in:')."\n"
 910                  .'<ul><li>'.implode( '</li><li>', $warning_message ).'</li></ul>'
 911                  .sprintf( T_('These will be assigned the default locale %s'), $current_default_locale );
 912              return false;
 913          }
 914      }
 915  
 916      $query = "REPLACE INTO T_locales ( loc_locale, loc_charset, loc_datefmt, loc_timefmt, loc_startofweek, loc_name, loc_messages, loc_priority, loc_transliteration_map, loc_enabled ) VALUES ";
 917      foreach( $locales as $localekey => $lval )
 918      {
 919          if( empty($lval['messages']) )
 920          { // if not explicit messages file is given we'll translate the locale
 921              $lval['messages'] = strtr($localekey, '-', '_');
 922          }
 923  
 924          $transliteration_map = '';
 925          if( !empty($lval['transliteration_map']) )
 926          {
 927              if( is_string($lval['transliteration_map']) )
 928              {    // The value is already serialized and encoded
 929                  $transliteration_map = $lval['transliteration_map'];
 930              }
 931              else
 932              {    // Encode the value
 933                  $transliteration_map = base64_encode( serialize($lval['transliteration_map']) );
 934              }
 935          }
 936  
 937          $query .= '(
 938              '.$DB->quote($localekey).',
 939              '.$DB->quote($lval['charset']).',
 940              '.$DB->quote($lval['datefmt']).',
 941              '.$DB->quote($lval['timefmt']).',
 942              '.$DB->quote($lval['startofweek']).',
 943              '.$DB->quote($lval['name']).',
 944              '.$DB->quote($lval['messages']).',
 945              '.$DB->quote($lval['priority']).',
 946              '.$DB->quote($transliteration_map).',
 947              '.$DB->quote($lval['enabled']).'
 948          ), ';
 949      }
 950      $query = substr($query, 0, -2);
 951  
 952      $DB->begin();
 953  
 954      $result = true;
 955      if( $action == 'confirm_update' )
 956      { // Update users and blogs locale to the default if the prevously used locale was disabled
 957          $users_update = $DB->query( 'UPDATE T_users
 958              SET user_locale = '.$DB->quote( $current_default_locale ).'
 959              WHERE  user_locale IN ( '.$disabled_locales.' )'
 960          );
 961          $blogs_update = $DB->query( 'UPDATE T_blogs
 962              SET blog_locale = '.$DB->quote( $current_default_locale ).'
 963              WHERE  blog_locale IN ( '.$disabled_locales.' )'
 964          );
 965          $result = ( $users_update !== false ) && ( $blogs_update !== false );
 966      }
 967  
 968      if( $result && ( $DB->query($query) !== false ) )
 969      { // Commit after successful update
 970          $DB->commit();
 971          return true;
 972      }
 973  
 974      // Some error occured, rollback the transaction and add error message
 975      $DB->rollback();
 976      $Messages->add( T_('Unexpected error occured, regional settings could not be updated.') );
 977      return false;
 978  }

1023   * @param string Target charset (TO)
1024   * @param string Source charset (FROM)
1025   * @return boolean
1026   */
1027  function can_convert_charsets( $dest_charset, $src_charset )
1028  {
1029      if( empty($dest_charset) || empty($src_charset) )
1030      {
1031          return false;
1032      }
1033  
1034      if( function_exists('mb_internal_encoding') )
1035      { // mb_string extension:
1036          $orig = mb_internal_encoding();

1040          { // we can set both encodings, so we should be able to convert:
1041              $r = true;
1042          }
1043  
1044          mb_internal_encoding($orig);
1045          return $r;
1046      }
1047  
1048      return false;
1049  }

1081  /**
1082   * Init charset handling between Input/Output ($io_charset) and the internal
1083   * handling ($evo_charset).
1084   *
1085   * Check and possibly adjust {@link $evo_charset}.
1086   *
1087   * @staticvar boolean Used to only start mb_output_handler once
1088   * @param string I/O (input/output) charset to use
1089   * @return boolean true, if encoding has been changed
1090   */
1091  function init_charsets( $req_io_charset )
1092  {
1093      static $mb_output_handler_started;
1094      global $io_charset, $evo_charset, $Debuglog, $DB;
1095      global $force_io_charset_if_accepted;
1096  
1097      if( $req_io_charset == $io_charset )
1098      { // no conversation/init needed
1099          return false;
1100      }
1101  
1102      // check, if we want to force a specific charset (e.g. 'utf-8'):
1103      if( ! empty($force_io_charset_if_accepted) )
1104      { // we want to force a specific charset:
1105          if( ! isset($_SERVER['HTTP_ACCEPT_CHARSET']) // all allowed
1106              || preg_match( '~\b(\*|'.preg_quote( $force_io_charset_if_accepted, '~' ).')\b~', $_SERVER['HTTP_ACCEPT_CHARSET'] ) )
1107          {
1108              $req_io_charset = $force_io_charset_if_accepted; // pretend that the first one has been requested
1109          }
1110      }
1111  
1112      if( $req_io_charset == $io_charset )
1113      { // no conversation/init needed
1114          return false;
1115      }
1116  
1117      $io_charset = $req_io_charset;
1118  
1119      if( empty($evo_charset) )
1120      { // empty evo_charset follows I/O charset:
1121          // TODO: $evo_charset will not follow, if it has followed before.. (because not empty anymore)
1122          $Debuglog->add( '$evo_charset follows $io_charset ('.$io_charset.').', array('locale') );
1123          $evo_charset = $io_charset;
1124      }
1125      elseif( $evo_charset != $io_charset )
1126      { // we have to convert for I/O
1127          // TODO: dh> $io_charset has to forcefully follow $evo_charset, if we cannot convert, e.g. utf-8/iso-8859-1
1128          if( ! function_exists('mb_convert_encoding') )
1129          {
1130              $Debuglog->add( '$evo_charset differs from $io_charset, but mbstrings is not available - cannot convert I/O to internal charset!', array('errors','locale') );
1131              $evo_charset = $io_charset; // we cannot convert I/O to internal charset
1132          }
1133          else
1134          { // check if the encodings are supported:
1135              // NOTE: mb_internal_encoding() is the best way to find out if the encoding is supported
1136              $old_mb_internal_encoding = mb_internal_encoding();
1137              if( ! @mb_internal_encoding($io_charset) )
1138              {
1139                  $Debuglog->add( 'Cannot I/O convert because I/O charset ['.$io_charset.'] is not supported by mbstring!', array('error','locale') );
1140                  $evo_charset = $io_charset;
1141                  mb_internal_encoding($old_mb_internal_encoding);
1142              }
1143              elseif( ! @mb_internal_encoding($evo_charset) )
1144              {
1145                  $Debuglog->add( 'Cannot I/O convert because $evo_charset='.$evo_charset.' is not supported by mbstring!', array('error','locale') );
1146                  $evo_charset = $io_charset;
1147                  mb_internal_encoding($old_mb_internal_encoding);
1148              }
1149              else
1150              { // we can convert between I/O
1151                  mb_http_output( $io_charset );
1152                  if( ! $mb_output_handler_started )
1153                  {
1154                      ob_start( 'mb_output_handler' ); // NOTE: this will send a Content-Type header by itself for "text/..."
1155                      $mb_output_handler_started = true;
1156                      $Debuglog->add( 'Started mb_output_handler.', 'locale' );
1157                  }
1158              }
1159          }
1160      }
1161  
1162      // Make sure the DB send us text in the same charset as $evo_charset (override whatever may have been set before)
1163      if( isset($DB) ) // not available in /install/index.php
1164      {    // Set encoding for MySQL connection:
1165          $DB->set_connection_charset( $evo_charset );
1166      }
1167  
1168      $Debuglog->add( 'evo_charset: '.$evo_charset, 'locale' );
1169      $Debuglog->add( 'io_charset: '.$io_charset, 'locale' );
1170  
1171      return true;
1172  }

1175  /**
1176   * Load available locale definitions
1177   */
1178  function locales_load_available_defs()
1179  {
1180      global $locales_path;
1181      global $locales;
1182  
1183      // This is where language packs will store their locale defintions:
1184      $locale_defs = array();
1185  
1186      // Get all locale folder names:
1187      $filename_params = array(
1188              'inc_files'    => false,
1189              'recurse'    => false,
1190              'basename'    => true,
1191          );
1192      $locale_folders = get_filenames( $locales_path, $filename_params );
1193      // Go through all locale folders:
1194      foreach( $locale_folders as $locale_folder )
1195      {
1196          //pre_dump( $locale_folder );
1197          $ad_locale_folder = $locales_path.'/'.$locale_folder;
1198          // Get files in folder:
1199          $filename_params = array(
1200                  'inc_dirs'    => false,
1201                  'recurse'    => false,
1202                  'basename'    => true,
1203              );
1204          $locale_def_files = get_filenames( $ad_locale_folder, $filename_params );
1205          // Go through files in locale folder:
1206          foreach( $locale_def_files as $locale_def_file )
1207          {    // Check if it's a definition file:
1208              // pre_dump( $locale_def_file );
1209              if( preg_match( '~[a-z0-9\-.]\.locale\.php$~i', $locale_def_file ) )
1210              {    // We found a definition file:
1211                  // pre_dump( $locale_def_file );
1212                  include $ad_locale_folder.'/'.$locale_def_file;
1213              }
1214          }
1215      }
1216  
1217      // Copy any new locale definitions over to $locales:
1218      foreach( $locale_defs as $locale_code => $locale_def )
1219      {
1220          if( !isset( $locales[$locale_code] ) )
1221          {    // New locale, add to main array:
1222              $locales[$locale_code] = $locale_def;
1223              // ... but mark as not enabled!
1224              $locales[$locale_code]['enabled'] = 0;
1225          }
1226      }
1227  
1228  
1229      // Assign priorities:
1230  
1231      // Find highest used priority:
1232      $max_prio = 0;
1233      foreach( $locales as $locale )
1234      {
1235          if( isset($locale['priority']) && $locale['priority'] > $max_prio )
1236          {
1237              $max_prio = $locale['priority'];
1238          }
1239      }
1240  
1241      foreach( $locales as $lkey=>$locale )
1242      {
1243          if( !isset($locale['priority']) )
1244          {
1245              $locales[$lkey]['priority'] = ++$max_prio;
1246          }
1247      }
1248  
1249  }

1252  /**
1253   * Get a number of the messages in the .PO & .POT file
1254   *
1255   * @param string File name (messages.po)
1256   * @param boolean TRUE - to calc a percent of translated messages
1257   * @return integer Number of the messages
1258   */
1259  function locale_file_po_info( $po_file_name, $calc_percent_done = false )
1260  {
1261      $all = 0;
1262      $fuzzy = 0;
1263      $this_fuzzy = false;
1264      $untranslated = 0;
1265      $translated = 0;
1266      $status = '-';
1267      $matches = array();
1268  
1269      if( file_exists( $po_file_name ) )
1270      {
1271          $lines = file( $po_file_name );
1272          $lines[] = '';    // Adds a blank line at the end in order to ensure complete handling of the file
1273  
1274          foreach( $lines as $line )
1275          {
1276              if( trim( $line ) == '' )
1277              {    // Blank line, go back to base status:
1278                  if( $status == 't' )
1279                  {    // ** End of a translation ** :
1280                      if( $msgstr == '' )
1281                      {
1282                          $untranslated++;
1283                          // echo 'untranslated: ', $msgid, '<br />';
1284                      }
1285                      else
1286                      {
1287                          $translated++;
1288                      }
1289                      if( $msgid == '' && $this_fuzzy )
1290                      {    // It's OK if first line is fuzzy
1291                          $fuzzy--;
1292                      }
1293                      $msgid = '';
1294                      $msgstr = '';
1295                      $this_fuzzy = false;
1296                  }
1297                  $status = '-';
1298              }
1299              elseif( ( $status == '-' ) && preg_match( '#^msgid "(.*)"#', $line, $matches ) )
1300              {    // Encountered an original text
1301                  $status = 'o';
1302                  $msgid = $matches[1];
1303                  // echo 'original: "', $msgid, '"<br />';
1304                  $all++;
1305              }
1306              elseif( ( $status == 'o' ) && preg_match( '#^msgstr "(.*)"#', $line, $matches ) )
1307              {    // Encountered a translated text
1308                  $status = 't';
1309                  $msgstr = $matches[1];

1311              }

1315                      $msgid .= $matches[1];
1316                  elseif( $status == 't' )
1317                      $msgstr .= $matches[1];
1318              }
1319              elseif( strpos( $line,'#, fuzzy' ) === 0 )
1320              {
1321                  $this_fuzzy = true;

1323              }
1324          }
1325      }

1328              'all'          => $all,
1329              'fuzzy'        => $fuzzy,
1330              'translated'   => $translated,

1340  }
