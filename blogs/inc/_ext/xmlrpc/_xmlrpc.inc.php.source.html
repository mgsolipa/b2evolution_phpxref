 393      /// xml parser handler function for opening element tags
 394  	function xmlrpc_se($parser, $name, $attrs, $accept_single_vals=false)
 395      {
 396          // if invalid xmlrpc already detected, skip all processing
 397          if ($GLOBALS['_xh']['isf'] < 2)
 398          {
 399              // check for correct element nesting
 400              // top level element can only be of 2 types
 401              /// @todo optimization creep: save this check into a bool variable, instead of using count() every time:
 402              ///       there is only a single top level element in xml anyway
 403              if (count($GLOBALS['_xh']['stack']) == 0)
 404              {
 405                  if ($name != 'METHODRESPONSE' && $name != 'METHODCALL' && (
 406                      $name != 'VALUE' && !$accept_single_vals))
 407                  {
 408                      $GLOBALS['_xh']['isf'] = 2;
 409                      $GLOBALS['_xh']['isf_reason'] = 'missing top level xmlrpc element';
 410                      return;
 411                  }
 412                  else
 413                  {
 414                      $GLOBALS['_xh']['rt'] = strtolower($name);
 415                  }
 416              }
 417              else
 418              {
 419                  // not top level element: see if parent is OK
 420                  $parent = end($GLOBALS['_xh']['stack']);
 421                  if (!array_key_exists($name, $GLOBALS['xmlrpc_valid_parents']) || !in_array($parent, $GLOBALS['xmlrpc_valid_parents'][$name]))
 422                  {
 423                      $GLOBALS['_xh']['isf'] = 2;
 424                      $GLOBALS['_xh']['isf_reason'] = "xmlrpc element $name cannot be child of $parent";
 425                      return;
 426                  }
 427              }
 428  
 429              switch($name)
 430              {
 431                  // optimize for speed switch cases: most common cases first
 432                  case 'VALUE':
 433                      /// @todo we could check for 2 VALUE elements inside a MEMBER or PARAM element
 434                      $GLOBALS['_xh']['vt']='value'; // indicator: no value found yet
 435                      $GLOBALS['_xh']['ac']='';
 436                      $GLOBALS['_xh']['lv']=1;
 437                      $GLOBALS['_xh']['php_class']=null;
 438                      break;
 439                  case 'I4':
 440                  case 'INT':
 441                  case 'STRING':
 442                  case 'BOOLEAN':
 443                  case 'DOUBLE':
 444                  case 'DATETIME.ISO8601':
 445                  case 'BASE64':
 446                      if ($GLOBALS['_xh']['vt']!='value')
 447                      {
 448                          //two data elements inside a value: an error occurred!
 449                          $GLOBALS['_xh']['isf'] = 2;
 450                          $GLOBALS['_xh']['isf_reason'] = "$name element following a {$GLOBALS['_xh']['vt']} element inside a single value";
 451                          return;
 452                      }
 453                      $GLOBALS['_xh']['ac']=''; // reset the accumulator
 454                      break;
 455                  case 'STRUCT':
 456                  case 'ARRAY':
 457                      if ($GLOBALS['_xh']['vt']!='value')
 458                      {
 459                          //two data elements inside a value: an error occurred!
 460                          $GLOBALS['_xh']['isf'] = 2;
 461                          $GLOBALS['_xh']['isf_reason'] = "$name element following a {$GLOBALS['_xh']['vt']} element inside a single value";
 462                          return;
 463                      }
 464                      // create an empty array to hold child values, and push it onto appropriate stack
 465                      $cur_val = array();
 466                      $cur_val['values'] = array();
 467                      $cur_val['type'] = $name;
 468                      // check for out-of-band information to rebuild php objs
 469                      // and in case it is found, save it
 470                      if (@isset($attrs['PHP_CLASS']))
 471                      {
 472                          $cur_val['php_class'] = $attrs['PHP_CLASS'];
 473                      }
 474                      $GLOBALS['_xh']['valuestack'][] = $cur_val;
 475                      $GLOBALS['_xh']['vt']='data'; // be prepared for a data element next
 476                      break;
 477                  case 'DATA':
 478                      if ($GLOBALS['_xh']['vt']!='data')
 479                      {
 480                          //two data elements inside a value: an error occurred!
 481                          $GLOBALS['_xh']['isf'] = 2;
 482                          $GLOBALS['_xh']['isf_reason'] = "found two data elements inside an array element";
 483                          return;
 484                      }
 485                  case 'METHODCALL':
 486                  case 'METHODRESPONSE':
 487                  case 'PARAMS':
 488                      // valid elements that add little to processing
 489                      break;
 490                  case 'METHODNAME':
 491                  case 'NAME':
 492                      /// @todo we could check for 2 NAME elements inside a MEMBER element
 493                      $GLOBALS['_xh']['ac']='';
 494                      break;
 495                  case 'FAULT':
 496                      $GLOBALS['_xh']['isf']=1;
 497                      break;
 498                  case 'MEMBER':
 499                      $GLOBALS['_xh']['valuestack'][count($GLOBALS['_xh']['valuestack'])-1]['name']=''; // set member name to null, in case we do not find in the xml later on
 500                      //$GLOBALS['_xh']['ac']='';
 501                      // Drop trough intentionally
 502                  case 'PARAM':
 503                      // clear value type, so we can check later if no value has been passed for this param/member
 504                      $GLOBALS['_xh']['vt']=null;
 505                      break;
 506                  case 'NIL':
 507                  case 'EX:NIL':
 508                      if ($GLOBALS['xmlrpc_null_extension'])
 509                      {
 510                          if ($GLOBALS['_xh']['vt']!='value')
 511                          {
 512                              //two data elements inside a value: an error occurred!
 513                              $GLOBALS['_xh']['isf'] = 2;
 514                              $GLOBALS['_xh']['isf_reason'] = "$name element following a {$GLOBALS['_xh']['vt']} element inside a single value";
 515                              return;
 516                          }
 517                          $GLOBALS['_xh']['ac']=''; // reset the accumulator
 518                          break;
 519                      }
 520                      // we do not support the <NIL/> extension, so
 521                      // drop through intentionally
 522                  default:
 523                      /// INVALID ELEMENT: RAISE ISF so that it is later recognized!!!
 524                      $GLOBALS['_xh']['isf'] = 2;
 525                      $GLOBALS['_xh']['isf_reason'] = "found not-xmlrpc xml element $name";
 526                      break;
 527              }
 528  
 529              // Save current element name to stack, to validate nesting
 530              $GLOBALS['_xh']['stack'][] = $name;
 531  
 532              /// @todo optimization creep: move this inside the big switch() above
 533              if($name!='VALUE')
 534              {
 535                  $GLOBALS['_xh']['lv']=0;
 536              }
 537          }
 538      }

 547  	function xmlrpc_ee($parser, $name, $rebuild_xmlrpcvals = true)
 548      {
 549          if ($GLOBALS['_xh']['isf'] < 2)
 550          {
 551              // push this element name from stack
 552              // NB: if XML validates, correct opening/closing is guaranteed and
 553              // we do not have to check for $name == $curr_elem.
 554              // we also checked for proper nesting at start of elements...
 555              $curr_elem = array_pop($GLOBALS['_xh']['stack']);
 556  
 557              switch($name)
 558              {
 559                  case 'VALUE':
 560                      // This if() detects if no scalar was inside <VALUE></VALUE>
 561                      if ($GLOBALS['_xh']['vt']=='value')
 562                      {
 563                          $GLOBALS['_xh']['value']=$GLOBALS['_xh']['ac'];
 564                          $GLOBALS['_xh']['vt']=$GLOBALS['xmlrpcString'];
 565                      }
 566  
 567                      if ($rebuild_xmlrpcvals)
 568                      {
 569                          // build the xmlrpc val out of the data received, and substitute it
 570                          $temp = new xmlrpcval($GLOBALS['_xh']['value'], $GLOBALS['_xh']['vt']);
 571                          // in case we got info about underlying php class, save it
 572                          // in the object we're rebuilding
 573                          if (isset($GLOBALS['_xh']['php_class']))
 574                              $temp->_php_class = $GLOBALS['_xh']['php_class'];
 575                          // check if we are inside an array or struct:
 576                          // if value just built is inside an array, let's move it into array on the stack
 577                          $vscount = count($GLOBALS['_xh']['valuestack']);
 578                          if ($vscount && $GLOBALS['_xh']['valuestack'][$vscount-1]['type']=='ARRAY')
 579                          {
 580                              $GLOBALS['_xh']['valuestack'][$vscount-1]['values'][] = $temp;
 581                          }
 582                          else
 583                          {
 584                              $GLOBALS['_xh']['value'] = $temp;
 585                          }
 586                      }
 587                      else
 588                      {
 589                          /// @todo this needs to treat correctly php-serialized objects,
 590                          /// since std deserializing is done by php_xmlrpc_decode,
 591                          /// which we will not be calling...
 592                          if (isset($GLOBALS['_xh']['php_class']))
 593                          {
 594                          }
 595  
 596                          // check if we are inside an array or struct:
 597                          // if value just built is inside an array, let's move it into array on the stack
 598                          $vscount = count($GLOBALS['_xh']['valuestack']);
 599                          if ($vscount && $GLOBALS['_xh']['valuestack'][$vscount-1]['type']=='ARRAY')
 600                          {
 601                              $GLOBALS['_xh']['valuestack'][$vscount-1]['values'][] = $GLOBALS['_xh']['value'];
 602                          }
 603                      }
 604                      break;
 605                  case 'BOOLEAN':
 606                  case 'I4':
 607                  case 'INT':
 608                  case 'STRING':
 609                  case 'DOUBLE':
 610                  case 'DATETIME.ISO8601':
 611                  case 'BASE64':
 612                      $GLOBALS['_xh']['vt']=strtolower($name);
 613                      /// @todo: optimization creep - remove the if/elseif cycle below
 614                      /// since the case() in which we are already did that
 615                      if ($name=='STRING')
 616                      {
 617                          $GLOBALS['_xh']['value']=$GLOBALS['_xh']['ac'];
 618                      }
 619                      elseif ($name=='DATETIME.ISO8601')
 620                      {
 621                          if (!preg_match('/^[0-9]{8}T[0-9]{2}:[0-9]{2}:[0-9]{2}$/', $GLOBALS['_xh']['ac']))
 622                          {
 623                              error_log('XML-RPC: invalid value received in DATETIME: '.$GLOBALS['_xh']['ac']);
 624                          }
 625                          $GLOBALS['_xh']['vt']=$GLOBALS['xmlrpcDateTime'];
 626                          $GLOBALS['_xh']['value']=$GLOBALS['_xh']['ac'];
 627                      }
 628                      elseif ($name=='BASE64')
 629                      {
 630                          /// @todo check for failure of base64 decoding / catch warnings
 631                          $GLOBALS['_xh']['value']=base64_decode($GLOBALS['_xh']['ac']);
 632                      }
 633                      elseif ($name=='BOOLEAN')
 634                      {
 635                          // special case here: we translate boolean 1 or 0 into PHP
 636                          // constants true or false.
 637                          // Strings 'true' and 'false' are accepted, even though the
 638                          // spec never mentions them (see eg. Blogger api docs)
 639                          // NB: this simple checks helps a lot sanitizing input, ie no
 640                          // security problems around here
 641                          if ($GLOBALS['_xh']['ac']=='1' || strcasecmp($GLOBALS['_xh']['ac'], 'true') == 0)
 642                          {
 643                              $GLOBALS['_xh']['value']=true;
 644                          }
 645                          else
 646                          {
 647                              // log if receiveing something strange, even though we set the value to false anyway
 648                              if ($GLOBALS['_xh']['ac']!='0' && strcasecmp($GLOBALS['_xh']['ac'], 'false') != 0)
 649                                  error_log('XML-RPC: invalid value received in BOOLEAN: '.$GLOBALS['_xh']['ac']);
 650                              $GLOBALS['_xh']['value']=false;
 651                          }
 652                      }
 653                      elseif ($name=='DOUBLE')
 654                      {
 655                          // we have a DOUBLE
 656                          // we must check that only 0123456789-.<space> are characters here
 657                          // NOTE: regexp could be much stricter than this...
 658                          if (!preg_match('/^[+-eE0123456789 \t.]+$/', $GLOBALS['_xh']['ac']))
 659                          {
 660                              /// @todo: find a better way of throwing an error than this!
 661                              error_log('XML-RPC: non numeric value received in DOUBLE: '.$GLOBALS['_xh']['ac']);
 662                              $GLOBALS['_xh']['value']='ERROR_NON_NUMERIC_FOUND';
 663                          }
 664                          else
 665                          {
 666                              // it's ok, add it on
 667                              $GLOBALS['_xh']['value']=(double)$GLOBALS['_xh']['ac'];
 668                          }
 669                      }
 670                      else
 671                      {
 672                          // we have an I4/INT
 673                          // we must check that only 0123456789-<space> are characters here
 674                          if (!preg_match('/^[+-]?[0123456789 \t]+$/', $GLOBALS['_xh']['ac']))
 675                          {
 676                              /// @todo find a better way of throwing an error than this!
 677                              error_log('XML-RPC: non numeric value received in INT: '.$GLOBALS['_xh']['ac']);
 678                              $GLOBALS['_xh']['value']='ERROR_NON_NUMERIC_FOUND';
 679                          }
 680                          else
 681                          {
 682                              // it's ok, add it on
 683                              $GLOBALS['_xh']['value']=(int)$GLOBALS['_xh']['ac'];
 684                          }
 685                      }
 686                      //$GLOBALS['_xh']['ac']=''; // is this necessary?
 687                      $GLOBALS['_xh']['lv']=3; // indicate we've found a value
 688                      break;
 689                  case 'NAME':
 690                      $GLOBALS['_xh']['valuestack'][count($GLOBALS['_xh']['valuestack'])-1]['name'] = $GLOBALS['_xh']['ac'];
 691                      break;
 692                  case 'MEMBER':
 693                      //$GLOBALS['_xh']['ac']=''; // is this necessary?
 694                      // add to array in the stack the last element built,
 695                      // unless no VALUE was found
 696                      if ($GLOBALS['_xh']['vt'])
 697                      {
 698                          $vscount = count($GLOBALS['_xh']['valuestack']);
 699                          $GLOBALS['_xh']['valuestack'][$vscount-1]['values'][$GLOBALS['_xh']['valuestack'][$vscount-1]['name']] = $GLOBALS['_xh']['value'];
 700                      } else
 701                          error_log('XML-RPC: missing VALUE inside STRUCT in received xml');
 702                      break;
 703                  case 'DATA':
 704                      //$GLOBALS['_xh']['ac']=''; // is this necessary?
 705                      $GLOBALS['_xh']['vt']=null; // reset this to check for 2 data elements in a row - even if they're empty
 706                      break;
 707                  case 'STRUCT':
 708                  case 'ARRAY':
 709                      // fetch out of stack array of values, and promote it to current value
 710                      $curr_val = array_pop($GLOBALS['_xh']['valuestack']);
 711                      $GLOBALS['_xh']['value'] = $curr_val['values'];
 712                      $GLOBALS['_xh']['vt']=strtolower($name);
 713                      if (isset($curr_val['php_class']))
 714                      {
 715                          $GLOBALS['_xh']['php_class'] = $curr_val['php_class'];
 716                      }
 717                      break;
 718                  case 'PARAM':
 719                      // add to array of params the current value,
 720                      // unless no VALUE was found
 721                      if ($GLOBALS['_xh']['vt'])
 722                      {
 723                          $GLOBALS['_xh']['params'][]=$GLOBALS['_xh']['value'];
 724                          $GLOBALS['_xh']['pt'][]=$GLOBALS['_xh']['vt'];
 725                      }
 726                      else
 727                          error_log('XML-RPC: missing VALUE inside PARAM in received xml');
 728                      break;
 729                  case 'METHODNAME':
 730                      $GLOBALS['_xh']['method']=preg_replace('/^[\n\r\t ]+/', '', $GLOBALS['_xh']['ac']);
 731                      break;
 732                  case 'NIL':
 733                  case 'EX:NIL':
 734                      if ($GLOBALS['xmlrpc_null_extension'])
 735                      {
 736                          $GLOBALS['_xh']['vt']='null';
 737                          $GLOBALS['_xh']['value']=null;
 738                          $GLOBALS['_xh']['lv']=3;
 739                          break;
 740                      }
 741                      // drop through intentionally if nil extension not enabled
 742                  case 'PARAMS':
 743                  case 'FAULT':
 744                  case 'METHODCALL':
 745                  case 'METHORESPONSE':
 746                      break;
 747                  default:
 748                      // End of INVALID ELEMENT!
 749                      // shall we add an assert here for unreachable code???
 750                      break;
 751              }
 752          }
 753      }

 906                  }

 908                  {
 909                      $this->password = $parts['pass'];
 910                  }

 913              {
 914                  $this->path='/'.$path;
 915              }
 916              else
 917              {

 919              }

1704          /**
1705          * Send an array of request messages and return an array of responses.
1706          * Unless $this->no_multicall has been set to true, it will try first
1707          * to use one single xmlrpc call to server method system.multicall, and
1708          * revert to sending many successive calls in case of failure.
1709          * This failure is also stored in $this->no_multicall for subsequent calls.
1710          * Unfortunately, there is no server error code universally used to denote
1711          * the fact that multicall is unsupported, so there is no way to reliably
1712          * distinguish between that and a temporary failure.
1713          * If you are sure that server supports multicall and do not want to
1714          * fallback to using many single calls, set the fourth parameter to FALSE.
1715          *
1716          * NB: trying to shoehorn extra functionality into existing syntax has resulted
1717          * in pretty much convoluted code...
1718          *
1719          * @param array $msgs an array of xmlrpcmsg objects
1720          * @param integer $timeout connection timeout (in seconds)
1721          * @param string $method the http protocol variant to be used
1722          * @param boolean fallback When true, upon receiveing an error during multicall, multiple single calls will be attempted
1723          * @return array
1724          * @access public
1725          */
1726  		function multicall($msgs, $timeout=0, $method='', $fallback=true)
1727          {
1728              if ($method == '')

1730                  $method = $this->method;
1731              }

1733              {
1734                  $results = $this->_try_multicall($msgs, $timeout, $method);
1735                  if(is_array($results))
1736                  {
1737                      // System.multicall succeeded
1738                      return $results;
1739                  }
1740                  else
1741                  {
1742                      // either system.multicall is unsupported by server,
1743                      // or call failed for some other reason.
1744                      if ($fallback)
1745                      {
1746                          // Don't try it next time...
1747                          $this->no_multicall = true;
1748                      }
1749                      else
1750                      {
1751                          if (is_a($results, 'xmlrpcresp'))
1752                          {
1753                              $result = $results;
1754                          }
1755                          else
1756                          {
1757                              $result = new xmlrpcresp(0, $GLOBALS['xmlrpcerr']['multicall_error'], $GLOBALS['xmlrpcstr']['multicall_error']);
1758                          }
1759                      }
1760                  }
1761              }
1762              else
1763              {
1764                  // override fallback, in case careless user tries to do two
1765                  // opposite things at the same time
1766                  $fallback = true;
1767              }
1768  
1769              $results = array();
1770              if ($fallback)
1771              {
1772                  // system.multicall is (probably) unsupported by server:
1773                  // emulate multicall via multiple requests
1774                  foreach($msgs as $msg)
1775                  {
1776                      $results[] =& $this->send($msg, $timeout, $method);
1777                  }
1778              }

1782                  // since we should always return an array of responses,
1783                  // return an array with the same error repeated n times
1784                  foreach($msgs as $msg)
1785                  {
1786                      $results[] = $result;
1787                  }
1788              }
1789              return $results;

3684          //   where value is of type token, no LWS allowed between 'charset' and value

3697          // 2 - scan the first bytes of the data for a UTF-16 (or other) BOM pattern
3698          //     (source: http://www.w3.org/TR/2000/REC-xml-20001006)
3699          //     NOTE: actually, according to the spec, even if we find the BOM and determine

3718          // Details:
3719          // SPACE:         (#x20 | #x9 | #xD | #xA)+ === [ \x9\xD\xA]+
3720          // EQ:            SPACE?=SPACE? === [ \x9\xD\xA]*=[ \x9\xD\xA]*

3722              '\s+encoding\s*=\s*' . "((?:\"[A-Za-z][A-Za-z0-9._-]*\")|(?:'[A-Za-z][A-Za-z0-9._-]*'))/",
3723              $xmlchunk, $matches))
3724          {
3725              return strtoupper(substr($matches[2], 1, -1));
3726          }

3728          // 4 - if mbstring is available, let it do the guesswork
3729          // NB: we favour finding an encoding that is compatible with what we can process
3730          if(extension_loaded('mbstring'))
3731          {
3732              if($encoding_prefs)
3733              {

3735              }
3736              else
3737              {
3738                  $enc = mb_detect_encoding($xmlchunk);
3739              }
3740              // NB: mb_detect likes to call it ascii, xml parser likes to call it US_ASCII...
3741              // IANA also likes better US-ASCII, so go with it
3742              if($enc == 'ASCII')
3743              {
3744                  $enc = 'US-'.$enc;
3745              }
3746              return $enc;
3747          }
3748          else
3749          {
3750              // no encoding specified: as per HTTP1.1 assume it is iso-8859-1?
3751              // Both RFC 2616 (HTTP 1.1) and 1945 (HTTP 1.0) clearly state that for text/xxx content types
3752              // this should be the standard. And we should be getting text/xml as request and response.
3753              // BUT we have to be backward compatible with the lib, which always used UTF-8 as default...
3754              return $GLOBALS['xmlrpc_defencoding'];
3755          }
3756      }
