  59                  {
  60                      $params .= 'text_'.$r.'=';
  61                      $first   = true;
  62                      foreach ( $v[$r] as $tk => $tv )
  63                      {
  64                          if ( $first )
  65                          {
  66                              $first = false;
  67                          }
  68                          else
  69                          {
  70                              $params .= ';';
  71                          }
  72                          $params .= $tk.':'.$tv;
  73                      }
  74                      $params .= '&';
  75                  }
  76                  break;
  77  
  78              case 'link':

  80                  {
  81                      $params .= 'link_'.$r.'=';
  82                      $first   = true;
  83                      foreach ( $v as $lk => $lt )
  84                      {
  85                          if ( $first )
  86                          {
  87                              $first = false;
  88                          }
  89                          else
  90                          {
  91                              $params .= ';';
  92                          }
