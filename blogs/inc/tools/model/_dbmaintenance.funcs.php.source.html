 115  /**

 116   * Optimize DB tables (MyISAM & InnoDB)

 117   *

 118   * @param boolean Display messages

 119   * @param boolean TRUE - to make optimize query for each table separately

 120   * @return array Results of the mysql command 'OPTIMIZE'

 121   */
 122  function dbm_optimize_tables( $display_messages = true, $separate_tables = true )
 123  {
 124      global $tableprefix, $DB, $Timer, $dbm_tables_count;
 125  
 126      $tables = $DB->get_results( 'SHOW TABLE STATUS FROM `'.$DB->dbname.'` LIKE \''.$tableprefix.'%\'');
 127  
 128      $myisam_tables = array();
 129      $innodb_tables = array();
 130      foreach( $tables as $table )
 131      {
 132          // Before MySQL 4.1.2, the "Engine" field was labeled as "Type".

 133          if( ( ( isset( $table->Engine ) && $table->Engine == 'MyISAM' )
 134                  || ( isset( $table->Type ) && $table->Type == 'MyISAM' ) )
 135              && $table->Data_free )
 136          {    // Optimization needed for MyISAM table
 137              $myisam_tables[] = $table->Name;
 138          }
 139          else if( ( ( isset( $table->Engine ) && $table->Engine == 'InnoDB' )
 140                  || ( isset( $table->Type ) && $table->Type == 'InnoDB' ) )
 141              && $table->Data_free )
 142          {    // Optimization needed for InnoDB table
 143              $innodb_tables[] = $table->Name;
 144          }
 145      }
 146      $dbm_tables_count = count( $myisam_tables ) + count( $innodb_tables );
 147  
 148      // Optimize MyISAM tables

 149      $myisam_results = dbm_optimize_tables_process( $display_messages, $separate_tables, $myisam_tables, 'MyISAM' );
 150      $results = $myisam_results;
 151  
 152      // Optimize InnoDB tables

 153      $innodb_results = dbm_optimize_tables_process( $display_messages, $separate_tables, $innodb_tables, 'InnoDB' );
 154      $results = array_merge( $results, $innodb_results );
 155  
 156      return $results;
 157  }

 160  /**

 161   * Optimize process DB tables (MyISAM & InnoDB)

 162   *

 163   * @param boolean Display messages

 164   * @param boolean TRUE - to make optimize query for each table separately

 165   * @param array Tables

 166   * @param string Table type: 'MyISAM' or 'InnoDB'

 167   * @return array Results of the mysql command 'OPTIMIZE'

 168   */
 169  function dbm_optimize_tables_process( $display_messages = true, $separate_tables = true, $tables = array(), $table_type = '' )
 170  {
 171      global $DB;
 172  
 173      load_class( '_core/model/_timer.class.php', 'Timer' );
 174      $Timer = new Timer('dbm_optimize');
 175  
 176      $results = array();
 177  
 178      if( $display_messages )
 179      { // Display messages
 180          echo '<b>'.sprintf( T_('Optimize %s tables...'), $table_type ).'</b><br />';
 181          evo_flush();
 182      }
 183      $timer_name = 'optimize_'.strtolower( $table_type );
 184  
 185      $Timer->start( $timer_name );
 186  
 187      if( !empty( $tables ) )
 188      { // Optimize tables
 189          if( $separate_tables )
 190          { // Optimize each table in separate query
 191              foreach( $tables as $table )
 192              {
 193                  $Timer->start( $timer_name.'_table' );
 194                  $table_results = $DB->get_results( 'OPTIMIZE NO_WRITE_TO_BINLOG TABLE '.$table );
 195                  $Timer->stop( $timer_name.'_table' );
 196                  if( $display_messages )
 197                  { // Display messages
 198                      dbm_display_result_messages( $table_results, 'optimize' );
 199                      echo '<b>'.sprintf( T_('Time: %s seconds'), $Timer->get_duration( $timer_name.'_table' ) ).'</b><br /><br />';
 200                  }
 201                  evo_flush();
 202                  $results = array_merge( $results, $table_results );
 203              }
 204          }
 205          else
 206          { // Optimize all table by one query, Used for cron job
 207              $results = $DB->get_results( 'OPTIMIZE NO_WRITE_TO_BINLOG TABLE '.implode( ', ', $tables ) );
 208          }
 209      }
 210  
 211      $Timer->stop( $timer_name );
 212  
 213      if( $display_messages )
 214      { // Display messages
 215          if( !$separate_tables || empty( $tables ) )
 216          { // Display full report log for case when the tables were optimized by one query
 217              dbm_display_result_messages( $results, 'optimize' );
 218          }
 219          echo '<b>'.sprintf( T_('Full execution time: %s seconds'), $Timer->get_duration( $timer_name ) ).'</b><br /><br />';
 220          evo_flush();
 221      }
 222  
 223      return $results;
 224  }

 227  /**

 228   * Print on screen the messages of 'OPTIMIZE', 'CHECK' OR 'ANALYZE' commands

 229   *

 230   * @param array Results of the mysql commands 'OPTIMIZE', 'CHECK' OR 'ANALYZE'

 231   * @param string mysql command type: optimize, check, analyze

 232   */
 233  function dbm_display_result_messages( $results, $command_type )
 234  {
 235      switch( $command_type )
 236      {
 237          case 'optimize':
 238              $params = array(
 239                      'message_ok'    => T_('Database table %s is optimized.'),
 240                      'message_error' => T_('Database table %s could not be optimized. Message: %s'),
 241                      'message_done'  => T_('All database tables are already optimized.'),
 242                  );
 243              break;
 244  
 245          case 'check':
 246              $params = array(
 247                      'message_ok'    => T_('Database table %s is checked.'),
 248                      'message_error' => T_('Database table %s could not be checked. Message: %s'),
 249                      'message_done'  => T_('All database tables are already checked.'),
 250                  );
 251              break;
 252  
 253          case 'analyze':
 254              $params = array(
 255                      'message_ok'    => T_('Database table %s is analyzed.'),
 256                      'message_error' => T_('Database table %s could not be analyzed. Message: %s'),
 257                      'message_done'  => T_('All database tables are already analyzed.'),
 258                  );
 259              break;
 260  
 261          default:
 262              // Don't support unknown commants, Exit here

 263              return;
 264      }
 265  
 266      if( count( $results ) > 0 )
 267      {
 268          foreach( $results as $result )
 269          {
 270              if( $result->Msg_type == 'status' && $result->Msg_text == 'OK' )
 271              { // OK
 272                  echo sprintf( $params['message_ok'], '<b>'.$result->Table.'</b>' ).'<br />';
 273              }
 274              elseif( $result->Msg_type == 'note' && $result->Msg_text == 'Table does not support optimize, doing recreate + analyze instead' )
 275              { // This warning is comming for every innodb table, but that is normal, Display for info
 276                  echo sprintf( T_('Database table %s does not support optimize, doing recreate + analyze instead'), '<b>'.$result->Table.'</b>' ).'<br />';
 277              }
 278              else
 279              { // Some errors
 280                  $message_class = $result->Msg_type == 'status' ? 'orange' : 'red';
 281                  echo '<span class="'.$message_class.'">'.sprintf( $params['message_error'], '<b>'.$result->Table.'</b>', '"'.$result->Msg_text.'"' ).'</span><br />';
 282              }
 283          }
 284      }
 285      else
 286      { // No tables found to optimize, probably all tables already were optimized
 287          echo $params['message_done'].'<br />';
 288      }
 289  }

 304      // Delete the posts only by these IDs

 315          $SQL->WHERE_and( 'post_ID IN ( '.$post_IDs.' )' );

 317

 323                  $num_deleted++;

 325              else

 331                  echo '. ';
 332                  evo_flush();
 333              }
 334          }
 335      }
 336

 417   * Find and delete orphan File objects with no matching file on disk

 418   */
 419  function dbm_delete_orphan_files()
 420  {
 421      global $DB, $admin_url;
 422  
 423      $FileCache = & get_FileCache();
 424      $FileCache->clear();
 425  
 426      echo T_('Deleting of the orphan File objects from the database...');
 427      evo_flush();
 428  
 429      $files_SQL = new SQL();
 430      $files_SQL->SELECT( '*' );
 431      $files_SQL->FROM( 'T_files' );
 432      $files_SQL->ORDER_BY( 'file_ID' );
 433  
 434      $count_files_valid = 0;
 435      $count_files_invalid = 0;
 436      $count_files_deleted = 0;
 437  
 438      $page_size = 100;
 439      $current_page = 0;
 440      // Search the files by page to save memory

 441      $files_SQL->LIMIT( '0, '.$page_size );
 442      while( $loaded_Files = $FileCache->load_by_sql( $files_SQL ) )
 443      { // Check all loaded files
 444          foreach( $loaded_Files as $File )
 445          {
 446              if( is_null( $File ) )
 447              { // The File object couldn't be created because the db entry is invalid
 448                  $count_files_invalid++;
 449                  continue;
 450              }
 451              if( $File->exists() )
 452              { // File exists on the disk
 453                  $count_files_valid++;
 454              }
 455              else
 456              { // File doesn't exist on the disk, Remove it from DB
 457                  $File->dbdelete();
 458                  $count_files_deleted++;
 459              }
 460          }
 461  
 462          echo ' .';
 463          evo_flush();
 464  
 465          // Clear cache after each page to save memory

 466          $FileCache->clear();
 467  
 468          $current_page++;
 469          $files_SQL->LIMIT( ( $current_page * $page_size ).', '.$page_size );
 470      }
 471  
 472      echo 'OK<p>';
 473      echo sprintf( T_('The number of deleted orphan File objects: %d.'), $count_files_deleted ).'<br />';
 474      echo sprintf( T_('The number of valid File objects in the database: %d.'), $count_files_valid ).'</p>';
 475  
 476      if( $count_files_invalid )
 477      { // There are invalid files in the database
 478          // Display warning to show that the 'Remove orphan file roots' tool should be also called

 479          $remove_orphan_file_roots = 'href="'.$admin_url.'ctrl=tools&amp;action=delete_orphan_file_roots&amp;'.url_crumb('tools').'"';
 480          $invalid_files_note = ( $count_files_invalid == 1 ) ? T_('An invalid File object was found in the database.') : sprintf( T_('%d invalid File objects were found in the database.'), $count_files_invalid );
 481          echo '<p class="warning">'.$invalid_files_note."<br/>"
 482              .sprintf( T_('It is strongly recommended to also execute the &lt;<a %s>Remove orphan file roots</a>&gt; tool to remove invalid files from the database and from the disk as well!'), $remove_orphan_file_roots )
 483              .'</p>';
 484      }
 485  }

 491  function dbm_delete_orphan_file_roots()
 492  {
 493      global $DB, $media_path;
 494  
 495      echo T_('Removing of the orphan file roots recursively with all of the content... ');
 496      evo_flush();
 497  
 498      // Store all directories that must be deleted

 499      $delete_dirs = array();
 500  
 501      /* BLOGS */

 502  
 503      // Get the media diretories of all existing blogs

 504      $BlogCache = & get_BlogCache();
 505      $BlogCache->load_all();
 506      $blog_dirs = array();
 507      foreach( $BlogCache->cache as $Blog )
 508      {
 509          $blog_dirs[] = $Blog->get_media_dir();
 510      }
 511      $BlogCache->clear();
 512  
 513      $media_path_blogs = $media_path.'blogs/';
 514  
 515      if( ( $media_dir_blogs = @opendir( $media_path_blogs ) ) === false )

 518      }
 519      else
 520      {
 521          // Find the blog dirs that must be deleted

 522          while( ( $folder = readdir( $media_dir_blogs ) ) !== false )
 523          {
 524              if( $folder == '.' || $folder == '..' || ! is_dir( $media_path_blogs.$folder ) )
 525              { // Skip files
 526                  continue;
 527              }
 528              if( ! in_array( $media_path_blogs.$folder.'/', $blog_dirs ) )
 529              { // This dir must be deleted because it is not media dir of the existing blogs
 530                  $delete_dirs[] = $media_path_blogs.$folder.'/';
 531              }
 532          }
 533  
 534          closedir( $media_dir_blogs );
 535      }
 536  
 537      /* USERS */

 538      echo '. ';
 539      evo_flush();
 540  
 541      // Get logins of all existing users

 542      $SQL = new SQL();
 543      $SQL->SELECT( 'user_login' );
 544      $SQL->FROM( 'T_users' );
 545      $user_logins = $DB->get_col( $SQL->get() );
 546  
 547      $media_path_users = $media_path.'users/';
 548  
 549      if( ( $media_dir_users = @opendir( $media_path_users ) ) === false )
 550      { // Could not open users media dir
 551          echo '<p class="red">'.sprintf( T_('Cannot open users media directory %s'), '<b>'.$media_path_users.'</b>' ).'</p>';
 552      }
 553      else
 554      {
 555          // Find the user dirs that must be deleted

 556          while( ( $folder = readdir( $media_dir_users ) ) !== false )
 557          {
 558              if( $folder == '.' || $folder == '..' || ! is_dir( $media_path_users.$folder ) )
 559              { // Skip files
 560                  continue;
 561              }
 562              if( ! in_array( $folder, $user_logins ) )
 563              { // This dir must be deleted because it is not media dir of the existing users
 564                  $delete_dirs[] = $media_path_users.$folder.'/';
 565              }
 566          }
 567  
 568          closedir( $media_dir_users );
 569      }
 570  
 571      /* DELETE broken  file roots */

 572      echo '. ';
 573      evo_flush();
 574  
 575      foreach( $delete_dirs as $delete_dir )
 576      {
 577          if( rmdir_r( $delete_dir ) )
 578          { // Success deleting
 579              echo '<p class="green">'.sprintf( T_('Invalid file root %s was found and removed with all of its content.'), '<b>'.$delete_dir.'</b>' ).'</p>';
 580          }
 581          else
 582          { // Failed deleting
 583              echo '<p class="red">'.sprintf( T_('Cannot delete directory %s. Please check the permissions or delete it manually.'), '<b>'.$delete_dir.'</b>' ).'</p>';
 584          }
 585      }
 586  
 587      /* DELETE orphan DB file records of the blogs and the users */

 588      echo '. ';
 589      evo_flush();
 590  
 591      $count_files_deleted = $DB->query( 'DELETE f, l, lv FROM T_files AS f
 592               LEFT JOIN T_links AS l ON l.link_file_ID = f.file_ID
 593               LEFT JOIN T_links__vote AS lv ON l.link_ID = lv.lvot_link_ID
 594          WHERE ( file_root_type = "collection"
 595                  AND file_root_ID NOT IN ( SELECT blog_ID FROM T_blogs ) )
 596             OR ( file_root_type = "user"
 597                  AND file_root_ID NOT IN ( SELECT user_ID FROM T_users ) )' );
 598  
 599      echo 'OK.<p>';
 600      echo sprintf( T_('%d File roots have been removed from the disk.'), count( $delete_dirs ) ).'<br />';
 601      echo sprintf( T_('%d File objects have been deleted from DB.'), intval( $count_files_deleted ) ).'</p>';
 602  }

 618      {
 619          $Item = $ItemCache->get_by_ID($item_ID);
 620  
 621          $prev_urltitle = $Item->get( 'urltitle' );
 622          $item_title = $Item->get( 'title' );

 669  function dbm_check_tables( $display_messages = true, $separate_tables = true )
 670  {
 671      global $DB, $tableprefix, $dbm_tables_count;

 673      load_class( '_core/model/_timer.class.php', 'Timer' );
 674      $Timer = new Timer('dbm_check');
 675  
 676      // Get all table names from DB

 678      $tables_names = array();
 679      foreach( $tables as $table )

 684

 688          evo_flush();

 692      if( $separate_tables )
 693      { // Check each table in separate query
 694          $check_results = array();
 695          foreach( $tables_names as $table )
 696          {
 697              $Timer->start( 'check_one_table' );
 698              $table_results = $DB->get_results( 'CHECK TABLE '.$table.' FAST' );
 699              $Timer->stop( 'check_one_table' );
 700              if( $display_messages )
 701              { // Display messages
 702                  dbm_display_result_messages( $table_results, 'check' );
 703                  echo '<b>'.sprintf( T_('Time: %s seconds'), $Timer->get_duration( 'check_one_table' ) ).'</b><br /><br />';
 704              }
 705              evo_flush();
 706              $check_results = array_merge( $check_results, $table_results );
 707          }
 708      }
 709      else
 710      { // Check all table by one query, Used for cron job
 711          $check_results = $DB->get_results( 'CHECK TABLE '.implode( ', ', $tables_names ).' FAST' );
 712      }
 713      $Timer->stop( 'check_tables' );
 714  
 715      if( $display_messages )
 716      { // Display messages
 717          if( !$separate_tables )
 718          { // Display full report log for case when the tables were checked by one query
 719              dbm_display_result_messages( $check_results, 'check' );
 720          }
 721          echo '<b>'.sprintf( T_('Full execution time: %s seconds'), $Timer->get_duration( 'check_tables' ) ).'</b><br />';
 722      }
 723  
 724      return $check_results;
