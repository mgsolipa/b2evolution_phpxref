 190  /**

 191   * Get result info of email log

 192   *

 193   * @param string Result ( 'ok', 'error', 'blocked' )

 194   * @param boolean Params

 195   * @return string Result info

 196   */
 197  function emlog_result_info( $result, $params = array() )
 198  {
 199      $params = array_merge( array(
 200              'display_icon'  => true,    // Display icon
 201              'display_text'  => true,    // Display text
 202              'link_blocked'  => false, // TRUE - to display 'Blocked' as link to go to page with Blocked email adresses with filter by email
 203              'email'         => '',    // Email address to filter
 204          ), $params );
 205  
 206      $result_info = '';
 207  
 208      switch( $result )
 209      {
 210          case 'ok':
 211              if( $params['display_icon'] )
 212              {
 213                  $result_info .= get_icon( 'bullet_green', 'imgtag', array( 'alt' => T_('Ok') ) );
 214              }
 215              if( $params['display_text'] )
 216              {
 217                  $result_info .= ' '.T_('Ok');
 218              }
 219              break;
 220  
 221          case 'error':
 222              if( $params['display_icon'] )
 223              {
 224                  $result_info .= get_icon( 'bullet_red', 'imgtag', array( 'alt' => T_('Error') ) );
 225              }
 226              if( $params['display_text'] )
 227              {
 228                  $result_info .= ' '.T_('Error');
 229              }
 230              break;
 231  
 232          case 'blocked':
 233              if( $params['display_icon'] )
 234              {
 235                  $result_info .= get_icon( 'bullet_black', 'imgtag', array( 'alt' => T_('Blocked') ) );
 236              }
 237              if( $params['display_text'] )
 238              {
 239                  if( $params['link_blocked'] && !empty( $params['email'] ) )
 240                  {    // Create a link for email address
 241                      global $admin_url;
 242                      $result_info .= ' <a href="'.$admin_url.'?ctrl=email&amp;tab=blocked&amp;email='.$params['email'].'">'.T_('Blocked').'</a>';
 243                  }
 244                  else
 245                  {    // Display only text
 246                      $result_info .= ' '.T_('Blocked');
 247                  }
 248              }
 249              break;
 250      }
 251  
 252      return $result_info;
 253  }

 259   * @param integer User ID

 261   * @param string Subject

 263   * @param string Headers

 266  function mail_log( $user_ID, $to, $subject, $message, $headers, $result )
 267  {
 268      global $DB, $servertimenow;
 269  
 270      /**

 271       * @var integer|NULL This global var stores ID of the last inserted mail log

 272       */
 273      global $mail_log_insert_ID;
 274      $mail_log_insert_ID = NULL;
 275  
 276      if( empty( $user_ID ) )
 277      {
 278          $user_ID = NULL;
 279      }
 280  
 281      $to = evo_strtolower( $to );
 282  
 283      // Insert mail log

 284      $DB->query( 'INSERT INTO T_email__log
 285          ( emlog_user_ID, emlog_to, emlog_result, emlog_subject, emlog_message, emlog_headers )
 286          VALUES
 287          ( '.$DB->quote( $user_ID ).',
 288            '.$DB->quote( $to ).',
 289            '.$DB->quote( $result ).',
 290            '.$DB->quote( evo_substr( $subject, 0, 255 ) ).',
 291            '.$DB->quote( $message ).',
 292            '.$DB->quote( $headers ).' )' );
 293  
 294      // Store ID of new inserted mail log

 295      $mail_log_insert_ID = $DB->insert_id;
 296  
 297      if( $result == 'ok' )
 298      { // Save a report about sending of this message in the table T_email__address
 299          // The mail sending is susccess. Update last sent date and increase a counter

 300          $DB->query( 'INSERT INTO T_email__address ( emadr_address, emadr_sent_count, emadr_sent_last_returnerror, emadr_last_sent_ts )
 301              VALUES( '.$DB->quote( $to ).', 1, 1, '.$DB->quote( date( 'Y-m-d H:i:s', $servertimenow ) ).' )
 302              ON DUPLICATE KEY UPDATE
 303                  emadr_sent_count = emadr_sent_count + 1,
 304                  emadr_sent_last_returnerror = emadr_sent_last_returnerror + 1,
 305                  emadr_last_sent_ts = '.$DB->quote( date( 'Y-m-d H:i:s', $servertimenow ) ) );
 306      }

 310  /**

 311   * Load the blocked emails from DB in cache

 312   *

 313   * @param array User IDs

 314   * @param array Blocked statuses to know what emails are blocked to send

 315   *     'unknown'     - Unknown

 316   *     'warning'     - Warning

 317   *     'suspicious1' - Suspicious 1

 318   *     'suspicious2' - Suspicious 2

 319   *     'suspicious3' - Suspicious 3

 320   *     'prmerror'    - Permament error

 321   *     'spammer'     - Spammer

 322   */
 323  function load_blocked_emails( $user_IDs, $blocked_statuses = array() )
 324  {
 325      global $DB, $cache_mail_is_blocked_status;
 326  
 327      if( empty( $user_IDs ) )
 328      { // No users, Exit here
 329          return;
 330      }
 331  
 332      if( !isset( $cache_mail_is_blocked_status ) )
 333      { // Init array first time
 334          $cache_mail_is_blocked_status = array();
 335      }
 336  
 337      $status_filter_name = implode( '_', $blocked_statuses );
 338      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ] ) )
 339      { // Init subarray for each filter by statuses
 340          $cache_mail_is_blocked_status[ $status_filter_name ] = array();
 341      }
 342  
 343      $SQL = new SQL();
 344      $SQL->SELECT( 'user_email, emadr_ID' );
 345      $SQL->FROM( 'T_users' );
 346      $SQL->FROM_add( 'LEFT JOIN T_email__address
 347           ON user_email = emadr_address
 348          AND '.get_mail_blocked_condition( true, $blocked_statuses ) );
 349      $SQL->WHERE( 'user_ID IN ( '.$DB->quote( $user_IDs ).' )' );
 350      $blocked_emails = $DB->get_assoc( $SQL->get() );
 351  
 352      foreach( $blocked_emails as $email => $email_blocked_ID )
 353      { // The blocked email has TRUE value; Trust emails - FALSE
 354          $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] = (boolean) $email_blocked_ID;
 355      }
 356  }

 359  /**

 360   * Check if the email address is blocked

 361   *

 362   * @param string Email address

 363   * @param array Blocked statuses to know what emails are blocked to send

 364   *     'unknown'     - Unknown

 365   *     'warning'     - Warning

 366   *     'suspicious1' - Suspicious 1

 367   *     'suspicious2' - Suspicious 2

 368   *     'suspicious3' - Suspicious 3

 369   *     'prmerror'    - Permament error

 370   *     'spammer'     - Spammer

 371   * @return boolean TRUE

 372   */
 373  function mail_is_blocked( $email, $blocked_statuses = array() )
 374  {
 375      global $cache_mail_is_blocked_status;
 376  
 377      if( !isset( $cache_mail_is_blocked_status ) )
 378      { // Init array first time
 379          $cache_mail_is_blocked_status = array();
 380      }
 381  
 382      $status_filter_name = implode( '_', $blocked_statuses );
 383      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ] ) )
 384      { // Init subarray for each filter by statuses
 385          $cache_mail_is_blocked_status[ $status_filter_name ] = array();
 386      }
 387  
 388      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] ) )
 389      { // If we check status of this email first time - get it from DB and store in cache
 390          global $DB;
 391          $SQL = new SQL();
 392          $SQL->SELECT( 'emadr_ID' );
 393          $SQL->FROM( 'T_email__address' );
 394          $SQL->WHERE( 'emadr_address = '.$DB->quote( evo_strtolower( $email ) ) );
 395          $SQL->WHERE_and( get_mail_blocked_condition( true, $blocked_statuses ) );
 396          $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] = (boolean) $DB->get_var( $SQL->get() );
 397      }
 398  
 399      // Get email block status from cache variable

 400      return $cache_mail_is_blocked_status[ $status_filter_name ][ $email ];
 401  }

 418  function get_mail_blocked_condition( $is_blocked = true, $blocked_statuses = array() )
 419  {

 438  function blocked_emails_memorize( $email )

 451              if( isset( $cache_blocked_emails[ $email ] ) )
 452              { // Icrease a count of blocked email
 453                  $cache_blocked_emails[ $email ]++;
 454              }
 455              else
 456              {
 457                  $cache_blocked_emails[ $email ] = 1;

 481  /**

 482   * Parse message of mail if content is in multipart or HTML format

 483   *

 484   * @param string Headers

 485   * @param string Message

 486   * @return array|boolean Mail data or FALSE when message has only Plain Text content

 487   */
 488  function mail_log_parse_message( $headers, $message )
 489  {
 490      preg_match( '/Content-Type: ([^;]+);/i', $headers, $header_matches );
 491  
 492      if( empty( $header_matches[1] ) )
 493      { // Incorrect headers, Exit here
 494          return false;
 495      }
 496  
 497      // Get only <body> content of html email message

 498      $html_search_body = '#.+<body[^>]*>(.+)</body>.+#is';
 499      $html_replace_body = '$1';
 500  
 501      $data = array();
 502  
 503      if( $header_matches[1] == 'text/html' )
 504      { // Message has one content in HTML format
 505          $data['html'] = array(
 506                  'type' => $header_matches[0],
 507                  'content' => preg_replace( $html_search_body, $html_replace_body, $message ),
 508              );
 509  
 510          return $data;
 511      }
 512      elseif( $header_matches[1] != 'multipart/mixed' )
 513      { // Message content is not multipart
 514          return false;
 515      }
 516  
 517      preg_match( '/Content-Type: multipart\/alternative; boundary="([^;]+)"/i', $message, $boundary_matches );
 518  
 519      if( empty( $boundary_matches ) || empty( $boundary_matches[1] ) )
 520      { // No found boundary delimiter of message contents
 521          return false;
 522      }
 523  
 524      $boundary_delimiter = '--'.$boundary_matches[1];
 525  
 526      $contents = explode( $boundary_delimiter, $message );
 527      unset( $contents[0] );
 528      unset( $contents[3] );
 529  
 530      foreach( $contents as $content )
 531      {
 532          preg_match( '/(Content-Type: ([^;]+);(.+)){1}\n([\s\S\n]+)/i', $content, $type_matches );
 533  
 534          switch( $type_matches[2] )
 535          {
 536              case 'text/html':
 537                  // Get data of Plain Text content

 538                  $data['html'] = array(
 539                          'type' => $type_matches[1],
 540                          'content' => preg_replace( $html_search_body, $html_replace_body, $type_matches[4] ),
 541                      );
 542                  break;
 543  
 544              case 'text/plain':
 545                  // Get data of HTML content

 546                  $data['text'] = array(
 547                          'type' => $type_matches[1],
 548                          'content' => $type_matches[4],
 549                      );
 550                  break;
 551          }
 552      }
 553  
 554      return $data;
 555  }
