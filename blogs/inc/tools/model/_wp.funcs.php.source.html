 529          $new_Chapter->set( 'urlname', $wp_Blog->get( 'urlname' ).'-main' );
 530          $new_Chapter->dbinsert();
 531          $category_default = $new_Chapter->ID;
 532      }
 533  
 534      /* Import tags */

 535      $tags = array();
 536      if( isset( $xml_data['tags'] ) && count( $xml_data['tags'] ) > 0 )
 537      {
 538          echo T_('Importing the tags... ');
 539          evo_flush();
 540  
 541          // Get existing tags

 542          $SQL = new SQL();
 543          $SQL->SELECT( 'tag_name, tag_ID' );
 544          $SQL->FROM( 'T_items__tag' );
 545          $tags = $DB->get_assoc( $SQL->get() );
 546  
 547          $tags_count = 0;
 548          foreach( $xml_data['tags'] as $tag )
 549          {
 550              if( empty( $tags[ (string) $tag['tag_name'] ] ) )
 551              {    // Insert new tag into DB if tag doesn't exist with current name
 552                  mysql_query( 'INSERT INTO '.$tableprefix.'items__tag ( tag_name )
 553                      VALUES ( '.$DB->quote( $tag['tag_name'] ).' )',
 554                      $DB->dbhandle );
 555                  $tag_ID = mysql_insert_id( $DB->dbhandle );
 556                  // Save new tag

 557                  $tags[ $tag['tag_name'] ] = (string) $tag_ID;
 558                  $tags_count++;
 559              }
 560          }
 561          echo sprintf( T_('%d records'), $tags_count ).'<br />';
 562      }
 563  
 564  
 565      /* Import posts */

 566      $posts = array();
 567      $comments = array();
 568      if( isset( $xml_data['posts'] ) && count( $xml_data['posts'] ) > 0 )
 569      {
 570          load_class( 'items/model/_item.class.php', 'Item' );
 571  
 572          // Set status's links between WP and b2evo names

 573          $post_statuses = array(
 574              'publish'    => 'published',
 575              'pending'    => 'deprecated',
 576              'deprecated' => 'deprecated',
 577              'protected'  => 'protected',
 578              'private'    => 'private',
 579              'redirected' => 'redirected',
 580              'draft'      => 'draft',
 581          );
 582  
 583          // Get post types

 584          $SQL = new SQL();
 585          $SQL->SELECT( 'LOWER( ptyp_name ), ptyp_ID' );
 586          $SQL->FROM( 'T_items__type' );
 587          $post_types = $DB->get_assoc( $SQL->get() );
 588  
 589          echo T_('Importing the posts... ');
 590          evo_flush();
 591  
 592          foreach( $xml_data['posts'] as $post )
 593          {
 594              $author_ID = isset( $authors[ (string) $post['post_author'] ] ) ? $authors[ (string) $post['post_author'] ] : 1;
 595              $last_edit_user_ID = isset( $authors[ (string) $post['post_lastedit_user'] ] ) ? $authors[ (string) $post['post_lastedit_user'] ] : $author_ID;
 596  
 597              $post_main_cat_ID = $category_default;
 598              $post_extra_cat_IDs = array( $post_main_cat_ID );
 599              $post_tags = array();
 600              if( !empty( $post['terms'] ) )
 601              {    // Set categories and tags
 602                  foreach( $post['terms'] as $term )
 603                  {
 604                      switch( $term['domain'] )
 605                      {
 606                          case 'category':
 607                              if( isset( $categories[ (string) $term['slug'] ] ) )
 608                              {
 609                                  if( $post_main_cat_ID == $category_default )
 610                                  {    // Set main category
 611                                      $post_main_cat_ID = $categories[ (string) $term['slug'] ];
 612                                  }
 613                                  else
 614                                  {    // Set extra categories
 615                                      $post_extra_cat_IDs[] = $categories[ (string) $term['slug'] ];
 616                                  }
 617                              }
 618                              break;
 619  
 620                          case 'post_tag':
 621                              if( isset( $tags[ (string) $term['slug'] ] ) )
 622                              {    // Set tag
 623                                  $post_tags[] = $term['slug'];
 624                              }
 625                              break;
 626                      }
 627                  }
 628              }
 629  
 630              // Set post type ID

 631              $post_type_ID = isset( $post_types[ strtolower( $post['post_type'] ) ] ) ? $post_types[ strtolower( $post['post_type'] ) ] : '1';
 632  
 633              // Get regional IDs by their names

 634              $item_regions = wp_get_regional_data( $post['post_country'], $post['post_region'], $post['post_subregion'], $post['post_city'] );
 635  
 636              $Item = new Item();
 637              $Item->set( 'main_cat_ID', $post_main_cat_ID );
 638              $Item->set( 'creator_user_ID', $author_ID );
 639              $Item->set( 'lastedit_user_ID', $last_edit_user_ID );
 640              $Item->set( 'title', $post['post_title'] );
 641              $Item->set( 'content', $post['post_content'] );
 642              $Item->set( 'excerpt', $post['post_excerpt'] );
 643              $Item->set( 'datestart', $post['post_date'] );
 644              $Item->set( 'datecreated', !empty( $post['post_datecreated'] ) ? $post['post_datecreated'] : $post['post_date'] );
 645              $Item->set( 'datemodified', !empty( $post['post_datemodified'] ) ? $post['post_datemodified'] : $post['post_date'] );
 646              $Item->set( 'urltitle', !empty( $post['post_urltitle'] ) ? $post['post_urltitle'] : $post['post_title'] );
 647              $Item->set( 'url', $post['post_url'] );
 648              $Item->set( 'status', isset( $post_statuses[ (string) $post['status'] ] ) ? $post_statuses[ (string) $post['status'] ] : 'draft' );
 649              // If 'comment_status' has the unappropriate value set it to 'open'

 650              $Item->set( 'comment_status', ( in_array( $post['comment_status'], array( 'open', 'closed', 'disabled' ) ) ? $post['comment_status'] : 'open' ) );
 651              $Item->set( 'ptyp_ID', $post_type_ID );
 652              if( empty( $post['post_excerpt'] ) && !empty( $post['post_content'] ) )
 653              {    // Generate excerpt
 654                  $Item->set( 'excerpt', wp_generate_excerpt( $post['post_content'] ) );
 655                  $Item->set( 'excerpt_autogenerated', '1' );
 656              }
 657              $Item->set( 'extra_cat_IDs', $post_extra_cat_IDs );
 658              $Item->set( 'dateset', $post['post_date_mode'] == 'set' ? 1 : 0 );
 659              if( isset( $authors[ (string) $post['post_assigned_user'] ] ) )
 660              {
 661                  $Item->set( 'assigned_user', $authors[ (string) $post['post_assigned_user'] ] );
 662              }
 663              $Item->set( 'datedeadline', $post['post_datedeadline'] );
 664              $Item->set( 'locale', $post['post_locale'] );
 665              $Item->set( 'excerpt_autogenerated', $post['post_excerpt_autogenerated'] );
 666              $Item->set( 'titletag', $post['post_titletag'] );
 667              $Item->set( 'notifications_status', empty( $post['post_notifications_status'] ) ? 'noreq' : $post['post_notifications_status'] );
 668              $Item->set( 'renderers', array( $post['post_renderers'] ) );
 669              $Item->set( 'priority', $post['post_priority'] );
 670              $Item->set( 'featured', $post['post_featured'] );
 671              $Item->set( 'order', $post['post_order'] );
 672              if( !empty( $item_regions['country'] ) )
 673              {    // Country
 674                  $Item->set( 'ctry_ID', $item_regions['country'] );
 675                  if( !empty( $item_regions['region'] ) )
 676                  {    // Region
 677                      $Item->set( 'rgn_ID', $item_regions['region'] );
 678                      if( !empty( $item_regions['subregion'] ) )
 679                      {    // Subregion
 680                          $Item->set( 'subrg_ID', $item_regions['subregion'] );
 681                      }
 682                      if( !empty( $item_regions['city'] ) )
 683                      {    // City
 684                          $Item->set( 'city_ID', $item_regions['city'] );
 685                      }
 686                  }
 687              }
 688  
 689              if( count( $post_tags ) > 0 )
 690              {
 691                  $Item->tags = $post_tags;
 692              }
 693  
 694              $Item->dbinsert();
 695              $posts[ $post['post_id'] ] = $Item->ID;
 696  
 697              if( ! empty( $files ) && ! empty( $post['links'] ) )
 698              { // Link the files to the Item if it has them
 699                  foreach( $post['links'] as $link )
 700                  {
 701                      if( isset( $files[ $link['link_file_ID'] ] ) )
 702                      { // Link a file to Item
 703                          $File = $files[ $link['link_file_ID'] ];
 704                          $LinkOwner = new LinkItem( $Item );
 705                          $File->link_to_Object( $LinkOwner, $link['link_order'], $link['link_position'] );
 706                      }
 707                  }
 708              }
 709  
 710              if( !empty( $post['comments'] ) )
 711              { // Set comments
 712                  $comments[ $Item->ID ] = $post['comments'];
 713              }
 714          }
 715  
 716          foreach( $xml_data['posts'] as $post )
 717          {    // Set post parents
 718              if( !empty( $post['post_parent'] ) && isset( $posts[ (string) $post['post_parent'] ] ) )
 719              {
 720                  mysql_query( 'UPDATE '.$tableprefix.'items__item
 721                            SET post_parent_ID = '.$DB->quote( $posts[ (string) $post['post_parent'] ] ).'
 722                          WHERE post_ID = '.$DB->quote( $posts[ (string) $post['post_id'] ] ),
 723                      $DB->dbhandle );
 724              }
 725          }
 726  
 727          echo sprintf( T_('%d records'), count( $xml_data['posts'] ) ).'<br />';
 728      }
 729  
 730  
 731      /* Import comments */

 732      if( !empty( $comments ) )
 733      {
 734          echo T_('Importing the comments... ');
 735          evo_flush();
 736  
 737          $comments_count = 0;
 738          $comments_IDs = array();
 739          foreach( $comments as $post_ID => $comments )
 740          {
 741              if( empty( $comments ) )
 742              {    // Skip if no comments
 743                  continue;
 744              }
 745  
 746              foreach( $comments as $comment )
 747              {
 748                  $comment_author_user_ID = 0;
 749                  if( !empty( $comment['comment_user_id'] ) && isset( $authors_IDs[ (string) $comment['comment_user_id'] ] ) )
 750                  {    // Author ID
 751                      $comment_author_user_ID = $authors_IDs[ (string) $comment['comment_user_id'] ];
 752                  }
 753  
 754                  $comment_parent_ID = 0;
 755                  if( !empty( $comment['comment_parent'] ) && isset( $comments_IDs[ (string) $comment['comment_parent'] ] ) )
 756                  {    // Parent comment ID
 757                      $comment_parent_ID = $comments_IDs[ (string) $comment['comment_parent'] ];
 758                  }
 759  
 760                  unset( $comment_IP_country );
 761                  if( !empty( $comment['comment_IP_country'] ) )
 762                  {    // Get country ID by code
 763                      $CountryCache = & get_CountryCache();
 764                      if( $Country = & $CountryCache->get_by_name( $comment['comment_IP_country'], false ) )
 765                      {
 766                          $comment_IP_country = $Country->ID;
 767                      }
 768                  }
 769  
 770                  $Comment = new Comment();
 771                  $Comment->set( 'item_ID', $post_ID );
 772                  if( !empty( $comment_parent_ID ) )
 773                  {
 774                      $Comment->set( 'in_reply_to_cmt_ID', $comment_parent_ID );
 775                  }
 776                  $Comment->set( 'date', $comment['comment_date'] );
 777                  if( !empty( $comment_author_user_ID ) )
 778                  {
 779                      $Comment->set( 'author_user_ID', $comment_author_user_ID );
 780                  }
 781                  $Comment->set( 'author', evo_substr( $comment['comment_author'], 0, 100 ) );
 782                  $Comment->set( 'author_IP', $comment['comment_author_IP'] );
 783                  $Comment->set( 'author_email', $comment['comment_author_email'] );
 784                  $Comment->set( 'content', $comment['comment_content'] );
 785                  if( empty( $comment['comment_status'] ) )
 786                  {    // If comment status is empty (the export of wordpress doesn't provide this field)
 787                      $Comment->set( 'status', $comment['comment_approved'] == '1' ? 'published' : 'draft' );
 788                  }
 789                  else
 790                  {    // Set status when we have predefined value
 791                      $Comment->set( 'status', $comment['comment_status'] );
 792                  }
 793                  if( !empty( $comment_IP_country ) )
 794                  {    // Country
 795                      $Comment->set( 'IP_ctry_ID', $comment_IP_country );
 796                  }
 797                  $Comment->set( 'rating', $comment['comment_rating'] );
 798                  $Comment->set( 'featured', $comment['comment_featured'] );
 799                  $Comment->set( 'nofollow', $comment['comment_nofollow'] );
 800                  $Comment->set( 'helpful_addvotes', $comment['comment_helpful_addvotes'] );
 801                  $Comment->set( 'helpful_countvotes', $comment['comment_helpful_countvotes'] );
 802                  $Comment->set( 'spam_addvotes', $comment['comment_spam_addvotes'] );
 803                  $Comment->set( 'spam_countvotes', $comment['comment_spam_countvotes'] );
 804                  $Comment->set( 'karma', $comment['comment_karma'] );
 805                  $Comment->set( 'spam_karma', $comment['comment_spam_karma'] );
 806                  $Comment->set( 'allow_msgform', $comment['comment_allow_msgform'] );
 807                  $Comment->set( 'notif_status', empty( $comment['comment_notif_status'] ) ? 'noreq' : $comment['comment_notif_status'] );
 808                  $Comment->dbinsert();
 809  
 810                  $comments_IDs[ $comment['comment_id'] ] = $Comment->ID;
 811                  $comments_count++;
 812              }
 813          }
 814  
 815          echo sprintf( T_('%d records'), $comments_count ).'<br />';
 816      }
 817  
 818      echo '<p>'.T_('Import complete.').'</p>';
 819  
 820      $DB->commit();
 821  }

 824  /**

 825   * Parse WordPress XML file into array

 826   *

 827   * @param string File path

 828   * @return array XML data:

 829   *          authors

 830   *          posts

 831   *          categories

 832   *          tags

 833   *          terms

 834   *          base_url

 835   *          wxr_version

 836   */
 837  function wpxml_parser( $file )
 838  {
 839      $authors = array();
 840      $posts = array();
 841      $categories = array();
 842      $tags = array();
 843      $terms = array();
 844      $files = array();
 845  
 846      $xml = simplexml_load_file( $file );
 847  
 848      // Get WXR version

 849      $wxr_version = $xml->xpath( '/rss/channel/wp:wxr_version' );
 850      $wxr_version = (string) trim( $wxr_version[0] );
 851  
 852      $base_url = $xml->xpath( '/rss/channel/wp:base_site_url' );
 853      $base_url = (string) trim( $base_url[0] );
 854  
 855      // Check language

 856      global $evo_charset, $xml_import_convert_to_latin;
 857      $language = $xml->xpath( '/rss/channel/language' );
 858      $language = (string) trim( $language[0] );
 859      if( $evo_charset != 'utf-8' && ( strpos( $language, 'utf8' ) !== false ) )
 860      { // We should convert the text values from utf8 to latin1
 861          $xml_import_convert_to_latin = true;
 862      }
 863      else
 864      { // Don't convert, it is correct encoding
 865          $xml_import_convert_to_latin = false;
 866      }
 867  
 868      $namespaces = $xml->getDocNamespaces();
 869      if( !isset( $namespaces['wp'] ) )
 870      {
 871          $namespaces['wp'] = 'http://wordpress.org/export/1.1/';
 872      }
 873      if( !isset( $namespaces['evo'] ) )
 874      {
 875          $namespaces['evo'] = 'http://b2evolution.net/export/1.0/';
 876      }
 877      if( !isset( $namespaces['excerpt'] ) )
 878      {
 879          $namespaces['excerpt'] = 'http://wordpress.org/export/1.1/excerpt/';
 880      }
 881  
 882      // Get authors

 883      foreach( $xml->xpath('/rss/channel/wp:author') as $author_arr )
 884      {
 885          $a = $author_arr->children( $namespaces['wp'] );
 886          $ae = $author_arr->children( $namespaces['evo'] );
 887          $login = (string) $a->author_login;
 888          $authors[$login] = array(
 889              'author_id'                   => (int) $a->author_id,
 890              'author_login'                => $login,
 891              'author_email'                => (string) $a->author_email,
 892              'author_display_name'         => wpxml_convert_value( $a->author_display_name ),
 893              'author_first_name'           => wpxml_convert_value( $a->author_first_name ),
 894              'author_last_name'            => wpxml_convert_value( $a->author_last_name ),
 895              'author_pass'                 => (string) $ae->author_pass,
 896              'author_group'                => (string) $ae->author_group,
 897              'author_status'               => (string) $ae->author_status,
 898              'author_nickname'             => wpxml_convert_value( $ae->author_nickname ),
 899              'author_url'                  => (string) $ae->author_url,
 900              'author_level'                => (int) $ae->author_level,
 901              'author_locale'               => (string) $ae->author_locale,
 902              'author_gender'               => (string) $ae->author_gender,
 903              'author_age_min'              => (int) $ae->author_age_min,
 904              'author_age_max'              => (int) $ae->author_age_max,
 905              'author_created_from_country' => (string) $ae->author_created_from_country,
 906              'author_country'              => (string) $ae->author_country,
 907              'author_region'               => (string) $ae->author_region,
 908              'author_subregion'            => (string) $ae->author_subregion,
 909              'author_city'                 => (string) $ae->author_city,
 910              'author_source'               => (string) $ae->author_source,
 911              'author_created_ts'           => (string) $ae->author_created_ts,
 912              'author_lastseen_ts'          => (string) $ae->author_lastseen_ts,
 913              'author_created_fromIPv4'     => (string) $ae->author_created_fromIPv4,
 914              'author_profileupdate_date'   => (string) $ae->author_profileupdate_date,
 915          );
 916      }
 917  
 918      // Get files

 919      foreach( $xml->xpath('/rss/channel/file') as $file_arr )
 920      {
 921          $t = $file_arr->children( $namespaces['evo'] );
 922          $files[] = array(
 923              'file_ID'        => (int) $t->file_ID,
 924              'file_root_type' => (string) $t->file_root_type,
 925              'file_root_ID'   => (int) $t->file_root_ID,
 926              'file_path'      => (string) $t->file_path,
 927              'file_title'     => wpxml_convert_value( $t->file_title ),
 928              'file_alt'       => wpxml_convert_value( $t->file_alt ),
 929              'file_desc'      => wpxml_convert_value( $t->file_desc ),
 930              'zip_path'       => (string) $t->zip_path,
 931          );
 932      }
 933  
 934      // Get categories

 935      foreach( $xml->xpath('/rss/channel/wp:category') as $term_arr )
 936      {
 937          $t = $term_arr->children( $namespaces['wp'] );
 938          $categories[] = array(
 939              'term_id'              => (int) $t->term_id,
 940              'category_nicename'    => wpxml_convert_value( $t->category_nicename ),
 941              'category_parent'      => (string) $t->category_parent,
 942              'cat_name'             => wpxml_convert_value( $t->cat_name ),
 943              'category_description' => wpxml_convert_value( $t->category_description )
 944          );
 945      }
 946  
 947      // Get tags

 948      foreach( $xml->xpath('/rss/channel/wp:tag') as $term_arr )
 949      {
 950          $t = $term_arr->children( $namespaces['wp'] );
 951          $tags[] = array(
 952              'term_id'         => (int) $t->term_id,
 953              'tag_slug'        => (string) $t->tag_slug,
 954              'tag_name'        => wpxml_convert_value( $t->tag_name ),
 955              'tag_description' => wpxml_convert_value( $t->tag_description )
 956          );
 957      }

 959      // Get terms

 960      foreach( $xml->xpath('/rss/channel/wp:term') as $term_arr )
 961      {
 962          $t = $term_arr->children( $namespaces['wp'] );
 963          $terms[] = array(
 964              'term_id'          => (int) $t->term_id,
 965              'term_taxonomy'    => (string) $t->term_taxonomy,
 966              'slug'             => (string) $t->term_slug,
 967              'term_parent'      => (string) $t->term_parent,
 968              'term_name'        => wpxml_convert_value( $t->term_name ),
 969              'term_description' => wpxml_convert_value( $t->term_description )
 970          );
 971      }
 972  
 973      // Get posts

 974      foreach( $xml->channel->item as $item )
 975      {
 976          $post = array(
 977              'post_title' => wpxml_convert_value( $item->title ),
 978              'guid'       => (string) $item->guid,
 979          );
 980  
 981          $dc = $item->children( 'http://purl.org/dc/elements/1.1/' );
 982          $post['post_author'] = (string) $dc->creator;
 983  
 984          $content = $item->children( 'http://purl.org/rss/1.0/modules/content/' );
 985          $excerpt = $item->children( $namespaces['excerpt'] );
 986          $post['post_content'] = wpxml_convert_value( $content->encoded );
 987          $post['post_excerpt'] = wpxml_convert_value( $excerpt->encoded );
 988  
 989          $wp = $item->children( $namespaces['wp'] );
 990          $evo = $item->children( $namespaces['evo'] );
 991  
 992          $post['post_id']        = (int) $wp->post_id;
 993          $post['post_date']      = (string) $wp->post_date;
 994          $post['post_date_gmt']  = (string) $wp->post_date_gmt;
 995          $post['comment_status'] = (string) $wp->comment_status;
 996          $post['ping_status']    = (string) $wp->ping_status;
 997          $post['post_name']      = (string) $wp->post_name;
 998          $post['status']         = (string) $wp->status;
 999          $post['post_parent']    = (int) $wp->post_parent;
1000          $post['menu_order']     = (int) $wp->menu_order;
1001          $post['post_type']      = (string) $wp->post_type;
1002          $post['post_password']  = (string) $wp->post_password;
1003          $post['is_sticky']      = (int) $wp->is_sticky;
1004          $post['post_date_mode']     = (string) $evo->post_date_mode;
1005          $post['post_lastedit_user'] = (string) $evo->post_lastedit_user;
1006          $post['post_assigned_user'] = (string) $evo->post_assigned_user;
1007          $post['post_datedeadline']  = (string) $evo->post_datedeadline;
1008          $post['post_datecreated']   = (string) $evo->post_datecreated;
1009          $post['post_datemodified']  = (string) $evo->post_datemodified;
1010          $post['post_locale']        = (string) $evo->post_locale;
1011          $post['post_excerpt_autogenerated'] = (int) $evo->post_excerpt_autogenerated;
1012          $post['post_urltitle']      = (string) $evo->post_urltitle;
1013          $post['post_titletag']      = (string) $evo->post_titletag;
1014          $post['post_url']           = (string) $evo->post_url;
1015          $post['post_notifications_status'] = (string) $evo->post_notifications_status;
1016          $post['post_renderers']     = (string) $evo->post_renderers;
