 104  	function load_from_Request()
 105      {
 106          // Address

 107          global $emadr_address;
 108          param_string_not_empty( 'emadr_address', T_('Please enter email address.') );
 109          $emadr_address = evo_strtolower( get_param( 'emadr_address' ) );
 110          param_check_email( 'emadr_address', true );
 111          if( $existing_emadr_ID = $this->dbexists( 'emadr_address', get_param( 'emadr_address' ) ) )
 112          {    // Check if a email address already exists with the same address
 113              global $admin_url;
 114              param_error( 'emadr_address', sprintf( T_('This email address already exists. Do you want to <a %s>edit the existing email address</a>?'),
 115                  'href="'.$admin_url.'?ctrl=email&amp;tab=blocked&amp;emadr_ID='.$existing_emadr_ID.'"' ) );
 116          }
 117          $this->set_from_Request( 'address' );
 118  
 119          // Status

 120          $emadr_status = param( 'emadr_status', 'string', true );
 121          if( !empty( $emadr_status ) )
 122          {
 123              $this->set( 'status', $emadr_status );
 124          }
 125  
 126          // Sent count

 127          param( 'emadr_sent_count', 'integer', '' );
 128          param_check_number( 'emadr_sent_count', T_('The count must be a number.'), true );
 129          $this->set_from_Request( 'sent_count', 'emadr_sent_count', true );

 159      /**

 160       * Increase a counter field

 161       *

 162       * @param string Counter name ( 'prmerror', 'tmperror', 'spamerror', 'othererror' )

 163       */
 164  	function increase_counter( $counter_name )
 165      {
 166          $counter_name .= '_count';
 167          if( isset( $this->$counter_name ) )
 168          {
 169              $this->set( $counter_name, $this->get( $counter_name ) + 1 );
 170              if( $counter_name != 'sent_count' )
 171              {    // Update last error date when we increase an error counter
 172                  global $servertimenow;
 173                  $this->set( 'last_error_ts', date( 'Y-m-d H:i:s', $servertimenow ) );
 174                  $this->set( 'sent_last_returnerror', '0' );
 175              }
 176          }
 177      }

 181       * Set status

 182       *

 183       * @param string New status value

 184       */
 185  	function set_status( $new_status )
 186      {
 187          if( $this->ID == 0 )
 188          {    // New record is creating, we can assign any status without level restrictions
 189              $this->set( 'status', $new_status );
 190          }
 191          else
 192          {    // The record is updating, we should update only status which has hight level then old status
 193  
 194              // Get statuses which we can update depending on levels of the statuses

 195              $available_statuses = emadr_get_statuses_less_level( $new_status );
 196  
 197              if( in_array( $this->get( 'status' ), $available_statuses ) )
 198              {    // Check if we can update this status
 199                  $this->set( 'status', $new_status );
 200              }
 201          }
 202      }
