 489  	function get_default_group_permissions( $grp_ID )

 511                  $permslugs = 'none';
 512                  $permtemplates = 'denied';
 513                  $permemails = 'view';
 514                  $def_notification = 'short';
 515                  break;
 516

 518              case 4:     // Normal Users (group ID 4) have permission by default:

 520                  $permusers = 'none';
 521                  $permoptions = 'none';
 522                  $permspam = 'view';
 523                  $permslugs = 'none';
 524                  $permtemplates = 'denied';
 525                  $permemails = 'none';
 526                  $def_notification = 'short';
 527                  break;
 528  
 529              // case 5:        // Misbehaving/Suspect users (group ID 5) have permission by default:

 532                  // Other groups have no permission by default

 537                  $permslugs = 'none';
