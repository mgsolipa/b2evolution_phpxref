 154  	function Form( $form_action = NULL, $form_name = '', $form_method = 'post', $layout = NULL, $enctype = '' )
 155      {
 156          global $AdminUI, $pagenow, $Skin;
 157  
 158          $this->form_name = $form_name;
 159          $this->form_action = (is_null($form_action) ? $pagenow : $form_action );
 160          $this->form_method = $form_method;
 161          $this->enctype = $enctype;
 162  
 163          if( is_object($AdminUI) )
 164          {
 165              if( empty( $layout ) || $layout == 'split' || $layout == 'none' || $layout == 'fieldset' )
 166              { // Get default skin setting:
 167                  $template = $AdminUI->get_template( 'Form' );
 168                  $layout = $template['layout'];
 169              }
 170              else
 171              {
 172                  $template = $AdminUI->get_template( $layout.'_form' );
 173                  $layout = $template['layout'];
 174              }
 175          }
 176          else
 177          {    // This happens for comment forms & login screen for example...
 178              $template_is_empty = true;
 179              if( is_object( $Skin ) )
 180              { // Get skin setting:
 181                  if( empty( $layout ) || $layout == 'split' || $layout == 'none' || $layout == 'fieldset' )
 182                  { // Get default skin setting:
 183                      $template = $Skin->get_template( 'Form' );
 184                  }
 185                  else
 186                  {
 187                      $template = $Skin->get_template( $layout.'_form' );
 188                  }
 189                  $template_is_empty = empty( $template );
 190                  if( !$template_is_empty )
 191                  {
 192                      $layout = $template['layout'];
 193                  }
 194              }
 195  
 196              if( $template_is_empty )
 197              {
 198                  $template = array(
 199                      'layout' => 'fieldset',
 200                      'formstart' => '<div>',// required before (no_)title_fmt for validation
 201                      'title_fmt' => '<span style="float:right">$global_icons$</span><h2>$title$</h2>'."\n",
 202                      'no_title_fmt' => '<span style="float:right">$global_icons$</span>'."\n",
 203                      'fieldset_begin' => '<fieldset $fieldset_attribs$>'."\n"

 210                      'labelempty' => '<div class="label"></div>', // so that IE6 aligns DIV.input correcctly
 211                      'inputstart' => '<div class="input">',

 213                      'inputend' => "</div>\n",

 216                      'buttonsend' => "</div></fieldset>\n\n",
 217                      'customstart' => '<div class="custom_content">',
 218                      'customend' => "</div>\n",
 219                      'note_format' => ' <span class="notes">%s</span>',
 220                      'formend' => '</div>',
 221                  );
 222                  $layout = 'fieldset';
 223              }
 224          }
 225  
 226          $this->saved_layouts = array($layout);
 227          $this->saved_templates = array($template);
 228          $this->switch_layout( NULL );    // "restore" saved layout.
 229  
 230  
 231          // Add any GET params from $form_action as hidden inputs (GET forms only)
 232          // Otherwise those params would be overwritten by the form submit and lost.
 233          if( strpos($this->form_action, '?') !== false && $this->form_method == 'get' )
 234          {
 235              $pos_args = strpos($this->form_action, '?');
 236              $query_str = substr($this->form_action, $pos_args+1);
 237              // Split args by "&" (but leave "&amp;" alone).
 238              $query_args = preg_split('~&(?!amp;)~', $query_str, -1, PREG_SPLIT_NO_EMPTY);

 242                  // Remember all pairs, and add them in end_form (so that used fieldnames can be skipped):
 243                  $this->possible_hiddens_from_action[] = array($field_name, $field_value);
 244              }
 245              // Remove params.
 246              $this->form_action = substr($form_action, 0, $pos_args);
 247          }

 475      /**
 476       * Start an input field.
 477       *
 478       * A field is a fielset containing a label div and an input div.
 479       *
 480       * @uses $_common_params
 481       * @param string The name of the field
 482       * @param string The field label
 483       * @param boolean Reset {@link $_common_params}? This should be used if you build a field by yourself.
 484       * @return The generated HTML
 485       */
 486  	function begin_field( $field_name = NULL, $field_label = NULL, $reset_common_params = false )
 487      {
 488          if( $reset_common_params )
 489          {
 490              $this->_common_params = array();
 491          }
 492  
 493          $field_classes = array();
 494          if( !empty( $this->_common_params['wide'] ) && $this->_common_params['wide'] )
 495          { // Wide field
 496              $field_classes[] = 'field_wide';
 497          }
 498          if( !empty( $this->_common_params['inline'] ) && $this->_common_params['inline'] )
 499          { // Inline field
 500              $field_classes[] = 'field_inline';
 501          }
 502  
 503          // Remember these, to make them available to get_label()
 504          if( isset($field_name) )
 505          {
 506              $this->_common_params['name'] = $field_name;
 507          }
 508          if( isset($field_label) )
 509          {
 510              $this->_common_params['label'] = $field_label;
 511          }
 512  
 513          // Start the new form field and inject an automatic DOM id
 514          // This is useful to show/hide the whole field by JS.
 515          if( !empty( $this->_common_params['id'] ) )
 516          {
 517              $ffield_id = ' id="ffield_'.$this->_common_params['id'].'" ';
 518          }
 519          else
 520          { // No ID in case there's no id/name given for a field.
 521              $ffield_id = '';
 522          }
 523          // quick and dirty "required_field" addition, needs tidying once tested
 524          if( !empty( $this->_common_params['required'] ) )
 525          { // required field
 526              $field_classes[] = 'field_required';
 527          }
 528  
 529          $r = $this->fieldstart;
 530  
 531          if( count( $field_classes ) > 0 )
 532          {
 533              if( strpos( $r, 'class="' ) === false )
 534              { // Add class attribute
 535                  $ffield_id .= ' class="'.implode( ' ', $field_classes ).'"';
 536              }
 537              else
 538              { // Append classes to attribute
 539                  $r = str_replace( ' class="', ' class="'.implode( ' ', $field_classes ).' ', $r );
 540              }
 541          }
 542  
 543          $r = str_replace( '$ID$', $ffield_id, $r );
 544  
 545          if( isset($this->_common_params['field_prefix']) )
 546          {
 547              $r .= $this->_common_params['field_prefix'];
 548          }
 549  
 550          $r .= $this->get_label();
 551  
 552          $r .= $this->inputstart;
 553  
 554          return $r;
 555      }

 561       * A field is a fielset containing a label div and an input div.

 563       * @return The generated HTML

 565  	function end_field()
 566      {
 567  
 568          $r = '';
 569  
 570          if( !empty($this->_common_params['note']) )
 571          { // We have a note
 572              $r .= sprintf( $this->_common_params['note_format'], $this->_common_params['note'] );
 573          }
 574  
 575          if( isset($this->_common_params['field_suffix']) )
 576          {
 577              $r .= $this->_common_params['field_suffix'];
 578          }

 896      /**
 897       * Builds a date input field.
 898       *
 899       * @param string the name of the input field
 900       * @param string initial value (ISO datetime (YYYY-MM-DD HH:MM:SS)
 901       *               or erroneous if the field is in error state)
 902       * @param string label displayed in front of the field
 903       * @param array Optional params. Additionally to {@link $_common_params} you can use:
 904       *              - date_format: Format of the date (string, PHP format, default taken from {@link locale_datefmt()})
 905       *              - add_date_format_note: If true, date format note gets prepended to the field's note
 906       * @return mixed true (if output) or the generated HTML if not outputting
 907       */
 908  	function date_input( $field_name, $field_value, $field_label, $field_params = array() )
 909      {
 910          global $month, $weekday_letter;
 911  
 912          if( empty($field_params['date_format']) )
 913          {    // Use locale date format:
 914              $date_format = locale_datefmt();
 915          }
 916          else
 917          {
 918              $date_format = $field_params['date_format'];
 919          }
 920          // Don't keep that attrib in the list:
 921          unset( $field_params['date_format'] );
 922  
 923          // Convert PHP date format to JS library date format:
 924          // NOTE: when editing/extending this here, you probably also have to adjust param_check_date()!
 925          $js_date_format = preg_replace_callback( '~(\\\)?(\w)~', create_function( '$m', '
 926              if( $m[1] == "\\\" ) return "\\\".$m[0]; // leave escaped
 927              switch( $m[2] )
 928              {
 929                  case "d": return "dd"; // day, 01-31
 930                  case "j": return "d"; // day, 1-31
 931                  case "l": return "EE"; // weekday (name)
 932                  case "D": return "E"; // weekday (abbr)
 933                  case "e": return ""; // weekday letter, not supported
 934  
 935                  case "m": return "MM"; // month, 01-12
 936                  case "n": return "M"; // month, 1-12
 937                  case "F": return "MMM"; // full month name; "name or abbr" in date.js
 938                  case "M": return "NNN"; // month name abbr
 939  
 940                  case "y": return "yy"; // year, 00-99
 941                  case "Y": return "yyyy"; // year, XXXX
 942                  default:
 943                      return $m[0];
 944              }' ), $date_format );
 945  
 946          $field_params['type'] = 'text';
 947  
 948          if( param_has_error( $field_name )
 949              &&  ! preg_match('~^\d\d\d\d-\d\d-\d\d(?: \d\d:\d\d:\d\d)?$~', $field_value) )
 950          { // There is an error message for this field:
 951  
 952              // We do not try to format the date, we keep the erroneous date (if it is not obviously valid).
 953              // We could have used param_error() ourself (e.g. "date outside of range"), and the erroneous
 954              // field should have the correct format.
 955  
 956              //echo 'error on '.$field_name.' keep erroneous entry intact ';
 957  
 958              // Keep original value, but strip off the time part (if any).
 959              $field_params['value'] = preg_replace( '~ \d\d:\d\d:\d\d$~', '', $field_value );
 960          }
 961          else
 962          { // Make the date value clean for display:
 963  
 964              // The date value may be compact, in this case we have to decompact it
 965              if( preg_match( '/^[0-9]+$/', $field_value ) )
 966              {    // The date is compact, so we decompact it
 967                  $field_value = decompact_date( $field_value );
 968              }
 969  
 970              // Get DATE part of datetime and format it to locale format:
 971              $field_params['value'] = mysql2date( $date_format, $field_value );
 972              if( ! $field_params['value'] )
 973              { // Conversion failed (e.g. for dates before ~1902 / PHP < 5.1 (Windows) / 32bit / without DateTime support), use the original value (better than 1970-01-01 anyway!).
 974                  $field_params['value'] = $field_value;
 975              }
 976          }
 977  
 978  
 979          if( !empty($field_params['add_date_format_note']) )
 980          { // Prepend $date_format to note
 981              $field_params['note'] = empty($field_params['note'])
 982                  ? '('.$date_format.')'
 983                  : '('.$date_format.') '.$field_params['note'];
 984          }
 985          unset( $field_params['add_date_format_note'] );
 986  
 987  
 988          if( !isset($field_params['size']) )
 989          { // Get size out of $date_format if not explicitly set
 990              $field_params['size'] = strlen( $js_date_format );
 991          }
 992  
 993          /*
 994          dh> do not use maxlength by default. Makes no sense IMHO and fails with dateformats like "j \d\e F, Y"
 995          if( !isset($field_params['maxlength']) )
 996          {
 997              $field_params['maxlength'] = $field_params['size'];
 998          }
 999          */
1000          /*
1001           Afwas > In the existing locales only d m y and Y are used. Currently the jQuery Datepicker
1002           can't handle other dateformats. I will see to some basic check or enable all.
1003           @TODO ^^ fp> It might make sense to have 2 date formats for locales: 1 for display and 1 for inputs. Input formats could be forced to used numeric data only.
1004           */
1005          // Give it a class, so it can be selected for CSS in IE6
1006          $field_params['class'] = ( empty( $field_params['class'] ) ? '' : $field_params['class'].' ' ).'form_date_input form-control';
1007  
1008          $this->handle_common_params( $field_params, $field_name, $field_label );
1009  
1010          $r = $this->begin_field() . $this->get_input_element($field_params, false);
1011          $r .= $this->end_field();
1012  
1013          return $this->display_or_return( $r );
1014      }

1097      /**
1098       * Builds a time select input field
1099       *
1100       * @param string field name
1101       * @param string initial value (ISO datetime or time only)
1102       * @param string precison xmn or xsec (x:integer) for the options minutes or secondes
1103       * @param string field label to be display before the field
1104       * @param string note to be displayed after the field
1105       * @param string CSS class for select
1106       * @param string Javascript to add for onchange event (trailing ";").
1107       */
1108  	function time_select( $field_name, $field_value = NULL, $precision = '5mn', $field_label, $field_note = NULL, $field_class = NULL, $field_onchange = NULL )
1109      {
1110          preg_match( '#([0-9]+)(mn|s)#', $precision, $matches );
1111  
1112          if( !isset( $matches[1] ) && !isset( $matches[2] ) )
1113          {    // precison has a bad format
1114              return;
1115          }
1116  
1117          $field_params = array(
1118          'note' => $field_note,
1119          'class' => $field_class,
1120          'onchange' => $field_onchange );
1121  
1122          /***  instantiate the precison for the minutes and secondes select options  ****/
1123  
1124          if( $matches[2] == 'mn' )
1125          {
1126              $precision_mn = $matches[1];
1127              $precision_s = 0;
1128              // convert the precision in sec
1129              $precision *= 60;
1130          }
1131          else
1132          {
1133              $precision_mn = 1;
1134              $precision_s = $matches[1];
1135          }
1136  
1137          // Check if field value is only a time
1138          if( strlen( $field_value ) <= 8 )
1139          {    // Add date part:
1140              $field_value = '2000-01-01 '.$field_value;
1141          }
1142  
1143          /***  set round time with the precision  ***/
1144          // Get nb sec since unix...
1145          $nbsec = mysql2timestamp( $field_value );
1146          $modulo =  $nbsec % $precision;
1147  
1148          if( $modulo < ( $precision / 2 ) )
1149          { // The round time is before
1150              $nbsec -= $modulo;
1151          }
1152          else
1153          { // The round time is after
1154              $nbsec += $precision - $modulo;
