 444  	function query( $create_default_cols_if_needed = true, $append_limit = true, $append_order_by = true,
 445                                          $query_title = 'Results::Query()' )
 446      {
 447          global $DB, $Debuglog;
 448          if( !is_null( $this->rows ) )
 449          { // Query has already executed:
 450              return;
 451          }
 452  
 453          if( empty( $this->sql ) )
 454          { // the sql query is empty so we can't process it
 455              return;
 456          }
 457  
 458          if( isset( $this->total_rows ) && intval( $this->total_rows ) === 0 )
 459          { // The number of total rows was counted and the sql query has no results
 460              return;
 461          }
 462  
 463          // Make sure we have colum definitions:
 464          if( is_null( $this->cols ) && $create_default_cols_if_needed )
 465          { // Let's create default column definitions:
 466              $this->cols = array();
 467  
 468              if( !preg_match( '#^(SELECT.*?(\([^)]*?FROM[^)]*\).*)*)FROM#six', $this->sql, $matches ) )
 469              {
 470                  debug_die( 'Results->query() : No SELECT clause!' );
 471              }
 472              // Split requested columns by commata
 473              foreach( preg_split( '#\s*,\s*#', $matches[1] ) as $l_select )
 474              {
 475                  if( is_numeric( $l_select ) )
 476                  { // just a single value (would produce parse error as '$x$')
 477                      $this->cols[] = array( 'td' => $l_select );
 478                  }
 479                  elseif( preg_match( '#^(\w+)$#i', $l_select, $match ) )
 480                  { // regular column
 481                      $this->cols[] = array( 'td' => '$'.$match[1].'$' );
 482                  }
 483                  elseif( preg_match( '#^(.*?) AS (\w+)#i', $l_select, $match ) )
 484                  { // aliased column
 485                      $this->cols[] = array( 'td' => '$'.$match[2].'$' );
 486                  }
 487              }
 488  
 489              if( !isset($this->cols[0]) )
 490              {
 491                  debug_die( 'No columns selected!' );
 492              }
 493          }
 494  
 495  
 496          // Make a copy of the SQL, that we may change and that gets executed:
 497          $sql = $this->sql;
 498  
 499          // Append ORDER clause if necessary:
 500          if( $append_order_by && ($orders = $this->get_order_field_list()) )
 501          {    // We have orders to append
 502  
 503              if( strpos( $sql, 'ORDER BY') === false )
 504              { // there is no ORDER BY clause in the original SQL query
 505                  $sql .= ' ORDER BY '.$orders.' ';
 506              }
 507              else
 508              {    // try to insert the chosen order at an existing '*' point
 509                  $inserted_sql = preg_replace( '# \s ORDER \s+ BY (.+) \* #xi', ' ORDER BY $1 '.$orders, $sql );
 510  
 511                  if( $inserted_sql != $sql )
 512                  {    // Insertion ok:
 513                      $sql = $inserted_sql;
 514                  }
 515                  else
 516                  {    // No insert point found:
 517                      // the chosen order must be appended to an existing ORDER BY clause
 518                      $sql .= ', '.$orders;
 519                  }
 520              }
 521          }
 522          else
 523          {    // Make sure there is no * in order clause:
 524              $sql = preg_replace( '# \s ORDER \s+ BY (.+) \* #xi', ' ORDER BY $1 ', $sql );
 525          }
 526  
 527          $add_limit = $append_limit && ! empty( $this->limit );
 528  
 529          if( $add_limit && ! $this->order_callbacks )
 530          {    // No callbacks to be called, so we can limit the line range to the requested page:
 531              $Debuglog->add( 'LIMIT requested and no callbacks - adding LIMIT to query.', 'results' );

 534  
 535          // Execute query and store results

 542          }

 547              $Debuglog->add( 'Sorting with callbacks.', 'results' );

 549              {
 550                  #echo 'order_callback: '; var_dump($order_callback);
 551  
 552                  $this->order_callback_wrapper_data = $order_callback; // to pass ASC/DESC param and callback itself through the wrapper to the callback
 553  
 554                  if( empty($order_callback['use_rows']) )
 555                  { // default: instantiate objects for the callback:
 556                      usort( $this->rows, array( &$this, 'order_callback_wrapper_objects' ) );
 557                  }
 558                  else
 559                  {
 560                      usort( $this->rows, array( &$this, 'order_callback_wrapper_rows' ) );
 561                  }
 562              }
 563  
 564              if ( $add_limit )
 565              {
 566                  $Debuglog->add( 'Callback sorting: LIMIT needed, extracting slice from array', 'results' );
 567                  $this->rows = array_slice( $this->rows, max( 0, ( $this->page - 1 ) * $this->limit ), $this->limit );
 568                  $this->result_num_rows = count( $this->rows );
 569              }
 570          }
 571  
 572          // Group by object property:
 573          if( ! empty($this->group_by_obj_prop) )
 574          {
 575              if( ! is_array($this->group_by_obj_prop) )
 576              {
 577                  $this->group_by_obj_prop = array($this->group_by_obj_prop);
 578              }
 579  
 580              $this->mergesort( $this->rows, array( &$this, 'callback_group_by_obj_prop' ) );
 581          }
 582

 584      }

 587      /**
 588       * Merge sort. This is required to not re-order items when sorting for e.g. grouping at the end.
 589       *
 590       * @see http://de2.php.net/manual/en/function.usort.php#38827
 591       *
 592       * @param array List of items to sort
 593       * @param callback Sort function/method
 594       */
 595  	function mergesort(&$array, $cmp_function)
 596      {
 597          // Arrays of size < 2 require no action.
 598          if (count($array) < 2) return;
 599          // Split the array in half
 600          $halfway = count($array) / 2;
 601          $array1 = array_slice($array, 0, $halfway);
 602          $array2 = array_slice($array, $halfway);
 603          // Recurse to sort the two halves
 604          $this->mergesort($array1, $cmp_function);
 605          $this->mergesort($array2, $cmp_function);
 606          // If all of $array1 is <= all of $array2, just append them.
 607          if (call_user_func($cmp_function, end($array1), $array2[0]) < 1) {
 608                  $array = array_merge($array1, $array2);
 609                  return;
 610          }
 611          // Merge the two sorted arrays into a single sorted array
 612          $array = array();
 613          $ptr1 = $ptr2 = 0;
 614          while ($ptr1 < count($array1) && $ptr2 < count($array2)) {
 615                  if (call_user_func($cmp_function, $array1[$ptr1], $array2[$ptr2]) < 1) {
 616                          $array[] = $array1[$ptr1++];
 617                  }
 618                  else {
 619                          $array[] = $array2[$ptr2++];
 620                  }
 621          }
 622          // Merge the remainder
 623          while ($ptr1 < count($array1)) $array[] = $array1[$ptr1++];
 624          while ($ptr2 < count($array2)) $array[] = $array2[$ptr2++];
 625          return;
 626       }

 638      {
 639          $obj_prop = $this->group_by_obj_prop[$depth];
 640  
 641          $a = & $this->Cache->instantiate($row_a);
 642          $a_value = $a->$obj_prop;
 643          $b = & $this->Cache->instantiate($row_b);

 645

 649              {

1581      /**
1582       * Handle variable subtitutions for column contents.
1583       *
1584       * This is one of the key functions to look at when you want to use the Results class.
1585       * - $var$
1586       * - £var£ - replaced by its utf-8 hex character (c2 a3)
1587       * - ²var² - replaced by its utf-8 hex character (c2 b2)
1588       * - #var#
1589       * - {row}
1590       * - %func()%
1591       * - ~func()~
1592       * - ¤func()¤ - @deprecated by ~func()~ - replaced by its utf-8 hex character (c2 a4)
1593       */
1594  	function parse_col_content( $content )
1595      {
1596          // Make variable substitution for STRINGS:
1597          $content = preg_replace( '#\$ (\w+) \$#ix', "'.format_to_output(\$row->$1).'", $content );
1598          // Make variable substitution for URL STRINGS:
1599          $content = preg_replace( '#\x{c2}\x{a3} (\w+) \x{c2}\x{a3}#ix', "'.format_to_output(\$row->$1, 'urlencoded').'", $content );
1600          // Make variable substitution for escaped strings:
1601          $content = preg_replace( '#\x{c2}\x{b2} (\w+) \x{c2}\x{b2}#ix', "'.evo_htmlentities(\$row->$1).'", $content );
1602          // Make variable substitution for RAWS:
1603          $content = preg_replace( '!\# (\w+) \#!ix', "\$row->$1", $content );
1604          // Make variable substitution for full ROW:
1605          $content = str_replace( '{row}', '$row', $content );
1606          // Make callback function substitution:
1607          $content = preg_replace( '#% (.+?) %#ix', "'.$1.'", $content );
1608          // Make variable substitution for intanciated Object:
1609          $content = str_replace( '{Obj}', "\$this->current_Obj", $content );
1610          // Make callback for Object method substitution:
1611          $content = preg_replace( '#@ (.+?) @#ix', "'.\$this->current_Obj->$1.'", $content );
1612          // Sometimes we need embedded function call, so we provide a second sign:
1613          $content = preg_replace( '#~ (.+?) ~#ix', "'.$1.'", $content );
1614  
1615          // @deprecated by ~func()~. Left here for backward compatibility only, to be removed in future versions.
1616          $content = preg_replace( '#\x{c2}\x{a4} (.+?) \x{c2}\x{a4}#ix', "'.$1.'", $content );
1617  
1618          // Make callback function move_icons for orderable lists // dh> what does it do?
1619          $content = str_replace( '{move}', "'.\$this->move_icons().'", $content );
1620  
1621          $content = str_replace( '{CUR_IDX}', $this->current_idx, $content );
1622          $content = str_replace( '{TOTAL_ROWS}', $this->total_rows, $content );
1623  
1624          return $content;
1625      }
