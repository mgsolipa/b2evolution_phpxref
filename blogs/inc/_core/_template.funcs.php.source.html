  98      // This cannot be enabled by default and requires admin switches.

 150              {
 151                  $redirect_to = $Blog->get('url');
 152              }
 153              else
 154              {
 155                  $redirect_to = $baseurl;
 156              }
 157          }
 158          elseif( $redirect_to[0] == '/' )
 159          { // relative URL, prepend current host:
 160              global $ReqHost;
 161              $redirect_to = $ReqHost.$redirect_to;
 162          }

 190          $redirect_url_parts = parse_url( $redirect_to );
 191          $host = empty( $redirect_url_parts['host'] ) ? null : $redirect_url_parts['host'];
 192          $baseurl_parts = parse_url( $baseurl );
 193          $base_host = empty( $baseurl_parts['host'] ) ? null : $baseurl_parts['host'];

 195          { // The main domain of the redirect_to url and the base url doesn't match

 206      {

 212      {    // Session is required here
 213  
 214          // Transfer of Debuglog to next page:

 233           // echo 'Passing Messages to next page';
 234          }
 235  
 236          $Session->dbsave(); // If we don't save now, we run the risk that the redirect goes faster than the PHP script shutdown.
 237      }
 238  
 239      // see http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html
 240      switch( $http_response_code )
 241      {
 242          case 301:
 243              // This should be a permanent move redirect!
 244              header_http_response( '301 Moved Permanently' );
 245              break;
 246  
 247          case 303:
 248              // This should be a "follow up" redirect
 249              // Note: Also see http://de3.php.net/manual/en/function.header.php#50588 and the other comments around
 250              header_http_response( '303 See Other' );
 251              break;
 252  
 253          case 302:
 254          default:
 255              header_http_response( '302 Found' );
 256      }
 257  
 258      // debug_die($redirect_to);
 259      if( headers_sent($filename, $line) )
 260      {
 261          debug_die( sprintf('Headers have already been sent in %s on line %d.', basename($filename), $line)
 262                          .'<br />Cannot <a href="'.evo_htmlspecialchars($redirect_to).'">redirect</a>.' );
 263      }
 264      header( 'Location: '.$redirect_to, true, $http_response_code ); // explictly setting the status is required for (fast)cgi
 265      exit(0);
 266  }

 270  /**
 271   * Sends HTTP headers to avoid caching of the page at the browser level
 272   * (at least without revalidating with the server to make sure whether the content has changed or not).
 273   *
 274   * @see http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html
 275   */
 276  function header_nocache( $timestamp = NULL )
 277  {
 278      global $servertimenow;
 279      if( empty($timestamp) )
 280      {
 281          $timestamp = $servertimenow;
 282      }
 283  
 284      header('Expires: '.gmdate('r',$timestamp));
 285      header('Last-Modified: '.gmdate('r',$timestamp));
 286      header('Cache-Control: no-cache, must-revalidate');
 287      header('Pragma: no-cache');
 288  }

 305  /**
 306   * Generate an etag to identify the version of the current page.
 307   * We use this primarily to make a difference between the same page that has been generated for anonymous users
 308   * and a version that has been generated for a specific user.
 309   *
 310   * A common problem without this would be that when users log out, the page cache would tell them "304 Not Modified"
 311   * based on the date of the cache and then the browser would show a locally cached version of the page that includes
 312   * the evobar.
 313   *
 314   * When a specific user logs out, the browser will send back the Etag of the logged in version it got and we will
 315   * be able to detect that this is not a "304 Not Modified" case -> we will send back the anonymou version of the page.
 316   */
 317  function gen_current_page_etag()
 318  {
 319      global $current_User, $Messages;
 320  
 321      if( isset($current_User) )
 322      {
 323          $etag = 'user:'.$current_User->ID;
 324      }
 325      else
 326      {
 327          $etag = 'user:anon';
 328      }
 329  
 330      if( $Messages->count() )
 331      {    // This case has never been observed yet, but let's forward protect us against client side cached messages
 332          $etag .= '-msg:'.md5($Messages->get_string('',''));
 333      }
 334  
 335      return '"'.$etag.'"';
 336  }

 839  /**
 840   * Get library url of JS or CSS file by file name or alias
 841   *
 842   * @param string File or Alias name
 843   * @param boolean|string Is the file's path relative to the base path/url?
 844   * @param string File type: 'js' or 'css'
 845   * @return string URL
 846   */
 847  function get_require_url( $lib_file, $relative_to = 'rsc_url', $type = 'js' )
 848  {
 849      global $library_local_urls, $library_cdn_urls, $debug, $rsc_url;
 850  
 851      if( ! empty( $library_cdn_urls[ $lib_file ] ) )
 852      { // Rewrite local urls with CDN urls if they are defined in _advanced.php
 853          $library_local_urls[ $lib_file ] = $library_cdn_urls[ $lib_file ];
 854      }
 855  
 856      if( ! empty( $library_local_urls[ $lib_file ] ) )
 857      { // We are requesting an alias
 858          if( $debug && ! empty( $library_local_urls[ $lib_file ][1] ) )
 859          { // Load JS file for debug mode (optional)
 860              $lib_file = $library_local_urls[ $lib_file ][1];
 861          }
 862          else
 863          { // Load JS file for production mode
 864              $lib_file = $library_local_urls[ $lib_file ][0];
 865          }
 866  
 867          if( $relative_to === 'relative' || $relative_to === true )
 868          { // Aliases cannot be relative, make it relative to $rsc_url
 869              $relative_to = 'rsc_url';
 870          }
 871      }
 872  
 873      if( $relative_to === 'relative' || $relative_to === true )
 874      { // Make the file relative to current page <base>:
 875          $lib_url = $lib_file;
 876      }
 877      elseif( preg_match('~^(https?:)?//~', $lib_file ) )
 878      { // It's an absolute url, keep it as is:
 879          $lib_url = $lib_file;
 880      }
 881      elseif( $relative_to === 'rsc_url' || $relative_to === false )
 882      { // Get the file from $rsc_url:
 883          $lib_url = $rsc_url.$type.'/'.$lib_file;
 884      }
 885      elseif( $relative_to === 'blog' )
 886      { // Get the file from $rsc_url:
 887          global $Blog;
 888          if( !empty( $Blog ) )
 889          {
 890              $lib_url = $Blog->get_local_rsc_url().$type.'/'.$lib_file;
 891          }
 892          else
 893          {
 894              $lib_url = $rsc_url.$type.'/'.$lib_file;
 895          }
 896      }
 897      else
 898      {
 899          debug_die( 'Unknown $relative to argument in get_require_url()' );
 900      }
 901  
 902      return $lib_url;
 903  }

 906   * Memorize that a specific javascript file will be required by the current page.
 907   * All requested files will be included in the page head only once (when headlines is called)
 908   *
 909   * Accepts absolute urls, filenames relative to the rsc/js directory and certain aliases, like 'jquery' and 'jquery_debug'
 910   * If 'jquery' is used and $debug is set to true, the 'jquery_debug' is automatically swapped in.

 912   *
 913   * @todo dh>merge with require_css()
 914   * @param string alias, url or filename (relative to rsc/js) for javascript file

 920      static $required_js;
 921  
 922      if( is_admin_page() && ( $relative_to == 'blog' ) )

 924          $relative_to = 'rsc_url';

 928      { // Dependency : ensure jQuery is loaded
 929          require_js( '#jquery#', $relative_to );

 936      $js_url = url_add_param( $js_url, 'v='.$app_version_long );
 937  
 938      // Add to headlines, if not done already:
 939      if( empty( $required_js ) || ! in_array( strtolower( $js_url ), $required_js ) )
 940      {
 941          $required_js[] = strtolower( $js_url );

 943      }

1955  /**
1956   * Check if the array given as the first param contains recursion
1957   *
1958   * @param array what to check
1959   * @param array contains object which were already seen
1960   * @return boolean true if contains recursion false otherwise
1961   */
1962  function is_recursive( /*array*/ & $array, /*array*/ & $alreadySeen = array() )
1963  {
1964      static $uniqueObject;
1965      if( !$uniqueObject )
1966      {
1967          $uniqueObject = new stdClass;
1968      }
1969  
1970      // Set main array as already seen
1971      $alreadySeen[] = & $array;
1972  
1973      foreach( $array as & $item )
1974      { // for each item in array
1975          if( !is_array( $item ) )
1976          { // if not array, we don't have to check it
1977              continue;
1978          }
1979  
1980          // put the unique object into the end of the array
1981          $item[] = $uniqueObject;
1982          $recursionDetected = false;
1983          foreach( $alreadySeen as $candidate )
1984          {
1985              if( end( $candidate ) === $uniqueObject )
1986              { // In the end of an already scanned array is the same unique Obect, this means that recursion was detected
1987                  $recursionDetected = true;
1988                  break;
1989              }
1990          }
1991  
1992          array_pop( $item );
1993  
1994          if( $recursionDetected || is_recursive( $item, $alreadySeen ) )
1995          { // Check until recursion detected or there are not more arrays
1996              return true;
1997          }
1998      }
1999  
2000      return false;
2001  }
