  41  /**
  42   * Messages class. For displaying notes, successful actions & errors.
  43   *
  44   * @todo CLEAN UP A LOT because of previous over factorization with Log class.
  45   * 
  46   * Messages can be logged into different categories (aka levels)
  47   * Examples: 'note', 'error'. Note: 'all' is reserved to display all categories together.
  48   * Messages can later be displayed grouped by category/level.
  49   *
  50   * @package evocore
  51   */
  52  class Messages
  53  {

  55       * The stored messages text.
  56       * array of Strings

  60      var $messages_text = array();

  62      /**
  63       * The stored messages type.
  64       * array of Strings
  65       *
  66       * @var array
  67       */
  68      var $messages_type = array();

  70      /**
  71       * The number of messages
  72       * 
  73       * @var integer
  74       */
  75      var $count = 0;
  76  
  77      /**
  78       * Error message was added or not.

  80       * @var boolean
  81       */
  82      var $has_errors = false;

  85       * Params

  92      /**
  93       * Constructor.
  94       *
  95       * @param array Params
  96       */
  97  	function Messages( $params = array() )
  98      {
  99          // Default params of messages
 100          $this->params = array_merge( array(
 101                  'class_success'  => 'log_success',
 102                  'class_warning'  => 'log_warning',
 103                  'class_error'    => 'log_error',
 104                  'class_note'     => 'log_note',
 105                  'before_message' => '',
 106              ), $params );
 107      }

 109      /**
 110       * Clears messages content
 111       */
 112  	function clear()
 113      {
 114          $this->messages_text = array();
 115          $this->messages_type = array();
 116          $this->count = 0;
 117          $this->has_errors = false;
 118      }

 121      /**
 122       * Add a message.
 123       *
 124       * @param string the message
 125       * @param string the message type, it can have this values: 'success', 'warning', 'error', 'note'
 126       */
 127  	function add( $text, $type = 'error' )
 128      {
 129          $this->messages_text[$this->count] = $text;
 130          $this->messages_type[$this->count] = $type;
 131          $this->count++;
 132          if( !$this->has_errors )
 133          {
 134              $this->has_errors = ( $type == 'error' );
 135          }
 136      }

 139      /**
 140       * Add a Messages object to this.
 141       *
 142       * @param Messages object
 143       */
 144  	function add_messages( $p_Messages )
 145      {
 146          $this->count = $this->count + $p_Messages->count;
 147          for( $i = 0; $i < $p_Messages->count; $i++ )
 148          {
 149              $this->messages_text[] = $p_Messages->messages_text[$i];
 150              $this->messages_type[] = $p_Messages->messages_type[$i];
 151              if( !$this->has_errors )
 152              {
 153                  $this->has_errors = ( $p_Messages->messages_type[$i] == 'error' );
 154              }
 155          }
 156      }

 160       * TEMPLATE TAG

 162       * The purpose here is to have a tag which is simple yet flexible.

 190      /**
 191       * Display messages of the object.
 192       *
 193       * - You can output/get the messages
 194       * - Head/Foot will be displayed on top/bottom of the messages.
 195       * - You can suppress the outer div or set a css class for it (defaults to
 196       *   'log_container').
 197       *
 198       * @todo Make this simple!
 199       * start by getting rid of the $category selection and the special cases for 'all'. If you don't want to display ALL messages,
 200       * then you should not log them in the same Log object and you should instantiate separate logs instead.
 201       *
 202       * @param string|NULL Header/title
 203       * @param string|NULL Footer
 204       * @param boolean to display or return (default: display)
 205       * @param mixed the outer div, may be false
 206       * @return boolean false, if no messages; else true (and outputs if $display)
 207       */
 208  	function display( $head = NULL, $foot = NULL, $display = true, $outerdivclass = 'log_container' )
 209      {
 210          if( $this->count == 0 ) {
 211              return false;
 212          }
 213  
 214          $disp = '';
 215  
 216          if( $outerdivclass )
 217          {
 218              $disp .= "\n<div class=\"$outerdivclass\">";
 219          }
 220  
 221          if( !empty( $head ) )
 222          {
 223              $disp .= '<h3>'.$head.'</h3>';
 224          }
 225  
 226          $disp .= '<ul>';
 227          for( $i = 0; $i < $this->count; $i++ )
 228          {
 229              $class = isset( $this->params['class_'.$this->messages_type[$i]] ) ?
 230                      $this->params['class_'.$this->messages_type[$i]] :
 231                      $this->params['class_note'];
 232              $disp .= "<li>\t<div class=\"{$class}\"".'>'
 233                      .$this->params['before_message']
 234                      .$this->messages_text[$i]."</div></li>\n";
 235          }
 236          $disp .= '</ul>';
 237  
 238          if( !empty( $foot ) )
 239          {
 240              $disp .= "\n<p>".$foot."</p>";
 241          }
 242  
 243          if( $outerdivclass )
 244          {
 245              $disp .= "</div>\n";
 246          }
 247  
 248          if( $display )
 249          {
 250              echo $disp;
 251              return true;
 252          }
 253          return $disp;
 254      }

 257      /**
 258       * Concatenates messages of a given category to a string
 259       *
 260       * @param string prefix of the string
 261       * @param string suffic of the string
 262       * @param string the glue
 263       * @param string result format
 264       * @return string the messages, imploded. Tags stripped.
 265       */
 266  	function get_string( $head = '', $foot = '', $implodeBy = ', ', $format = 'striptags' )
 267      {
 268          if( !$this->count )
 269          {
 270              return false;
 271          }
 272  
 273          $r = '';
 274          if( '' != $head )
 275          {
 276              $r .= $head.' ';
 277          }
 278          $r .= implode( $implodeBy, $this->messages_text );
 279          if( '' != $foot )
 280          {
 281              $r .= ' '.$foot;
 282          }
 283  
 284          switch( $format )
 285          {
 286              case 'xmlrpc':
 287                  $r = strip_tags( $r );    // get rid of <code>
 288                  $r = str_replace( '&lt;', '<', $r );
 289                  $r = str_replace( '&gt;', '>', $r );
 290                  $r = str_replace( '&quot;', '"', $r );
 291                  break;
 292  
 293              case 'striptags':
 294                  $r = strip_tags( $r );
 295                  break;
 296          }
 297  
 298          return $r;
 299      }
