 375      /**
 376       * Initialize sql query
 377       *
 378       * @todo count?
 379       *
 380       * @param boolean
 381       */
 382  	function query_init( $force_init = false )
 383      {
 384          global $DB;
 385  
 386          if( ! $force_init && ! empty( $this->query_is_initialized ) )
 387          { // Don't initialize query because it was already done
 388              return;
 389          }
 390  
 391          // Save to know the query init was done
 392          $this->query_is_initialized = true;
 393  
 394          if( empty( $this->filters ) )
 395          {    // Filters have not been set before, we'll use the default filterset:
 396              // If there is a preset filter, we need to activate its specific defaults:
 397              $this->filters['filter_preset'] = param( $this->param_prefix.'filter_preset', 'string', $this->default_filters['filter_preset'], true );
 398              $this->activate_preset_filters();
 399  
 400              // Use the default filters:
 401              $this->set_filters( $this->default_filters );
 402          }
 403  
 404          // GENERATE THE QUERY:
 405  
 406          /*
 407           * Resrict to selected blog
 408           */
 409          // If we dont have specific comment or post ids, we have to restric to blog
 410          if( !is_null( $this->Blog ) &&
 411              ( $this->filters['post_ID'] == NULL || ( ! empty($this->filters['post_ID']) && substr( $this->filters['post_ID'], 0, 1 ) == '-') ) &&
 412              ( $this->filters['comment_ID'] == NULL || ( ! empty($this->filters['comment_ID']) && substr( $this->filters['comment_ID'], 0, 1 ) == '-') ) &&
 413              ( $this->filters['comment_ID_list'] == NULL || ( ! empty($this->filters['comment_ID_list']) && substr( $this->filters['comment_ID_list'], 0, 1 ) == '-') ) )
 414          { // restriction for blog
 415              $this->ItemQuery->where_chapter( $this->Blog->ID );
 416          }
 417  
 418          /*
 419           * filtering stuff:
 420           */
 421          $this->CommentQuery->where_author( $this->filters['author_IDs'] );
 422          $this->CommentQuery->where_author_email( $this->filters['author_email'] );
 423          $this->CommentQuery->where_author_url( $this->filters['author_url'], $this->filters['url_match'], $this->filters['include_emptyurl'] );
 424          $this->CommentQuery->where_author_IP( $this->filters['author_IP'] );
 425          $this->ItemQuery->where_ID( $this->filters['post_ID'] );
 426          $this->CommentQuery->where_ID( $this->filters['comment_ID'], $this->filters['author'] );
 427          $this->CommentQuery->where_ID_list( $this->filters['comment_ID_list'] );
 428          $this->CommentQuery->where_rating( $this->filters['rating_toshow'], $this->filters['rating_turn'], $this->filters['rating_limit'] );
 429          $this->CommentQuery->where_keywords( $this->filters['keywords'], $this->filters['phrase'], $this->filters['exact'] );
 430          $this->CommentQuery->where_statuses( $this->filters['statuses'] );
 431          $this->CommentQuery->where_types( $this->filters['types'] );
 432          $this->ItemQuery->where_datestart( '', '', '', '', $this->filters['timestamp_min'], $this->filters['timestamp_max'] );
 433  
 434          if( !is_null( $this->Blog ) && isset( $this->filters['user_perm'] ) )
 435          { // If Blog and required user permission is set, add the corresponding restriction
 436              $this->CommentQuery->user_perm_restrict( $this->filters['user_perm'], $this->Blog->ID );
 437          }
 438  
 439  
 440          /*
 441           * ORDER BY stuff:
 442           */
 443          $available_sort_options = array( 'date', 'type', 'author', 'author_url', 'author_email', 'author_IP', 'spam_karma', 'status', 'item_ID' );
 444          $order_by = gen_order_clause( $this->filters['orderby'], $this->filters['order'], $this->Cache->dbprefix, $this->Cache->dbIDname, $available_sort_options );
 445  
 446          if( $this->filters['threaded_comments'] )
 447          {    // In mode "Threaded comments" we should get all replies in the begining of the list
 448              $order_by = $this->Cache->dbprefix.'in_reply_to_cmt_ID DESC, '.$order_by;
 449          }
 450  
 451          $this->CommentQuery->order_by( $order_by );
 452  
 453          // GET Item IDs, this way we don't have to JOIN two times the items and the categories table into the comment query
 454          if( isset( $this->filters['post_statuses'] ) )
 455          { // Set post statuses by filters
 456              $post_show_statuses = $this->filters['post_statuses'];
 457          }
 458          elseif( is_admin_page() )
 459          { // Allow all kind of post status ( This statuses will be filtered later by user perms )
 460              $post_show_statuses = get_visibility_statuses( 'keys' );
 461          }
 462          else
 463          { // Allow only inskin statuses for posts
 464              $post_show_statuses = get_inskin_statuses( isset( $this->Blog ) ? $this->Blog->ID : NULL, 'post');
 465          }
 466          // Restrict post filters to available statuses. When blog = 0 we will check visibility statuses for each blog separately ( on the same query ).
 467          $this->ItemQuery->where_visibility( $post_show_statuses );
 468          $sql_item_IDs = 'SELECT DISTINCT post_ID'
 469                          .$this->ItemQuery->get_from()
 470                          .$this->ItemQuery->get_where();
 471          $item_IDs = $DB->get_col( $sql_item_IDs, 0, 'Get CommentQuery Item IDs' );
 472          if( empty( $item_IDs ) )
 473          { // There is no item which belongs to the given blog and user may view it, so there are no comments either
 474              parent::count_total_rows( 0 );
 475              $this->CommentQuery->WHERE_and( 'FALSE' );
 476              return;
 477          }
 478          $this->CommentQuery->where_post_ID( implode( ',', $item_IDs ) );
 479  
 480          /*
 481           * Restrict to active comments by default, show expired comments only if it was requested
 482           * Note: This condition makes the CommentQuery a lot slower!
 483           */
 484          $this->CommentQuery->expiry_restrict( $this->filters['expiry_statuses'] );
 485  
 486          /*
 487           * GET TOTAL ROW COUNT:
 488           */
 489          $sql_count = '
 490                  SELECT COUNT( '.$this->Cache->dbIDname.') '
 491                      .$this->CommentQuery->get_from()
 492                      .$this->CommentQuery->get_where();
 493  
 494          parent::count_total_rows( $sql_count );
 495  
 496          /*
 497           * Page set up:
 498           */
 499          if( $this->page > 1 )
 500          { // We have requested a specific page number
 501              if( $this->limit > 0 )

 693      /**
 694       * Returns values needed to make sort links for a given column
 695       * This is needed because the order is not handled by the result class.
 696       * Reason: Sometimes the comment list needs to be ordered without having a display table, and columns. The result class order is based on columns.
 697       *
 698       * Returns an array containing the following values:
 699       *  - current_order : 'ASC', 'DESC' or ''
 700       *  - order_asc : url needed to order in ascending order
 701       *  - order_desc
 702       *  - order_toggle : url needed to toggle sort order
 703       *
 704       * @param integer column to sort
 705       * @return array
 706       */
 707  	function get_col_sort_values( $col_idx )
 708      {
 709          $col_order_fields = $this->cols[$col_idx]['order'];
 710  
 711          // Current order:
 712          if( $this->filters['orderby'] == $col_order_fields || $this->param_prefix.$this->filters['orderby'] == $col_order_fields  )
 713          {
 714              $col_sort_values['current_order'] = $this->filters['order'];
 715          }
 716          else
 717          {
 718              $col_sort_values['current_order'] = '';
 719          }
 720  
 721  
 722          // Generate sort values to use for sorting on the current column:
 723          $col_sort_values['order_asc'] = regenerate_url( array($this->param_prefix.'order',$this->param_prefix.'orderby'),
 724                                                                              $this->param_prefix.'order=ASC&amp;'.$this->param_prefix.'orderby='.$col_order_fields );
 725          $col_sort_values['order_desc'] = regenerate_url(  array($this->param_prefix.'order',$this->param_prefix.'orderby'),
 726                                                                              $this->param_prefix.'order=DESC&amp;'.$this->param_prefix.'orderby='.$col_order_fields );
 727  
 728          if( !$col_sort_values['current_order'] && isset( $this->cols[$col_idx]['default_dir'] ) )
 729          {    // There is no current order on this column and a default order direction is set for it
 730              // So set a default order direction for it
 731  
 732              if( $this->cols[$col_idx]['default_dir'] == 'A' )
 733              {    // The default order direction is A, so set its toogle  order to the order_asc
 734                  $col_sort_values['order_toggle'] = $col_sort_values['order_asc'];
 735              }
 736              else
 737              { // The default order direction is A, so set its toogle order to the order_desc
 738                  $col_sort_values['order_toggle'] = $col_sort_values['order_desc'];
 739              }
 740          }
 741          elseif( $col_sort_values['current_order'] == 'ASC' )
 742          {    // There is an ASC current order on this column, so set its toogle order to the order_desc
 743              $col_sort_values['order_toggle'] = $col_sort_values['order_desc'];
 744          }
 745          else
 746          { // There is a DESC or NO current order on this column,  so set its toogle order to the order_asc
 747              $col_sort_values['order_toggle'] = $col_sort_values['order_asc'];
 748          }
 749  
 750          // pre_dump( $col_sort_values );
 751  
 752          return $col_sort_values;

 776      /**
 777       * Auto pruning of recycled comments.
 778       *
 779       * It uses a general setting to store the day of the last prune, avoiding multiple prunes per day.
 780       * fplanque>> Check: How much faster is this than DELETING right away with an INDEX on the date field?
 781       *
 782       * NOTE: do not call this directly, but only in conjuction with auto_prune_stats_mode.
 783       *
 784       * @static
 785       * @return string Empty, if ok.
 786       */
 787  	function dbprune()
 788      {
 789          /**
 790           * @var DB
 791           */
 792          global $DB;
 793          global $Debuglog, $Settings, $localtimenow;
 794  
 795          // Prune when $localtime is a NEW day (which will be the 1st request after midnight):
 796          $last_prune = $Settings->get( 'auto_empty_trash_done' );
 797          if( $last_prune >= date('Y-m-d', $localtimenow) && $last_prune <= date('Y-m-d', $localtimenow+86400) )
 798          { // Already pruned today (and not more than one day in the future -- which typically never happens)
 799              return T_('Pruning of recycled comments has already been done today');
 800          }
 801  
 802          $time_prune_before = $localtimenow - ( $Settings->get('auto_empty_trash') * 86400 ); // 1 day = 86400 seconds
 803  
 804          $rows_affected = comments_delete_where( 'comment_status = "trash"
 805              AND comment_last_touched_ts < '.$DB->quote( date2mysql( $time_prune_before ) ) );
 806          $Debuglog->add( 'CommentList2::dbprune(): autopruned '.$rows_affected.' rows from T_comments.', 'request' );
 807  
 808          // Optimizing tables
 809          $DB->query('OPTIMIZE TABLE T_comments');
 810  
 811          $Settings->set( 'auto_empty_trash_done', date('Y-m-d H:i:s', $localtimenow) ); // save exact datetime
 812          $Settings->dbupdate();
 813  
 814          return ''; /* ok */
 815      }
