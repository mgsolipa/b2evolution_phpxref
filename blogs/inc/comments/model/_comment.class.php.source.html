 384      /**
 385       * Set the vote, as a number.
 386       *
 387       * @param string Vote type (spam, helpful)
 388       * @param string Vote value (spam, notsure, ok, yes, no)
 389       * @access protected
 390       */
 391  	function set_vote( $vote_type, $vote_value )
 392      {
 393          global $DB, $current_User;
 394  
 395          if( ! in_array( $vote_type, array( 'spam', 'helpful' ) ) )
 396          { // Restrict access for bad requests
 397              return;
 398          }
 399  
 400          switch ( $vote_value )
 401          { // Set a value for spam vote
 402              case "spam":
 403              case "yes":
 404                  $vote = '1';
 405                  break;
 406              case "notsure":
 407                  $vote = '0';
 408                  break;
 409              case "ok":
 410              case "no":
 411                  $vote = '-1';
 412                  break;
 413              default:
 414                  // $vote_value is not correct from ajax request
 415                  return;
 416          }
 417  
 418          if( empty( $this->ID ) )
 419          { // If comment doesn't exist
 420              return;
 421          }
 422  
 423          $DB->begin();
 424  
 425          $SQL = new SQL( 'Check if current user already voted on this comment' );
 426          $SQL->SELECT( 'cmvt_cmt_ID' );
 427          $SQL->FROM( 'T_comments__votes' );
 428          $SQL->WHERE( 'cmvt_cmt_ID = '.$DB->quote( $this->ID ) );
 429          $SQL->WHERE_and( 'cmvt_user_ID = '.$DB->quote( $current_User->ID ) );
 430          if( !$DB->get_row( $SQL->get() ) )
 431          { // Add a new vote for first time
 432              $DB->query( 'INSERT INTO T_comments__votes
 433                                       ( cmvt_cmt_ID, cmvt_user_ID, cmvt_'.$vote_type.' )
 434                                VALUES ( '.$DB->quote( $this->ID ).', '.$DB->quote( $current_User->ID ).', '.$DB->quote( $vote ).' )' );
 435          }
 436          else
 437          { // Update a vote
 438              $DB->query( 'UPDATE T_comments__votes
 439                              SET cmvt_'.$vote_type.' = '.$DB->quote( $vote ).'
 440                            WHERE cmvt_cmt_ID = '.$DB->quote( $this->ID ).'
 441                              AND cmvt_user_ID = '.$DB->quote( $current_User->ID ) );
 442          }
 443  
 444          $vote_SQL = new SQL( 'Get voting results of this comment' );
 445          $vote_SQL->SELECT( 'COUNT( cmvt_'.$vote_type.' ) AS c, SUM( cmvt_'.$vote_type.' ) AS s' );
 446          $vote_SQL->FROM( 'T_comments__votes' );
 447          $vote_SQL->WHERE( 'cmvt_cmt_ID = '.$DB->quote( $this->ID ) );
 448          $vote_SQL->WHERE_and( 'cmvt_'.$vote_type.' IS NOT NULL' );
 449          $vote = $DB->get_row( $vote_SQL->get() );
 450  
 451          // Update fields with vote counters for this comment
 452          $DB->query( 'UPDATE T_comments
 453                          SET comment_'.$vote_type.'_addvotes = '.$DB->quote( $vote->s ).',
 454                              comment_'.$vote_type.'_countvotes = '.$DB->quote( $vote->c ).'
 455                        WHERE comment_ID = '.$DB->quote( $this->ID ) );
 456          $this->{$vote_type.'_addvotes'} = $vote->s;
 457          $this->{$vote_type.'_countvotes'} = $vote->c;
 458  
 459          $DB->commit();
 460  
 461          return;
 462      }

 566      /**
 567       * Get the vote summary, as a string.
 568       *
 569       * @param type Vote type (spam, helpful)
 570       * @param srray Params
 571       * @return string
 572       */
 573  	function get_vote_summary( $type, $params = array() )
 574      {
 575          $params = array_merge( array(
 576                  'result_title'           => '',
 577                  'result_title_undecided' => '',
 578                  'after_result'           => '',
 579              ), $params );
 580  
 581          if( ! in_array( $type, array( 'spam', 'helpful' ) ) )
 582          {    // Bad request
 583              return '';
 584          }
 585  
 586          if( $this->{$type.'_countvotes'} == 0 )
 587          {    // No votes for current comment
 588              return '';
 589          }
 590  
 591          // Calculate a vote summary
 592          $summary = ceil( $this->{$type.'_addvotes'} / $this->{$type.'_countvotes'} * 100 );
 593  
 594          if( $summary < -20 )
 595          {    // Comment is OK
 596              $summary = abs($summary).'% '.( $type == 'spam' ? T_('OK') : T_('not helpful') );
 597          }
 598          else if( $summary >= -20 && $summary <= 20 )
 599          {    // Comment is UNDECIDED
 600              $summary = T_('UNDECIDED');
 601              if( !empty( $params['result_title_undecided'] ) )
 602              {    // Display title before undecided results
 603                  $summary = $params['result_title_undecided'].' '.$summary;
 604              }
 605          }
 606          else if( $summary > 20 )
 607          {    // Comment is SPAM
 608              $summary .= '% '.( $type == 'spam' ? T_('SPAM') : T_('helpful') );
 609          }
 610  
 611          if( !empty( $params['result_title'] ) )
 612          {    // Display title before results
 613              $summary = $params['result_title'].' '.$summary;
 614          }
 615  
 616          return $summary.$params['after_result'].' ';
 617      }
