  18  function fadeIn( selector, color )
  19  {
  20      if( jQuery( selector ).length == 0 )
  21      {
  22          return;
  23      }
  24      if( jQuery( selector ).get(0).tagName == 'TR' )
  25      { // Fix selector, <tr> cannot have a css property background-color
  26          selector = selector + ' td';
  27      }
  28      var bg_color = jQuery( selector ).css( 'backgroundColor' );
  29      jQuery( selector ).animate( { backgroundColor: color }, 200 );
  30      return bg_color;
  31  }
