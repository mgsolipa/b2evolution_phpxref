  30  // Create SQL query to find all file hashes that satisfy to filters

  31  $hash_SQL = new SQL();
  32  $hash_SQL->SELECT( 'f.file_hash, COUNT( f.file_ID )' );
  33  $hash_SQL->FROM( 'T_files AS f' );
  34  $hash_SQL->FROM_add( 'INNER JOIN (
  35          SELECT file_hash
  36            FROM T_files
  37           GROUP BY file_hash
  38          HAVING COUNT( file_ID ) > 1
  39      ) AS fd ON fd.file_hash = f.file_hash' );
  40  $hash_SQL->GROUP_BY( 'f.file_hash' );
  41  
  42  if( ( $min_inappropriate_votes > 0 ) || ( $min_spam_votes > 0 ) )
  43  { // we need to join files vote table to be able to filter by votes
  44      $hash_SQL->FROM_add( 'LEFT JOIN T_links AS l ON l.link_file_ID = f.file_ID' );
  45      $hash_SQL->FROM_add( 'LEFT JOIN T_links__vote AS lv ON lv.lvot_link_ID = l.link_ID' );
  46      if( $min_inappropriate_votes > 0 )
  47      {    // Filter by minimum count of inappropriate votes
  48          $hash_SQL->HAVING_and( 'SUM( lvot_inappropriate ) >= '.$DB->quote( $min_inappropriate_votes ) );
  49      }
  50      if( $min_spam_votes > 0 )
  51      {    // Filter by minimum count of spam votes
  52          $hash_SQL->HAVING_and( 'SUM( lvot_spam ) >= '.$DB->quote( $min_spam_votes ) );
  53      }
  54  }
  55  if( $file_ID > 0 )
  56  {    // Filter by hash of File ID
  57      $hash_SQL->WHERE_and( '( SELECT fh.file_hash FROM T_files AS fh WHERE fh.file_ID ='.$DB->quote( $file_ID ).' ) = f.file_hash' );
  58  }
  59  
  60  // Get all distinct hash values from what we have duplicates and at least one file from duplicates corresponds to filters

  61  $hash_results = $DB->get_assoc( $hash_SQL->get() );
  62  $file_hash_values = array_keys( $hash_results );
  63  // set the number of count all files which will be displayed

  64  $num_file_results = array_sum( $hash_results );
  65  
  66  if( $num_file_results > 0 )
  67  { // Create SQL query to build a results table
  68      $SQL = new SQL();
  69      $SQL->SELECT( 'f.*,
  70          SUM( IFNULL( lvot_like, 0 ) ) as total_like,
  71          SUM( IFNULL( lvot_inappropriate, 0 ) ) as total_inappropriate,
  72          SUM( IFNULL( lvot_spam, 0 ) ) as total_spam,
  73          ( SELECT COUNT( file_ID ) FROM T_files AS f2 WHERE f.file_hash = f2.file_hash ) AS total_duplicates' );
  74      $SQL->FROM( 'T_files AS f' );
  75      $SQL->FROM_add( 'LEFT JOIN T_links AS l ON l.link_file_ID = f.file_ID' );
  76      $SQL->FROM_add( 'LEFT JOIN T_links__vote AS lv ON lv.lvot_link_ID = l.link_ID' );
  77      $SQL->WHERE( 'f.file_hash IN ( '.$DB->quote( $file_hash_values ).' )' );
  78      $SQL->GROUP_BY( 'f.file_ID, f.file_hash' );
  79      $SQL->ORDER_BY( 'f.file_hash, *, total_spam DESC, total_inappropriate DESC' );
  80  }
