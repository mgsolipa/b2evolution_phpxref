 142  	function get_default_group_permissions( $grp_ID )
 143      {
 144          switch( $grp_ID )
 145          {
 146              case 1: // Administrators group ID equals 1
 147                  $permfiles = 'all'; // Files permissions
 148                  $permshared = 'edit'; // Access to shared root
 149                  $permimport = 'edit'; // Access to import root
 150                  break;
 151              case 2: // Moderators group equals 2
 152                  $permfiles = 'add';
 153                  $permshared = 'add';
 154                  $permimport = 'none';
 155                  break;
 156              case 3: // Trusted Users (group ID 3) have permission by default:
 157                  $permfiles = 'view';
 158                  $permshared = 'view';
 159                  $permimport = 'none';
 160                  break;
 161              default: // Other groups
 162                  $permfiles = 'none';
 163                  $permshared = 'none';
 164                  $permimport = 'none';
 165                  break;
 166          }
 167  
 168          // We can return as many default permissions as we want:
 169          // e.g. array ( permission_name => permission_value, ... , ... )
 170          return $permissions = array(
 171                  'perm_files' => $permfiles,
 172                  'perm_shared_root' => $permshared,
 173                  'perm_import_root' => $permimport,
 174              );

 266      /**
 267       * Check a permission for the group. ( see 'group_func' in get_available_group_permissions() function )
 268       *
 269       * @param string Requested permission level
 270       * @param string Permission value
 271       * @param mixed Permission target (blog ID, array of cat IDs...)
 272       * @return boolean True on success (permission is granted), false if permission is not granted
 273       */
 274  	function check_files_group_perm( $permlevel, $permvalue, $permtarget )
 275      {
 276          $perm = false;
 277          switch ( $permvalue )
 278          {
 279              case 'all':
 280                  global $demo_mode;
 281                  if( ( ! $demo_mode ) && ( $permlevel == 'all' ) )
 282                  { // All permissions granted
 283                      $perm = true;
 284                      break;
 285                  }
 286  
 287              case 'edit':
 288                  // User can ask for normal edit perm...
 289                  if( $permlevel == 'edit' )
 290                  {
 291                      $perm = true;
 292                      break;
 293                  }
 294  
 295              case 'add':
 296                  // User can ask for add perm...
 297                  if( $permlevel == 'add' )
 298                  {
 299                      $perm = true;
 300                      break;
 301                  }
 302  
 303              case 'view':
 304                  // User can ask for view perm...
 305                  if( $permlevel == 'view' )
 306                  {
 307                      $perm = true;
 308                      break;
 309                  }
 310          }
 311  
 312          if( $perm && isset($permtarget) && ( is_a( $permtarget, 'FileRoot' ) ) )
 313          {
 314              global $current_User;
 315              switch( $permtarget->type )
 316              {
 317                  case 'shared':
 318                      return $current_User->check_perm( 'shared_root', $permlevel );
 319                  case 'import':
 320                      return $current_User->check_perm( 'import_root', $permlevel );
 321                  case 'user':
 322                      return $permtarget->in_type_ID == $current_User->ID;
 323              }
 324          }
 325  
 326          return $perm;
 327      }
