  35  /**
  36   * This class provides info about File Roots.
  37   *
  38   * These are root directories available for media file storage, under access permission.
  39   *
  40   * @package evocore
  41   */
  42  class FileRootCache
  43  {
  44      /**
  45       * Internal cache
  46       * @var array
  47       */
  48      var $cache = array();
  49  
  50  
  51      /**
  52       * Get an array of ALL available Fileroots (not just the cached ones).
  53       *
  54       * @todo fp> it would probably make sense to refactor this as the constructor for the file roots
  55       * and initialize the whole cache at construction time
  56       *
  57       * @static
  58       *
  59       * @return array of FileRoots (key being the FileRoot's ID)
  60       */
  61  	function get_available_FileRoots()
  62      {
  63          global $current_User;
  64          global $collections_Module;
  65  
  66          $r = array();
  67  
  68          // The user's blog (if available) is the default/first one:
  69          $user_FileRoot = & $this->get_by_type_and_ID( 'user', $current_User->ID, true );
  70          if( $user_FileRoot )
  71          { // We got a user media dir:
  72              $r[ $user_FileRoot->ID ] = & $user_FileRoot;
  73          }
  74  
  75          if( isset($collections_Module) )
  76          {    // Blog/collection media dirs:
  77              $BlogCache = & get_BlogCache();
  78              $bloglist = $BlogCache->load_user_blogs( 'blog_media_browse', $current_User->ID );
  79              foreach( $bloglist as $blog_ID )
  80              {
  81                  if( $Root = & $this->get_by_type_and_ID( 'collection', $blog_ID, true ) )
  82                  {
  83                      $r[ $Root->ID ] = & $Root;
  84                  }
  85              }
  86          }
  87  
  88          // Shared root:
  89          $shared_FileRoot = & $this->get_by_type_and_ID( 'shared', 0, true );
  90          if( $shared_FileRoot )
  91          { // We got a shared dir:
  92              $r[ $shared_FileRoot->ID ] = & $shared_FileRoot;
  93          }
  94  
  95          if( isset($collections_Module) )
  96          { // Skins root:
  97              $skins_FileRoot = & $this->get_by_type_and_ID( 'skins', 0, false );
  98              if( $skins_FileRoot )
  99              { // We got a skins dir:
 100                  $r[ $skins_FileRoot->ID ] = & $skins_FileRoot;
 101              }
 102          }
 103  
 104          // Import root:
 105          $import_FileRoot = & $this->get_by_type_and_ID( 'import', 0, true );
 106          if( $import_FileRoot )
 107          { // We got an import dir:
 108              $r[ $import_FileRoot->ID ] = & $import_FileRoot;
 109          }
 110  
 111          return $r;
 112      }

 115      /**
 116       * Get a FileRoot (cached) by ID.
 117       *
 118       * @uses FileRootCache::get_by_type_and_ID()
 119       * @param string ID of the FileRoot (e.g. 'user_X' or 'collection_X')
 120       * @param boolean Create the directory, if it does not exist yet?
 121       * @return FileRoot|false FileRoot on success, false on failure (ads_path is false).
 122       */
 123      function & get_by_ID( $id, $create = false )
 124      {
 125          $part = explode( '_', $id );
 126          $root_type = $part[0];
 127          $root_in_type_ID = $part[1];
 128  
 129          return $this->get_by_type_and_ID( $root_type, $root_in_type_ID, $create );
 130      }
 131  
 132  
 133      /**
 134       * Get a FileRoot (cached).
 135       *
 136       * @param string Root type: 'user', 'group', 'collection' or 'absolute'
 137       * @param integer ID of the user, the group or the collection the file belongs to...
 138       * @param boolean Create the directory, if it does not exist yet?
 139       * @return FileRoot|false FileRoot on success, false on failure (ads_path is false).
 140       */
 141      function & get_by_type_and_ID( $root_type, $root_in_type_ID, $create = false )
 142      {
 143          $root_ID = FileRoot::gen_ID( $root_type, $root_in_type_ID );
 144  
 145          if( ! isset( $this->cache[$root_ID] ) )
 146          {    // Not in Cache, let's instantiate:
 147              $Root = new FileRoot( $root_type, $root_in_type_ID, $create ); // COPY (func)
 148              if( empty($Root->ads_path) ) // false
 149              {
 150                  $Root = false;
 151              }
 152              $this->cache[$root_ID] = & $Root;
 153          }
 154  
 155          return $this->cache[$root_ID];
 156      }
 157  
 158  
 159      /**
 160       * Get the absolute path (FileRoot::ads_path) to a given root (with ending slash).
 161       *
 162       * @deprecated since 1.9
 163       * @param boolean Create the directory, if it does not exist yet?
 164       * @return string
 165       */
 166  	function get_root_dir( $root_type, $root_in_type_ID, $create = false )
 167      {
 168          $tmp_FileRoot = & $this->get_by_type_and_ID( $root_type, $root_in_type_ID, $create );
 169          return $tmp_FileRoot->ads_path;
 170      }
 171  
 172  
 173      /**
 174       * Clear the cache
 175       */
 176  	function clear()
 177      {
 178          $this->cache = array();
 179      }
 180  }
