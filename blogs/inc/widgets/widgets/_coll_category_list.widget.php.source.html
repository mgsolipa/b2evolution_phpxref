 165              'before_level' => array( $this, 'cat_before_level' ),
 166              'after_level'  => array( $this, 'cat_after_level' )
 167          );
 168  
 169          if( !empty( $params['callback_posts'] ) )

 175          echo $this->disp_params['block_start'];

 178          $this->disp_title();
 179

 185          }

 194                  $tmp_disp .= $this->add_cat_class_attr( $this->disp_params['item_start'], 'all' );

 196                  if( $this->disp_params['link_type'] == 'context' )

 216          }

 329      /**
 330       * Callback: Generate category line when it has children
 331       *
 332       * @param Chapter generic category we want to display
 333       * @param int level of the category in the recursive tree
 334       * @return string HTML
 335       */
 336  	function cat_line( $Chapter, $level )
 337      {
 338          global $cat_array;
 339  
 340          if( !isset($cat_array) )
 341          {
 342              $cat_array = array();
 343          }
 344  
 345          $exclude_cats = sanitize_id_list($this->disp_params['exclude_cats'], true);
 346          if( in_array( $Chapter->ID, $exclude_cats ) )
 347          { // Cat ID is excluded, skip it
 348              return;
 349          }
 350  
 351          if( in_array( $Chapter->ID, $cat_array ) )
 352          { // This category is in the current selection
 353              $start_tag = $this->disp_params['item_selected_start'];
 354          }
 355          else if( empty( $Chapter->children ) )
 356          { // This category has no children
 357              $start_tag = $this->disp_params['item_last_start'];
 358          }
 359          else
 360          {
 361              $start_tag = $this->disp_params['item_start'];
 362          }
 363  
 364          if( $Chapter->meta )
 365          { // Add class name "meta" for meta categories
 366              $start_tag = $this->add_cat_class_attr( $start_tag, 'meta' );
 367          }
 368  
 369          $r = $start_tag;
 370  
 371          if( $this->disp_params['use_form'] || $this->disp_params['display_checkboxes'] )
 372          { // We want to add form fields:
 373              $cat_checkbox_params = '';
 374              if( $Chapter->meta )
 375              { // Disable the checkbox of meta category ( and hide it by css )
 376                  $cat_checkbox_params = ' disabled="disabled"';
 377              }
 378  
 379              $r .= '<label><input type="checkbox" name="catsel[]" value="'.$Chapter->ID.'" class="checkbox middle"';
 380              if( in_array( $Chapter->ID, $cat_array ) )
 381              { // This category is in the current selection
 382                  $r .= ' checked="checked"';
 383              }
 384              $r .= $cat_checkbox_params.' /> ';
 385          }
 386  
 387          $cat_name = $Chapter->dget('name');
 388          if( $Chapter->lock && isset( $this->disp_params['show_locked'] ) && $this->disp_params['show_locked'] )
 389          {
 390              $cat_name .= '<span style="padding:0 5px;" >'.get_icon( 'file_not_allowed', 'imgtag', array( 'title' => T_('Locked') ) ).'</span>';
 391          }
 392  
 393          // Make a link from category name
 394          $r .= '<a href="';
 395          if( $this->disp_params['link_type'] == 'context' )
 396          { // We want to preserve current browsing context:
 397              $r .= regenerate_url( 'cats,catsel', 'cat='.$Chapter->ID );
 398          }
 399          else
 400          {
 401              $r .= $Chapter->get_permanent_url();
 402          }
 403          $r .= '">'.$cat_name.'</a>';
 404  
 405          if( $this->disp_params['use_form'] || $this->disp_params['display_checkboxes'] )
 406          { // We want to add form fields:
 407              $r .= '</label>';
 408          }
 409  
 410          // Do not end line here because we need to include children first!
 411          // $r .= $this->disp_params['item_end'];
 412  
 413          return $r;
 414      }

 467      /**
 468       * Add new class name for start tag
 469       *
 470       * @param string HTML start tag: e.g. <div class="div_class"> or <div>
 471       * @param string New class name
 472       * @return string HTML start tag with new added class name
 473       */
 474  	function add_cat_class_attr( $start_tag, $class_name )
 475      {
 476          if( preg_match( '/ class="[^"]*"/i', $start_tag ) )
 477          { // Append to already existing attribute
 478              return preg_replace( '/ class="([^"]*)"/i', ' class="$1 '.$class_name.'"', $start_tag );
 479          }
 480          else
 481          { // Add new attribute for meta class
 482              return preg_replace( '/^<([^\s>]+)/', '<$1 class="'.$class_name.'"', $start_tag );
 483          }
 484      }
