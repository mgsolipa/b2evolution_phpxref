 168      /**
 169       * Display the widget!
 170       *
 171       * @param array MUST contain at least the basic display params
 172       */
 173  	function display( $params )
 174      {
 175          /**
 176          * @var Blog
 177          */
 178          global $Blog;
 179  
 180          $this->init_display( $params );
 181  
 182          // Collection common links:
 183          echo $this->disp_params['block_start'];
 184  
 185          // Display title if requested
 186          $this->disp_title();
 187  
 188          echo $this->disp_params['block_body_start'];
 189  
 190          echo $this->disp_params['list_start'];
 191  
 192          if( $this->disp_params['show_home'] )
 193          {
 194              echo $this->disp_params['item_start'];
 195              echo '<strong><a href="'.$Blog->get('url').'">'.T_('Home').'</a></strong>';
 196              echo $this->disp_params['item_end'];
 197          }
 198  
 199          if( $this->disp_params['show_recently'] )
 200          {
 201              echo $this->disp_params['item_start'];
 202              echo '<strong><a href="'.$Blog->get('recentpostsurl').'">'.T_('Recently').'</a></strong>';
 203              echo $this->disp_params['item_end'];
 204          }
 205  
 206          if( $this->disp_params['show_search'] )
 207          {
 208              echo $this->disp_params['item_start'];
 209              echo '<strong><a href="'.$Blog->get('searchurl').'">'.T_('Search').'</a></strong>';
 210              echo $this->disp_params['item_end'];
 211          }
 212  
 213          if( $this->disp_params['show_postidx'] )
 214          {
 215              echo $this->disp_params['item_start'];
 216              echo '<strong><a href="'.$Blog->get('postidxurl').'">'.T_('Post index').'</a></strong>';
 217              echo $this->disp_params['item_end'];
 218          }
 219  
 220          if( $this->disp_params['show_archives'] )
 221          {
 222              // fp> TODO: don't display this if archives plugin not installed... or depluginize archives (I'm not sure)
 223              echo $this->disp_params['item_start'];
 224              echo '<strong><a href="'.$Blog->get('arcdirurl').'">'.T_('Archives').'</a></strong>';
 225              echo $this->disp_params['item_end'];
 226          }
 227  
 228          if( $this->disp_params['show_categories'] )
 229          {
 230              echo $this->disp_params['item_start'];
 231              echo '<strong><a href="'.$Blog->get('catdirurl').'">'.T_('Categories').'</a></strong>';
 232              echo $this->disp_params['item_end'];
 233          }
 234  
 235          if( $this->disp_params['show_mediaidx'] )
 236          {
 237              echo $this->disp_params['item_start'];
 238              echo '<strong><a href="'.$Blog->get('mediaidxurl').'">'.T_('Photo index').'</a></strong>';
 239              echo $this->disp_params['item_end'];
 240          }
 241  
 242          if( $this->disp_params['show_latestcomments'] && $Blog->get_setting( 'comments_latest' ) )
 243          { // Display link to latest comments if this feature is enabled for current blog
 244              echo $this->disp_params['item_start'];
 245              echo '<strong><a href="'.$Blog->get('lastcommentsurl').'">'.T_('Latest comments').'</a></strong>';
 246              echo $this->disp_params['item_end'];
 247          }
 248  
 249          if( $this->disp_params['show_owneruserinfo'] )
 250          {
 251              echo $this->disp_params['item_start'];
 252              echo '<strong><a href="'.url_add_param( $Blog->get('userurl'), 'user_ID='.$Blog->owner_user_ID ).'">'.T_('Owner details').'</a></strong>';
 253              echo $this->disp_params['item_end'];
 254          }
 255  
 256          if( $this->disp_params['show_ownercontact'] && $url = $Blog->get_contact_url( true ) )
 257          {    // owner allows contact:
 258              echo $this->disp_params['item_start'];
 259              echo '<strong><a href="'.$url.'">'.T_('Contact').'</a></strong>';
 260              echo $this->disp_params['item_end'];
 261          }
 262  
 263          if( $this->disp_params['show_sitemap'] )
 264          {
 265              echo $this->disp_params['item_start'];
 266              echo '<strong><a href="'.$Blog->get('sitemapurl').'">'.T_('Site map').'</a></strong>';
 267              echo $this->disp_params['item_end'];
 268          }
 269  
 270          echo $this->disp_params['list_end'];
 271  
 272          echo $this->disp_params['block_body_end'];
 273  
 274          echo $this->disp_params['block_end'];
 275  
 276          return true;
 277      }
