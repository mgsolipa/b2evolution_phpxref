 169      /**
 170       * Display the widget!
 171       *
 172       * @param array MUST contain at least the basic display params
 173       */
 174  	function display( $params )
 175      {
 176          global $localtimenow, $DB;
 177  
 178          $this->init_display( $params );
 179  
 180          global $Blog;
 181          $blog_ID = ( $this->disp_params[ 'blog_ID' ] ? $this->disp_params[ 'blog_ID' ] : $Blog->ID );
 182          //pre_dump( $blog_ID );
 183  
 184          // Display photos:
 185          // TODO: permissions, complete statuses...
 186          // TODO: A FileList object based on ItemListLight but adding File data into the query?
 187          //          overriding ItemListLigth::query() for starters ;)
 188  
 189          // Init caches
 190          $ItemCache = & get_ItemCache();
 191          $FileCache = & get_FileCache();
 192  
 193          // Query list of files:
 194          // Note: We use ItemQuery to get attachments from all posts which should be visible ( even in case of aggregate blogs )
 195          $ItemQuery = new ItemQuery( $ItemCache->dbtablename, $ItemCache->dbprefix, $ItemCache->dbIDname );
 196          $ItemQuery->SELECT( 'post_ID, post_datestart, post_datemodified, post_main_cat_ID, post_urltitle, post_canonical_slug_ID,
 197                                      post_tiny_slug_ID, post_ptyp_ID, post_title, post_excerpt, post_url, file_ID,
 198                                      file_title, file_root_type, file_root_ID, file_path, file_alt, file_desc, file_path_hash' );
 199          $ItemQuery->FROM_add( 'INNER JOIN T_links ON post_ID = link_itm_ID' );
 200          $ItemQuery->FROM_add( 'INNER JOIN T_files ON link_file_ID = file_ID' );
 201          $ItemQuery->where_chapter( $blog_ID );
 202          $ItemQuery->where_visibility( NULL );
 203          $ItemQuery->WHERE_and( 'post_datestart <= \''.remove_seconds( $localtimenow ).'\'' );
 204          if( !empty( $this->disp_params[ 'item_type' ] ) )
 205          { // Get items only with specified type
 206              $ItemQuery->WHERE_and( 'post_ptyp_ID = '.intval( $this->disp_params[ 'item_type' ] ) );
 207          }
 208          $ItemQuery->GROUP_BY( 'link_ID' );
 209          $ItemQuery->LIMIT( intval( $this->disp_params[ 'limit' ] ) * 4 ); // fp> TODO: because no way of getting images only, we get 4 times more data than requested and hope that 25% at least will be images :/
 210          $ItemQuery->ORDER_BY(    gen_order_clause( $this->disp_params['order_by'], $this->disp_params['order_dir'],
 211                                              'post_', 'post_ID '.$this->disp_params['order_dir'].', link_ID' ) );
 212  
 213          // Init FileList with the above defined query
 214          $FileList = new DataObjectList2( $FileCache );
 215          $FileList->sql = $ItemQuery->get();
 216  
 217          $FileList->query( false, false, false, 'Media index widget' );
 218  
 219          $layout = $this->disp_params[ 'thumb_layout' ];
 220  
 221          $nb_cols = $this->disp_params[ 'grid_nb_cols' ];
 222          $count = 0;
 223          $r = '';
 224          /**
 225           * @var File
 226           */
 227          while( $File = & $FileList->get_next() )
 228          {
 229              if( $count >= $this->disp_params[ 'limit' ] )
 230              {    // We have enough images already!
 231                  break;
 232              }
 233  
 234              if( ! $File->is_image() )
 235              { // Skip anything that is not an image
 236                  // Only images are selected or those files where we don't know the file type yet.
 237                  // This check is only for those files where we don't know the filte type. The file type will be set during the check.
 238                  continue;
 239              }
 240  
 241              if( $layout == 'grid' )
 242              {
 243                  if( $count % $nb_cols == 0 )
 244                  {
 245                      $r .= $this->disp_params[ 'grid_colstart' ];
 246                  }
 247                  $r .= $this->disp_params[ 'grid_cellstart' ];
 248              }
 249              else
 250              {
 251                  $r .= $this->disp_params[ 'item_start' ];
 252              }
 253  
 254              // 1/ Hack a dirty permalink( will redirect to canonical):
 255              // $link = url_add_param( $Blog->get('url'), 'p='.$post_ID );
 256  
 257              // 2/ Hack a link to the right "page". Very daring!!
 258              // $link = url_add_param( $Blog->get('url'), 'paged='.$count );
 259  
 260              // 3/ Instantiate a light object in order to get permamnent url:
 261              $ItemLight = new ItemLight( $FileList->get_row_by_idx( $FileList->current_idx - 1 ) );    // index had already been incremented
 262  
 263              $r .= '<a href="'.$ItemLight->get_permanent_url().'">';
 264              // Generate the IMG THUMBNAIL tag with all the alt, title and desc if available
 265              $r .= $File->get_thumb_imgtag( $this->disp_params['thumb_size'], '', '', $ItemLight->title );
 266              $r .= '</a>';
 267              if( $this->disp_params[ 'disp_image_title' ] )
 268              { // Dislay title of image or item
 269                  $title = ( $File->get( 'title' ) ) ? $File->get( 'title' ) : $ItemLight->title;
 270                  if( ! empty( $title ) )
 271                  {
 272                      $r .= '<span class="note">'.$title.'</span>';
 273                  }
 274              }
 275  
 276              ++$count;
 277  
 278              if( $layout == 'grid' )
 279              {
 280                  $r .= $this->disp_params[ 'grid_cellend' ];
 281                  if( $count % $nb_cols == 0 )
 282                  {
 283                      $r .= $this->disp_params[ 'grid_colend' ];
 284                  }
 285              }
 286              else
 287              {
 288                  $r .= $this->disp_params[ 'item_end' ];
 289              }
 290          }
 291  
 292          // Exit if no files found
 293          if( empty($r) ) return;
 294  
 295          echo $this->disp_params[ 'block_start'];
 296  
 297          // Display title if requested
 298          $this->disp_title();
 299  
 300          echo $this->disp_params['block_body_start'];
 301  
 302          if( $layout == 'grid' )
 303          {
 304              echo $this->disp_params[ 'grid_start' ];
 305          }
 306          else
 307          {
 308              echo $this->disp_params[ 'list_start' ];
 309          }
 310  
 311          echo $r;
 312  
 313          if( $layout == 'grid' )
 314          {
 315              if( $count && ( $count % $nb_cols != 0 ) )
 316              {
 317                  echo $this->disp_params[ 'grid_colend' ];
 318              }
 319  
 320              echo $this->disp_params[ 'grid_end' ];
 321          }
 322          else
 323          {
 324              echo $this->disp_params[ 'list_end' ];
 325          }
 326  
 327          echo $this->disp_params['block_body_end'];
 328  
 329          echo $this->disp_params[ 'block_end' ];
 330  
 331          return true;
 332      }
