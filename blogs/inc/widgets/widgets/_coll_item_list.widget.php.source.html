 445      /**
 446       * Support function for above
 447       *
 448       * @param Item
 449       * @return boolean TRUE - if content is displayed
 450       */
 451  	function disp_contents( & $disp_Item )
 452      {
 453          // Check if only the title was displayed before the first picture
 454          $displayed_only_title = false;
 455  
 456          // Set this var to TRUE when some content(title, excerpt or picture) is displayed
 457          $content_is_displayed = false;
 458  
 459          // Is this the current item?
 460          global $disp, $Item;
 461          if( !empty($Item) && $disp_Item->ID == $Item->ID )
 462          {    // The current page is currently displaying the Item this link is pointing to
 463              // Let's display it as selected
 464              $link_class = $this->disp_params['link_selected_class'];
 465          }
 466          else
 467          {    // Default link class
 468              $link_class = $this->disp_params['link_default_class'];
 469          }
 470  
 471          if( $link_class == $this->disp_params['link_selected_class'] )
 472          {
 473              echo $this->disp_params['item_selected_start'];
 474          }
 475          else
 476          {
 477              echo $this->disp_params['item_start'];
 478          }
 479  
 480          if( $this->disp_params[ 'disp_title' ] )
 481          {    // Display title
 482              $disp_Item->title( array(
 483                      'link_type'  => $this->disp_params['item_title_link_type'],
 484                      'link_class' => $link_class,
 485                  ) );
 486              $displayed_only_title = true;
 487              $content_is_displayed = true;
 488          }
 489  
 490          if( $this->disp_params[ 'disp_excerpt' ] )
 491          {
 492              $excerpt = $disp_Item->dget( 'excerpt', 'htmlbody' );
 493              if( !empty($excerpt) )
 494              {    // Note: Excerpts are plain text -- no html (at least for now)
 495                  echo '<div class="item_excerpt">'.$excerpt.'</div>';
 496                  $displayed_only_title = false;
 497                  $content_is_displayed = true;
 498              }
 499          }
 500  
 501          if( $this->disp_params['disp_teaser'] )
 502          { // we want to show some or all of the post content
 503              $content = $disp_Item->get_content_teaser( 1, false, 'htmlbody' );
 504  
 505              if( $words = $this->disp_params['disp_teaser_maxwords'] )
 506              { // limit number of words
 507                  $content = strmaxwords( $content, $words, array(
 508                          'continued_link' => $disp_Item->get_permanent_url(),
 509                          'continued_text' => '&hellip;',
 510                       ) );
 511              }
 512              echo '<div class="item_content">'.$content.'</div>';
 513              $displayed_only_title = false;
 514              $content_is_displayed = true;
 515  
 516              /* fp> does that really make sense?
 517                  we're no longer in a linkblog/linkroll use case here, are we?
 518              $disp_Item->more_link( array(
 519                      'before'    => '',
 520                      'after'     => '',
 521                      'link_text' => T_('more').' &raquo;',
 522                  ) );
 523                  */
 524          }
 525  
 526          if( in_array( $this->disp_params[ 'attached_pics' ], array( 'first', 'all' ) ) )
 527          {    // Display attached pictures
 528              $picture_limit = $this->disp_params[ 'attached_pics' ] == 'first' ? 1 : 1000;
 529              $LinkOnwer = new LinkItem( $disp_Item );
 530              if( $FileList = $LinkOnwer->get_attachment_FileList( $picture_limit ) )
 531              {    // Get list of attached files
 532                  while( $File = & $FileList->get_next() )
 533                  {
 534                      if( $File->is_image() )
 535                      {    // Get only images
 536                          switch( $this->disp_params[ 'item_pic_link_type' ] )
 537                          {    // Set url for picture link
 538                              case 'none':
 539                                  $pic_url = NULL;
 540                                  break;
 541  
 542                              case 'permalink':
 543                                  $pic_url = $disp_Item->get_permanent_url();
 544                                  break;
 545  
 546                              case 'linkto_url':
 547                                  $pic_url = $disp_Item->url;
 548                                  break;
 549  
 550                              case 'auto':
 551                              default:
 552                                  $pic_url = ( empty( $disp_Item->url ) ? $disp_Item->get_permanent_url() : $disp_Item->url );
 553                                  break;
 554                          }
 555  
 556                          if( $displayed_only_title )
 557                          { // If only the title was displayed - Insert new line before the first picture
 558                              echo '<br />';
 559                              $displayed_only_title = false;
 560                          }
 561  
 562                          // Print attached picture
 563                          echo $File->get_tag( '', '', '', '', $this->disp_params['thumb_size'], $pic_url );
 564  
 565                          $content_is_displayed = true;
 566                      }
 567                  }
 568              }
 569          }
 570  
 571          if( $link_class == $this->disp_params['link_selected_class'] )
 572          {
 573              echo $this->disp_params['item_selected_end'];
 574          }
 575          else
 576          {
 577              echo $this->disp_params['item_end'];
 578          }
 579  
 580          return $content_is_displayed;
 581      }
