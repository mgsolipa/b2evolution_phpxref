 105          }

 543      /**
 544       * Maybe be overriden by some widgets, depending on what THEY depend on..
 545       *
 546       * @return array of keys this widget depends on
 547       */
 548  	function get_cache_keys()
 549      {
 550          global $Blog, $current_User;
 551  
 552          $keys = array(
 553                  'wi_ID'   => $this->ID,                    // Have the widget settings changed ?
 554                  'set_coll_ID' => $Blog->ID            // Have the settings of the blog changed ? (ex: new owner, new skin)
 555              );
 556  
 557          switch( $this->disp_params['link_type'] )
 558          {
 559              case 'login':          /* This one will probably abort caching by itself anyways */
 560              case 'register':    /* This one will probably abort caching by itself anyways */
 561              case 'profile':        // This can be cached
 562              case 'avatar':
 563                  // This link also depends on whether or not someone is logged in:
 564                  $keys['loggedin'] = (is_logged_in() ? 1 : 0);
 565  
 566          }
 567  
 568          return $keys;
 569      }
