  97  	function get_param_definitions( $params )

 101          $r = array_merge( array(

 105                      'type' => 'select',

 108                      'onchange' => '

 183      /**
 184       * Display the widget!
 185       *
 186       * @param array MUST contain at least the basic display params
 187       */
 188  	function display( $params )
 189      {
 190          global $current_User, $unread_messages_count;
 191          global $disp;
 192  
 193          $this->init_display( $params );
 194  
 195          $blog_ID = intval( $this->disp_params['blog_ID'] );
 196          if( $blog_ID > 0 )
 197          { // Try to use blog from widget setting
 198              $BlogCache = & get_BlogCache();
 199              $current_Blog = & $BlogCache->get_by_ID( $blog_ID, false, false );
 200          }
 201  
 202          if( empty( $current_Blog ) )
 203          { // Blog is not defined in setting or it doesn't exist in DB
 204              global $Blog;
 205              // Use current blog
 206              $current_Blog = & $Blog;
 207          }
 208  
 209          switch( $this->disp_params['show_to'] )
 210          {
 211              case 'any':
 212                  break;
 213              case 'loggedin':
 214                  if( !is_logged_in() )
 215                  {
 216                      return false;
 217                  }
 218                  break;
 219              case 'perms':
 220                  if( !is_logged_in() || !$current_User->check_perm( 'perm_messaging', 'reply', false ) )
 221                  {
 222                      return false;
 223                  }
 224                  break; // display
 225              case 'default':
 226                  debug_die( 'Invalid params!' );
 227          }
 228  
 229          // Default link class
 230          $link_class = $this->disp_params['link_default_class'];
 231  
 232          switch( $this->disp_params[ 'link_type' ] )
 233          {
 234              case 'messages':
 235                  $url = $current_Blog->get( 'threadsurl' );
 236                  $text = T_( 'Messages' );
 237                  // set allow blockcache to 0, this way make sure block cache is never allowed for messages
 238                  $this->disp_params[ 'allow_blockcache' ] = 0;
 239                  // Is this the current display?
 240                  if( $disp == 'threads' )
 241                  { // The current page is currently displaying the messages:
 242                      // Let's display it as selected
 243                      $link_class = $this->disp_params['link_selected_class'];
 244                  }
 245                  break;
 246  
 247              case 'contacts':
 248                  $url = $current_Blog->get( 'contactsurl' );
 249                  $text = T_( 'Contacts' );
 250                  // set show badge to 0, this way make sure badge won't be displayed
 251                  $this->disp_params[ 'show_badge' ] = 0;
 252                  // Is this the current display?
 253                  if( $disp == 'contacts' )
 254                  { // The current page is currently displaying the contacts:
 255                      // Let's display it as selected
 256                      $link_class = $this->disp_params['link_selected_class'];
 257                  }
 258                  break;
 259          }
 260  
 261          if( !empty( $this->disp_params[ 'link_text' ] ) )
 262          {
 263              $text = $this->disp_params[ 'link_text' ];
 264          }
 265  
 266          if( ( $this->disp_params[ 'show_badge' ] ) && ( $unread_messages_count > 0 ) )
 267          {
 268              $badge = ' <span class="badge">'.$unread_messages_count.'</span>';
 269              if( isset( $this->BlockCache ) )
 270              { // Do not cache if bage is displayed because the number of unread messages are always changing
 271                  $this->BlockCache->abort_collect();
 272              }
 273          }
 274          else
 275          {
 276              $badge = '';
 277          }
 278  
 279          echo $this->disp_params['block_start'];
 280          echo $this->disp_params['block_body_start'];
 281          echo $this->disp_params['list_start'];
 282  
 283          if( $link_class == $this->disp_params['link_selected_class'] )
 284          {
 285              echo $this->disp_params['item_selected_start'];
 286          }
 287          else
 288          {
 289              echo $this->disp_params['item_start'];
 290          }
 291          echo '<a href="'.$url.'" class="'.$link_class.'">'.$text.$badge.'</a>';
 292          if( $link_class == $this->disp_params['link_selected_class'] )
 293          {
 294              echo $this->disp_params['item_selected_end'];
 295          }
 296          else
 297          {
 298              echo $this->disp_params['item_end'];
 299          }
 300  
 301          echo $this->disp_params['list_end'];
 302          echo $this->disp_params['block_body_end'];
 303          echo $this->disp_params['block_end'];
 304  
 305          return true;
 306      }
