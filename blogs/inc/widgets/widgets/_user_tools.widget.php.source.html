 252      /**
 253       * Display the widget!
 254       *
 255       * @param array MUST contain at least the basic display params
 256       */
 257  	function display( $params )
 258      {
 259          if( !is_logged_in() )
 260          {    // Only logged in users can see this tools panel
 261              return false;
 262          }
 263  
 264          $this->init_display( $params ); // just in case it hasn't been done before
 265  
 266          $this->disp_params['item_start'] .= '<strong>';
 267          $this->disp_params['item_end'] = '</strong>'.$this->disp_params['item_end'];
 268  
 269          $tools_links = '';
 270          if( $this->get_param('user_postnew_link_show') )
 271          {    // Write new post - disp=edit
 272              $tools_links .= get_item_new_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_postnew_link' ] );
 273          }
 274          if( $this->get_param('user_messaging_link_show') )
 275          {    // Messaging - disp=threads
 276              $tools_links .= get_user_messaging_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_messaging_link' ], $this->disp_params[ 'user_messaging_link' ], $this->disp_params[ 'show_badge' ] );
 277          }
 278          if( $this->get_param('user_contacts_link_show') )
 279          {    // Contacts - disp=contacts
 280              $tools_links .= get_user_contacts_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_contacts_link' ], $this->disp_params[ 'user_contacts_link' ] );
 281          }
 282          if( $this->get_param('user_view_link_show') )
 283          {    // See profile - disp=user
 284              $tools_links .= get_user_tab_link( 'user', $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_view_link' ], $this->disp_params[ 'user_view_link' ] );
 285          }
 286          if( $this->get_param('user_profile_link_show') )
 287          {    // Edit profile - disp=profile
 288              $tools_links .= get_user_profile_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_profile_link' ], $this->disp_params[ 'user_profile_link' ] );
 289          }
 290          if( $this->get_param('user_picture_link_show') )
 291          {    // Edit picture - disp=avatar
 292              $tools_links .= get_user_tab_link( 'avatar', $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_picture_link' ], $this->disp_params[ 'user_picture_link' ] );
 293          }
 294          if( $this->get_param('user_password_link_show') )
 295          {    // Edit password - disp=pwdchange
 296              $tools_links .= get_user_tab_link( 'pwdchange', $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_password_link' ], $this->disp_params[ 'user_password_link' ] );
 297          }
 298          if( $this->get_param('user_preferences_link_show') )
 299          {    // Edit preferences - disp=userprefs
 300              $tools_links .= get_user_tab_link( 'userprefs', $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_preferences_link' ], $this->disp_params[ 'user_preferences_link' ] );
 301          }
 302          if( $this->get_param('user_subs_link_show') )
 303          {    // Edit notifications - disp=subs
 304              $tools_links .= get_user_subs_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_subs_link' ], $this->disp_params[ 'user_subs_link' ] );
 305          }
 306          if( $this->get_param('user_admin_link_show') )
 307          {    // Admin
 308              $tools_links .= get_user_admin_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_admin_link' ] );
 309          }
 310          if( $this->get_param('user_logout_link_show') )
 311          {    // Logout
 312              $tools_links .= get_user_logout_link( $this->disp_params['item_start'], $this->disp_params['item_end'], $this->disp_params[ 'user_logout_link' ] );
 313          }
 314  
 315          if( empty( $tools_links ) )
 316          {    // No available links to display
 317              return false;
 318          }
 319  
 320          // User tools:
 321          echo $this->disp_params['block_start'];
 322  
 323          if( !empty( $this->disp_params['title'] ) )
 324          {    // Display title
 325              echo $this->disp_params['block_title_start'];
 326              echo $this->disp_params['title'];
 327              echo $this->disp_params['block_title_end'];
 328          }
 329  
 330          echo $this->disp_params['block_body_start'];
 331  
 332          echo $this->disp_params['list_start'];
 333  
 334          echo $tools_links;
 335  
 336          if( isset($this->BlockCache) )
 337          {    // Do NOT cache because some of these links are using a redirect_to param, which makes it page dependent.
 338              // Note: also beware of the source param.
 339              // so this will be cached by the PageCache; there is no added benefit to cache it in the BlockCache
 340              // (which could have been shared between several pages):
 341              $this->BlockCache->abort_collect();
 342          }
 343  
 344          echo $this->disp_params['list_end'];
 345  
 346          echo $this->disp_params['block_body_end'];
 347  
 348          echo $this->disp_params['block_end'];
 349      }
