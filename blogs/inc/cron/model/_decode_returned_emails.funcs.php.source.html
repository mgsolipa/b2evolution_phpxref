  48  function dre_connect()
  49  {
  50      if( !extension_loaded( 'imap' ) )
  51      {    // Exit here if imap extension is not loaded
  52          dre_msg('<b class="red">IMAP extension is NOT loaded!</b>');
  53          return false;
  54      }
  55

  57

  59      $mailserver = '{'.$host;
  60  
  61      dre_msg('Connecting and authenticating to mail server <b>'.$host.'</b>');
  62  
  63      switch( $Settings->get('repath_encrypt') )
  64      {
  65          case 'ssl':
  66              $mailserver .= '/ssl';
  67              break;
  68  
  69          case 'tls':
  70              $mailserver .= '/tls';
  71              break;
  72  
  73          case 'none':
  74          default:
  75              $mailserver .= '/notls';
  76              break;
  77      }
  78  
  79      switch( $Settings->get('repath_method') )
  80      {
  81          case 'pop3':
  82          case 'pop3a':
  83              $mailserver .= '/pop3';
  84              break;
  85  
  86          case 'imap':
  87          default:
  88              // imap needs no additional options

  89              break;
  90      }
  91  
  92      if( $Settings->get('repath_novalidatecert') )
  93      {
  94          $mailserver .= '/novalidate-cert';
  95      }
  96  
  97      $mailserver .= '}INBOX';
  98  
  99      // Connect to mail server (one retry)

 100      $mbox = @imap_open( $mailserver, $Settings->get('repath_username'), $Settings->get('repath_password'), NULL, 1 );
 101  
 102      if( is_null(@get_resource_type($mbox)) )
 103      {    // Not a resource
 104          $error = imap_errors();
 105          if( is_array($error) )
 106          {
 107              $error = implode( "<br />\n", $error );
 108          }
 109  
 110          dre_msg( sprintf( /* TRANS: %s is the error message */ T_('Connection failed: %s'), $error), true );
 111          return false;
 112      }
 113      dre_msg('<b class="green">Successfully connected!</b>');
 114  
 115      @imap_errors();
 116

 121  /**

 122   * Read messages from server and save returned emails into DB

 123   *

 124   * @param resource $mbox created by dre_connect() (by reference)

 125   * @param integer the number of messages to process

 126   * @return boolean true on success

 127   */
 128  function dre_process_messages( & $mbox, $limit )
 129  {
 130      //return; // Exit, in development...

 131  
 132  
 133      global $Settings;
 134      global $dre_messages, $dre_emails, $email_cntr, $del_cntr, $is_cron_mode;
 135  
 136      // No execution time limit

 137      set_max_execution_time(0);
 138  
 139      $email_cntr = 0;
 140      $del_cntr = 0;
 141      for( $index = 1; $index <= $limit; $index++ )
 142      {
 143          dre_msg('<hr /><h3>Processing message #'.$index.':</h3>');
 144  
 145          $html_body = '';
 146          $strbody = '';
 147          $hasAttachment = false;
 148          $hasRelated = false;
 149  
 150          // Save email to hard drive, otherwise attachments may take a lot of RAM

 151          if( ! ($tmpMIME = tempnam( sys_get_temp_dir(), 'b2evoMail' )) )
 152          {
 153              dre_msg( T_('Could not create temporary file.'), true );
 154              continue;
 155          }
 156          imap_savebody( $mbox, $tmpMIME, $index );
 157  
 158          // Create random temp directory for message parts

 159          $tmpDirMIME = dre_tempdir( sys_get_temp_dir(), 'b2evo_' );
 160  
 161          $mimeParser = new mime_parser_class;
 162          $mimeParser->mbox = 0;                // Set to 0 for parsing a single message file

 163          $mimeParser->decode_headers = 1;
 164          $mimeParser->ignore_syntax_errors = 1;
 165          $mimeParser->extract_addresses = 0;
 166  
 167          $MIMEparameters = array(
 168                  'File' => $tmpMIME,
 169                  'SaveBody' => $tmpDirMIME,    // Save message body parts to a directory
 170                  'SkipBody' => 1,            // Do not retrieve or save message body parts
 171              );
 172  
 173          if( !$mimeParser->Decode( $MIMEparameters, $decodedMIME ) )
 174          {
 175              dre_msg( sprintf( 'MIME message decoding error: %s at position %d.', $mimeParser->error, $mimeParser->error_position), true );
 176              rmdir_r( $tmpDirMIME );
 177              unlink( $tmpMIME );
 178              continue;
 179          }
 180          else
 181          {
 182              dre_msg('MIME message decoding successful');
 183  
 184              if( ! $mimeParser->Analyze( $decodedMIME[0], $parsedMIME ) )
 185              {
 186                  dre_msg( sprintf( 'MIME message analyse error: %s', $mimeParser->error), true );
 187                  rmdir_r( $tmpDirMIME );
 188                  unlink( $tmpMIME );
 189                  continue;
 190              }
 191  
 192              // Get message $subject and $post_date from headers (by reference)

 193              if( ! dre_process_header( $parsedMIME, $subject, $post_date ) )
 194              {    // Couldn't process message headers
 195                  rmdir_r( $tmpDirMIME );
 196                  unlink($tmpMIME);
 197                  continue;
 198              }
 199  
 200              // TODO: handle type == "message" recursively

 201              // sam2kb> For some reason imap_qprint() demages HTML text... needs more testing

 202  
 203              if( $parsedMIME['Type'] == 'html' )
 204              {    // Mail is HTML
 205                  dre_msg( 'HTML message part saved as '.$parsedMIME['DataFile'] );
 206                  $html_body = file_get_contents($parsedMIME['DataFile']);
 207  
 208                  foreach( $parsedMIME['Alternative'] as $alternative )
 209                  {    // First try to get HTML alternative (when possible)
 210                      if( $alternative['Type'] == 'html' )
 211                      {    // HTML text
 212                          dre_msg('HTML alternative message part saved as '.$alternative['DataFile']);
 213                          // sam2kb> TODO: we may need to use $html_body here instead

 214                          $strbody = file_get_contents($alternative['DataFile']);
 215                          break; // stop after first alternative

 216                      }
 217                      elseif( $alternative['Type'] == 'text' )
 218                      {    // Plain text
 219                          dre_msg('Text alternative message part saved as '.$alternative['DataFile']);
 220                          $strbody = imap_qprint( file_get_contents($alternative['DataFile']) );
 221                          break; // stop after first alternative

 222                      }
 223                  }
 224              }
 225              elseif( $parsedMIME['Type'] == 'text' )
 226              {    // Mail is plain text
 227                  dre_msg('Plain-text message part saved as '.$parsedMIME['DataFile']);
 228                  $strbody = imap_qprint( file_get_contents($parsedMIME['DataFile']) );
 229              }
 230              elseif( $parsedMIME['Type'] == 'delivery-status' )
 231              {    // Mail is delivery-status
 232                  $strbody = '';
 233                  foreach( $decodedMIME[0]['Parts'] as $part )
 234                  {
 235                      $strbody .= imap_qprint( file_get_contents( $part['BodyFile'] ) );
 236                  }
 237              }
 238  
 239              if( count($mimeParser->warnings) > 0 )
 240              {
 241                  dre_msg( sprintf('<h4>%d warnings during decode:</h4>', count($mimeParser->warnings)) );
 242                  foreach( $mimeParser->warnings as $k => $v )
 243                  {
 244                      dre_msg('Warning: '.$v.' at position '.$k);
 245                  }
 246              }
 247          }
 248          unlink( $tmpMIME );
 249  
 250          if( empty($html_body) )
 251          {    // Plain text message
 252              dre_msg('Message type: TEXT');
 253              dre_msg('Message body: <pre style="font-size:10px">'.evo_htmlspecialchars($strbody).'</pre>');
 254  
 255              // Process body. First fix different line-endings (dos, mac, unix), remove double newlines

 256              $content = str_replace( array("\r", "\n\n"), "\n", trim($strbody) );
 257          }
 258          else
 259          {    // HTML message
 260              dre_msg('Message type: HTML');
 261  
 262              if( ($parsed_message = dre_prepare_html_message( $html_body )) === false )
 263              {    // No 'auth' tag provided, skip to the next message
 264                  rmdir_r( $tmpDirMIME );
 265                  continue;
 266              }
 267              list($auth, $content) = $parsed_message;
 268          }
 269  
 270          dre_msg('<b class="green">Success</b>');
 271  
 272          $message_text = $content;
 273  
 274          // Remove content after terminators

 275          $content = dre_limit_by_terminators( $content );
 276  
 277          global $Messages;
 278          if( $Messages->has_errors() )
 279          {
 280              // Make it easier for user to find and correct the errors

 281              dre_msg( "\n".sprintf( T_('Processing message: %s'), $post_title ), true );
 282              dre_msg( $Messages->get_string( T_('Cannot post, please correct these errors:'), 'error' ), true );
 283  
 284              $Messages->clear();
 285              rmdir_r( $tmpDirMIME );
 286              continue;
 287          }
 288  
 289          global $dre_emails, $DB, $localtimenow;
 290  
 291          dre_msg( sprintf('<h4>Saving the returned email in the database</h4>' ) );
 292  
 293          // Insert a returned email's data into DB

 294          if( $returned_email = dre_insert_returned_email( $content, $message_text, dre_get_headers( $decodedMIME ) ) )
 295          {
 296              dre_msg( 'Error Type: '.dre_decode_error_type( $returned_email['errtype'] ) );
 297              dre_msg( 'Error Message: '.$returned_email['errormsg'] );
 298  
 299              ++$email_cntr;
 300          }
 301  
 302          // Delete temporary directory

 303          rmdir_r( $tmpDirMIME );
 304  
 305          if( $Settings->get('repath_delete_emails') )
 306          {
 307              dre_msg( 'Marking message for deletion from inbox: '.$index );
 308              imap_delete( $mbox, $index );
 309              ++$del_cntr;
 310          }
 311      }
 312  
 313      // Expunge messages market for deletion

 314      imap_expunge( $mbox );
 315  
 316      return true;
 317  }

 378  /**

 379   * Process Header information like subject and date of a mail.

 380   *

 381   * @param array $header header as set by mime_parser_class::Analyze()

 382   * @param string message subject by reference

 383   * @param string message date by reference

 384   * @return bool true if valid subject prefix is detected

 385   */
 386  function dre_process_header( $header, & $subject, & $post_date )
 387  {
 388      global $Settings;
 389  
 390      $subject = $header['Subject'];
 391      $ddate = $header['Date'];
 392  
 393      dre_msg('Subject: '.$subject);
 394  
 395      // Check subject to match in titles to identify return path emails

 396      $subject_is_correct = false;
 397      $repath_subjects = explode( "\n", str_replace( array( '\r\n', '\n\n' ), '\n', $Settings->get( 'repath_subject' ) ) );
 398      foreach( $repath_subjects as $repath_subject )
 399      {
 400          if( strpos( $subject, $repath_subject ) !== false )
 401          {
 402              $subject_is_correct = true;
 403              break;
 404          }
 405      }
 406  
 407      if( !$subject_is_correct )
 408      {    // Subject is not match to identify return email
 409          dre_msg( 'Subject prefix is not  "'.implode( '", "', $repath_subjects ).'", skip this email' );
 410          return false;
 411      }
 412  
 413      // Parse Date

 414      if( !preg_match('#^(.{3}, )?(\d{2}) (.{3}) (\d{4}) (\d{2}):(\d{2}):(\d{2})#', $ddate, $match) )
 415      {
 416          $ddate_U = @strtotime($ddate);
 417          if( empty($ddate_U) || strlen($ddate_U) < 2 )
 418          {
 419              dre_msg( sprintf( T_('Could not parse date header "%s"'), $ddate ), true );
 420              return false;
 421          }
 422      }
 423  
 424      if( empty($ddate_U) )
 425      {
 426          $dmonths = array(
 427              'Jan' => 1,
 428              'Feb' => 2,
 429              'Mar' => 3,
 430              'Apr' => 4,
 431              'May' => 5,
 432              'Jun' => 6,
 433              'Jul' => 7,
 434              'Aug' => 8,
 435              'Sep' => 9,
 436              'Oct' => 10,
 437              'Nov' => 11,
 438              'Dec' => 12,
 439          );
 440  
 441          $ddate_H = $match[5];
 442          $ddate_i = $match[6];
 443          $ddate_s = $match[7];
 444  
 445          if( ! isset( $dmonths[$match[3]] ) )
 446          {
 447              dre_msg( T_('Invalid month name in message date string.'), true );
 448              return false;
 449          }
 450          $ddate_m = $dmonths[$match[3]];
 451          $ddate_d = $match[2];
 452          $ddate_Y = $match[4];
 453  
 454          $ddate_U = mktime( $ddate_H, $ddate_i, $ddate_s, $ddate_m, $ddate_d, $ddate_Y );
 455      }
 456  
 457      $post_date = date( 'Y-m-d H:i:s', $ddate_U );
 458  
 459      return true;
 460  }

 463  /**

 464   * Extract emails from a message body

 465   *

 466   * @param string Message body

 467   * @param integer Max count emails

 468   * @param string Delimeter between emails

 469   * @return string Emails separated by delimeter

 470   */
 471  function dre_get_emails( $content, $max_count = 1, $delimeter = ', ' )
 472  {
 473      if( preg_match_all( '/([a-z0-9&\-_.]+?@[\w\-]+\.([\w\-\.]+\.)?[\w]+)/is', $content, $emails ) )
 474      {    // Get the returned emails
 475          global $Settings;
 476  
 477          $emails = array_unique( $emails[1] );
 478          foreach( $emails as $e => $email )
 479          {
 480              if( in_array( $email, array( $Settings->get( 'notification_sender_email' ), $Settings->get( 'notification_return_path' ) ) ) )
 481              {    // Exclude Sender email & Return path
 482                  unset( $emails[$e] );
 483              }
 484          }
 485          // Limit by max count emails

 486          $emails = array_slice( $emails, 0, $max_count );
 487  
 488          return implode( $delimeter, $emails );
 489      }
 490  
 491      return '';
 492  }

 495  /**

 496   * Prepare html message

 497   *

 498   * @param string Message

 499   * @return string Content

 500   */
 501  function dre_prepare_html_message( $message )
 502  {
 503      dre_msg('Message body (original): <pre style="font-size:10px">'.evo_htmlspecialchars($message).'</pre>');
 504  
 505      $marker = 0;
 506      if( preg_match( '~<body[^>]*>(.*?)</body>~is', $message, $result ) )
 507      {    // First see if we can get contents of <body> tag
 508          $content = $result[1];
 509          $marker = 1;
 510      }
 511      elseif( preg_match( '~<html[^>]*>(.*?)</html>~is', $message, $result ) )
 512      {    // <body> was not found, use <html> contents and delete <head> section from it
 513          $content = preg_replace( '~<head[^>]*>(.*?)</head>~is', '', $result[1] );
 514          $marker = 1;
 515      }
 516  
 517      if( empty($marker) )
 518      {    // None of the above methods worked, just use the original message body
 519          $content = $message;
 520      }
 521  
 522      // First fix different line-endings (dos, mac, unix), remove double newlines

 523      $content = str_replace( array("\r", "\n\n"), "\n", trim($content) );
 524  
 525      // Decode 'category', 'title' and 'auth' tags

 526      $content = preg_replace( '~&lt;(/)?(category|title|auth)&gt;~i', '<\\1\\2>', $content );
 527  
 528      // Balance tags

 529      $content = balance_tags($content);
 530  
 531      // Remove markup that cause validator errors

 532      $patterns = array(
 533          '~ moz-do-not-send="true"~',            // Thunderbird inline image with absolute "src"
 534          '~ class="moz-signature" cols="\d+"~',    // Thunderbird signature in HTML message
 535          '~ goomoji="[^"]+"~',                    // Gmail smilies
 536      );
 537      $content = preg_replace( $patterns, '', $content );
 538  
 539      dre_msg('Message body (processed): <pre style="font-size:10px">'.evo_htmlspecialchars($content).'</pre>');
 540  
 541      return $content;
 542  }

 567  /**

 568   * Get content after email and before terminator line

 569   *

 570   * @param mixed $message

 571   * @param mixed $emails

 572   * @return string Content

 573   */
 574  function dre_get_processing_content( $content, $emails )
 575  {
 576      $error_text = '';
 577  
 578      $emails = explode( ', ', $emails );
 579      if( count( $emails ) > 0 )
 580      {    // If emails exist
 581          // Get last email

 582          $email = $emails[ count( $emails ) - 1 ];
 583          if( !empty( $email ) )
 584          {    // Get error text after last email address
 585              $error_text = trim( evo_substr( $content, evo_strpos( $content, $email ) + evo_strlen( $email ) ) );
 586          }
 587          if( empty( $error_text ) )
 588          {    // If error text is empty we should get all content before email OR full content if no email address in content
 589              $error_text = empty( $email ) ? $content : trim( evo_substr( $content, 0, evo_strpos( $content, $email ) ) );
 590          }
 591      }
 592      else
 593      {    // If no emails - get full content as error text
 594          $error_text = $content;
 595      }
 596  
 597      if( !empty( $error_text ) )
 598      {    // Replace all new line sumbols with space symbol
 599          $error_text = str_replace( array( "\r\n\r\n", "\r\n", "\n\n", "\n" ), " ", $error_text );
 600      }
 601  
 602      return $error_text;
 603  }

 629  /**

 630   * Get Error from Message by defined patterns ($Settings->get( 'repath_errtype' ))

 631   *

 632   * @param string Content

 633   * @return array ( 'Error Type', 'Error Mask' )

 634   */
 635  function dre_get_error_by_pattern( $content )
 636  {
 637      global $Settings;
 638  
 639      $error_types = trim( $Settings->get( 'repath_errtype' ) );
 640  
 641      if( empty( $error_types ) )
 642      {    // Error types are not defined
 643          return false;
 644      }
 645  
 646      $error_types = explode( "\n", str_replace( array( "\r", "\n\n" ), "\n", $error_types ) );
 647  
 648      foreach( $error_types as $error_type )
 649      {
 650          list( $error_type, $error_mask ) = explode( ' ', $error_type, 2 );
 651          if( preg_match( '#'.$error_mask.'#i', $content ) )
 652          {
 653              return array( $error_type, $error_mask );
 654          }
 655      }
 656  
 657      // Not found error

 658      return false;
 659  }

 840  /**

 841   * Insert/Update a blocked email's data into DB

 842   *

 843   * @param array Data of returned email:

 844   *               'address'

 845   *               'errormsg'

 846   *               'message'

 847   *               'headers'

 848   *               'errtype'

 849   */
 850  function dre_save_blocked_email( $email_returned )
 851  {
 852      global $DB;
 853  
 854      if( empty( $email_returned['address'] ) )
 855      {    // No emails, Exit here
 856          return;
 857      }
 858  
 859      $EmailAddressCache = & get_EmailAddressCache();
 860      // Get an existing email address to update if it exist

 861      $EmailAddress = & $EmailAddressCache->get_by_name( $email_returned['address'], false );
 862      if( !$EmailAddress )
 863      {    // Insert new email address
 864          $EmailAddress = new EmailAddress();
 865          $EmailAddress->set( 'address', $email_returned['address'] );
 866      }
 867  
 868      switch( $email_returned['errtype'] )
 869      {    // Error type of the returned email:
 870          case 'P':    // Permanent error
 871              $EmailAddress->increase_counter( 'prmerror' );
 872              // Update only the adresses with NOT spammer statuses

 873              $EmailAddress->set_status( 'prmerror' );
 874              break;
 875  
 876          case 'T':    // Temporary error
 877              if( in_array( $EmailAddress->get( 'status' ), array( 'suspicious1', 'suspicious2', 'suspicious3' ) ) )
 878              { // If current status already is defined as 'suspicious1', 'suspicious2' or 'suspicious3'
 879                  if( $EmailAddress->get( 'sent_last_returnerror' ) <= 1 )
 880                  {
 881                      if( $EmailAddress->get( 'status' ) == 'suspicious1' )
 882                      {    // Increase status from suspicious1 to suspicious2
 883                          $EmailAddress->set( 'status', 'suspicious2' );
 884                      }
 885                      elseif( $EmailAddress->get( 'status' ) == 'suspicious2' )
 886                      {    // Increase status from suspicious2 to suspicious3
 887                          $EmailAddress->set( 'status', 'suspicious3' );
 888                      }
 889                  }
 890              }
 891              elseif( $EmailAddress->get( 'status' ) == 'redemption' )
 892              { // IF current status is 'redemption' we should set it as 'suspicious3'
 893                  $EmailAddress->set_status( 'suspicious3' );
 894              }
 895              else
 896              { // Update only the email addresses with level status less then Suspicious 1
 897                  $EmailAddress->set_status( 'suspicious1' );
 898              }
 899              $EmailAddress->increase_counter( 'tmperror' );
 900              break;
 901  
 902          case 'S':    // Spam suspicion
 903              $EmailAddress->increase_counter( 'spamerror' );
 904              // Update only the email addresses with 'unknown' status

 905              $EmailAddress->set_status( 'warning' );
 906              break;
 907  
 908          default:    // Other errors
 909              $EmailAddress->increase_counter( 'othererror' );
 910              // Update only the email addresses with 'unknown' status

 911              $EmailAddress->set_status( 'warning' );
 912              break;
 913      }
 914  
 915      // Insert/Update an email address

 916      $EmailAddress->dbsave();
 917  }
