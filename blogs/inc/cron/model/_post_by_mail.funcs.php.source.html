  45   * Connect to a mail server

  48   * @return resource $mbox

  50  function pbm_connect()
  51  {
  52      if( !extension_loaded( 'imap' ) )
  53      {    // Exit here if imap extension is not loaded
  54          pbm_msg('<b class="red">IMAP extension is NOT loaded!</b>');
  55          return false;
  56      }
  57  
  58      global $Settings;
  59  
  60      $host = $Settings->get('eblog_server_host').':'.$Settings->get('eblog_server_port');
  61      $mailserver = '{'.$host;
  62  
  63      pbm_msg('Connecting and authenticating to mail server <b>'.$host.'</b>');
  64  
  65      switch( $Settings->get('eblog_encrypt') )
  66      {
  67          case 'ssl':
  68              $mailserver .= '/ssl';
  69              break;
  70  
  71          case 'tls':
  72              $mailserver .= '/tls';
  73              break;
  74  
  75          case 'none':
  76          default:
  77              $mailserver .= '/notls';
  78              break;
  79      }
  80  
  81      switch( $Settings->get('eblog_method') )
  82      {
  83          case 'pop3':
  84          case 'pop3a':
  85              $mailserver .= '/pop3';
  86              break;
  87  
  88          case 'imap':
  89          default:
  90              // imap needs no additional options
  91              break;
  92      }
  93  
  94      if( $Settings->get('eblog_novalidatecert') )
  95      {
  96          $mailserver .= '/novalidate-cert';
  97      }
  98  
  99      $mailserver .= '}INBOX';
 100  
 101      // Connect to mail server (one retry)
 102      $mbox = @imap_open( $mailserver, $Settings->get('eblog_username'), $Settings->get('eblog_password'), NULL, 1 );
 103  
 104      if( is_null(@get_resource_type($mbox)) )
 105      {    // Not a resource
 106          $error = imap_errors();
 107          if( is_array($error) )
 108          {
 109              $error = implode( "<br />\n", $error );
 110          }
 111  
 112          pbm_msg( sprintf( /* TRANS: %s is the error message */ T_('Connection failed: %s'), $error), true );
 113          return false;
 114      }
 115      pbm_msg('<b class="green">Successfully connected!</b>');
 116  
 117      @imap_errors();
 118  
 119      return $mbox;
 120  }

 558  /**
 559   * Process Header information like subject and date of a mail.
 560   *
 561   * @param array $header header as set by mime_parser_class::Analyze()
 562   * @param string message subject by reference
 563   * @param string message date by reference
 564   * @return bool true if valid subject prefix is detected
 565   */
 566  function pbm_process_header( $header, & $subject, & $post_date )
 567  {
 568      global $Settings;
 569  
 570      $subject = $header['Subject'];
 571      $ddate = $header['Date'];
 572  
 573      $prefix = $Settings->get( 'eblog_subject_prefix' );
 574      pbm_msg('Subject: '.$subject);
 575  
 576      if( evo_substr($subject, 0, evo_strlen($prefix)) !== $prefix )
 577      {
 578          pbm_msg('Subject prefix is not "'.$prefix.'", skip this email');
 579          return false;
 580      }
 581  
 582      $subject = evo_substr($subject, evo_strlen($prefix));
 583  
 584      // Parse Date
 585      if( !preg_match('#^(.{3}, )?(\d{2}) (.{3}) (\d{4}) (\d{2}):(\d{2}):(\d{2})#', $ddate, $match) )
 586      {
 587          $ddate_U = @strtotime($ddate);
 588          if( empty($ddate_U) || strlen($ddate_U) < 2 )
 589          {
 590              pbm_msg( sprintf( T_('Could not parse date header "%s"'), $ddate ), true );
 591              return false;
 592          }
 593      }
 594  
 595      if( empty($ddate_U) )
 596      {
 597          $dmonths = array(
 598              'Jan' => 1,
 599              'Feb' => 2,
 600              'Mar' => 3,
 601              'Apr' => 4,
 602              'May' => 5,
 603              'Jun' => 6,
 604              'Jul' => 7,
 605              'Aug' => 8,
 606              'Sep' => 9,
 607              'Oct' => 10,
 608              'Nov' => 11,
 609              'Dec' => 12,
 610          );
 611  
 612          $ddate_H = $match[5];
 613          $ddate_i = $match[6];
 614          $ddate_s = $match[7];
 615  
 616          if( ! isset( $dmonths[$match[3]] ) )
 617          {
 618              pbm_msg( T_('Invalid month name in message date string.'), true );
 619              return false;
 620          }
 621          $ddate_m = $dmonths[$match[3]];
 622          $ddate_d = $match[2];
 623          $ddate_Y = $match[4];
 624  
 625          $ddate_U = mktime( $ddate_H, $ddate_i, $ddate_s, $ddate_m, $ddate_d, $ddate_Y );
 626      }
 627  
 628      $post_date = date( 'Y-m-d H:i:s', $ddate_U );
 629  
 630      return true;
 631  }

 634  /**
 635   * process attachments by saving into media directory and optionally creating image tag in post
 636   *
 637   * @param string message content that is optionally manipulated by adding image tags (by reference)
 638   * @param  array $mailAttachments array containing path to attachment files
 639   * @param  string $mediadir path to media directory of blog as seen by file system
 640   * @param  string $media_url url to media directory as seen by user
 641   * @param  bool $add_img_tags should img tags be added to the post (instead of linking through the file manager)
 642   * @param  string $type defines attachment type: 'attach' or 'related'
 643   */
 644  function pbm_process_attachments( & $content, $mailAttachments, $mediadir, $media_url, $add_img_tags = true, $type = 'attach' )
 645  {
 646      global $Settings, $pbm_item_files, $filename_max_length;
 647  
 648      pbm_msg('<h4>Processing attachments</h4>');
 649  
 650      foreach( $mailAttachments as $attachment )
 651      {
 652          if( isset($attachment['FileName']) )
 653          {
 654              $filename = trim( evo_strtolower($attachment['FileName']) );
 655          }
 656          else
 657          {    // Related attachments may not have file name, we'll generate one below
 658              $filename = '';
 659          }
 660  
 661          if( $filename == '' )
 662          {
 663              $filename = 'upload_'.uniqid().'.'.$attachment['SubType'];
 664              pbm_msg( sprintf('Attachment without name. Using "%s".', evo_htmlspecialchars($filename)) );
 665          }
 666  
 667          // Check valid filename/extension: (includes check for locked filenames)
 668          if( $error_filename = process_filename( $filename, true ) )
 669          {
 670              pbm_msg('Invalid filename: '.$error_filename);
 671              continue;
 672          }
 673  
 674          // If file exists count up a number
 675          $cnt = 0;
 676          $prename = substr( $filename, 0, strrpos( $filename, '.' ) ).'-';
 677          $sufname = strrchr( $filename, '.' );
 678          $error_in_filename = false;
 679  
 680          while( file_exists( $mediadir.$filename ) )
 681          {
 682              $filename = $prename.$cnt.$sufname;
 683              if( strlen( $filename ) > $filename_max_length )
 684              { // This is a special case, when the filename is longer then the maximum allowed
 685                  // Cut as many characters as required before the counter on the file name
 686                  $filename = fix_filename_length( $filename, strlen( $prename ) - 1 );
 687                  if( $error_in_filename = process_filename( $filename, true ) )
 688                  { // The file name is not valid, this is an unexpected situation, because the file name was already validated before
 689                      pbm_msg('Invalid filename: '.$error_filename);
 690                      break;
 691                  }
 692              }
 693              ++$cnt;
 694          }
 695          if( $error_in_filename )
 696          { // Don't create file with invalid file name
 697              continue;
 698          }
 699          pbm_msg( sprintf('New file name is <b>%s</b>', $filename) );
 700  
 701          $imginfo = NULL;
 702          if( ! $Settings->get('eblog_test_mode') )
 703          {
 704              pbm_msg( 'Saving file to: '.evo_htmlspecialchars($mediadir.$filename) );
 705              if( !copy( $attachment['DataFile'], $mediadir.$filename ) )
 706              {
 707                  pbm_msg( 'Unable to copy uploaded file to '.evo_htmlspecialchars($mediadir.$filename) );
 708                  continue;
 709              }
 710  
 711              // chmod uploaded file:
 712              $chmod = $Settings->get('fm_default_chmod_file');
 713              @chmod( $mediadir.$filename, octdec( $chmod ) );
 714  
 715              $imginfo = @getimagesize($mediadir.$filename);
 716              pbm_msg( 'Is this an image?: '.(is_array($imginfo) ? 'yes' : 'no') );
 717          }
 718  
 719          if( $type == 'attach' )
 720          {
 721              $content .= "\n";
 722              if( is_array($imginfo) && $add_img_tags )
 723              {
 724                  $content .= '<img src="'.$media_url.$filename.'" '.$imginfo[3].' />';
 725              }
 726              else
 727              {
 728                  pbm_msg( sprintf('The file <b>%s</b> will be attached to the post later, after we save the post in the database.', $filename) );
 729                  $pbm_item_files[] = $filename;
 730              }
 731              $content .= "\n";
 732          }
 733          elseif( !empty($attachment['ContentID']) )
 734          {    // Replace relative "cid:xxxxx" URIs with absolute URLs to media files
 735              $content = str_replace( 'cid:'.$attachment['ContentID'], $media_url.$filename, $content );
 736          }
 737      }
 738  }

 775   * Extract <auth> tag with login and password
 776   *
 777   * @param string Message body (by reference)
 778   * @return array login and password
 779   */
 780  function pbm_get_auth_tag( & $content )
 781  {
 782      if( preg_match( '~<(auth)>(.+?)</\\1>~is', $content, $match ) )
 783      {
 784          // tblue> splitting only into 2 parts allows colons in the user PW
 785          // Note: login and password cannot include '<' !
 786          $auth = explode( ':', strip_tags($match[2]), 2 );
 787  
 788          // Delete 'auth' tag from post content
 789          $content = preg_replace( '~<(auth)>(.+?)</\\1>~is', '', $content );
 790

 797  function pbm_prepare_html_message( $message )
 798  {
 799      pbm_msg('Message body (original): <pre style="font-size:10px">'.evo_htmlspecialchars($message).'</pre>');
 800  
 801      $marker = 0;
 802      if( preg_match( '~<body[^>]*>(.*?)</body>~is', $message, $result ) )
 803      {    // First see if we can get contents of <body> tag
 804          $content = $result[1];
 805          $marker = 1;
 806      }
 807      elseif( preg_match( '~<html[^>]*>(.*?)</html>~is', $message, $result ) )
 808      {    // <body> was not found, use <html> contents and delete <head> section from it
 809          $content = preg_replace( '~<head[^>]*>(.*?)</head>~is', '', $result[1] );
 810          $marker = 1;
 811      }
 812  
 813      if( empty($marker) )
 814      {    // None of the above methods worked, just use the original message body
 815          $content = $message;
 816      }
 817  
 818      // First fix different line-endings (dos, mac, unix), remove double newlines
 819      $content = str_replace( array("\r", "\n\n"), "\n", trim($content) );
 820  
 821      // Decode 'category', 'title' and 'auth' tags
 822      $content = preg_replace( '~&lt;(/)?(category|title|auth)&gt;~i', '<\\1\\2>', $content );
 823  
 824      if( ($auth = pbm_get_auth_tag($content)) === false )
 825      {    // No 'auth' tag provided, exit
 826          pbm_msg( sprintf( T_('&lt;auth&gt; tag not found! Please add username and password in message body in format %s.'),
 827                      '"&lt;auth&gt;username:password&lt;/auth&gt;"' ), true );
 828          return false;
 829      }
 830  
 831      // Balance tags
 832      $content = balance_tags($content);
 833  
 834      // Remove markup that cause validator errors
 835      $patterns = array(
 836          '~ moz-do-not-send="true"~',            // Thunderbird inline image with absolute "src"
 837          '~ class="moz-signature" cols="\d+"~',    // Thunderbird signature in HTML message
 838          '~ goomoji="[^"]+"~',                    // Gmail smilies
 839      );
 840      $content = preg_replace( $patterns, '', $content );
 841  
 842      pbm_msg('Message body (processed): <pre style="font-size:10px">'.evo_htmlspecialchars($content).'</pre>');
 843  
 844      return array( $auth, $content );
 845  }

 848  function pbm_validate_user_password( $user_login, $user_pass )
 849  {
 850      $UserCache = & get_UserCache();
 851      $User = & $UserCache->get_by_login( $user_login );
 852      if( ! $User )
 853      {
 854          return false;
 855      }
 856  
 857      // First check unhashed password
 858      if( ! $User->check_password( $user_pass, false ) )
 859      {
 860          if( preg_match( '~^[a-f0-9]{32}$~i', $user_pass ) )
 861          {    // This is a hashed password, see if it's valid
 862              // We check it here because some crazy user may use a real 32-chars password!
 863              if( $User->check_password( $user_pass, true ) )
 864              {    // Valid password
 865                  return $User;
 866              }
 867          }
 868          return false;
 869      }
 870      return $User;
 871  }
