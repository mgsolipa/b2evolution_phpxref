  32  /**
  33   * Log a message from cron.
  34   * @param string Message
  35   * @param integer Level of importance. The higher the more important.
  36   *        (if $quiet (number of "-q" params passed to cron_exec.php)
  37   *         is higher than this, the message gets skipped)

  39  function cron_log( $message, $level = 0 )
  40  {
  41      global $is_web, $quiet;
  42  
  43      if( $quiet > $level )
  44      {
  45          return;
  46      }
  47  
  48      if( $is_web )
  49      {
  50          echo '<p>'.$message.'</p>';
  51      }
  52      else
  53      {
  54          echo "\n".$message."\n";
  55      }
  56  }

  59  /**
  60   * Call a cron job.
  61   *
  62   * @param string Name of the job
  63   * @param string Params for the job:
  64   *               'ctsk_ID'   - task ID
  65   *               'ctsk_name' - task name
  66   * @return string Error message
  67   */
  68  function call_job( $job_name, $job_params = array() )
  69  {
  70      global $DB, $inc_path, $Plugins, $admin_url;
  71  
  72      global $result_message, $result_status, $timestop, $time_difference;
  73  
  74      $error_message = '';
  75      $result_message = NULL;
  76      $result_status = 'error';
  77  
  78      apm_name_transaction( 'CRON/'.$job_name );
  79  
  80      if( preg_match( '~^plugin_(\d+)_(.*)$~', $job_name, $match ) )
  81      { // Cron job provided by a plugin:
  82          if( ! is_object($Plugins) )
  83          {
  84              load_class( 'plugins/model/_plugins.class.php', 'Plugins' );
  85              $Plugins = new Plugins();
  86          }
  87  
  88          $Plugin = & $Plugins->get_by_ID( $match[1] );
  89          if( ! $Plugin )
  90          {
  91              $result_message = 'Plugin for controller ['.$job_name.'] could not get instantiated.';
  92              cron_log( $result_message, 2 );
  93              return $result_message;
  94          }
  95  
  96          // CALL THE PLUGIN TO HANDLE THE JOB:
  97          $tmp_params = array( 'ctrl' => $match[2], 'params' => $job_params );
  98          $sub_r = $Plugins->call_method( $Plugin->ID, 'ExecCronJob', $tmp_params );
  99  
 100          $error_code = (int)$sub_r['code'];
 101          $result_message = $sub_r['message'];
 102      }
 103      else
 104      {
 105          $controller = $inc_path.$job_name;
 106          if( ! is_file( $controller ) )
 107          {
 108              $result_message = 'Controller ['.$job_name.'] does not exist.';
 109              cron_log( $result_message, 2 );
 110              return $result_message;
 111          }
 112  
 113          // INCLUDE THE JOB FILE AND RUN IT:
 114          $error_code = require $controller;
 115      }
 116  
 117      if( is_array( $result_message ) )
 118      {    // If result is array (we should store it as serialized data later)
 119          // array keys: 'message' - Result message
 120          //             'table_cols' - Columns names of the table to display on the Execution details of the cron job log
 121          //             'table_data' - Array
 122          $result_message_text = $result_message['message'];
 123      }
 124      else
 125      {    // Result is text string
 126          $result_message_text = $result_message;
 127      }
 128  
 129      if( $error_code != 1 )
 130      {    // We got an error
 131          $result_status = 'error';
 132          $result_message_text = '[Error code: '.$error_code.' ] '.$result_message_text;
 133          if( is_array( $result_message ) )
 134          { // If result is array
 135              $result_message['message'] = $result_message_text;
 136          }
 137          $cron_log_level = 2;
 138  
 139          $error_message = $result_message_text;
 140      }
 141      else
 142      {
 143          $result_status = 'finished';
 144          $cron_log_level = 1;
 145      }
 146  
 147      $timestop = time() + $time_difference;
 148      cron_log( 'Task finished at '.date( 'H:i:s', $timestop ).' with status: '.$result_status
 149          ."\nMessage: $result_message_text", $cron_log_level );
 150  
 151      return $error_message;
 152  }

 167              'error'    => 'FF0000',

 213  /**
 214   * Detect timed out cron jobs and Send notifications
 215   *
 216   * @param array Task with error
 217   *             'name'
 218   *             'message'
 219   */
 220  function detect_timeout_cron_jobs( $error_task = NULL )
 221  {
 222      global $DB, $time_difference, $cron_timeout_delay, $admin_url;
 223  
 224      $SQL = new SQL( 'Find cron timeouts' );
 225      $SQL->SELECT( 'ctsk_ID, ctsk_name' );
 226      $SQL->FROM( 'T_cron__log' );
 227      $SQL->FROM_add( 'INNER JOIN T_cron__task ON ctsk_ID = clog_ctsk_ID' );
 228      $SQL->WHERE( 'clog_status = "started"' );
 229      $SQL->WHERE_and( 'clog_realstart_datetime < '.$DB->quote( date2mysql( time() + $time_difference - $cron_timeout_delay ) ) );
 230      $SQL->GROUP_BY( 'ctsk_ID' );
 231      $timeouts = $DB->get_assoc( $SQL->get(), OBJECT, $SQL->title );
 232  
 233      $tasks = array();
 234  
 235      if( count( $timeouts ) > 0 )
 236      {
 237          foreach( $timeouts as $task_ID => $task_name )
 238          {
 239              $tasks[ $task_ID ] = array(
 240                      'name'    => $task_name,
 241                      'message' => T_('Cron job was timed out.'),
 242                  );
 243          }
 244  
 245          // Update timed out cron jobs
 246          $DB->query( 'UPDATE T_cron__log
 247                SET clog_status = "timeout"
 248              WHERE clog_ctsk_ID IN ( '.$DB->quote( array_keys( $tasks ) ).' )', 'Detect cron timeouts.' );
 249      }
 250  
 251      if( !is_null( $error_task ) )
 252      { // Send notification with error task
 253          $tasks[ $error_task['ID'] ] = $error_task;
 254      }
 255  
 256      if( count( $tasks ) > 0 )
 257      { // Send notification email about timed out and error cron jobs to users with edit options permission
 258          $email_template_params = array(
 259                  'tasks' => $tasks,
 260              );
 261          send_admin_notification( NT_('Scheduled task error'), 'scheduled_task_error_report', $email_template_params );
 262      }
 263  }
