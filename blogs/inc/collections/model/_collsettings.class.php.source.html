 240      /**
 241       * Loads the settings. Not meant to be called directly, but gets called
 242       * when needed.
 243       *
 244       * @access protected
 245       * @param string First column key
 246       * @param string Second column key
 247       * @return boolean
 248       */
 249  	function _load( $coll_ID, $arg )
 250      {
 251          if( empty( $coll_ID ) )
 252          {
 253              return false;
 254          }
 255  
 256          return parent::_load( $coll_ID, $arg );
 257      }

 260      /**
 261       * Commit changed settings to DB.
 262       *
 263       * @return boolean true, if settings have been updated; false otherwise
 264       */
 265  	function dbupdate()
 266      {
 267          global $DB;
 268  
 269          if( !parent::dbupdate() )
 270          { // Collection settings couldn't been updated successful
 271              return false;
 272          }
 273  
 274          if( !empty( $this->update_cascade ) )
 275          { // process update cascade commands
 276              foreach( $this->update_cascade as $query )
 277              {
 278                  if( $DB->query( $query ) === false )
 279                  { // error occured
 280                      return false;
 281                  }
 282              }
 283              $this->update_cascade = array();
 284          }
 285  
 286          return true;
 287      }

 290      /**
 291       * Add an update/delete query which must be processed if collection settings will be updated
 292       *
 293       * @param string query
 294       */
 295  	function add_update_cascade( $query )
 296      {
 297          if( empty( $query ) )
 298          {
 299              return;
 300          }
 301          $this->update_cascade[] = $query;
 302      }
 303  
 304  
 305      /**
 306       * Remove all registered update query
 307       */
 308  	function clear_update_cascade()
 309      {
 310          $this->update_cascade = array();
 311      }
