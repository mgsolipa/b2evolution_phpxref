  73       * @return boolean True on add, false if already existing.

  75  	function add( & $Blog )
  76      {
  77          if( ! empty($Blog->siteurl) && preg_match( '~^https?://~', $Blog->siteurl ) )
  78          { // absolute siteurl
  79              $this->cache_siteurl_abs[ $Blog->siteurl ] = & $Blog;
  80          }
  81  
  82          $this->cache_urlname[ $Blog->urlname ] = & $Blog;
  83  
  84          return parent::add( $Blog );

  88      /**
  89       * Get an object from cache by its url ("siteurl")
  90       *
  91       * Load the cache if necessary
  92       *
  93       * This gets used in /index_multi.php to detect blogs according to the requested HostWithPath
  94       *
  95       * @todo fp> de-factorize. cleanup. make efficient. split access types.
  96       *
  97       * @param string URL of blog to load (should be the whole requested URL/path, e.g. "http://mr.example.com/permalink")
  98       * @param boolean false if you want to return false on error
  99       * @return Blog A Blog object on success, false on failure (may also halt!)
 100       */
 101      function & get_by_url( $req_url, $halt_on_error = true )
 102      {
 103          global $DB, $Debuglog, $baseurl, $basedomain;
 104  
 105          foreach( array_keys($this->cache_siteurl_abs) as $siteurl_abs )
 106          {
 107              if( strpos( $req_url, $siteurl ) === 0 )
 108              { // found in cache
 109                  return $this->cache_siteurl_abs[$siteurl_abs];
 110              }
 111          }
 112  
 113          // Load just the requested object:
 114          $Debuglog->add( "Loading <strong>$this->objtype($req_url)</strong> into cache", 'dataobjects' );
 115  
 116          $req_url_wo_proto = substr( $req_url, strpos( $req_url, '://' ) ); // req_url without protocol, so it matches http and https below
 117  
 118          $sql = 'SELECT *
 119                FROM T_blogs
 120               WHERE ( blog_access_type = "absolute"
 121                       AND ( '.$DB->quote('http'.$req_url_wo_proto).' LIKE CONCAT( blog_siteurl, "%" )
 122                           OR '.$DB->quote('https'.$req_url_wo_proto).' LIKE CONCAT( blog_siteurl, "%" ) ) )
 123                  OR ( blog_access_type = "subdom"
 124                       AND '.$DB->quote($req_url_wo_proto).' LIKE CONCAT( "://", blog_urlname, ".'.$basedomain.'/%" ) )';
 125  
 126          // Match stubs like "http://base/url/STUB?param=1" on $baseurl
 127          /*
 128          if( preg_match( "#^$baseurl([^/?]+)#", $req_url, $match ) )
 129          {
 130              $sql .= "\n OR ( blog_access_type = 'stub' AND blog_stub = ".$DB->quote($match[1])." )";
 131          }
 132          */
 133  
 134          $row = $DB->get_row( $sql, OBJECT, 0, 'Blog::get_by_url()' );
 135  
 136          if( empty( $row ) )
 137          { // Requested object does not exist
 138              if( $halt_on_error ) debug_die( "Requested $this->objtype does not exist!" );
 139  
 140              $r = false;
 141              return $r; // we return by reference!
 142          }
 143  
 144          $Blog = new Blog( $row );
 145          $this->add( $Blog );
 146  
 147          return $Blog;
 148      }

 181              return $r;

 295      /**
 296       * Load blogs a user has permissions for.
 297       *
 298       * @param string permission: 'member' (default), 'browse' (files)
 299       * @param string
 300       * @param integer user ID
 301       * @return array The blog IDs
 302       */
 303  	function load_user_blogs( $permname = 'blog_ismember', $permlevel = 'view', $user_ID = NULL, $order_by = '', $order_dir = '', $limit = NULL )
 304      {
 305          global $DB, $Settings, $Debuglog;
 306  
 307          $Debuglog->add( "Loading <strong>$this->objtype(permission: $permname)</strong> into cache", 'dataobjects' );
 308  
 309          if( $order_by == '' )
 310          {    // Use default value from settings
 311              $order_by = $Settings->get('blogs_order_by');
 312          }
 313  
 314          if( $order_dir == '' )
 315          {    // Use default value from settings
 316              $order_dir = $Settings->get('blogs_order_dir');
 317          }
 318  
 319          if( is_null($user_ID) )
 320          {
 321              global $current_User;
 322              $user_ID = $current_User->ID;
 323              $for_User = $current_User;
 324          }
 325          else
 326          {
 327              $UserCache = & get_UserCache();
 328              $for_User = & $UserCache->get_by_ID( $user_ID );
 329          }
 330          $for_User->get_Group();// ensure Group is set
 331  
 332          $Group = $for_User->Group;
 333          // First check if we have a global access perm:
 334           if( $Group->check_perm( 'blogs', $permlevel ) )
 335          { // If group grants a global permission:
 336              $this->load_all( $order_by, $order_dir );
 337              return $this->get_ID_array();
 338          }
 339  
 340          // Note: We only JOIN in the advanced perms if any given blog has them enabled,
 341          // otherwise they are ignored!
 342          $sql = "SELECT DISTINCT T_blogs.*
 343                    FROM T_blogs LEFT JOIN T_coll_user_perms ON (blog_advanced_perms <> 0
 344                                                                                                    AND blog_ID = bloguser_blog_ID
 345                                                                                                    AND bloguser_user_ID = {$user_ID} )
 346                             LEFT JOIN T_coll_group_perms ON (blog_advanced_perms <> 0
 347                                                                                        AND blog_ID = bloggroup_blog_ID
 348                                                                                        AND bloggroup_group_ID = {$Group->ID} )
 349                   WHERE ";
 350  
 351          if( $permname != 'blog_admin' )
 352          {    // Only the admin perm is not convered by being the owner of the blog:
 353              $sql .= "blog_owner_user_ID = {$user_ID} ";
 354          }
 355  
 356          switch( $permname )
 357          {
 358              case 'blog_ismember':
 359                  $sql .= "OR bloguser_ismember <> 0
 360                                   OR bloggroup_ismember <> 0";
 361                  break;
 362  
 363              case 'blog_post_statuses':
 364                  $sql .= "OR bloguser_perm_poststatuses <> ''
 365                                 OR bloggroup_perm_poststatuses <> ''";
 366                  break;
 367  
 368              case 'blog_comments':
 369                  // user needs to have permission for at least one kind of comments
 370                  $sql .= "OR bloguser_perm_cmtstatuses <> ''
 371                          OR bloggroup_perm_cmtstatuses <> ''";
 372                  break;
 373  
 374              case 'stats':
 375                  $permname = 'blog_properties';    // TEMP
 376              case 'blog_cats':
 377              case 'blog_properties':
 378              case 'blog_admin':
 379              case 'blog_media_browse':
 380                  $short_permname = substr( $permname, 5 );
 381                  $sql .= "OR bloguser_perm_{$short_permname} <> 0
 382                                   OR bloggroup_perm_{$short_permname} <> 0";
 383                  break;
 384  
 385              default:
 386                  debug_die( 'BlogCache::load_user_blogs() : Unsupported perm ['.$permname.']!' );
 387          }
 388  
 389          $sql .= " ORDER BY ".gen_order_clause( $order_by, $order_dir, $this->dbprefix, $this->dbIDname );
 390  
 391          if( $limit )
 392          {
 393              $sql .= " LIMIT {$limit}";
 394          }
 395  
 396          foreach( $DB->get_results( $sql, OBJECT, 'Load user blog list' ) as $row )
 397          {
 398              // Instantiate a custom object
 399              $this->instantiate( $row );
 400          }
 401  
 402          return $DB->get_col( NULL, 0 );
 403      }
