 249      global $blog;

 251    /**
 252       * @var User
 253       */
 254      global $current_User;
 255  
 256      $autoselected_blog = $blog;

 259      { // a blog is already selected
 260          if( !$current_User->check_perm( $permname, $permlevel, false, $autoselected_blog ) )

 262               // echo 'current blog was invalid';
 263              $autoselected_blog = 0;
 264          }
 265      }
 266

 390  /**
 391   * Enable/Disable the given cache
 392   *
 393   * @param string cache key name, 'general_cache_enabled', blogs 'cache_enabled'
 394   * @param boolean status to set
 395   * @param integer the id of the blog, if we want to set a blog's cache. Let it NULL to set general caching.
 396   * @param boolean true to save db changes, false if db update will be called outside from this function
 397   */
 398  function set_cache_enabled( $cache_key, $new_status, $coll_ID = NULL, $save_setting = true )
 399  {
 400      load_class( '_core/model/_pagecache.class.php', 'PageCache' );
 401      global $Settings;
 402  
 403      if( empty( $coll_ID ) )
 404      { // general cache
 405          $Blog = NULL;
 406          $old_cache_status = $Settings->get( $cache_key );
 407      }
 408      else
 409      { // blog page cache
 410          $BlogCache = & get_BlogCache();
 411          $Blog = $BlogCache->get_by_ID( $coll_ID );
 412          $old_cache_status = $Blog->get_setting( $cache_key );
 413      }
 414  
 415      $PageCache = new PageCache( $Blog );
 416      if( $old_cache_status == false && $new_status == true )
 417      { // Caching has been turned ON:
 418          if( $PageCache->cache_create( false ) )
 419          { // corresponding cache folder was created
 420              if( empty( $coll_ID ) )
 421              { // general cache
 422                  $result = array( 'success', T_( 'General caching has been enabled.' ) );
 423              }
 424              else
 425              { // blog page cache
 426                  $result = array( 'success', T_( 'Page caching has been enabled.' ) );
 427              }
 428          }
 429          else
 430          { // error creating cache folder
 431              if( empty( $coll_ID ) )
 432              { // general cache
 433                  $result = array( 'error', T_( 'General caching could not be enabled. Check /cache/ folder file permissions.' ) );
 434              }
 435              else
 436              { // blog page cache
 437                  $result = array( 'error', T_( 'Page caching could not be enabled. Check /cache/ folder file permissions.' ) );
 438              }
 439              $new_status = false;
 440          }
 441      }
 442      elseif( $old_cache_status == true && $new_status == false )
 443      { // Caching has been turned OFF:
 444          $PageCache->cache_delete();
 445          if( empty( $coll_ID ) )
 446          { // general cache
 447              $result = array( 'note',  T_( 'General caching has been disabled. Cache contents have been purged.' ) );
 448          }
 449          else
 450          { // blog page cache
 451              $result = array( 'note',  T_( 'Page caching has been disabled. Cache contents have been purged.' ) );
 452          }
 453      }
 454      else
 455      { // nothing was changed
 456          // check if ajax_form_enabled has correct state after b2evo upgrade
 457          if( ( $Blog != NULL ) && ( $new_status ) && ( !$Blog->get_setting( 'ajax_form_enabled' ) ) )
 458          { // if page cache is enabled, ajax form must be enabled to
 459              $Blog->set_setting( 'ajax_form_enabled', true );
 460              $Blog->dbupdate();
 461          }
 462          return NULL;
 463      }
 464  
 465      // set db changes
 466      if( $Blog == NULL )
 467      {
 468          $Settings->set( 'general_cache_enabled', $new_status );
 469          if( $save_setting )
 470          { // save
 471              $Settings->dbupdate();
 472          }
 473      }
 474      else
 475      {
 476          $Blog->set_setting( $cache_key, $new_status );
 477          if( ( $cache_key == 'cache_enabled' ) && $new_status )
 478          { // if page cache is enabled, ajax form must be enabled to
 479              $Blog->set_setting( 'ajax_form_enabled', true );
 480          }
 481          if( $save_setting )
 482          { // save
 483              $Blog->dbupdate();
 484          }
 485      }
 486      return $result;
 487  }

 490  /**
 491   * Initialize global $blog variable to the requested blog
 492   *
 493   * @return boolean true if $blog was initialized successful, false otherwise
 494   */
 495  function init_requested_blog()
 496  {
 497      global $blog, $ReqHost, $ReqPath;
 498      global $Settings;
 499      global $Debuglog;
 500  
 501      if( !empty( $blog ) )
 502      { // blog was already initialized
 503          return true;
 504      }
 505  
 506      // Check if a specific blog has been requested in the URL:
 507      $blog = param( 'blog', 'integer', '', true );
 508  
 509      if( !empty($blog) )
 510      { // a specific blog has been requested in the URL
 511          return true;
 512      }
 513  
 514      // No blog requested by URL param, let's try to match something in the URL
 515      $Debuglog->add( 'No blog param received, checking extra path...', 'detectblog' );
 516      $BlogCache = & get_BlogCache();
 517      if( preg_match( '#^(.+?)index.php/([^/]+)#', $ReqHost.$ReqPath, $matches ) )
 518      { // We have an URL blog name:
 519          $Debuglog->add( 'Found a potential URL blog name: '.$matches[2], 'detectblog' );
 520          if( (($Blog = & $BlogCache->get_by_urlname( $matches[2], false )) !== false) )
 521          { // We found a matching blog:
 522              $blog = $Blog->ID;
 523              return true;
 524          }
 525      }
 526  
 527      // No blog identified by URL name, let's try to match the absolute URL
 528      if( preg_match( '#^(.+?)index.php#', $ReqHost.$ReqPath, $matches ) )
 529      { // Remove what's not part of the absolute URL
 530          $ReqAbsUrl = $matches[1];
 531      }
 532      else
 533      {
 534          $ReqAbsUrl = $ReqHost.$ReqPath;
 535      }
 536      $Debuglog->add( 'Looking up absolute url : '.$ReqAbsUrl, 'detectblog' );
 537      if( (($Blog = & $BlogCache->get_by_url( $ReqAbsUrl, false )) !== false) )
 538      { // We found a matching blog:
 539          $blog = $Blog->ID;
 540          $Debuglog->add( 'Found matching blog: '.$blog, 'detectblog' );
 541          return true;
 542      }
 543  
 544      // Still no blog requested, use default
 545      $blog = $Settings->get('default_blog_ID');
 546      if( (($Blog = & $BlogCache->get_by_ID( $blog, false, false )) !== false) )
 547      { // We found a matching blog:
 548          $Debuglog->add( 'Using default blog '.$blog, 'detectblog' );
 549          return true;
 550      }
 551  
 552      $blog = NULL;
 553      return false;
 554  }

 697  /**
 698   * Get the highest public status and action button label of a new post or comment in the given blog what the current User may create.
 699   * We assume here that the User should be able to create a post/comment with at least the lowest level status.
 700   *
 701   * @param string 'post' or 'comment'
 702   * @param integer blog ID
 703   * @param boolean set false to get only the status without the action button label
 704   * @return mixed string status if with_label is false, array( status, label ) if with_label is true
 705   */
 706  function get_highest_publish_status( $type, $blog, $with_label = true )
 707  {
 708      global $current_User;
 709  
 710      if( ( $type != 'post' ) && ( $type != 'comment' ) )
 711      { // $type is invalid
 712          debug_die( 'Invalid type parameter!' );
 713      }
 714  
 715      $BlogCache = & get_BlogCache();
 716      $requested_Blog = $BlogCache->get_by_ID( $blog );
 717      $default_status = ( $type == 'post' ) ? $requested_Blog->get_setting( 'default_post_status' ) : $requested_Blog->get_setting( 'new_feedback_status' );
 718  
 719      if( empty( $current_User ) || ( ( !$requested_Blog->get( 'advanced_perms' ) ) && ( !$current_User->check_perm_blog_global( $blog, 'editall' ) ) ) )
 720      { // current User is not set or collection advanced perms are not enabled and user has no global perms on the given blog, set status to the default status
 721          return ( $with_label ? array( $default_status, '' ) : $default_status );
 722      }
 723  
 724      $status_order = get_visibility_statuses( 'ordered-array' );
 725      $highest_index = count( $status_order ) - 1;
 726      $default_status_label = '';
 727      $result = false;
 728      for( $index = $highest_index; $index > 0; $index-- )
 729      {
 730          $curr_status = $status_order[$index][0];
 731          if( $curr_status == $default_status )
 732          { // Set default status label for later use
 733              $default_status_label =  $status_order[$index][1];
 734          }
 735          if( $current_User->check_perm( 'blog_'.$type.'!'.$curr_status, 'create', false, $blog ) )
 736          { // The highest available publish status has been found
 737              $result = array( $curr_status, $status_order[$index][1] );
 738              break;
 739          }
 740      }
 741  
 742      if( !$result )
 743      { // There are no available public status
 744          if( $current_User->check_perm( 'blog_'.$type.'!private', 'create', false, $blog ) )
 745          { // Check private status
 746              $result = array( 'private', T_('Make private!') );

 749          { // None of the statuses were allowed avove the 'draft' status, so we return the lowest level status.
 750              $result = array( 'draft', '' );
 751          }
 752      }
 753  
 754      if( $with_label )
 755      {
 756          return $result;
 757      }
 758  
 759      // Return only the highest available visibility status without label
 760      return $result[0];
 761  }
