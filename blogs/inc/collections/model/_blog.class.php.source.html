 238                  $this->set_setting( 'archive_mode', 'postbypost' );
 239                  break;
 240  
 241              case 'group':
 242                  $this->set( 'type', 'group' );
 243                  $this->set( 'name', empty($name) ? T_('Our blog') : $name );
 244                  $this->set( 'shortname', empty($shortname) ? T_('Group') : $shortname );
 245                  $this->set( 'urlname', empty($urlname) ? 'group' : $urlname );
 246                  $this->set_setting( 'use_workflow', 1 );
 247                  break;
 248  
 249              case 'forum':
 250                  $this->set( 'type', 'forum' );
 251                  $this->set( 'name', empty($name) ? T_('My forum') : $name );
 252                  $this->set( 'shortname', empty($shortname) ? T_('Forum') : $shortname );
 253                  $this->set( 'urlname', empty($urlname) ? 'forum' : $urlname );
 254                  $this->set( 'advanced_perms', 1 );
 255                  $this->set_setting( 'post_navigation', 'same_category' );
 256                  $this->set_setting( 'allow_comments', 'registered' );
 257                  $this->set_setting( 'in_skin_editing', '1' );
 258                  $this->set_setting( 'posts_per_page', 30 );
 259                  $this->set_setting( 'allow_html_post', 0 );
 260                  $this->set_setting( 'allow_html_comment', 0 );
 261                  $this->set_setting( 'orderby', 'last_touched_ts' );
 262                  $this->set_setting( 'orderdir', 'DESC' );
 263                  $this->set_setting( 'enable_goto_blog', 'post' );
 264                  $this->set_setting( 'front_disp', 'front' );
 265                  break;
 266  
 267              case 'manual':
 268                  $this->set( 'type', 'manual' );
 269                  $this->set( 'name', empty($name) ? T_('Manual') : $name );
 270                  $this->set( 'shortname', empty($shortname) ? T_('Manual') : $shortname );
 271                  $this->set( 'urlname', empty($urlname) ? 'manual' : $urlname );
 272                  $this->set_setting( 'post_navigation', 'same_category' );
 273                  $this->set_setting( 'single_links', 'chapters' );
 274                  $this->set_setting( 'enable_goto_blog', 'post' );
 275                  $this->set_setting( 'front_disp', 'front' );
 276                  break;
 277  
 278              case 'std':
 279              default:
 280                  $this->set( 'type', 'std' );
 281                  $this->set( 'name', empty($name) ? T_('My weblog') : $name );
 282                  $this->set( 'shortname', empty($shortname) ? T_('Blog') : $shortname );
 283                  $this->set( 'urlname', empty($urlname) ? 'blog' : $urlname );
 284                  break;
 285          }
 286  
 287          if( empty($name) && empty($shortname) && empty($urlname) )
 288          {    // Not in installation mode, init custom collection kinds.
 289              global $Plugins;
 290  
 291              // Defines blog settings by its kind.
 292              $Plugins->trigger_event( 'InitCollectionKinds', array(
 293                              'Blog' => & $this,
 294                              'kind' => & $kind,
 295                          ) );
 296          }
 297      }
