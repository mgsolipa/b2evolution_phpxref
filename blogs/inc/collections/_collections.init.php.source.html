 387      /**
 388       * Get default module permissions
 389       *
 390       * @param integer Group ID
 391       * @return array
 392       */
 393  	function get_default_group_permissions( $grp_ID )
 394      {
 395          switch( $grp_ID )
 396          {
 397              case 1:        // Administrators (group ID 1) have permission by default:
 398                  $permapi = 'always'; // Can use APIs
 399                  $permcreateblog = 'allowed'; // Creating new blogs
 400                  $permgetblog = 'denied'; // Automatically add a new blog to the new users
 401                  break;
 402  
 403              case 2:        // Moderators (group ID 2) have permission by default:
 404                  $permapi = 'always';
 405                  $permcreateblog = 'allowed';
 406                  $permgetblog = 'denied';
 407                  break;
 408  
 409              case 3:        // Trusted Users (group ID 3) have permission by default:
 410              case 4:     // Normal Users (group ID 4) have permission by default:
 411                  $permapi = 'always';
 412                  $permcreateblog = 'denied';
 413                  $permgetblog = 'denied';
 414                  break;
 415  
 416              default:
 417                  // Other groups have no permission by default
 418                  $permapi = 'never';
 419                  $permcreateblog = 'denied';
 420                  $permgetblog = 'denied';
 421                  break;
 422          }
 423  
 424          // We can return as many default permissions as we want:
 425          // e.g. array ( permission_name => permission_value, ... , ... )
 426          return $permissions = array( 'perm_api' => $permapi, 'perm_createblog' => $permcreateblog, 'perm_getblog' => $permgetblog );
 427      }

 491       *
 492       * @param string Requested permission level
 493       * @param string Permission value
 494       * @param mixed Permission target (blog ID, array of cat IDs...)
 495       * @return boolean True on success (permission is granted), false if permission is not granted
 496       */
 497  	function check_api_group_perm( $permlevel, $permvalue, $permtarget )
 498      {
 499          $perm = false;
 500          switch ( $permvalue )
 501          {
 502              case 'always':
 503                  // Users can use APIs
 504                  if( $permlevel == 'always' )
 505                  {
 506                      $perm = true;
 507                      break;
 508                  }
 509  
 510              case 'never':
 511                  // Users can`t use APIs
 512                  if( $permlevel == 'never' )
 513                  {
 514                      $perm = false;
 515                      break;
 516                  }
 517          }
 518  
 519          return $perm;
 520      }

 871      /**
 872       * Handle collections module htsrv actions
 873       */
 874  	function handle_htsrv_action()
 875      {
 876          global $demo_mode, $current_User, $DB, $Session, $Messages;
 877          global $UserSettings, $samedomain_htsrv_url;
 878  
 879          if( !is_logged_in() )
 880          { // user must be logged in
 881              bad_request_die( $this->T_( 'You are not logged in.' ) );
 882          }
 883  
 884          // Init the objects we want to work on.
 885          $action = param_action( true );
 886  
 887          // Check that this action request is not a CSRF hacked request:
 888          $Session->assert_received_crumb( 'collections_'.$action );
 889  
 890          switch( $action )
 891          {
 892              case 'unlink':
 893                  // Unlink a file from a LinkOwner ( Item, Comment ) object, and delete that file if it's not linked to any other object
 894  
 895                  $link_ID = param( 'link_ID', 'integer', true );
 896                  $redirect_to = param( 'redirect_to', 'url', '' );
 897                  $LinkCache = & get_LinkCache();
 898                  $edited_Link = & $LinkCache->get_by_ID( $link_ID, false );
 899  
 900                  if( !$edited_Link )
 901                  { // the edited Link object doesn't exists
 902                      $Messages->add( sprintf( T_('Requested &laquo;%s&raquo; object does not exist any longer.'), T_('Link') ), 'error' );
 903                      header_redirect();
 904                  }
 905  
 906                  // We have a link, get the LinkOwner it is attached to:
 907                  $LinkOwner = & $edited_Link->get_LinkOwner();
 908                  $linked_File = & $edited_Link->get_File();
 909  
 910                  // Load the blog we're in:
 911                  $Blog = & $LinkOwner->get_Blog();
 912                  set_working_blog( $Blog->ID );
 913  
 914                  // Check permission:
 915                  $LinkOwner->check_perm( 'edit', true );
 916  
 917                  $confirmed = param( 'confirmed', 'integer', 0 );
 918                  if( $confirmed )
 919                  { // Unlink File from Item:
 920                      $deleted_link_ID = $edited_Link->ID;
 921                      $edited_Link->dbdelete( true );
 922                      unset($edited_Link);
 923  
 924                      $LinkOwner->after_unlink_action( $deleted_link_ID );
 925  
 926                      $Messages->add( $LinkOwner->translate( 'Link has been deleted from $ownerTitle$.' ), 'success' );
 927  
 928                      if( $current_User->check_perm( 'files', 'edit' ) )
 929                      { // current User has permission to edit/delete files
 930                          $file_name = $linked_File->get_name();
 931                          // Get number of objects where this file is attahced to
 932                          // TODO: attila>this must be handled with a different function
 933                          $file_links = get_file_links( $linked_File->ID, array( 'separator' => '<br />' ) );
 934                          $links_count = ( strlen( $file_links ) > 0 ) ? substr_count( $file_links, '<br />' ) + 1 : 0;
 935                          if( $links_count > 0 )
 936                          { // File is linked to other objects
 937                              $Messages->add( sprintf( T_('File %s is still linked to %d other objects'), $file_name, $links_count ), 'note' );
 938                          }
 939                          else
 940                          { // File is not linked to other objects
 941                              if( $linked_File->unlink() )
 942                              { // File removed successful ( removed from db and from storage device also )
 943                                  $Messages->add( sprintf( T_('File %s has been deleted.'), $file_name ), 'success' );
 944                              }
 945                              else
 946                              { // Could not completly remove the file
 947                                  $Messages->add( sprintf( T_('File %s could not be deleted.'), $file_name ), 'error' );
 948                              }
 949                          }
 950                      }
 951                  }
 952                  else
 953                  { // Display confirm unlink/delete message
 954                      $delete_url = $samedomain_htsrv_url.'action.php?mname=collections&action=unlink&link_ID='.$edited_Link->ID.'&confirmed=1&crumb_collections_unlink='.get_crumb( 'collections_unlink' );
 955                      $ok_button = '<span class="linkbutton"><a href="'.$delete_url.'">'.T_( 'I am sure!' ).'!</a></span>';
 956                      $cancel_button = '<span class="linkbutton"><a href="'.$redirect_to.'">CANCEL</a></span>';
 957                      $msg = sprintf( T_( 'You are about to unlink and delete the attached file from %s path.' ), $linked_File->get_root_and_rel_path() );
 958                      $msg .= '<br />'.T_( 'This CANNOT be undone!').'<br />'.T_( 'Are you sure?' ).'<br /><br />'.$ok_button."\t".$cancel_button;
 959                      $Messages->add( $msg, 'error' );
 960                  }
 961                  header_redirect( $redirect_to );
 962                  break;
 963  
 964              case 'isubs_update':
 965                  // Subscribe/Unsubscribe user on the selected item
 966  
 967                  if( $demo_mode && ( $current_User->ID <= 3 ) )
 968                  { // don't allow default users profile change on demo mode
 969                      bad_request_die( 'Demo mode: you can\'t edit the admin and demo users profile!<br />[<a href="javascript:history.go(-1)">'
 970                                  . T_('Back to profile') . '</a>]' );
 971                  }
 972  
 973                  // Get params
 974                  $item_ID = param( 'p', 'integer', true );
 975                  $notify = param( 'notify', 'integer', 0 );
 976  
 977                  if( ( $notify < 0 ) || ( $notify > 1 ) )
 978                  { // Invalid notify param. It should be 0 for unsubscribe and 1 for subscribe.
 979                      $Messages->add( 'Invalid params!', 'error' );
 980                  }
 981  
 982                  if( ! is_email( $current_User->get( 'email' ) ) )
 983                  { // user doesn't have a valid email address
 984                      $Messages->add( T_( 'Your email address is invalid. Please set your email address first.' ), 'error' );
 985                  }
 986  
 987                  if( $Messages->has_errors() )
 988                  { // errors detected
 989                      header_redirect();
 990                      // already exited here
 991                  }
 992  
 993                  if( set_user_isubscription( $current_User->ID, $item_ID, $notify ) )
 994                  {
 995                      if( $notify == 0 )
 996                      {
 997                          $Messages->add( T_( 'You have successfully unsubscribed.' ), 'success' );
 998                      }
 999                      else
1000                      {
1001                          $Messages->add( T_( 'You have successfully subscribed to notifications.' ), 'success' );
1002                      }
1003                  }
1004                  else
1005                  {
1006                      $Messages->add( T_( 'Could not subscribe to notifications.' ), 'error' );
1007                  }
1008  
1009                  header_redirect();
1010                  break; // already exited here
1011          }
1012      }
