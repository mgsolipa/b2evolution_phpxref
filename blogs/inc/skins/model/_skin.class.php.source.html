 238      /**
 239       * Discover containers included in skin files
 240       */
 241  	function discover_containers()
 242      {
 243          global $skins_path, $Messages;
 244  
 245          $this->container_list = array();
 246  
 247          if( ! $dir = @opendir($skins_path.$this->folder) )
 248          {    // Skin directory not found!
 249              $Messages->add( 'Cannot open skin directory.', 'error' ); // No trans
 250              return false;
 251          }
 252  
 253          // Go through all files in the skin directory:
 254          while( ( $file = readdir($dir) ) !== false )
 255          {
 256              $rf_main_subpath = $this->folder.'/'.$file;
 257              $af_main_path = $skins_path.$rf_main_subpath;
 258  
 259              if( !is_file( $af_main_path ) || ! preg_match( '~\.php$~', $file ) )
 260              { // Not a php template file, go to next:
 261                  continue;
 262              }
 263  
 264              if( ! is_readable($af_main_path) )
 265              {    // Cannot open PHP file:
 266                  $Messages->add( sprintf( T_('Cannot read skin file &laquo;%s&raquo;!'), $rf_main_subpath ), 'error' );
 267                  continue;
 268              }
 269  
 270              $file_contents = @file_get_contents( $af_main_path );
 271              if( ! is_string($file_contents) )
 272              {    // Cannot get contents:
 273                  $Messages->add( sprintf( T_('Cannot read skin file &laquo;%s&raquo;!'), $rf_main_subpath ), 'error' );
 274                  continue;
 275              }
 276  
 277              // DETECT if the file contains containers:
 278              // if( ! preg_match_all( '~ \$Skin->container\( .*? (\' (.+?) \' )|(" (.+?) ") ~xmi', $file_contents, $matches ) )
 279              if( ! preg_match_all( '~ (\$Skin->|skin_)container\( .*? ((\' (.+?) \')|(" (.+?) ")) ~xmi', $file_contents, $matches ) )
 280              {    // No containers in this file, go to next:
 281                  continue;
 282              }
 283  
 284              // Merge matches from the two regexp parts (due to regexp "|" )
 285              $container_list = array_merge( $matches[4], $matches[6] );
 286  
 287              $c = 0;
 288              foreach( $container_list as $container )
 289              {
 290                  if( empty($container) )
 291                  {    // regexp empty match -- NOT a container:
 292                      continue;
 293                  }
 294  
 295                  // We have one more container:
 296                  $c++;
 297  
 298                  if( in_array( $container, $this->container_list ) )
 299                  {    // we already have that one
 300                      continue;
 301                  }
 302  
 303                  $this->container_list[] = $container;
 304              }
 305  
 306              if( $c )
 307              {
 308                  $Messages->add( sprintf( T_('%d containers have been found in skin template &laquo;%s&raquo;.'), $c, $rf_main_subpath ), 'success' );
 309              }
 310          }
 311  
 312          // pre_dump( $this->container_list );
 313  
 314          if( empty($this->container_list) )
 315          {
 316              $Messages->add( T_('No containers found in this skin!'), 'error' );
 317              return false;
 318          }
 319  
 320          return true;
 321      }

 392      /**
 393       * Save containers
 394       *
 395       * to be called by dbinsert / dbupdate
 396       */
 397  	function db_save_containers()
 398      {
 399          global $DB;
 400  
 401          // Get a list of all currently empty containers:
 402          $sql = 'SELECT sco_name
 403                            FROM T_skins__container LEFT JOIN T_widget ON ( sco_name = wi_sco_name )
 404                           WHERE sco_skin_ID = '.$this->ID.'
 405                           GROUP BY sco_name
 406                          HAVING COUNT(wi_ID) = 0';
 407          $empty_containers_list = $DB->get_col( $sql, 0, 'Get empty containers' );
 408          //pre_dump( $empty_containers_list );
 409  
 410          // Look for containers in skin file:
 411          $this->discover_containers();
 412  
 413          // Delete empty containers:
 414          foreach( $empty_containers_list as $empty_container )
 415          {
 416              if( !in_array( $empty_container, $this->container_list ) )
 417              {    // This container has been removed from the skin + it's empty, so delete it from DB:
 418                  $DB->query( 'DELETE FROM T_skins__container
 419                                      WHERE sco_name = '.$DB->quote($empty_container) );
 420              }
 421          }
 422  
 423          // Make sure new containers are added:
 424          if( ! empty( $this->container_list ) )
 425          {
 426              $values = array();
 427              foreach( $this->container_list as $container_name )
 428              {
 429                  $values [] = '( '.$this->ID.', '.$DB->quote($container_name).' )';
 430              }
 431  
 432              $DB->query( 'REPLACE INTO T_skins__container( sco_skin_ID, sco_name )
 433                                          VALUES '.implode( ',', $values ), 'Insert containers' );
 434          }
 435      }

 672      /**
 673       * Translate a given string, in the Skin's context.
 674       *
 675       * This means, that the translation is obtained from the Skin's
 676       * "locales" folder.
 677       *
 678       * It uses the global/regular {@link T_()} function as a fallback.
 679       *
 680       * @param string The string (english), that should be translated
 681       * @param string Requested locale ({@link $current_locale} gets used by default)
 682       * @return string The translated string.
 683       *
 684       * @uses T_()
 685       * @since 3.2.0 (after beta)
 686       */
 687      function T_( $string, $req_locale = '' )
 688      {
 689          global $skins_path;
 690  
 691          if( ( $return = T_( $string, $req_locale, array(
 692                                  'ext_transarray' => & $this->_trans,
 693                                  'alt_basedir'    => $skins_path.$this->folder,
 694                              ) ) ) == $string )
 695          {    // This skin did not provide a translation - fallback to global T_():
 696              return T_( $string, $req_locale );
 697          }
 698  
 699          return $return;
 700      }
