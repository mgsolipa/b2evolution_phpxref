 298   * @param integer File ID

 299   * @param array Params

 300   * @return string The links to that posts, comments and users where the file is used

 301   */
 302  function get_file_links( $file_ID, $params = array() )
 303  {
 304      global $DB, $current_User, $baseurl, $admin_url;
 305  
 306      $params = array_merge( array(
 307              'separator'       => '<br />',
 308              'post_prefix'     => T_('Post').' - ',
 309              'comment_prefix'  => T_('Comment on').' - ',
 310              'user_prefix'     => T_('Profile picture').' - ',
 311              'current_link_ID' => 0,
 312              'current_before'  => '<b>',
 313              'current_after'   => '</b>',
 314          ), $params );
 315  
 316      // Create result array

 317      $attached_to = array();
 318  
 319      // Get all links with posts and comments

 320      $links_SQL = new SQL();
 321      $links_SQL->SELECT( 'link_ID, link_itm_ID, link_cmt_ID, link_usr_ID' );
 322      $links_SQL->FROM( 'T_links' );
 323      $links_SQL->WHERE( 'link_file_ID = '.$DB->quote( $file_ID ) );
 324      $links = $DB->get_results( $links_SQL->get() );
 325  
 326      if( !empty( $links ) )
 327      { // File is linked with some posts or comments
 328          $ItemCache = & get_ItemCache();
 329          $CommentCache = & get_CommentCache();
 330          $UserCache = & get_UserCache();
 331          foreach( $links as $link )
 332          {
 333              $r = '';
 334              if( $params['current_link_ID'] == $link->link_ID )
 335              {
 336                  $r .= $params['current_before'];
 337              }
 338              if( !empty( $link->link_itm_ID ) )
 339              { // File is linked to a post
 340                  if( $Item = & $ItemCache->get_by_ID( $link->link_itm_ID, false ) )
 341                  {
 342                      $Blog = $Item->get_Blog();
 343                      if( $current_User->check_perm( 'item_post!CURSTATUS', 'view', false, $Item ) )
 344                      { // Current user can edit the linked post
 345                          $r .= $params['post_prefix'].'<a href="'.url_add_param( $admin_url, 'ctrl=items&amp;blog='.$Blog->ID.'&amp;p='.$link->link_itm_ID ).'">'.$Item->get( 'title' ).'</a>';
 346                      }
 347                      else
 348                      { // No access to edit the linked post
 349                          $r .= $params['post_prefix'].$Item->get( 'title' );
 350                      }
 351                  }
 352              }
 353              if( !empty( $link->link_cmt_ID ) )
 354              { // File is linked to a comment
 355                  if( $Comment = & $CommentCache->get_by_ID( $link->link_cmt_ID, false ) )
 356                  {
 357                      $Item = $Comment->get_Item();
 358                      if( $current_User->check_perm( 'comment!CURSTATUS', 'moderate', false, $Comment ) )
 359                      { // Current user can edit the linked Comment
 360                          $r .= $params['comment_prefix'].'<a href="'.url_add_param( $admin_url, 'ctrl=comments&amp;action=edit&amp;comment_ID='.$link->link_cmt_ID ).'">'.$Item->get( 'title' ).'</a>';
 361                      }
 362                      else
 363                      { // No access to edit the linked Comment
 364                          $r .= $params['comment_prefix'].$Item->get( 'title' );
 365                      }
 366                  }
 367              }
 368              if( !empty( $link->link_usr_ID ) )
 369              { // File is linked to user
 370                  if( $User = & $UserCache->get_by_ID( $link->link_usr_ID, false ) )
 371                  {
 372                      if( $current_User->ID != $User->ID && !$current_User->check_perm( 'users', 'view' ) )
 373                      { // No permission to view other users in admin form
 374                          $r .= $params['user_prefix'].'<a href="'.url_add_param( $baseurl, 'disp=user&amp;user_ID='.$User->ID ).'">'.$User->login.'</a>';
 375                      }
 376                      else
 377                      { // Build a link to display a user in admin form
 378                          $r .= $params['user_prefix'].'<a href="?ctrl=user&amp;user_tab=profile&amp;user_ID='.$User->ID.'">'.$User->login.'</a>';
 379                      }
 380                  }
 381              }
 382              if( $params['current_link_ID'] == $link->link_ID )
 383              {
 384                  $r .= $params['current_after'];
 385              }
 386              if( !empty( $r ) )
 387              {
 388                  $attached_to[] = $r;
 389              }
 390          }
 391      }
 392  
 393      return implode( $params['separator'], $attached_to );
 394  }

 397  /**

 398   * Save a vote for the link of file by user

 399   *

 400   * @param string Link ID

 401   * @param integer User ID

 402   * @param string Action of the voting ( 'like', 'noopinion', 'dontlike', 'inappropriate', 'spam' )

 403   * @param integer 1 = checked, 0 = unchecked (for checkboxes: 'Inappropriate' & 'Spam' )

 404   */
 405  function link_vote( $link_ID, $user_ID, $vote_action, $checked = 1 )
 406  {
 407      global $DB;
 408  
 409      // Set modified field name and value

 410      switch( $vote_action )
 411      {
 412          case 'like':
 413              $field_name = 'lvot_like';
 414              $field_value = '1';
 415              break;
 416  
 417          case 'noopinion':
 418              $field_name = 'lvot_like';
 419              $field_value = '0';
 420              break;
 421  
 422          case 'dontlike':
 423              $field_name = 'lvot_like';
 424              $field_value = '-1';
 425              break;
 426  
 427          case 'inappropriate':
 428              $field_name = 'lvot_inappropriate';
 429              $field_value = $checked;
 430              break;
 431  
 432          case 'spam':
 433              $field_name = 'lvot_spam';
 434              $field_value = $checked;
 435              break;
 436  
 437          default:
 438              // invalid vote action

 439              return;
 440      }
 441  
 442      $DB->begin();
 443  
 444      $SQL = new SQL();
 445      $SQL->SELECT( 'lvot_link_ID' );
 446      $SQL->FROM( 'T_links__vote' );
 447      $SQL->WHERE( 'lvot_link_ID = '.$DB->quote( $link_ID ) );
 448      $SQL->WHERE_and( 'lvot_user_ID = '.$DB->quote( $user_ID ) );
 449      $vote = $DB->get_row( $SQL->get() );
 450  
 451      // Save a voting results in DB

 452      if( empty( $vote ) )
 453      { // User replace into to avoid duplicate key conflict in case when user clicks two times fast one after the other
 454          $result = $DB->query( 'REPLACE INTO T_links__vote ( lvot_link_ID, lvot_user_ID, '.$field_name.' )
 455                          VALUES ( '.$DB->quote( $link_ID ).', '.$DB->quote( $user_ID ).', '.$DB->quote( $field_value ).' )' );
 456      }
 457      else
 458      { // Update existing record, because user already has a vote for this file
 459          $result = $DB->query( 'UPDATE T_links__vote
 460                      SET '.$field_name.' = '.$DB->quote( $field_value ).'
 461                      WHERE lvot_link_ID = '.$DB->quote( $link_ID ).'
 462                          AND lvot_user_ID = '.$DB->quote( $user_ID ) );
 463      }
 464  
 465      if( $result )
 466      {
 467          $DB->commit();
 468      }
 469      else
 470      {
 471          $DB->rollback();
 472      }
 473  }
