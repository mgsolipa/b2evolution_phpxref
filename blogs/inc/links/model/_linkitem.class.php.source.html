 105      /**

 106       * Add new link to owner Item

 107       *

 108       * @param integer file ID

 109       * @param integer link position ( 'teaser', 'aftermore' )

 110       * @param int order of the link

 111       */
 112  	function add_link( $file_ID, $position, $order = 1 )
 113      {
 114          $edited_Link = new Link();
 115          $edited_Link->set( 'itm_ID', $this->Item->ID );
 116          $edited_Link->set( 'file_ID', $file_ID );
 117          $edited_Link->set( 'position', $position );
 118          $edited_Link->set( 'order', $order );
 119          if( $edited_Link->dbinsert() )
 120          {
 121              // New link was added to the item, invalidate blog's media BlockCache

 122              BlockCache::invalidate_key( 'media_coll_ID', $this->Item->get_blog_ID() );
 123  
 124              // Update last touched date of the Item

 125              $this->update_last_touched_date();
 126  
 127              // Reset the Links

 128              $this->Links = NULL;
 129              $this->load_Links();
 130  
 131              return true;
 132          }
 133  
 134          return false;
 135      }

 200      /**

 201       * This function is called after when some file was unlinked from item

 202       *

 203       * @param integer Link ID

 204       */
 205  	function after_unlink_action( $link_ID = 0 )
 206      {
 207          if( empty( $this->Item ) )
 208          { // No existing Item, Exit here
 209              return;
 210          }
 211  
 212          if( ! empty( $link_ID ) )
 213          { // Find inline image placeholders if link ID is defined
 214              preg_match_all( '/\[image:'.$link_ID.':?[^\]]*\]/i', $this->Item->content, $inline_images );
 215              if( ! empty( $inline_images[0] ) )
 216              { // There are inline image placeholders in the post content
 217                  $this->Item->set( 'content', str_replace( $inline_images[0], '', $this->Item->content ) );
 218                  $this->Item->dbupdate();
 219                  return;
 220              }
 221          }
 222  
 223          // Update last touched date of the Item

 224          $this->update_last_touched_date();
 225      }
