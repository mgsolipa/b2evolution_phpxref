 119      function & get_File()
 120      {
 121          if( ! isset($this->File) )
 122          {
 123              if( isset($GLOBALS['files_Module']) )
 124              {
 125                  $FileCache = & get_FileCache();
 126                  // fp> do not halt on error. For some reason (ahem bug) a file can disappear and if we fail here then we won't be
 127                  // able to delete the link
 128                  $this->File = & $FileCache->get_by_ID( $this->file_ID, false, false );
 129              }
 130              else
 131              {
 132                  $this->File = NULL;
 133              }
 134          }

 225      /**
 226       * Get an url to download file
 227       *
 228       * @param array Params
 229       * @return string|boolean URL or FALSE when Link object is broken
 230       */
 231  	function get_download_url( $params = array() )
 232      {
 233          $params = array_merge( array(
 234                  'glue' => '&amp;', // Glue between url params
 235                  'type' => 'page', // 'page' - url of the download page, 'action' - url to force download
 236              ), $params );
 237  
 238          if( ! ( $File = & $this->get_File() ) ||
 239              ! ( $LinkOwner = & $this->get_LinkOwner() ) )
 240          { // Broken Link
 241              return false;
 242          }
 243  
 244          if( $LinkOwner->type == 'item' && $LinkOwner->Item )
 245          { // Use specific url for Item to download
 246              switch( $params['type'] )
 247              {
 248                  case 'action':
 249                      // Get URL to froce download a file
 250                      if( $File->get_ext() == 'zip' )
 251                      { // Provide direct url to ZIP files
 252                          return $File->get_url();
 253                      }
 254                      else
 255                      { // For other files use url through special file that forces a download action
 256                          return get_samedomain_htsrv_url().'download.php?link_ID='.$this->ID;
 257                      }
 258  
 259                  case 'page':
 260                  default:
 261                      // Get URL to display a page with info about file and item
 262                      return url_add_param( $LinkOwner->Item->get_permanent_url( '', '', $params['glue'] ), 'download='.$this->ID, $params['glue'] );
 263              }
 264          }
 265          else
 266          { // Use standard url for all other types
 267              return $File->get_view_url( false );
 268          }
 269      }
