 246      /**

 247       * Get default position for a new link

 248       *

 249       * @param boolean new link file is image or not

 250       */
 251  	function get_default_position( $is_image = false )
 252      {
 253          $this->load_Links();
 254  
 255          if( $this->Links )
 256          {
 257              $position = NULL;
 258              $last_Link = array_pop( $this->Links );
 259              $last_File = & $last_Link->get_File();
 260  
 261              if( $last_File && $is_image && $last_File->is_image() && ! count( $this->Links ) )
 262              { // there's only one image attached yet, the second becomes "aftermore"
 263                  $position = 'aftermore';
 264              }
 265              else
 266              { // default: use position of previous link/attachment
 267                  $position = $last_Link->get('position');
 268              }
 269  
 270              // Re-add popped link.

 271              $this->Links[] = $last_Link;
 272          }
 273          else
 274          { // no attachment yet
 275              $position = $is_image ? 'teaser' : 'aftermore';
 276          }
 277  
 278          return $position;
 279      }

 334      /**

 335       * Get list of attached Links

 336       *

 337       * @param integer Limit max result

 338       * @param string Restrict to files/images linked to a specific position.

 339       *               Position can be 'teaser'|'aftermore'|'inline'

 340       *               Use comma as separator

 341       * @param string File type: 'image', 'audio', 'other'; NULL - to select all

 342       * @param array Params

 343       * @return DataObjectList2 on success or NULL if no linked files found

 344       */
 345  	function get_attachment_LinkList( $limit = 1000, $position = NULL, $file_type = NULL, $params = array() )
 346      {
 347          if( ! isset($GLOBALS['files_Module']) )
 348          {
 349              return NULL;
 350          }
 351  
 352          $params = array_merge( array(
 353                  'sql_select_add' => '', // Additional fields for SELECT clause
 354                  'sql_order_by'   => 'link_order', // ORDER BY clause
 355              ), $params );
 356  
 357          global $DB;
 358  
 359          load_class( '_core/model/dataobjects/_dataobjectlist2.class.php', 'DataObjectList2' );
 360  
 361          $LinkCache = & get_LinkCache();
 362  
 363          $LinkList = new DataObjectList2( $LinkCache ); // IN FUNC

 364  
 365          $SQL = new SQL();
 366          $SQL->SELECT( 'l.*' );
 367          $SQL->SELECT_add( $params['sql_select_add'] );
 368          $SQL->FROM( 'T_links AS l' );
 369          $SQL->WHERE( $this->get_where_condition() );
 370          if( !empty($position) )
 371          {
 372              $position = explode( ',', $position );
 373              $SQL->WHERE_and( 'link_position IN ( '.$DB->quote( $position ).' )' );
 374          }
 375          $SQL->ORDER_BY( $params['sql_order_by'] );
 376          $SQL->LIMIT( $limit );
 377  
 378          if( ! is_null( $file_type ) )
 379          { // Restrict the Links by File type
 380              $SQL->FROM_add( 'INNER JOIN T_files ON link_file_ID = file_ID' );
 381              $SQL->WHERE_and( 'file_type = '.$DB->quote( $file_type ).' OR file_type IS NULL' );
 382          }
 383  
 384          $LinkList->sql = $SQL->get();
 385  
 386          $LinkList->query( false, false, false, 'get_attachment_LinkList' );
 387  
 388          if( $LinkList->result_num_rows == 0 )
 389          { // Nothing found
 390              $LinkList = NULL;
 391          }
 392  
 393          return $LinkList;
 394      }
