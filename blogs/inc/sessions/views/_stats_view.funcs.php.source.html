 266  /**
 267   * Helper function for "Requested URI" column
 268   *
 269   * @param integer Blog ID
 270   * @param string Requested URI
 271   * @param integer Output string lenght
 272   * @param string Display
 273   * @param string Controller
 274   * @return string
 275   */
 276  function stats_format_req_URI( $hit_coll_ID, $hit_uri, $max_len = 40, $hit_disp = NULL, $hit_ctrl = NULL, $hit_action = NULL)
 277  {
 278      if( ! empty( $hit_coll_ID ) )
 279      {
 280          $BlogCache = & get_BlogCache();
 281          $tmp_Blog = & $BlogCache->get_by_ID( $hit_coll_ID );
 282          $full_url = $tmp_Blog->get_baseurl_root().$hit_uri;
 283      }
 284      else
 285      {
 286          $full_url = $hit_uri;
 287      }
 288  
 289      $int_search_uri = urldecode($hit_uri);
 290      if( ( evo_strpos( $int_search_uri , '?s=' ) !== false )
 291       || ( evo_strpos( $int_search_uri , '&s=' ) !== false ) )
 292      { // This is an internal search:
 293          preg_match( '~[?&]s=([^&#]*)~', $int_search_uri, $res );
 294          $hit_uri = sprintf( T_( 'Internal search: %s' ), $res[1] );
 295      }
 296      elseif( evo_strlen($hit_uri) > $max_len )
 297      {
 298          $hit_uri = '...'.evo_substr( $hit_uri, -$max_len );
 299      }
 300  
 301      if( $hit_disp != NULL || $hit_ctrl != NULL || $hit_action != NULL)
 302      {
 303          $hit_uri = '';
 304          if( $hit_disp != NULL )
 305          {
 306              $hit_uri .= '[disp=<a href="'.$full_url.'">'.$hit_disp.'</a>]';
 307          }
 308          if( $hit_ctrl != NULL )
 309          {
 310              $hit_uri .= ' [ctrl=<a href="'.$full_url.'">'.$hit_ctrl.'</a>]';
 311          }
 312          if( $hit_action != NULL )
 313          {
 314              $hit_uri .= ' [action=<a href="'.$full_url.'">'.$hit_action.'</a>]';
 315          }
 316  
 317          return $hit_uri;
 318      }
 319  
 320      return '<a href="'.$full_url.'">'.$hit_uri.'</a>';
 321  }

 530  /**
 531   * Get status code of IP range by IP address
 532   *
 533   * @param string IP address in format xxx.xxx.xxx.xxx
 534   * @return string Status value as it is stored in DB
 535   */
 536  function hit_iprange_status( $IP_address )
 537  {
 538      global $DB, $hit_iprange_statuses_cache;
 539  
 540      $IP_address = ip2int( $IP_address );
 541  
 542      if( ! is_array( $hit_iprange_statuses_cache ) )
 543      { // Initialize it only first time
 544          $SQL = new SQL();
 545          $SQL->SELECT( 'aipr_status AS status, aipr_IPv4start AS start, aipr_IPv4end AS end' );
 546          $SQL->FROM( 'T_antispam__iprange' );
 547          $hit_iprange_statuses_cache = $DB->get_results( $SQL->get() );
 548      }
 549  
 550      // Use this empty value for IPs without detected ranges in DB
 551      $ip_range_status = '';
 552  
 553      // Find status in the cache
 554      foreach( $hit_iprange_statuses_cache as $hit_iprange_status )
 555      {
 556          if( $IP_address >= $hit_iprange_status->start &&
 557              $IP_address <= $hit_iprange_status->end )
 558          { // IP is detected in this range
 559              $ip_range_status = $hit_iprange_status->status;
 560              break;
 561          }
 562      }
 563  
 564      return $ip_range_status;
 565  }
