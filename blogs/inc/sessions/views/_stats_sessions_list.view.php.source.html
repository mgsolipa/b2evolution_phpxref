 117  function display_sess_length( $sess_ID, $sess_length )
 118  {
 119      $result = '';
 120      $second = $sess_length % 60;
 121      $sess_length = ( $sess_length - $second ) / 60;
 122      $minute = $sess_length % 60;
 123      $sess_length = ( $sess_length - $minute ) / 60;
 124      $hour = $sess_length % 24;
 125      $day = ( $sess_length - $hour ) / 24;
 126  
 127      if( $day > 0 )
 128      {
 129          $result = sprintf( ( ( $day > 1 ) ? T_( '%d days' ) : T_( '%d day' ) ), $day ).' ';
 130      }
 131      if( $hour < 10 )
 132      {
 133          $hour = '0'.$hour;
 134      }
 135      if( $minute < 10 )
 136      {
 137          $minute = '0'.$minute;
 138      }
 139      if( $second < 10 )

 141          $second = '0'.$second;
 142      }
 143  
 144      $result .= $hour.':'.$minute.':'.$second;
 145      return stat_session_hits( $sess_ID, $result );
 146  }
