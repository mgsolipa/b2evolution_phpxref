 368       * Get & Analyze referer
 369       *
 370       * Due to potential non-thread safety with getenv() (fallback), we'd better do this early.
 371       *
 372       * referer_type: enum('search', 'blacklist', 'referer', 'direct'); 'spam' gets used internally
 373       */
 374  	function detect_referer( $referer = NULL )
 375      {
 376          global $Debuglog, $debug;
 377          global $self_referer_list, $SpecialList;  // used to detect $referer_type
 378          global $skins_path, $siteskins_path;
 379          global $Settings;
 380  
 381          if( isset($referer) )
 382          {
 383              $this->referer = $referer;
 384          }
 385          else
 386          {    // Get referer from HTTP request
 387              $this->referer = $this->get_referer();
 388          }
 389  
 390  
 391          if( empty($this->referer) )
 392          {    // NO referer
 393              // This type may be superseeded and set to 'admin'
 394              if( ! $this->detect_admin_page() )
 395              {    // Not an admin page:
 396                  $this->referer_type = 'direct';
 397              }
 398              else
 399              {
 400                  if (empty($this->hit_type))
 401                  {
 402                      $this->hit_type = 'admin';
 403                  }
 404                  $this->referer_type = 'direct';
 405              }
 406              return;
 407          }
 408  
 409  
 410          // ANALYZE referer...
 411  
 412  
 413          // Check self referer list, see {@link $self_referer_list}
 414          // fplanque: we log these (again), because if we didn't we woudln't detect
 415          // reloads on these... and that would be a problem!
 416          foreach( $self_referer_list as $self_referer )
 417          {
 418              $pos = strpos( $this->referer, $self_referer );
 419              // If not starting within in the first 12 chars it's probably an url param as in &url=http://this_blog.com
 420              if( $pos !== false && $pos <= 12
 421                  && ! ($debug && strpos( $this->referer, '/search.html' ) ) ) // search simulation
 422              {
 423                  // This type may be superseeded by admin page
 424                  if( ! $this->detect_admin_page() )
 425                  {    // Not an admin page:
 426                      $Debuglog->add( 'Hit: detect_referer(): self referer ('.$self_referer.')', 'request' );
 427                      $this->referer_type = 'self';
 428                  }
 429                  else
 430                  {
 431                      if (empty($this->hit_type))
 432                      {
 433                          $this->hit_type = 'admin';
 434                      }
 435                      $this->referer_type = 'self';
 436                  }
 437                  return;
 438              }
 439          }
 440  
 441  
 442          // Check Special list, see {@link $SpecialList}
 443          // NOTE: This is NOT the antispam!!
 444          // fplanque: we log these (again), because if we didn't we woudln't detect
 445          // reloads on these... and that would be a problem!
 446          foreach( $SpecialList as $lSpeciallist )
 447          {
 448              $pos = strpos( $this->referer, $lSpeciallist );
 449              // If not starting within in the first 12 chars it's probably an url param as in &url=http://this_blog.com
 450              if( $pos !== false && $pos <= 12 )
 451              {
 452                  // This type may be superseeded by admin page
 453                  // fp> 2009-05-10: because of the 12 char limit above the following is probably no longer needed. Please enable it back if anyone has a problem with admin being detected as blacklist
 454                  // if( ! $this->detect_admin_page() )
 455                  {    // Not an admin page:
 456                      $Debuglog->add( 'Hit: detect_referer(): blacklist ('.$lSpeciallist.')', 'request' );
 457                      $this->referer_type = 'special';
 458                  }
 459                  return;
 460              }
 461          }
 462  
 463  
 464          // Check if the referer is valid and does not match the antispam blacklist:
 465          // NOTE: requests to admin pages should not arrive here, because they should be matched above through $self_referer_list!
 466          load_funcs('_core/_url.funcs.php');
 467          if( $error = validate_url( $this->referer, 'commenting' ) )
 468          { // This is most probably referer spam!!
 469              $Debuglog->add( 'Hit: detect_referer(): '.$error.' (SPAM)', 'hit');
 470              $this->referer_type = 'spam';
 471  
 472              if( $Settings->get('antispam_block_spam_referers') )
 473              { // In order to preserve server resources, we're going to stop processing immediatly (no logging)!!
 474                  require $siteskins_path.'_403_referer_spam.main.php';    // error & exit
 475                  exit(0); // just in case.
 476                  // THIS IS THE END!!
 477              }
 478  
 479              return; // type "spam"
 480          }
 481  
 482  
 483          // Is the referer a search engine?
 484          if( $this->is_search_referer($this->referer) )
 485          {
 486              $Debuglog->add( 'Hit: detect_referer(): search engine', 'request' );
 487              $this->referer_type = 'search';
 488              return;
 489          }
 490  
 491          $this->referer_type = 'referer';
 492      }

 495      /**
 496       * Get referer_domain_ID (ID of the referer in T_basedomains).
 497       *
 498       * @return integer (may be NULL, but should never).
 499       */
 500  	function get_referer_domain_ID()
 501      {
 502          if( ! isset($this->referer_domain_ID) )
 503          {
 504              global $DB;
 505              // Check if we know the base domain:
 506              $referer_basedomain = get_base_domain($this->referer);
 507              if( $referer_basedomain )
 508              {    // This referer has a base domain
 509                  // Check if we have met it before:
 510                  $hit_basedomain = $DB->get_row( '
 511                      SELECT dom_ID
 512                          FROM T_basedomains
 513                       WHERE dom_name = '.$DB->quote($referer_basedomain) );
 514                  if( !empty( $hit_basedomain->dom_ID ) )
 515                  {    // This basedomain has visited before:
 516                      $this->referer_domain_ID = $hit_basedomain->dom_ID;
 517                      // fp> The blacklist handling that was here made no sense.
 518                  }
 519                  else
 520                  {    // This is the first time this base domain visits:
 521  
 522                      // The INSERT below can fail, probably if we get two simultaneous hits (seen in the demo logfiles)
 523                      if ($this->agent_type == 'robot' || $this->hit_type == 'rss')
 524                      {
 525                          $this->dom_type = 'aggregator';
 526                      }
 527                      elseif($this->referer_type == 'search')
 528                      {
 529                          $this->dom_type = 'searcheng';
 530                      }
 531                      elseif($this->referer_type == 'referer' || $this->referer_type == 'self')
 532                      {
 533                          $this->dom_type = 'normal';
 534                      }
 535  
 536  
 537                      $DB->save_error_state();
 538  
 539                      if( $DB->query( '
 540                          INSERT INTO T_basedomains( dom_name, dom_type)
 541                              VALUES( '.$DB->quote($referer_basedomain).', '.$DB->quote($this->dom_type).' )' ) )
 542                      { // INSERTed ok:
 543                          $this->referer_domain_ID = $DB->insert_id;
 544                      }
 545                      else
 546                      { // INSERT failed: see, try to select again (may become/stay NULL)
 547                          $this->referer_domain_ID = $DB->get_var( '
 548                              SELECT dom_ID
 549                                  FROM T_basedomains
 550                               WHERE dom_name = '.$DB->quote($referer_basedomain) );
 551                      }
 552  
 553                      $DB->restore_error_state();
 554                  }
 555              }
 556          }
 557          return $this->referer_domain_ID;
 558      }

 672                  $this->agent_type = 'browser';

 676                  $this->is_opera = 1;
 677                  $this->agent_name = 'opera';
 678                  $this->agent_type = 'browser';
 679              }

 683                  $this->agent_name = 'nav4';

 735      /**
 736       * Log a hit on a blog page / rss feed.
 737       *
 738       * This function should be called at the end of the page, otherwise if the page
 739       * is displaying previous hits, it may display the current one too.
 740       *
 741       * The hit will not be logged in special occasions, see {@link $ignore} and {@link is_good_hit()}.
 742       *
 743       * It will call {@link Hitlist::dbprune()} to do the automatic pruning of old hits in case
 744       * of auto_prune_stats_mode == "page".
 745       *
 746       * @return boolean true if the hit gets logged; false if not
 747       */
 748  	function log()
 749      {
 750          global $Plugins;
 751  
 752          if( $this->logged )
 753          {    // Already logged, don't log twice:
 754              return false;
 755          }
 756  
 757          // Remember we have already attempted to log:
 758          $this->logged = true;
 759  
 760          // Check if this hit should be logged:
 761          if( ! $this->should_be_logged() )
 762          {
 763              return false;
 764          }
 765  
 766          if( $Plugins->trigger_event_first_true('AppendHitLog', array( 'Hit' => &$this ) ) )
 767          {    // Plugin has handled recording
 768              return true;
 769          }
 770  
 771          // Check if this hit should STILL be logged after plugin call:
 772          if( ! $this->should_be_logged() )
 773          {
 774              return false;
 775          }
 776  
 777          // Record the HIT now:
 778          $this->record_the_hit();
 779  
 780          return true;
 781      }
