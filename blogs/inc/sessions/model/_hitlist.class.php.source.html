  54  	function delete( $hit_ID )

  62      /**
  63       * Delete all hits for a specific date
  64       *
  65       * @static
  66       * @param int unix timestamp to delete hits for
  67       * @return mixed Return value of {@link DB::query()}
  68       */
  69  	function prune( $date )
  70      {
  71          global $DB;
  72  
  73          $iso_date = date ('Y-m-d', $date);
  74          $sql = "
  75              DELETE FROM T_hitlog
  76               WHERE DATE_FORMAT(hit_datetime,'%Y-%m-%d') = '$iso_date'";
  77  
  78          return $DB->query( $sql, 'Prune hits for a specific date' );
  79      }

  82      /**
  83       * Change type for a hit
  84       *
  85       * @static
  86       * @param int ID to change
  87       * @param string new type, must be valid ENUM for hit_referer_type field
  88       * @return mixed Return value of {@link DB::query()}
  89       */
  90  	function change_type( $hit_ID, $type )
  91      {
  92          global $DB;
  93  
  94          $sql = '
  95                  UPDATE T_hitlog
  96                     SET hit_referer_type = '.$DB->quote($type).",
  97                         hit_datetime = hit_datetime " /* prevent mySQL from updating timestamp */ ."
  98                   WHERE hit_ID = $hit_ID";
  99          return $DB->query( $sql, 'Change type for a specific hit' );
 100      }

 103      /**
 104       * Auto pruning of old stats.
 105       *
 106       * It uses a general setting to store the day of the last prune, avoiding multiple prunes per day.
 107       * fplanque>> Check: How much faster is this than DELETING right away with an INDEX on the date field?
 108       *
 109       * Note: we're using {@link $localtimenow} to log hits, so use this for pruning, too.
 110       *
 111       * NOTE: do not call this directly, but only in conjuction with auto_prune_stats_mode.
 112       *
 113       * @static
 114       * @return array array(
 115       *   'result'  => 'error' | 'ok'
 116       *   'message' => Message of the error or result data
 117       * )
 118       */
 119  	function dbprune()
 120      {
 121          /**
 122           * @var DB
 123           */
 124          global $DB;
 125          global $Debuglog, $Settings, $localtimenow;
 126          global $Plugins, $Messages;
 127  
 128          // Prune when $localtime is a NEW day (which will be the 1st request after midnight):
 129          $last_prune = $Settings->get( 'auto_prune_stats_done' );
 130          if( $last_prune >= date( 'Y-m-d', $localtimenow ) && $last_prune <= date( 'Y-m-d', $localtimenow + 86400 ) )
 131          { // Already pruned today (and not more than one day in the future -- which typically never happens)
 132              $message = T_('Pruning has already been done today');
 133              $Messages->add( $message, 'error' );
 134              return array(
 135                      'result'  => 'error',
 136                      'message' => $message
 137                  );
 138          }
 139  
 140          // Init Timer for hitlist
 141          // Note: Don't use a global $Timer because it works only in debug mode
 142          load_class( '_core/model/_timer.class.php', 'Timer' );
 143          $hitlist_Timer = new Timer( 'prune_hits' );
 144  
 145          $time_prune_before = ( $localtimenow - ( $Settings->get( 'auto_prune_stats' ) * 86400 ) ); // 1 day = 86400 seconds
 146  
 147          $hitlist_Timer->start( 'hitlog' );
 148          $hitlog_rows_affected = $DB->query( "
 149              DELETE FROM T_hitlog
 150              WHERE hit_datetime < '".date( 'Y-m-d', $time_prune_before )."'", 'Autopruning hit log' );
 151          $hitlist_Timer->stop( 'hitlog' );
 152          $Debuglog->add( 'Hitlist::dbprune(): autopruned '.$hitlog_rows_affected.' rows from T_hitlog.', 'request' );
 153  
 154          // Prune sessions that have timed out and are older than auto_prune_stats
 155          $sess_prune_before = ($localtimenow - $Settings->get( 'timeout_sessions' ));
 156          $smaller_time = min( $sess_prune_before, $time_prune_before );
 157          // allow plugins to prune session based data
 158          $Plugins->trigger_event( 'BeforeSessionsDelete', $temp_array = array( 'cutoff_timestamp' => $smaller_time ) );
 159  
 160          $hitlist_Timer->start( 'sessions' );
 161          $sessions_rows_affected = $DB->query( 'DELETE FROM T_sessions WHERE sess_lastseen_ts < '.$DB->quote( date( 'Y-m-d H:i:s', $smaller_time ) ), 'Autoprune sessions' );
 162          $hitlist_Timer->stop( 'sessions' );
 163          $Debuglog->add( 'Hitlist::dbprune(): autopruned '.$sessions_rows_affected.' rows from T_sessions.', 'request' );
 164  
 165          // Prune non-referrered basedomains (where the according hits got deleted)
 166          // BUT only those with unknown dom_type/dom_status, because otherwise this
 167          //     info is useful when we get hit again.
 168          // Note: MySQL server version >= 4 is required for multi-table deletes, but v 4.1 is now a requirement for b2evolution:
 169          $hitlist_Timer->start( 'basedomains' );
 170          $basedomains_rows_affected = $DB->query( "
 171              DELETE T_basedomains
 172                FROM T_basedomains LEFT JOIN T_hitlog ON hit_referer_dom_ID = dom_ID
 173               WHERE hit_referer_dom_ID IS NULL
 174               AND dom_type = 'unknown'
 175               AND dom_status = 'unknown'" );
 176          $hitlist_Timer->stop( 'basedomains' );
 177          $Debuglog->add( 'Hitlist::dbprune(): autopruned '.$basedomains_rows_affected.' rows from T_basedomains.', 'request' );
 178  
 179          // Optimizing tables
 180          $hitlist_Timer->start( 'optimize' );
 181          $DB->query('OPTIMIZE TABLE T_hitlog');
 182          $DB->query('OPTIMIZE TABLE T_sessions');
 183          $DB->query('OPTIMIZE TABLE T_basedomains');
 184          $hitlist_Timer->stop( 'optimize' );
 185  
 186          // Stop total hitlist timer
 187          $hitlist_Timer->stop( 'prune_hits' );
 188  
 189          $Settings->set( 'auto_prune_stats_done', date( 'Y-m-d H:i:s', $localtimenow ) ); // save exact datetime
 190          $Settings->dbupdate();
 191  
 192          $Messages->add( T_('The old hits & sessions have been pruned.'), 'success' );
 193          return array(
 194                  'result'  => 'ok',
 195                  'message' => T_('The following data have been autopruned:')."\n"
 196                      .sprintf( T_( '%s rows from T_hitlog, Execution time: %s seconds' ), $hitlog_rows_affected, $hitlist_Timer->get_duration( 'hitlog' ) )."\n"
 197                      .sprintf( T_( '%s rows from T_sessions, Execution time: %s seconds' ), $sessions_rows_affected, $hitlist_Timer->get_duration( 'sessions' ) )."\n"
 198                      .sprintf( T_( '%s rows from T_basedomains, Execution time: %s seconds' ), $basedomains_rows_affected, $hitlist_Timer->get_duration( 'basedomains' ) )."\n"
 199                      .sprintf( T_( 'Execution time for tables optimizing: %s seconds' ), $hitlist_Timer->get_duration( 'optimize' ) )."\n"
 200                      .sprintf( T_( 'Total execution time: %s seconds' ), $hitlist_Timer->get_duration( 'prune_hits' ) )
 201              );
 202      }
 203  }
