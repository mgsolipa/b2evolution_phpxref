  58      if( !$current_User->check_perm( 'stats', 'view' ) )
  59      {    // Current user has no permission to view all stats (aggregated stats)

  63      /**
  64       * View funcs

  74      $tab = param( 'tab', 'string', 'summary', true );
  75      $tab3 = param( 'tab3', 'string', '', true );
  76  
  77      switch( $tab )
  78      {
  79          case 'other':
  80              $preset_results_title = T_('Direct browser hits');
  81              $preset_referer_type = 'direct';
  82              $preset_agent_type = 'browser';

  96              {
  97                  $preset_results_title = T_('Search hits');
  98                  $preset_referer_type = 'search';

 229      $resuts_param_prefix = 'hits_';

 266  function refererList(
 267      $howMany = 5,
 268      $visitURL = '',
 269      $disp_blog = 0,
 270      $disp_uri = 0,
 271      $type = "'referer'",        // was: 'referer' normal refer, 'invalid', 'badchar', 'blacklist', 'rss', 'robot', 'search'
 272                                                      // new: 'search', 'blacklist', 'referer', 'direct', ('spam' but spam is not logged)
 273      $groupby = '',     // dom_name
 274      $blog_ID = '',
 275      $get_total_hits = false, // Get total number of hits (needed for percentages)
 276      $get_user_agent = false ) // Get the user agent
 277  {
 278      global $DB, $res_stats, $stats_total_hits, $ReqURI;
 279  
 280      if( strpos( $type, "'" ) !== 0 )
 281      { // no quote at position 0
 282          $type = "'".$type."'";
 283      }
 284  
 285      //if no visitURL, will show links to current page.
 286      //if url given, will show links to that page.
 287      //if url="global" will show links to all pages
 288      if (!$visitURL)
 289      {
 290          $visitURL = $ReqURI;
 291      }
 292  
 293      if( $groupby == '' )
 294      { // No grouping:
 295          $sql = 'SELECT hit_ID, UNIX_TIMESTAMP(hit_datetime) AS hit_datetime, hit_referer, dom_name';
 296      }
 297      else
 298      { // group by
 299          if( $groupby == 'baseDomain' )
 300          { // compatibility HACK!
 301              $groupby = 'dom_name';
 302          }
 303          $sql = 'SELECT COUNT(*) AS totalHits, hit_referer, dom_name';
 304      }
 305      if( $disp_blog )
 306      {
 307          $sql .= ', hit_coll_ID';
 308      }
 309      if( $disp_uri )
 310      {
 311          $sql .= ', hit_uri';
 312      }
 313      if( $get_user_agent )
 314      {
 315          $sql .= ', agnt_signature';
 316      }
 317  
 318      $sql_from_where = "
 319                FROM T_hitlog LEFT JOIN T_basedomains ON dom_ID = hit_referer_dom_ID
 320               WHERE hit_referer_type IN (".$type.")
 321                 AND hit_agent_type = 'browser'";
 322      if( !empty($blog_ID) )
 323      {
 324          $sql_from_where .= " AND hit_coll_ID = '".$blog_ID."'";
 325      }
 326      if ( $visitURL != 'global' )
 327      {
 328          $sql_from_where .= " AND hit_uri = '".$DB->escape($visitURL, 0, 250)."'";
 329      }
 330  
 331      $sql .= $sql_from_where;
 332  
 333      if( $groupby == '' )
 334      { // No grouping:
 335          $sql .= ' ORDER BY hit_ID DESC';
 336      }
 337      else
 338      { // group by
 339          $sql .= " GROUP BY ".$groupby." ORDER BY totalHits DESC";
 340      }
 341      $sql .= ' LIMIT '.$howMany;
 342  
 343      $res_stats = $DB->get_results( $sql, ARRAY_A );
 344  
 345      if( $get_total_hits )
 346      { // we need to get total hits
 347          $sql = 'SELECT COUNT(*) '.$sql_from_where;
 348          $stats_total_hits = $DB->get_var( $sql );
 349      }
 350      else
 351      { // we're not getting total hits
 352          $stats_total_hits = 1;        // just in case some tries a percentage anyway (avoid div by 0)
 353      }
 354  
 355  }

 478   * Displays keywords used for search leading to this page

 480  function stats_search_keywords( $keyphrase, $length = 45 )
 481  {
 482      global $evo_charset;
 483  
 484      if( empty( $keyphrase ) )
 485      {
 486          return '<span class="note">['.T_('n.a.').']</span>';
 487      }
 488  
 489      // Save original string
 490      $keyphrase_orig = $keyphrase;
 491  
 492      $keyphrase = strmaxlen($keyphrase, $length, '...', 'raw');
 493  
 494      // Convert keyword encoding, some charsets are supported only in PHP 4.3.2 and later.
 495      // This fixes encoding problem for Cyrillic keywords
 496      // See http://forums.b2evolution.net/viewtopic.php?t=17431
 497      $keyphrase = evo_htmlentities( $keyphrase, ENT_COMPAT, $evo_charset );
 498  
 499      return '<span title="'.format_to_output( $keyphrase_orig, 'htmlattr' ).'">'.$keyphrase.'</span>';
 500  }
