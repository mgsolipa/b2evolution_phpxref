 110      /**
 111       * Load data from Request form fields.
 112       *
 113       * @return boolean true if loaded data seems valid.
 114       */
 115  	function load_from_Request()
 116      {
 117          // Category
 118          param( 'goal_gcat_ID', 'integer', true );
 119          param_check_not_empty( 'goal_gcat_ID', T_('Please select a category.') );
 120          $this->set_from_Request( 'gcat_ID' );
 121  
 122          // Name
 123          $this->set_string_from_param( 'name', true );
 124  
 125          // Key
 126          $this->set_string_from_param( 'key', true );
 127  
 128          // Temporary Redirection URL:
 129          $this->set_string_from_param( 'temp_redir_url' );
 130  
 131          // Normal Redirection URL:
 132          param( 'goal_redir_url', 'string' );
 133          if( $this->get( 'temp_redir_url' ) != '' )
 134          { // Normal Redirection URL is required when Temporary Redirection URL is not empty
 135              param_check_not_empty( 'goal_redir_url', T_('Please enter Normal Redirection URL.') );
 136          }
 137          $this->set_from_Request( 'redir_url' );
 138  
 139          if( $this->get( 'temp_redir_url' ) != '' && $this->get( 'temp_redir_url' ) == $this->get( 'redir_url' ) )
 140          { // Compare normal and temp urls
 141              param_error( 'goal_temp_redir_url', T_( 'Temporary Redirection URL should not be equal to Normal Redirection URL' ) );
 142              param_error( 'goal_redir_url', NULL, '' );
 143          }
 144  
 145          // Temporary Start
 146          $temp_start_date = param_date( 'goal_temp_start_date', T_('Please enter a valid date.'), false );
 147          if( ! empty( $temp_start_date ) )
 148          {
 149              $temp_start_time = param( 'goal_temp_start_time', 'string' );
 150              $temp_start_time = empty( $temp_start_time ) ? '00:00:00' : param_time( 'goal_temp_start_time' );
 151              $this->set( 'temp_start_ts', form_date( $temp_start_date, $temp_start_time ) );
 152          }
 153          else
 154          {
 155              $this->set( 'temp_start_ts', NULL );
 156          }
 157  
 158          // Temporary End
 159          $temp_end_date = param_date( 'goal_temp_end_date', T_('Please enter a valid date.'), false );
 160          if( ! empty( $temp_end_date ) )
 161          {
 162              $temp_end_time = param( 'goal_temp_end_time', 'string' );
 163              $temp_end_time = empty( $temp_end_time ) ? '00:00:00' : param_time( 'goal_temp_end_time' );
 164              $this->set( 'temp_end_ts', form_date( $temp_end_date, $temp_end_time ) );
 165          }
 166          else
 167          {
 168              $this->set( 'temp_end_ts', NULL );
 169          }
 170  
 171          if( $this->get( 'temp_start_ts' ) !== NULL && $this->get( 'temp_end_ts' ) !== NULL &&
 172              strtotime( $this->get( 'temp_start_ts' ) ) >= strtotime( $this->get( 'temp_end_ts' ) ) )
 173          { // Compare Start and End dates
 174              param_error( 'goal_temp_start_date', NULL, '' );
 175              param_error( 'goal_temp_start_time', NULL, '' );
 176              param_error( 'goal_temp_end_date', NULL, '' );
 177              param_error( 'goal_temp_end_time', T_( 'Temporary Start Date/Time should not be greater than Temporary End Date/Time' ) );
 178          }
 179  
 180          // Default value:
 181          param( 'goal_default_value', 'string' );
 182          param_check_decimal( 'goal_default_value', T_('Default value must be a number.') );
 183          $this->set_from_Request( 'default_value', 'goal_default_value', true  );
 184  
 185          // Notes
 186          param( 'goal_notes', 'text' );
 187          $this->set_from_Request( 'notes', 'goal_notes' );
 188  
 189          return ! param_errors_detected();
 190      }

 239      /**
 240       * Get redirection URL that is active now
 241       *
 242       * @param array Params
 243       * @return string $this->redir_url or $this->temp_redir_url
 244       */
 245  	function get_active_url( $params = array() )
 246      {
 247          $params = array_merge( array(
 248              'before_temp' => '',
 249              'after_temp'  => '',
 250              ), $params );
 251  
 252          if( empty( $this->temp_redir_url ) )
 253          { // Use normal redirection URL when temporary redirection URL is empty
 254              return $this->redir_url;
 255          }
 256          else
 257          { // Check if we can use temporary redirection URL
 258              global $localtimenow;
 259  
 260              if( ( empty( $this->temp_start_ts ) || $this->temp_start_ts <= $localtimenow ) &&
 261                  ( empty( $this->temp_end_ts ) || $this->temp_end_ts >= $localtimenow ) )
 262              { // Use temporary redirection URL now
 263                  return $params['before_temp'].$this->temp_redir_url.$params['after_temp'];
 264              }
 265              else
 266              { // Temporary redirection URL is out date, Use normal URL now
 267                  return $this->redir_url;
 268              }
 269          }
 270      }

 273      /**
 274       * Record goal hit
 275       *
 276       * @param string Extra params, Use '#' to get params from the $_SERVER['QUERY_STRING']
 277       */
 278  	function record_hit( $extra_params = '#' )
 279      {
 280          global $DB, $Hit;
 281  
 282          if( $extra_params == '#' )
 283          { // Use standard extra params from query string
 284              if( isset( $_SERVER['QUERY_STRING'] ) )
 285              { // Set additional params in goal hit
 286                  $extra_params = '&'.$_SERVER['QUERY_STRING'].'&';
 287                  $extra_params = preg_replace( '/&key=[^&]+(&)?/i', '$1', $extra_params );
 288                  $extra_params = trim( $extra_params, '&' );
 289              }
 290              if( empty( $extra_params ) )
 291              { // No extra params
 292                  $extra_params = NULL;
 293              }
 294          }
 295  
 296          // We need to log the HIT now! Because we need the hit ID!
 297          $Hit->log();
 298  
 299          // Insert a goal hit:
 300          $DB->query( 'INSERT INTO T_track__goalhit( ghit_goal_ID, ghit_hit_ID, ghit_params )
 301              VALUES( '.$this->ID.', '.$Hit->ID.', '.$DB->quote( $extra_params ).' )',
 302              'Record goal hit' );
 303      }
