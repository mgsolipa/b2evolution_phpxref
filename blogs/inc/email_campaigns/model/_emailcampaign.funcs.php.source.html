  29  /**

  30   * Get number of users for newsletter from UserList filterset

  31   *

  32   * @return array 

  33   *         'all' - Number of accounts in filterset

  34   *         'active' - Number of active accounts in filterset

  35   *         'newsletter' - Number of active accounts which accept newsletter email

  37  function get_newsletter_users_numbers()
  38  {
  39      $numbers = array(
  40              'all'        => 0,
  41              'active'     => 0,
  42              'newsletter' => 0,
  43          );
  44  
  45      $users_IDs = get_filterset_user_IDs();
  46  
  47      if( count( $users_IDs ) )
  48      { // Found users in the filterset
  49          global $DB;
  50  
  51          $numbers['all'] = count( $users_IDs );
  52  
  53          // Get number of all active users

  54          $SQL = new SQL();
  55          $SQL->SELECT( 'COUNT( * )' );
  56          $SQL->FROM( 'T_users' );
  57          $SQL->WHERE( 'user_ID IN ( '.implode( ', ', $users_IDs ).' )' );
  58          $SQL->WHERE_and( 'user_status IN ( \'activated\', \'autoactivated\' )' );
  59          $numbers['active'] = $DB->get_var( $SQL->get() );
  60  
  61          // Get number of all active users which accept newsletter email

  62          $SQL = get_newsletter_users_sql( $users_IDs );
  63          $SQL->SELECT( 'COUNT( * )' );
  64          $numbers['newsletter'] = $DB->get_var( $SQL->get() );
  65      }
  66  
  67      return $numbers;
  68  }

  71  /**

  72   * Get SQL for active users which accept newsletter email

  73   *

  74   * @param array users IDs

  75   * @return object SQL

  76   */
  77  function get_newsletter_users_sql( $users_IDs )
  78  {
  79      global $Settings;
  80  
  81      $SQL = new SQL();
  82      $SQL->SELECT( 'u.user_ID' );
  83      $SQL->FROM( 'T_users u' );
  84      $SQL->FROM_add( 'LEFT OUTER JOIN T_users__usersettings us ON u.user_ID = us.uset_user_ID' );
  85      $SQL->FROM_add( 'AND us.uset_name = \'newsletter_news\'' );
  86      $SQL->WHERE( 'u.user_ID IN ( '.implode( ', ', $users_IDs ).' )' );
  87      $SQL->WHERE_and( 'u.user_status IN ( \'activated\', \'autoactivated\' )' );
  88      if( $Settings->get( 'def_newsletter_news' ) )
  89      {    // If General setting "newsletter_news" = 1 we also should include all users without defined user's setting "newsletter_news"
  90          $SQL->WHERE_and( '( us.uset_value = 1 OR us.uset_value IS NULL )' );
  91      }
  92      else
  93      {    // If General setting "newsletter_news" = 0 we take only users which accept newsletter email
  94          $SQL->WHERE_and( 'us.uset_value = 1' );
  95      }
  96  
  97      return $SQL;
  98  }

 176  /**

 177   * Get URL for current/next tab of edit campaign view

 178   *

 179   * @param string Current tab: 'info', 'html', 'text', 'send'

 180   * @param integer Campaign ID

 181   * @param string Type of tab: 'current', 'next'

 182   * @param string Glue

 183   * @return string URL

 184   */
 185  function get_campaign_tab_url( $current_tab, $campaign_ID, $type = 'current', $glue = '&' )
 186  {
 187      $modes = get_campaign_edit_modes( $campaign_ID, $glue );
 188  
 189      switch( $type )
 190      {
 191          case 'current':
 192              // Get URL of current tab

 193              if( !empty( $modes[ $current_tab ] ) )
 194              {
 195                  return $modes[ $current_tab ]['href'];
 196              }
 197              break;
 198  
 199          case 'next':
 200          default:
 201              // Get URL of next tab

 202              $this_tab = false;
 203              foreach( $modes as $tab_name => $tab_info )
 204              {
 205                  if( $this_tab )
 206                  { // We find URL for next tab
 207                      return $tab_info['href'];
 208                  }
 209                  if( $tab_name == $current_tab )
 210                  { // The next tab will be what we find
 211                      $this_tab = true;
 212                  }
 213              }
 214          break;
 215      }
 216  
 217      return '';
 218  }
