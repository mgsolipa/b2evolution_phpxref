  87  	function add_users()
  88      {
  89          global $DB;
  90  
  91          // Get user IDs from current filterset of users list

  92          $new_users_IDs = get_filterset_user_IDs();
  93  
  94          if( count( $new_users_IDs ) )
  95          { // Users are found in the filterset
  96  
  97              // Get all active users which accept newsletter email

  98              $new_users_SQL = get_newsletter_users_sql( $new_users_IDs );
  99              $new_users = $DB->get_col( $new_users_SQL->get() );
 100  
 101              // Remove the users which didn't accept email before

 102              $DB->query( 'DELETE FROM T_email__campaign_send
 103                  WHERE csnd_camp_ID = '.$DB->quote( $this->ID ).'
 104                   AND csnd_emlog_ID IS NULL' );
 105  
 106              // Get users which already accept newsletter email

 107              $old_users = $this->get_users( 'accept' );
 108  
 109              // Exclude old users from new users (To store value of csnd_emlog_ID)

 110              $new_users = array_diff( $new_users, $old_users );
 111  
 112              if( count( $new_users ) )
 113              { // Insert new users for this campaign
 114                  $insert_SQL = 'INSERT INTO T_email__campaign_send ( csnd_camp_ID, csnd_user_ID ) VALUES';
 115                  foreach( $new_users as $user_ID )
 116                  {
 117                      $insert_SQL .= "\n".'( '.$DB->quote( $this->ID ).', '.$DB->quote( $user_ID ).' ),';
 118                  }
 119                  $DB->query( substr( $insert_SQL, 0, -1 ) );
 120              }
 121          }

 125      /**

 126       * Get user IDs of this campaign

 127       *

 128       * @param string Type of users:

 129       *   'all'    - All users which assigned to this campaign

 130       *   'accept' - Users which already receive email newsletter

 131       *   'wait'   - Users which still didn't receive email by some reason (Probably their newsletter limit was full)

 132       * @return array user IDs

 133       */
 134  	function get_users( $type = 'all' )
 135      {
 136          global $DB;
 137  
 138          if( !is_null( $this->users ) )
 139          { // Get users from cache
 140              return $this->users[ $type ];
 141          }
 142  
 143          // Get users from DB

 144          $users_SQL = new SQL();
 145          $users_SQL->SELECT( 'csnd_user_ID, csnd_emlog_ID' );
 146          $users_SQL->FROM( 'T_email__campaign_send' );
 147          $users_SQL->WHERE( 'csnd_camp_ID = '.$DB->quote( $this->ID ) );
 148          $users = $DB->get_assoc( $users_SQL->get() );
 149  
 150          $this->users['all'] = array();
 151          $this->users['accept'] = array();
 152          $this->users['wait'] = array();
 153  
 154          foreach( $users as $user_ID => $emlog_ID )
 155          {
 156              $this->users['all'][] = $user_ID;
 157              if( $emlog_ID > 0 )
 158              { // This user already accepted newsletter email
 159                  $this->users['accept'][] = $user_ID;
 160              }
 161              else
 162              { // This user didn't still accept email
 163                  $this->users['wait'][] = $user_ID;
 164              }
 165          }
 166  
 167          return $this->users[ $type ];
 168      }

 206          { // Email HTML message

 208              $this->set_from_Request( 'email_html' );

 212          { // Email Plain Text message
 213              $this->set_from_Request( 'email_text' );
 214          }
 215

 297      /**

 298       * Send email newsletter for all users of this campaign

 299       *

 300       * @param boolean

 301       */
 302  	function send_all_emails( $display_messages = true )
 303      {
 304          global $DB, $localtimenow, $mail_log_insert_ID;
 305  
 306          // Send emails only for users which still don't accept emails

 307          $user_IDs = $this->get_users( 'wait' );
 308  
 309          if( empty( $user_IDs ) )
 310          { // No users, Exit here
 311              return;
 312          }
 313  
 314          $DB->begin();
 315  
 316          // Update date of sending

 317          $this->set( 'sent_ts', date( 'Y-m-d H:i:s', $localtimenow ) );
 318          $this->dbupdate();
 319  
 320          if( $display_messages )
 321          { // We need in this cache when display the messages
 322              $UserCache = & get_UserCache();
 323          }
 324  
 325          foreach( $user_IDs as $user_ID )
 326          {
 327              $result = $this->send_email( $user_ID );
 328  
 329              if( $result )
 330              { // Email newsletter was sent for user successfully
 331                  if( !empty( $mail_log_insert_ID ) )
 332                  { // ID of last inserted mail log is defined in function mail_log()
 333                      $DB->query( 'UPDATE T_email__campaign_send
 334                              SET csnd_emlog_ID = '.$DB->quote( $mail_log_insert_ID ).'
 335                          WHERE csnd_camp_ID = '.$DB->quote( $this->ID ).'
 336                              AND csnd_user_ID = '.$DB->quote( $user_ID ) );
 337  
 338                      // Update arrays where we store which users accepted email and who waiting it now

 339                      $this->users['accept'][] = $user_ID;
 340                      if( ( $wait_user_ID_key = array_search( $user_ID, $this->users['wait'] ) ) !== false )
 341                      {
 342                          unset( $this->users['wait'][ $wait_user_ID_key ] );
 343                      }
 344                  }
 345              }
 346  
 347              if( $display_messages )
 348              { // Print the messages
 349                  $User = & $UserCache->get_by_ID( $user_ID, false, false );
 350  
 351                  if( $result === true )
 352                  { // Success
 353                      echo sprintf( T_('Email was sent to user: %s'), $User->get_identity_link() ).'<br />';
 354                  }
 355                  else
 356                  { // Failed, Email was NOT sent
 357                      if( ! check_allow_new_email( 'newsletter_limit', 'last_newsletter', $user_ID ) )
 358                      { // Newsletter email is limited today for this user
 359                          echo '<span class="orange">'.sprintf( T_('User %s has already received max # of newsletters today.'), $User->get_identity_link() ).'</span><br />';
 360                      }
 361                      else
 362                      { // Another error
 363                          echo '<span class="red">'.sprintf( T_('Email was not sent to user: %s'), $User->get_identity_link() ).'</span><br />';
 364                      }
 365                  }
 366  
 367                  evo_flush();
 368              }
 369          }
 370  
 371          $DB->commit();
 372      }
