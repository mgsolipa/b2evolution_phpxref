  54  	function get_content()
  55      {
  56          $input = fopen("php://input", "rb");
  57          $temp = tmpfile();
  58          stream_copy_to_stream( $input, $temp );
  59          fclose( $input );
  60  
  61          fseek($temp, 0, SEEK_SET);
  62          $contents = '';
  63  
  64          load_funcs( 'tools/model/_system.funcs.php' );
  65          $memory_limit = system_check_memory_limit();
  66  
  67          while( ! feof( $temp ) )
  68          {
  69              $curr_mem_usage = memory_get_usage( true );
  70              if( ( $memory_limit - $curr_mem_usage ) < 8192 )
  71              { // Don't try to load the next portion of image into the memory because it would cause 'Allowed memory size exhausted' error
  72                  fclose( $temp );
  73                  return false;
  74              }
  75              $contents .= fread( $temp, 8192 );
  76          }
  77          fclose( $temp );
  78          return $contents;
