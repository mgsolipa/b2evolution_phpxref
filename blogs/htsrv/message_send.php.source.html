  98  elseif( $antispam_on_message_form && antispam_check( $message ) )

 105  if( !empty( $comment_id ) || !empty( $post_id ) )

 107      $Blog = & $BlogCache->get_by_ID( $blog );    // Required
 108  }
 109  else
 110  {
 111      $Blog = & $BlogCache->get_by_ID( $blog, true, false );    // Optional
 112  }

 115  if( ! empty( $recipient_id ) )

 274      // exited here
 275  }

 282  $unsaved_message_params[ 'message' ] = $original_content;
 283  save_message_params_to_session( $unsaved_message_params );
 284  
 285  if( param_errors_detected() || empty( $redirect_to ) )
 286  {
 287      $redirect_to = url_add_param( $Blog->gen_blogurl(), 'disp=msgform&recipient_id='.$recipient_id );
 288  }
 289  header_redirect( $redirect_to );
