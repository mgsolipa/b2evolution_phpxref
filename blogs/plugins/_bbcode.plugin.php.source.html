 186      /**
 187       * Parse BB code
 188       *   ( The main purpose of this function is to parsing multilevel lists tags,
 189       *     i.e. when one [list] is contained inside other [list] )
 190       *
 191       * @param string Content
 192       * @param array Search list
 193       * @param array Replace list
 194       * @return string Content
 195       */
 196  	function parse_bbcode( $content, $search_list, $replace_list )
 197      {
 198          if( empty( $content ) )
 199          { // No content, Exit here
 200              return $content;
 201          }
 202  
 203          $complex = array();
 204          $simple = array();
 205  
 206          foreach( $search_list as $i => $search_text )
 207          { // Split the masks to separate things
 208              preg_match( '#(\\\\\[.+\])\(.+\)(\\\\\[.+\])#is', $search_text, $search_tags );
 209              preg_match( '#(<.+>)\$.+(<.+>)#is', $replace_list[ $i ], $replace_tags );
 210              if( !empty( $search_tags ) )
 211              {
 212                  $complex[] = array(
 213                          's_full'  => $search_text,
 214                          's_start' => '!^'.$search_tags[1].'!is',
 215                          's_end'   => '!^'.$search_tags[2].'!is',
 216                          'r_full'  => $replace_list[ $i ],
 217                          'r_start' => $replace_tags[1],
 218                          'r_end'   => $replace_tags[2],
 219                      );
 220              }
 221              else
 222              {
 223                  $simple[] = array(
 224                          's_full' => $search_text,
 225                          'r_full' => $replace_list[ $i ],
 226                      );
 227              }
 228          }
 229          //pre_dump($complex);
 230  
 231          foreach( $simple as $tag_info )
 232          { // Make simple replacements
 233              $content = preg_replace( $tag_info['s_full'], $tag_info['r_full'], $content );
 234          }
 235  
 236          // Complex parsing with multilevel tags structure
 237          $c_pos = 0;
 238          $content_length = strlen( $content );
 239          $opened_tags = array();
 240          $new_content = $content;
 241          while( $c_pos < $content_length - 2 )
 242          {
 243              if( $content[ $c_pos ].$content[ $c_pos + 1 ] == '[/' )
 244              { // Closed tag
 245                  if( !empty( $opened_tags ) )
 246                  {    // Get closed tag that was opened as last
 247                      $sub_content = substr( $content, $c_pos );
 248                      $closed_tag_info = $complex[ $opened_tags[ count( $opened_tags ) - 1 ] ];
 249                      if( preg_match( $closed_tag_info['s_end'], $sub_content ) )
 250                      {
 251                          $first_part = str_replace( $sub_content, '', $new_content );
 252                          $sub_content = preg_replace( $closed_tag_info['s_end'], $closed_tag_info['r_end'], $sub_content );
 253                          $new_content = $first_part.$sub_content;
 254                          //pre_dump( 'CLOSE', $opened_tags, $closed_tag_info['r_end'], $closed_tag_info['s_end'], $new_content );
 255                          // Remove this tag from opened tags array
 256                          array_pop( $opened_tags );
 257                      }
 258                  }
 259              }
 260              elseif( $content[ $c_pos ] == '[' )
 261              { // Opened tag
 262                  $sub_content = substr( $content, $c_pos );
 263                  foreach( $complex as $tag_num => $tag_info )
 264                  {
 265                      if( preg_match( $tag_info['s_start'], $sub_content ) )
 266                      {
 267                          $first_part = str_replace( $sub_content, '', $new_content );
 268                          $sub_content = preg_replace( $tag_info['s_start'], $tag_info['r_start'], $sub_content );
 269                          $new_content = $first_part.$sub_content;
 270                          // Add this tag to know what tag was opened as last
 271                          $opened_tags[] = $tag_num;
 272                          //pre_dump( 'OPEN', $opened_tags, $tag_info['s_start'], $new_content );
 273                          break;
 274                      }
 275                  }
 276              }
 277              $c_pos++;
 278          }
 279          //pre_dump($new_content);
 280  
 281          $new_content = $this->parse_anchor_links( $new_content );
 282  
 283          return $new_content;
 284      }

 287      /**
 288       * Parse anchor links, Set absolute path for each link with relative anchor like <a href="#">
 289       *
 290       * @param string Content
 291       * @return string Content
 292       */
 293  	function parse_anchor_links( $content )
 294      {
 295          if( preg_match_all( '/ href="#(c|p)([0-9]+)"/i', $content, $matches ) )
 296          {
 297              $CommentCache = & get_CommentCache();
 298              $ItemCache = & get_ItemCache();
 299              foreach( $matches[0] as $m => $full_match )
 300              {
 301                  $object_ID = $matches[2][$m];
 302                  $new_url = '';
 303                  switch( $matches[1][$m] )
 304                  { // Object type:
 305                      case 'p':
 306                          // Item
 307                          if( $Item = & $ItemCache->get_by_ID( $object_ID, false, false ) )
 308                          { // Replace anchor url with item permanent url
 309                              $new_url = $Item->get_permanent_url().'#p'.$object_ID;
 310                          }
 311                          break;
 312  
 313                      case 'c':
 314                          // Comment
 315                          if( $Comment = & $CommentCache->get_by_ID( $object_ID, false, false ) )
 316                          { // Replace anchor url with comment permanent url
 317                              $new_url = $Comment->get_permanent_url();
 318                          }
 319                          break;
 320  
 321                      default:
 322                          // Incorrect object type, Skip this url
 323                          continue;
 324                  }
 325                  if( !empty( $new_url ) )
 326                  { // Replace relative anchor url with new absolute url
 327                      $content = str_replace( $full_match, ' href="'.$new_url.'"', $content );
 328                  }
 329              }
 330          }
 331  
 332          return $content;
 333      }

 582  		function bbInsertTag( myField, i )
 583          {
 584              // we need to know if something is selected.
 585              // First, ask plugins, then try IE and Mozilla.
 586              var sel_text = b2evo_Callbacks.trigger_callback("get_selected_text_for_"+myField.id);
 587              var focus_when_finished = false; // used for IE
 588  
 589              if( sel_text == null )
 590              { // detect selection:
 591                  //IE support
 592                  if(document.selection)
 593                  {
 594                      myField.focus();
 595                      var sel = document.selection.createRange();
 596                      sel_text = sel.text;
 597                      focus_when_finished = true;
 598                  }
 599                  //MOZILLA/NETSCAPE support
 600                  else if(myField.selectionStart || myField.selectionStart == '0')
 601                  {
 602                      var startPos = myField.selectionStart;
 603                      var endPos = myField.selectionEnd;
 604                      sel_text = (startPos != endPos);
 605                  }
 606              }
 607

 689  	function prepare_search_list( $setting_name, $Blog )
 690      {
 691          $search_list = explode( "\n", str_replace( "\r", '', $this->get_coll_setting( $setting_name, $Blog ) ) );
 692  
 693          foreach( $search_list as $l => $line )

 695              $line = explode( ' ', $line, 2 );

 699                  unset( $search_list[ $l ] );
