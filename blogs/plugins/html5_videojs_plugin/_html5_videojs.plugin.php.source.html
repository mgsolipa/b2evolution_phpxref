 129      /**
 130       * Event handler: Called when displaying item attachment.
 131       *
 132       * @param array Associative array of parameters. $params['File'] - attachment, $params['data'] - output
 133       * @param boolean TRUE - when render in comments
 134       * @return boolean true if plugin rendered this attachment
 135       */
 136  	function RenderItemAttachment( & $params, $in_comments = false )
 137      {
 138          $File = $params['File'];
 139  
 140          if( ! $this->is_flp_video( $File ) )
 141          {
 142              return false;
 143          }
 144  
 145          $Item = & $params['Item'];
 146          $item_Blog = $Item->get_Blog();
 147  
 148          if( $File->exists() )
 149          {
 150              /**
 151               * @var integer A number to assign each video player new id attribute
 152               */
 153              global $html5_videojs_number;
 154              $html5_videojs_number++;
 155  
 156              if( $in_comments )
 157              {
 158                  $params['data'] .= '<div style="clear: both; height: 0px; font-size: 0px"></div>';
 159              }
 160  
 161              /**
 162               * Video options:
 163               *
 164               * controls:  true // The controls option sets whether or not the player has controls that the user can interact with.
 165               * autoplay:  true // If autoplay is true, the video will start playing as soon as page is loaded (without any interaction from the user). NOT SUPPORTED BY APPLE iOS DEVICES
 166               * preload:   'auto'|'metadata'|'none' // The preload attribute informs the browser whether or not the video data should begin downloading as soon as the video tag is loaded.
 167               * poster:    'myPoster.jpg' // The poster attribute sets the image that displays before the video begins playing.
 168               * loop:      true // The loop attribute causes the video to start over as soon as it ends.
 169               */
 170              $video_options = array();
 171              $video_options['controls'] = true;
 172              $video_options['preload'] = 'auto';
 173  
 174              $params['data'] .= '<div class="videojs_block">';
 175  
 176              $params['data'] .= '<video id="html5_videojs_'.$html5_videojs_number.'" class="video-js '.$this->get_coll_setting( 'skin', $item_Blog ).'" data-setup=\''.evo_json_encode( $video_options ).'\'>'.
 177                  '<source src="'.$File->get_url().'" type="'.$this->get_video_mimetype( $File ).'" />'.
 178                  '</video>';
 179  
 180              if( $File->get( 'desc' ) != '' && $this->get_coll_setting( 'disp_caption', $item_Blog ) )
 181              { // Display caption
 182                  $params['data'] .= '<div class="videojs_text">'.$File->get( 'desc' ).'</div>';
 183              }
 184  
 185              if( $this->get_coll_setting( 'allow_download', $item_Blog ) )
 186              { // Allow to download the video files
 187                  $params['data'] .= '<div class="videojs_text"><a href="'.$File->get_url().'">'.T_('Download this video').'</a></div>';
 188              }
 189  
 190              $params['data'] .= '</div>';
 191  
 192              return true;
 193          }
 194  
 195          return false;
 196      }

 213      /**
 214       * Get a list of the skins
 215       *
 216       * @return array Skins
 217       */
 218  	function get_skins_list()
 219      {
 220          $skins_path = dirname( $this->classfile_path ).'/skins';
 221  
 222          $skins = array();
 223          // Set this skin permanently, because it is a default skin name
 224          $skins['vjs-default-skin'] = 'vjs-default-skin';
 225  
 226          $files = scandir( $skins_path );
 227          foreach( $files as $file )
 228          {
 229              if( $file != '.' && $file != '..' && is_dir( $skins_path.'/'.$file ) )
 230              {    // Use folder name as skin name
 231                  $skins[ $file ] = $file;
 232              }
 233          }
 234  
 235          return $skins;
 236      }

 241  	function require_skin()
 242      {
 243          global $Blog;
 244  
 245          $skin = $this->get_coll_setting( 'skin', $Blog );
 246          if( !empty( $skin ) && $skin != 'vjs-default-skin')
 247          {
 248              $skins_path = dirname( $this->classfile_path ).'/skins';
 249              if( file_exists( $skins_path.'/'.$skin.'/style.css' ) )
 250              {    // Require css file only if it exists
 251                  require_css( $this->get_plugin_url().'skins/'.$skin.'/style.css', 'relative' );
 252              }
 253          }

 256      /**
 257       * Get video mimetype
 258       *
 259       * @param object File
 260       * @return string Mimetype
 261       */
 262  	function get_video_mimetype( $File )
 263      {
 264          switch( $File->get_ext() )
 265          {
 266              case 'flv':
 267                  $mimetype = 'video/flv';
 268                  break;
 269  
 270              case 'm4v':
 271                  $mimetype = 'video/m4v';
 272                  break;
 273  
 274              case 'ogv':
 275                  $mimetype = 'video/ogg';
 276                  break;
 277  
 278              case 'webm':
 279                  $mimetype = 'video/webm';
 280                  break;
 281  
 282              case 'f4v':
 283              case 'mp4':
 284              default:
 285                  $mimetype = 'video/mp4';
 286                  break;
 287          }
 288  
 289          return $mimetype;
 290      }
