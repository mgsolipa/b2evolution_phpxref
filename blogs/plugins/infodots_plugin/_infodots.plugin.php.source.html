 210      /**

 211       * Callback function to load a dot from NOT rendered content

 212       *

 213       * @param array Matches

 214       * @param boolean TRUE is used only to load dot without returning of tooltip template

 215       * @return string Empty string to don't display the dot template in content, It is printed out before image tag

 216       */
 217  	function load_infodot_from_source( $matches, $only_load_dot = false )
 218      {
 219          $link_ID = intval( $matches[3] );
 220  
 221          if( empty( $link_ID ) || empty( $matches ) || empty( $this->object_ID ) )
 222          { // Skip this incorrect match
 223              return;
 224          }
 225  
 226          $LinkCache = & get_LinkCache();
 227          $Link = & $LinkCache->get_by_ID( $link_ID, false, false );
 228          if( ! $Link )
 229          { // Inform about invalid Link ID
 230              return '<div style="color:#F00"><b>'.T_('Invalid Link ID').' - '.$matches[0].'</b></div>';
 231          }
 232  
 233          if( $this->dot_numbers === NULL )
 234          { // Init dot numbers array first time
 235              $this->dot_numbers = array();
 236          }
 237  
 238          if( ! isset( $this->dot_numbers[ $link_ID ] ) )
 239          { // Start to calculate number of the dots for current Link object
 240              $this->dot_numbers[ $link_ID ] = 1;
 241          }
 242  
 243          if( ! isset( $this->loaded_objects[ $this->object_ID ] ) )
 244          { // Load dots only once
 245              if( $this->dots === NULL )
 246              { // Init dots array first time
 247                  $this->dots = array();
 248              }
 249  
 250              if( ! isset( $this->dots[ $link_ID ] ) )
 251              { // Init sub array for each Link
 252                  $this->dots[ $link_ID ] = array();
 253              }
 254  
 255              // Add dot

 256              $this->dots[ $link_ID ][] = array(
 257                      'x' => intval( $matches[4] ), // Left
 258                      'y' => intval( $matches[5] ), // Top
 259                  );
 260          }
 261  
 262          if( $only_load_dot )
 263          { // Exit here to don't execute a code below
 264              return;
 265          }
 266  
 267          $dot_num = $this->dot_numbers[ $link_ID ];
 268          if( empty( $matches[6] ) )
 269          { // No defined width
 270              $tooltip_width = '';
 271          }
 272          else
 273          { // Set css style for width
 274              $tooltip_width = substr( $matches[6], 1 );
 275              $tooltip_width = ( strlen( intval( $tooltip_width ) ) == strlen( $tooltip_width ) ? $tooltip_width.'px' : $tooltip_width );
 276              $tooltip_width = ' style="width:'.$tooltip_width.'"';
 277          }
 278          $dot_xy = ' xy="'.$this->dots[ $link_ID ][ $dot_num - 1 ]['x'].':'.$this->dots[ $link_ID ][ $dot_num - 1 ]['y'].'"';
 279  
 280          $this->dot_numbers[ $link_ID ]++;
 281  
 282          // Print this element that will be used for tooltip of the dot

 283          return '<div class="infodots_info" id="infodot_'.$link_ID.'_'.$dot_num.'"'.$dot_xy.$tooltip_width.'>'
 284                  .balance_tags( $matches[7] )
 285              .'</div>'."\n";
 286      }

 306      /**

 307       * Render the dots before <img> tag

 308       *

 309       * @param array Associative array of parameters. $params['File'] - attachment, $params['data'] - output

 310       * @param string Content of the Item/Comment

 311       */
 312  	function render_infodots( & $params, $content )
 313      {
 314          if( empty( $params['File'] ) || empty( $params['Link'] ) )
 315          { // Check input data
 316              return;
 317          }
 318  
 319          $File = $params['File'];
 320          $Link = $params['Link'];
 321  
 322          if( ! $File->is_image() )
 323          { // This plugin works only with image files
 324              return;
 325          }
 326  
 327          if( ( $LinkOwner = & $Link->get_LinkOwner() ) === false || ( $Blog = & $LinkOwner->get_Blog() ) === false )
 328          { // Couldn't get Blog object
 329              return;
 330          }
 331  
 332          global $thumbnail_sizes;
 333          $thumbnail_width = isset( $thumbnail_sizes[ $params['image_size'] ] ) ? $thumbnail_sizes[ $params['image_size'] ][1] : 0;
 334  
 335          if( $File->get_image_size( 'width' ) < $this->get_coll_setting( 'coll_min_width', $Blog ) ||
 336              $thumbnail_width < $this->get_coll_setting( 'coll_min_width', $Blog ) )
 337          { // Don't draw a dot on image if width is less than setting value
 338              return;
 339          }
 340  
 341          if( ! isset( $this->loaded_objects[ $this->object_ID ] ) )
 342          { // Load the info dots if they were not loaded before
 343              replace_content_outcode( '#<div class="infodots_info" id="infodot_(\d+)_(\d+)" xy="(-?\d+):(-?\d+)"[^>]*>(.+?)</div>#is', array( $this, 'load_infodot_from_rendered_content' ), $content, 'replace_content_callback' );
 344              $this->loaded_objects[ $this->object_ID ] = 1;
 345          }
 346  
 347          if( empty( $this->dots[ $Link->ID ] ) )
 348          { // No dots for this Link
 349              return;
 350          }
 351  
 352          $before_image = '<div class="infodots_image">'."\n";
 353          foreach( $this->dots[ $Link->ID ] as $d => $dot )
 354          { // Init html element for each dot
 355              $before_image .= '<div class="infodots_dot" rel="infodot_'.$Link->ID.'_'.( $d + 1 ).'" style="left:'.$dot['x'].'px;top:'.$dot['y'].'px"></div>'."\n";
 356          }
 357  
 358          // Append info dots html to current image tag

 359          $params['before_image'] = $params['before_image'].$before_image;
 360          $params['after_image'] = '</div>'.$params['after_image'];
 361      }
