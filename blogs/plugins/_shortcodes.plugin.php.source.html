  98          {
  99              var tag = shortcodes_buttons[i].tag;
 100              if( typeof( tinyMCE ) != 'undefined' && typeof( tinyMCE.activeEditor ) != 'undefined' && tinyMCE.activeEditor )
 101              { // tinyMCE plugin is active now, we should insert separators by plugin buttons to render them
 102                  if( tag == '[teaserbreak]' && typeof( tinyMCE.activeEditor.execCommands.mceMore ) != 'undefined' )
 103                  { // MorePlugin exists in tinyMCE
 104                      tinyMCE.execCommand( 'mceMore', false, tinyMCE.activeEditor.id );
 105                      return;

 108                  { // PageBreakPlugin exists in tinyMCE
 109                      tinyMCE.execCommand( 'mcePageBreak', false, tinyMCE.activeEditor.id );
 110                      return;
 111                  }
 112              }
 113              // Insert plain text in simple textarea

 114              textarea_wrap_selection( canvas_field, tag, '', 0 );
 115          }
