 194      /**
 195       * This gets called before an image thumbnail gets created.
 196       *
 197       * This is useful to post-process the thumbnail image (add a watermark or change colors).
 198       *
 199       * @param array Associative array of parameters
 200       *   - 'File': the related File (by reference)
 201       *   - 'imh': image resource (by reference)
 202       *   - 'size': size name (by reference)
 203       *   - 'mimetype': mimetype of thumbnail (by reference)
 204       *   - 'quality': JPEG image quality [0-100] (by reference)
 205       *   - 'root_type': file root type 'user', 'group', 'collection' etc. (by reference)
 206       *   - 'root_type_ID': ID of user, group or collection (by reference)
 207       */
 208  	function BeforeThumbCreate( & $params )
 209      {
 210          if( $params['root_type'] == 'collection' )
 211          {
 212              $BlogCache = & get_BlogCache();
 213              $Blog = & $BlogCache->get_by_ID( $params['root_type_ID'], false, false );
 214          }
 215  
 216          if( ! empty( $Blog ) )
 217          { // Use blog settings when possible
 218              $font = $this->get_coll_setting( 'coll_font', $Blog );
 219              $text = $this->get_coll_setting( 'coll_text', $Blog );
 220              $font_size = $this->get_coll_setting( 'coll_font_size', $Blog );
 221              $min_dimension = $this->get_coll_setting( 'coll_min_dimension', $Blog );
 222          }
 223          else
 224          { // Use global settings
 225              $font = $this->Settings->get( 'font' );
 226              $text = $this->Settings->get( 'text' );
 227              $font_size = $this->Settings->get( 'font_size' );
 228              $min_dimension = $this->Settings->get( 'min_dimension' );
 229          }
 230  
 231          if( ! function_exists( 'imagettftext' ) )
 232          { // The function imagettftext() is not available.
 233              return;
 234          }
 235  
 236          // Canvas width & height
 237          $width = imagesx( $params['imh'] ) ;
 238          $height = imagesy( $params['imh'] );
 239  
 240          if( $width < $min_dimension || $height < $min_dimension )
 241          { // Skip small thumbnails
 242              return;
 243          }
 244  
 245          if( $font )
 246          {    // Custom font
 247              $font = $this->fonts_dir.'/'.$font;
 248          }
 249          else
 250          {
 251              $font = $this->get_default_font();
 252          }
 253  
 254          if( !is_readable($font) )
 255          {    // Font file not found
 256              $this->debug_log( sprintf('Font file (%s) is not readable!', $font) );
 257              return;
 258          }
 259  
 260          $text = evo_html_entity_decode( $text );
 261  
 262          // Text margins
 263          $margin_left = 10;
 264          $margin_bottom = 10;
 265  
 266          // Create a transparent image
 267          $im = imagecreatetruecolor($width, $height);
 268          imagealphablending($im, true);
 269          imagefill($im, 0, 0, imagecolortransparent($im, imagecolorallocatealpha($im, 0, 0, 0, 127)));
 270          imagesavealpha($im, true);
 271  
 272          // Create some colors
 273          $white = imagecolorallocate($im, 255, 255, 255);
 274          $light_grey = imagecolorallocate($im, 230, 230, 230);
 275          $grey = imagecolorallocate($im, 60, 60, 60);
 276          $black = imagecolorallocate($im, 0, 0, 0);
 277  
 278          // Add text shadow
 279          imagettftext($im, $font_size, 0, $margin_left + 2, $height - $margin_bottom, $grey, $font, $text);
 280  
 281          // Add text
 282          imagettftext($im, $font_size, 0, $margin_left, $height - $margin_bottom, $light_grey, $font, $text);
 283  
 284          // Merge images
 285          imagecopyresampled($params['imh'], $im, 0, 0, 0, 0, $width, $height, $width, $height);
 286  
 287          // Low quality :(
 288          //imagecopymerge($params['imh'], $im, 0, 0, 0, 0, $width, $height, 90);
 289  
 290          return true;
 291      }
