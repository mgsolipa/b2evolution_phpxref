 232  	function display_smiley_bar()
 233      {

 237          $smiled = array();
 238          foreach( $this->smilies as $smiley )
 239          {
 240              if( ! in_array($smiley['image'], $smiled) )
 241              { // include any smiley only once
 242                  $smiled[] = $smiley['image'];

 245                      'class' => 'top',
 246                      'onclick' => 'textarea_wrap_selection( b2evoCanvas, \''. str_replace("'", "\'", $smiley['code']). '\', \'\', 1 );' ) )
 247                      .' ';
 248              }
 249          }
 250  
 251          echo '<div class="edit_toolbar" id="smiley_toolbar">'.$grins.'</div>' ;
 252  
 253          return true;

 257      /**
 258       * Perform rendering
 259       *
 260       * @see Plugin::RenderItemAsHtml()
 261       */
 262  	function RenderItemAsHtml( & $params )
 263      {
 264          $this->InitSmilies();    // check smilies are already cached
 265  
 266  
 267          if( ! isset( $this->search ) )
 268          {    // We haven't prepared the smilies yet
 269              $this->search = array();
 270  
 271              $tmpsmilies = $this->smilies;
 272              usort($tmpsmilies, array(&$this, 'smiliescmp'));
 273  
 274              foreach( $tmpsmilies as $smiley )
 275              {
 276                  // Detect html entities if smile code, to find encoded code version when HTML is disabled in content
 277                  $html_entities_exist = ( strpos( $smiley['code'], '>' ) !== false || strpos( $smiley['code'], '>' ) !== false );
 278  
 279                  $this->search[] = $smiley['code'];
 280                  $smiley_masked = '';
 281                  for( $i = 0; $i < strlen( $smiley['code'] ); $i++ )
 282                  {
 283                      $smiley_masked .= '&#'.ord( substr( $smiley['code'], $i, 1 ) ).';';
 284                  }
 285                  if( $html_entities_exist )
 286                  { // Add code version with encoded html entities
 287                      $this->search[] = str_replace( array( '<', '>' ), array( '&lt;', '&gt;' ), $smiley['code'] );
 288                  }
 289                  $smiley['code'] = $smiley_masked;
 290                  $this->replace[] = $this->get_smiley_img_tag( $smiley );
 291                  if( $html_entities_exist )
 292                  { // Add a duplicate of replacement when smile code has html entities
 293                      $this->replace[] = $this->get_smiley_img_tag( $smiley );
 294                  }
 295              }
 296          }
 297  
 298  
 299          // REPLACE:  But only in non-HTML blocks, totally excluding <CODE>..</CODE> and <PRE>..</PRE>
 300  
 301          $content = & $params['data'];
 302  
 303          // Lazy-check first, using stristr() (stripos() is only available since PHP5):
 304          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false )
 305          { // Call ReplaceTagSafe() on everything outside code/pre:
 306              $content = callback_on_non_matching_blocks( $content,
 307                      '~<(code|pre)[^>]*>.*?</\1>~is',
 308                      array( & $this, 'ReplaceTagSafe' ) );
 309          }
 310          else
 311          { // No code/pre blocks, replace on the whole thing
 312              $content = $this->ReplaceTagSafe($content);
 313          }
 314  
 315          return true;
 316      }

 321       * @param array Override params, e.g. "class"

 323  	function get_smiley_img_tag($smiley, $override_fields = array())
 324      {
 325          $attribs = array(
 326              'src' => $smiley['image'],
 327              'title' => format_to_output( $smiley['code'], 'htmlattr' ),
 328              'alt' => format_to_output( $smiley['code'], 'htmlattr' ),
 329              'class' => 'middle',
 330              );
 331  
 332          if( $smiley_wh = imgsize($smiley['path'], 'widthheight_assoc') )
 333              $attribs += $smiley_wh;
 334  
 335          if( $override_fields )
 336              $attribs = $override_fields + $attribs;
 337  
 338          return '<img'.get_field_attribs_as_string($attribs).' />';
 339      }

 363       * sorts the smilies' array by length
 364       * this is important if you want :)) to superseede :) for example
 365       */
 366  	function smiliescmp($a, $b)
 367      {
 368          if( ($diff = strlen( $b[ 'code' ] ) - strlen( $a[ 'code' ] ) ) == 0)
 369          {
 370              return strcmp( $a[ 'code' ], $b[ 'code' ] );
 371          }
 372          return $diff;
 373      }

 376      /**
 377       * Initiates the smiley array if not already initiated
 378       *
 379       * Attempts to use skin specific smileys where available
 380       *    - skins_adm/skin/rsc/smilies/
 381       *    - skins/skin/smilies/
 382       *
 383       * Attempts to fallback to default smilies
 384       *    - rsc/smilies/
 385       *
 386       * If no image file found the smiley is not added
 387       *
 388       * @return array of available smilies( code, image url )
 389       */
 390  	function InitSmilies()
 391      {
 392          if( isset( $this->smilies ) )
 393          { // smilies are already cached
 394              return;
 395          }
 396  
 397          global $admin_skin, $adminskins_path, $adminskins_url, $rsc_path, $rsc_url, $skin, $skins_path, $skins_url;
 398  
 399          // set the skin path/url and the default (rsc) path/url
 400          $currentskin_path = ( is_admin_page() ? $adminskins_path.$admin_skin.'/rsc' : $skins_path.$skin ).'/smilies/';
 401          $currentskin_url = ( is_admin_page() ? $adminskins_url.$admin_skin.'/rsc' : $skins_url.$skin ).'/smilies/';
 402          $default_path = $rsc_path.'smilies/';
 403          $default_url = $rsc_url.'smilies/';
 404  
 405          $skin_has_smilies = is_dir( $currentskin_path );    // check if skin has a /smilies/ folder
 406  
 407          $this->smilies = array();
 408          $temp_list = explode( "\n", str_replace( array( "\r", "\t" ), '', $this->Settings->get( 'smiley_list' ) ) );
 409  
 410          foreach( $temp_list as $temp_smiley )
 411          {
 412              $a_smiley = explode( '<->',    preg_replace_callback( '#^(\S.+?\s)(.+?)(\/\/.*?)*$#', array( $this, 'get_smiley' ),$temp_smiley ) );
 413              if( isset( $a_smiley[0] ) and isset( $a_smiley[1] ) )
 414              {
 415                  // lets see if the file exists
 416                  $temp_img = trim( $a_smiley[1] );
 417                  if( $skin_has_smilies && is_file( $currentskin_path.$temp_img ) )
 418                  {
 419                      $temp_url = $currentskin_url.$temp_img;    // skin has it's own smiley, use it
 420                      $temp_path = $currentskin_path.$temp_img;
 421                  }
 422                  elseif ( is_file( $default_path.$temp_img ) )
 423                  {
 424                      $temp_url = $default_url.$temp_img; // no skin image, but default smiley found so use it
 425                      $temp_path = $default_path.$temp_img;
 426                  }
 427                  else
 428                  {
 429                      $temp_url = ''; // no smiley image found, so don't add the smiley
 430                  }
 431  
 432                  if( $temp_url )
 433                      $this->smilies[] = array( 'code' => trim( $a_smiley[0] ), 'image' => $temp_url, 'path' => $temp_path );
 434              }
 435          }
 436      }
