  60  	function parse($text)
  61      {    
  62          # Removes UTF-8 BOM and marker characters.
  63          $text = preg_replace('{^\xEF\xBB\xBF|\x1A}', '', $text);
  64          
  65          # Removes \r characters.
  66          $text = str_replace("\r\n", "\n", $text);
  67          $text = str_replace("\r", "\n", $text);
  68          
  69          # Replaces tabs with spaces.
  70          $text = str_replace("\t", '    ', $text);
  71          
  72          # Encodes escape sequences.
  73          
  74          if (strpos($text, '\\') !== FALSE)
  75          {
  76              $escape_sequences = array('\\\\', '\`', '\*', '\_', '\{', '\}', '\[', '\]', '\(', '\)', '\>', '\#', '\+', '\-', '\.', '\!');
  77              
  78              foreach ($escape_sequences as $index => $escape_sequence)
  79              {
  80                  if (strpos($text, $escape_sequence) !== FALSE)
  81                  {
  82                      $code = "\x1A".'\\'.$index;
  83                      
  84                      $text = str_replace($escape_sequence, $code, $text);
  85                      
  86                      $this->escape_sequence_map[$code] = $escape_sequence;
  87                  }
  88              }
  89          }
  90          
  91          # Extracts link references.
  92          
  93          if (preg_match_all('/^[ ]{0,3}\[(.+)\][ ]?:[ ]*\n?[ ]*(.+)$/m', $text, $matches, PREG_SET_ORDER))
  94          {
  95              foreach ($matches as $matches)
  96              {
  97                  $this->reference_map[strtolower($matches[1])] = $matches[2];
  98                  
  99                  $text = str_replace($matches[0], '', $text);
 100              }
 101          }
 102          
 103          # ~ 
 104          
 105          $text = preg_replace('/\n\s*\n/', "\n\n", $text);
 106          $text = trim($text, "\n");
 107          
 108          $lines = explode("\n", $text);
 109          
 110          $text = $this->parse_block_elements($lines);
 111          
 112          # Decodes escape sequences (leaves out backslashes).
 113          
 114          foreach ($this->escape_sequence_map as $code => $escape_sequence)
 115          {
 116              $text = str_replace($code, $escape_sequence[1], $text);
 117          }
 118          
 119          $text = rtrim($text, "\n");
 120          
 121          return $text;
 122      }

 128  	private function parse_block_elements(array $lines, $context = '')
 129      {
 130          $elements = array();
 131          
 132          $element = array(
 133              'type' => '',
 134          );
 135          
 136          foreach ($lines as $line)
 137          {
 138              # Empty 
 139              
 140              if( $line === '' )
 141              {
 142                  $element['interrupted'] = true;
 143                  
 144                  $element['type'] === 'code' and $element['text'] .= "\n";
 145                  
 146                  continue;
 147              }
 148  
 149              # Codeblock
 150              if( $element['type'] != 'codeblock' && preg_match('/^```([a-z0-9]+)*$/i', $line, $matches ) )
 151              { // Codeblock is opening
 152                  if( ! empty( $element['type'] ) )
 153                  { // Save any previous element that has been started but not finished yet
 154                      $elements[] = $element;
 155                  }
 156                  // Start new codeblock element
 157                  $element = array(
 158                          'type' => 'codeblock',
 159                          'lang' => strtolower( $matches[1] ),
 160                          'lines' => array(),
 161                      );
 162                  continue;
 163              }
 164              if( $element['type'] == 'codeblock' )
 165              {
 166                  if( preg_match('/^```$/i', $line ) )
 167                  { // Codeblock is closing
 168                      $elements[] = $element;
 169                      $element = array( 'type' => '' );
 170                  }
 171                  else
 172                  { // Save each line of codeblock
 173                      $element['lines'][] = $line;
 174                  }
 175                  continue;
 176              }
 177  
 178              # Lazy Blockquote 
 179              
 180              if ($element['type'] === 'blockquote' and ! isset($element['interrupted']))
 181              {
 182                  $line = preg_replace('/^[ ]*(>|&gt;)[ ]?/', '', $line);
 183                  
 184                  $element['lines'] []= $line;
 185                  
 186                  continue;
 187              }
 188              
 189              # Lazy List Item 
 190              
 191              if ($element['type'] === 'li')
 192              {
 193                  if (preg_match('/^([ ]{0,3})(\d+[.]|[*+-])[ ](.*)/', $line, $matches))
 194                  {
 195                      if ($element['indentation'] !== $matches[1]) 
 196                      {
 197                          $element['lines'] []= $line;
 198                      }
 199                      else 
 200                      {
 201                          unset($element['last']);
 202                          
 203                          $elements []= $element;
 204                          
 205                          $element = array(
 206                              'type' => 'li',
 207                              'indentation' => $matches[1],
 208                              'last' => true,
 209                              'lines' => array(
 210                                  preg_replace('/^[ ]{0,4}/', '', $matches[3]),
 211                              ),
 212                          );
 213                      }
 214                      
 215                      continue;
 216                  }
 217                  
 218                  if (isset($element['interrupted']))
 219                  {
 220                      if ($line[0] === ' ')
 221                      {
 222                          $element['lines'] []= '';
 223                          
 224                          $line = preg_replace('/^[ ]{0,4}/', '', $line);;
 225                          
 226                          $element['lines'] []= $line;
 227                          
 228                          continue;
 229                      }
 230                  }
 231                  elseif( ! preg_match( '/^[\s\t]*<\//', $line ) )
 232                  { // Don't append the second line if it is an end of HTML tag
 233                      $line = preg_replace('/^[ ]{0,4}/', '', $line);;
 234                      
 235                      $element['lines'] []= $line;
 236                      
 237                      continue;
 238                  }
 239              }
 240              
 241              # Quick Paragraph 
 242              
 243              if( ! ( $line[0] >= 'A' and $line['0'] !== '_' ) )
 244              { // 
 245  
 246                  # Setext Header (---) 
 247                  
 248                  if ($element['type'] === 'p' and ! isset($element['interrupted']) and preg_match('/^[-]+[ ]*$/', $line))
 249                  {
 250                      $element['type'] = 'h.';
 251                      $element['level'] = 2;
 252                      
 253                      continue;
 254                  }
 255                  
 256                  # Horizontal Rule  
 257                  
 258                  if (preg_match('/^[ ]{0,3}([-*_])([ ]{0,2}\1){2,}[ ]*$/', $line))
 259                  {
 260                      $elements []= $element;
 261                      
 262                      $element = array(
 263                          'type' => 'hr',
 264                      );
 265                      
 266                      continue;
 267                  }
 268                  
 269                  # List Item 
 270                  
 271                  if (preg_match('/^([ ]{0,3})(\d+[.]|[*+-])[ ](.*)/', $line, $matches))
 272                  {
 273                      $elements []= $element;
 274                      
 275                      $element = array(
 276                          'type' => 'li',
 277                          'ordered' => isset($matches[2][1]),
 278                          'indentation' => $matches[1],
 279                          'last' => true,
 280                          'lines' => array(
 281                              preg_replace('/^[ ]{0,4}/', '', $matches[3]),
 282                          ),
 283                      );
 284                      
 285                      continue;
 286                  }
 287                  
 288                  # Code 
 289                  
 290                  if (preg_match('/^[ ]{4}(.*)/', $line, $matches))
 291                  {
 292                      if ($element['type'] === 'code')
 293                      {
 294                          $element['text'] .= "\n".$matches[1];
 295                      }
 296                      else
 297                      {
 298                          $elements []= $element;
 299                          
 300                          $element = array(
 301                              'type' => 'code', 
 302                              'text' => $matches[1],
 303                          );
 304                      }
 305                      
 306                      continue;
 307                  }
 308                  
 309                  # Atx Header (#)
 310                  
 311                  if ($line[0] === '#' and preg_match('/^(#{1,6}) (.+?)[ ]*#*$/', $line, $matches))
 312                  {
 313                      $elements []= $element;
 314                      
 315                      $level = strlen($matches[1]);
 316                      
 317                      $element = array(
 318                          'type' => 'h.', 
 319                          'text' => $matches[2], 
 320                          'level' => $level,
 321                      );
 322                      
 323                      continue;
 324                  }
 325                  
 326                  # Blockquote 
 327                  
 328                  if (preg_match('/^[ ]*(>|&gt;)[ ]?(.*)/', $line, $matches))
 329                  {
 330                      if ($element['type'] === 'blockquote')
 331                      {
 332                          if (isset($element['interrupted']))
 333                          {
 334                              $element['lines'] []= '';
 335                              
 336                              unset($element['interrupted']);
 337                          }
 338                          
 339                          $element['lines'] []= $matches[2];
 340                      }
 341                      else
 342                      {
 343                          $elements []= $element;
 344                          
 345                          $element = array(
 346                              'type' => 'blockquote',
 347                              'lines' => array(
 348                                  $matches[2],
 349                              ),
 350                          );
 351                      }
 352                      
 353                      continue;
 354                  }
 355                  
 356                  # Setext Header (===) 
 357                  
 358                  if ($element['type'] === 'p' and ! isset($element['interrupted']) and preg_match('/^[=]+[ ]*$/', $line))
 359                  {
 360                      $element['type'] = 'h.';
 361                      $element['level'] = 1;
 362                      
 363                      continue;
 364                  }
 365  
 366              }
 367  
 368              if ($element['type'] === 'p')
 369              {
 370                  if (isset($element['interrupted']))
 371                  {
 372                      $elements []= $element;
 373                      
 374                      $element['text'] = $line;
 375                      
 376                      unset($element['interrupted']);
 377                  }
 378                  else
 379                  {
 380                      $element['text'] .= "\n".$line;
 381                  }
 382              }
 383              else
 384              {
 385                  $elements []= $element;
 386                  
 387                  $element = array(
 388                      'type' => 'p', 
 389                      'text' => $line,
 390                  );
 391              }
 392          }
 393          
 394          $elements []= $element;
 395          
 396          array_shift($elements);
 397          
 398          # 
 399          # ~ 
 400          # 
 401          
 402          $markup = '';
 403          
 404          foreach ($elements as $index => $element)
 405          {
 406              switch ($element['type'])
 407              {
 408                  case 'li':
 409                      
 410                      if (isset($element['ordered'])) # first
 411                      {
 412                          $list_type = $element['ordered'] ? 'ol' : 'ul';
 413                          
 414                          $markup .= '<'.$list_type.'>'."\n";
 415                      }
 416                      
 417                      if (isset($element['interrupted']) and ! isset($element['last']))
 418                      {
 419                          $element['lines'] []= '';
 420                      }
 421                      
 422                      $text = $this->parse_block_elements($element['lines'], 'li');
 423                      
 424                      $markup .= '<li>'.$text.'</li>'."\n";
 425                      
 426                      isset($element['last']) and $markup .= '</'.$list_type.'>'."\n";
 427                      
 428                      break;
 429                  
 430                  case 'p':
 431                      
 432                      $text = $this->parse_inline_elements($element['text']);
 433                      
 434                      $text = preg_replace('/[ ]{2}\n/', '<br />'."\n", $text);
 435                      
 436                      if( $text == '[teaserbreak]' || $text == '[pagebreak]' )
 437                      { // Don't apply <p> around item content separators
 438                          $markup .= $text."\n";
 439                      }
 440                      elseif( $context === 'li' && $index === 0 )
 441                      {
 442                          if (isset($element['interrupted']))
 443                          {
 444                              $markup .= "\n".'<p>'.$text.'</p>'."\n";
 445                          }
 446                          else 
 447                          {
 448                              $markup .= $text;
 449                          }
 450                      }
 451                      else 
 452                      {
 453                          if( preg_match( '/^<table.+<\\/table>$/is', $text ) )
 454                          { // Apply <p> tag around full table
 455                              $markup .= '<p>'.$text.'</p>'."\n";
 456                          }
 457                          elseif( preg_match( '/(.*<td>)(.+)/is', $text, $text_match ) &&
 458                                ! preg_match( '#(</?table|</?tr|</?th|</?td)#is', $text_match[2] ) )
 459                          { // Apply <p> tag for content of <td>
 460                              $markup .= $text_match[1].'<p>'.$text_match[2].'</p>'."\n";
 461                          }
 462                          elseif( ! preg_match( '#(</?table|</?tr|</?th|</?td)#i', $text ) )
 463                          { // Apply <p> tag for text that is not a part of a <table>
 464                              if( count( $elements ) > 1 )
 465                              { // Only if a content contains many elements
 466                                  $markup .= '<p>'.$text.'</p>'."\n";
 467                              }
 468                              else
 469                              { // Don't add <p> tag around whole content
 470                                  $markup .= $text;
 471                              }
 472                          }
 473                          else
 474                          { // Don't apply <p> tag
 475                              $markup .= $text."\n";
 476                          }
 477                      }
 478                      
 479                      break;
 480                  
 481                  case 'code':
 482                      
 483                      $text = rtrim($element['text'], "\n");
 484                      
 485                      $text = htmlentities($text, ENT_NOQUOTES);
 486                      
 487                      strpos($text, "\x1A\\") !== FALSE and $text = strtr($text, $this->escape_sequence_map);
 488                      
 489                      $markup .= '<pre class="codeblock"><code>'.$text.'</code></pre>'."\n";
 490                      
 491                      break;
 492                  
 493                  case 'blockquote':
 494                      
 495                      $text = $this->parse_block_elements($element['lines']);
 496                      
 497                      $markup .= '<blockquote>'."\n".$text.'</blockquote>'."\n";
 498                      
 499                      break;
 500  
 501                  case 'codeblock':
 502                      // Codeblock
 503                      $attrs = empty( $element['lang'] ) ? '' : ' lang='.$element['lang'];
 504                      $attrs .= ' line=1'; // set this param because codehighlight plugin doesn't detect language without this
 505                      $text = implode( "\n", $element['lines'] );
 506                      $markup .= '<!-- codeblock '.$attrs.'--><pre class="codeblock"><code>'."\n".$text."\n".'</code></pre><!-- /codeblock -->'."\n";
 507                      break;
 508  
 509                  case 'h.':
 510                      
 511                      $text = $this->parse_inline_elements($element['text']);
 512                      
 513                      $markup .= '<h'.$element['level'].'>'.$text.'</h'.$element['level'].'>'."\n";
 514                      
 515                      break;
 516                  
 517                  case 'hr':
 518                      
 519                      $markup .= '<hr />'."\n";
 520                      
 521                      break;
 522              }
 523          }
 524          
 525          return $markup;
 526      }

 528  	private function parse_inline_elements($text)
 529      {
 530          $map = array();
 531          
 532          $index = 0;
 533          
 534          # Code Span 
 535          
 536          if (strpos($text, '`') !== FALSE and preg_match_all('/`(.+?)`/', $text, $matches, PREG_SET_ORDER)) 
 537          {
 538              foreach ($matches as $matches)
 539              {
 540                  $element_text = $matches[1];
 541                  $element_text = htmlentities($element_text, ENT_NOQUOTES);
 542                  
 543                  # Decodes escape sequences.
 544                  
 545                  $this->escape_sequence_map 
 546                      and strpos($element_text, "\x1A") !== FALSE 
 547                      and $element_text = strtr($element_text, $this->escape_sequence_map);
 548                  
 549                  # Composes element.
 550                  
 551                  $element = '<code class="codespan">'.$element_text.'</code>';
 552                  
 553                  # Encodes element.
 554                  
 555                  $code = "\x1A".'$'.$index;
 556                  
 557                  $text = str_replace($matches[0], $code, $text);
 558                  
 559                  $map[$code] = $element;
 560                  
 561                  $index ++;
 562              }
 563          }
 564  
 565          if( $this->parse_images || $this->parse_links )
 566          { // Parse images or links
 567  
 568              # Inline Link / Image 
 569              if( strpos($text, '](') !== FALSE ) # inline 
 570              {
 571                  $text = str_replace( '&quot;', '"', $text ); // revert from html entity
 572                  if( preg_match_all( '/(!?)(\[((?:[^][]+|(?2))*)\])\(([^"]*?)( "([^"]+)")?\)/', $text, $matches, PREG_SET_ORDER ) )
 573                  {
 574                      foreach ($matches as $matches)
 575                      {
 576                          if ($matches[1]) # image 
 577                          {
 578                              if( $this->parse_images )
 579                              { // Parse images only if it is enabled
 580                                  $element = '<img src="'.$matches[4].'" alt="'.$matches[3].'"'.( ! empty( $matches[6] ) ? ' title="'.$matches[6].'"' : '' ).'>';
 581                              }
 582                          }
 583                          else
 584                          {
 585                              if( $this->parse_links )
 586                              { // Parse links only if it is enabled
 587                                  $element_text = $this->parse_inline_elements($matches[3]);
 588                                  $element = '<a href="'.$matches[4].'"'.( ! empty( $matches[6] ) ? ' title="'.$matches[6].'"' : '' ).'>'.$element_text.'</a>';
 589                              }
 590                          }
 591  
 592                          if( ! isset( $element ) )
 593                          {
 594                              continue;
 595                          }
 596  
 597                          $element_text = $this->parse_inline_elements($matches[1]);
 598                          
 599                          # ~ 
 600  
 601                          $code = "\x1A".'$'.$index;
 602  
 603                          $text = str_replace($matches[0], $code, $text);
 604

 606

 617                  foreach ($matches as $matches)
 618                  {
 619                      $link_difinition = isset($matches[3]) && $matches[3]

 626                      {
 627                          $url = $this->reference_map[$link_difinition];
 628                          
 629                          if ($matches[1]) # image 
 630                          {
 631                              if( $this->parse_images )
 632                              { // Parse images only if it is enabled

 634                              }
 635                          }

 638                              if( $this->parse_links )

 641                                  $element = '<a href="'.$url.'">'.$element_text.'</a>';
 642                              }
 643                          }
 644  
 645                          if( ! isset( $element ) )
