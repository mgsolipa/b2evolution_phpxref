  94       * Perform rendering

  95       *

  96       * @param array Associative array of parameters

  97       *   'data': the data (by reference). You probably want to modify this.

  98       *   'format': see {@link format_to_output()}. Only 'htmlbody' and 'entityencoded' will arrive here.

  99       * @return boolean true if we can render something for the required output format

 100       */
 101  	function RenderItemAsHtml( & $params )
 102      {
 103          $content = & $params['data'];
 104  
 105          if( !empty( $params['Item'] ) )
 106          { // Get Item from params
 107              $Item = & $params['Item'];
 108          }
 109          elseif( !empty( $params['Comment'] ) )
 110          { // Get Item from Comment
 111              $Comment = & $params['Comment'];
 112              $Item = & $Comment->get_Item();
 113          }
 114  
 115          if( ! empty( $Item ) )
 116          { // We are rendering Item or Comment now, Get the settings depending on Blog
 117              $item_Blog = & $Item->get_Blog();
 118              $text_styles_enabled = $this->get_coll_setting( 'text_styles', $item_Blog );
 119              $links_enabled = $this->get_coll_setting( 'links', $item_Blog );
 120              $images_enabled = $this->get_coll_setting( 'images', $item_Blog );
 121          }
 122          elseif( ! empty( $params['Message'] ) )
 123          { // We are rendering Message now, Use FALSE by default because we don't have the settings
 124              $text_styles_enabled = false;
 125              $links_enabled = false;
 126              $images_enabled = false;
 127          }
 128          else
 129          { // Unknown call, Don't render this case
 130              return;
 131          }
 132  
 133          // Init parser class with blog settings

 134          $Parsedown = Parsedown::instance();
 135          $Parsedown->parse_font_styles = $text_styles_enabled;
 136          $Parsedown->parse_links = $links_enabled;
 137          $Parsedown->parse_images = $images_enabled;
 138  
 139          // Parse markdown code to HTML

 140          if( stristr( $content, '<code' ) !== false || stristr( $content, '<pre' ) !== false )
 141          { // Call replace_content() on everything outside code/pre:
 142              $content = callback_on_non_matching_blocks( $content,
 143                  '~<(code|pre)[^>]*>.*?</\1>~is',
 144                  array( $Parsedown, 'parse' ) );
 145          }
 146          else
 147          { // No code/pre blocks, replace on the whole thing
 148              $content = $Parsedown->parse( $content );
 149          }
 150  
 151          return true;
 152      }

 195  	function AdminDisplayToolbar( & $params )
 196      {
 197          if( ! empty( $params['Item'] ) )
 198          { // Item is set, get Blog from post
 199              $edited_Item = & $params['Item'];
 200              $Blog = & $edited_Item->get_Blog();
 201          }
 202  
 203          if( empty( $Blog ) )
 204          { // Item is not set, try global Blog
 205              global $Blog;
 206              if( empty( $Blog ) )
 207              { // We can't get a Blog, this way "apply_rendering" plugin collection setting is not available
 208                  return false;
 209              }
 210          }
 211  
 212          $coll_setting_name = ( $params['target_type'] == 'Comment' ) ? 'coll_apply_comment_rendering' : 'coll_apply_rendering';
 213          $apply_rendering = $this->get_coll_setting( $coll_setting_name, $Blog );
 214          if( empty( $apply_rendering ) || $apply_rendering == 'never' )
 215          { // Plugin is not enabled for current case, so don't display a toolbar:
 216              return false;
 217          }
 218  
 219          // Print toolbar on screen

 220          return $this->DisplayCodeToolbar( $Blog );

 499  		function markdown_insert_tag( field, i )
 500          {
 501              // we need to know if something is selected.

 502              // First, ask plugins, then try IE and Mozilla.

 503              var sel_text = b2evo_Callbacks.trigger_callback( "get_selected_text_for_" + field.id );
 504              var focus_when_finished = false; // used for IE

 505  
 506              if( sel_text == null )
 507              { // detect selection:
 508                  //IE support

 509                  if( document.selection )
 510                  {
 511                      field.focus();
 512                      var sel = document.selection.createRange();
 513                      sel_text = sel.text;
 514                      focus_when_finished = true;
 515                  }
 516                  //MOZILLA/NETSCAPE support

 517                  else if( field.selectionStart || field.selectionStart == '0' )
 518                  {
 519                      var startPos = field.selectionStart;
 520                      var endPos = field.selectionEnd;
 521                      sel_text = ( startPos != endPos );
 522                  }
 523              }
 524  
 525              if( sel_text )
 526              { // some text selected
 527                  textarea_wrap_selection( field, markdown_btns[i].tag_start, markdown_btns[i].tag_end, 0 );
 528              }
 529              else
 530              {
 531                  if( !markdown_check_open_tags(i) || markdown_btns[i].tag_end == '' )
 532                  {
 533                      textarea_wrap_selection( field, markdown_btns[i].tag_start, '', 0 );
 534                      markdown_add_tag(i);
 535                  }
 536                  else
 537                  {
 538                      textarea_wrap_selection( field, '', markdown_btns[i].tag_end, 0 );
 539                      markdown_remove_tag(i);
 540                  }
 541              }
 542              if( focus_when_finished )
 543              {
 544                  field.focus();
 545              }
 546          }
