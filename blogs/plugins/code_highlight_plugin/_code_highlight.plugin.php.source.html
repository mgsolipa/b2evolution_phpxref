 476      /**
 477       * Tidys up a block of code ready for numbering
 478       *
 479       * @param string $block - the code to be tidied up
 480       * @param string $line_seperator - the seperator between lines of code ( default \n )
 481       * @return string - the tidied code
 482       */
 483  	function tidy_code_output( $block, $line_seperator = "\n" )
 484      {
 485          // lets split the block into individual lines
 486          $code = explode( $line_seperator,
 487              // after removing windows garbage
 488              str_replace( "\r", '', $block ) );
 489  
 490          // time to rock and roll ;)
 491          $still_open = array(); // this holds all the spans that need closing and re-opening on the following code line
 492          for( $i = 0; $i < count( $code ); $i++ )
 493          {
 494              // we need to note all opening spans
 495              $spans =
 496                  // get rid of the first element, it's always empty
 497                  array_slice(
 498                  // split line at each opening span
 499                  explode( '<span class="',
 500                  // add any open spans back in
 501                  implode( '', $still_open )
 502                  .$code[$i] )
 503                  , 1 );
 504  //            pre_dump( $spans );
 505              // reset still_open array
 506              $still_open = array();
 507              // $spans now contains a list of opening spans
 508              for( $z = 0; $z < count( $spans ); $z++ )
 509              {
 510                  // add the span to the still_open array
 511                  $still_open[] = '<span class="'.substr( $spans[$z], 0, strpos( $spans[$z], '"' ) ).'">';
 512  //                pre_dump( $still_open );
 513                  // count all closing spans and remove them from the open spans list
 514                  if( $closed = substr_count( $spans[$z], '</span>' ) )
 515                      $still_open = array_slice( $still_open, 0, $closed * -1 );
 516              }
 517              // lets rebuild the code line and close any remaining spans
 518              $code[$i] = '<span class="'.implode( '<span class="', $spans ).str_repeat( '</span>', count( $still_open ) );
 519          }
 520          // lets stitch it all back together again
 521          $cleaned = implode( "\n", $code );
 522          // and get rid of any empty spans
 523          while( preg_match( '#\<span[^>]+?>\</span>#', $cleaned ) )
 524              $cleaned = preg_replace( '#\<span[^>]+?>\</span>#', '', $cleaned );
 525  //        pre_dump( $cleaned );
 526          // return the cleaned up code
 527          return $cleaned;
 528      }

 598      /**
 599       * Formats code ready for displaying
 600       * With "SwipeFriendly" line numbers
 601       *
 602       * @todo fp> no table should be needed. Also since we have odd/even coloring, word wrapping may be ok.
 603       * yabs > unfortunately the tables are required to make the odd/even colouring and swipeable numbers work :(
 604       *
 605       * an example :
 606       * Danny uploaded a page to demo his colours of choice ( http://brendoman.com/dev/youtube.html )
 607       * if you scroll down to RenderItemAsHtml() you'll see it has 4 problems
 608       * 1) some lines wrap - but they wrap underneath the line number
 609       * 2) some lines don't wrap so they force a browser scrollbar
 610       *   ( although a scrollable div will cure this)
 611       * 3) the lines that don't wrap lose their background colour once they become longer than the browser width
 612       *   ( a scrollable div has the same problem )
 613       * 4) If you try and copy/paste the code you also get the line numbers
 614       *   ( like every other code formatter that we've seen on the web that uses line numbers ... which is why most don't ;) )
 615       *
 616       * compare that to the same section of code using this plugin ( http://cvs.astonishme.co.uk/index.php/2007/04/01/a_demo )
 617       *
 618       * fp> almost all that can be fixed with proper CSS and will have the advantage of not adding extra space in front of the lines when copy/pasting. I'll work on it when I have time.
 619       *
 620       * yabs> The operative word in that sentance is "almost" ;) ... I've removed the trailing spaces for all but empty lines ( required to "prop them open" ) ... I'd forgotten all about them :p
 621       *
 622       * @param string the code block to be wrapped up
 623       * @param string the attributes - currently only the starting line number
 624       * @param string the code type ( code || php )
 625       *
 626       * @return string formatted code
 627       */
 628  	function do_numbering( $code, $offset = 0, $type = 'code' )
 629      {
 630          $temp = str_replace( array( '&nbsp;&nbsp;', '  ', "\t", '[x', '[/x' ),  array( '&#160;&#160;', '&#160;&#160;', '&#160;&#160;', '[', '[/' ), $code );
 631          $temp = explode( "\n", $temp );
 632          $count = 0;
 633          $output = '';
 634          $odd_line = false;
 635          foreach( $temp as $line )
 636          {
 637              $output .= '<tr class="amc_code_'.( ( $odd_line = !$odd_line ) ? 'odd' : 'even' ).'"><td class="amc_line">'
 638                                      .$this->create_number( ++$count + $offset ).'</td><td><code>'.$line
 639                                      // add an &nbsp; to empty lines to stop them "collapsing"
 640                                      .( empty( $line ) ? '&nbsp;' : '' )
 641                                      .'</code></td></tr>';//."\n"; yura: I commented this because Auto-P plugin creates the tags <p></p> from this symbol
 642          }
 643          // make "long" value a setting ? - yabs
 644          return '<p class="codeblock_title">'.$this->languageCache[ $type ]->language_title.'</p><div class="codeblock codeblock_with_title amc_'.$type.' '.( $count < 26 ? 'amc_short' : 'amc_long' ).'"><table>'.$output.'</table></div>';
 645      }

 654  	function create_number( $num )

 669      /**
 670       * Formats codeblock ready for displaying
 671       * Each language is stored as a classfile
 672       * This would allow new languages to be added more easily
 673       * It would also allow Geshi to be used as the highlighter with no code changes ;)
 674       *
 675       * Replaces both (current) highlighter functions
 676       *
 677       * ..... still requires some more thought though :p
 678       *
 679       * @param array $block ( 2 - attributes, 4 - the code )
 680       * @return string formatted code
 681       */
 682  	function render_codeblock_callback( $block )
 683      {
 684          // set the offset if present - default : 0
 685          preg_match( '#line=("|\'?)([0-9]+?)(["\']?)$#', $block[2], $match );
 686          $offset = ( empty( $match[2] ) ? 0 : $match[2] - 1 );
 687  
 688          // set the language if present - default : code
 689          preg_match( '#lang=("|\'?)([^\1]+?)([\s"\']+?)#', $block[2], $match );
 690          $language = strtolower( ( empty( $match[2] ) ? 'code' : $match[2] ) );
 691  
 692          if( $code = trim( $block[4] ) )
 693          { // we have a code block
 694              if( $this->allow_html )
 695              { // If HTML is allowed in content we should disallow this for <code> content
 696                  $code = evo_htmlspecialchars( $code );
 697              }
 698              // is the relevant language highlighter already cached?
 699              if( empty( $this->languageCache[ $language ] ) )
 700              { // lets attempt to load the language
 701                  $language_file = dirname(__FILE__).'/highlighters/'.$language.'.highlighter.php';
 702                  if( is_file( $language_file ) )
 703                  { // language class exists, lets load and cache an instance of it
 704                      require_once $language_file;
 705                      $class = 'am_'.$language.'_highlighter';
 706                      $this->languageCache[ $language ] = new $class( $this );
 707                  }
 708                  else
 709                  { // language class doesn't exists, fallback to default highlighter
 710                      $language = 'code';
 711                      if( empty( $this->languageCache[ $language ] ) )
 712                      { // lets attempt to load the default language
 713                          $language_file = dirname(__FILE__).'/highlighters/'.$language.'.highlighter.php';
 714                          if( is_file( $language_file ) )
 715                          { // default lanugage exists
 716                              require_once $language_file;
 717                              $class = 'am_'.$language.'_highlighter';
 718                              // add the language to the cache
 719                              $this->languageCache[ $language ] = new $class( $this );
 720                          }
 721                          else
 722                          { // if we hit this we might as well go to the pub ;)
 723                              // echo '***** error ***** no language or default language file is present';
 724                              return $code;
 725                          }
 726                      }
 727                  }
 728              }
 729              $this->languageCache[ $language ]->requested_language = $language;
 730              $this->languageCache[ $language ]->strict_mode = $this->Settings->get( 'strict' );
 731              $code = $this->languageCache[ $language ]->highlight_code( $code );
 732              // add the line numbers
 733              $code = $this->do_numbering( $code, $offset, $language );
 734          }
 735          return $code;
 736      }
