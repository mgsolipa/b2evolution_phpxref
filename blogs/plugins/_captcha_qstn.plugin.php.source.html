 149  	function PluginSettingsUpdateAction()
 150      {
 151          global $DB, $Messages;
 152  
 153          // Store in this array all questions IDs which are inserted or updated on currect action
 154          $questions_IDs = array( -1 );
 155          // Store in this array only questions with correct format
 156          $questions_formated = array();
 157  
 158          $questions = $this->Settings->get( 'questions' );
 159          if( empty( $questions ) )
 160          { // No questions
 161              // Delete all questions from DB:
 162              $DB->query( 'DELETE FROM '.$this->get_sql_table( 'questions' ) );
 163              return true;
 164          }
 165  
 166          $questions = explode( "\n", str_replace( "\r\n", "\n", $questions ) );
 167  
 168          $insert_data = array();
 169          $update_data = array();
 170          foreach( $questions as $question )
 171          {
 172              if( preg_match( '/(.+\?)(.+)/i', $question, $match ) )
 173              { // Check a question for correct format
 174                  $question = trim( $match[1] );
 175                  $answers = trim( $match[2] );
 176                  if( !empty( $question ) && !empty( $answers ) )
 177                  { // Save this question in DB
 178                      $SQL = new SQL();
 179                      $SQL->SELECT( 'cptq_ID' );
 180                      $SQL->FROM( $this->get_sql_table( 'questions' ) );
 181                      $SQL->WHERE( 'cptq_question = '.$DB->quote( $question ) );
 182  
 183                      if( $question_ID = $DB->get_var( $SQL->get() ) )
 184                      { // This question already exists, we should only update the answers
 185                          $update_data[ $question_ID ] = $answers;
 186                      }
 187                      else
 188                      { // New question
 189                          $insert_data[] = '( '.$DB->quote( $question ).', '.$DB->quote( $answers ).' )';
 190                      }
 191  
 192                      $questions_formated[] = $question.' '.$answers;
 193                  }
 194              }
 195              elseif( ! empty( $question ) )
 196              { // Incorrect format of question
 197                  $Messages->add( sprintf( T_('Invalid question format in the line %s.'), '"<b>'.$question.'</b>"' ), 'error' );
 198              }
 199          }
 200  
 201          if( param_errors_detected() )
 202          { // Don't save the changes on errors
 203              return false;
 204          }
 205          else
 206          { // Save the changes on success
 207              if( count( $insert_data ) )
 208              { // Insert new questions in DB:
 209                  foreach( $insert_data as $qstn_data )
 210                  {
 211                      $DB->query( 'INSERT INTO '.$this->get_sql_table( 'questions' ).'
 212                          ( cptq_question, cptq_answers ) VALUES '.$qstn_data );
 213                      $questions_IDs[] = $DB->insert_id;
 214                  }
 215              }
 216  
 217              if( count( $update_data ) )
 218              { // Update the questions in DB:
 219                  foreach( $update_data as $qstn_ID => $qstn_answers )
 220                  {
 221                      $DB->query( 'UPDATE '.$this->get_sql_table( 'questions' ).'
 222                            SET cptq_answers = '.$DB->quote( $qstn_answers ).'
 223                          WHERE cptq_ID = '.$qstn_ID );
 224                      $questions_IDs[] = $qstn_ID;
 225                  }
 226              }
 227  
 228              // Delete the old questions from DB:
 229              $DB->query( 'DELETE FROM '.$this->get_sql_table( 'questions' ).'
 230                  WHERE cptq_ID NOT IN ( '.implode( ', ', $questions_IDs ).' )' );
 231  
 232              // Resave the questions in order to delere all whitespaces and questions with incorrect format
 233              $this->Settings->set( 'questions', implode( "\r\n", $questions_formated ) );
 234  
 235              return true;
 236          }
 237      }

 240      /**
 241       * Validate the given answer against our stored one.
 242       *
 243       * This event is provided for other plugins and gets used internally
 244       * for other events we're hooking into.
 245       *
 246       * @param array Associative array of parameters.
 247       * @param string Form type ( comment|register|message )
 248       * @return boolean|NULL
 249       */
 250  	function CaptchaValidated( & $params, $form_type )
 251      {
 252          global $DB, $localtimenow, $Session;
 253  
 254          if( ! $this->does_apply( $params, $form_type ) )
 255          {
 256              return;
 257          }
 258  
 259          $posted_answer = evo_strtolower( param( 'captcha_qstn_'.$this->ID.'_answer', 'string', '' ) );
 260  
 261          if( empty( $posted_answer ) )
 262          {
 263              $this->debug_log( 'captcha_qstn_'.$this->ID.'_answer' );
 264              $params['validate_error'] = $this->T_('Please enter the captcha answer.');
 265              return false;
 266          }
 267  
 268          $question = $this->CaptchaQuestion();
 269  
 270          $posted_answer_is_correct = false;
 271  
 272          $answers = explode( '|', evo_strtolower( $question->cptq_answers ) );
 273          foreach( $answers as $answer )
 274          {
 275              if( $posted_answer == $answer )
 276              {    // Correct answer is found in DB
 277                  $posted_answer_is_correct = true;
 278                  break;
 279              }
 280          }
 281  
 282          if( !$posted_answer_is_correct )
 283          {
 284              $this->debug_log( 'Posted ('.$posted_answer.') and saved ('.$question->cptq_answers.') answer do not match!' );
 285              $params['validate_error'] = $this->T_('The entered answer is incorrect.');
 286              return false;
 287          }
 288  
 289          // If answer is correct:
 290          //   We should clean the question ID that was assigned for current session and IP address
 291          //   It gives to assign new question on the next captcha event
 292          $this->CaptchaQuestionCleanup();
 293  
 294          return true;
 295      }

 299       * Get question for current user
 300       *
 301       * @return object Question row from DB table "questions"

 303  	function CaptchaQuestion()
 304      {
 305          global $DB, $Session;
 306  
 307          $question = NULL;
 308  
 309          $IP = ip2int( $_SERVER['REMOTE_ADDR'] );
 310  
 311          // Get question ID from Session
 312          $this->question_ID = $Session->get( 'captcha_qstn_'.$this->ID.'_ID' );
 313  
 314          if( empty( $this->question_ID ) )
 315          {    // Get question from DB by current IP address
 316              $SQL = new SQL();
 317              $SQL->SELECT( 'cq.*' );
 318              $SQL->FROM( $this->get_sql_table( 'ip_question' ) );
 319              $SQL->FROM_add( 'INNER JOIN '.$this->get_sql_table( 'questions' ).' AS cq ON cptip_cptq_ID = cptq_ID' );
 320              $SQL->WHERE( 'cptip_IP = '.$DB->quote( $IP ) );
 321  
 322              if( $question = $DB->get_row( $SQL->get() ) )
 323              {
 324                  $this->question_ID = $question->cptq_ID;
 325              }
 326          }
 327  
 328          if( empty( $this->question_ID ) )
 329          {    // Assign new random question for current IP address
 330              $question = $this->CaptchaQuestionNew();
 331          }
 332  
 333          if( empty( $question ) && !empty( $this->question_ID ) )
 334          {    // Get question data
 335              $SQL = new SQL();
 336              $SQL->SELECT( '*' );
 337              $SQL->FROM( $this->get_sql_table( 'questions' ) );
 338              $SQL->WHERE( 'cptq_ID = '.$DB->quote( $this->question_ID ) );
 339              $question = $DB->get_row( $SQL->get() );
 340  
 341              if( empty( $question ) )
 342              {    // Assign random question if previous question doesn't exist in DB
 343                  // This case may happens when admin changed the questions but user has the old question ID in the session or in DB table "ip_question"
 344                  $question = $this->CaptchaQuestionNew();
 345              }
 346          }
 347  
 348          return $question;
 349      }

 356  	function CaptchaQuestionNew()
 357      {
 358          global $DB, $Session;
 359  
 360          $IP = ip2int( $_SERVER['REMOTE_ADDR'] );
 361  
 362          // Get new random question from DB
 363          $SQL = new SQL();
 364          $SQL->SELECT( '*' );
 365          $SQL->FROM( $this->get_sql_table( 'questions' ) );
 366          $SQL->ORDER_BY( 'RAND()' );
 367          $SQL->LIMIT( 1 );
 368          $question = $DB->get_row( $SQL->get() );
 369  
 370          // Insert a record for current IP address with assigned question ID
 371          $DB->query( 'INSERT INTO '.$this->get_sql_table( 'ip_question' ).'
 372              ( cptip_IP, cptip_cptq_ID ) VALUES
 373              ( '.$DB->quote( $IP ).', '.$DB->quote( $question->cptq_ID ).' )' );
 374  
 375          // Save the assigned question ID in the session
 376          $Session->set( 'captcha_qstn_'.$this->ID.'_ID', $question->cptq_ID );
 377          $Session->dbsave();
 378  
 379          $this->question_ID = $question->cptq_ID;
 380  
 381          return $question;
 382      }

 403      /**
 404       * When a comment form gets displayed, we inject our captcha and an input field to
 405       * enter the answer.
 406       *
 407       * The question ID is saved into the user's Session and in the DB table "ip_question".
 408       *
 409       * @param array Associative array of parameters
 410       *   - 'Form': the form where payload should get added (by reference, OPTIONALLY!)
 411       *   - 'form_use_fieldset':
 412       *   - 'key': A key that is associated to the caller of the event (string, OPTIONALLY!)
 413       * @param string Form type ( comment|register|message )
 414       * @return boolean|NULL true, if displayed; false, if error; NULL if it does not apply
 415       */
 416  	function CaptchaPayload( & $params, $form_type )
 417      {
 418          global $DB, $Session;
 419  
 420          if( ! $this->does_apply( $params, $form_type ) )
 421          {
 422              return;
 423          }
 424  
 425          $question = $this->CaptchaQuestion();
 426  
 427          if( empty( $question ) )
 428          {    // No the defined questions
 429              return;
 430          }
 431  
 432          $this->debug_log( 'Question ID is: ('.$this->question_ID.')' );
 433  
 434          if( ! isset( $params['Form'] ) )
 435          {    // there's no Form where we add to, but we create our own form:
 436              $Form = new Form( regenerate_url() );
 437              $Form->begin_form();
 438          }
 439          else
 440          {
 441              $Form = & $params['Form'];
 442              if( ! isset( $params['form_use_fieldset'] ) || $params['form_use_fieldset'] )
 443              {
 444                  $Form->begin_fieldset();
 445              }
 446          }
 447  
 448          $Form->info( $this->T_('Captcha question'), $question->cptq_question );
 449          $Form->text_input( 'captcha_qstn_'.$this->ID.'_answer', param( 'captcha_qstn_'.$this->ID.'_answer', 'string', '' ), 10, $this->T_('Captcha answer'), $this->T_('Please answer on question above.') );
 450  
 451          if( ! isset($params['Form']) )
 452          {    // there's no Form where we add to, but our own form:
 453              $Form->end_form( array( array( 'submit', 'submit', $this->T_('Validate me'), 'ActionButton' ) ) );
 454          }
 455          else
 456          {
 457              if( ! isset($params['form_use_fieldset']) || $params['form_use_fieldset'] )
 458              {
 459                  $Form->end_fieldset();
 460              }
 461          }
 462  
 463          return true;
 464      }

 569                  {
