  93  	function RenderItemAsHtml( & $params )
  94      {
  95          $content = & $params['data'];
  96  
  97          $this->allow_html = false;
  98          // Get a setting "Allow HTML" for:

  99          if( ! empty( $params['Comment'] ) )
 100          { // Comment
 101              $Comment = & $params['Comment'];
 102              $comment_Item = & $Comment->get_Item();
 103              $item_Blog = $comment_Item->get_Blog();
 104              $this->allow_html = $item_Blog->get_setting( 'allow_html_comment' );
 105          }
 106          else if( ! empty( $params['Item'] ) )
 107          { // Item
 108              $Item = & $params['Item'];
 109              $item_Blog = $Item->get_Blog();
 110              $this->allow_html = $item_Blog->get_setting( 'allow_html_post' );
 111          }
 112  
 113          if( $this->allow_html )
 114          { // If HTML is allowed in content we should disallow this for <code> content
 115              $content = preg_replace_callback( '#(<code class="language-[a-z]+">)([\s\S]+?)(</code>)#i',
 116                  array( $this, 'encode_html_entities' ), $content );
 117          }
 118      }

 134       * Convert code blocks to html tags

 135       *

 136       * @param string Content

 137       * @return string Content

 139  	function filter_code( $content )
 140      {
 141          // change all [codeblock]  segments before format_to_post() gets a hold of them

 142          // 1 - codeblock or codespan

 143          // 2 - attribs : lang &| line

 144          // 3 - code content

 145          $content = preg_replace_callback( '#\[(codeblock|codespan)([^\]]*?)\]([\s\S]+?)?\[/\1\]#i',
 146                                  array( $this, 'filter_code_callback' ), $content );
 147  
 148          return $content;
 149      }

 152      /**

 153       * Formats code ready for rendering

 154       *

 155       * @param array $block ( 1 - type, 2 - attributes, 3 - content )

 156       * @return string formatted code || empty

 157       */
 158  	function filter_code_callback( $block )
 159      {
 160          $content = trim( $block[3] );
 161  
 162          if( empty( $content ) )
 163          { // Don't render if no code content
 164              return '';
 165          }
 166  
 167          // Type of code: 'codeblock' OR 'codespan'

 168          $type = $block[1];
 169  
 170          // Language:

 171          $lang = strtolower( preg_replace( '/.*lang="?([a-z]+)"?.*/i', '$1', html_entity_decode( $block[2] ) ) );
 172          if( ! in_array( $lang, array( 'php', 'css', 'javascript', 'sql' ) ) )
 173          { // Use Markup for unknown language
 174              $lang = 'markup';
 175          }
 176  
 177          $r = '<code class="language-'.$lang.'">'.$block[3].'</code>';
 178  
 179          if( $type == 'codeblock' )
 180          { // Set special template and attributes only for codeblock
 181  
 182              // Detect number of start line:

 183              $line = intval( preg_replace( '/.*line="?(-?[0-9]+)"?.*/i', '$1', html_entity_decode( $block[2] ) ) );
 184              $line = $line != 1 ? ' data-start="'.$line.'"' : '';
 185  
 186              // Put <pre> around <code> to render codeblock

 187              $r = '<pre class="line-numbers"'.$line.'>'.$r.'</pre>';
 188          }
 189  
 190          return $r;
 191      }

 209      /**

 210       * Formats code ready for editing

 211       *

 212       * @param array $block ( 1 - start of <pre> tag, 4 - language, 5 - content )

 213       * @return string formatted code || empty

 214       */
 215  	function unfilter_code_callback( $block )
 216      {
 217          if( empty( $block[1] ) )
 218          { // [codespan]
 219              $code_tag = 'codespan';
 220              // codespan doesn't provide line numbers

 221              $line = '';
 222          }
 223          else
 224          { // [codeblock]
 225              $code_tag = 'codeblock';
 226              // Detect number of start line:

 227              preg_match( '/.*data-start="(-?[0-9]+)".*/i', html_entity_decode( $block[1] ), $line );
 228              $line = ' line="'.( isset( $line[1] ) ? intval( $line[1] ) : '1' ).'"';
 229          }
 230  
 231          // Build codeblock

 232          $r = '['.$code_tag.' lang="'.strtolower( $block[4] ).'"'.$line.']';
 233          $r .= $block[5];
 234          $r .= '[/'.$code_tag.']';
 235  
 236          return $r;
 237      }

 252       * Event handler: Called when displaying editor toolbars.

 255       * @return boolean did we display a toolbar?

 257  	function DisplayCommentToolbar( & $params )
 258      {
 259          if( !empty( $params['Comment'] ) )
 260          { // Comment is set, get Blog from comment
 261              $Comment = & $params['Comment'];
 262              if( !empty( $Comment->item_ID ) )
 263              {
 264                  $comment_Item = & $Comment->get_Item();
 265                  $Blog = & $comment_Item->get_Blog();
 266              }
 267          }
 268  
 269          if( empty( $Blog ) )
 270          { // Comment is not set, try global Blog
 271              global $Blog;
 272              if( empty( $Blog ) )
 273              { // We can't get a Blog, this way "apply_comment_rendering" plugin collection setting is not available
 274                  return false;
 275              }
 276          }
 277  
 278          $apply_rendering = $this->get_coll_setting( 'coll_apply_comment_rendering', $Blog );
 279          if( ! empty( $apply_rendering ) && $apply_rendering != 'never' )
 280          {
 281              return $this->display_toolbar();
 282          }
 283          return false;
 284      }

 293  	function AdminDisplayToolbar( & $params )
 294      {
 295          if( !empty( $params['Item'] ) )
 296          { // Item is set, get Blog from post
 297              $edited_Item = & $params['Item'];
 298              $Blog = & $edited_Item->get_Blog();
 299          }
 300  
 301          if( empty( $Blog ) )
 302          { // Item is not set, try global Blog
 303              global $Blog;
 304              if( empty( $Blog ) )
 305              { // We can't get a Blog, this way "apply_rendering" plugin collection setting is not available
 306                  return false;
 307              }
 308          }
 309  
 310          $coll_setting_name = ( $params['target_type'] == 'Comment' ) ? 'coll_apply_comment_rendering' : 'coll_apply_rendering';
 311          $apply_rendering = $this->get_coll_setting( $coll_setting_name, $Blog );
 312          if( empty( $apply_rendering ) || $apply_rendering == 'never' )
 313          { // Don't display toolbar
 314              return false;
 315          }
 316  
 317          // Display toolbar

 318          $this->display_toolbar();
 319      }
