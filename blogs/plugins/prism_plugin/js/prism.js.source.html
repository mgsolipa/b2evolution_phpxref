 192      tokenize: function(text, grammar, language) {
 193          var Token = _.Token;
 194  
 195          var strarr = [text];
 196  
 197          var rest = grammar.rest;
 198  
 199          if (rest) {
 200              for (var token in rest) {
 201                  grammar[token] = rest[token];
 202              }
 203  
 204              delete grammar.rest;
 205          }
 206  
 207          tokenloop: for (var token in grammar) {
 208              if(!grammar.hasOwnProperty(token) || !grammar[token]) {
 209                  continue;
 210              }
 211  
 212              var pattern = grammar[token],
 213                  inside = pattern.inside,
 214                  lookbehind = !!pattern.lookbehind,
 215                  lookbehindLength = 0;
 216  
 217              pattern = pattern.pattern || pattern;
 218  
 219              for (var i=0; i<strarr.length; i++) { // Don’t cache length as it changes during the loop
 220  
 221                  var str = strarr[i];
 222  
 223                  if (strarr.length > text.length) {
 224                      // Something went terribly wrong, ABORT, ABORT!
 225                      break tokenloop;
 226                  }
 227  
 228                  if (str instanceof Token) {
 229                      continue;
 230                  }
 231  
 232                  pattern.lastIndex = 0;
 233  
 234                  var match = pattern.exec(str);
 235  
 236                  if (match) {
 237                      if(lookbehind) {
 238                          lookbehindLength = match[1].length;
 239                      }
 240  
 241                      var from = match.index - 1 + lookbehindLength,
 242                          match = match[0].slice(lookbehindLength),
 243                          len = match.length,
 244                          to = from + len,
 245                          before = str.slice(0, from + 1),
 246                          after = str.slice(to + 1);
 247  
 248                      var args = [i, 1];
 249  
 250                      if (before) {
 251                          args.push(before);
 252                      }
 253  
 254                      var wrapped = new Token(token, inside? _.tokenize(match, inside) : match);
 255  
 256                      args.push(wrapped);
 257  
 258                      if (after) {
 259                          args.push(after);
 260                      }
 261  
 262                      Array.prototype.splice.apply(strarr, args);
 263                  }
 264              }
 265          }
 266  
 267          return strarr;
 268      },
