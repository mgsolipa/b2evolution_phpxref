 182      /**

 183       * Get country by IP address

 184       *

 185       * @param string IP in format xxx.xxx.xxx.xxx

 186       * @return object Country

 187       */
 188  	function get_country_by_IP( $IP )
 189      {
 190          if( function_exists('geoip_country_code_by_name') )
 191          {    // GeoIP extension
 192  
 193              // Get country code by user IP address

 194              $country_code = @geoip_country_code_by_name($IP);
 195          }
 196          else
 197          {    // Include GeoIP API
 198              require_once( dirname( __FILE__ ).'/geoip.inc' );
 199  
 200              // Open GeoIP database

 201              $GeoIP = geoip_open( $this->geoip_file_path, GEOIP_STANDARD );
 202  
 203              // Get country code by user IP address

 204              $country_code = geoip_country_code_by_addr( $GeoIP, $IP );
 205  
 206              // Close GeoIP DB

 207              geoip_close( $GeoIP );
 208          }
 209  
 210          if( ! $country_code )
 211          {    // No found country with current IP address
 212              return false;
 213          }
 214  
 215          global $DB;
 216  
 217          // Get country ID by code

 218          $SQL = new SQL();
 219          $SQL->SELECT( 'ctry_ID' );
 220          $SQL->FROM( 'T_regional__country' );
 221          $SQL->WHERE( 'ctry_code = '.$DB->quote( strtolower( $country_code ) ) );
 222          $country_ID = $DB->get_var( $SQL->get() );
 223  
 224          if( !$country_ID )
 225          { // No found country in the b2evo DB
 226              return false;
 227          }
 228  
 229          // Load Country class (PHP4):

 230          load_class( 'regional/model/_country.class.php', 'Country' );
 231  
 232          $CountryCache = & get_CountryCache();
 233          $Country = $CountryCache->get_by_ID( $country_ID, false );
 234  
 235          return $Country;
 236      }

 289      /**

 290       * AJAX callback to load country.

 291       *

 292       * @param array Associative array of parameters

 293       *   - 'user_ID': User ID

 294       */
 295  	function htsrv_load_country( $params )
 296      {
 297          global $debug, $debug_jslog;
 298          // Do not append Debuglog to response!

 299          $debug = false;
 300          // Do not append Debug JSlog to response!

 301          $debug_jslog = false;
 302  
 303          $user_ID = $params['user_ID'];
 304  
 305          if( empty( $user_ID ) )
 306          {    // Bad request
 307              return;
 308          }
 309  
 310          $UserCache = & get_UserCache();
 311          if( ! ( $User = & $UserCache->get_by_ID( $user_ID ) ) )
 312          {    // No user exists
 313              return;
 314          }
 315  
 316          global $UserSettings;
 317          // Get Country by IP address

 318          $Country = $this->get_country_by_IP( int2ip( $UserSettings->get( 'created_fromIPv4', $User->ID ) ) );
 319  
 320          if( empty( $Country ) )
 321          {    // No found country
 322              echo sprintf( T_('No country found for IP address %s'), int2ip( $UserSettings->get( 'created_fromIPv4', $User->ID ) ) );
 323          }
 324          else
 325          {    // Display country name with flag and Update user's field 'From Country'
 326              load_funcs( 'regional/model/_regional.funcs.php' );
 327              country_flag( $Country->get( 'code' ), $Country->get_name() );
 328              echo ' '.$Country->get_name();
 329  
 330              // Update user

 331              global $DB;
 332              $DB->query( 'UPDATE T_users
 333                        SET user_reg_ctry_ID = '.$DB->quote( $Country->ID ).'
 334                      WHERE user_ID = '.$DB->quote( $User->ID ) );
 335  
 336              // Move user to suspect group by Country ID

 337              antispam_suspect_user_by_country( $Country->ID, $User->ID );
 338          }
 339      }

 381  	function DisplayRegisterFormBefore( & $params )
 382      {
 383          global $Settings;
 384  
 385          $registration_require_country = (bool)$Settings->get('registration_require_country');
 386          if( !$registration_require_country )
 387          {    // Country is not required on registration form. Exit here.
 388              return;
 389          }
 390  
 391          $detect_registration = $this->Settings->get( 'detect_registration' );
 392          if( $detect_registration == 'no' )
 393          {    // No detect country on registration
 394              return;
 395          }
 396  
 397          $Country = $this->get_country_by_IP( get_ip_list( true ) );
 398  
 399          if( !$Country )
 400          {    // No found country by IP address
 401              return;
 402          }
 403  
 404          switch( $detect_registration )
 405          {
 406              case 'auto':
 407                  // Auto select current country in list

 408                  $country = param( 'country', 'integer', 0 );
 409                  if( empty( $country ) )
 410                  {    // Set country ID if user didn't select country yet
 411                      set_param( 'country', $Country->ID );
 412                  }
 413                  break;
 414  
 415              case 'hide':
 416                  // Hide country selector if a country has been detected

 417                  if( ! isset( $params['Form'] ) )
 418                  {    // there's no Form where we add to, but we create our own form:
 419                      $Form = new Form( regenerate_url() );
 420  
 421                  }
 422                  else
 423                  {
 424                      $Form = & $params['Form'];
 425                  }
 426  
 427                  // Disable this setting temporary to hide a select list with countries

 428                  $Settings->set( 'registration_require_country', 0 );
 429  
 430                  // Append a hidden input element with autodetected country ID

 431                  $Form->hidden( 'country', $Country->ID );
 432                  break;
 433          }
 434      }

 507      /**

 508       * Event handler: Called when handling actions for the "Tools" menu.

 509       *

 510       * Use {@link $Messages} to add Messages for the user.

 511       *

 512       * @see Plugin::AdminToolAction()

 513       */
 514  	function AdminToolAction()
 515      {
 516          $action = param_action();
 517  
 518          if( !empty( $action ) )
 519          {    // If form is submitted
 520              global $DB;
 521  
 522              switch( $action )
 523              {
 524                  case 'geoip_find_country':
 525                  // Find and Assign Registration Country for all Users

 526  
 527                      $SQL = new SQL( 'Find all users without registration country' );
 528                      $SQL->SELECT( 'user_ID, uset_value' );
 529                      $SQL->FROM( 'T_users' );
 530                      $SQL->FROM_add( 'LEFT JOIN T_users__usersettings
 531                           ON user_ID = uset_user_ID
 532                          AND uset_name = "created_fromIPv4"' );
 533                      $SQL->WHERE( 'user_reg_ctry_ID IS NULL' );
 534                      $users = $DB->get_assoc( $SQL->get() );
 535  
 536                      $total_users = count( $users );
 537                      if( $total_users == 0 )
 538                      {    // No users
 539                          $this->text_from_AdminTabAction = T_('No found users without registration country.');
 540                          break;
 541                      }
 542                      $count_nofound_country = 0;
 543  
 544                      $users_report = '';
 545                      foreach( $users as $user_ID => $created_fromIPv4 )
 546                      {
 547                          $users_report .= sprintf( T_('User #%s, IP:%s' ), $user_ID, int2ip( $created_fromIPv4 ) );
 548                          if( empty( $created_fromIPv4 ) )
 549                          {    // No defined IP, Skip this user
 550                              $count_nofound_country++;
 551                              $users_report .= ' - <b class="orange">'.T_('IP is not defined!').'</b><br />';
 552                              continue;
 553                          }
 554  
 555                          // Get Country by IP address

 556                          $Country = $this->get_country_by_IP( int2ip( $created_fromIPv4 ) );
 557  
 558                          if( !$Country )
 559                          {    // No found country by IP address
 560                              $count_nofound_country++;
 561                              $users_report .= ' - <b class="red">'.T_('Country is not detected!').'</b><br />';
 562                              continue;
 563                          }
 564  
 565                          // Update user's registration country

 566                          $DB->query( 'UPDATE T_users
 567                                    SET user_reg_ctry_ID = '.$DB->quote( $Country->ID ).'
 568                                  WHERE user_ID = '.$DB->quote( $user_ID ) );
 569  
 570                          // Move user to suspect group by Country ID

 571                          antispam_suspect_user_by_country( $Country->ID, $user_ID );
 572  
 573                          $users_report .= ' - '.sprintf( T_('Country: <b>%s</b>'), $Country->get( 'name' ) ).'<br />';
 574                      }
 575  
 576                      $this->text_from_AdminTabAction = '<div>'.sprintf( T_('Count of users without registration country: <b>%s</b>' ), $total_users ).'</div>';
 577                      if( $count_nofound_country > 0 )
 578                      {    // If some users have IP address with unknown country
 579                          $this->text_from_AdminTabAction .= '<div>'.sprintf( T_('Count of users whose country could not be identified: <b>%s</b>' ), $count_nofound_country ).'</div>';
 580                      }
 581                      $this->text_from_AdminTabAction .= '<div style="margin-top:20px">'.$users_report.'</div>';
 582  
 583                      break;
 584              }
 585          }
 586      }

 635  /**

 636   * Get country by IP address (Used in the method GetAdditionalColumnsTable() in the table column)

 637   *

 638   * @param string|integer IP address

 639   * @return string Country with flag

 640   */
 641  function geoip_get_country_by_IP( $IP )
 642  {
 643      global $Plugins, $Timer;
 644  
 645      $Timer->resume( 'plugin_geoip' );
 646  
 647      $country = '';
 648  
 649      if( $Plugins && $geoip_Plugin = & $Plugins->get_by_code( 'evo_GeoIP' ) )
 650      {
 651          if( strlen( intval( $IP ) ) == strlen( $IP ) )
 652          { // IP is in integer format, We should convert it to normal IP
 653              $IP = int2ip( $IP );
 654          }
 655  
 656          if( $Country = & $geoip_Plugin->get_country_by_IP( $IP ) )
 657          { // Get country flag + name
 658              load_funcs( 'regional/model/_regional.funcs.php' );
 659              $country = country_flag( $Country->get( 'code' ), $Country->get_name(), 'w16px', 'flag', '', false ).
