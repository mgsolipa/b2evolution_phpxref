 260  		function b2evoCheckOpenTags(button)
 261          {
 262              var tag = 0;
 263              for (i = 0; i < b2evoOpenTags.length; i++)
 264              {
 265                  if (b2evoOpenTags[i] == button)
 266                  {
 267                      tag++;
 268                  }
 269              }
 270  
 271              if (tag > 0)
 272              {
 273                  return true; // tag found
 274              }
 275              else
 276              {
 277                  return false; // tag not found
 278              }

 304  		function b2evoInsertTag( myField, i )
 305          {
 306              // we need to know if something is selected.
 307              // First, ask plugins, then try IE and Mozilla.
 308              var sel_text = b2evo_Callbacks.trigger_callback("get_selected_text_for_"+myField.id);
 309              var focus_when_finished = false; // used for IE
 310  
 311              if( sel_text == null )
 312              { // detect selection:
 313                  //IE support
 314                  if(document.selection)
 315                  {
 316                      myField.focus();
 317                      var sel = document.selection.createRange();
 318                      sel_text = sel.text;
 319                      focus_when_finished = true;
 320                  }
 321                  //MOZILLA/NETSCAPE support
 322                  else if(myField.selectionStart || myField.selectionStart == '0')
 323                  {
 324                      var startPos = myField.selectionStart;
 325                      var endPos = myField.selectionEnd;
 326                      sel_text = (startPos != endPos);
 327                  }
 328              }
 329  
 330              if( sel_text )
 331              { // some text selected
 332                  textarea_wrap_selection( myField, b2evoButtons[i].tagStart, b2evoButtons[i].tagEnd, 0 );
 333              }
 334              else
 335              {
 336                  if( !b2evoCheckOpenTags(i) || b2evoButtons[i].tagEnd == '')
 337                  {
 338                      textarea_wrap_selection( myField, b2evoButtons[i].tagStart, '', 0 );
 339                      b2evoAddTag(i);
 340                  }
 341                  else
 342                  {
 343                      textarea_wrap_selection( myField, '', b2evoButtons[i].tagEnd, 0 );
 344                      b2evoRemoveTag(i);
 345                  }
 346              }
 347              if(focus_when_finished)
 348              {
 349                  myField.focus();
 350              }
 351          }

 355          {
 356              if (!defaultValue)

 359              }
 360  
 361              if (!b2evoCheckOpenTags(i)) {
 362                  var URL = prompt( '<?php echo TS_('URL') ?>:', defaultValue);
 363                  if (URL)
 364                  {
 365                      b2evoButtons[i].tagStart = '<a href="' + URL + '">';
 366                      b2evoInsertTag(myField, i);
 367                  }
 368              }
 369              else
 370              {
 371                  b2evoInsertTag( myField, i );
 372              }
