  88  function debug_get_config( $close_handle = true )
  89  {
  90      global $errors, $file_overrides_name, $file_section_start, $file_section_end, $file_overrides_handle;
  91  
  92      // Get debug config from file

  93      $file_overrides_handle = @fopen( $file_overrides_name, 'r+' );
  94      $file_overrides_content = '';
  95      if( $file_overrides_handle )
  96      {
  97          while( !feof( $file_overrides_handle ) )
  98          {
  99              $file_overrides_content .= fgets( $file_overrides_handle, 4096 );
 100          }
 101          if( $close_handle )
 102          {
 103              fclose( $file_overrides_handle );
 104          }
 105      }
 106      else
 107      {    // Bad file name or file has not the rights to read/write
 108          $errors[] = T_('Config file cannot be opened');
 109          return false;
 110      }
 111  
 112      return $file_overrides_content;
 113  }
