 153      /**
 154       * Decode a given ACE domain name
 155       * @param    string   Domain name (ACE string)
 156       * [@param    string   Desired output encoding, see {@link set_parameter}]
 157       * @return   string   Decoded Domain name (UTF-8 or UCS-4)
 158       */
 159      public function decode($input, $one_time_encoding = false)
 160      {
 161          // Optionally set
 162          if ($one_time_encoding) {
 163              switch ($one_time_encoding) {
 164              case 'utf8':
 165              case 'ucs4_string':
 166              case 'ucs4_array':
 167                  break;
 168              default:
 169                  $this->_error('Unknown encoding '.$one_time_encoding);
 170                  return false;
 171              }
 172          }
 173          // Make sure to drop any newline characters around
 174          $input = trim($input);
 175  
 176          // Negotiate input and try to determine, whether it is a plain string,
 177          // an email address or something like a complete URL
 178          if (strpos($input, '@')) { // Maybe it is an email address
 179              // No no in strict mode
 180              if ($this->_strict_mode) {
 181                  $this->_error('Only simple domain name parts can be handled in strict mode');
 182                  return false;
 183              }
 184              list ($email_pref, $input) = explode('@', $input, 2);
 185              $arr = explode('.', $input);
 186              foreach ($arr as $k => $v) {
 187                  if (preg_match('!^'.preg_quote($this->_punycode_prefix, '!').'!', $v)) {
 188                      $conv = $this->_decode($v);
 189                      if ($conv) $arr[$k] = $conv;
 190                  }
 191              }
 192              $input = join('.', $arr);
 193              $arr = explode('.', $email_pref);
 194              foreach ($arr as $k => $v) {
 195                  if (preg_match('!^'.preg_quote($this->_punycode_prefix, '!').'!', $v)) {
 196                      $conv = $this->_decode($v);
 197                      if ($conv) $arr[$k] = $conv;
 198                  }
 199              }
 200              $email_pref = join('.', $arr);
 201              $return = $email_pref . '@' . $input;
 202          } elseif (preg_match('![:\./]!', $input)) { // Or a complete domain name (with or without paths / parameters)
 203              // No no in strict mode
 204              if ($this->_strict_mode) {
 205                  $this->_error('Only simple domain name parts can be handled in strict mode');
 206                  return false;
 207              }
 208              $parsed = parse_url($input);
 209              if (isset($parsed['host'])) {
 210                  $arr = explode('.', $parsed['host']);
 211                  foreach ($arr as $k => $v) {
 212                      $conv = $this->_decode($v);
 213                      if ($conv) $arr[$k] = $conv;
 214                  }
 215                  $parsed['host'] = join('.', $arr);
 216                  $return =
 217                          (empty($parsed['scheme']) ? '' : $parsed['scheme'].(strtolower($parsed['scheme']) == 'mailto' ? ':' : '://'))
 218                          .(empty($parsed['user']) ? '' : $parsed['user'].(empty($parsed['pass']) ? '' : ':'.$parsed['pass']).'@')
 219                          .$parsed['host']
 220                          .(empty($parsed['port']) ? '' : ':'.$parsed['port'])
 221                          .(empty($parsed['path']) ? '' : $parsed['path'])
 222                          .(empty($parsed['query']) ? '' : '?'.$parsed['query'])
 223                          .(empty($parsed['fragment']) ? '' : '#'.$parsed['fragment']);
 224              } else { // parse_url seems to have failed, try without it
 225                  $arr = explode('.', $input);
 226                  foreach ($arr as $k => $v) {
 227                      $conv = $this->_decode($v);
 228                      $arr[$k] = ($conv) ? $conv : $v;
 229                  }
 230                  $return = join('.', $arr);
 231              }
 232          } else { // Otherwise we consider it being a pure domain name string
 233              $return = $this->_decode($input);
 234              if (!$return) $return = $input;
 235          }
 236          // The output is UTF-8 by default, other output formats need conversion here
 237          // If one time encoding is given, use this, else the objects property
 238          switch (($one_time_encoding) ? $one_time_encoding : $this->_api_encoding) {
 239          case 'utf8':
 240              return $return;
 241              break;
 242          case 'ucs4_string':
 243             return $this->_ucs4_to_ucs4_string($this->_utf8_to_ucs4($return));
 244             break;
 245          case 'ucs4_array':
 246              return $this->_utf8_to_ucs4($return);
 247              break;
 248          default:
 249              $this->_error('Unsupported output format');
 250              return false;
 251          }
 252      }

 254      /**
 255       * Encode a given UTF-8 domain name
 256       * @param    string   Domain name (UTF-8 or UCS-4)
 257       * [@param    string   Desired input encoding, see {@link set_parameter}]
 258       * @return   string   Encoded Domain name (ACE string)

 260      public function encode($decoded, $one_time_encoding = false)
 261      {
 262          // Forcing conversion of input to UCS4 array
 263          // If one time encoding is given, use this, else the objects property
 264          switch ($one_time_encoding ? $one_time_encoding : $this->_api_encoding) {
 265          case 'utf8':
 266              $decoded = $this->_utf8_to_ucs4($decoded);
 267              break;
 268          case 'ucs4_string':
 269             $decoded = $this->_ucs4_string_to_ucs4($decoded);
 270          case 'ucs4_array':
 271             break;
 272          default:
 273              $this->_error('Unsupported input format: '.($one_time_encoding ? $one_time_encoding : $this->_api_encoding));
 274              return false;
 275          }
 276  
 277          // No input, no output, what else did you expect?
 278          if (empty($decoded)) return '';
 279  
 280          // Anchors for iteration
 281          $last_begin = 0;
 282          // Output string
 283          $output = '';
 284          foreach ($decoded as $k => $v) {
 285              // Make sure to use just the plain dot
 286              switch($v) {
 287              case 0x3002:
 288              case 0xFF0E:
 289              case 0xFF61:
 290                  $decoded[$k] = 0x2E;
 291                  // Right, no break here, the above are converted to dots anyway
 292              // Stumbling across an anchoring character
 293              case 0x2E:
 294              case 0x2F:
 295              case 0x3A:
 296              case 0x3F:
 297              case 0x40:
 298                  // Neither email addresses nor URLs allowed in strict mode
 299                  if ($this->_strict_mode) {
 300                     $this->_error('Neither email addresses nor URLs are allowed in strict mode.');
 301                     return false;
 302                  } else {
 303                      // Skip first char
 304                      if ($k) {
 305                          $encoded = '';
 306                          $encoded = $this->_encode(array_slice($decoded, $last_begin, (($k)-$last_begin)));
 307                          if ($encoded) {
 308                              $output .= $encoded;
 309                          } else {
 310                              $output .= $this->_ucs4_to_utf8(array_slice($decoded, $last_begin, (($k)-$last_begin)));
 311                          }
 312                          $output .= chr($decoded[$k]);
 313                      }
 314                      $last_begin = $k + 1;
 315                  }
 316              }
 317          }
 318          // Catch the rest of the string
 319          if ($last_begin) {
 320              $inp_len = sizeof($decoded);
 321              $encoded = '';
 322              $encoded = $this->_encode(array_slice($decoded, $last_begin, (($inp_len)-$last_begin)));
 323              if ($encoded) {
 324                  $output .= $encoded;
 325              } else {
 326                  $output .= $this->_ucs4_to_utf8(array_slice($decoded, $last_begin, (($inp_len)-$last_begin)));
 327              }
 328              return $output;
 329          } else {
 330              if ($output = $this->_encode($decoded)) {
 331                  return $output;
 332              } else {
 333                  return $this->_ucs4_to_utf8($decoded);
 334              }
 335          }
 336      }

 348      /**
 349       * The actual decoding algorithm
 350       * @param string
 351       * @return mixed
 352       */
 353      private function _decode($encoded)
 354      {
 355          $decoded = array();
 356          // find the Punycode prefix
 357          if (!preg_match('!^'.preg_quote($this->_punycode_prefix, '!').'!', $encoded)) {
 358              $this->_error('This is not a punycode string');
 359              return false;
 360          }
 361          $encode_test = preg_replace('!^'.preg_quote($this->_punycode_prefix, '!').'!', '', $encoded);
 362          // If nothing left after removing the prefix, it is hopeless
 363          if (!$encode_test) {
 364              $this->_error('The given encoded string was empty');
 365              return false;
 366          }
 367          // Find last occurence of the delimiter
 368          $delim_pos = strrpos($encoded, '-');
 369          if ($delim_pos > strlen($this->_punycode_prefix)) {
 370              for ($k = strlen($this->_punycode_prefix); $k < $delim_pos; ++$k) {
 371                  $decoded[] = ord($encoded{$k});
 372              }
 373          }
 374          $deco_len = count($decoded);
 375          $enco_len = strlen($encoded);
 376  
 377          // Wandering through the strings; init
 378          $is_first = true;
 379          $bias = $this->_initial_bias;
 380          $idx = 0;
 381          $char = $this->_initial_n;
 382  
 383          for ($enco_idx = ($delim_pos) ? ($delim_pos + 1) : 0; $enco_idx < $enco_len; ++$deco_len) {
 384              for ($old_idx = $idx, $w = 1, $k = $this->_base; 1 ; $k += $this->_base) {
 385                  $digit = $this->_decode_digit($encoded{$enco_idx++});
 386                  $idx += $digit * $w;
 387                  $t = ($k <= $bias) ? $this->_tmin :
 388                          (($k >= $bias + $this->_tmax) ? $this->_tmax : ($k - $bias));
 389                  if ($digit < $t) break;
 390                  $w = (int) ($w * ($this->_base - $t));
 391              }
 392              $bias = $this->_adapt($idx - $old_idx, $deco_len + 1, $is_first);
 393              $is_first = false;
 394              $char += (int) ($idx / ($deco_len + 1));
 395              $idx %= ($deco_len + 1);
 396              if ($deco_len > 0) {
 397                  // Make room for the decoded char
 398                  for ($i = $deco_len; $i > $idx; $i--) $decoded[$i] = $decoded[($i - 1)];
 399              }
 400              $decoded[$idx++] = $char;
 401          }
 402          return $this->_ucs4_to_utf8($decoded);
 403      }

 405      /**
 406       * The actual encoding algorithm
 407       * @param  string
 408       * @return mixed
 409       */
 410      private function _encode($decoded)
 411      {
 412          // We cannot encode a domain name containing the Punycode prefix
 413          $extract = strlen($this->_punycode_prefix);
 414          $check_pref = $this->_utf8_to_ucs4($this->_punycode_prefix);
 415          $check_deco = array_slice($decoded, 0, $extract);
 416  
 417          if ($check_pref == $check_deco) {
 418              $this->_error('This is already a punycode string');
 419              return false;
 420          }
 421          // We will not try to encode strings consisting of basic code points only
 422          $encodable = false;
 423          foreach ($decoded as $k => $v) {
 424              if ($v > 0x7a) {
 425                  $encodable = true;
 426                  break;
 427              }
 428          }
 429          if (!$encodable) {
 430              $this->_error('The given string does not contain encodable chars');
 431              return false;
 432          }
 433          // Do NAMEPREP
 434          $decoded = $this->_nameprep($decoded);
 435          if (!$decoded || !is_array($decoded)) return false; // NAMEPREP failed
 436          $deco_len  = count($decoded);
 437          if (!$deco_len) return false; // Empty array
 438          $codecount = 0; // How many chars have been consumed
 439          $encoded = '';
 440          // Copy all basic code points to output
 441          for ($i = 0; $i < $deco_len; ++$i) {
 442              $test = $decoded[$i];
 443              // Will match [-0-9a-zA-Z]
 444              if ((0x2F < $test && $test < 0x40) || (0x40 < $test && $test < 0x5B)
 445                      || (0x60 < $test && $test <= 0x7B) || (0x2D == $test)) {
 446                  $encoded .= chr($decoded[$i]);
 447                  $codecount++;
 448              }
 449          }
 450          if ($codecount == $deco_len) return $encoded; // All codepoints were basic ones
 451  
 452          // Start with the prefix; copy it to output
 453          $encoded = $this->_punycode_prefix.$encoded;
 454          // If we have basic code points in output, add an hyphen to the end
 455          if ($codecount) $encoded .= '-';
 456          // Now find and encode all non-basic code points
 457          $is_first = true;
 458          $cur_code = $this->_initial_n;
 459          $bias = $this->_initial_bias;
 460          $delta = 0;
 461          while ($codecount < $deco_len) {
 462              // Find the smallest code point >= the current code point and
 463              // remember the last ouccrence of it in the input
 464              for ($i = 0, $next_code = $this->_max_ucs; $i < $deco_len; $i++) {
 465                  if ($decoded[$i] >= $cur_code && $decoded[$i] <= $next_code) {
 466                      $next_code = $decoded[$i];
 467                  }
 468              }
 469              $delta += ($next_code - $cur_code) * ($codecount + 1);
 470              $cur_code = $next_code;
 471  
 472              // Scan input again and encode all characters whose code point is $cur_code
 473              for ($i = 0; $i < $deco_len; $i++) {
 474                  if ($decoded[$i] < $cur_code) {
 475                      $delta++;
 476                  } elseif ($decoded[$i] == $cur_code) {
 477                      for ($q = $delta, $k = $this->_base; 1; $k += $this->_base) {
 478                          $t = ($k <= $bias) ? $this->_tmin :
 479                                  (($k >= $bias + $this->_tmax) ? $this->_tmax : $k - $bias);
 480                          if ($q < $t) break;
 481                          $encoded .= $this->_encode_digit(intval($t + (($q - $t) % ($this->_base - $t)))); //v0.4.5 Changed from ceil() to intval()
 482                          $q = (int) (($q - $t) / ($this->_base - $t));
 483                      }
 484                      $encoded .= $this->_encode_digit($q);
 485                      $bias = $this->_adapt($delta, $codecount+1, $is_first);
 486                      $codecount++;
 487                      $delta = 0;
 488                      $is_first = false;
 489                  }
 490              }
 491              $delta++;
 492              $cur_code++;
 493          }
 494          return $encoded;
 495      }

 497      /**
 498       * Adapt the bias according to the current code point and position
 499       * @param int $delta
 500       * @param int $npoints
 501       * @param int $is_first
 502       * @return int
 503       */
 504      private function _adapt($delta, $npoints, $is_first)
 505      {
 506          $delta = intval($is_first ? ($delta / $this->_damp) : ($delta / 2));
 507          $delta += intval($delta / $npoints);
 508          for ($k = 0; $delta > (($this->_base - $this->_tmin) * $this->_tmax) / 2; $k += $this->_base) {
 509              $delta = intval($delta / ($this->_base - $this->_tmin));
 510          }
 511          return intval($k + ($this->_base - $this->_tmin + 1) * $delta / ($delta + $this->_skew));
 512      }

 544      /**
 545       * Do Nameprep according to RFC3491 and RFC3454
 546       * @param    array    Unicode Characters
 547       * @return   string   Unicode Characters, Nameprep'd
 548       */
 549      private function _nameprep($input)
 550      {
 551          $output = array();
 552          $error = false;
 553          //
 554          // Mapping
 555          // Walking through the input array, performing the required steps on each of
 556          // the input chars and putting the result into the output array
 557          // While mapping required chars we apply the cannonical ordering
 558          foreach ($input as $v) {
 559              // Map to nothing == skip that code point
 560              if (in_array($v, $this->NP['map_nothing'])) continue;
 561              // Try to find prohibited input
 562              if (in_array($v, $this->NP['prohibit']) || in_array($v, $this->NP['general_prohibited'])) {
 563                  $this->_error('NAMEPREP: Prohibited input U+'.sprintf('%08X', $v));
 564                  return false;
 565              }
 566              foreach ($this->NP['prohibit_ranges'] as $range) {
 567                  if ($range[0] <= $v && $v <= $range[1]) {
 568                      $this->_error('NAMEPREP: Prohibited input U+'.sprintf('%08X', $v));
 569                      return false;
 570                  }
 571              }
 572              // Hangul syllable decomposition
 573              if (0xAC00 <= $v && $v <= 0xD7AF) {
 574                  foreach ($this->_hangul_decompose($v) as $out) $output[] = (int) $out;
 575              // There's a decomposition mapping for that code point
 576              } elseif (isset($this->NP['replacemaps'][$v])) {
 577                  foreach ($this->_apply_cannonical_ordering($this->NP['replacemaps'][$v]) as $out) {
 578                      $output[] = (int) $out;
 579                  }
 580              } else {
 581                  $output[] = (int) $v;
 582              }
 583          }
 584          // Before applying any Combining, try to rearrange any Hangul syllables
 585          $output = $this->_hangul_compose($output);
 586          //
 587          // Combine code points
 588          //
 589          $last_class = 0;
 590          $last_starter = 0;
 591          $out_len = count($output);
 592          for ($i = 0; $i < $out_len; ++$i) {
 593              $class = $this->_get_combining_class($output[$i]);
 594              if ((!$last_class || $last_class > $class) && $class) {
 595                  // Try to match
 596                  $seq_len = $i - $last_starter;
 597                  $out = $this->_combine(array_slice($output, $last_starter, $seq_len));
 598                  // On match: Replace the last starter with the composed character and remove
 599                  // the now redundant non-starter(s)
 600                  if ($out) {
 601                      $output[$last_starter] = $out;
 602                      if (count($out) != $seq_len) {
 603                          for ($j = $i+1; $j < $out_len; ++$j) $output[$j-1] = $output[$j];
 604                          unset($output[$out_len]);
 605                      }
 606                      // Rewind the for loop by one, since there can be more possible compositions
 607                      $i--;
 608                      $out_len--;
 609                      $last_class = ($i == $last_starter) ? 0 : $this->_get_combining_class($output[$i-1]);
 610                      continue;
 611                  }
 612              }
 613              // The current class is 0
 614              if (!$class) $last_starter = $i;
 615              $last_class = $class;
 616          }
 617          return $output;
 618      }

 620      /**
 621       * Decomposes a Hangul syllable
 622       * (see http://www.unicode.org/unicode/reports/tr15/#Hangul
 623       * @param    integer  32bit UCS4 code point
 624       * @return   array    Either Hangul Syllable decomposed or original 32bit value as one value array
 625       */
 626      private function _hangul_decompose($char)
 627      {
 628          $sindex = (int) $char - $this->_sbase;
 629          if ($sindex < 0 || $sindex >= $this->_scount) return array($char);
 630          $result = array();
 631          $result[] = (int) $this->_lbase + $sindex / $this->_ncount;
 632          $result[] = (int) $this->_vbase + ($sindex % $this->_ncount) / $this->_tcount;
 633          $T = intval($this->_tbase + $sindex % $this->_tcount);
 634          if ($T != $this->_tbase) $result[] = $T;
 635          return $result;
 636      }
 637      /**
 638       * Ccomposes a Hangul syllable
 639       * (see http://www.unicode.org/unicode/reports/tr15/#Hangul
 640       * @param    array    Decomposed UCS4 sequence
 641       * @return   array    UCS4 sequence with syllables composed
 642       */
 643      private function _hangul_compose($input)
 644      {
 645          $inp_len = count($input);
 646          if (!$inp_len) return array();
 647          $result = array();
 648          $last = (int) $input[0];
 649          $result[] = $last; // copy first char from input to output
 650  
 651          for ($i = 1; $i < $inp_len; ++$i) {
 652              $char = (int) $input[$i];
 653              $sindex = $last - $this->_sbase;
 654              $lindex = $last - $this->_lbase;
 655              $vindex = $char - $this->_vbase;
 656              $tindex = $char - $this->_tbase;
 657              // Find out, whether two current characters are LV and T
 658              if (0 <= $sindex && $sindex < $this->_scount && ($sindex % $this->_tcount == 0)
 659                      && 0 <= $tindex && $tindex <= $this->_tcount) {
 660                  // create syllable of form LVT
 661                  $last += $tindex;
 662                  $result[(count($result) - 1)] = $last; // reset last
 663                  continue; // discard char
 664              }
 665              // Find out, whether two current characters form L and V
 666              if (0 <= $lindex && $lindex < $this->_lcount && 0 <= $vindex && $vindex < $this->_vcount) {
 667                  // create syllable of form LV
 668                  $last = (int) $this->_sbase + ($lindex * $this->_vcount + $vindex) * $this->_tcount;
 669                  $result[(count($result) - 1)] = $last; // reset last
 670                  continue; // discard char
 671              }
 672              // if neither case was true, just add the character
 673              $last = $char;
 674              $result[] = $char;
 675          }
 676          return $result;
 677      }

 689      /**
 690       * Apllies the cannonical ordering of a decomposed UCS4 sequence
 691       * @param    array      Decomposed UCS4 sequence
 692       * @return   array      Ordered USC4 sequence
 693       */
 694      private function _apply_cannonical_ordering($input)
 695      {
 696          $swap = true;
 697          $size = count($input);
 698          while ($swap) {
 699              $swap = false;
 700              $last = $this->_get_combining_class(intval($input[0]));
 701              for ($i = 0; $i < $size-1; ++$i) {
 702                  $next = $this->_get_combining_class(intval($input[$i+1]));
 703                  if ($next != 0 && $last > $next) {
 704                      // Move item leftward until it fits
 705                      for ($j = $i + 1; $j > 0; --$j) {
 706                          if ($this->_get_combining_class(intval($input[$j-1])) <= $next) break;
 707                          $t = intval($input[$j]);
 708                          $input[$j] = intval($input[$j-1]);
 709                          $input[$j-1] = $t;
 710                          $swap = true;
 711                      }
 712                      // Reentering the loop looking at the old character again
 713                      $next = $last;
 714                  }
 715                  $last = $next;
 716              }
 717          }
 718          return $input;
 719      }

 721      /**
 722       * Do composition of a sequence of starter and non-starter
 723       * @param    array      UCS4 Decomposed sequence
 724       * @return   array      Ordered USC4 sequence
 725       */
 726      private function _combine($input)
 727      {
 728          $inp_len = count($input);
 729          foreach ($this->NP['replacemaps'] as $np_src => $np_target) {
 730              if ($np_target[0] != $input[0]) continue;
 731              if (count($np_target) != $inp_len) continue;
 732              $hit = false;
 733              foreach ($input as $k2 => $v2) {
 734                  if ($v2 == $np_target[$k2]) {
 735                      $hit = true;
 736                  } else {
 737                      $hit = false;
 738                      break;
 739                  }
 740              }
 741              if ($hit) return $np_src;
 742          }
 743          return false;
 744      }

 746      /**
 747       * This converts an UTF-8 encoded string to its UCS-4 representation
 748       * By talking about UCS-4 "strings" we mean arrays of 32bit integers representing
 749       * each of the "chars". This is due to PHP not being able to handle strings with
 750       * bit depth different from 8. This apllies to the reverse method _ucs4_to_utf8(), too.
 751       * The following UTF-8 encodings are supported:
 752       * bytes bits  representation
 753       * 1        7  0xxxxxxx
 754       * 2       11  110xxxxx 10xxxxxx
 755       * 3       16  1110xxxx 10xxxxxx 10xxxxxx
 756       * 4       21  11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
 757       * 5       26  111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
 758       * 6       31  1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
 759       * Each x represents a bit that can be used to store character data.
 760       * The five and six byte sequences are part of Annex D of ISO/IEC 10646-1:2000
 761       * @param string $input
 762       * @return string
 763       */
 764      private function _utf8_to_ucs4($input)
 765      {
 766          $output = array();
 767          $out_len = 0;
 768          // Patch by Daniel Hahler; work around prolbem with mbstring.func_overload
 769          if (function_exists('mb_strlen')) {
 770              $inp_len = mb_strlen($input, '8bit');
 771          } else {
 772              $inp_len = strlen($input);
 773          }
 774          $mode = 'next';
 775          $test = 'none';
 776          for ($k = 0; $k < $inp_len; ++$k) {
 777              $v = ord($input{$k}); // Extract byte from input string
 778              if ($v < 128) { // We found an ASCII char - put into stirng as is
 779                  $output[$out_len] = $v;
 780                  ++$out_len;
 781                  if ('add' == $mode) {
 782                      $this->_error('Conversion from UTF-8 to UCS-4 failed: malformed input at byte '.$k);
 783                      return false;
 784                  }
 785                  continue;
 786              }
 787              if ('next' == $mode) { // Try to find the next start byte; determine the width of the Unicode char
 788                  $start_byte = $v;
 789                  $mode = 'add';
 790                  $test = 'range';
 791                  if ($v >> 5 == 6) { // &110xxxxx 10xxxxx
 792                      $next_byte = 0; // Tells, how many times subsequent bitmasks must rotate 6bits to the left
 793                      $v = ($v - 192) << 6;
 794                  } elseif ($v >> 4 == 14) { // &1110xxxx 10xxxxxx 10xxxxxx
 795                      $next_byte = 1;
 796                      $v = ($v - 224) << 12;
 797                  } elseif ($v >> 3 == 30) { // &11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
 798                      $next_byte = 2;
 799                      $v = ($v - 240) << 18;
 800                  } elseif ($v >> 2 == 62) { // &111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
 801                      $next_byte = 3;
 802                      $v = ($v - 248) << 24;
 803                  } elseif ($v >> 1 == 126) { // &1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
 804                      $next_byte = 4;
 805                      $v = ($v - 252) << 30;
 806                  } else {
 807                      $this->_error('This might be UTF-8, but I don\'t understand it at byte '.$k);
 808                      return false;
 809                  }
 810                  if ('add' == $mode) {
 811                      $output[$out_len] = (int) $v;
 812                      ++$out_len;
 813                      continue;
 814                  }
 815              }
 816              if ('add' == $mode) {
 817                  if (!$this->_allow_overlong && $test == 'range') {
 818                      $test = 'none';
 819                      if (($v < 0xA0 && $start_byte == 0xE0) || ($v < 0x90 && $start_byte == 0xF0) || ($v > 0x8F && $start_byte == 0xF4)) {
 820                          $this->_error('Bogus UTF-8 character detected (out of legal range) at byte '.$k);
 821                          return false;
 822                      }
 823                  }
 824                  if ($v >> 6 == 2) { // Bit mask must be 10xxxxxx
 825                      $v = ($v - 128) << ($next_byte * 6);
 826                      $output[($out_len - 1)] += $v;
 827                      --$next_byte;
 828                  } else {
 829                      $this->_error('Conversion from UTF-8 to UCS-4 failed: malformed input at byte '.$k);
 830                      return false;
 831                  }
 832                  if ($next_byte < 0) {
 833                      $mode = 'next';
 834                  }
 835              }
 836          } // for
 837          return $output;
 838      }

 840      /**
 841       * Convert UCS-4 string into UTF-8 string
 842       * See _utf8_to_ucs4() for details
 843       * @param string  $input
 844       * @return string
 845       */
 846      private function _ucs4_to_utf8($input)
 847      {
 848          $output = '';
 849          foreach ($input as $k => $v) {
 850              if ($v < 128) { // 7bit are transferred literally
 851                  $output .= chr($v);
 852              } elseif ($v < (1 << 11)) { // 2 bytes
 853                  $output .= chr(192+($v >> 6)).chr(128+($v & 63));
 854              } elseif ($v < (1 << 16)) { // 3 bytes
 855                  $output .= chr(224+($v >> 12)).chr(128+(($v >> 6) & 63)).chr(128+($v & 63));
 856              } elseif ($v < (1 << 21)) { // 4 bytes
 857                  $output .= chr(240+($v >> 18)).chr(128+(($v >> 12) & 63)).chr(128+(($v >> 6) & 63)).chr(128+($v & 63));
 858              } elseif (self::$safe_mode) {
 859                  $output .= self::$safe_char;
 860              } else {
 861                  $this->_error('Conversion from UCS-4 to UTF-8 failed: malformed input at byte '.$k);
 862                  return false;
 863              }
 864          }
 865          return $output;
 866      }

 868      /**
 869        * Convert UCS-4 array into UCS-4 string
 870        *
 871        * @param array $input
 872        * @return string
 873        */
 874      private function _ucs4_to_ucs4_string($input)
 875      {
 876          $output = '';
 877          // Take array values and split output to 4 bytes per value
 878          // The bit mask is 255, which reads &11111111
 879          foreach ($input as $v) {
 880              $output .= chr(($v >> 24) & 255).chr(($v >> 16) & 255).chr(($v >> 8) & 255).chr($v & 255);
 881          }
 882          return $output;
 883      }

 885      /**
 886        * Convert UCS-4 strin into UCS-4 garray
 887        *
 888        * @param  string $input
 889        * @return array
 890        */
 891      private function _ucs4_string_to_ucs4($input)
 892      {
 893          $output = array();
 894          $inp_len = strlen($input);
 895          // Input length must be dividable by 4
 896          if ($inp_len % 4) {
 897              $this->_error('Input UCS4 string is broken');
 898              return false;
 899          }
 900          // Empty input - return empty output
 901          if (!$inp_len) return $output;
 902          for ($i = 0, $out_len = -1; $i < $inp_len; ++$i) {
 903              // Increment output position every 4 input bytes
 904              if (!($i % 4)) {
 905                  $out_len++;
 906                  $output[$out_len] = 0;
 907              }
 908              $output[$out_len] += ord($input{$i}) << (8 * (3 - ($i % 4) ) );
 909          }
 910          return $output;
 911      }
