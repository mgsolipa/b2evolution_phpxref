  23   * @since 1.0

  32  {
  33      $str = utf8_clean( $str, true ); //True == $remove_bom
  34  
  35      $str = utf8_strtolower( $str );
  36  
  37      if( $trns && iconv_loaded() )

  39          $str = iconv( 'UTF-8', 'US-ASCII//TRANSLIT//IGNORE', $str );

  70   *
  71   * Checks whether the passed string contains only byte sequances that
  72   * appear valid UTF-8 characters.
  73   * @since 1.0
  74   *
  75   * @param    string $str The string to be checked
  76   * @return   bool True if the check succeeds, False Otherwise
  77   */
  78  function is_utf8( $str )
  79  {
  80      if( pcre_utf8_support() )
  81      {
  82          return ( bool ) preg_match( '//u', $str );
  83      }
  84  
  85      if( mbstring_loaded() )
  86      {
  87          return mb_check_encoding( $str, 'UTF-8' );
  88      }
  89  
  90      //Fallback
  91  
  92      $len = strlen( $str );
  93  
  94      for( $i = 0; $i < $len; $i ++ )
  95      {
  96          if( ( $str[ $i ] & "\x80" ) === "\x00" )
  97          {
  98              continue;
  99          }
 100          else if( ( ( $str[ $i ] & "\xE0" ) === "\xC0" ) && ( isset( $str[ $i + 1 ] ) ) )
 101          {
 102              if( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" )
 103              {
 104                  $i ++;
 105                  continue;
 106              }
 107  
 108              return false;
 109          }
 110          else if( ( ( $str[ $i ] & "\xF0" ) === "\xE0" ) && ( isset( $str[ $i + 2 ] ) ) )
 111          {
 112              if( ( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 2 ] & "\xC0" ) === "\x80" ) )
 113              {
 114                  $i = $i + 2;
 115                  continue;
 116              }
 117  
 118              return false;
 119          }
 120          else if( ( ( $str[ $i ] & "\xF8" ) === "\xF0" ) && ( isset( $str[ $i + 3 ] ) ) )
 121          {
 122              if( ( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 2 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 3 ] & "\xC0" ) === "\x80" ) )
 123              {
 124                  $i = $i + 3;
 125                  continue;
 126              }
 127  
 128              return false;
 129          }
 130          else
 131          {
 132              return false;
 133          }
 134      }

 139  /**
 140   * utf8_ord( )
 141   *
 142   * Calculates Unicode Code Point of the given UTF-8 encoded character
 143   * @since 1.0
 144   *
 145   * @param    string $chr The character of which to calculate Code Point
 146   * @return   int Unicode Code Point of the given character
 147   * 0 on invalid UTF-8 byte sequence
 148   */
 149  function utf8_ord( $chr )
 150  {
 151      $chr = utf8_split( $chr );
 152  
 153      $chr = $chr[ 0 ];
 154  
 155      switch( strlen( $chr ) )
 156      {
 157          case 1 :
 158              return ord( $chr );
 159  
 160          case 2 :
 161              return ( ( ord( $chr[ 0 ] ) & 0x1F ) << 6 ) | ( ord( $chr[ 1 ] ) & 0x3F );
 162  
 163          case 3 :
 164              return ( ( ord( $chr[ 0 ] ) & 0x0F ) << 12 ) | ( ( ord( $chr[ 1 ] ) & 0x3F ) << 6 ) | ( ord( $chr[ 2 ] ) & 0x3F );
 165  
 166          case 4 :
 167              return ( ( ord( $chr[ 0 ] ) & 0x07 ) << 18 ) | ( ( ord( $chr[ 1 ] ) & 0x3F ) << 12 ) | ( ( ord( $chr[ 2 ] ) & 0x3F ) << 6 ) | ( ord( $chr[ 3 ] ) & 0x3F );
 168      }
 169  
 170      return 0;
 171  }

 214  /**
 215   * utf8_chr( )
 216   *
 217   * Generates a UTF-8 encoded character from the given Code Point
 218   * @since 1.0
 219   *
 220   * @param    int $code_point The code point for which to generate a character
 221   * @return   string Milti-Byte character
 222   * returns empty string on failure to encode
 223   */
 224  function utf8_chr( $code_point )
 225  {
 226      if( ( $i = ( int ) $code_point ) !== $code_point )
 227      {
 228          //$code_point is a string, lets extract int code point from it
 229          if( ! ( $i = ( int ) utf8_hex_to_int( $code_point ) ) )
 230          {
 231              return '';
 232          }
 233      }
 234  
 235      //json not working properly for larger code points
 236      //See the original post.
 237  
 238      //if( extension_loaded( 'json' ) )
 239      //{
 240      //    $hex    = dechex( $i );
 241      //
 242      //    return json_decode('"\u'. ( strlen( $hex ) < 4 ? substr( '000' . $hex , -4 ) : $hex ) .'"');
 243      //}
 244      //else
 245  
 246      if( mbstring_loaded() /*extension_loaded( 'mbstring' )*/ )
 247          {
 248          return mb_convert_encoding( "&#$i;", 'UTF-8', 'HTML-ENTITIES' );
 249      }
 250      else if( version_compare( phpversion(), '5.0.0' ) === 1 )
 251      {
 252          //html_entity_decode did not support Multi-Byte before PHP 5.0.0
 253          return html_entity_decode( "&#{$i};", ENT_QUOTES, 'UTF-8' );
 254      }
 255  
 256      //Fallback
 257  
 258      $bits = ( int ) ( log( $i, 2 ) + 1 );
 259  
 260      if( $bits <= 7 ) //Single Byte
 261      {
 262          return chr( $i );
 263      }
 264      else if( $bits <= 11 ) //Two Bytes
 265      {
 266          return chr( ( ( $i >> 6 ) & 0x1F ) | 0xC0 ) . chr( ( $i & 0x3F ) | 0x80 );
 267      }
 268      else if( $bits <= 16 ) //Three Bytes
 269      {
 270          return chr( ( ( $i >> 12 ) & 0x0F ) | 0xE0 ) . chr( ( ( $i >> 6 ) & 0x3F ) | 0x80 ) . chr( ( $i & 0x3F ) | 0x80 );
 271      }
 272      else if( $bits <= 21 ) //Four Bytes
 273      {
 274          return chr( ( ( $i >> 18 ) & 0x07 ) | 0xF0 ) . chr( ( ( $i >> 12 ) & 0x3F ) | 0x80 ) . chr( ( ( $i >> 6 ) & 0x3F ) | 0x80 ) . chr( ( $i & 0x3F ) | 0x80 );
 275      }
 276      else
 277      {
 278          return ''; //Cannot be encoded as Valid UTF-8
 279      }
 280  }

 333   * @since 1.0

 336   * @param    int $split_length Max character length of each array element
 337   * @return   array An array containing chunks of the string

 339  function utf8_split( $str, $split_length = 1 )
 340  {
 341      $str = ( string ) $str;
 342  
 343      $ret = array();
 344  
 345      if( pcre_utf8_support() )
 346      {
 347          $str = utf8_clean( $str );
 348  
 349          //    http://stackoverflow.com/a/8780076/369005
 350          $ret = preg_split( '/(?<!^)(?!$)/u', $str );
 351  
 352          // \X is buggy in many recent versions of PHP
 353          //preg_match_all( '/\X/u' , $str , $ret );
 354          //$ret    = $ret[0];
 355      }
 356      else
 357      {
 358          //Fallback
 359  
 360          $len = strlen( $str );
 361  
 362          for( $i = 0; $i < $len; $i ++ )
 363          {
 364              if( ( $str[ $i ] & "\x80" ) === "\x00" )
 365              {
 366                  $ret[] = $str[ $i ];
 367              }
 368              else if( ( ( $str[ $i ] & "\xE0" ) === "\xC0" ) && ( isset( $str[ $i + 1 ] ) ) )
 369              {
 370                  if( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" )
 371                  {
 372                      $ret[] = $str[ $i ] . $str[ $i + 1 ];
 373  
 374                      $i ++;
 375                  }
 376              }
 377              else if( ( ( $str[ $i ] & "\xF0" ) === "\xE0" ) && ( isset( $str[ $i + 2 ] ) ) )
 378              {
 379                  if( ( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 2 ] & "\xC0" ) === "\x80" ) )
 380                  {
 381                      $ret[] = $str[ $i ] . $str[ $i + 1 ] . $str[ $i + 2 ];
 382  
 383                      $i = $i + 2;
 384                  }
 385              }
 386              else if( ( ( $str[ $i ] & "\xF8" ) === "\xF0" ) && ( isset( $str[ $i + 3 ] ) ) )
 387              {
 388                  if( ( ( $str[ $i + 1 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 2 ] & "\xC0" ) === "\x80" ) && ( ( $str[ $i + 3 ] & "\xC0" ) === "\x80" ) )
 389                  {
 390                      $ret[] = $str[ $i ] . $str[ $i + 1 ] . $str[ $i + 2 ] . $str[ $i + 3 ];
 391  
 392                      $i = $i + 3;
 393                  }
 394              }
 395          }
 396      }
 397  
 398      if( $split_length > 1 )
 399      {
 400          $ret = array_chunk( $ret, $split_length );
 401  
 402          $ret = array_map( 'implode', $ret );
 403      }
 404  
 405      if( $ret[ 0 ] === '' )
 406      {
 407          return array();
 408      }
 409  
 410      return $ret;
 411  }

 507   *
 508   * UTF-8 aware substr
 509   * @since 1.0

 517  {

 536          }

 703   * utf8_strpos( )
 704   *
 705   * Finds the number of Characters to the left of first occurance of the needle
 706   * @since 1.0
 707   *
 708   * For detailed documentation see php.net/strpos
 709   *
 710   * strpos works with bytes, while utf8_strpos works with characters
 711   * and are identical in all other aspects.
 712   */
 713  function utf8_strpos( $haystack, $needle, $offset = 0 )
 714  {
 715      //iconv and mbstring do not support integer $needle
 716      if( ( ( int ) $needle ) === $needle && ( $needle >= 0 ) )
 717      {
 718          $needle = utf8_chr( $needle );
 719      }
 720  
 721      $needle = utf8_clean( ( string ) $needle );
 722      $offset = ( int ) $offset;
 723  
 724      //mb_strpos returns wrong position if invalid characters are found in $haystack before $needle
 725      //iconv_strpos is not tolerant to invalid characters
 726  
 727      $haystack = utf8_clean( $haystack );
 728  
 729      if( mbstring_loaded() )
 730      {
 731          return mb_strpos( $haystack, $needle, $offset, 'UTF-8' );
 732      }
 733  
 734      if( iconv_loaded() )
 735      {
 736          return iconv_strpos( $haystack, $needle, $offset, 'UTF-8' );
 737      }
 738  
 739      if( $offset > 0 )
 740      {
 741          $haystack = utf8_substr( $haystack, $offset );
 742      }
 743  
 744      //Negative Offset not supported in PHP strpos(), ignoring
 745      if( ( $pos = strpos( $haystack, $needle ) ) !== false )
 746      {
 747          $left = substr( $haystack, 0, $pos );
 748          return ( $offset > 0 ? $offset : 0 ) + utf8_strlen( $left );
 749      }
 750  
 751      return false;

1159   *
1160   * Pad a UTF-8 string to given length with another string
1161   * @since 1.3

1163   * @param    string $input The input string

1166   * @param    int $pad_type can be STR_PAD_RIGHT, STR_PAD_LEFT or STR_PAD_BOTH
1167   * @return   string Returns the padded string
1168   */
1169  function utf8_str_pad( $input, $pad_length, $pad_string = ' ', $pad_type = STR_PAD_RIGHT )
1170  {
1171      $input_length = utf8_strlen( $input );
1172  
1173      if( is_int( $pad_length ) && ( $pad_length > 0 ) && ( $pad_length >= $input_length ) )

1177

1190                  $post = utf8_substr( $post, 0, ( int ) ceil( $diff / 2 ) );
