 228  function CanvasDonutChart( $chart )
 229  {
 230  ?>
 231  <div id="canvasdonutchart" style="height:<?php echo $chart['height']; ?>px;width:<?php echo $chart['width']; ?>px;"></div>
 232  <script type="text/javascript">
 233  jQuery( window ).load( function()
 234  {
 235  <?php
 236      $data_vars = array();
 237      foreach( $chart[ 'data' ] as $d => $data_level )
 238      { // Init the chart data
 239          $data = array();
 240          foreach( $data_level as $label => $value )
 241          {
 242              $data[] = '[\''.$label.'\','.$value.']';
 243          }
 244          echo '    var data'.$d.' = ['.implode( ',', $data ).'];'."\n";
 245          $data_vars[] = 'data'.$d;
 246      }
 247      ?>
 248  
 249      jQuery.jqplot( 'canvasdonutchart', [<?php echo implode( ',', $data_vars ); ?>], {
 250          series: [
 251          <?php
 252          foreach( $chart[ 'series_color' ] as $colors )
 253          { // Init the colors for each level
 254          ?>
 255              { seriesColors: [ '#<?php echo implode( '\', \'#', $colors ); ?>' ] },
 256          <?php } ?>
 257          ],
 258          seriesDefaults: {
 259              renderer:$.jqplot.DonutRenderer,
 260              rendererOptions: {
 261                  sliceMargin: 1,
 262                  startAngle: -90,
 263                  shadow: false,
 264                  innerDiameter: 100,
 265              },
 266              highlighter: {
 267                  tooltipAxes: 'y',
 268                  useAxesFormatters: false
 269              }
 270          },
 271          grid: {
 272              shadow: false,
 273              borderWidth: 1,
 274              borderColor: '#e5e5e5',
 275              background: '#fff'
 276          },
 277          legend: {
 278              renderer: $.jqplot.EnhancedLegendRenderer,
 279              rendererOptions: {
 280                  numberRows: <?php echo ( isset( $chart['legend_numrows'] ) ? $chart['legend_numrows'] : '1' ); ?>
 281              },
 282              labels: ['<?php echo implode( "','", $chart['legends'] ); ?>'],
 283              show: true,
 284              predraw: true,
 285              location: 'e',
 286              placement: 'outside',
 287              yoffset: 10
 288          },
 289          highlighter: { show: true },
 290      } );
 291  
 292      // Highlight legend
 293      jQuery( '#canvasdonutchart' ).bind( 'jqplotDataHighlight', function( ev, seriesIndex, pointIndex, data )
 294      {
 295          //console.log( ev, seriesIndex, pointIndex, data );
 296          jQuery( '#canvasdonutchart td.jqplot-table-legend' ).removeClass( 'legend-text-highlighted' );
 297          if( data[0].length == 1 )
 298          {
 299              var x = data[0] == 'F' ? 0 : ( data[0] == 'M' ? 1 : 2 );
 300              var obj = jQuery( '#canvasdonutchart tr:eq(' + ( x * 3 ) + '), #canvasdonutchart tr:eq(' + ( x * 3 + 1) + '), #canvasdonutchart tr:eq(' + ( x * 3 + 2 ) + ')' );
 301              obj.children( 'td' ).addClass( 'legend-text-highlighted' );
 302          }
 303          else
 304          {
 305              var x = 0;
 306              var y = 0;
 307              var td_selector = 'td';
 308              switch( data[0][0] )
 309              {
 310                  case 'M': y = 1; break;
 311                  case 'G': y = 2; break;
 312              }
 313              switch( data[0][1] )
 314              {
 315                  case 'i': x = 1; break;
 316                  case 'c': x = 2; break;
 317              }
 318  
 319              if( data[0].length == 3 )
 320              {
 321                  var z = 0;
 322                  if( data[0][2] == 'n' )
 323                  {
 324                      z = 2;
 325                  }
 326                  td_selector = 'td:eq(' + z + '), td:eq(' + ( z + 1 ) + ')';
 327              }
 328  
 329              var obj = jQuery( '#canvasdonutchart tr:eq(' + ( x + ( y * 3 ) ) + ')' );
 330              obj.children( td_selector ).addClass( 'legend-text-highlighted' );
 331          }
 332      } );
 333      jQuery( '#canvasdonutchart' ).bind( 'jqplotDataUnhighlight', function( ev, seriesIndex, pointIndex, data )
 334      {
 335          jQuery( '#canvasdonutchart td.jqplot-table-legend' ).removeClass( 'legend-text-highlighted' );
 336      } );
 337  } );
 338  </script>
 339  <?php
 340  }
