  75      public function send(Swift_Mime_Message $message, &$failedRecipients = null)
  76      {
  77          $failedRecipients = (array) $failedRecipients;
  78  
  79          if (!$this->_transport->isStarted()) {
  80              $this->_transport->start();
  81          }
  82  
  83          $sent = 0;
  84  
  85          try {
  86              $sent = $this->_transport->send($message, $failedRecipients);
  87          } catch (Swift_RfcComplianceException $e) {
  88              foreach ($message->getTo() as $address => $name) {
  89                  $failedRecipients[] = $address;
  90              }
  91          }
  92  
  93          return $sent;
  94      }
