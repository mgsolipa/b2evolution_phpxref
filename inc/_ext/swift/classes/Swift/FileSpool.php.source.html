  85       * Queues a message.
  86       *
  87       * @param Swift_Mime_Message $message The message to store
  88       *
  89       * @return boolean
  90       *
  91       * @throws Swift_IoException

  93      public function queueMessage(Swift_Mime_Message $message)
  94      {
  95          $ser = serialize($message);
  96          $fileName = $this->_path . '/' . $this->getRandomString(10);
  97          for ($i = 0; $i < $this->_retryLimit; ++$i) {
  98              /* We try an exclusive creation of the file. This is an atomic operation, it avoid locking mechanism */
  99              $fp = @fopen($fileName . '.message', 'x');
 100              if (false !== $fp) {
 101                  if (false === fwrite($fp, $ser)) {
 102                      return false;
 103                  }
 104  
 105                  return fclose($fp);
 106              } else {
 107                  /* The file already exists, we try a longer fileName */
 108                  $fileName .= $this->getRandomString(1);
 109              }
 110          }
 111  
 112          throw new Swift_IoException('Unable to create a file for enqueuing Message');
 113      }

 115      /**
 116       * Execute a recovery if for any reason a process is sending for too long.
 117       *
 118       * @param integer $timeout in second Defaults is for very slow smtp responses
 119       */
 120      public function recover($timeout = 900)
 121      {
 122          foreach (new DirectoryIterator($this->_path) as $file) {
 123              $file = $file->getRealPath();
 124  
 125              if (substr($file, - 16) == '.message.sending') {
 126                  $lockedtime = filectime($file);
 127                  if ((time() - $lockedtime) > $timeout) {
 128                      rename($file, substr($file, 0, - 8));
 129                  }
 130              }
 131          }
 132      }

 134      /**
 135       * Sends messages using the given transport instance.
 136       *
 137       * @param Swift_Transport $transport        A transport instance
 138       * @param string[]        $failedRecipients An array of failures by-reference
 139       *
 140       * @return integer The number of sent e-mail's
 141       */
 142      public function flushQueue(Swift_Transport $transport, &$failedRecipients = null)
 143      {
 144          $directoryIterator = new DirectoryIterator($this->_path);
 145  
 146          /* Start the transport only if there are queued files to send */
 147          if (!$transport->isStarted()) {
 148              foreach ($directoryIterator as $file) {
 149                  if (substr($file->getRealPath(), -8) == '.message') {
 150                      $transport->start();
 151                      break;
 152                  }
 153              }
 154          }
 155  
 156          $failedRecipients = (array) $failedRecipients;
 157          $count = 0;
 158          $time = time();
 159          foreach ($directoryIterator as $file) {
 160              $file = $file->getRealPath();
 161  
 162              if (substr($file, -8) != '.message') {
 163                  continue;
 164              }
 165  
 166              /* We try a rename, it's an atomic operation, and avoid locking the file */
 167              if (rename($file, $file.'.sending')) {
 168                  $message = unserialize(file_get_contents($file.'.sending'));
 169  
 170                  $count += $transport->send($message, $failedRecipients);
 171  
 172                  unlink($file.'.sending');
 173              } else {
 174                  /* This message has just been catched by another process */
 175                  continue;
 176              }
 177  
 178              if ($this->getMessageLimit() && $count >= $this->getMessageLimit()) {
 179                  break;
 180              }
 181  
 182              if ($this->getTimeLimit() && (time() - $time) >= $this->getTimeLimit()) {
 183                  break;
 184              }
 185          }
 186  
 187          return $count;
 188      }

 190      /**
 191       * Returns a random string needed to generate a fileName for the queue.
 192       *
 193       * @param integer $count
 194       *
 195       * @return string
 196       */
 197      protected function getRandomString($count)
 198      {
 199          // This string MUST stay FS safe, avoid special chars
 200          $base = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_-.";
 201          $ret = '';
 202          $strlen = strlen($base);
 203          for ($i = 0; $i < $count; ++$i) {
 204              $ret .= $base[((int) rand(0, $strlen - 1))];
 205          }
 206  
 207          return $ret;
 208      }
