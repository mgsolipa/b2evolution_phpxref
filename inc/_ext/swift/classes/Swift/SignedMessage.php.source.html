  22       * @var Swift_Signers_HeaderSigner[]

  26      /**

  29      private $bodySigners = array();

  32       * @var array

  51      /**
  52       * Attach a new signature handler to the message.

  55       * @return Swift_SignedMessage

  57      public function attachSigner(Swift_Signer $signer)
  58      {
  59          if ($signer instanceof Swift_Signers_HeaderSigner) {
  60              $this->headerSigners[] = $signer;
  61          }
  62          elseif ($signer instanceof Swift_Signers_BodySigner) {
  63              $this->bodySigners[] = $signer;
  64          }
  65  
  66          return $this;
  67      }

  74      public function toString()

  81  
  82          return $string;
  83      }
  84  
  85      /**
  86       * Write this message to a {@link Swift_InputByteStream}.
  87       *
  88       * @param Swift_InputByteStream $is
  89       */
  90      public function toByteStream(Swift_InputByteStream $is)
  91      {
  92          $this->saveMessage();
  93          $this->doSign();
  94  
  95          parent::toByteStream($is);
  96          $this->restoreMessage();
  97      }

  99      protected function doSign()
 100      {
 101          foreach ($this->bodySigners as $signer) {
 102              $altered = $signer->getAlteredHeaders();
 103              $this->saveHeaders($altered);
 104              $signer->signMessage($this);
 105          }
 106  
 107          foreach ($this->headerSigners as $signer) {
 108              $altered = $signer->getAlteredHeaders();
 109              $this->saveHeaders($altered);
 110              $signer->reset();
 111  
 112              $signer->setHeaders($this->getHeaders());
 113  
 114              $signer->startBody();
 115              $this->_bodyToByteStream($signer);
 116              $signer->endBody();
 117  
 118              $signer->addSignature($this->getHeaders());
 119          }
 120      }
 121  
 122      protected function saveMessage()
 123      {
 124          $this->savedMessage = array('headers'=> array());
 125          $this->savedMessage['body'] = $this->getBody();
 126          $this->savedMessage['children'] = $this->getChildren();
 127          if (count($this->savedMessage['children']) > 0 && $this->getBody() != '') {
 128              $this->setChildren(array_merge(array($this->_becomeMimePart()), $this->savedMessage['children']));
 129              $this->setBody('');
 130          }
 131      }
 132  
 133      protected function saveHeaders(array $altered)
 134      {
 135          foreach ($altered as $head) {
 136              $lc = strtolower($head);
 137  
 138              if (!isset($this->savedMessage['headers'][$lc])) {
 139                  $this->savedMessage['headers'][$lc] = $this->getHeaders()->getAll($head);
 140              }
 141          }
 142      }
 143  
 144      protected function restoreHeaders()
 145      {
 146          foreach ($this->savedMessage['headers'] as $name => $savedValue) {
 147              $headers = $this->getHeaders()->getAll($name);
 148  
 149              foreach ($headers as $key => $value) {
 150                  if (!isset($savedValue[$key])) {
 151                      $this->getHeaders()->remove($name, $key);
 152                  }
 153              }
 154          }
 155      }
 156  
 157      protected function restoreMessage()
 158      {
 159          $this->setBody($this->savedMessage['body']);
 160          $this->setChildren($this->savedMessage['children']);
 161  
 162          $this->restoreHeaders();
 163          $this->savedMessage = array();
 164      }
 165  }
