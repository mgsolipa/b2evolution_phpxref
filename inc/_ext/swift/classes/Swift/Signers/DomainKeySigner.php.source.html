 350      public function setHeaders(Swift_Mime_HeaderSet $headers)
 351      {
 352          $this->_startHash();
 353          $this->_canonData = '';
 354          // Loop through Headers
 355          $listHeaders = $headers->listAll();
 356          foreach ($listHeaders as $hName) {
 357              // Check if we need to ignore Header
 358              if (! isset($this->_ignoredHeaders[strtolower($hName)])) {
 359                  if ($headers->has($hName)) {
 360                      $tmp = $headers->getAll($hName);
 361                      foreach ($tmp as $header) {
 362                          if ($header->getFieldBody() != '') {
 363                              $this->_addHeader($header->toString());
 364                              $this->_signedHeaders[] = $header->getFieldName();
 365                          }
 366                      }
 367                  }
 368              }
 369          }
 370          $this->_endOfHeaders();
 371  
 372          return $this;
 373      }

 381      public function addSignature(Swift_Mime_HeaderSet $headers)
 382      {
 383          // Prepare the DomainKey-Signature Header
 384          $params = array('a' => $this->_hashAlgorithm, 'b' => chunk_split(base64_encode($this->_getEncryptedHash()), 73, " "), 'c' => $this->_canon, 'd' => $this->_domainName, 'h' => implode(': ', $this->_signedHeaders), 'q' => 'dns', 's' => $this->_selector);
 385          $string = '';
 386          foreach ($params as $k => $v) {
 387              $string .= $k . '=' . $v . '; ';

 389          $string = trim($string);
 390          $headers->addTextHeader('DomainKey-Signature', $string);

 397      protected function _addHeader($header)
 398      {
 399          switch ($this->_canon) {
 400              case 'nofws' :
 401                  // Prepare Header and cascade
 402                  $exploded = explode(':', $header, 2);
 403                  $name = strtolower(trim($exploded[0]));
 404                  $value = str_replace("\r\n", "", $exploded[1]);
 405                  $value = preg_replace("/[ \t][ \t]+/", " ", $value);
 406                  $header = $name . ":" . trim($value) . "\r\n";
 407              case 'simple' :
 408                  // Nothing to do
 409          }
 410          $this->_addToHash($header);
 411      }

 418      protected function _canonicalizeBody($string)
 419      {
 420          $len = strlen($string);
 421          $canon = '';
 422          $nofws = ($this->_canon == "nofws");
 423          for ($i = 0; $i < $len; ++$i) {
 424              if ($this->_bodyCanonIgnoreStart > 0) {
 425                  --$this->_bodyCanonIgnoreStart;
 426                  continue;
 427              }
 428              switch ($string[$i]) {
 429                  case "\r" :
 430                      $this->_bodyCanonLastChar = "\r";
 431                      break;
 432                  case "\n" :
 433                      if ($this->_bodyCanonLastChar == "\r") {
 434                          if ($nofws) {
 435                              $this->_bodyCanonSpace = false;
 436                          }
 437                          if ($this->_bodyCanonLine == '') {
 438                              ++$this->_bodyCanonEmptyCounter;
 439                          } else {
 440                              $this->_bodyCanonLine = '';
 441                              $canon .= "\r\n";
 442                          }
 443                      } else {
 444                          // Wooops Error
 445                          throw new Swift_SwiftException('Invalid new line sequence in mail found \n without preceding \r');
 446                      }
 447                      break;
 448                  case " " :
 449                  case "\t" :
 450                  case "\x09": //HTAB
 451                      if ($nofws) {
 452                          $this->_bodyCanonSpace = true;
 453                          break;
 454                      }
 455                  default :
 456                      if ($this->_bodyCanonEmptyCounter > 0) {
 457                          $canon .= str_repeat("\r\n", $this->_bodyCanonEmptyCounter);
 458                          $this->_bodyCanonEmptyCounter = 0;
 459                      }
 460                      $this->_bodyCanonLine .= $string[$i];
 461                      $canon .= $string[$i];
 462              }
 463          }
 464          $this->_addToHash($canon);
 465      }

 493      private function _getEncryptedHash()
 494      {
 495          $signature = '';
 496          $pkeyId=openssl_get_privatekey($this->_privateKey);
 497          if (!$pkeyId) {
 498              throw new Swift_SwiftException('Unable to load DomainKey Private Key ['.openssl_error_string().']');
 499          }
 500          if (openssl_sign($this->_canonData, $signature, $pkeyId, 'sha1')) {
 501              return $signature;
 502          }
 503          throw new Swift_SwiftException('Unable to sign DomainKey Hash  ['.openssl_error_string().']');
 504      }
