 492      /**
 493       * Add the signature to the given Headers
 494       *
 495       * @param Swift_Mime_HeaderSet $headers
 496       * @return Swift_Signers_DKIMSigner
 497       */
 498      public function addSignature(Swift_Mime_HeaderSet $headers)
 499      {
 500          // Prepare the DKIM-Signature
 501          $params = array('v' => '1', 'a' => $this->_hashAlgorithm, 'bh' => base64_encode($this->_bodyHash), 'd' => $this->_domainName, 'h' => implode(': ', $this->_signedHeaders), 'i' => $this->_signerIdentity, 's' => $this->_selector);
 502          if ($this->_bodyCanon != 'simple') {
 503              $params['c'] = $this->_headerCanon . '/' . $this->_bodyCanon;
 504          } elseif ($this->_headerCanon != 'simple') {
 505              $params['c'] = $this->_headerCanon;
 506          }
 507          if ($this->_showLen) {
 508              $params['l'] = $this->_bodyLen;
 509          }
 510          if ($this->_signatureTimestamp === true) {
 511              $params['t'] = time();
 512              if ($this->_signatureExpiration !== false) {
 513                  $params['x'] = $params['t'] + $this->_signatureExpiration;
 514              }
 515          } else {
 516              if ($this->_signatureTimestamp !== false) {
 517                  $params['t'] = $this->_signatureTimestamp;
 518              }
 519              if ($this->_signatureExpiration !== false) {
 520                  $params['x'] = $this->_signatureExpiration;
 521              }
 522          }
 523          if ($this->_debugHeaders) {
 524              $params['z'] = implode('|', $this->_debugHeadersData);
 525          }
 526          $string = '';
 527          foreach ($params as $k => $v) {
 528              $string .= $k . '=' . $v . '; ';
 529          }
 530          $string = trim($string);
 531          $headers->addTextHeader('DKIM-Signature', $string);
 532          // Add the last DKIM-Signature
 533          $tmp = $headers->getAll('DKIM-Signature');
 534          $this->_dkimHeader = end($tmp);
 535          $this->_addHeader(trim($this->_dkimHeader->toString()) . "\r\n b=", true);
 536          $this->_endOfHeaders();
 537          if ($this->_debugHeaders) {
 538              $headers->addTextHeader('X-DebugHash', base64_encode($this->_headerHash));
 539          }
 540          $this->_dkimHeader->setValue($string . " b=" . trim(chunk_split(base64_encode($this->_getEncryptedHash()), 73, " ")));
 541  
 542          return $this;
 543      }

 547      protected function _addHeader($header, $is_sig = false)
 548      {
 549          switch ($this->_headerCanon) {
 550              case 'relaxed' :
 551                  // Prepare Header and cascade
 552                  $exploded = explode(':', $header, 2);
 553                  $name = strtolower(trim($exploded[0]));
 554                  $value = str_replace("\r\n", "", $exploded[1]);
 555                  $value = preg_replace("/[ \t][ \t]+/", " ", $value);
 556                  $header = $name . ":" . trim($value) . ($is_sig ? '' : "\r\n");
 557              case 'simple' :
 558                  // Nothing to do
 559          }
 560          $this->_addToHeaderHash($header);
 561      }

 568      protected function _canonicalizeBody($string)
 569      {
 570          $len = strlen($string);
 571          $canon = '';
 572          $method = ($this->_bodyCanon == "relaxed");
 573          for ($i = 0; $i < $len; ++$i) {
 574              if ($this->_bodyCanonIgnoreStart > 0) {
 575                  --$this->_bodyCanonIgnoreStart;
 576                  continue;
 577              }
 578              switch ($string[$i]) {
 579                  case "\r" :
 580                      $this->_bodyCanonLastChar = "\r";
 581                      break;
 582                  case "\n" :
 583                      if ($this->_bodyCanonLastChar == "\r") {
 584                          if ($method) {
 585                              $this->_bodyCanonSpace = false;
 586                          }
 587                          if ($this->_bodyCanonLine == '') {
 588                              ++$this->_bodyCanonEmptyCounter;
 589                          } else {
 590                              $this->_bodyCanonLine = '';
 591                              $canon .= "\r\n";
 592                          }
 593                      } else {
 594                          // Wooops Error
 595                          // todo handle it but should never happen
 596                      }
 597                      break;
 598                  case " " :
 599                  case "\t" :
 600                      if ($method) {
 601                          $this->_bodyCanonSpace = true;
 602                          break;
 603                      }
 604                  default :
 605                      if ($this->_bodyCanonEmptyCounter > 0) {
 606                          $canon .= str_repeat("\r\n", $this->_bodyCanonEmptyCounter);
 607                          $this->_bodyCanonEmptyCounter = 0;
 608                      }
 609                      if ($this->_bodyCanonSpace) {
 610                          $this->_bodyCanonLine .= ' ';
 611                          $canon .= ' ';
 612                          $this->_bodyCanonSpace = false;
 613                      }
 614                      $this->_bodyCanonLine .= $string[$i];
 615                      $canon .= $string[$i];
 616              }
 617          }
 618          $this->_addToBodyHash($canon);
 619      }

 630      private function _addToBodyHash($string)
 631      {
 632          $len = strlen($string);
 633          if ($len > ($new_len = ($this->_maxLen - $this->_bodyLen))) {
 634              $string = substr($string, 0, $new_len);
 635              $len = $new_len;
 636          }
 637          hash_update($this->_bodyHashHandler, $string);
 638          $this->_bodyLen += $len;
 639      }

 649      private function _getEncryptedHash()
 650      {
 651          $signature = '';
 652          switch ($this->_hashAlgorithm) {
 653              case 'rsa-sha1':
 654                  $algorithm = 'sha1';
 655                  break;
 656              case 'rsa-sha256':
 657                  $algorithm = 'sha256';
 658                  break;
 659          }
 660          $pkeyId=openssl_get_privatekey($this->_privateKey);
 661          if (!$pkeyId) {
 662              throw new Swift_SwiftException('Unable to load DKIM Private Key ['.openssl_error_string().']');
 663          }
 664          if (openssl_sign($this->_headerCanonData, $signature, $this->_privateKey, $algorithm)) {
 665              return $signature;
 666          }
 667          throw new Swift_SwiftException('Unable to sign DKIM Hash ['.openssl_error_string().']');
 668      }
