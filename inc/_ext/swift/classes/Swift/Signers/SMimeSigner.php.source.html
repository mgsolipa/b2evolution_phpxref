  96      public function setSignCertificate($certificate, $privateKey = null, $signOptions = PKCS7_DETACHED)
  97      {
  98          $this->signCertificate = 'file://' . str_replace('\\', '/', realpath($certificate));
  99  
 100          if (null !== $privateKey) {
 101              if (is_array($privateKey)) {
 102                  $this->signPrivateKey = $privateKey;
 103                  $this->signPrivateKey[0] = 'file://' . str_replace('\\', '/', realpath($privateKey[0]));
 104              } else {
 105                  $this->signPrivateKey = 'file://' . str_replace('\\', '/', realpath($privateKey));
 106              }
 107          }
 108  
 109          $this->signOptions = $signOptions;
 110  
 111          return $this;

 196      /**
 197       * Change the Swift_SignedMessage to apply the singing.
 198       *
 199       * @param Swift_SignedMessage $message
 200       *
 201       * @return Swift_Signers_SMimeSigner
 202       */
 203      public function signMessage(Swift_SignedMessage $message)
 204      {
 205          if (null === $this->signCertificate && null === $this->encryptCert) {
 206              return $this;
 207          }
 208  
 209          // Store the message using ByteStream to a file{1}
 210          // Remove all Children
 211          // Sign file{1}, parse the new MIME headers and set them on the primary MimeEntity
 212          // Set the singed-body as the new body (without boundary)
 213  
 214          $messageStream = new Swift_ByteStream_TemporaryFileByteStream();
 215          $this->toSMimeByteStream($messageStream, $message);
 216          $message->setEncoder(Swift_DependencyContainer::getInstance()->lookup('mime.rawcontentencoder'));
 217  
 218          $message->setChildren(array());
 219          $this->streamToMime($messageStream, $message);
 220  
 221      }

 233      /**
 234       * @param Swift_InputByteStream $inputStream
 235       * @param Swift_SignedMessage   $mimeEntity
 236       */
 237      protected function toSMimeByteStream(Swift_InputByteStream $inputStream, Swift_SignedMessage $message)
 238      {
 239          $mimeEntity = $this->createMessage($message);
 240          $messageStream = new Swift_ByteStream_TemporaryFileByteStream();
 241  
 242          $mimeEntity->toByteStream($messageStream);
 243          $messageStream->commit();
 244  
 245          if (null !== $this->signCertificate && null !== $this->encryptCert) {
 246              $temporaryStream = new Swift_ByteStream_TemporaryFileByteStream();
 247  
 248              if ($this->signThenEncrypt) {
 249                  $this->messageStreamToSignedByteStream($messageStream, $temporaryStream);
 250                  $this->messageStreamToEncryptedByteStream($temporaryStream, $inputStream);
 251              } else {
 252                  $this->messageStreamToEncryptedByteStream($messageStream, $temporaryStream);
 253                  $this->messageStreamToSignedByteStream($temporaryStream, $inputStream);
 254              }
 255          } elseif ($this->signCertificate !== null) {
 256              $this->messageStreamToSignedByteStream($messageStream, $inputStream);
 257          } else {
 258              $this->messageStreamToEncryptedByteStream($messageStream, $inputStream);
 259          }
 260      }

 267      protected function createMessage(Swift_SignedMessage $message)
 268      {
 269          $mimeEntity = new Swift_Message('', $message->getBody(), $message->getContentType(), $message->getCharset());
 270          $mimeEntity->setChildren($message->getChildren());
 271  
 272          $messageHeaders = $mimeEntity->getHeaders();
 273          $messageHeaders->remove('Message-ID');
 274          $messageHeaders->remove('Date');
 275          $messageHeaders->remove('Subject');
 276          $messageHeaders->remove('MIME-Version');
 277          $messageHeaders->remove('To');
 278          $messageHeaders->remove('From');

 280          return $mimeEntity;

 289      protected function messageStreamToSignedByteStream(Swift_FileStream $outputStream, Swift_InputByteStream $inputStream)
 290      {
 291          $signedMessageStream = new Swift_ByteStream_TemporaryFileByteStream();
 292  
 293          if (!openssl_pkcs7_sign($outputStream->getPath(), $signedMessageStream->getPath(), $this->signCertificate, $this->signPrivateKey, array(), $this->signOptions)) {
 294              throw new Swift_IoException(sprintf('Failed to sign S/Mime message. Error: "%s".', openssl_error_string()));
 295          }
 296  
 297          $this->copyFromOpenSSLOutput($signedMessageStream, $inputStream);
 298      }

 308          $encryptedMessageStream = new Swift_ByteStream_TemporaryFileByteStream();
 309  
 310          if (!openssl_pkcs7_encrypt($outputStream->getPath(), $encryptedMessageStream->getPath(), $this->encryptCert, array())) {
 311              throw new Swift_IoException(sprintf('Failed to encrypt S/Mime message. Error: "%s".', openssl_error_string()));

 318       * @param Swift_OutputByteStream $fromStream
 319       * @param Swift_InputByteStream  $toStream
 320       */
 321      protected function copyFromOpenSSLOutput(Swift_OutputByteStream $fromStream, Swift_InputByteStream $toStream)
 322      {
 323          $bufferLength = 4096;
 324          $filteredStream = new Swift_ByteStream_TemporaryFileByteStream();
 325          $filteredStream->addFilter($this->replacementFactory->createFilter("\r\n", "\n"), 'CRLF to LF');
 326          $filteredStream->addFilter($this->replacementFactory->createFilter("\n", "\r\n"), 'LF to CRLF');
 327  
 328          while (false !== ($buffer = $fromStream->read($bufferLength))) {
 329              $filteredStream->write($buffer);
 330          }
 331  
 332          $filteredStream->flushBuffers();
 333  
 334          while (false !== ($buffer = $filteredStream->read($bufferLength))) {
 335              $toStream->write($buffer);
 336          }
 337  
 338          $toStream->commit();
 339      }

 341      /**
 342       * Merges an OutputByteStream to Swift_SignedMessage.
 343       *
 344       * @param Swift_OutputByteStream $fromStream
 345       * @param Swift_Message          $message
 346       */
 347      protected function streamToMime(Swift_OutputByteStream $fromStream, Swift_Message $message)
 348      {
 349          $bufferLength = 78;
 350          $headerData = '';
 351  
 352          $fromStream->setReadPointer(0);
 353  
 354          while (($buffer = $fromStream->read($bufferLength)) !== false) {
 355              $headerData .= $buffer;
 356  
 357              if (false !== strpos($buffer, "\r\n\r\n")) {
 358                  break;
 359              }
 360          }
 361  
 362          $headersPosEnd = strpos($headerData, "\r\n\r\n");
 363          $headerData = trim($headerData);
 364          $headerData = substr($headerData, 0, $headersPosEnd);
 365          $headerLines = explode("\r\n", $headerData);
 366          unset($headerData);
 367  
 368          $headers = array();
 369          $currentHeaderName = '';
 370  
 371          foreach ($headerLines as $headerLine) {
 372              // Line separated
 373              if (ctype_space($headerLines[0]) || false === strpos($headerLine, ':')) {
 374                  $headers[$currentHeaderName] .= ' ' . trim($headerLine);
 375                  continue;
 376              }
 377  
 378              $header = explode(':', $headerLine, 2);
 379              $currentHeaderName = strtolower($header[0]);
 380              $headers[$currentHeaderName] = trim($header[1]);
 381          }
 382  
 383          $messageStream = new Swift_ByteStream_TemporaryFileByteStream();
 384          $messageStream->addFilter($this->replacementFactory->createFilter("\r\n", "\n"), 'CRLF to LF');
 385          $messageStream->addFilter($this->replacementFactory->createFilter("\n", "\r\n"), 'LF to CRLF');
 386  
 387          $messageHeaders = $message->getHeaders();
 388  
 389          // No need to check for 'application/pkcs7-mime', as this is always base64
 390          if ('multipart/signed;' === substr($headers['content-type'], 0, 17)) {
 391              if (!preg_match('/boundary=("[^"]+"|(?:[^\s]+|$))/is', $headers['content-type'], $contentTypeData)) {
 392                  throw new Swift_SwiftException('Failed to find Boundary parameter');
 393              }
 394  
 395              $boundary = trim($contentTypeData['1'], '"');
 396              $boundaryLen = strlen($boundary);
 397  
 398              // Skip the header and CRLF CRLF
 399              $fromStream->setReadPointer($headersPosEnd + 4);
 400  
 401              while (false !== ($buffer = $fromStream->read($bufferLength))) {
 402                  $messageStream->write($buffer);
 403              }
 404  
 405              $messageStream->commit();
 406  
 407              $messageHeaders->remove('Content-Transfer-Encoding');
 408              $message->setContentType($headers['content-type']);
 409              $message->setBoundary($boundary);
 410              $message->setBody($messageStream);
 411          } else {
 412              $fromStream->setReadPointer($headersPosEnd + 4);
 413  
 414              if (null === $this->headerFactory) {
 415                  $this->headerFactory = Swift_DependencyContainer::getInstance()->lookup('mime.headerfactory');
 416              }
 417  
 418              $message->setContentType($headers['content-type']);
 419              $messageHeaders->set($this->headerFactory->createTextHeader('Content-Transfer-Encoding', $headers['content-transfer-encoding']));
 420              $messageHeaders->set($this->headerFactory->createTextHeader('Content-Disposition', $headers['content-disposition']));
 421  
 422              while (false !== ($buffer = $fromStream->read($bufferLength))) {
 423                  $messageStream->write($buffer);
 424              }
 425  
 426              $messageStream->commit();
 427              $message->setBody($messageStream);
 428          }
 429      }
