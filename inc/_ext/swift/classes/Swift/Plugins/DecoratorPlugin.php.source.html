  36      /**
  37       * Create a new DecoratorPlugin with $replacements.
  38       *
  39       * The $replacements can either be an associative array, or an implementation
  40       * of {@link Swift_Plugins_Decorator_Replacements}.
  41       *
  42       * When using an array, it should be of the form:
  43       * <code>
  44       * $replacements = array(
  45       *  "address1@domain.tld" => array("{a}" => "b", "{c}" => "d"),
  46       *  "address2@domain.tld" => array("{a}" => "x", "{c}" => "y")
  47       * )
  48       * </code>
  49       *
  50       * When using an instance of {@link Swift_Plugins_Decorator_Replacements},
  51       * the object should return just the array of replacements for the address
  52       * given to {@link Swift_Plugins_Decorator_Replacements::getReplacementsFor()}.
  53       *
  54       * @param mixed $replacements Array or Swift_Plugins_Decorator_Replacements
  55       */
  56      public function __construct($replacements)
  57      {
  58          $this->setReplacements($replacements);
  59      }

  68      public function setReplacements($replacements)

  70          if (!($replacements instanceof \Swift_Plugins_Decorator_Replacements)) {

  72          } else {
  73              $this->_replacements = $replacements;

  77      /**
  78       * Invoked immediately before the Message is sent.
  79       *
  80       * @param Swift_Events_SendEvent $evt
  81       */
  82      public function beforeSendPerformed(Swift_Events_SendEvent $evt)
  83      {
  84          $message = $evt->getMessage();
  85          $this->_restoreMessage($message);
  86          $to = array_keys($message->getTo());
  87          $address = array_shift($to);
  88          if ($replacements = $this->getReplacementsFor($address)) {
  89              $body = $message->getBody();
  90              $search = array_keys($replacements);
  91              $replace = array_values($replacements);
  92              $bodyReplaced = str_replace(
  93                  $search, $replace, $body
  94                  );
  95              if ($body != $bodyReplaced) {
  96                  $this->_originalBody = $body;
  97                  $message->setBody($bodyReplaced);
  98              }
  99  
 100              foreach ($message->getHeaders()->getAll() as $header) {
 101                  $body = $header->getFieldBodyModel();
 102                  $count = 0;
 103                  if (is_array($body)) {
 104                      $bodyReplaced = array();
 105                      foreach ($body as $key => $value) {
 106                          $count1 = 0;
 107                          $count2 = 0;
 108                          $key = is_string($key) ? str_replace($search, $replace, $key, $count1) : $key;
 109                          $value = is_string($value) ? str_replace($search, $replace, $value, $count2) : $value;
 110                          $bodyReplaced[$key] = $value;
 111  
 112                          if (!$count && ($count1 || $count2)) {
 113                              $count = 1;
 114                          }
 115                      }
 116                  } else {
 117                      $bodyReplaced = str_replace($search, $replace, $body, $count);
 118                  }
 119  
 120                  if ($count) {
 121                      $this->_originalHeaders[$header->getFieldName()] = $body;
 122                      $header->setFieldBodyModel($bodyReplaced);
 123                  }
 124              }
 125  
 126              $children = (array) $message->getChildren();
 127              foreach ($children as $child) {
 128                  list($type, ) = sscanf($child->getContentType(), '%[^/]/%s');
 129                  if ('text' == $type) {
 130                      $body = $child->getBody();
 131                      $bodyReplaced = str_replace(
 132                          $search, $replace, $body
 133                          );
 134                      if ($body != $bodyReplaced) {
 135                          $child->setBody($bodyReplaced);
 136                          $this->_originalChildBodies[$child->getId()] = $body;
 137                      }
 138                  }
 139              }
 140              $this->_lastMessage = $message;
 141          }
 142      }

 144      /**
 145       * Find a map of replacements for the address.
 146       *
 147       * If this plugin was provided with a delegate instance of
 148       * {@link Swift_Plugins_Decorator_Replacements} then the call will be
 149       * delegated to it.  Otherwise, it will attempt to find the replacements
 150       * from the array provided in the constructor.
 151       *
 152       * If no replacements can be found, an empty value (NULL) is returned.
 153       *
 154       * @param string $address
 155       *
 156       * @return array
 157       */
 158      public function getReplacementsFor($address)
 159      {
 160          if ($this->_replacements instanceof Swift_Plugins_Decorator_Replacements) {
 161              return $this->_replacements->getReplacementsFor($address);
 162          } else {
 163              return isset($this->_replacements[$address])
 164                  ? $this->_replacements[$address]
 165                  : null
 166                  ;
 167          }
 168      }

 182      /** Restore a changed message back to its original state */
 183      private function _restoreMessage(Swift_Mime_Message $message)
 184      {
 185          if ($this->_lastMessage === $message) {
 186              if (isset($this->_originalBody)) {
 187                  $message->setBody($this->_originalBody);
 188                  $this->_originalBody = null;
 189              }
 190              if (!empty($this->_originalHeaders)) {
 191                  foreach ($message->getHeaders()->getAll() as $header) {
 192                      if (array_key_exists($header->getFieldName(), $this->_originalHeaders)) {
 193                          $header->setFieldBodyModel($this->_originalHeaders[$header->getFieldName()]);
 194                      }
 195                  }
 196                  $this->_originalHeaders = array();
 197              }
 198              if (!empty($this->_originalChildBodies)) {
 199                  $children = (array) $message->getChildren();
 200                  foreach ($children as $child) {
 201                      $id = $child->getId();
 202                      if (array_key_exists($id, $this->_originalChildBodies)) {
 203                          $child->setBody($this->_originalChildBodies[$id]);
 204                      }
 205                  }
 206                  $this->_originalChildBodies = array();
 207              }
 208              $this->_lastMessage = null;
 209          }
 210      }
