 141      /**
 142       * Connect to the POP3 host and authenticate.
 143       *
 144       * @throws Swift_Plugins_Pop_Pop3Exception if connection fails
 145       */
 146      public function connect()
 147      {
 148          if (isset($this->_connection)) {
 149              $this->_connection->connect();
 150          } else {
 151              if (!isset($this->_socket)) {
 152                  if (!$socket = fsockopen(
 153                      $this->_getHostString(), $this->_port, $errno, $errstr, $this->_timeout))
 154                  {
 155                      throw new Swift_Plugins_Pop_Pop3Exception(
 156                          sprintf('Failed to connect to POP3 host [%s]: %s', $this->_host, $errstr)
 157                      );
 158                  }
 159                  $this->_socket = $socket;
 160  
 161                  if (false === $greeting = fgets($this->_socket)) {
 162                      throw new Swift_Plugins_Pop_Pop3Exception(
 163                          sprintf('Failed to connect to POP3 host [%s]', trim($greeting))
 164                      );
 165                  }
 166  
 167                  $this->_assertOk($greeting);
 168  
 169                  if ($this->_username) {
 170                      $this->_command(sprintf("USER %s\r\n", $this->_username));
 171                      $this->_command(sprintf("PASS %s\r\n", $this->_password));
 172                  }
 173              }
 174          }
 175      }

 181      {
 182          if (isset($this->_connection)) {

 186              if (!fclose($this->_socket)) {
 187                  throw new Swift_Plugins_Pop_Pop3Exception(
 188                      sprintf('POP3 host [%s] connection could not be stopped', $this->_host)
 189                  );
 190              }
 191              $this->_socket = null;
 192          }

 235      private function _command($command)
 236      {
 237          if (!fwrite($this->_socket, $command)) {
 238              throw new Swift_Plugins_Pop_Pop3Exception(
 239                  sprintf('Failed to write command [%s] to POP3 host', trim($command))
 240              );
 241          }
 242  
 243          if (false === $response = fgets($this->_socket)) {
 244              throw new Swift_Plugins_Pop_Pop3Exception(
 245                  sprintf('Failed to read from POP3 host after command [%s]', trim($command))
 246              );
 247          }
 248  
 249          $this->_assertOk($response);
 250  
 251          return $response;
 252      }

 274          }
