  91      public function beforeSendPerformed(Swift_Events_SendEvent $evt)
  92      {
  93          $message = $evt->getMessage();
  94          $headers = $message->getHeaders();
  95  
  96          // conditionally save current recipients
  97  
  98          if ($headers->has('to')) {
  99              $headers->addMailboxHeader('X-Swift-To', $message->getTo());
 100          }
 101  
 102          if ($headers->has('cc')) {
 103              $headers->addMailboxHeader('X-Swift-Cc', $message->getCc());
 104          }
 105  
 106          if ($headers->has('bcc')) {
 107              $headers->addMailboxHeader('X-Swift-Bcc', $message->getBcc());
 108          }
 109  
 110          // Add hard coded recipient
 111          $message->addTo($this->_recipient);
 112  
 113          // Filter remaining headers against whitelist
 114          $this->_filterHeaderSet($headers, 'To');
 115          $this->_filterHeaderSet($headers, 'Cc');
 116          $this->_filterHeaderSet($headers, 'Bcc');
 117      }
 118  
 119      /**
 120       * Filter header set against a whitelist of regular expressions
 121       *
 122       * @param Swift_Mime_HeaderSet $headerSet
 123       * @param string $type
 124       */
 125      private function _filterHeaderSet(Swift_Mime_HeaderSet $headerSet, $type)
 126      {
 127          foreach ($headerSet->getAll($type) as $headers) {
 128              $headers->setNameAddresses($this->_filterNameAddresses($headers->getNameAddresses()));
 129          }
 130      }
 131  
 132      /**
 133       * Filtered list of addresses => name pairs
 134       *
 135       * @param array $recipients
 136       * @return array
 137       */
 138      private function _filterNameAddresses(array $recipients)
 139      {
 140          $filtered = array();
 141  
 142          foreach ($recipients as $address => $name) {
 143              if ($this->_isWhitelisted($address)) {
 144                  $filtered[$address] = $name;
 145              }
 146          }
 147  
 148          return $filtered;
 149      }
 150  
 151      /**
 152       * Matches address against whitelist of regular expressions
 153       *
 154       * @param $recipient
 155       * @return bool
 156       */
 157      protected function _isWhitelisted($recipient)
 158      {
 159          if ($recipient === $this->_recipient) {
 160              return true;
 161          }
 162  
 163          foreach ($this->_whitelist as $pattern) {
 164              if (preg_match($pattern, $recipient)) {
 165                  return true;
 166              }
 167          }
 168  
 169          return false;
 170      }
 171  
 172      /**

 175       * @param Swift_Events_SendEvent $evt
 176       */
 177      public function sendPerformed(Swift_Events_SendEvent $evt)

 179          $this->_restoreMessage($evt->getMessage());

 184      private function _restoreMessage(Swift_Mime_Message $message)
 185      {
 186          // restore original headers
 187          $headers = $message->getHeaders();
 188  
 189          if ($headers->has('X-Swift-To')) {
 190              $message->setTo($headers->get('X-Swift-To')->getNameAddresses());
 191              $headers->removeAll('X-Swift-To');
 192          }
 193  
 194          if ($headers->has('X-Swift-Cc')) {
 195              $message->setCc($headers->get('X-Swift-Cc')->getNameAddresses());
 196              $headers->removeAll('X-Swift-Cc');
 197          }
 198  
 199          if ($headers->has('X-Swift-Bcc')) {
 200              $message->setBcc($headers->get('X-Swift-Bcc')->getNameAddresses());
 201              $headers->removeAll('X-Swift-Bcc');
 202          }
 203      }
