 291       * @return resource
 292       */
 293      private function _getHandle($nsKey, $itemKey, $position)
 294      {
 295          if (!isset($this->_keys[$nsKey][$itemKey])) {
 296              $openMode = $this->hasKey($nsKey, $itemKey)
 297                  ? 'r+b'
 298                  : 'w+b'
 299                  ;
 300              $fp = fopen($this->_path . '/' . $nsKey . '/' . $itemKey, $openMode);
 301              $this->_keys[$nsKey][$itemKey] = $fp;
 302          }
 303          if (self::POSITION_START == $position) {
 304              fseek($this->_keys[$nsKey][$itemKey], 0, SEEK_SET);
 305          } elseif (self::POSITION_END == $position) {
 306              fseek($this->_keys[$nsKey][$itemKey], 0, SEEK_END);
 307          }
 308  
 309          return $this->_keys[$nsKey][$itemKey];
 310      }
