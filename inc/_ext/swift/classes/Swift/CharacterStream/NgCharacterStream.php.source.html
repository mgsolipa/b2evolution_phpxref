 158       * @see Swift_CharacterStream::read()
 159       *
 160       * @param integer $length
 161       *
 162       * @return string
 163       */
 164      public function read($length)
 165      {
 166          if ($this->_currentPos>=$this->_charCount) {
 167            return false;
 168          }
 169          $ret=false;
 170          $length = ($this->_currentPos+$length > $this->_charCount)
 171            ? $this->_charCount - $this->_currentPos
 172            : $length;
 173            switch ($this->_mapType) {
 174              case Swift_CharacterReader::MAP_TYPE_FIXED_LEN:
 175                  $len = $length*$this->_map;
 176                  $ret = substr($this->_datas,
 177                          $this->_currentPos * $this->_map,
 178                          $len);
 179                  $this->_currentPos += $length;
 180                  break;
 181  
 182              case Swift_CharacterReader::MAP_TYPE_INVALID:
 183                  $end = $this->_currentPos + $length;
 184                  $end = $end > $this->_charCount
 185                      ?$this->_charCount
 186                      :$end;
 187                  $ret = '';
 188                  for (; $this->_currentPos < $length; ++$this->_currentPos) {
 189                      if (isset ($this->_map[$this->_currentPos])) {
 190                          $ret .= '?';
 191                      } else {
 192                          $ret .= $this->_datas[$this->_currentPos];
 193                      }
 194                  }
 195                  break;
 196  
 197              case Swift_CharacterReader::MAP_TYPE_POSITIONS:
 198                  $end = $this->_currentPos + $length;
 199                  $end = $end > $this->_charCount
 200                      ?$this->_charCount
 201                      :$end;
 202                  $ret = '';
 203                  $start = 0;
 204                  if ($this->_currentPos>0) {
 205                      $start = $this->_map['p'][$this->_currentPos-1];
 206                  }
 207                  $to = $start;
 208                  for (; $this->_currentPos < $end; ++$this->_currentPos) {
 209                      if (isset($this->_map['i'][$this->_currentPos])) {
 210                          $ret .= substr($this->_datas, $start, $to - $start).'?';
 211                          $start = $this->_map['p'][$this->_currentPos];
 212                      } else {
 213                          $to = $this->_map['p'][$this->_currentPos];
 214                      }
 215                  }
 216                  $ret .= substr($this->_datas, $start, $to - $start);
 217                  break;
 218          }
 219  
 220          return $ret;
 221      }
