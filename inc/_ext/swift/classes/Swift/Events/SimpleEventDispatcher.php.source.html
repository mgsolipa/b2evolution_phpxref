 114      public function bindEventListener(Swift_Events_EventListener $listener)
 115      {
 116          foreach ($this->_listeners as $l) {
 117              //Already loaded

 125      /**
 126       * Dispatch the given Event to all suitable listeners.
 127       *
 128       * @param Swift_Events_EventObject $evt
 129       * @param string                   $target method
 130       */
 131      public function dispatchEvent(Swift_Events_EventObject $evt, $target)
 132      {
 133          $this->_prepareBubbleQueue($evt);
 134          $this->_bubble($evt, $target);
 135      }

 137      // -- Private methods

 139      /** Queue listeners on a stack ready for $evt to be bubbled up it */
 140      private function _prepareBubbleQueue(Swift_Events_EventObject $evt)
 141      {
 142          $this->_bubbleQueue = array();
 143          $evtClass = get_class($evt);
 144          foreach ($this->_listeners as $listener) {
 145              if (array_key_exists($evtClass, $this->_eventMap)
 146                  && ($listener instanceof $this->_eventMap[$evtClass]))
 147              {
 148                  $this->_bubbleQueue[] = $listener;
 149              }
 150          }
 151      }
 152  
 153      /** Bubble $evt up the stack calling $target() on each listener */
 154      private function _bubble(Swift_Events_EventObject $evt, $target)
 155      {
 156          if (!$evt->bubbleCancelled() && $listener = array_shift($this->_bubbleQueue)) {
 157              $listener->$target($evt);
 158              $this->_bubble($evt, $target);
 159          }
 160      }
