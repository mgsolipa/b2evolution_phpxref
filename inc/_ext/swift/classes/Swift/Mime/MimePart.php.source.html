 199      /** Encode charset when charset is not utf-8 */
 200      protected function _convertString($string)
 201      {
 202          $charset = strtolower($this->getCharset());
 203          if (!in_array($charset, array('utf-8', 'iso-8859-1', ''))) {
 204              // mb_convert_encoding must be the first one to check, since iconv cannot convert some words.
 205              if (function_exists('mb_convert_encoding')) {
 206                  $string = mb_convert_encoding($string, $charset, 'utf-8');
 207              } elseif (function_exists('iconv')) {
 208                  $string = iconv($charset, 'utf-8//TRANSLIT//IGNORE', $string);
 209              } else {
 210                  throw new Swift_SwiftException('No suitable convert encoding function (use UTF-8 as your charset or install the mbstring or iconv extension).');
 211              }
 212  
 213              return $string;
 214          }
 215  
 216          return $string;
 217      }
