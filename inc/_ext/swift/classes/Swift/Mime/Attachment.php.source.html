 139      public function setFile(Swift_FileStream $file, $contentType = null)
 140      {
 141          $this->setFilename(basename($file->getPath()));
 142          $this->setBody($file, $contentType);
 143          if (!isset($contentType)) {
 144              $extension = strtolower(substr(
 145                  $file->getPath(), strrpos($file->getPath(), '.') + 1
 146                  ));
 147  
 148              if (array_key_exists($extension, $this->_mimeTypes)) {
 149                  $this->setContentType($this->_mimeTypes[$extension]);
 150              }
 151          }
 152  
 153          return $this;
 154      }
