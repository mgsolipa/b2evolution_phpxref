 444      /**
 445       * Set the priority of this message.
 446       *
 447       * The value is an integer where 1 is the highest priority and 5 is the lowest.
 448       *
 449       * @param integer $priority
 450       *
 451       * @return Swift_Mime_SimpleMessage
 452       */
 453      public function setPriority($priority)
 454      {
 455          $priorityMap = array(
 456              1 => 'Highest',
 457              2 => 'High',
 458              3 => 'Normal',
 459              4 => 'Low',
 460              5 => 'Lowest'
 461              );
 462          $pMapKeys = array_keys($priorityMap);
 463          if ($priority > max($pMapKeys)) {
 464              $priority = max($pMapKeys);
 465          } elseif ($priority < min($pMapKeys)) {
 466              $priority = min($pMapKeys);
 467          }
 468          if (!$this->_setHeaderFieldModel('X-Priority',
 469              sprintf('%d (%s)', $priority, $priorityMap[$priority])))
 470          {
 471              $this->getHeaders()->addTextHeader('X-Priority',
 472                  sprintf('%d (%s)', $priority, $priorityMap[$priority]));
 473          }
 474  
 475          return $this;
 476      }
