 300      public function toString()
 301      {
 302          $string = '';
 303          $headers = $this->_headers;
 304          if ($this->_canSort()) {
 305              uksort($headers, array($this, '_sortHeaders'));
 306          }
 307          foreach ($headers as $collection) {
 308              foreach ($collection as $header) {
 309                  if ($this->_isDisplayed($header) || $header->getFieldBody() != '') {
 310                      $string .= $header->toString();
 311                  }
 312              }
 313          }

 332      /** Save a Header to the internal collection */
 333      private function _storeHeader($name, Swift_Mime_Header $header, $offset = null)
 334      {
 335          if (!isset($this->_headers[strtolower($name)])) {
 336              $this->_headers[strtolower($name)] = array();
 337          }
 338          if (!isset($offset)) {
 339              $this->_headers[strtolower($name)][] = $header;
 340          } else {
 341              $this->_headers[strtolower($name)][$offset] = $header;
 342          }
 343      }

 345      /** Test if the headers can be sorted */
 346      private function _canSort()
 347      {
 348          return count($this->_order) > 0;
 349      }

 351      /** uksort() algorithm for Header ordering */
 352      private function _sortHeaders($a, $b)
 353      {
 354          $lowerA = strtolower($a);
 355          $lowerB = strtolower($b);
 356          $aPos = array_key_exists($lowerA, $this->_order)
 357              ? $this->_order[$lowerA]
 358              : -1;
 359          $bPos = array_key_exists($lowerB, $this->_order)
 360              ? $this->_order[$lowerB]
 361              : -1;
 362  
 363          if ($aPos == -1) {
 364              return 1;
 365          } elseif ($bPos == -1) {
 366              return -1;
 367          }
 368  
 369          return ($aPos < $bPos) ? -1 : 1;
 370      }

 372      /** Test if the given Header is always displayed */
 373      private function _isDisplayed(Swift_Mime_Header $header)
 374      {
 375          return array_key_exists(strtolower($header->getFieldName()), $this->_required);
 376      }

 384              }
 385          }
