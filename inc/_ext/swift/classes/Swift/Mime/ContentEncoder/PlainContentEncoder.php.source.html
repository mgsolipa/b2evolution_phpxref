  34      /**
  35       * Creates a new PlainContentEncoder with $name (probably 7bit or 8bit).

  37       * @param string  $name
  38       * @param boolean $canonical If canonicalization transformation should be done.

  46      /**
  47       * Encode a given string to produce an encoded string.
  48       *
  49       * @param string  $string
  50       * @param integer $firstLineOffset ignored
  51       * @param integer $maxLineLength   - 0 means no wrapping will occur
  52       *
  53       * @return string
  54       */
  55      public function encodeString($string, $firstLineOffset = 0, $maxLineLength = 0)
  56      {
  57          if ($this->_canonical) {
  58              $string = $this->_canonicalize($string);
  59          }
  60  
  61          return $this->_safeWordWrap($string, $maxLineLength, "\r\n");
  62      }

  64      /**
  65       * Encode stream $in to stream $out.
  66       *
  67       * @param Swift_OutputByteStream $os
  68       * @param Swift_InputByteStream  $is
  69       * @param integer                $firstLineOffset ignored
  70       * @param integer                $maxLineLength   optional, 0 means no wrapping will occur
  71       */
  72      public function encodeByteStream(Swift_OutputByteStream $os, Swift_InputByteStream $is, $firstLineOffset = 0, $maxLineLength = 0)
  73      {
  74          $leftOver = '';
  75          while (false !== $bytes = $os->read(8192)) {
  76              $toencode = $leftOver . $bytes;
  77              if ($this->_canonical) {
  78                  $toencode = $this->_canonicalize($toencode);
  79              }
  80              $wrapped = $this->_safeWordWrap($toencode, $maxLineLength, "\r\n");
  81              $lastLinePos = strrpos($wrapped, "\r\n");
  82              $leftOver = substr($wrapped, $lastLinePos);
  83              $wrapped = substr($wrapped, 0, $lastLinePos);
  84  
  85              $is->write($wrapped);
  86          }
  87          if (strlen($leftOver)) {
  88              $is->write($leftOver);
  89          }
  90      }

 111      /**
 112       * A safer (but weaker) wordwrap for unicode.
 113       *
 114       * @param string  $string
 115       * @param integer $length
 116       * @param string  $le
 117       *
 118       * @return string
 119       */
 120      private function _safeWordwrap($string, $length = 75, $le = "\r\n")
 121      {
 122          if (0 >= $length) {
 123              return $string;
 124          }
 125  
 126          $originalLines = explode($le, $string);
 127  
 128          $lines = array();
 129          $lineCount = 0;
 130  
 131          foreach ($originalLines as $originalLine) {
 132              $lines[] = '';
 133              $currentLine =& $lines[$lineCount++];
 134  
 135              //$chunks = preg_split('/(?<=[\ \t,\.!\?\-&\+\/])/', $originalLine);
 136              $chunks = preg_split('/(?<=\s)/', $originalLine);
 137  
 138              foreach ($chunks as $chunk) {
 139                  if (0 != strlen($currentLine)
 140                      && strlen($currentLine . $chunk) > $length)
 141                  {
 142                      $lines[] = '';
 143                      $currentLine =& $lines[$lineCount++];
 144                  }
 145                  $currentLine .= $chunk;
 146              }
 147          }
 148  
 149          return implode("\r\n", $lines);
 150      }
 151  
 152      /**
 153       * Canonicalize string input (fix CRLF).
 154       *
 155       * @param string $string
 156       *
 157       * @return string
 158       */
 159      private function _canonicalize($string)
 160      {
 161          return str_replace(
 162              array("\r\n", "\r", "\n"),
 163              array("\n", "\n", "\r\n"),
 164              $string
 165              );
 166      }
