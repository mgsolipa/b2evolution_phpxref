 236      /**
 237       * Produces a compliant, formatted RFC 2822 'phrase' based on the string given.
 238       *
 239       * @param Swift_Mime_Header        $header
 240       * @param string                   $string  as displayed
 241       * @param string                   $charset of the text
 242       * @param Swift_Mime_HeaderEncoder $encoder
 243       * @param boolean                  $shorten the first line to make remove for header name
 244       *
 245       * @return string
 246       */
 247      protected function createPhrase(Swift_Mime_Header $header, $string, $charset, Swift_Mime_HeaderEncoder $encoder = null, $shorten = false)
 248      {
 249          //Treat token as exactly what was given
 250          $phraseStr = $string;
 251          //If it's not valid
 252          if (!preg_match('/^' . $this->getGrammar()->getDefinition('phrase') . '$/D', $phraseStr)) {
 253              // .. but it is just ascii text, try escaping some characters
 254              // and make it a quoted-string
 255              if (preg_match('/^' . $this->getGrammar()->getDefinition('text') . '*$/D', $phraseStr)) {
 256                  $phraseStr = $this->getGrammar()->escapeSpecials(
 257                      $phraseStr, array('"'), $this->getGrammar()->getSpecials()
 258                      );
 259                  $phraseStr = '"' . $phraseStr . '"';
 260              } else { // ... otherwise it needs encoding
 261                  //Determine space remaining on line if first line
 262                  if ($shorten) {
 263                      $usedLength = strlen($header->getFieldName() . ': ');
 264                  } else {
 265                      $usedLength = 0;
 266                  }
 267                  $phraseStr = $this->encodeWords($header, $string, $usedLength);
 268              }
 269          }
 270  
 271          return $phraseStr;
 272      }

 274      /**
 275       * Encode needed word tokens within a string of input.
 276       *
 277       * @param Swift_Mime_Header $header
 278       * @param string            $input
 279       * @param string            $usedLength optional
 280       *
 281       * @return string
 282       */
 283      protected function encodeWords(Swift_Mime_Header $header, $input, $usedLength = -1)
 284      {
 285          $value = '';
 286  
 287          $tokens = $this->getEncodableWordTokens($input);
 288  
 289          foreach ($tokens as $token) {
 290              //See RFC 2822, Sect 2.2 (really 2.2 ??)
 291              if ($this->tokenNeedsEncoding($token)) {
 292                  //Don't encode starting WSP
 293                  $firstChar = substr($token, 0, 1);
 294                  switch ($firstChar) {
 295                      case ' ':
 296                      case "\t":
 297                          $value .= $firstChar;
 298                          $token = substr($token, 1);
 299                  }
 300  
 301                  if (-1 == $usedLength) {
 302                      $usedLength = strlen($header->getFieldName() . ': ') + strlen($value);
 303                  }
 304                  $value .= $this->getTokenAsEncodedWord($token, $usedLength);
 305  
 306                  $header->setMaxLineLength(76); //Forcefully override
 307              } else {
 308                  $value .= $token;
 309              }
 310          }
 311  
 312          return $value;
 313      }

 327      /**
 328       * Splits a string into tokens in blocks of words which can be encoded quickly.
 329       *
 330       * @param string $string
 331       *
 332       * @return string[]
 333       */
 334      protected function getEncodableWordTokens($string)
 335      {
 336          $tokens = array();
 337  
 338          $encodedToken = '';
 339          //Split at all whitespace boundaries
 340          $boundaries = is_array( $string ) ? $string : preg_split('~(?=[\t ])~', $string);
 341          foreach ($boundaries as $token) {
 342              if ($this->tokenNeedsEncoding($token)) {
 343                  $encodedToken .= $token;
 344              } else {
 345                  if (strlen($encodedToken) > 0) {
 346                      $tokens[] = $encodedToken;
 347                      $encodedToken = '';
 348                  }
 349                  $tokens[] = $token;
 350              }
 351          }
 352          if (strlen($encodedToken)) {
 353              $tokens[] = $encodedToken;
 354          }
 355  
 356          return $tokens;
 357      }

 359      /**
 360       * Get a token as an encoded word for safe insertion into headers.
 361       *
 362       * @param string  $token           token to encode
 363       * @param integer $firstLineOffset optional
 364       *
 365       * @return string
 366       */
 367      protected function getTokenAsEncodedWord($token, $firstLineOffset = 0)
 368      {
 369          //Adjust $firstLineOffset to account for space needed for syntax
 370          $charsetDecl = $this->_charset;
 371          if (isset($this->_lang)) {
 372              $charsetDecl .= '*' . $this->_lang;
 373          }
 374          $encodingWrapperLength = strlen(
 375              '=?' . $charsetDecl . '?' . $this->_encoder->getName() . '??='
 376              );
 377  
 378          if ($firstLineOffset >= 75) { //Does this logic need to be here?
 379              $firstLineOffset = 0;
 380          }
 381  
 382          $encodedTextLines = explode("\r\n",
 383              $this->_encoder->encodeString(
 384                  $token, $firstLineOffset, 75 - $encodingWrapperLength, $this->_charset
 385                  )
 386          );
 387  
 388          if (strtolower($this->_charset) !== 'iso-2022-jp') { // special encoding for iso-2022-jp using mb_encode_mimeheader
 389              foreach ($encodedTextLines as $lineNum => $line) {
 390                  $encodedTextLines[$lineNum] = '=?' . $charsetDecl .
 391                      '?' . $this->_encoder->getName() .
 392                      '?' . $line . '?=';
 393              }
 394          }
 395  
 396          return implode("\r\n ", $encodedTextLines);
 397      }

 445      /**
 446       * Generate a list of all tokens in the final header.
 447       *
 448       * @param string $string The string to tokenize
 449       *
 450       * @return array An array of tokens as strings
 451       */
 452      protected function toTokens($string = null)
 453      {
 454          if (is_null($string)) {
 455              $string = $this->getFieldBody();
 456          }
 457  
 458          $tokens = array();
 459  
 460          //Generate atoms; split at all invisible boundaries followed by WSP
 461          foreach (preg_split('~(?=[ \t])~', $string) as $token) {
 462              $newTokens = $this->generateTokenLines($token);
 463              foreach ($newTokens as $newToken) {
 464                  $tokens[] = $newToken;
 465              }
 466          }
 467          return $tokens;
 468      }

 470      /**
 471       * Takes an array of tokens which appear in the header and turns them into
 472       * an RFC 2822 compliant string, adding FWSP where needed.
 473       *
 474       * @param string[] $tokens
 475       *
 476       * @return string
 477       */
 478      private function _tokensToString(array $tokens)
 479      {
 480          $lineCount = 0;
 481          $headerLines = array();
 482          $headerLines[] = $this->_name . ': ';
 483          $currentLine =& $headerLines[$lineCount++];
 484  
 485          //Build all tokens back into compliant header
 486          foreach ($tokens as $i => $token) {
 487              //Line longer than specified maximum or token was just a new line
 488              if (("\r\n" == $token) ||
 489                  ($i > 0 && strlen($currentLine . $token) > $this->_lineLength)
 490                  && 0 < strlen($currentLine))
 491              {
 492                  $headerLines[] = '';
 493                  $currentLine =& $headerLines[$lineCount++];
 494              }
 495  
 496              //Append token to the line
 497              if ("\r\n" != $token) {
 498                  $currentLine .= $token;
 499              }
 500          }
 501  
 502          //Implode with FWS (RFC 2822, 2.2.3)
 503          return implode("\r\n", $headerLines) . "\r\n";
 504      }
