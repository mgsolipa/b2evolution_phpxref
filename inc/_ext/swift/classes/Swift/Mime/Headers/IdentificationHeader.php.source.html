  90      }

  92      /**
  93       * Get the ID used in the value of this Header.
  94       *
  95       * If multiple IDs are set only the first is returned.
  96       *
  97       * @return string
  98       */
  99      public function getId()
 100      {
 101          if (count($this->_ids) > 0) {
 102              return $this->_ids[0];
 103          }
 104      }
 105  
 106      /**
 107       * Set a collection of IDs to use in the value of this Header.
 108       *
 109       * @param string[] $ids
 110       *
 111       * @throws Swift_RfcComplianceException
 112       */
 113      public function setIds(array $ids)
 114      {
 115          $actualIds = array();
 116  
 117          foreach ($ids as $id) {
 118              $this->_assertValidId($id);
 119              $actualIds[] = $id;
 120          }
 121  
 122          $this->clearCachedValueIf($this->_ids != $actualIds);
 123          $this->_ids = $actualIds;
 124      }
 125  
 126      /**
 127       * Get the list of IDs used in this Header.
 128       *
 129       * @return string[]
 130       */
 131      public function getIds()
 132      {
 133          return $this->_ids;
 134      }
 135  
 136      /**
 137       * Get the string value of the body in this Header.
 138       *
 139       * This is not necessarily RFC 2822 compliant since folding white space will
 140       * not be added at this stage (see {@see toString()} for that).
 141       *
 142       * @see toString()
 143       *
 144       * @return string
 145       *
 146       * @throws Swift_RfcComplianceException
 147       */
 148      public function getFieldBody()
 149      {
 150          if (!$this->getCachedValue()) {
 151              $angleAddrs = array();
 152  
 153              foreach ($this->_ids as $id) {
 154                  $angleAddrs[] = '<' . $id . '>';
 155              }
 156  
 157              $this->setCachedValue(implode(' ', $angleAddrs));
 158          }
 159  
 160          return $this->getCachedValue();
 161      }
 162  
 163      /**
 164       * Throws an Exception if the id passed does not comply with RFC 2822.
 165       *
 166       * @param string $id
 167       *
 168       * @throws Swift_RfcComplianceException
 169       */
 170      private function _assertValidId($id)
 171      {
 172          if (!preg_match(
 173              '/^' . $this->getGrammar()->getDefinition('id-left') . '@' .
 174              $this->getGrammar()->getDefinition('id-right') . '$/D',
 175              $id
 176              ))
 177          {
 178              throw new Swift_RfcComplianceException(
 179                  'Invalid ID given <' . $id . '>'
 180                  );
 181          }
 182      }
