  83       * Set a list of mailboxes to be shown in this Header.

  85       * The mailboxes can be a simple array of addresses, or an array of
  86       * key=>value pairs where (email => personalName).

  88       * <code>
  89       * <?php
  90       * //Sets two mailboxes in the Header, one with a personal name

 106      public function setNameAddresses($mailboxes)
 107      {
 108          $this->_mailboxes = $this->normalizeMailboxes((array) $mailboxes);
 109          $this->setCachedValue(null); //Clear any cached value

 245      /**
 246       * Normalizes a user-input list of mailboxes into consistent key=>value pairs.
 247       *
 248       * @param string[] $mailboxes
 249       *
 250       * @return string[]
 251       */
 252      protected function normalizeMailboxes(array $mailboxes)
 253      {
 254          $actualMailboxes = array();
 255  
 256          foreach ($mailboxes as $key => $value) {
 257              if (is_string($key)) { //key is email addr
 258                  $address = $key;
 259                  $name = $value;
 260              } else {
 261                  $address = $value;
 262                  $name = null;
 263              }
 264              $this->_assertValidAddress($address);
 265              $actualMailboxes[$address] = $name;
 266          }
 267  
 268          return $actualMailboxes;
 269      }

 272       * Produces a compliant, formatted display-name based on the string given.

 274       * @param string  $displayName as displayed
 275       * @param boolean $shorten     the first line to make remove for header name
 276       *
 277       * @return string
 278       */
 279      protected function createDisplayNameString($displayName, $shorten = false)
 280      {
 281          return $this->createPhrase($this, $displayName,
 282              $this->getCharset(), $this->getEncoder(), $shorten

 295      protected function createMailboxListString(array $mailboxes)
 296      {
 297          return implode(', ', $this->_createNameAddressStrings($mailboxes));
 298      }

 300      /**

 302       *
 303       * Commas and semicolons are used to separate
 304       * multiple addresses, and should therefore be encoded
 305       *
 306       * @param string $token
 307       *
 308       * @return boolean
 309       */
 310      protected function tokenNeedsEncoding($token)
 311      {
 312          return preg_match('/[,;]/', $token) || parent::tokenNeedsEncoding($token);
 313      }

 317      /**
 318       * Return an array of strings conforming the the name-addr spec of RFC 2822.
 319       *
 320       * @param string[] $mailboxes
 321       *
 322       * @return string[]
 323       */
 324      private function _createNameAddressStrings(array $mailboxes)
 325      {
 326          $strings = array();
 327  
 328          foreach ($mailboxes as $email => $name) {
 329              $mailboxStr = $email;
 330              if (!is_null($name)) {
 331                  $nameStr = $this->createDisplayNameString($name, empty($strings));
 332                  $mailboxStr = $nameStr . ' <' . $mailboxStr . '>';
 333              }
 334              $strings[] = $mailboxStr;
 335          }
 336  
 337          return $strings;
 338      }

 340      /**
 341       * Throws an Exception if the address passed does not comply with RFC 2822.
 342       *
 343       * @param string $address
 344       *
 345       * @throws Swift_RfcComplianceException If invalid.
 346       */
 347      private function _assertValidAddress($address)
 348      {
 349          if (!preg_match('/^' . $this->getGrammar()->getDefinition('addr-spec') . '$/D',
 350              $address))
 351          {
 352              throw new Swift_RfcComplianceException(
 353                  'Address in mailbox given [' . $address .
 354                  '] does not comply with RFC 2822, 3.6.2.'
 355                  );
 356          }
 357      }
