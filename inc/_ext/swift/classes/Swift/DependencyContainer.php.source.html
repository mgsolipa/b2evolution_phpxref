  11  /**
  12   * Dependency Injection container.
  13   *
  14   * @package Swift
  15   * @author  Chris Corbyn
  16   */
  17  class Swift_DependencyContainer
  18  {
  19      /** Constant for literal value types */
  20      const TYPE_VALUE = 0x0001;
  21  
  22      /** Constant for new instance types */
  23      const TYPE_INSTANCE = 0x0010;
  24  
  25      /** Constant for shared instance types */
  26      const TYPE_SHARED = 0x0100;
  27  
  28      /** Constant for aliases */
  29      const TYPE_ALIAS = 0x1000;
  30  
  31      /** Singleton instance */
  32      private static $_instance = null;
  33  
  34      /** The data container */
  35      private $_store = array();
  36  
  37      /** The current endpoint in the data container */
  38      private $_endPoint;
  39  
  40      /**
  41       * Constructor should not be used.
  42       *
  43       * Use {@link getInstance()} instead.
  44       */
  45      public function __construct() { }
  46  
  47      /**
  48       * Returns a singleton of the DependencyContainer.
  49       *
  50       * @return Swift_DependencyContainer
  51       */
  52      public static function getInstance()
  53      {
  54          if (!isset(self::$_instance)) {
  55              self::$_instance = new self();
  56          }
  57  
  58          return self::$_instance;
  59      }
  60  
  61      /**
  62       * List the names of all items stored in the Container.
  63       *
  64       * @return array
  65       */
  66      public function listItems()
  67      {
  68          return array_keys($this->_store);
  69      }
  70  
  71      /**
  72       * Test if an item is registered in this container with the given name.
  73       *
  74       * @see register()
  75       *
  76       * @param string $itemName
  77       *
  78       * @return boolean
  79       */
  80      public function has($itemName)
  81      {
  82          return array_key_exists($itemName, $this->_store)
  83              && isset($this->_store[$itemName]['lookupType']);
  84      }
  85  
  86      /**
  87       * Lookup the item with the given $itemName.
  88       *
  89       * @see register()
  90       *
  91       * @param string $itemName
  92       *
  93       * @return mixed
  94       *
  95       * @throws Swift_DependencyException If the dependency is not found
  96       */
  97      public function lookup($itemName)
  98      {
  99          if (!$this->has($itemName)) {
 100              throw new Swift_DependencyException(
 101                  'Cannot lookup dependency "' . $itemName . '" since it is not registered.'
 102                  );
 103          }
 104  
 105          switch ($this->_store[$itemName]['lookupType']) {
 106              case self::TYPE_ALIAS:
 107                  return $this->_createAlias($itemName);
 108              case self::TYPE_VALUE:
 109                  return $this->_getValue($itemName);
 110              case self::TYPE_INSTANCE:
 111                  return $this->_createNewInstance($itemName);
 112              case self::TYPE_SHARED:
 113                  return $this->_createSharedInstance($itemName);
 114          }
 115      }
 116  
 117      /**
 118       * Create an array of arguments passed to the constructor of $itemName.
 119       *
 120       * @param string $itemName
 121       *
 122       * @return array
 123       */
 124      public function createDependenciesFor($itemName)
 125      {
 126          $args = array();
 127          if (isset($this->_store[$itemName]['args'])) {
 128              $args = $this->_resolveArgs($this->_store[$itemName]['args']);
 129          }
 130  
 131          return $args;
 132      }
 133  
 134      /**
 135       * Register a new dependency with $itemName.
 136       *
 137       * This method returns the current DependencyContainer instance because it
 138       * requires the use of the fluid interface to set the specific details for the
 139       * dependency.
 140       * @see asNewInstanceOf(), asSharedInstanceOf(), asValue()
 141       *
 142       * @param string $itemName
 143       *
 144       * @return Swift_DependencyContainer
 145       */
 146      public function register($itemName)
 147      {
 148          $this->_store[$itemName] = array();
 149          $this->_endPoint =& $this->_store[$itemName];
 150  
 151          return $this;
 152      }
 153  
 154      /**
 155       * Specify the previously registered item as a literal value.
 156       *
 157       * {@link register()} must be called before this will work.
 158       *
 159       * @param mixed $value
 160       *
 161       * @return Swift_DependencyContainer
 162       */
 163      public function asValue($value)
 164      {
 165          $endPoint =& $this->_getEndPoint();
 166          $endPoint['lookupType'] = self::TYPE_VALUE;
 167          $endPoint['value'] = $value;
 168  
 169          return $this;
 170      }
 171  
 172      /**
 173       * Specify the previously registered item as an alias of another item.
 174       *
 175       * @param string $lookup
 176       *
 177       * @return Swift_DependencyContainer
 178       */
 179      public function asAliasOf($lookup)
 180      {
 181          $endPoint =& $this->_getEndPoint();
 182          $endPoint['lookupType'] = self::TYPE_ALIAS;
 183          $endPoint['ref'] = $lookup;
 184  
 185          return $this;
 186      }
 187  
 188      /**
 189       * Specify the previously registered item as a new instance of $className.
 190       *
 191       * {@link register()} must be called before this will work.
 192       * Any arguments can be set with {@link withDependencies()},
 193       * {@link addConstructorValue()} or {@link addConstructorLookup()}.
 194       *
 195       * @see withDependencies(), addConstructorValue(), addConstructorLookup()
 196       *
 197       * @param string $className
 198       *
 199       * @return Swift_DependencyContainer
 200       */
 201      public function asNewInstanceOf($className)
 202      {
 203          $endPoint =& $this->_getEndPoint();
 204          $endPoint['lookupType'] = self::TYPE_INSTANCE;
 205          $endPoint['className'] = $className;
 206  
 207          return $this;
 208      }
 209  
 210      /**
 211       * Specify the previously registered item as a shared instance of $className.
 212       *
 213       * {@link register()} must be called before this will work.
 214       *
 215       * @param string $className
 216       *
 217       * @return Swift_DependencyContainer
 218       */
 219      public function asSharedInstanceOf($className)
 220      {
 221          $endPoint =& $this->_getEndPoint();
 222          $endPoint['lookupType'] = self::TYPE_SHARED;
 223          $endPoint['className'] = $className;
 224  
 225          return $this;
 226      }
 227  
 228      /**
 229       * Specify a list of injected dependencies for the previously registered item.
 230       *
 231       * This method takes an array of lookup names.
 232       *
 233       * @see addConstructorValue(), addConstructorLookup()
 234       *
 235       * @param array $lookups
 236       *
 237       * @return Swift_DependencyContainer
 238       */
 239      public function withDependencies(array $lookups)
 240      {
 241          $endPoint =& $this->_getEndPoint();
 242          $endPoint['args'] = array();
 243          foreach ($lookups as $lookup) {
 244              $this->addConstructorLookup($lookup);
 245          }
 246  
 247          return $this;
 248      }
 249  
 250      /**
 251       * Specify a literal (non looked up) value for the constructor of the
 252       * previously registered item.
 253       *
 254       * @see withDependencies(), addConstructorLookup()
 255       *
 256       * @param mixed $value
 257       *
 258       * @return Swift_DependencyContainer
 259       */
 260      public function addConstructorValue($value)
 261      {
 262          $endPoint =& $this->_getEndPoint();
 263          if (!isset($endPoint['args'])) {
 264              $endPoint['args'] = array();
 265          }
 266          $endPoint['args'][] = array('type' => 'value', 'item' => $value);
 267  
 268          return $this;
 269      }
 270  
 271      /**
 272       * Specify a dependency lookup for the constructor of the previously
 273       * registered item.
 274       *
 275       * @see withDependencies(), addConstructorValue()
 276       *
 277       * @param string $lookup
 278       *
 279       * @return Swift_DependencyContainer
 280       */
 281      public function addConstructorLookup($lookup)
 282      {
 283          $endPoint =& $this->_getEndPoint();
 284          if (!isset($this->_endPoint['args'])) {
 285              $endPoint['args'] = array();
 286          }
 287          $endPoint['args'][] = array('type' => 'lookup', 'item' => $lookup);
 288  
 289          return $this;
 290      }

 292      // -- Private methods
 293  
 294      /** Get the literal value with $itemName */
 295      private function _getValue($itemName)
 296      {
 297          return $this->_store[$itemName]['value'];
 298      }
 299  
 300      /** Resolve an alias to another item */
 301      private function _createAlias($itemName)
 302      {
 303          return $this->lookup($this->_store[$itemName]['ref']);
 304      }
 305  
 306      /** Create a fresh instance of $itemName */
 307      private function _createNewInstance($itemName)
 308      {
 309          $reflector = new ReflectionClass($this->_store[$itemName]['className']);
 310          if ($reflector->getConstructor()) {
 311              return $reflector->newInstanceArgs(
 312                  $this->createDependenciesFor($itemName)
 313                  );
 314          } else {
 315              return $reflector->newInstance();
 316          }
 317      }
 318  
 319      /** Create and register a shared instance of $itemName */
 320      private function _createSharedInstance($itemName)
 321      {
 322          if (!isset($this->_store[$itemName]['instance'])) {
 323              $this->_store[$itemName]['instance'] = $this->_createNewInstance($itemName);
 324          }
 325  
 326          return $this->_store[$itemName]['instance'];
 327      }
 328  
 329      /** Get the current endpoint in the store */
 330      private function &_getEndPoint()
 331      {
 332          if (!isset($this->_endPoint)) {
 333              throw new BadMethodCallException(
 334                  'Component must first be registered by calling register()'
 335                  );
 336          }
 337  
 338          return $this->_endPoint;
 339      }
 340  
 341      /** Get an argument list with dependencies resolved */
 342      private function _resolveArgs(array $args)
 343      {
 344          $resolved = array();
 345          foreach ($args as $argDefinition) {
 346              switch ($argDefinition['type']) {
 347                  case 'lookup':
 348                      $resolved[] = $this->_lookupRecursive($argDefinition['item']);
 349                      break;
 350                  case 'value':
 351                      $resolved[] = $argDefinition['item'];
 352                      break;
 353              }
 354          }
 355  
 356          return $resolved;
 357      }
 358  
 359      /** Resolve a single dependency with an collections */
 360      private function _lookupRecursive($item)
 361      {
 362          if (is_array($item)) {
 363              $collection = array();
 364              foreach ($item as $k => $v) {
 365                  $collection[$k] = $this->_lookupRecursive($v);
 366              }
 367  
 368              return $collection;
 369          } else {
 370              return $this->lookup($item);
 371          }
 372      }
