 220       * Run a command against the buffer, expecting the given response codes.

 222       * If no response codes are given, the response will not be validated.

 231      public function executeCommand($command, $codes = array(), &$failures = null)
 232      {
 233          $failures = (array) $failures;
 234          $stopSignal = false;
 235          $response = null;
 236          foreach ($this->_getActiveHandlers() as $handler) {
 237              $response = $handler->onCommand(

 239                  );
 240              if ($stopSignal) {

 242              }
 243          }

 346      /** Determine ESMTP capabilities by function group */
 347      private function _getCapabilities($ehloResponse)
 348      {
 349          $capabilities = array();
 350          $ehloResponse = trim($ehloResponse);
 351          $lines = explode("\r\n", $ehloResponse);
 352          array_shift($lines);
 353          foreach ($lines as $line) {
 354              if (preg_match('/^[0-9]{3}[ -]([A-Z0-9-]+)((?:[ =].*)?)$/Di', $line, $matches)) {
 355                  $keyword = strtoupper($matches[1]);
 356                  $paramStr = strtoupper(ltrim($matches[2], ' ='));
 357                  $params = !empty($paramStr) ? explode(' ', $paramStr) : array();
 358                  $capabilities[$keyword] = $params;
 359              }
 360          }
 361  
 362          return $capabilities;
 363      }
