 102       * Start the SMTP connection.

 104      public function start()
 105      {
 106          if (!$this->_started) {
 107              if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this)) {
 108                  $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStarted');
 109                  if ($evt->bubbleCancelled()) {
 110                      return;
 111                  }
 112              }
 113  
 114              try {
 115                  $this->_buffer->initialize($this->_getBufferParams());
 116              } catch (Swift_TransportException $e) {
 117                  $this->_throwException($e);
 118              }
 119              $this->_readGreeting();
 120              $this->_doHeloCommand();
 121  
 122              if ($evt) {
 123                  $this->_eventDispatcher->dispatchEvent($evt, 'transportStarted');
 124              }
 125  
 126              $this->_started = true;
 127          }
 128      }

 140      /**
 141       * Send the given Message.
 142       *
 143       * Recipient/sender data will be retrieved from the Message API.
 144       * The return value is the number of recipients who were accepted for delivery.
 145       *
 146       * @param Swift_Mime_Message $message
 147       * @param string[]           $failedRecipients An array of failures by-reference
 148       *
 149       * @return int
 150       */
 151      public function send(Swift_Mime_Message $message, &$failedRecipients = null)
 152      {
 153          $sent = 0;
 154          $failedRecipients = (array) $failedRecipients;
 155  
 156          if ($evt = $this->_eventDispatcher->createSendEvent($this, $message)) {
 157              $this->_eventDispatcher->dispatchEvent($evt, 'beforeSendPerformed');
 158              if ($evt->bubbleCancelled()) {
 159                  return 0;
 160              }
 161          }
 162  
 163          if (!$reversePath = $this->_getReversePath($message)) {
 164              throw new Swift_TransportException(
 165                  'Cannot send message without a sender address'
 166                  );
 167          }
 168  
 169          $to = (array) $message->getTo();
 170          $cc = (array) $message->getCc();
 171          $bcc = (array) $message->getBcc();
 172  
 173          $message->setBcc(array());
 174  
 175          try {
 176              $sent += $this->_sendTo($message, $reversePath, $to, $failedRecipients);
 177              $sent += $this->_sendCc($message, $reversePath, $cc, $failedRecipients);
 178              $sent += $this->_sendBcc($message, $reversePath, $bcc, $failedRecipients);
 179          } catch (Exception $e) {
 180              $message->setBcc($bcc);
 181              throw $e;
 182          }
 183  
 184          $message->setBcc($bcc);
 185  
 186          if ($evt) {
 187              if ($sent == count($to) + count($cc) + count($bcc)) {
 188                  $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
 189              } elseif ($sent > 0) {
 190                  $evt->setResult(Swift_Events_SendEvent::RESULT_TENTATIVE);
 191              } else {
 192                  $evt->setResult(Swift_Events_SendEvent::RESULT_FAILED);
 193              }
 194              $evt->setFailedRecipients($failedRecipients);
 195              $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
 196          }
 197  
 198          $message->generateId(); //Make sure a new Message ID is used
 199  
 200          return $sent;
 201      }

 206      public function stop()
 207      {
 208          if ($this->_started) {
 209              if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this)) {
 210                  $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStopped');
 211                  if ($evt->bubbleCancelled()) {
 212                      return;
 213                  }
 214              }
 215  
 216              try {
 217                  $this->executeCommand("QUIT\r\n", array(221));
 218              } catch (Swift_TransportException $e) {}
 219  
 220              try {
 221                  $this->_buffer->terminate();
 222  
 223                  if ($evt) {
 224                      $this->_eventDispatcher->dispatchEvent($evt, 'transportStopped');
 225                  }
 226              } catch (Swift_TransportException $e) {
 227                  $this->_throwException($e);
 228              }
 229          }
 230          $this->_started = false;
 231      }

 261      /**
 262       * Run a command against the buffer, expecting the given response codes.
 263       *
 264       * If no response codes are given, the response will not be validated.
 265       * If codes are given, an exception will be thrown on an invalid response.
 266       *
 267       * @param string   $command
 268       * @param int[]    $codes
 269       * @param string[] $failures An array of failures by-reference
 270       *
 271       * @return string
 272       */
 273      public function executeCommand($command, $codes = array(), &$failures = null)
 274      {
 275          $failures = (array) $failures;
 276          $seq = $this->_buffer->write($command);
 277          $response = $this->_getFullResponse($seq);
 278          if ($evt = $this->_eventDispatcher->createCommandEvent($this, $command, $codes)) {
 279              $this->_eventDispatcher->dispatchEvent($evt, 'commandSent');
 280          }
 281          $this->_assertResponseCode($response, $codes);
 282  
 283          return $response;
 284      }

 338      /** Determine the best-use reverse path for this message */
 339      protected function _getReversePath(Swift_Mime_Message $message)
 340      {
 341          $return = $message->getReturnPath();
 342          $sender = $message->getSender();
 343          $from = $message->getFrom();
 344          $path = null;
 345          if (!empty($return)) {
 346              $path = $return;
 347          } elseif (!empty($sender)) {
 348              // Don't use array_keys
 349              reset($sender); // Reset Pointer to first pos
 350              $path = key($sender); // Get key
 351          } elseif (!empty($from)) {
 352              reset($from); // Reset Pointer to first pos
 353              $path = key($from); // Get key
 354          }
 355  
 356          return $path;
 357      }

 372      /** Throws an Exception if a response code is incorrect */
 373      protected function _assertResponseCode($response, $wanted)
 374      {
 375          list($code) = sscanf($response, '%3d');
 376          $valid = (empty($wanted) || in_array($code, $wanted));
 377  
 378          if ($evt = $this->_eventDispatcher->createResponseEvent($this, $response,
 379              $valid))
 380          {
 381              $this->_eventDispatcher->dispatchEvent($evt, 'responseReceived');
 382          }
 383  
 384          if (!$valid) {
 385              $this->_throwException(
 386                  new Swift_TransportException(
 387                      'Expected response code ' . implode('/', $wanted) . ' but got code ' .
 388                      '"' . $code . '", with message "' . $response . '"',
 389                      $code)
 390                  );
 391          }
 392      }

 395      protected function _getFullResponse($seq)
 396      {
 397          $response = '';
 398          try {
 399              do {
 400                  $line = $this->_buffer->readLine($seq);
 401                  $response .= $line;
 402              } while (null !== $line && false !== $line && ' ' != $line{3});
 403          } catch (Swift_TransportException $e) {
 404              $this->_throwException($e);

 412      /** Send an email to the given recipients from the given reverse path */
 413      private function _doMailTransaction($message, $reversePath, array $recipients, array &$failedRecipients)
 414      {
 415          $sent = 0;
 416          $this->_doMailFromCommand($reversePath);
 417          foreach ($recipients as $forwardPath) {
 418              try {
 419                  $this->_doRcptToCommand($forwardPath);
 420                  $sent++;
 421              } catch (Swift_TransportException $e) {
 422                  $failedRecipients[] = $forwardPath;
 423              }
 424          }
 425  
 426          if ($sent != 0) {
 427              $this->_doDataCommand();
 428              $this->_streamMessage($message);
 429          } else {
 430              $this->reset();
 431          }
 432  
 433          return $sent;
 434      }

 459      private function _sendBcc(Swift_Mime_Message $message, $reversePath, array $bcc, array &$failedRecipients)

 472      /** Try to determine the hostname of the server this is run on */
 473      private function _lookupHostname()
 474      {
 475          if (!empty($_SERVER['SERVER_NAME'])
 476              && $this->_isFqdn($_SERVER['SERVER_NAME']))
 477          {
 478              $this->_domain = $_SERVER['SERVER_NAME'];

 484      /** Determine is the $hostname is a fully-qualified name */
 485      private function _isFqdn($hostname)
 486      {
 487          //We could do a really thorough check, but there's really no point
 488          if (false !== $dotPos = strpos($hostname, '.')) {
 489              return ($dotPos > 0) && ($dotPos != strlen($hostname) - 1);
 490          } else {
 491              return false;
 492          }
