 104      {
 105          if (isset($this->_stream)) {
 106              switch ($this->_params['type']) {
 107                  case self::TYPE_PROCESS:
 108                      fclose($this->_in);

 123      /**
 124       * Set an array of string replacements which should be made on data written
 125       * to the buffer.
 126       *
 127       * This could replace LF with CRLF for example.

 131      public function setWriteTranslations(array $replacements)
 132      {
 133          foreach ($this->_translations as $search => $replace) {
 134              if (!isset($replacements[$search])) {
 135                  $this->removeFilter($search);
 136                  unset($this->_translations[$search]);
 137              }
 138          }
 139  
 140          foreach ($replacements as $search => $replace) {
 141              if (!isset($this->_translations[$search])) {
 142                  $this->addFilter(
 143                      $this->_replacementFactory->createFilter($search, $replace), $search
 144                      );
 145                  $this->_translations[$search] = true;
 146              }
 147          }
 148      }

 164          if (isset($this->_out) && !feof($this->_out)) {

 167                  $metas = stream_get_meta_data($this->_out);

 174                  }

 182       * Reads $length bytes from the stream into a string and moves the pointer
 183       * through the stream by $length.
 184       *
 185       * If less bytes exist than are requested the remaining bytes are given instead.

 240      /**
 241       * Establishes a connection to a remote server.
 242       */
 243      private function _establishSocketConnection()
 244      {
 245          $host = $this->_params['host'];
 246          if (!empty($this->_params['protocol'])) {
 247              $host = $this->_params['protocol'] . '://' . $host;
 248          }
 249          $timeout = 15;
 250          if (!empty($this->_params['timeout'])) {
 251              $timeout = $this->_params['timeout'];
 252          }
 253          $options = array();
 254          if (!empty($this->_params['sourceIp'])) {
 255              $options['socket']['bindto']=$this->_params['sourceIp'].':0';
 256          }
 257          $this->_stream = @stream_socket_client($host.':'.$this->_params['port'], $errno, $errstr, $timeout, STREAM_CLIENT_CONNECT, stream_context_create($options));
 258          if (false === $this->_stream) {
 259              throw new Swift_TransportException(
 260                  'Connection could not be established with host ' . $this->_params['host'] .
 261                  ' [' . $errstr . ' #' . $errno . ']'
 262                  );
 263          }
 264          if (!empty($this->_params['blocking'])) {
 265              stream_set_blocking($this->_stream, 1);
 266          } else {
 267              stream_set_blocking($this->_stream, 0);
 268          }
 269          stream_set_timeout($this->_stream, $timeout);
 270          $this->_in =& $this->_stream;
 271          $this->_out =& $this->_stream;
 272      }

 274      /**
 275       * Opens a process for input/output.
 276       */
 277      private function _establishProcessConnection()
 278      {
 279          $command = $this->_params['command'];
 280          $descriptorSpec = array(
 281              0 => array('pipe', 'r'),
 282              1 => array('pipe', 'w'),
 283              2 => array('pipe', 'w')
 284              );
 285          $this->_stream = proc_open($command, $descriptorSpec, $pipes);
 286          stream_set_blocking($pipes[2], 0);
 287          if ($err = stream_get_contents($pipes[2])) {
 288              throw new Swift_TransportException(
 289                  'Process could not be started [' . $err . ']'
 290                  );
 291          }
 292          $this->_in =& $pipes[0];
 293          $this->_out =& $pipes[1];
 294      }
 295  
 296      private function _getReadConnectionDescription()
 297      {
 298          switch ($this->_params['type']) {
 299              case self::TYPE_PROCESS:
 300                  return 'Process '.$this->_params['command'];
 301                  break;
 302  
 303              case self::TYPE_SOCKET:
 304              default:
 305                  $host = $this->_params['host'];
 306                  if (!empty($this->_params['protocol'])) {
 307                      $host = $this->_params['protocol'] . '://' . $host;
 308                  }
 309                  $host.=':'.$this->_params['port'];
 310  
 311                  return $host;
 312                  break;
 313          }
 314      }
