 165      /**
 166       * Runs immediately after a EHLO has been issued.
 167       *
 168       * @param Swift_Transport_SmtpAgent $agent to read/write
 169       */
 170      public function afterEhlo(Swift_Transport_SmtpAgent $agent)
 171      {
 172          if ($this->_username) {
 173              $count = 0;
 174              foreach ($this->_getAuthenticatorsForAgent() as $authenticator) {
 175                  if (in_array(strtolower($authenticator->getAuthKeyword()),
 176                      array_map('strtolower', $this->_esmtpParams)))
 177                  {
 178                      $count++;
 179                      if ($authenticator->authenticate($agent, $this->_username, $this->_password)) {
 180                          return;
 181                      }
 182                  }
 183              }
 184              throw new Swift_TransportException(
 185                  'Failed to authenticate on SMTP server with username "' .
 186                  $this->_username . '" using ' . $count . ' possible authenticators'
 187                  );
 188          }
 189      }

 254      protected function _getAuthenticatorsForAgent()
 255      {
 256          if (!$mode = strtolower($this->_auth_mode)) {

 261              if (strtolower($authenticator->getAuthKeyword()) == $mode) {
