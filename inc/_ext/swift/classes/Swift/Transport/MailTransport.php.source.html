  99      /**
 100       * Send the given Message.
 101       *
 102       * Recipient/sender data will be retrieved from the Message API.
 103       * The return value is the number of recipients who were accepted for delivery.
 104       *
 105       * @param Swift_Mime_Message $message
 106       * @param string[]           $failedRecipients An array of failures by-reference
 107       *
 108       * @return int
 109       */
 110      public function send(Swift_Mime_Message $message, &$failedRecipients = null)
 111      {
 112          $failedRecipients = (array) $failedRecipients;
 113  
 114          if ($evt = $this->_eventDispatcher->createSendEvent($this, $message)) {
 115              $this->_eventDispatcher->dispatchEvent($evt, 'beforeSendPerformed');
 116              if ($evt->bubbleCancelled()) {
 117                  return 0;
 118              }
 119          }
 120  
 121          $count = (
 122              count((array) $message->getTo())
 123              + count((array) $message->getCc())
 124              + count((array) $message->getBcc())
 125              );
 126  
 127          $toHeader = $message->getHeaders()->get('To');
 128          $subjectHeader = $message->getHeaders()->get('Subject');
 129  
 130          if (!$toHeader) {
 131              throw new Swift_TransportException(
 132                  'Cannot send message without a recipient'
 133                  );
 134          }
 135          $to = $toHeader->getFieldBody();
 136          $subject = $subjectHeader ? $subjectHeader->getFieldBody() : '';
 137  
 138          $reversePath = $this->_getReversePath($message);
 139  
 140          //Remove headers that would otherwise be duplicated
 141          $message->getHeaders()->remove('To');
 142          $message->getHeaders()->remove('Subject');
 143  
 144          $messageStr = $message->toString();
 145  
 146          $message->getHeaders()->set($toHeader);
 147          $message->getHeaders()->set($subjectHeader);
 148  
 149          //Separate headers from body
 150          if (false !== $endHeaders = strpos($messageStr, "\r\n\r\n")) {
 151              $headers = substr($messageStr, 0, $endHeaders) . "\r\n"; //Keep last EOL
 152              $body = substr($messageStr, $endHeaders + 4);
 153          } else {
 154              $headers = $messageStr . "\r\n";
 155              $body = '';
 156          }
 157  
 158          unset($messageStr);
 159  
 160          if ("\r\n" != PHP_EOL) {
 161              //Non-windows (not using SMTP)
 162              $headers = str_replace("\r\n", PHP_EOL, $headers);
 163              $body = str_replace("\r\n", PHP_EOL, $body);
 164          } else {
 165              //Windows, using SMTP
 166              $headers = str_replace("\r\n.", "\r\n..", $headers);
 167              $body = str_replace("\r\n.", "\r\n..", $body);
 168          }
 169  
 170          if ($this->_invoker->mail($to, $subject, $body, $headers,
 171              sprintf($this->_extraParams, $reversePath)))
 172          {
 173              if ($evt) {
 174                  $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
 175                  $evt->setFailedRecipients($failedRecipients);
 176                  $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
 177              }
 178          } else {
 179              $failedRecipients = array_merge(
 180                  $failedRecipients,
 181                  array_keys((array) $message->getTo()),
 182                  array_keys((array) $message->getCc()),
 183                  array_keys((array) $message->getBcc())
 184                  );
 185  
 186              if ($evt) {
 187                  $evt->setResult(Swift_Events_SendEvent::RESULT_FAILED);
 188                  $evt->setFailedRecipients($failedRecipients);
 189                  $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
 190              }
 191  
 192              $message->generateId();
 193  
 194              $count = 0;
 195          }
 196  
 197          return $count;
 198      }

 212      /** Determine the best-use reverse path for this message */
 213      private function _getReversePath(Swift_Mime_Message $message)
 214      {
 215          $return = $message->getReturnPath();
 216          $sender = $message->getSender();
 217          $from = $message->getFrom();
 218          $path = null;
 219          if (!empty($return)) {
 220              $path = $return;
 221          } elseif (!empty($sender)) {
 222              $keys = array_keys($sender);
 223              $path = array_shift($keys);
 224          } elseif (!empty($from)) {
 225              $keys = array_keys($from);
 226              $path = array_shift($keys);
 227          }
 228  
 229          return $path;
 230      }
