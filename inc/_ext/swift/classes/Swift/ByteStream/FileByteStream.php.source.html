 182      /** Streams in a readOnly stream ensuring copy if needed */
 183      private function _seekReadStreamToPosition($offset)
 184      {
 185          if ($this->_seekable===null) {
 186              $this->_getReadStreamSeekableStatus();
 187          }
 188          if ($this->_seekable === false) {
 189              $currentPos = ftell($this->_reader);

 192                  fread($this->_reader, $toDiscard);

 202      private function _copyReadStream()
 203      {
 204          if ($tmpFile = fopen('php://temp/maxmemory:4096', 'w+b')) {
 205              /* We have opened a php:// Stream Should work without problem */
 206          } elseif (function_exists('sys_get_temp_dir') && is_writable(sys_get_temp_dir()) && ($tmpFile = tmpfile())) {
 207              /* We have opened a tmpfile */
 208          } else {
 209              throw new Swift_IoException('Unable to copy the file to make it seekable, sys_temp_dir is not writable, php://memory not available');
 210          }
 211          $currentPos = ftell($this->_reader);
 212          fclose($this->_reader);
 213          $source = fopen($this->_path, 'rb');
 214          if (!$source) {
 215              throw new Swift_IoException('Unable to open file for copying [' . $this->_path . ']');
 216          }
 217          fseek($tmpFile, 0, SEEK_SET);
 218          while (!feof($source)) {
 219              fwrite($tmpFile, fread($source, 4096));
 220          }
 221          fseek($tmpFile, $currentPos, SEEK_SET);
 222          fclose($source);
 223          $this->_reader = $tmpFile;
