  69  {
  70      global $blog, $DB;
  71      $result_Array = array();
  72  
  73      $SQL = new SQL();
  74      $SQL->SELECT( 'cat_ID, cat_name' );
  75      $SQL->FROM( 'T_categories' );
  76      $SQL->WHERE( 'cat_blog_ID = '.$DB->quote( $blog ) );
  77      if( $parent_category_ID == -1 )
  78      {
  79          $SQL->WHERE_and( 'cat_parent_ID IS NULL' );
  80      }
  81      else
  82      {
  83          $SQL->WHERE( 'cat_parent_ID = '.$DB->quote( $parent_category_ID ) );
  84      }
  85      $SQL->ORDER_BY( 'cat_name' );
  86      $categories = $DB->get_results( $SQL->get() );
  87  
  88      if( ! empty( $categories ) )
  89      {
  90          foreach( $categories as $category )
  91          {
  92              $result_Array[] = array(
  93                      'value' => $category->cat_ID,
  94                      'label' => str_repeat( '&nbsp;&nbsp;&nbsp;', $level ).$category->cat_name
  95                  );
  96  
  97              $child_Categories_opts = fcpf_categories_select( $category->cat_ID, $level + 1 );
  98              if( $child_Categories_opts != '' )
  99              {
 100                  foreach( $child_Categories_opts as $cat )
 101                  {
 102                      $result_Array[] = $cat;
 103                  }
 104              }
 105          }
 106      }
