 290      /**
 291       * Load custom fields from request
 292       */
 293  	function load_custom_fields_from_Request()
 294      {
 295          global $Messages;
 296  
 297          // Initialize the arrays
 298          $this->update_custom_fields = array();
 299          $this->insert_custom_fields = array();
 300          $this->delete_custom_fields = trim( param( 'deleted_custom_double', 'string', '' ).','.param( 'deleted_custom_varchar', 'string', '' ), ',' );
 301          $this->delete_custom_fields = empty( $this->delete_custom_fields ) ? array() : explode( ',', $this->delete_custom_fields );
 302  
 303          // Field names array is used to check the diplicates
 304          $field_names = array();
 305  
 306          // Empty and Initialize the custom fields from POST data
 307          $this->custom_fields = array();
 308  
 309          foreach( array( 'double', 'varchar' ) as $type )
 310          {
 311              $empty_title_error = false; // use this to display empty title fields error message only ones
 312              $custom_field_count = param( 'count_custom_'.$type, 'integer', 0 ); // all custom fields count ( contains even deleted fields )
 313  
 314              for( $i = 1 ; $i <= $custom_field_count; $i++ )
 315              {
 316                  $custom_field_ID = param( 'custom_'.$type.'_ID'.$i, '/^[a-z0-9\-_]+$/', NULL );
 317                  if( empty( $custom_field_ID ) || in_array( $custom_field_ID, $this->delete_custom_fields ) )
 318                  { // This field was deleted, don't neeed to update
 319                      continue;
 320                  }
 321  
 322                  $custom_field_label = param( 'custom_'.$type.'_'.$i, 'string', NULL );
 323                  $custom_field_name = param( 'custom_'.$type.'_fname'.$i, '/^[a-z0-9\-_]+$/', NULL );
 324                  $custom_field_order = param( 'custom_'.$type.'_order'.$i, 'integer', NULL );
 325                  $custom_field_is_new = param( 'custom_'.$type.'_new'.$i, 'integer', 0 );
 326  
 327                  // Add each new/existing custom field in this array
 328                  // in order to see all them on the form when post type is not updated because some errors
 329                  $this->custom_fields[] = array(
 330                          'temp_i'  => $i, // Used only on submit form to know the number of the field on the form
 331                          'ID'      => $custom_field_ID,
 332                          'ityp_ID' => $this->ID,
 333                          'label'   => $custom_field_label,
 334                          'name'    => $custom_field_name,
 335                          'type'    => $type,
 336                          'order'   => $custom_field_order,
 337                      );
 338  
 339                  if( empty( $custom_field_label ) )
 340                  { // Field title can't be emtpy
 341                      if( ! $empty_title_error )
 342                      { // This message was not displayed yet
 343                          $Messages->add( T_('Custom field titles can\'t be empty!') );
 344                          $empty_title_error = true;
 345                      }
 346                  }
 347                  elseif( empty( $custom_field_name ) )
 348                  { // Field identical name can't be emtpy
 349                      $Messages->add( sprintf( T_('Please enter name for custom field "%s"'), $custom_field_label ) );
 350                  }
 351                  elseif( in_array( $custom_field_name, $field_names ) )
 352                  { // Field name must be identical
 353                      $Messages->add( sprintf( T_('The field name "%s" is not identical, please use another.'), $custom_field_name ) );
 354                  }
 355                  else
 356                  {
 357                      $field_names[] = $custom_field_name;
 358                  }
 359                  if( $custom_field_is_new )
 360                  { // Insert custom field
 361                      $this->insert_custom_fields[ $custom_field_ID ] = array(
 362                          'type'  => $type,
 363                          'name'  => $custom_field_name,
 364                          'label' => $custom_field_label,
 365                          'order' => $custom_field_order );
 366                  }
 367                  else
 368                  { // Update custom field
 369                      $this->update_custom_fields[ $custom_field_ID ] = array(
 370                          'type'  => $type,
 371                          'name'  => $custom_field_name,
 372                          'label' => $custom_field_label,
 373                          'order' => $custom_field_order );
 374                  }
 375              }
 376          }
 377      }

 474       * Update/Insert/Delete custom fields
 475       */
 476  	function dbsave_custom_fields()
 477      {
 478          global $DB;
 479  
 480          if( ! empty( $this->insert_custom_fields ) )
 481          { // Insert new custom fields
 482              $sql_data = array();
 483              foreach( $this->insert_custom_fields as $itcf_ID => $custom_field )
 484              {
 485                  $DB->query( 'INSERT INTO T_items__type_custom_field ( itcf_ityp_ID, itcf_label, itcf_name, itcf_type, itcf_order )
 486                      VALUES ( '.$DB->quote( $this->ID ).', '
 487                          .$DB->quote( $custom_field['label'] ).', '
 488                          .$DB->quote( $custom_field['name'] ).', '
 489                          .$DB->quote( $custom_field['type'] ).', '
 490                          .( empty( $custom_field['order'] ) ? 'NULL' : $DB->quote( $custom_field['order'] ) ).' )' );
 491              }
 492          }
 493  
 494          if( ! empty( $this->update_custom_fields ) )
 495          { // Update custom fields
 496              foreach( $this->update_custom_fields as $itcf_ID => $custom_field )
 497              {
 498                  $DB->query( 'UPDATE T_items__type_custom_field
 499                      SET
 500                          itcf_label = '.$DB->quote( $custom_field['label'] ).',
 501                          itcf_name = '.$DB->quote( $custom_field['name'] ).',
 502                          itcf_order = '.( empty( $custom_field['order'] ) ? 'NULL' : $DB->quote( $custom_field['order'] ) ).'
 503                      WHERE itcf_ityp_ID = '.$DB->quote( $this->ID ).'
 504                          AND itcf_ID = '.$DB->quote( $itcf_ID ).'
 505                          AND itcf_type = '.$DB->quote( $custom_field['type'] ) );
 506              }
 507          }
 508  
 509          if( ! empty( $this->delete_custom_fields ) )
 510          { // Delete custom fields
 511              $sql_data = array();
 512              foreach( $this->delete_custom_fields as $itcf_ID )
 513              {
 514                  $sql_data[] = '( itcf_ityp_ID = '.$DB->quote( $this->ID ).' AND itcf_ID = '.$DB->quote( $itcf_ID ).' )';
 515              }
 516              $DB->query( 'DELETE FROM T_items__type_custom_field
 517                  WHERE '.implode( ' OR ', $sql_data ) );
 518          }
 519      }

 611       *
 612       * @param string Type of custom field: 'all', 'varchar', 'double'
 613       * @param string Field name that is used as key of array: 'ID', 'ityp_ID', 'label', 'name', 'type', 'order'
 614       * @return array Custom fields
 615       */
 616  	function get_custom_fields( $type = 'all', $array_key = 'name' )
 617      {
 618          if( ! isset( $this->custom_fields ) )
 619          { // Initialize an array only first time
 620              if( empty( $this->ID ) )
 621              { // Set an empty array for new creating post type
 622                  $this->custom_fields = array();
 623              }
 624              else
 625              { // Get the custom fields from DB
 626                  global $DB;
 627                  $SQL = new SQL();
 628                  $SQL->SELECT( 'itcf_ID AS ID, itcf_ityp_ID AS ityp_ID, itcf_label AS label, itcf_name AS name, itcf_type AS type, itcf_order AS `order`' );
 629                  $SQL->FROM( 'T_items__type_custom_field' );
 630                  $SQL->WHERE( 'itcf_ityp_ID = '.$DB->quote( $this->ID ) );
