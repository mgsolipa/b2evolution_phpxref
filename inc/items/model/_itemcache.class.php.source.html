  67      {

  87      /**
  88       * Load items by the given categories or collection ID
  89       * After the Items are loaded create a map of loaded items by categories
  90       *
  91       * @param array of category ids
  92       * @param integer collection ID
  93       * @return boolean true if load items was required and it was loaded successfully, false otherwise
  94       */
  95  	function load_by_categories( $cat_array, $coll_ID )
  96      {
  97          global $DB, $posttypes_specialtypes;
  98  
  99          if( empty( $cat_array ) && empty( $coll_ID ) )
 100          { // Nothing to load
 101              return false;
 102          }
 103  
 104          // In case of an empty cat_array param, use categoriesfrom the given collection
 105          if( empty( $cat_array ) )
 106          { // Get all categories from the given subset
 107              $ChapterCache = & get_ChapterCache();
 108              $subset_chapters = $ChapterCache->get_chapters_by_subset( $coll_ID );
 109              $cat_array = array();
 110              foreach( $subset_chapters as $Chapter )
 111              {
 112                  $cat_array[] = $Chapter->ID;
 113              }
 114          }
 115  
 116          // Check which category is not loaded
 117          $not_loaded_cat_ids = array();
 118          foreach( $cat_array as $cat_ID )
 119          {
 120              if( ! isset( $this->items_by_cat_map[$cat_ID] ) )
 121              { // This category is not loaded
 122                  $not_loaded_cat_ids[] = $cat_ID;
 123                  // Initialize items_by_cat_map for this cat_ID
 124                  $this->items_by_cat_map[$cat_ID] = array( 'items' => array(), 'sorted' => false );
 125              }
 126          }
 127  
 128          if( empty( $not_loaded_cat_ids ) )
 129          { // Requested categories items are all loaded
 130              return false;
 131          }
 132  
 133          // Query to load all Items from the given categories
 134          $sql = 'SELECT postcat_cat_ID as cat_ID, postcat_post_ID as post_ID FROM T_postcats
 135                      WHERE postcat_cat_ID IN ( '.implode( ', ', $not_loaded_cat_ids ).' )
 136                      ORDER BY postcat_post_ID';
 137  
 138          $cat_posts = $DB->get_results( $sql, ARRAY_A, 'Get all category post ids pair by category' );
 139  
 140          // Initialize $Blog from coll_ID
 141          $BlogCache = & get_BlogCache();
 142          $Blog = $BlogCache->get_by_ID( $coll_ID );
 143  
 144          $visibility_statuses = is_admin_page() ? get_visibility_statuses( 'keys', array('trash') ) : get_inskin_statuses( $coll_ID, 'post' );
 145  
 146          // Create ItemQuery for loading visible items
 147          $ItemQuery = new ItemQuery( $this->dbtablename, $this->dbprefix, $this->dbIDname );
 148  
 149          // Set filters what to select
 150          $ItemQuery->SELECT( $this->dbtablename.'.*' );
 151          $ItemQuery->where_chapter2( $Blog, $not_loaded_cat_ids, "" );
 152          $ItemQuery->where_visibility( $visibility_statuses );
 153          $ItemQuery->where_datestart( NULL, NULL, NULL, NULL, $Blog->get_timestamp_min(), $Blog->get_timestamp_max() );
 154          $ItemQuery->where_types( '-'.implode(',',$posttypes_specialtypes) );
 155  
 156          // Clear previous items from the cache and load by the defined SQL
 157          $this->clear( true );
 158          $this->load_by_sql( $ItemQuery );
 159  
 160          foreach( $cat_posts as $row )
 161          { // Iterate through the post - cat pairs and fill the map
 162              if( empty( $this->cache[ $row['post_ID'] ] ) )
 163              { // The Item was not loaded because it does not correspond to the defined filters
 164                  continue;
 165              }
 166  
 167              // Add to the map
 168              $this->items_by_cat_map[$row['cat_ID']]['items'][] = $this->get_by_ID( $row['post_ID'] );
 169          }
 170      }

 173      /**
 174       * Get an object from cache by its urltitle
 175       *
 176       * Load into cache if necessary
 177       *
 178       * @param string stub of object to load
 179       * @param boolean false if you want to return false on error
 180       * @param boolean true if function should die on empty/null
 181       */
 182      function & get_by_urltitle( $req_urltitle, $halt_on_error = true, $halt_on_empty = true )
 183      {
 184          global $DB, $Debuglog;
 185  
 186          if( !isset( $this->urltitle_index[$req_urltitle] ) )
 187          { // not yet in cache:
 188          // Get from SlugCache
 189              $SlugCache = & get_SlugCache();
 190              $req_Slug =  $SlugCache->get_by_name( $req_urltitle, $halt_on_error, $halt_on_empty );
 191  
 192              if( $req_Slug && $req_Slug->get( 'type' ) == 'item' )
 193              {    // It is in SlugCache
 194                  $itm_ID = $req_Slug->get( 'itm_ID' );
 195                  if( $Item = $this->get_by_ID( $itm_ID, $halt_on_error, $halt_on_empty ) )
 196                  {
 197                      $this->urltitle_index[$req_urltitle] = $Item;
 198                  }
 199                  else
 200                  {    // Item does not exist
 201                      if( $halt_on_error ) debug_die( "Requested $this->objtype does not exist!" );
 202                      $this->urltitle_index[$req_urltitle] = false;
 203                  }
 204              }
 205              else
 206              {    // not in the slugCache
 207                  if( $halt_on_error ) debug_die( "Requested $this->objtype does not exist!" );
 208                  $this->urltitle_index[$req_urltitle] = false;
 209              }
 210          }
 211          else
 212          {
 213              $Debuglog->add( "Retrieving <strong>$this->objtype($req_urltitle)</strong> from cache" );
 214          }
 215  
 216          return $this->urltitle_index[$req_urltitle];
 217      }

 225  	function load_urltitle_array( $req_array )
 226      {
 227          global $DB, $Debuglog;
 228  
 229          $req_list = "'".implode( "','", $req_array)."'";
 230          $Debuglog->add( "Loading <strong>$this->objtype($req_list)</strong> into cache", 'dataobjects' );
 231          $sql = "SELECT * FROM $this->dbtablename WHERE post_urltitle IN ( $req_list )";
 232          $dbIDname = $this->dbIDname;
 233          $objtype = $this->objtype;
 234          foreach( $DB->get_results( $sql ) as $row )
 235          {
 236              $this->cache[ $row->$dbIDname ] = new $objtype( $row ); // COPY!
 237              // $obj = $this->cache[ $row->$dbIDname ];
 238              // $obj->disp( 'name' );
 239  
 240              // put into index:
 241              $this->urltitle_index[$row->post_urltitle] = & $this->cache[ $row->$dbIDname ];
 242  
 243              $Debuglog->add( "Cached <strong>$this->objtype($row->post_urltitle)</strong>" );
 244          }
 245  
 246          // Set cache from Slug table:
 247          foreach( $req_array as $urltitle )
 248          {
 249              if( !isset( $this->urltitle_index[$urltitle] ) )
 250              { // not yet in cache:
 251                  $SlugCache = & get_SlugCache();
 252                  if( $req_Slug = $SlugCache->get_by_name( $urltitle, false, false ) )
 253                  {
 254                      if( $req_Slug->get( 'type' ) == 'item' )
 255                      {    // Is item slug
 256                          if( $Item = $this->get_by_ID( $req_Slug->get( 'itm_ID' ), false ) )
 257                          {    // Set cache
 258                              $this->urltitle_index[$urltitle] = $Item;
 259                              $Debuglog->add( "Cached <strong>$this->objtype($urltitle)</strong>" );
 260                              continue;
 261                          }
 262                      }
 263                  }
 264                  // Set cache for non found objects:
 265                  $this->urltitle_index[$urltitle] = false; // Remember it doesn't exist in DB either
 266                  $Debuglog->add( "Cached <strong>$this->objtype($urltitle)</strong> as NON EXISTENT" );
 267              }
 268          }
 269      }
