 715                      $lastpostdate = mysql2date('U',$lastpostdate);

 718                      $this->ItemQuery->WHERE_and( $this->Cache->dbprefix.'datestart > \''. $otherdate.'\'' );
 719                  }
 720              }
 721              else
 722              { // We have a start date, we'll display x days starting from that point:
 723                  // $dstart_mysql has been calculated earlier
 724  
 725                  // TODO: this is redundant with previous dstart processing:
 726                  // Add trailing 0s: YYYYMMDDHHMMSS

 730                                                  .substr($dstart0,8,2).':'.substr($dstart0,10,2).':'.substr($dstart0,12,2);

 732                  // go forward x days
 733                  $enddate_ts = date('Y-m-d H:i:s', ($dstart_ts + ($this->limit * 86400)));
 734                  $this->ItemQuery->WHERE_and( $this->Cache->dbprefix.'datestart < \''. $enddate_ts.'\'' );
