 531  /**
 532   * Get a node from settings by path (e.g. "locales[0][questions]")
 533   *
 534   * @param Plugin
 535   * @param string Settings type ("Settings" or "UserSettings")
 536   * @param string The settings path, e.g. 'setting[0]foo[1]' or even 'setting[]'. (Is used as array internally for recursion.)
 537   * @return array Array(
 538   *          - 'set_name': setting name (string); key of the first level
 539   *          - 'set_node': selected setting node, may be NULL (by reference)
 540   *          - 'set_meta': meta info (from GetDefault[User]Settings()) for selected node (array)
 541   *          - 'set_parent': parent node (by reference)
 542   *          - 'set_key': key in parent node (by reference)
 543   *          - 'setting': whole settings (array)
 544   */
 545  function get_plugin_settings_node_by_path( & $Plugin, $set_type, $path, $create = false )
 546  {
 547      // Init:
 548      if( ! preg_match( '~^\w+(\[\w+\])+$~', $path ) )
 549      {
 550          debug_die( 'Invalid path param!' );
 551      }
 552  
 553      $path = preg_split( '~(\[|\]\[?)~', $path, -1 ); // split by "[" and "][", so we get an array with setting name and index alternating
 554      $foo = array_pop($path); // remove last one
 555      if( ! empty($foo) )
 556          debug_die('Assertion failed!');
 557  
 558      $set_name = $path[0];
 559  
 560      $setting = $Plugin->$set_type->get($set_name);  // $Plugin->Settings or $Plugin->UserSettings
 561      if( ! is_array($setting) )
 562      { // this may happen, if there was a non-array setting stored previously:
 563          // discard those!
 564          $setting = array();
 565      }
 566  
 567      // meta info for this setting:
 568      $method = 'GetDefault'.$set_type; // GetDefaultSettings or GetDefaultUserSettings
 569      $defaults = $Plugin->$method( $tmp_params = array('for_editing'=>true) );
 570      if( ! isset($defaults[ $set_name ]) )
 571      {
 572          //debug_die( 'Invalid setting ('.$set_name.') - no meta data!' );
 573          return false;
 574      }
 575  
 576      $found_node = & $setting;
 577      $defaults_node = & $defaults;
 578      $set_meta = $defaults[$set_name];
 579      $set_parent = NULL;
 580      $set_key = NULL;
 581  
 582      $count = 0;
 583      while( count($path) )
 584      {
 585          $count++;
 586  
 587          $loop_name = array_shift($path);
 588  
 589          if( $count > 1 )
 590          {
 591              $set_parent = & $found_node[$loop_name];
 592              $set_key = NULL;
 593              $defaults_node = & $defaults_node['entries'][$loop_name];
 594              $found_node = & $found_node[$loop_name];
 595          }
 596          else
 597          {
 598              $defaults_node = & $defaults_node[$loop_name];
 599              $set_parent = & $setting;
 600          }
 601  
 602          if( count($path) )
 603          { // has an index => array
 604              $loop_index = array_shift($path);
 605  
 606              #$set_parent = & $set_parent[$loop_name];
 607              $set_key = $loop_index;
 608  
 609              if( $set_key === '' )
 610              { // []-syntax: append entry
 611                  if( $create && ! count($path) )
 612                  { // only create, if at the end
 613                      $found_node[] = array();
 614                  }
 615                  $found_node = & $found_node[ array_pop(array_keys($found_node)) ];
 616              }
 617              else
 618              { // specific key:
 619                  if( ! isset($found_node[$loop_index]) )
 620                  {

 783   * Validates settings according to their meta info recursively.
 784   *
 785   * @todo Init "checkbox" values in "array" type settings (they do not get send) (dh)
 786   * @param string Param name
 787   * @param array Meta info
 788   * @return boolean
 789   */
 790  function autoform_validate_param_value( $param_name, $value, $meta )
 791  {
 792      global $Messages;
 793  
 794      if( is_array($value) && isset($meta['entries']) )
 795      {
 796          $r = true;
 797          if(isset($meta['key']))
 798          { // validate keys:
 799              foreach( array_keys($value) as $k )
 800              {
 801                  if( ! autoform_validate_param_value($param_name.'['.$k.'][__key__]', $k, $meta['key']) )
 802                  {
 803                      $r = false;
 804                  }
 805              }
 806          }
 807  
 808          // Check max_count/min_count
 809          // dh> TODO: find a way to link it to the form's fieldset (and add an "error" class to it)
 810          if( isset($meta['max_count']) && count($value) > $meta['max_count'] )
 811          {
 812              $r = false;
 813              $label = isset($meta['label']) ? $meta['label'] : $param_name;
 814              $Messages->add( sprintf( T_('Too many entries in the "%s" set. It must have %d at most.'), $label, $meta['max_count'] ), 'error' );
 815          }
 816          elseif( isset($meta['min_count']) && count($value) < $meta['min_count'] )
 817          {
 818              $r = false;
 819              $label = isset($meta['label']) ? $meta['label'] : $param_name;
 820              $Messages->add( sprintf( T_('Too few entries in the "%s" set. It must have %d at least.'), $label, $meta['min_count'] ), 'error' );
 821          }
 822  
 823          foreach( $meta['entries'] as $mk => $mv )
 824          {
 825              foreach( $value as $vk => $vv )
 826              {
 827                  if( ! isset($vv[$mk]) )
 828                      continue;
 829  
 830                  if( ! autoform_validate_param_value($param_name.'['.$vk.']['.$mk.']', $vv[$mk], $mv) )
 831                  {
 832                      $r = false;
 833                  }
 834              }
 835          }
 836          return $r;
 837      }
 838  
 839  
 840      if( isset( $meta['type'] ) )
 841      {
 842          if( isset( $meta['allow_empty'] ) && $meta['allow_empty'] && empty( $value ) )
 843          { // Allow an empty value
 844              return true;
 845          }
 846  
 847          switch( $meta['type'] )
 848          {
 849              case 'integer':
 850                  if( ! preg_match( '~^[-+]?\d+$~', $value ) )
 851                  {
 852                      param_error( $param_name, sprintf( T_('The value for &laquo;%s&raquo; must be numeric.'), $meta['label'] ), T_('The value must be numeric.') );
 853                      return false;
 854                  }
 855                  break;
 856  
 857              case 'float':
 858                  if( ! preg_match( '~^[-+]?\d+(\.\d+)?$~', $value ) )
 859                  {
 860                      param_error( $param_name, sprintf( T_('The value for &laquo;%s&raquo; must be numeric.'), $meta['label'] ), T_('The value must be numeric.') );
 861                      return false;
 862                  }
 863                  break;
 864  
 865              case 'radio':
 866                  $check_value = false;
 867                  foreach($meta['options'] as $arr)
 868                  {
 869                      if( ! is_array($arr) )
 870                      {
 871                          param_error( $param_name, sprintf( T_('Invalid option &laquo;%s&raquo;.'), $arr ) );
 872                          return false;
 873                      }
 874                      if( $value == $arr[0] )
 875                      {
 876                          $check_value = true;
 877                          break;
 878                      }
 879                  }
 880                  if ( ! $check_value )
 881                  {
 882                      param_error( $param_name, sprintf( T_('Invalid option &laquo;%s&raquo;.'), $value ) );
 883                      return false;
 884                  }
 885                  break;
 886  
 887              case 'select':
 888                  $check_options = $value;
 889                  if( ! is_array($check_options) )
 890                  { // no "multiple" select:
 891                      $check_options = array($check_options);
 892                  }
 893  
 894                  foreach($check_options as $v)
 895                  {
 896                      if( ! in_array( $v, array_keys($meta['options']) ) )
 897                      {
 898                          param_error( $param_name, sprintf( T_('Invalid option &laquo;%s&raquo;.'), $v ) );
 899                          return false;
 900                      }
 901                  }
 902                  break;
 903  
 904              case 'select_blog':
 905              case 'select_group':
 906              case 'select_user':
 907                  if( is_array($value) && empty($value) // empty "multiple" select
 908                      || ( ! is_array($value) && ! strlen($value) ) )
 909                  {
 910                      if( empty($meta['allow_none']) )
 911                      { // empty is not ok
 912                          param_error( $param_name, sprintf( T_('Invalid option &laquo;%s&raquo;.'), $value ) );
 913                          return false;
 914                      }
 915                  }
 916                  else
 917                  { // Try retrieving the value from the corresponding Cache:
 918                      switch( $meta['type'] )
 919                      {
 920                          case 'select_blog':
 921                              $Cache = & get_BlogCache();
 922                              break;
 923  
 924                          case 'select_group':
 925                              $Cache = & get_GroupCache();
 926                              break;
 927  
 928                          case 'select_user':
 929                              $Cache = & get_UserCache();
 930                              break;
 931                      }
 932  
 933                      $check_options = $value;
 934                      if( ! is_array($check_options) )
 935                      { // no "multiple" select:
 936                          $check_options = array($check_options);
 937                      }
 938  
 939                      foreach($check_options as $v)
 940                      {
 941                          if( empty($v) && ! empty($meta['allow_none']) )
 942                          { // empty is ok:
 943                              continue;
 944                          }
 945                          if( ! $Cache->get_by_ID($v, false, false) )
 946                          {
 947                              param_error( $param_name, sprintf( T_('Invalid option &laquo;%s&raquo;.'), $v ) );
 948                              return false;
 949                          }
 950                      }
 951                  }
 952                  break;
 953          }
 954      }
 955  
 956      // Check maxlength:
 957      if( isset($meta['maxlength']) )
 958      {
 959          if( utf8_strlen($value) > $meta['maxlength'] )
 960          {
 961              param_error( $param_name, sprintf( T_('The value is too long.'), $value ) );
 962          }
 963      }
 964  
 965      // Check valid pattern:
 966      if( isset($meta['valid_pattern']) )
 967      {
 968          $param_pattern = is_array($meta['valid_pattern']) ? $meta['valid_pattern']['pattern'] : $meta['valid_pattern'];
 969          if( ! preg_match( $param_pattern, $value ) )
 970          {
 971              $param_error = is_array($meta['valid_pattern']) ? $meta['valid_pattern']['error'] : sprintf(T_('The value is invalid. It must match the regular expression &laquo;%s&raquo;.'), $param_pattern);
 972              param_error( $param_name, $param_error );

 974          }
 975      }
 976  
 977      // Check valid range:

 979      {
 980          // Transform numeric indexes into associative keys:
 981          if( ! isset($meta['valid_range']['min'], $meta['valid_range']['max'])
 982              && isset($meta['valid_range'][0], $meta['valid_range'][1]) )
 983          {
 984              $meta['valid_range']['min'] = $meta['valid_range'][0];
 985              $meta['valid_range']['max'] = $meta['valid_range'][1];
 986          }

 988          {

 990          }
 991  
 992          if( (isset($meta['valid_range']['min']) && $value < $meta['valid_range']['min'])
 993                  || (isset($meta['valid_range']['max']) && $value > $meta['valid_range']['max']) )
 994          {
 995              if( isset($meta['valid_range']['error']) )
 996              {
 997                  $param_error = $meta['valid_range']['error'];
 998              }
 999              else
1000              {
1001                  if( isset($meta['valid_range']['min']) && isset($meta['valid_range']['max']) )
1002                  {
1003                      $param_error = sprintf(T_('The value is invalid. It must be in the range from %s to %s.'), $meta['valid_range']['min'], $meta['valid_range']['max']);
1004                  }
1005                  elseif( isset($meta['valid_range']['max']) )
1006                  {
1007                      $param_error = sprintf(T_('The value is invalid. It must be smaller than or equal to %s.'), $meta['valid_range']['max']);
1008                  }
1009                  else
1010                  {
1011                      $param_error = sprintf(T_('The value is invalid. It must be greater than or equal to %s.'), $meta['valid_range']['min']);
1012                  }
1013              }
1014  
1015              param_error( $param_name, $param_error );
1016              return false;
1017          }
1018      }
1019  
1020      return true;

1024  /**
1025   * This handles the special "__key__" index in all array type values
1026   * in the given array. It makes sure, that "__key__" is unique and
1027   * replaces the original key of the value with it.
1028   * @param array (by reference)
1029   */
1030  function handle_array_keys_in_plugin_settings( & $a )
1031  {
1032      if( ! is_array($a) )
1033      {
1034          return;
1035      }
1036  
1037      $new_arr = array(); // use a new array to maintain order, also for "numeric" keys
1038  
1039      foreach( array_keys($a) as $k )
1040      {
1041          $v = & $a[$k];
1042  
1043          if( is_array($v) && isset($v['__key__']) )
1044          {
1045              if( $k != $v['__key__'] )
1046              {
1047                  $k = $v['__key__'];
1048                  if( ! strlen($k) || isset($a[ $k ]) )
1049                  { // key already exists (or is empty):
1050                      $c = 1;
1051  
1052                      while( isset($a[ $k.'_'.$c ]) )
1053                      {
1054                          $c++;
1055                      }
1056                      $k = $k.'_'.$c;
1057                  }
1058              }
1059              unset($v['__key__']);
1060  
1061              $new_arr[$k] = $v;
1062          }
1063          else
1064          {
1065              $new_arr[$k] = $v;
1066          }
1067  
1068          // Recurse:
1069          foreach( array_keys($v) as $rk )
1070          {
1071              if( is_array($v[$rk]) )
1072              {
1073                  handle_array_keys_in_plugin_settings($v[$rk]);
1074              }
1075          }
1076      }
1077      $a = $new_arr;
1078  }

1092

1120                  return false;
1121              }
1122              elseif( $install_db_deltas_confirm_md5 == md5( implode('', $install_db_deltas) ) || $force_install_db_deltas )
1123              { // Confirmed in first step:
1124                  foreach( $install_db_deltas as $query )
1125                  {
1126                      $DB->query( $query );
1127                  }
1128  
1129                  $Messages->add( T_('The database has been updated.'), 'success' );

1133                  $Messages->add( T_('The DB schema has been changed since confirmation.'), 'error' );
1134

1136                  return false;
