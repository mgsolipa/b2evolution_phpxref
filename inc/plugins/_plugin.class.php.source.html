 330  	function load_template()
 331      {
 332          $params = array(
 333                  'toolbar_before'       => '<div class="edit_toolbar $toolbar_class$">',
 334                  'toolbar_after'        => '</div>',
 335                  'toolbar_title_before' => '',
 336                  'toolbar_title_after'  => '',
 337                  'toolbar_group_before' => '',
 338                  'toolbar_group_after'  => ' ',
 339                  'toolbar_button_class' => 'quicktags',
 340              );
 341  
 342          if( is_admin_page() )
 343          { // Get plugin template from backoffice skin
 344              if( is_logged_in() )
 345              {
 346                  global $current_User, $UserSettings, $adminskins_path;
 347                  $admin_skin = $UserSettings->get( 'admin_skin', $current_User->ID );
 348                  require_once $adminskins_path.$admin_skin.'/_adminUI.class.php';
 349                  $AdminUI = new AdminUI();
 350                  $skin_template = $AdminUI->get_template( 'plugin_template' );
 351              }
 352          }
 353          else
 354          { // Get plugin template from frontoffice skin
 355              global $Blog;
 356              if( ! empty( $Blog ) )
 357              {
 358                  $skin_ID = $Blog->get_skin_ID();
 359                  $SkinCache = & get_SkinCache();
 360                  if( $Skin = & $SkinCache->get_by_ID( $skin_ID, false, false ) )
 361                  {
 362                      $skin_template = $Skin->get_template( 'plugin_template' );
 363                  }
 364              }
 365          }
 366  
 367          if( empty( $skin_template ) )
 368          { // Use default template params when they are not set by skin
 369              $this->template = $params;
 370          }
 371          else
 372          { // Merge default params with current skin params
 373              $this->template = array_merge( $params, $skin_template );
 374          }
 375      }

 378      /**
 379       * Get param of the current loaded template
 380       *
 381       * @param string Param name
 382       * @param array What should be replaced in the param (e.g. array( '$toolbar_class$' => 'quicktags_toolbar' ) )
 383       */
 384  	function get_template( $param_name, $replaces = NULL )
 385      {
 386          if( is_null( $this->template ) )
 387          { // Load template only first time
 388              $this->load_template();
 389          }
 390  
 391          if( ! isset( $this->template[ $param_name ] ) )
 392          { // Return empty string if no defined param
 393              return '';
 394          }
 395  
 396          $param_value = $this->template[ $param_name ];
 397  
 398          if( is_array( $replaces ) )
 399          {
 400              foreach( $replaces as $search => $replace )
 401              {
 402                  $param_value = str_replace( $search, $replace, $param_value );
 403              }
 404          }
 405  
 406          return $param_value;
 407      }
