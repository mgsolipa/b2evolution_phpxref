  82  function plugin_results_td_status( $plug_status, $plug_ID )
  83  {
  84      global $admin_Plugins, $current_User, $admin_url;
  85  
  86      $perm_edit = $current_User->check_perm( 'options', 'edit', false );
  87  
  88      if( $plug_status == 'enabled' )
  89      { // Enabled
  90          if( $perm_edit )
  91          { // URL to disable the plugin
  92              $plugin_status_url = $admin_url.'?ctrl=plugins&amp;action=disable_plugin&amp;plugin_ID='.$plug_ID.'&amp;'.url_crumb( 'plugin' );
  93          }
  94          $plugin_status_icon = get_icon( 'bullet_green', 'imgtag', array( 'title' => T_('The plugin is enabled.') ) );
  95      }
  96      elseif( $plug_status == 'broken' )
  97      { // Broken
  98          $plugin_status_icon = get_icon( 'warning', 'imgtag', array(
  99              'title' => T_('The plugin is broken.')
 100                  .// Display load error from Plugins::register() (if any):
 101                  ( isset( $admin_Plugins->plugin_errors[ $plug_ID ] )
 102                      && ! empty( $admin_Plugins->plugin_errors[ $plug_ID ]['register'] )
 103                      ? ' '.$admin_Plugins->plugin_errors[ $plug_ID ]['register']
 104                      : '' )
 105              ) );
 106      }
 107      elseif( $plug_status == 'needs_config' )
 108      { // Needs config
 109          if( $perm_edit )
 110          { // URL to edit plugin settings
 111              $plugin_status_url = $admin_url.'?ctrl=plugins&amp;action=edit_settings&amp;plugin_ID='.$plug_ID;
 112          }
 113          $plugin_status_icon = get_icon( 'question', 'imgtag', array( 'title' => T_('The plugin is not installed completely.') ) );
 114      }
 115      else
 116      { // Disabled
 117          if( $perm_edit )
 118          { // URL to enable the plugin
 119              $plugin_status_url = $admin_url.'?ctrl=plugins&amp;action=enable_plugin&amp;plugin_ID='.$plug_ID.'&amp;'.url_crumb( 'plugin' );
 120          }
 121          $plugin_status_icon = get_icon( 'bullet_empty_grey', 'imgtag', array( 'title' => T_('The plugin is disabled.') ) );
 122      }
 123  
 124      if( isset( $plugin_status_url ) )
 125      {
 126          return '<a href="'.$plugin_status_url.'">'.$plugin_status_icon.'</a>';
 127      }
 128      else
 129      {
 130          return $plugin_status_icon;
 131      }
 132  }

 136          'order' => 'plug_status',
 137          'td' => '%plugin_results_td_status( #plug_status#, #plug_ID# )%',
