 123      /**
 124       * Check a permission for the group. ( see 'group_func' in get_available_group_permissions() function )
 125       *
 126       * @param string Requested permission level
 127       * @param string Permission value
 128       * @param mixed Permission target (blog ID, array of cat IDs...)
 129       * @return boolean True on success (permission is granted), false if permission is not granted
 130       */
 131  	function check_maintenance_group_perm( $permlevel, $permvalue, $permtarget )
 132      {
 133          $perm = false;
 134          switch ( $permvalue )
 135          {
 136              case 'upgrade':
 137                  // Users can create backups & upgrade the app.
 138                  if( $permlevel == 'upgrade' )
 139                  { // User can ask for delete perm...
 140                      $perm = true;
 141                      break;
 142                  }
 143  
 144              case 'backup':
 145                  //  Users can create backups
 146                  if( $permlevel == 'backup' )
 147                  {
 148                      $perm = true;
 149                      break;
 150                  }
 151          }
 152  
 153          return $perm;
 154      }

 162  	function build_menu_3()

 165

 167          {
 168              return;
 169          }

 171          if( $current_User->check_perm( 'perm_maintenance', 'backup' ) )
 172          {
 173              // Display Backup tab in System -> Maintenance menu
 174              $AdminUI->add_menu_entries( array( 'options', 'misc' ), array(
 175                                      'backup' => array(
 176                                      'text' => T_('Backup'),
 177                                      'href' => '?ctrl=backup'    ),
 178                              ) );
 179          }
 180  
 181          if( $current_User->check_perm( 'perm_maintenance', 'upgrade' ) )
 182          {
 183              // Display Updates tab in System -> Maintenance menu
 184              $AdminUI->add_menu_entries( array( 'options', 'misc' ), array(
 185                                      'upgrade' => array(
 186                                      'text' => T_('Auto Upgrade'),

 189              $AdminUI->add_menu_entries( array( 'options', 'misc' ), array(
