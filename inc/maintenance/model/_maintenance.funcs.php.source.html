  12  function get_upgrade_folder_path( $version_folder_name )
  13  {
  14      global $upgrade_path;
  15  
  16      if( empty( $version_folder_name ) || ! file_exists( $upgrade_path.$version_folder_name ) )
  17      { // Don't allow an invalid upgrade folder
  18          debug_die( 'Invalid name of upgrade folder' );
  19      }
  20  
  21      // Use a root path by default
  22      $upgrade_folder_path = $upgrade_path.$version_folder_name;
  23  
  24      if( file_exists( $upgrade_folder_path.'/b2evolution/blogs' ) )
  25      { // Use 'b2evolution/blogs' folder
  26          $upgrade_folder_path .= '/b2evolution/blogs';
  27      }
  28      else if( file_exists( $upgrade_folder_path.'/b2evolution/site' ) )
  29      { // Use 'b2evolution/site' folder
  30          $upgrade_folder_path .= '/b2evolution/site';
  31      }
  32      else if( file_exists( $upgrade_folder_path.'/b2evolution' ) )
  33      { // Use 'b2evolution' folder
  34          $upgrade_folder_path .= '/b2evolution';
  35      }

  95  /**
  96   * Enable/disable maintenance mode
  97   *
  98   * @param boolean true if maintenance mode need to be enabled
  99   * @param string Mode: 'all', 'install', 'upgrade'
 100   * @param string maintenance mode message
 101   * @param boolean TRUE to don't print out a message status
 102   */
 103  function switch_maintenance_mode( $enable, $mode = 'all', $msg = '', $silent = false )
 104  {
 105      global $conf_path;
 106  
 107      switch( $mode )
 108      {
 109          case 'install':
 110              // Use maintenance mode except of install actions
 111              $maintenance_mode_file = 'imaintenance.html';
 112              break;
 113  
 114          case 'upgrade':
 115              // Use maintenance mode except of upgrade actions
 116              $maintenance_mode_file = 'umaintenance.html';
 117              break;
 118  
 119          default:
 120              // Use full maintenance mode
 121              $maintenance_mode_file = 'maintenance.html';
 122              break;
 123      }
 124  
 125      if( $enable )
 126      {    // Create maintenance file
 127          echo '<p>'.T_('Switching to maintenance mode...');
 128          evo_flush();
 129  
 130          $content = '<html>
 131  <head>
 132      <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
 133      <title>Site temporarily down for maintenance.</title>
 134  </head>
 135  <body>
 136  <h1>503 Service Unavailable</h1>
 137  <p>'.$msg.'</p>
 138  <hr />
 139  <p>Site administrators: please view the source of this page for details.</p>
 140  <!--
 141  If you need to manually put b2evolution out of maintenance mode, delete or rename the file /conf/maintenance.html
 142  -->
 143  </body>
 144  </html>';
 145  
 146          if( save_to_file( $content, $conf_path.$maintenance_mode_file, 'w+' ) )
 147          { // Maintenance file has been created
 148              echo ' OK.</p>';
 149          }
 150          else
 151          { // Maintenance file has not been created
 152              echo '</p><p style="color:red">'.sprintf( T_( 'Unable to switch maintenance mode. Maintenance file can\'t be created: &laquo;%s&raquo;' ), $maintenance_mode_file ).'</p>';
 153              evo_flush();
 154  
 155              return false;
 156          }
 157      }
 158      else
 159      {    // Delete maintenance file
 160          if( ! $silent )
 161          {
 162              echo '<p>'.T_('Switching out of maintenance mode...');
 163          }
 164          if( is_writable( $conf_path.$maintenance_mode_file ) )
 165          { // Delete a maintenance file if it exists and writable
 166              if( @unlink( $conf_path.$maintenance_mode_file ) )
 167              { // Unlink was successful
 168                  if( ! $silent )
 169                  { // Dispaly OK message
 170                      echo ' OK.</p>';
 171                  }
 172              }
 173              else
 174              { // Unlink failed
 175                  echo '</p><p style="color:red">'.sprintf( T_( 'Unable to delete a maintenance file: &laquo;%s&raquo;' ), $maintenance_mode_file ).'</p>';
 176              }
 177          }
 178          evo_flush();
 179      }
 180  
 181      return true;
 182  }

 262   *

 264   * @param string destination directory path
 265   * @param boolean true if create destination directory
 266   * @param string Zip file name
 267   * @return boolean results

 269  function unpack_archive( $src_file, $dest_dir, $mk_dest_dir = false, $src_file_name = '' )
 270  {
 271      if( !file_exists( $dest_dir ) )
 272      { // We can create directory
 273          if ( ! mkdir_r( $dest_dir ) )
 274          {
 275              echo '<p style="color:red">'.sprintf( T_( 'Unable to create &laquo;%s&raquo; directory to extract files from ZIP archive.' ), $dest_dir ).'</p>';
 276              evo_flush();
 277  
 278              return false;
 279          }
 280      }
 281  
 282      if( function_exists( 'gzopen' ) )
 283      { // Unpack using 'zlib' extension and PclZip wrapper
 284  
 285          // Load PclZip class (PHP4):
 286          load_class( '_ext/pclzip/pclzip.lib.php', 'PclZip' );
 287  
 288          $PclZip = new PclZip( $src_file );
 289          if( $PclZip->extract( PCLZIP_OPT_PATH, $dest_dir ) == 0 )
 290          {
 291              if( empty( $src_file_name ) )
 292              { // Set zip file name
 293                  $src_file_name = $src_file;
 294              }
 295              echo '<p style="color:red">'
 296                  .sprintf( T_( 'Unable to decompress &laquo;%s&raquo; ZIP archive.' ), $src_file_name ).'<br />'
 297                  .sprintf( T_( 'Error: %s' ), $PclZip->errorInfo( true ) )
 298                  .'</p>';
 299              evo_flush();
 300  
 301              return false;
 302          }
 303      }
 304      else
 305      {
 306          debug_die( 'Unable to decompress the file because there is no \'zip\' or \'zlib\' extension installed in your PHP!' );
 307      }
 308  
 309      return true;
 310  }

 313  /**
 314   * Verify that destination files can be overwritten
 315   *
 316   * @param string source directory
 317   * @param string destination directory
 318   * @param string action name
 319   * @param boolean overwrite
 320   * @param array read only file list
 321   */
 322  function verify_overwrite( $src, $dest, $action = '', $overwrite = true, & $read_only_list )
 323  {
 324      global $basepath;
 325  
 326      /**
 327       * Result of this function is FALSE when some error was detected
 328       * @var boolean
 329       */
 330      $result = true;
 331  
 332      $dir = opendir( $src );
 333  
 334      if( $dir === false )
 335      { // $dir is not a valid directory or it can not be opened due to permission restrictions
 336          echo '<div class="red">The &laquo;'.htmlspecialchars( $src ).'&raquo; is not a valid direcotry or the directory can not be opened due to permission restrictions or filesystem errors.</div>';
 337          return false;
 338      }
 339  
 340      $dir_list = array();
 341      $file_list = array();
 342      while( false !== ( $file = readdir( $dir ) ) )
 343      {
 344          if ( ( $file != '.' ) && ( $file != '..' ) )
 345          {
 346              $srcfile = $src.'/'.$file;
 347              $destfile = $dest.'/'.$file;
 348  
 349              if( isset( $read_only_list ) && file_exists( $destfile ) && !is_writable( $destfile ) )
 350              { // Folder or file is not writable
 351                  $read_only_list[] = $destfile;
 352              }
 353  
 354              if ( is_dir( $srcfile ) )
 355              {
 356                  $dir_list[$srcfile] = $destfile;
 357              }
 358              elseif( $overwrite )
 359              { // Add to overwrite
 360                  $file_list[$srcfile] = $destfile;
 361              }
 362          }
 363      }
 364  
 365      $config_ignore_files = get_upgrade_config( 'ignore' );
 366      $config_softmove_files = get_upgrade_config( 'softmove');
 367      $config_forcemove_files = get_upgrade_config( 'forcemove' );
 368  
 369      if( ! empty( $action ) && $action == 'Copying' )
 370      { // Display errors about config file or the unknown and incorrect commands from config file
 371          $config_has_errors = false;
 372          if( is_string( $config_ignore_files ) )
 373          { // Config file has some errors, but the upgrade should not fail because of that
 374              echo '<div class="red">'.$config_ignore_files.'</div>';
 375              $config_has_errors = true;
 376          }
 377          else
 378          {
 379              $config_unknown_commands = get_upgrade_config( 'unknown' );
 380              $config_incorrect_commands = get_upgrade_config( 'incorrect' );
 381  
 382              if( ! empty( $config_unknown_commands ) && is_array( $config_unknown_commands ) )
 383              { // Unknown commands
 384                  foreach( $config_unknown_commands as $config_unknown_command )
 385                  {
 386                      echo '<div class="red">'.sprintf( T_('Unknown policy command: %s'), $config_unknown_command ).'</div>';
 387                  }
 388                  $config_has_errors = true;
 389              }
 390  
 391              if( ! empty( $config_incorrect_commands ) && is_array( $config_incorrect_commands ) )
 392              { // Incorrect commands
 393                  foreach( $config_incorrect_commands as $config_incorrect_command )
 394                  {
 395                      echo '<div class="red">'.sprintf( T_('Incorrect policy command: %s'), $config_incorrect_command ).'</div>';
 396                  }
 397                  $config_has_errors = true;
 398              }
 399          }
 400  
 401          if( $config_has_errors )
 402          { // The upgrade config file contains the errors, Stop the upgrading process
 403              echo '<div class="red">'.sprintf( T_('To continue the upgrade process please fix the issues of the file %s or delete it.'), '&laquo;<b>upgrade_policy.conf</b>&raquo;' ).'</div>';
 404              return false;
 405          }
 406      }
 407  
 408      foreach( $dir_list as $src_dir => $dest_dir )
 409      {
 410          $dest_dir_name = str_replace( $basepath, '', $dest_dir );
 411          // Detect if we should ignore this folder
 412          $ignore_dir = $overwrite && is_array( $config_ignore_files ) && in_array( $dest_dir_name, $config_ignore_files );
 413  
 414          $dir_success = false;
 415          if( !empty( $action ) )
 416          {
 417              if( $ignore_dir )
 418              { // Ignore folder
 419                  echo '<div class="orange">'.sprintf( T_('Ignoring %s because of upgrade_policy.conf'), '&laquo;<b>'.$dest_dir.'</b>&raquo;' ).'</div>';
 420              }
 421              else
 422              { // progressive display of what backup is doing
 423                  echo $action.' &laquo;<strong>'.$dest_dir.'</strong>&raquo;...';
 424                  $dir_success = true;
 425              }
 426              evo_flush();
 427          }
 428          elseif( $ignore_dir )
 429          { // This subfolder must be ingored, Display message about this
 430              echo '<div class="orange">'.sprintf( T_('Ignoring %s because of upgrade_policy.conf'), '&laquo;<b>'.$dest_dir_name.'</b>&raquo;' ).'</div>';
 431              $dir_success = false;
 432              evo_flush();
 433          }
 434  
 435          if( $ignore_dir )
 436          { // Skip the ignored folder
 437              continue;
 438          }
 439  
 440          if( $overwrite && !file_exists( $dest_dir ) )
 441          {
 442              // Create destination directory
 443              if( ! evo_mkdir( $dest_dir ) )
 444              { // No permission to create a folder
 445                  echo '<div class="red">'.sprintf( T_('Unavailable creating of folder %s, probably no permissions.'), '&laquo;<b>'.$dest_dir_name.'</b>&raquo;' ).'</div>';
 446                  $result = false;
 447                  $dir_success = false;
 448                  evo_flush();
 449                  continue;
 450              }
 451          }
 452  
 453          if( $dir_success )
 454          {
 455              echo ' OK.<br />';
 456              evo_flush();
 457          }
 458  
 459          $result = $result && verify_overwrite( $src_dir, $dest_dir, '', $overwrite, $read_only_list );
 460      }
 461  
 462      foreach( $file_list as $src_file => $dest_file )
 463      { // Overwrite destination file
 464          $dest_file_name = str_replace( $basepath, '', $dest_file );
 465          if( is_array( $config_ignore_files ) && in_array( $dest_file_name, $config_ignore_files ) )
 466          { // Ignore this file
 467              echo '<div class="orange">'.sprintf( T_('Ignoring %s because of upgrade_policy.conf'), '&laquo;<b>'.$dest_file_name.'</b>&raquo;' ).'</div>';
 468              evo_flush();
 469              continue;
 470          }
 471  
 472          if( is_array( $config_softmove_files ) && !empty( $config_softmove_files[ $dest_file_name ] ) )
 473          { // Action 'softmove': This file should be copied to other location with saving old file
 474              $copy_file_name = $config_softmove_files[ $dest_file_name ];
 475              // Don't rewrite old file
 476              $rewrite_old_file = false;
 477          }
 478          if( is_array( $config_forcemove_files ) && !empty( $config_forcemove_files[ $dest_file_name ] ) )
 479          { // Action 'forcemove': This file should be copied to other location with rewriting old file
 480              $copy_file_name = $config_forcemove_files[ $dest_file_name ];
 481              // Rewrite old file
 482              $rewrite_old_file = true;
 483          }
 484  
 485          if( ! empty( $copy_file_name ) )
 486          { // This file is marked in config to copy to other location
 487              $copy_file = $basepath.$copy_file_name;
 488              if( ! $rewrite_old_file && file_exists( $copy_file ) )
 489              { // Display warning if we cannot rewrite an existing file
 490                  echo '<div class="orange">'.sprintf( T_('Ignoring softmove of %s because %s is already in place (see upgrade_policy.conf)'),
 491                          '&laquo;<b>'.$dest_file_name.'</b>&raquo;',
 492                          '&laquo;<b>'.$copy_file_name.'</b>&raquo;' ).'</div>';
 493                  evo_flush();
 494                  unset( $copy_file_name );
 495                  continue; // Skip this file
 496              }
 497              else
 498              { // We can copy this file to other location
 499                  echo '<div class="orange">'.sprintf( T_('Moving %s to %s as stated in upgrade_policy.conf'),
 500                          '&laquo;<b>'.$dest_file_name.'</b>&raquo;',
 501                          '&laquo;<b>'.$copy_file_name.'</b>&raquo;' ).'</div>';
 502                  evo_flush();
 503                  // Set new location for a moving file
 504                  $dest_file = $copy_file;
 505                  $dest_file_name = $copy_file_name;
 506                  unset( $copy_file_name );
 507              }
 508          }
 509  
 510          // Copying
 511          if( ! @copy( $src_file, $dest_file ) )
 512          { // Display error if a copy command is unavailable
 513              echo '<div class="red">'.sprintf( T_('Unavailable copying to %s, probably no permissions.'), '&laquo;<b>'.$dest_file_name.'</b>&raquo;' ).'</div>';
 514              $result = false;
 515              evo_flush();
 516          }
 517      }
 518  
 519      closedir( $dir );
 520  
 521      return $result;
 522  }

 577  /**
 578   * Get a list of files and folders that must be ignored/removed on upgrade
 579   *
 580   * @param string Type of action: 'ignore', 'remove', 'softmove', 'forcemove'
 581   *                               'unknown' - Stores all unknown actions
 582   *                               'incorrect' - Stores all incorrect actions
 583   * @return array|string List of files and folders | Error message
 584   */
 585  function get_upgrade_config( $action )
 586  {
 587      global $conf_path, $upgrade_policy_config;
 588  
 589      if( ! isset( $upgrade_policy_config ) )
 590      { // Init global array first time
 591          $upgrade_policy_config = array();
 592      }
 593      elseif( is_string( $upgrade_policy_config ) )
 594      { // Return error about config file
 595          return $upgrade_policy_config;
 596      }
 597  
 598      if( isset( $upgrade_policy_config[ $action ] ) )
 599      { // The config files were already initialized before, Don't make it twice
 600          return $upgrade_policy_config[ $action ];
 601      }
 602  
 603      if( ! check_upgrade_config() )
 604      { // The upgrade config file is NOT mandatory
 605          // Return just an empty array without error text
 606          // Error message is displayed on top page by $Messages object
 607          return $upgrade_policy_config;
 608      }
 609  
 610      $config_handle = @fopen( $conf_path.'upgrade_policy.conf', 'r' );
 611      if( ! $config_handle )
 612      { // No permissions to open file
 613          $upgrade_policy_config = sprintf( T_('No permission to open the %s file.'), '<code>upgrade_policy.conf</code>' );
 614          return $upgrade_policy_config;
 615      }
 616  
 617      // Get content from config file
 618      $config_content = '';
 619      while( !feof( $config_handle ) )
 620      {
 621          $config_content .= fgets( $config_handle, 4096 );
 622      }
 623      fclose( $config_handle );
 624  
 625      if( empty( $config_content ) )
 626      { // Config file is empty for required action
 627          $upgrade_policy_config = sprintf( T_('The %s file is empty.'), '<code>upgrade_policy.conf</code>' );
 628          return $upgrade_policy_config;
 629      }
 630  
 631      // Only these actions are available in the upgrade_policy.conf
 632      $available_actions = array( 'ignore', 'remove', 'softmove', 'forcemove' );
 633  
 634      $all_actions = array_merge( $available_actions, array( 'unknown', 'incorrect' ) );
 635      foreach( $all_actions as $available_action )
 636      { // Init array for all actions only first time
 637          if( !isset( $upgrade_policy_config[ $available_action ] ) )
 638          {
 639              $upgrade_policy_config[ $available_action ] = array();
 640          }
 641      }
 642  
 643      $config_content = str_replace( "\r", '', $config_content );
 644      $config_content = explode( "\n", $config_content );
 645  
 646      foreach( $config_content as $config_line )
 647      {
 648          if( substr( $config_line, 0, 1 ) == ';' )
 649          { // This line is comment text, Skip it
 650              continue;
 651          }
 652  
 653          $config_line = trim( $config_line );
 654  
 655          $config_line_params = explode( ' ', $config_line );
 656          $line_action =  $config_line_params[0];
 657          if( in_array( $line_action, $available_actions ) )
 658          { // This line has an available action
 659              if( empty( $config_line_params[1] ) )
 660              { // Incorrect command
 661                  $upgrade_policy_config[ 'incorrect' ][] = $config_line;
 662                  continue;
 663              }
 664              if( $line_action == 'softmove' || $line_action == 'forcemove' )
 665              { // These actions have two params
 666                  if( empty( $config_line_params[1] ) || empty( $config_line_params[2] ) )
 667                  { // Incorrect command
 668                      $upgrade_policy_config[ 'incorrect' ][] = $config_line;
 669                      continue;
 670                  }
 671                  $upgrade_policy_config[ $line_action ][ $config_line_params[1] ] = $config_line_params[2];
 672              }
 673              else
 674              { // Actions 'ignore' & 'remove' have only one param
 675                  $upgrade_policy_config[ $line_action ][] = $config_line_params[1];
 676              }
 677          }
 678          elseif( !empty( $line_action ) )
 679          { // Also save all unknown actions to display error
 680              $upgrade_policy_config[ 'unknown' ][] = $config_line;
 681          }
 682      }
 683  
 684      return $upgrade_policy_config[ $action ];
 685  }

 689   * Remove files/folders after upgrade, See file upgrade_policy.conf
 690   */
 691  function remove_after_upgrade()
 692  {
 693      global $basepath, $conf_path;
 694  
 695      $upgrade_removed_files = get_upgrade_config( 'remove' );
 696  
 697      echo '<h4>'.T_('Cleaning up...').'</h4>';
 698      evo_flush();
 699  
 700      if( is_string( $upgrade_removed_files ) )
 701      { // Errors on opening of upgrade_policy.conf
 702          $config_error = $upgrade_removed_files;
 703      }
 704      elseif( empty( $upgrade_removed_files ) )
 705      { // No files/folders to remove, Exit here
 706          $config_error = sprintf( T_('No "remove" sections have been defined in the file %s.'), '<code>upgrade_policy.conf</code>' );
 707      }
 708  
 709      if( !empty( $config_error ) )
 710      { // Display config error
 711          echo '<div class="red">';
 712          echo $config_error;
 713          echo ' '.T_('No cleanup is being done. You should manually remove the <code>/install</code> folder and check for other unwanted files...');
 714          echo '</div>';
 715          return;
 716      }
 717  
 718      foreach( $upgrade_removed_files as $file_path )
 719      {
 720          $file_path = $basepath.$file_path;
 721          $log_message = sprintf( T_('Removing %s as stated in upgrade_policy.conf...'), '<code>'.$file_path.'</code>' ).' ';
 722          $success = true;
 723          if( file_exists( $file_path ) )
 724          { // File exists
 725              if( is_dir( $file_path ) )
 726              { // Remove folder recursively
 727                  if( rmdir_r( $file_path ) )
 728                  { // Success
 729                      $log_message .= T_('OK');
 730                  }
 731                  else
 732                  { // Failed
 733                      $log_message .= T_('Failed').': '.T_('No permissions to delete the folder');
 734                      $success = false;
 735                  }
 736              }
 737              elseif( is_writable( $file_path ) )
 738              { // Remove file
 739                  if( @unlink( $file_path ) )
 740                  { // Success
 741                      $log_message .= T_('OK');
 742                  }
 743                  else
 744                  { // Failed
 745                      $log_message .= T_('Failed').': '.T_('No permissions to delete the file');
 746                      $success = false;
 747                  }
 748              }
 749              else
 750              { // File is not writable
 751                  $log_message .= T_('Failed').': '.T_('No permissions to delete the file');
 752                  $success = false;
 753              }
 754          }
 755          else
 756          { // No file/folder
 757              $log_message .= T_('Failed').': '.T_('No file found');
 758              $success = false;
 759          }
 760  
 761          echo $success ? $log_message.'<br />' : '<div class="orange">'.$log_message.'</div>';
 762          evo_flush();
 763      }
 764  }

 771   * @return string

 773  function get_affected_paths( $path )
 774  {
 775      global $basepath;
 776  
 777      $affected_paths = T_( 'Affected paths:' ).' ';
 778      if( is_array( $path ) )
 779      {
 780          $paths = array();
 781          foreach( $path as $p )
 782              $paths[] = no_trailing_slash( $p );
 783  
 784          $affected_paths .= implode( ', ', $paths );
 785      }
 786      elseif( $path == '*' )
 787      {
 788          $filename_params = array(
 789                  'inc_files'    => false,
 790                  'recurse'    => false,
 791                  'basename'    => true,
 792              );
 793          $affected_paths .= implode( ', ', get_filenames( $basepath, $filename_params ) );
 794      }
 795      else
 796      {
 797          $affected_paths .= no_trailing_slash( $path );
 798      }
 799      return $affected_paths;
