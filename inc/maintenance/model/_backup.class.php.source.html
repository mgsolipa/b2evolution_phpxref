 159      /**
 160       * Load settings from request
 161       *
 162       * @param boolean TRUE to memorize all params
 163       */
 164  	function load_from_Request( $memorize_params = false )
 165      {
 166          global $backup_paths, $backup_tables, $Messages;
 167  
 168          // Load folders/files settings from request
 169          foreach( $backup_paths as $name => $settings )
 170          {
 171              if( array_key_exists( 'label', $settings ) && !is_null( $settings['label'] ) )
 172              {    // We can set param
 173                  $this->backup_paths[$name] = param( 'bk_'.$name, 'boolean', 0, $memorize_params );
 174              }
 175          }
 176  
 177          // Load tables settings from request
 178          foreach( $backup_tables as $name => $settings )
 179          {
 180              $this->backup_tables[$name] = param( 'bk_'.$name, 'boolean', 0, $memorize_params );
 181          }
 182  
 183          $this->pack_backup_files = param( 'bk_pack_backup_files', 'boolean', 0, $memorize_params );
 184  
 185          // Check are there something to backup
 186          if( !$this->has_included( $this->backup_paths ) && !$this->has_included( $this->backup_tables ) )
 187          {
 188              $Messages->add( T_('You have not selected anything to backup.'), 'error' );
 189              return false;
 190          }
 191  
 192          return true;
 193      }

 197       * Start backup

 199  	function start_backup()
 200      {
 201          global $basepath, $backup_path, $servertimenow;
 202  
 203          // Create current backup path
 204          $cbackup_path = $backup_path.date( 'Y-m-d-H-i-s', $servertimenow ).'/';
 205  
 206          echo '<p>'.sprintf( T_('Starting backup to: &laquo;%s&raquo; ...'), $cbackup_path ).'</p>';
 207          evo_flush();
 208  
 209          // Prepare backup directory
 210          $success = prepare_maintenance_dir( $backup_path, true );
 211  
 212          // Backup directories and files
 213          if( $success && $this->has_included( $this->backup_paths ) )
 214          {
 215              $backup_files_path = $this->pack_backup_files ? $cbackup_path : $cbackup_path.'files/';
 216  
 217              // Prepare files backup directory
 218              if( $success = prepare_maintenance_dir( $backup_files_path, false ) )
 219              {    // We can backup files
 220                  $success = $this->backup_files( $backup_files_path );
 221              }
 222          }
 223  
 224          // Backup database
 225          if( $success && $this->has_included( $this->backup_tables ) )
 226          {
 227              $backup_tables_path = $this->pack_backup_files ? $cbackup_path : $cbackup_path.'db/';
 228  
 229              // Prepare database backup directory
 230              if( $success = prepare_maintenance_dir( $backup_tables_path, false ) )
 231              {    // We can backup database
 232                  $success = $this->backup_database( $backup_tables_path );
 233              }
 234          }
 235  
 236          if( $success )
 237          {
 238              echo '<p>'.sprintf( T_('Backup complete. Directory: &laquo;%s&raquo;'), $cbackup_path ).'</p>';
 239              evo_flush();
 240  
 241              return true;
 242          }
 243  
 244          @rmdir_r( $cbackup_path );
 245          return false;

 249      /**
 250       * Backup files
 251       * @param string backup directory path
 252       */
 253  	function backup_files( $backup_dirpath )
 254      {
 255          global $basepath, $backup_paths, $inc_path;
 256  
 257          echo '<h4>'.T_('Creating folders/files backup...').'</h4>';
 258          evo_flush();
 259  
 260          // Find included and excluded files
 261  
 262          $included_files = array();
 263  
 264          if( $root_included = $this->backup_paths['application_files'] )
 265          {
 266              $filename_params = array(
 267                      'recurse'        => false,
 268                      'basename'       => true,
 269                      'trailing_slash' => true,
 270                      //'inc_evocache' => true, // Uncomment to backup ?evocache directories
 271                      'inc_temp'       => false,
 272                  );
 273              $included_files = get_filenames( $basepath, $filename_params );
 274          }
 275  
 276          // Prepare included/excluded paths
 277          $excluded_files = array();
 278  
 279          foreach( $this->backup_paths as $name => $included )
 280          {
 281              foreach( $this->path_to_array( $backup_paths[$name]['path'] ) as $path )
 282              {
 283                  if( $root_included && !$included )
 284                  {
 285                      $excluded_files[] = $path;
 286                  }
 287                  elseif( !$root_included && $included )
 288                  {
 289                      $included_files[] = $path;
 290                  }
 291              }
 292          }
 293  
 294          // Remove excluded list from included list
 295          $included_files = array_diff( $included_files, $excluded_files );
 296  
 297          if( $this->pack_backup_files )
 298          { // Create ZIPped backup
 299              $zip_filepath = $backup_dirpath.'files.zip';
 300  
 301              // Pack using 'zlib' extension and PclZip wrapper
 302  
 303              if( ! defined( 'PCLZIP_TEMPORARY_DIR' ) )
 304              { // Set path for temp files of PclZip
 305                  define( 'PCLZIP_TEMPORARY_DIR', $backup_dirpath );
 306              }
 307              // Load PclZip class (PHP4):
 308              load_class( '_ext/pclzip/pclzip.lib.php', 'PclZip' );
 309  
 310              $PclZip = new PclZip( $zip_filepath );
 311  
 312              echo sprintf( T_('Archiving files to &laquo;<strong>%s</strong>&raquo;...'), $zip_filepath ).'<br/>';
 313              evo_flush();
 314  
 315              foreach( $included_files as $included_file )
 316              {
 317                  echo sprintf( T_('Backing up &laquo;<strong>%s</strong>&raquo; ...'), $basepath.$included_file );
 318                  evo_flush();
 319  
 320                  $file_list = $PclZip->add( no_trailing_slash( $basepath.$included_file ), PCLZIP_OPT_REMOVE_PATH, no_trailing_slash( $basepath ) );
 321                  if( $file_list == 0 )
 322                  {
 323                      echo '<p style="color:red">'.sprintf( T_('Unable to create &laquo;%s&raquo;'), $zip_filepath ).'</p>';
 324                      evo_flush();
 325  
 326                      return false;
 327                  }
 328                  else
 329                  {
 330                      echo ' OK.<br />';
 331                      evo_flush();
 332                  }
 333              }
 334          }
 335          else
 336          {    // Copy directories and files to backup directory
 337              foreach( $included_files as $included_file )
 338              {
 339                  $this->recurse_copy( no_trailing_slash( $basepath.$included_file ),
 340                                          no_trailing_slash( $backup_dirpath.$included_file ) );
 341              }
 342          }
 343  
 344          return true;
 345      }

 348      /**
 349       * Backup database
 350       *
 351       * @param string backup directory path
 352       */
 353  	function backup_database( $backup_dirpath )
 354      {
 355          global $DB, $db_config, $backup_tables, $inc_path;
 356  
 357          echo '<h4>'.T_('Creating database backup...').'</h4>';
 358          evo_flush();
 359  
 360          // Collect all included tables
 361          $ready_to_backup = array();
 362          foreach( $this->backup_tables as $name => $included )
 363          {
 364              if( $included )
 365              {
 366                  $tables = aliases_to_tables( $backup_tables[$name]['table'] );
 367                  if( is_array( $tables ) )
 368                  {
 369                      $ready_to_backup = array_merge( $ready_to_backup, $tables );
 370                  }
 371                  elseif( $tables == '*' )
 372                  {
 373                      foreach( $DB->get_results( 'SHOW TABLES', ARRAY_N ) as $row )
 374                      {
 375                          $ready_to_backup[] = $row[0];
 376                      }
 377                  }
 378                  else
 379                  {
 380                      $ready_to_backup[] = $tables;
 381                  }
 382              }
 383          }
 384  
 385          // Ensure there are no duplicated tables
 386          $ready_to_backup = array_unique( $ready_to_backup );
 387  
 388          // Exclude tables
 389          foreach( $this->backup_tables as $name => $included )
 390          {
 391              if( !$included )
 392              {
 393                  $tables = aliases_to_tables( $backup_tables[$name]['table'] );
 394                  if( is_array( $tables ) )
 395                  {
 396                      $ready_to_backup = array_diff( $ready_to_backup, $tables );
 397                  }
 398                  elseif( $tables != '*' )
 399                  {
 400                      $index = array_search( $tables, $ready_to_backup );
 401                      if( $index )
 402                      {
 403                          unset( $ready_to_backup[$index] );
 404                      }
 405                  }
 406              }
 407          }
 408  
 409          // Create and save created SQL backup script
 410          $backup_sql_filename = 'db.sql';
 411          $backup_sql_filepath = $backup_dirpath.$backup_sql_filename;
 412  
 413          // Check if backup file exists
 414          if( file_exists( $backup_sql_filepath ) )
 415          {    // Stop tables backup, because backup file exists
 416              echo '<p style="color:red">'.sprintf( T_('Unable to write database dump. Database dump already exists: &laquo;%s&raquo;'), $backup_sql_filepath ).'</p>';
 417              evo_flush();
 418  
 419              return false;
 420          }
 421  
 422          $f = @fopen( $backup_sql_filepath , 'w+' );
 423          if( $f == false )
 424          {    // Stop backup, because it can't open backup file for writing
 425              echo '<p style="color:red">'.sprintf( T_('Unable to write database dump. Could not open &laquo;%s&raquo; for writing.'), $backup_sql_filepath ).'</p>';
 426              evo_flush();
 427  
 428              return false;
 429          }
 430  
 431          echo sprintf( T_('Dumping tables to &laquo;<strong>%s</strong>&raquo;...'), $backup_sql_filepath ).'<br/>';
 432          evo_flush();
 433  
 434          // Create and save created SQL backup script
 435          foreach( $ready_to_backup as $table )
 436          {
 437              // progressive display of what backup is doing
 438              echo sprintf( T_('Backing up table &laquo;<strong>%s</strong>&raquo; ...'), $table );
 439              evo_flush();
 440  
 441              $row_table_data = $DB->get_row( 'SHOW CREATE TABLE '.$table, ARRAY_N );
 442              fwrite( $f, $row_table_data[1].";\n\n" );
 443  
 444              $page = 0;
 445              $page_size = 500;
 446              $is_insert_sql_started = false;
 447              $is_first_insert_sql_value = true;
 448              while( ! empty( $rows ) || $page == 0 )
 449              { // Get the records by page(500) in order to save memory and avoid fatal error
 450                  $rows = $DB->get_results( 'SELECT * FROM '.$table.' LIMIT '.( $page * $page_size ).', '.$page_size, ARRAY_N );
 451  
 452                  if( $page == 0 && ! $is_insert_sql_started && ! empty( $rows ) )
 453                  { // Start SQL INSERT clause
 454                      fwrite( $f, 'INSERT INTO '.$table.' VALUES ' );
 455                      $is_insert_sql_started = true;
 456                  }
 457  
 458                  foreach( $rows as $row )
 459                  {
 460                      $values = '(';
 461                      $num_fields = count( $row );
 462                      for( $index = 0; $index < $num_fields; $index++ )
 463                      {
 464                          if( isset( $row[$index] ) )
 465                          {
 466                              $row[$index] = str_replace("\n","\\n", addslashes( $row[$index] ) );
 467                              $values .= '\''.$row[$index].'\'' ;
 468                          }
 469                          else
 470                          { // The $row[$index] value is not set or is NULL
 471                              $values .= 'NULL';
 472                          }
 473  
 474                          if( $index<( $num_fields-1 ) )
 475                          {
 476                              $values .= ',';
 477                          }
 478                      }
 479                      $values .= ')';
 480                      if( $is_first_insert_sql_value )
 481                      { // Don't write a comma before first row values
 482                          $is_first_insert_sql_value = false;
 483                      }
 484                      else
 485                      { // Write a comma between row values
 486                          $values = ','.$values;
 487                      }
 488  
 489                      fwrite( $f, $values );
 490                  }
 491                  unset( $rows );
 492                  $page++;
 493              }
 494  
 495              if( $is_insert_sql_started )
 496              { // End SQL INSERT clause
 497                  fwrite( $f, ";\n\n" );
 498              }
 499  
 500              // Flush the output to a file
 501              if( fflush( $f ) )
 502              {
 503                  echo ' OK.';
 504              }
 505              echo '<br />';
 506              evo_flush();
 507          }
 508  
 509          // Close backup file input stream
 510          fclose( $f );
 511  
 512          if( $this->pack_backup_files )
 513          { // Pack created backup SQL script
 514  
 515              // Pack using 'zlib' extension and PclZip wrapper
 516  
 517              if( ! defined( 'PCLZIP_TEMPORARY_DIR' ) )
 518              { // Set path for temp files of PclZip
 519                  define( 'PCLZIP_TEMPORARY_DIR', $backup_dirpath );
 520              }
 521              // Load PclZip class (PHP4):
 522              load_class( '_ext/pclzip/pclzip.lib.php', 'PclZip' );
 523  
 524              $zip_filepath = $backup_dirpath.'db.zip';
 525              $PclZip = new PclZip( $zip_filepath );
 526  
 527              $file_list = $PclZip->add( $backup_dirpath.$backup_sql_filename, PCLZIP_OPT_REMOVE_PATH, no_trailing_slash( $backup_dirpath ) );
 528              if( $file_list == 0 )
 529              {
 530                  echo '<p style="color:red">'.sprintf( T_('Unable to create &laquo;%s&raquo;'), $zip_filepath ).'</p>';
 531                  evo_flush();
 532  
 533                  return false;
 534              }
 535  
 536              unlink( $backup_sql_filepath );
 537          }
 538  
 539          return true;
 540      }

 543      /**
 544       * Copy directory recursively
 545       *
 546       * @param string source directory
 547       * @param string destination directory
 548       * @param array excluded directories
 549       */
 550  	function recurse_copy( $src, $dest, $root = true )
 551      {
 552          if( is_dir( $src ) )
 553          {
 554              if( ! ( $dir = opendir( $src ) ) )
 555              {
 556                  return false;
 557              }
 558              if( ! evo_mkdir( $dest ) )
 559              {
 560                  return false;
 561              }
 562              while( false !== ( $file = readdir( $dir ) ) )
 563              {
 564                  if( $file == '.' || $file == '..' )
 565                  { // Skip these reserved names
 566                      continue;
 567                  }
 568  
 569                  if( $file == 'upload-tmp' )
 570                  { // Skip temp folder
 571                      continue;
 572                  }
 573  
 574                  $srcfile = $src.'/'.$file;
 575                  if( is_dir( $srcfile ) )
 576                  {
 577                      if( $root )
 578                      { // progressive display of what backup is doing
 579                          echo sprintf( T_('Backing up &laquo;<strong>%s</strong>&raquo; ...'), $srcfile ).'<br/>';
 580                          evo_flush();
 581                      }
 582                      $this->recurse_copy( $srcfile, $dest . '/' . $file, false );
 583                  }
 584                  else
 585                  { // Copy file
 586                      copy( $srcfile, $dest.'/'. $file );
 587                  }
 588              }
 589              closedir( $dir );
 590          }
 591          else
 592          {
 593              copy( $src, $dest );
 594          }
 595      }
