  22   * Add a widget to global array in order to insert it in DB by single SQL query later
  23   *
  24   * @param integer Blog ID
  25   * @param string Container name
  26   * @param string Type
  27   * @param string Code

  30   * @param integer 1 - enabled, 0 - disabled
  31   */
  32  function add_basic_widget( $blog_ID, $container_name, $code, $type, $order, $params = NULL, $enabled = 1 )
  33  {
  34      global $basic_widgets_insert_sql_rows, $DB;
  35  
  36      if( is_null( $params ) )
  37      { // NULL
  38          $params = 'NULL';
  39      }
  40      elseif( is_array( $params ) )
  41      { // array
  42          $params = $DB->quote( serialize( $params ) );
  43      }
  44      else
  45      { // string
  46          $params = $DB->quote( $params );
  47      }
  48  
  49      $basic_widgets_insert_sql_rows[] = '( '
  50          .$blog_ID.', '
  51          .$DB->quote( $container_name ).', '
  52          .$order.', '
  53          .$enabled.', '
  54          .$DB->quote( $type ).', '
  55          .$DB->quote( $code ).', '
  56          .$params.' )';
  57  }
