 114      /**
 115       * Display the widget!
 116       *
 117       * @param array MUST contain at least the basic display params
 118       */
 119  	function display( $params )
 120      {
 121          global $DB, $Item, $Blog;
 122  
 123          $this->init_display( $params );
 124  
 125          // Initialise css classes for icons depending on widget setting
 126          $icon_colors_classes = '';
 127          if( ! empty( $this->disp_params['icon_colors'] ) )
 128          { // If at least one color status is selected
 129              foreach( $this->disp_params['icon_colors'] as $class_name => $is_selected )
 130              {
 131                  if( ! empty( $is_selected ) )
 132                  {
 133                      $icon_colors_classes .= ' ufld__'.$class_name.'color';
 134                  }
 135              }
 136          }
 137  
 138          $r = '';
 139  
 140          $widget_User = & $this->get_widget_User();
 141          if( empty( $widget_User ) )
 142          { // No user detected
 143              $r .= '<p class="red">'.sprintf( T_('User %s not found.'), '<b>'.format_to_output( $this->disp_params['login'], 'text' ).'</b>' ).'</p>';
 144          }
 145  
 146          if( ! empty( $widget_User ) )
 147          { // If we really have found user
 148              // Get all user extra field values with type "url"
 149              $url_fields = $widget_User->userfields_by_type( 'url' );
 150              if( count( $url_fields ) )
 151              {
 152                  $r .= '<div class="ufld_icon_links">';
 153                  foreach( $url_fields as $field )
 154                  {
 155                      $r .= '<a href="'.$field->uf_varchar.'"'.( empty( $icon_colors_classes ) ? '' : ' class="ufld_'.$field->ufdf_code.$icon_colors_classes.'"' ).'>'
 156                              .'<span class="'.$field->ufdf_icon_name.'"></span>'
 157                          .'</a>';
 158                  }
 159                  $r .= '</div>';
 160              }
 161          }
 162  
 163          if( empty( $r ) )
 164          { // Nothing to display
 165              return true;
 166          }
 167  
 168          echo $this->disp_params['block_start'];
 169  
 170          $this->disp_title();
 171  
 172          echo $this->disp_params['block_body_start'];
 173  
 174          echo $r;
 175  
 176          echo $this->disp_params['block_body_end'];
 177  
 178          echo $this->disp_params['block_end'];
 179  
 180          return true;
 181      }

 184      /**
 185       * Get User that should be used for this widget now
 186       *
 187       * @return object User
 188       */
 189      function & get_widget_User()
 190      {
 191          global $Item, $Blog;
 192  
 193          $widget_User = NULL;
 194  
 195          if( empty( $this->disp_params['login'] ) )
 196          { // No defined user in widget settings
 197              // Note: There is no 'in-item' context in i7
 198              if( ! empty( $Blog ) )
 199              { // Use an owner of the current $Blog
 200                  $widget_User = & $Blog->get_owner_User();
 201              }
 202          }
 203          else
 204          { // Try to get user by login from DB
 205              $UserCache = & get_UserCache();
 206              $widget_User = & $UserCache->get_by_login( $this->disp_params['login'] );
 207          }
 208  
 209          return $widget_User;
 210      }
