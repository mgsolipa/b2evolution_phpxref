 147      /**
 148       * Display the widget!
 149       *
 150       * @param array MUST contain at least the basic display params
 151       */
 152  	function display( $params )
 153      {
 154          global $DB, $Item, $Blog;
 155  
 156          $this->init_display( $params );
 157  
 158          echo $this->disp_params['block_start'];
 159  
 160          $this->disp_title();
 161  
 162          echo $this->disp_params['block_body_start'];
 163  
 164          $org_ID = intval( $this->disp_params['org_ID'] );
 165          if( $org_ID > 0 )
 166          {
 167              $OrganizationCache = & get_OrganizationCache();
 168              $Organization = & $OrganizationCache->get_by_ID( $org_ID, false, false );
 169          }
 170  
 171          if( empty( $Organization ) )
 172          { // No organization found
 173              global $Messages;
 174              $Messages->add( T_('Requested Team was not found.'), 'error' );
 175              $Messages->display();
 176          }
 177          else
 178          {
 179              // Get all users of the selected organization
 180              $users = $Organization->get_users();
 181  
 182              if( $this->disp_params['display_icons'] )
 183              { // Initialise css classes for icons depending on widget setting and only when they are displayed
 184                  $icon_colors_classes = '';
 185                  if( ! empty( $this->disp_params['icon_colors'] ) )
 186                  { // If at least one color status is selected
 187                      foreach( $this->disp_params['icon_colors'] as $class_name => $is_selected )
 188                      {
 189                          if( ! empty( $is_selected ) )
 190                          {
 191                              $icon_colors_classes .= ' ufld__'.$class_name.'color';
 192                          }
 193                      }
 194                  }
 195              }
 196  
 197              if( count( $users ) )
 198              {
 199                  foreach( $users as $org_User )
 200                  {
 201                      echo '<div class="col-lg-4 col-sm-6 text-center">';
 202  
 203                      $user_url = $this->disp_params['link_profile'] ? $org_User->get_userpage_url( $Blog->ID, true ) : '';
 204  
 205                      if( ! empty( $user_url ) )
 206                      { // Display url to user page only when it is allowed by widget setting
 207                          echo '<a href="'.$user_url.'" class="user_link">';
 208                      }
 209  
 210                      // Profile picture
 211                      echo $org_User->get_avatar_imgtag( 'crop-top-320x320', 'img-circle img-responsive img-center' );
 212  
 213                      if( ! empty( $user_url ) )
 214                      { // End of user link, see above
 215                           echo '</a>';
 216  
 217                          echo '<a href="'.$user_url.'" class="user_link">';
 218                      }
 219  
 220                      // Full name
 221                      echo '<h3>'.$org_User->get( 'fullname' ).'</h3>';
 222  
 223                      if( ! empty( $user_url ) )
 224                      { // End of user link, see above
 225                          echo '</a>';
 226                      }
 227  
 228                      if( $this->disp_params['display_icons'] )
 229                      { // Display user links as icons
 230                          $url_fields = $org_User->userfields_by_type( 'url' );
 231                          if( count( $url_fields ) )
 232                          {
 233                              echo '<div class="ufld_icon_links">';
 234                              foreach( $url_fields as $field )
 235                              {
 236                                  echo '<a href="'.$field->uf_varchar.'"'.( empty( $icon_colors_classes ) ? '' : ' class="ufld_'.$field->ufdf_code.$icon_colors_classes.'"' ).'>'
 237                                          .'<span class="'.$field->ufdf_icon_name.'"></span>'
 238                                      .'</a>';
 239                              }
 240                              echo '</div>';
 241                          }
 242                      }
 243  
 244                      // Info
 245                      if( ! empty( $this->disp_params['field_code'] ) && ( $field_values = $org_User->userfield_values_by_code( $this->disp_params['field_code'] ) ) )
 246                      {
 247                          $field_extra_lines = intval( $this->disp_params['field_extra_lines'] );
 248                          echo '<p class="user_field"'
 249                              .( empty( $field_extra_lines ) ? '' :
 250                                      ' style="height:'.( $field_extra_lines * 1.5 /* line-height */ ).'em;'
 251                                          .'-webkit-line-clamp:'.$field_extra_lines.'"' )
 252                              .'>';
 253                          foreach( $field_values as $f => $field_value )
 254                          {
 255                              if( $f > 0 )
 256                              { // Space between each field value
 257                                  echo ' ';
 258                              }
 259                              echo preg_replace( "/[\r\n]+/", ' ', $field_value );
 260                          }
 261                          echo '</p>';
 262                      }
 263  
 264                      echo '</div>';
 265                  }
 266  
 267                  echo '<div class="clear"></div>';
 268              }
 269          }
 270  
 271          echo $this->disp_params['block_body_end'];
 272  
 273          echo $this->disp_params['block_end'];
 274  
 275          return true;
 276      }
