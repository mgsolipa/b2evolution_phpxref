 167      /**
 168       * Display the widget!
 169       *
 170       * @param array MUST contain at least the basic display params
 171       */
 172  	function display( $params )
 173      {
 174          /**
 175          * @var Blog
 176          */
 177          global $Blog;
 178  
 179          $this->init_display( $params );
 180  
 181          // Collection common links:
 182          echo $this->disp_params['block_start'];
 183  
 184          // Display title if requested
 185          $this->disp_title();
 186  
 187          echo $this->disp_params['block_body_start'];
 188  
 189          echo $this->disp_params['list_start'];
 190  
 191          if( $this->disp_params['show_home'] )
 192          {
 193              echo $this->disp_params['item_start'];
 194              echo '<strong><a href="'.$Blog->get('url').'">'.T_('Home').'</a></strong>';
 195              echo $this->disp_params['item_end'];
 196          }
 197  
 198          if( $this->disp_params['show_recently'] )
 199          {
 200              echo $this->disp_params['item_start'];
 201              echo '<strong><a href="'.$Blog->get('recentpostsurl').'">'.T_('Recently').'</a></strong>';
 202              echo $this->disp_params['item_end'];
 203          }
 204  
 205          if( $this->disp_params['show_search'] )
 206          {
 207              echo $this->disp_params['item_start'];
 208              echo '<strong><a href="'.$Blog->get('searchurl').'">'.T_('Search').'</a></strong>';
 209              echo $this->disp_params['item_end'];
 210          }
 211  
 212          if( $this->disp_params['show_postidx'] )
 213          {
 214              echo $this->disp_params['item_start'];
 215              echo '<strong><a href="'.$Blog->get('postidxurl').'">'.T_('Post index').'</a></strong>';
 216              echo $this->disp_params['item_end'];
 217          }
 218  
 219          if( $this->disp_params['show_archives'] )
 220          {
 221              // fp> TODO: don't display this if archives plugin not installed... or depluginize archives (I'm not sure)
 222              echo $this->disp_params['item_start'];
 223              echo '<strong><a href="'.$Blog->get('arcdirurl').'">'.T_('Archives').'</a></strong>';
 224              echo $this->disp_params['item_end'];
 225          }
 226  
 227          if( $this->disp_params['show_categories'] )
 228          {
 229              echo $this->disp_params['item_start'];
 230              echo '<strong><a href="'.$Blog->get('catdirurl').'">'.T_('Categories').'</a></strong>';
 231              echo $this->disp_params['item_end'];
 232          }
 233  
 234          if( $this->disp_params['show_mediaidx'] )
 235          {
 236              echo $this->disp_params['item_start'];
 237              echo '<strong><a href="'.$Blog->get('mediaidxurl').'">'.T_('Photo index').'</a></strong>';
 238              echo $this->disp_params['item_end'];
 239          }
 240  
 241          if( $this->disp_params['show_latestcomments'] && $Blog->get_setting( 'comments_latest' ) )
 242          { // Display link to latest comments if this feature is enabled for current blog
 243              echo $this->disp_params['item_start'];
 244              echo '<strong><a href="'.$Blog->get('lastcommentsurl').'">'.T_('Latest comments').'</a></strong>';
 245              echo $this->disp_params['item_end'];
 246          }
 247  
 248          if( $this->disp_params['show_owneruserinfo'] )
 249          {
 250              echo $this->disp_params['item_start'];
 251              echo '<strong><a href="'.url_add_param( $Blog->get('userurl'), 'user_ID='.$Blog->owner_user_ID ).'">'.T_('Owner details').'</a></strong>';
 252              echo $this->disp_params['item_end'];
 253          }
 254  
 255          if( $this->disp_params['show_ownercontact'] && $url = $Blog->get_contact_url( true ) )
 256          {    // owner allows contact:
 257              echo $this->disp_params['item_start'];
 258              echo '<strong><a href="'.$url.'">'.T_('Contact').'</a></strong>';
 259              echo $this->disp_params['item_end'];
 260          }
 261  
 262          if( $this->disp_params['show_sitemap'] )
 263          {
 264              echo $this->disp_params['item_start'];
 265              echo '<strong><a href="'.$Blog->get('sitemapurl').'">'.T_('Site map').'</a></strong>';
 266              echo $this->disp_params['item_end'];
 267          }
 268  
 269          echo $this->disp_params['list_end'];
 270  
 271          echo $this->disp_params['block_body_end'];
 272  
 273          echo $this->disp_params['block_end'];
 274  
 275          return true;
 276      }
