 581  	function get_cache_keys()

 584  
 585          $keys = array(
 586                  'wi_ID'   => $this->ID,                    // Have the widget settings changed ?

 588              );
 589  
 590          switch( $this->disp_params['link_type'] )
 591          {
 592              case 'login':          /* This one will probably abort caching by itself anyways */
 593              case 'register':    /* This one will probably abort caching by itself anyways */
 594              case 'profile':        // This can be cached
 595              case 'avatar':
 596                  // This link also depends on whether or not someone is logged in:
 597                  $keys['loggedin'] = (is_logged_in() ? 1 : 0);
