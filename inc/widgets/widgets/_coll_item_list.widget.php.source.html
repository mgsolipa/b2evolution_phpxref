 548      /**
 549       * Display a chapter with all of its loaded items
 550       *
 551       * @param Chapter
 552       * @param array Items map by Chapter
 553       * @return boolean true if content was displayed, false otherwise
 554       */
 555  	function disp_chapter( $Chapter, & $items_map_by_chapter )
 556      {
 557          $content_is_displayed = false;
 558  
 559          if( isset( $items_map_by_chapter[$Chapter->ID] ) && ( count( $items_map_by_chapter[$Chapter->ID] ) > 0 ) )
 560          { // Display Chapter only if it has some items
 561              echo $this->disp_params['item_start'];
 562              $Chapter->get_Blog();
 563              echo '<a href="'.$Chapter->get_permanent_url().'">'.$Chapter->get('name').'</a>';
 564              echo $this->disp_params['item_end'];
 565              echo $this->disp_params['group_start'];
 566  
 567              foreach( $items_map_by_chapter[$Chapter->ID] as $iterator_Item )
 568              { // Display contents of the Item depending on widget params:
 569                  $content_is_displayed = $this->disp_contents( $iterator_Item, true ) || $content_is_displayed;
 570              }
 571  
 572              // Close cat group
 573              echo $this->disp_params['group_end'];
 574          }
 575  
 576          return $content_is_displayed;
 577      }

 592          $content_is_displayed = false;

 597          // Is this the current item?

 599          {    // The current page is currently displaying the Item this link is pointing to

 609          {

 611          }
 612          else
 613          {
 614              echo $this->disp_params[$disp_param_prefix.'item_start'];
 615          }
 616  
 617          if( $this->disp_params['disp_first_image'] == 'special' )

 744      /**
 745       * Display images of the selected item
 746       *
 747       * @param array Params
 748       * @param boolean Changed by reference when content is displayed
 749       */
 750  	function disp_images( $params = array(), & $content_is_displayed )
 751      {
 752          $params = array_merge( array(
 753                  'before'                     => '',
 754                  'after'                      => '',
 755                  'placeholder'                => '',
 756                  'Item'                       => NULL,
 757                  'start'                      => 1,
 758                  'limit'                      => 1,
 759                  'restrict_to_image_position' => 'teaser,teaserperm,teaserlink,aftermore,inline',
 760                  'links_sql_select'           => '',
 761                  'links_sql_orderby'          => 'link_order',
 762              ), $params );
 763  
 764          $links_params = array(
 765                  'sql_select_add' => $params['links_sql_select'],
 766                  'sql_order_by'   => $params['links_sql_orderby']
 767              );
 768  
 769          $disp_Item = & $params['Item'];
 770  
 771          // Get list of ALL attached files:
 772          $LinkOwner = new LinkItem( $disp_Item );
 773  
 774          $images = '';
 775  
 776          if( $LinkList = $LinkOwner->get_attachment_LinkList( $params['limit'], $params['restrict_to_image_position'], 'image', $links_params ) )
 777          {    // Get list of attached files
 778              $image_num = 1;
 779              while( $Link = & $LinkList->get_next() )
 780              {
 781                  if( ( $File = & $Link->get_File() ) && $File->is_image() )
 782                  {    // Get only images
 783                      if( $image_num < $params['start'] )
 784                      { // Skip these first images
 785                          $image_num++;
 786                          continue;
 787                      }
 788                      switch( $this->disp_params[ 'item_pic_link_type' ] )
 789                      {    // Set url for picture link
 790                          case 'none':
 791                              $pic_url = NULL;
 792                              break;
 793  
 794                          case 'permalink':
 795                              $pic_url = $disp_Item->get_permanent_url();
 796                              break;
 797  
 798                          case 'linkto_url':
 799                              $pic_url = $disp_Item->url;
 800                              break;
 801  
 802                          case 'auto':
 803                          default:
 804                              $pic_url = ( empty( $disp_Item->url ) ? $disp_Item->get_permanent_url() : $disp_Item->url );
 805                              break;
 806                      }
 807  
 808                      // Print attached picture
 809                      $images .= $File->get_tag( '', '', '', '', $this->disp_params['thumb_size'], $pic_url );
 810  
 811                      $content_is_displayed = true;
 812  
 813                      $image_num++;
 814                  }
 815              }
 816          }
 817  
 818          if( ! empty( $images ) )
 819          { // Print out images only when at least one exists
 820              echo $params['before'];
 821              echo $images;
 822              echo $params['after'];
 823          }
 824          else
 825          {    // Display placeholder if no images:
 826              // Replace mask $item_permaurl$ with the item permanent URL:
 827              echo str_replace( '$item_permaurl$', $disp_Item->get_permanent_url(), $params['placeholder'] );

 857          {    // If item title is displayed:
 858              $block_css_class .= ' evo_withtitle';

 886                  $block_css_class .= ' evo_1pic__normal';

 892  
 893          if( $this->disp_params['disp_excerpt'] )
 894          {    // If item excerpt is displayed:

 903          {    // If item teaser is displayed:

 905          }
 906          else
 907          {    // Item teaser is hidden:
