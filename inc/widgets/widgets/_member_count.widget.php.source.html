  92      /**
  93       * Get a count of blog members
  94       *
  95       * @return integer Members count
  96       */
  97  	function get_members_count()
  98      {
  99          global $Blog, $DB;
 100  
 101          // Get blog owner
 102          $blogowner_SQL = new SQL();
 103          $blogowner_SQL->SELECT( 'user_ID, user_status' );
 104          $blogowner_SQL->FROM( 'T_users' );
 105          $blogowner_SQL->FROM_add( 'INNER JOIN T_blogs ON blog_owner_user_ID = user_ID' );
 106          $blogowner_SQL->WHERE( 'blog_ID = '.$DB->quote( $Blog->ID ) );
 107  
 108          // Calculate what users are members of the blog
 109          $userperms_SQL = new SQL();
 110          $userperms_SQL->SELECT( 'user_ID, user_status' );
 111          $userperms_SQL->FROM( 'T_users' );
 112          $userperms_SQL->FROM_add( 'INNER JOIN T_coll_user_perms ON ( bloguser_user_ID = user_ID AND bloguser_ismember = 1 )' );
 113          $userperms_SQL->WHERE( 'bloguser_blog_ID = '.$DB->quote( $Blog->ID ) );
 114  
 115          // Calculate what user groups are members of the blog
 116          $usergroups_SQL = new SQL();
 117          $usergroups_SQL->SELECT( 'user_ID, user_status' );
 118          $usergroups_SQL->FROM( 'T_users' );
 119          $usergroups_SQL->FROM_add( 'INNER JOIN T_groups ON grp_ID = user_grp_ID' );
 120          $usergroups_SQL->FROM_add( 'LEFT JOIN T_coll_group_perms ON ( bloggroup_group_ID = grp_ID AND bloggroup_ismember = 1 )' );
 121          $usergroups_SQL->WHERE( 'bloggroup_blog_ID = '.$DB->quote( $Blog->ID ) );
 122  
 123          $members_count_sql = 'SELECT COUNT( user_ID ) AS member_count FROM ( '
 124              .$blogowner_SQL->get()
 125              .' UNION '
 126              .$userperms_SQL->get()
 127              .' UNION '
 128              .$usergroups_SQL->get().' ) AS members '
 129              .'WHERE members.user_status != "closed"';
 130  
 131          return intval( $DB->get_var( $members_count_sql ) );
 132      }
