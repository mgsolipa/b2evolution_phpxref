 199      /**
 200       * Display the widget!
 201       *
 202       * @param array MUST contain at least the basic display params
 203       */
 204  	function display( $params )
 205      {
 206          global $localtimenow, $DB, $Blog;
 207  
 208          $this->init_display( $params );
 209  
 210          $blog_ID = intval( $this->disp_params['blog_ID'] );
 211          if( empty( $blog_ID ) )
 212          { // Use current blog by default
 213              $blog_ID = $Blog->ID;
 214          }
 215  
 216          $BlogCache = & get_BlogCache();
 217          if( ! $BlogCache->get_by_ID( $blog_ID, false, false ) )
 218          { // No blog exists
 219              return;
 220          }
 221  
 222          // Display photos:
 223          // TODO: permissions, complete statuses...
 224          // TODO: A FileList object based on ItemListLight but adding File data into the query?
 225          //          overriding ItemListLigth::query() for starters ;)
 226  
 227          // Init caches
 228          $FileCache = & get_FileCache();
 229          $ItemCache = & get_ItemCache();
 230  
 231          // Query list of files and posts fields:
 232          // Note: We use ItemQuery to get attachments from all posts which should be visible ( even in case of aggregate blogs )
 233          $ItemQuery = new ItemQuery( $ItemCache->dbtablename, $ItemCache->dbprefix, $ItemCache->dbIDname );
 234          $ItemQuery->SELECT( 'post_ID, post_datestart, post_datemodified, post_main_cat_ID, post_urltitle, post_canonical_slug_ID,
 235                                      post_tiny_slug_ID, post_ityp_ID, post_title, post_excerpt, post_url, file_ID, file_type,
 236                                      file_title, file_root_type, file_root_ID, file_path, file_alt, file_desc, file_path_hash' );
 237          $ItemQuery->FROM_add( 'INNER JOIN T_links ON post_ID = link_itm_ID' );
 238          $ItemQuery->FROM_add( 'INNER JOIN T_files ON link_file_ID = file_ID' );
 239          $ItemQuery->where_chapter( $blog_ID );
 240          if( $this->disp_params['item_visibility'] == 'public' )
 241          { // Get images only of the public items
 242              $ItemQuery->where_visibility( array( 'published' ) );
 243          }
 244          else
 245          { // Get image of all available posts for current user
 246              $ItemQuery->where_visibility( NULL );
 247          }
 248          $ItemQuery->WHERE_and( '( file_type = "image" ) OR ( file_type IS NULL )' );
 249          $ItemQuery->WHERE_and( 'post_datestart <= \''.remove_seconds( $localtimenow ).'\'' );
 250          $ItemQuery->WHERE_and( 'link_position != "cover"' );
 251          if( !empty( $this->disp_params['item_type'] ) )
 252          { // Get items only with specified type
 253              $ItemQuery->WHERE_and( 'post_ityp_ID = '.intval( $this->disp_params['item_type'] ) );
 254          }
 255          $ItemQuery->GROUP_BY( 'link_ID' );
 256  
 257          // fp> TODO: because no way of getting images only, we get 4 times more data than requested and hope that 25% at least will be images :/
 258          // asimo> This was updated and we get images and those files where we don't know the file type yet. Now we get 2 times more data than requested.
 259          // Maybe it would be good to get only the requested amount of files, because after a very short period the file types will be set for all images.
 260          $ItemQuery->LIMIT( intval( $this->disp_params['limit'] ) * 2 );
 261  
 262          $ItemQuery->ORDER_BY(    gen_order_clause( $this->disp_params['order_by'], $this->disp_params['order_dir'],
 263                                              'post_', 'post_ID '.$this->disp_params['order_dir'].', link_ID' ) );
 264  
 265          // Init FileList with the above defined query
 266          $FileList = new DataObjectList2( $FileCache );
 267          $FileList->sql = $ItemQuery->get();
 268  
 269          $FileList->query( false, false, false, 'Media index widget' );
 270  
 271          $layout = $this->disp_params['thumb_layout'];
 272  
 273          $nb_cols = $this->disp_params['grid_nb_cols'];
 274          $count = 0;
 275          $r = '';
 276          /**
 277           * @var File
 278           */
 279          while( $File = & $FileList->get_next() )
 280          {
 281              if( $count >= $this->disp_params['limit'] )
 282              {    // We have enough images already!
 283                  break;
 284              }
 285  
 286              if( ! $File->is_image() )
 287              { // Skip anything that is not an image
 288                  // Only images are selected or those files where we don't know the file type yet.
 289                  // This check is only for those files where we don't know the filte type. The file type will be set during the check.
 290                  continue;
 291              }
 292  
 293              if( $layout == 'grid' )
 294              { // Grid layout
 295                  if( $count % $nb_cols == 0 )
 296                  {
 297                      $r .= $this->disp_params['grid_colstart'];
 298                  }
 299                  $r .= $this->disp_params['grid_cellstart'];
 300              }
 301              elseif( $layout == 'flow' )
 302              { // Flow block layout
 303                  $r .= $this->disp_params['flow_block_start'];
 304              }
 305              else
 306              { // List layout
 307                  $r .= $this->disp_params['item_start'];
 308              }
 309  
 310              // 1/ Hack a dirty permalink( will redirect to canonical):
 311              // $link = url_add_param( $Blog->get('url'), 'p='.$post_ID );
 312  
 313              // 2/ Hack a link to the right "page". Very daring!!
 314              // $link = url_add_param( $Blog->get('url'), 'paged='.$count );
 315  
 316              // 3/ Instantiate a light object in order to get permamnent url:
 317              $ItemLight = new ItemLight( $FileList->get_row_by_idx( $FileList->current_idx - 1 ) );    // index had already been incremented
 318  
 319              $r .= '<a href="'.$ItemLight->get_permanent_url().'">';
 320              // Generate the IMG THUMBNAIL tag with all the alt, title and desc if available
 321              $r .= $File->get_thumb_imgtag( $this->disp_params['thumb_size'], '', '', $ItemLight->title );
 322              $r .= '</a>';
 323              if( $this->disp_params['disp_image_title'] )
 324              { // Dislay title of image or item
 325                  $title = ( $File->get( 'title' ) ) ? $File->get( 'title' ) : $ItemLight->title;
 326                  if( ! empty( $title ) )
 327                  {
 328                      $r .= '<span class="note">'.$title.'</span>';
 329                  }
 330              }
 331  
 332              ++$count;
 333  
 334              if( $layout == 'grid' )
 335              { // Grid layout
 336                  $r .= $this->disp_params['grid_cellend'];
 337                  if( $count % $nb_cols == 0 )
 338                  {
 339                      $r .= $this->disp_params['grid_colend'];
 340                  }
 341              }
 342              elseif( $layout == 'flow' )
 343              { // Flow block layout
 344                  $r .= $this->disp_params['flow_block_end'];
 345              }
 346              else
 347              { // List layout
 348                  $r .= $this->disp_params['item_end'];
 349              }
 350          }
 351  
 352          // Exit if no files found
 353          if( empty($r) ) return;
 354  
 355          echo $this->disp_params['block_start'];
 356  
 357          // Display title if requested
 358          $this->disp_title();
 359  
 360          echo $this->disp_params['block_body_start'];
 361  
 362          if( $layout == 'grid' )
 363          { // Grid layout
 364              echo $this->disp_params['grid_start'];
 365          }
 366          elseif( $layout == 'flow' )
 367          { // Flow block layout
 368              echo $this->disp_params['flow_start'];
 369          }
 370          else
 371          { // List layout
 372              echo $this->disp_params['list_start'];
 373          }
 374  
 375          echo $r;
 376  
 377          if( $layout == 'grid' )
 378          { // Grid layout
 379              if( $count && ( $count % $nb_cols != 0 ) )
 380              {
 381                  echo $this->disp_params['grid_colend'];
 382              }
 383  
 384              echo $this->disp_params['grid_end'];
 385          }
 386          elseif( $layout == 'flow' )
 387          { // Flow block layout
 388              echo $this->disp_params['flow_end'];
 389          }
 390          else
 391          { // List layout
 392              echo $this->disp_params['list_end'];
 393          }
 394  
 395          echo $this->disp_params['block_body_end'];
 396  
 397          echo $this->disp_params['block_end'];
 398  
 399          return true;
 400      }

 419                  'wi_ID'         => $this->ID, // Have the widget settings changed?
 420                  'set_coll_ID'   => $Blog->ID, // Have the settings of the blog changed? (ex: new skin)
 421                  'cont_coll_ID'  => $blog_ID,  // Has the content of the displayed blog changed?
 422                  'media_coll_ID' => $blog_ID,  // Have some media files attached to one of the blogs item?
 423              );
