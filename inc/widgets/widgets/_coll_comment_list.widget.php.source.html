 129      /**
 130       * Display the widget!
 131       *
 132       * @param array MUST contain at least the basic display params
 133       */
 134  	function display( $params )
 135      {
 136          global $Blog;
 137  
 138          $this->init_display( $params );
 139  
 140          $blog_ID = intval( $this->disp_params['blog_ID'] );
 141  
 142          $blogCache = & get_BlogCache();
 143          $listBlog = ( $blog_ID ? $blogCache->get_by_ID( $blog_ID ) : $Blog );
 144  
 145          // Create ItemList
 146          // Note: we pass a widget specific prefix in order to make sure to never interfere with the mainlist
 147          $limit = intval( $this->disp_params['limit'] );
 148          $order = $this->disp_params['disp_order'];
 149  
 150          $CommentList = new CommentList2( $listBlog, $limit, 'CommentCache', $this->code.'_' );
 151  
 152          $filters = array(
 153                  'types' => array( 'comment','trackback','pingback' ),
 154                  'statuses' => array( 'published' ),
 155                  'order' => $order,
 156                  'comments' => $limit,
 157              );
 158  
 159          if( isset( $this->disp_params['page'] ) )
 160          {
 161              $filters['page'] = $this->disp_params['page'];
 162          }
 163  
 164          // Filter list:
 165          $CommentList->set_filters( $filters );
 166  
 167          // Get ready for display (runs the query):
 168          $CommentList->display_init();
 169  
 170          echo $this->disp_params[ 'block_start'];
 171  
 172          // Display title if requested
 173          $this->disp_title();
 174  
 175          echo $this->disp_params['block_body_start'];
 176  
 177          echo $this->disp_params[ 'list_start' ];
 178  
 179          if( empty( $this->disp_params[ 'author_link_text' ] ) )
 180          {
 181              $this->disp_params[ 'author_link_text' ] = 'login';
 182          }
 183  
 184          /**
 185           * @var Comment
 186           */
 187          while( $Comment = & $CommentList->get_next() )
 188          { // Loop through comments:
 189              // Load comment's Item object:
 190              $Comment->get_Item();
 191              echo $this->disp_params[ 'item_start' ];
 192              $Comment->author( '', ' ', '', ' ', 'htmlbody', $this->disp_params[ 'author_links' ], $this->disp_params[ 'author_link_text' ] );
 193              echo T_( 'on ' );
 194              $Comment->permanent_link( array(
 195                  'text'        => $Comment->Item->title,
 196                  'title'       => $this->disp_params[ 'hover_text' ],
 197                  ) );
 198              echo $this->disp_params[ 'item_end' ];
 199          }    // End of comment loop.}
 200  
 201          if( isset( $this->disp_params['page'] ) )
 202          {
 203              if( empty( $this->disp_params['pagination'] ) )
 204              {
 205                  $this->disp_params['pagination'] = array();
 206              }
 207              $CommentList->page_links( $this->disp_params['pagination'] );
 208          }
 209  
 210          echo $this->disp_params[ 'list_end' ];
 211  
 212          echo $this->disp_params['block_body_end'];
 213  
 214          echo $this->disp_params[ 'block_end' ];
 215  
 216          return true;
 217      }
