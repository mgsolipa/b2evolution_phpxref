 170      /**
 171       * Display the widget!
 172       *
 173       * @param array MUST contain at least the basic display params
 174       */
 175  	function display( $params )
 176      {
 177          $this->init_display( $params );
 178  
 179          global $blog;
 180  
 181          $BlogCache = & get_BlogCache();
 182  
 183          // Source collections:
 184          // Get a list of quoted blog IDs
 185          $blog_ids = sanitize_id_list($this->disp_params['blog_ids'], true);
 186          if( empty($blog) && empty($blog_ids) )
 187          {    // Nothing to display
 188              return;
 189          }
 190          elseif( empty($blog_ids) )
 191          {    // Use current Blog
 192              $blog_ids = $blog;
 193          }
 194  
 195          // Destination:
 196          if( $destination_coll_ID = $this->disp_params['destination_coll_ID'] )
 197          {    // Get destination Colelction, but allow error, in that case we'll get NULL
 198              $destination_Blog = $BlogCache->get_by_ID( $destination_coll_ID, false );
 199          }
 200          else
 201          { // Auto destination:
 202              $destination_Blog = NULL;
 203          }
 204  
 205          $results = get_tags( $blog_ids, $this->disp_params['max_tags'], /* $this->disp_params['filter_list'] */ NULL, false );
 206  
 207          if( empty($results) )
 208          {    // No tags!
 209              return;
 210          }
 211  
 212          $max_count = $results[0]->tag_count;
 213          $min_count = $results[count($results)-1]->tag_count;
 214          $count_span = max( 1, $max_count - $min_count );
 215          $max_size = $this->disp_params['tag_max_size'];
 216          $min_size = $this->disp_params['tag_min_size'];
 217          $size_span = $max_size - $min_size;
 218  
 219          if($this->disp_params['tag_ordering'] == 'ASC')
 220          {
 221              usort($results, array($this, 'tag_cloud_cmp'));
 222          }
 223          elseif($this->disp_params['tag_ordering'] == 'RAND')
 224          {
 225              shuffle( $results );
 226          }
 227  
 228          echo $this->disp_params['block_start'];
 229  
 230          $this->disp_title();
 231  
 232          echo $this->disp_params['block_body_start'];
 233  
 234          echo $this->disp_params['tag_cloud_start'];
 235  
 236          $count = 0;
 237          foreach( $results as $row )
 238          {
 239              if( $count > 0 )
 240              {
 241                  echo $this->disp_params['tag_separator'];
 242              }
 243  
 244              // If there's a space in the tag name, quote it:
 245              $tag_name_disp = strpos($row->tag_name, ' ')
 246                  ? '&laquo;'.format_to_output($row->tag_name, 'htmlbody').'&raquo;'
 247                  : format_to_output($row->tag_name, 'htmlbody');
 248  
 249              $font_size = floor( $row->tag_count * $size_span / $count_span + $min_size );
 250  
 251              if( !is_null($destination_Blog) )
 252              {
 253                  $l_Blog = $destination_Blog;
 254              }
 255              else
 256              {    // Automatic destination decision. Note: this may not be be best decision.
 257                  $l_Blog = $BlogCache->get_by_ID( $row->cat_blog_ID );
 258              }
 259  
 260              echo $l_Blog->get_tag_link( $row->tag_name, $tag_name_disp, array(
 261                  'style' => 'font-size:'.$font_size.'pt;',
 262                  'title' => sprintf( T_('Display posts tagged with "%s"'), $row->tag_name ) ) );
 263              $count++;
 264          }
 265          echo $this->disp_params['tag_cloud_end'];
 266  
 267          echo $this->disp_params['block_body_end'];
 268  
 269          echo $this->disp_params['block_end'];
 270  
 271          return true;
 272      }
