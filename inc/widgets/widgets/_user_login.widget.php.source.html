 195      /**
 196       * Display the widget!
 197       *
 198       * @param array MUST contain at least the basic display params
 199       */
 200  	function display( $params )
 201      {
 202          global $Blog;
 203  
 204          if( get_param( 'disp' ) == 'login' )
 205          {    // No display a duplicate form for inskin login mode
 206              return false;
 207          }
 208  
 209          $this->init_display( $params );
 210  
 211          if( isset( $this->BlockCache ) )
 212          {    // Do NOT cache some of these links are using a redirect_to param, which makes it page dependent.
 213              // Note: also beware of the source param.
 214              // so this will be cached by the PageCache; there is no added benefit to cache it in the BlockCache
 215              // (which could have been shared between several pages):
 216              $this->BlockCache->abort_collect();
 217          }
 218  
 219          echo $this->disp_params['block_start'];
 220  
 221          if( ! is_logged_in() )
 222          { // Login form:
 223              global $Settings;
 224  
 225              $source = 'user_login_widget';
 226  
 227              $redirect_to = $Settings->get( 'redirect_to_after_login' );
 228              if( empty( $redirect_to ) )
 229              {
 230                  $redirect_to = regenerate_url( '', '', '', '&' );
 231              }
 232  
 233              $this->disp_title();
 234  
 235              echo $this->disp_params['block_body_start'];
 236  
 237              // display widget login form
 238              require skin_template_path( '_widget_login.form.php' );
 239  
 240              echo $this->disp_params['block_body_end'];
 241          }
 242          else
 243          { // Display a greeting text
 244              global $current_User;
 245  
 246              echo $this->disp_params['block_body_start'];
 247  
 248              if( $this->get_param('profile_picture_size') != '' )
 249              {    // Display profile picture
 250                  echo $current_User->get_avatar_imgtag( $this->disp_params['profile_picture_size'], 'avatar', 'middle' );
 251              }
 252  
 253              if( $this->get_param('greeting_show') )
 254              {    // Display greeting text
 255                  $user_login = $current_User->get_identity_link( array( 'link_text' => 'name', 'display_bubbletip' => false ) );
 256                  echo ' <strong class="greeting">'.str_replace( '$login$', $user_login, $this->get_param('greeting_text') ).'</strong>';
 257              }
 258  
 259              if( $this->get_param('group_show') )
 260              {    // Display user group
 261                  $user_Group = $current_User->get_Group();
 262                  echo '<p class="user_group">'
 263                      .str_replace( '$group$', $user_Group->get( 'name' ), $this->get_param('group_text') )
 264                      .'</p>';
 265              }
 266  
 267              if( $this->get_param('level_show') )
 268              {    // Display user group
 269                  echo '<p class="user_level">'
 270                      .str_replace( '$level$', $current_User->get( 'level' ), $this->get_param('level_text') )
 271                      .'</p>';
 272              }
 273  
 274              echo $this->disp_params['block_body_end'];
 275          }
 276  
 277          echo $this->disp_params['block_end'];
 278  
 279          return true;
