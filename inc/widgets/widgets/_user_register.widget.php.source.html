 156      /**
 157       * Display the widget!
 158       *
 159       * @param array MUST contain at least the basic display params
 160       */
 161  	function display( $params )
 162      {
 163          global $Blog, $Settings, $Session, $redirect_to, $dummy_fields;
 164  
 165          if( is_logged_in() )
 166          {    // No display when user is already registered
 167              return false;
 168          }
 169  
 170          if( $Settings->get( 'newusers_canregister' ) != 'yes' || ! $Settings->get( 'quick_registration' ) )
 171          { // Display error message when quick registration is disabled
 172              echo '<p class="error">'.T_('Quick registration is currently disabled on this system.').'</p>';
 173              return false;
 174          }
 175  
 176          // Restore the typed in params from the redirected page:
 177          $widget_param_input_err_messages = $Session->get( 'param_input_err_messages_'.$this->ID );
 178          $widget_param_input_values = $Session->get( 'param_input_values_'.$this->ID );
 179          if( ! empty( $widget_param_input_err_messages ) )
 180          { // Convert param errors to global $param_input_err_messages that is used to display an error text under input field
 181              global $param_input_err_messages;
 182              $param_input_err_messages = $widget_param_input_err_messages;
 183          }
 184          // Clear the temp session vars
 185          $Session->delete( 'param_input_err_messages_'.$this->ID );
 186          $Session->delete( 'param_input_values_'.$this->ID );
 187          $Session->dbsave();
 188  
 189          $this->init_display( $params );
 190  
 191          if( isset( $this->BlockCache ) )
 192          {    // Do NOT cache some of these links are using a redirect_to param, which makes it page dependent.
 193              // Note: also beware of the source param.
 194              // so this will be cached by the PageCache; there is no added benefit to cache it in the BlockCache
 195              // (which could have been shared between several pages):
 196              $this->BlockCache->abort_collect();
 197          }
 198  
 199          echo $this->disp_params['block_start'];
 200  
 201          $redirect_to = param( 'redirect_to', 'url', regenerate_url( '', '', '', '&' ) );
 202  
 203          $this->disp_title();
 204  
 205          echo $this->disp_params['block_body_start'];
 206  
 207          if( ! empty( $this->disp_params['intro'] ) )
 208          { // Intro text
 209              echo '<p>'.$this->disp_params['intro'].'</p>';
 210          }
 211  
 212          $Form = new Form( get_secure_htsrv_url().'register.php', 'register_form', 'post' );
 213  
 214          $Form->begin_form();
 215  
 216          $Form->add_crumb( 'regform' );
 217          $Form->hidden( 'action', 'quick_register' );
 218          $Form->hidden( 'inskin', true );
 219          $Form->hidden( 'blog', $Blog->ID );
 220          $Form->hidden( 'widget', $this->ID );
 221          $Form->hidden( 'redirect_to', $redirect_to );
 222  
 223          // E-mail
 224          $email_value = isset( $widget_param_input_values[ $dummy_fields['email'] ] ) ? $widget_param_input_values[ $dummy_fields['email'] ] : '';
 225          $Form->text_input( $dummy_fields['email'], $email_value, 50, T_('Your email'), '', array( 'maxlength' => 255, 'class' => 'input_text', 'required' => true ) );
 226  
 227          if( $this->disp_params['ask_firstname'] != 'no' )
 228          { // First name
 229              $firstname_value = isset( $widget_param_input_values['firstname'] ) ? $widget_param_input_values['firstname'] : '';
 230              $Form->text_input( 'firstname', $firstname_value, 18, T_('First name'), '', array( 'maxlength' => 50, 'class' => 'input_text', 'required' => ( $this->disp_params['ask_firstname'] == 'required' ) ) );
 231          }
 232  
 233          if( $this->disp_params['ask_lastname'] != 'no' )
 234          { // Last name
 235              $lastname_value = isset( $widget_param_input_values['lastname'] ) ? $widget_param_input_values['lastname'] : '';
 236              $Form->text_input( 'lastname', $lastname_value, 18, T_('Last name'), '', array( 'maxlength' => 50, 'class' => 'input_text', 'required' => ( $this->disp_params['ask_lastname'] == 'required' ) ) );
 237          }
