  96  	function get_param_definitions( $params )
  97      {
  98          global $msg_menu_link_widget_link_types;
  99  
 100          $r = array_merge( array(
 101                  'link_type' => array(
 102                      'label' => T_( 'Link Type' ),
 103                      'note' => T_('What do you want to link to?'),
 104                      'type' => 'select',
 105                      'options' => $msg_menu_link_widget_link_types,
 106                      'defaultvalue' => 'messages',
 107                      'onchange' => '
 108                          var curr_link_type = this.value;
 109                          var show_badge = jQuery("[id$=\'_set_show_badge\']");
 110                          if( curr_link_type == "messages" )
 111                          {
 112                              show_badge.removeAttr(\'disabled\');
 113                              show_badge.attr( \'checked\', \'checked\' );
 114                          }
 115                          else
 116                          {
 117                              show_badge.attr( \'disabled\', \'disabled\' );
 118                              show_badge.removeAttr(\'checked\');
 119                          };'
 120                  ),
 121                  'link_text' => array(
 122                      'label' => T_( 'Link text' ),
 123                      'note' => T_('Text to use for the link (leave empty for default).'),
 124                      'type' => 'text',
 125                      'size' => 20,
 126                      'defaultvalue' => '',
 127                  ),
 128                  'blog_ID' => array(
 129                      'label' => T_('Collection ID'),
 130                      'note' => T_('Leave empty for current collection.'),
 131                      'type' => 'integer',
 132                      'allow_empty' => true,
 133                      'size' => 5,
 134                      'defaultvalue' => '',
 135                  ),
 136                  'show_to' => array(
 137                      'label' => T_( 'Show to' ),
 138                      'note' => '',
 139                      'type' => 'radio',
 140                      'options' => array( array( 'any', T_( 'All users') ),
 141                                          array( 'loggedin', T_( 'Logged in users' ) ),
 142                                          array( 'perms', T_( 'Users with messaging permissions only' ) ) ),
 143                      'defaultvalue' => 'perms',
 144                  ),
 145                  'show_badge' => array(
 146                      'label' => T_( 'Show Badge' ),
 147                      'note' => T_( 'Show a badge with the count of unread messages.' ),
 148                      'type' => 'checkbox',
 149                      'defaultvalue' => true,
 150                  ),
 151              ), parent::get_param_definitions( $params )    );
 152  
 153          // Do not modify anything during update because the editing form contains all of the required modifications
 154          if( !isset( $params['for_updating'] ) )
 155          { // Not called from the update process
 156              // Turn off allow blockcache by default, because it is forbidden in case of messages
 157              // Note: we may call $this->get_param() only if this function was not called from there. This way we prevent infinite recursion/loop.
 158              $link_type = ( empty( $this->params ) || isset( $params['infinite_loop'] ) ) ? 'messages' : $this->get_param( 'link_type', true );

 161                  $r['show_badge']['defaultvalue'] = false;
 162                  $r['show_badge']['disabled'] = 'disabled';
 163              }

 166          if( isset( $r['allow_blockcache'] ) )
 167          { // Disable "allow blockcache" because this widget uses the selected items
 168              $r['allow_blockcache']['defaultvalue'] = false;
 169              $r['allow_blockcache']['disabled'] = 'disabled';
 170              $r['allow_blockcache']['note'] = T_('This widget cannot be cached in the block cache.');
 171          }

 173          return $r;

 191      /**
 192       * Display the widget!
 193       *
 194       * @param array MUST contain at least the basic display params
 195       */
 196  	function display( $params )
 197      {
 198          global $current_User, $unread_messages_count;
 199          global $disp;
 200  
 201          $this->init_display( $params );
 202  
 203          $blog_ID = intval( $this->disp_params['blog_ID'] );
 204          if( $blog_ID > 0 )
 205          { // Try to use blog from widget setting
 206              $BlogCache = & get_BlogCache();
 207              $current_Blog = & $BlogCache->get_by_ID( $blog_ID, false, false );
 208          }
 209  
 210          if( empty( $current_Blog ) )
 211          { // Blog is not defined in setting or it doesn't exist in DB
 212              global $Blog;
 213              // Use current blog
 214              $current_Blog = & $Blog;
 215          }
 216  
 217          if( empty( $current_Blog ) )
 218          { // Don't use this widget without current collection:
 219              return false;
 220          }
 221  
 222          switch( $this->disp_params['show_to'] )
 223          {
 224              case 'any':
 225                  break;
 226              case 'loggedin':
 227                  if( !is_logged_in() )
 228                  {
 229                      return false;
 230                  }
 231                  break;
 232              case 'perms':
 233                  if( !is_logged_in() || !$current_User->check_perm( 'perm_messaging', 'reply', false ) )
 234                  {
 235                      return false;
 236                  }
 237                  break; // display
 238              case 'default':
 239                  debug_die( 'Invalid params!' );
 240          }
 241  
 242          // Default link class
 243          $link_class = $this->disp_params['link_default_class'];
 244  
 245          switch( $this->disp_params[ 'link_type' ] )
 246          {
 247              case 'messages':
 248                  $url = $current_Blog->get( 'threadsurl' );
 249                  $text = T_( 'Messages' );
 250                  // set allow blockcache to 0, this way make sure block cache is never allowed for messages
 251                  $this->disp_params[ 'allow_blockcache' ] = 0;
 252                  // Is this the current display?
 253                  if( ( $disp == 'threads' && ( ! isset( $_GET['disp'] ) || $_GET['disp'] != 'msgform' ) ) || $disp == 'messages' )
 254                  { // The current page is currently displaying the messages:
 255                      // Let's display it as selected
 256                      $link_class = $this->disp_params['link_selected_class'];
 257                  }
 258                  break;
 259  
 260              case 'contacts':
 261                  $url = $current_Blog->get( 'contactsurl' );
 262                  $text = T_( 'Contacts' );
 263                  // set show badge to 0, this way make sure badge won't be displayed
 264                  $this->disp_params[ 'show_badge' ] = 0;
 265                  // Is this the current display?
 266                  if( $disp == 'contacts' )
 267                  { // The current page is currently displaying the contacts:
 268                      // Let's display it as selected
 269                      $link_class = $this->disp_params['link_selected_class'];
 270                  }
 271                  break;
 272          }
 273  
 274          if( !empty( $this->disp_params[ 'link_text' ] ) )
 275          {
 276              $text = $this->disp_params[ 'link_text' ];
 277          }
 278  
 279          if( ( $this->disp_params[ 'show_badge' ] ) && ( $unread_messages_count > 0 ) )
 280          {
 281              $badge = ' <span class="badge badge-important">'.$unread_messages_count.'</span>';
 282              if( isset( $this->BlockCache ) )
 283              { // Do not cache if bage is displayed because the number of unread messages are always changing
 284                  $this->BlockCache->abort_collect();
 285              }
 286          }
 287          else
 288          {
 289              $badge = '';
 290          }
 291  
 292          echo $this->disp_params['block_start'];
 293          echo $this->disp_params['block_body_start'];
 294          echo $this->disp_params['list_start'];
 295  
 296          if( $link_class == $this->disp_params['link_selected_class'] )
 297          {
 298              echo $this->disp_params['item_selected_start'];
 299          }
 300          else
 301          {
 302              echo $this->disp_params['item_start'];
 303          }
 304          echo '<a href="'.$url.'" class="'.$link_class.'">'.$text.$badge.'</a>';
 305          if( $link_class == $this->disp_params['link_selected_class'] )
 306          {
 307              echo $this->disp_params['item_selected_end'];
 308          }
 309          else
 310          {
 311              echo $this->disp_params['item_end'];
 312          }
 313  
 314          echo $this->disp_params['list_end'];
 315          echo $this->disp_params['block_body_end'];
 316          echo $this->disp_params['block_end'];
 317  
 318          return true;
 319      }
