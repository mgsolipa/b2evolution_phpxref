  95      /**
  96       * Display the widget!
  97       *
  98       * @param array MUST contain at least the basic display params
  99       */
 100  	function display( $params )
 101      {
 102          global $DB, $Settings, $UserSettings, $localtimenow;
 103  
 104          if( ( !$this->get_param( 'allow_anonymous' ) ) && ( !is_logged_in() ) )
 105          { // display only for logged in users
 106              return;
 107          }
 108  
 109          // load online Users
 110          $UserCache = & get_UserCache();
 111          $online_threshold = $localtimenow - ( 2 * $Settings->get( 'timeout_online' ) );
 112          $UserCache->load_where( 'user_lastseen_ts > '.$DB->quote( date2mysql( $online_threshold ).' AND user_status <> '.$DB->quote( 'closed' ) ) );
 113  
 114          $this->init_display( $params );
 115  
 116          // START DISPLAY:
 117          echo $this->disp_params['block_start'];
 118  
 119          // Display title if requested
 120          $this->disp_title();
 121  
 122          echo $this->disp_params['block_body_start'];
 123  
 124          $r = '';
 125  
 126          while( ( $iterator_User = & $UserCache->get_next() ) != NULL )
 127          { // Iterate through UserCache
 128              $user_lastseen_ts = mysql2timestamp( $iterator_User->get( 'lastseen_ts' ) );
 129              if( ( $user_lastseen_ts > $online_threshold )
 130                  && ( $UserSettings->get( 'show_online', $iterator_User->ID ) )
 131                  && ( !$iterator_User->check_status( 'is_closed' ) ) )
 132              {
 133                  if( empty($r) )
 134                  { // first user
 135                      $r .= $params['list_start'];
 136                  }
 137  
 138                  $r .= $params['item_start'];
 139                  $r .= $iterator_User->get_identity_link( array( 'login_mask' => '$login$' ) );
 140                  $r .= $params['item_end'];
 141              }
 142          }
 143  
 144          if( !empty( $r ) )
 145          {
 146              $r .= $params['list_end'];
 147              echo $r;
 148          }
 149  
 150          echo $this->disp_params['block_body_end'];
 151  
 152          echo $this->disp_params['block_end'];
 153  
 154          return true;
 155      }
