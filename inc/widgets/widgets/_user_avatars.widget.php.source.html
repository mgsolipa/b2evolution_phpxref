 183      {
 184          $this->init_display( $params );
 185  
 186          $UserCache = & get_UserCache();
 187  
 188          $UserList = new DataObjectList2( $UserCache );
 189  
 190          switch( $this->disp_params[ 'order_by' ] )
 191          {
 192              case 'regdate':
 193                  $sql_order = 'user_created_datetime DESC';
 194                  break;
 195              case 'moddate':
 196                  $sql_order = 'user_profileupdate_date DESC';
 197                  break;
 198              case 'random':
 199              default:
 200                  $sql_order = 'RAND()';
 201                  break;
 202          }
 203  
 204          // Query list of users with picture and not closed:
 205          $SQL = new SQL();
 206          $SQL->SELECT( '*' );
 207          $SQL->FROM( 'T_users' );
 208          $SQL->WHERE( 'user_avatar_file_ID IS NOT NULL' );
 209          $SQL->WHERE_and( 'user_status <> "closed"' );
 210          if( is_logged_in() )
 211          { // Add filters
 212              global $current_User, $DB;
 213              switch( $this->disp_params[ 'gender' ] )
 214              { // Filter by gender
 215                  case 'same':
 216                      $SQL->WHERE_and( 'user_gender = "'.$current_User->gender.'"' );
 217                      break;
 218                  case 'opposite':
 219                      $SQL->WHERE_and( 'user_gender != "'.$current_User->gender.'"' );
 220                      break;
 221              }
 222              switch( $this->disp_params[ 'location' ] )
 223              { // Filter by location
 224                  case 'city':
 225                      $SQL->WHERE_and( 'user_city_ID '.( empty( $current_User->city_ID ) ? 'IS NULL' : '= "'.$current_User->city_ID.'"' ) );
 226                  case 'subregion':
 227                      $SQL->WHERE_and( 'user_subrg_ID '.( empty( $current_User->subrg_ID  ) ? 'IS NULL' : '= "'.$current_User->subrg_ID .'"' ) );
 228                  case 'region':
 229                      $SQL->WHERE_and( 'user_rgn_ID '.( empty( $current_User->rgn_ID  ) ? 'IS NULL' : '= "'.$current_User->rgn_ID .'"' ) );
 230                  case 'country':
 231                      $SQL->WHERE_and( 'user_ctry_ID '.( empty( $current_User->ctry_ID ) ? 'IS NULL' : '= "'.$current_User->ctry_ID.'"' ) );
 232                      break;
 233                  case 'closest':
 234                      if( !empty( $current_User->city_ID ) )
 235                      { // Check if users exist with same city
 236                          $user_exists = $DB->get_var( 'SELECT user_ID
 237                               FROM T_users
 238                              WHERE user_city_ID ="'.$current_User->city_ID.'"
 239                                AND user_ID != "'.$current_User->ID.'"
 240                              LIMIT 1' );
 241                          if( !empty( $user_exists ) )
 242                          {
 243                              $SQL->WHERE_and( 'user_city_ID = "'.$current_User->city_ID.'"' );
 244                              $SQL->WHERE_and( 'user_subrg_ID = "'.$current_User->subrg_ID .'"' );
 245                              $SQL->WHERE_and( 'user_rgn_ID = "'.$current_User->rgn_ID .'"' );
 246                              $SQL->WHERE_and( 'user_ctry_ID = "'.$current_User->ctry_ID.'"' );
 247                              break;
 248                          }
 249                      }

 251                      { // Check if users exist with same sub-region
 252                          $user_exists = $DB->get_var( 'SELECT user_ID
 253                               FROM T_users
 254                              WHERE user_subrg_ID ="'.$current_User->subrg_ID.'"
 255                                AND user_ID != "'.$current_User->ID.'"
 256                              LIMIT 1' );
 257                          if( !empty( $user_exists ) )
 258                          {
 259                              $SQL->WHERE_and( 'user_subrg_ID = "'.$current_User->subrg_ID .'"' );
 260                              $SQL->WHERE_and( 'user_rgn_ID = "'.$current_User->rgn_ID .'"' );
 261                              $SQL->WHERE_and( 'user_ctry_ID = "'.$current_User->ctry_ID.'"' );
 262                              break;
 263                          }
 264                      }
 265                      if( !empty( $current_User->rgn_ID ) && empty( $user_exists ) )
 266                      { // Check if users exist with same region
 267                          $user_exists = $DB->get_var( 'SELECT user_ID
 268                               FROM T_users
 269                              WHERE user_rgn_ID ="'.$current_User->rgn_ID.'"
 270                                AND user_ID != "'.$current_User->ID.'"
 271                              LIMIT 1' );
 272                          if( !empty( $user_exists ) )
 273                          {
 274                              $SQL->WHERE_and( 'user_rgn_ID = "'.$current_User->rgn_ID .'"' );
 275                              $SQL->WHERE_and( 'user_ctry_ID = "'.$current_User->ctry_ID.'"' );
 276                              break;
 277                          }
 278                      }

 284                                AND user_ID != "'.$current_User->ID.'"
 285                              LIMIT 1' );

 289                          }

 292              }
