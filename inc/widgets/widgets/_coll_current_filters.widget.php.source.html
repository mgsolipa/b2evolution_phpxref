 113      /**
 114       * Display the widget!
 115       *
 116       * @param array MUST contain at least the basic display params
 117       */
 118  	function display( $params )
 119      {
 120          global $MainList;
 121  
 122          $params = array_merge( array(
 123                  'ItemList'             => $MainList,
 124                  'display_button_reset' => true, // Display a button to reset all filters
 125                  'display_empty_filter' => false, // Display a block with text "No filters"
 126              ), $params );
 127  
 128          if( empty( $params['ItemList'] ) )
 129          { // Empty ItemList object
 130              return false;
 131          }
 132  
 133          if( isset( $params['show_filters'] ) )
 134          { // Get the predefined filters
 135              $show_filters = $params['show_filters'];
 136              unset( $params['show_filters'] );
 137          }
 138  
 139          $this->init_display( $params );
 140  
 141          if( isset( $show_filters ) )
 142          { // Rewrite default filters by predefined
 143              $this->disp_params['show_filters'] = array_merge( $this->disp_params['show_filters'], $show_filters );
 144          }
 145  
 146          $filters =  implode( ' '.T_('AND').' ', $params['ItemList']->get_filter_titles( array(), array(
 147                  'categories_text'     => '',
 148                  'categories_nor_text' => T_('NOT').' ',
 149                  'tags_nor_text'       => T_('NOT').' ',
 150                  'authors_nor_text'    => T_('NOT').' ',
 151                  'group_mask'          => '$filter_items$',
 152                  'filter_mask'         => '<div class="filter_item $filter_class$">'."\n"
 153                          .'<div class="group">$group_title$</div>'."\n"
 154                          .'<div class="name">$filter_name$</div>'."\n"
 155                          .'<div class="icon">$clear_icon$</div>'."\n"
 156                      .'</div>',
 157                  'filter_mask_nogroup' => '<div class="filter_item $filter_class$">'."\n"
 158                          .'<div class="name">$filter_name$</div>'."\n"
 159                          .'<div class="icon">$clear_icon$</div>'."\n"
 160                      .'</div>',
 161                  'before_items'       => '( ',
 162                  'after_items'        => ' )',
 163                  'separator_and'      => ' '.T_('AND').' ',
 164                  'separator_or'       => ' '.T_('OR').' ',
 165                  'separator_nor'      => ' '.T_('NOR').' ',
 166                  'separator_comma'    => ' '.T_('OR').' ',
 167                  'display_category'   => ! empty( $this->disp_params['show_filters']['category'] ),
 168                  'display_archive'    => ! empty( $this->disp_params['show_filters']['archive'] ),
 169                  'display_keyword'    => ! empty( $this->disp_params['show_filters']['keyword'] ),
 170                  'display_tag'        => ! empty( $this->disp_params['show_filters']['tag'] ),
 171                  'display_author'     => ! empty( $this->disp_params['show_filters']['author'] ),
 172                  'display_assignee'   => ! empty( $this->disp_params['show_filters']['assignee'] ),
 173                  'display_locale'     => ! empty( $this->disp_params['show_filters']['locale'] ),
 174                  'display_status'     => ! empty( $this->disp_params['show_filters']['status'] ),
 175                  'display_visibility' => ! empty( $this->disp_params['show_filters']['visibility'] ),
 176                  'display_time'       => ! empty( $this->disp_params['show_filters']['time'] ),
 177                  'display_limit'      => ! empty( $this->disp_params['show_filters']['limit'] ),
 178              ) ) );
 179  
 180          if( empty( $filters ) && ! $this->disp_params['display_empty_filter'] )
 181          { // No filters
 182              return;
 183          }
 184  
 185          // START DISPLAY:
 186          echo $this->disp_params['block_start'];
 187  
 188          // Display title if requested
 189          $this->disp_title();
 190  
 191          echo $this->disp_params['block_body_start'];
 192  
 193          if( empty( $filters ) )
 194          { // No filters
 195              if( $this->disp_params['display_empty_filter'] )
 196              {
 197                  if( is_admin_page() && get_param( 'tab' ) == 'type' )
 198                  { // Try to get a title for current selected post type on back-office pages:
 199                      $current_post_type_title = '"'.get_param( 'tab_type' ).'"';
 200                  }
 201                  if( empty( $current_post_type_title ) )
 202                  { // Use this title by default for unknown selected post type:
 203                      $current_post_type_title = T_('items');
 204                  }
 205                  echo sprintf( T_('No filters - Showing all %s'), $current_post_type_title );
 206              }
 207          }
 208          else
 209          { // Display the filters
 210              echo $filters;
 211  
 212              if( $params['display_button_reset'] )
 213              { // Button to reset all filters
 214                  echo '<p>'.action_icon( T_('Reset all filters'), 'reset_filters',
 215                      regenerate_url( 'catsel,cat,'
 216                          .$params['ItemList']->param_prefix.'tag,'
 217                          .$params['ItemList']->param_prefix.'author,'
 218                          .$params['ItemList']->param_prefix.'author_login,'
 219                          .$params['ItemList']->param_prefix.'assgn,'
 220                          .$params['ItemList']->param_prefix.'assgn_login,'
 221                          .$params['ItemList']->param_prefix.'author_assignee,'
 222                          .$params['ItemList']->param_prefix.'lc,'
 223                          .$params['ItemList']->param_prefix.'status,'
 224                          .$params['ItemList']->param_prefix.'show_statuses,'
 225                          .$params['ItemList']->param_prefix.'types,'
 226                          .$params['ItemList']->param_prefix.'s,'
 227                          .$params['ItemList']->param_prefix.'sentence,'
 228                          .$params['ItemList']->param_prefix.'exact,'
 229                          .$params['ItemList']->param_prefix.'p,'
 230                          .$params['ItemList']->param_prefix.'title,'
 231                          .$params['ItemList']->param_prefix.'pl,'
 232                          .$params['ItemList']->param_prefix.'m,'
 233                          .$params['ItemList']->param_prefix.'w,'
 234                          .$params['ItemList']->param_prefix.'dstart,'
 235                          .$params['ItemList']->param_prefix.'dstop,'
 236                          .$params['ItemList']->param_prefix.'show_past,'
 237                          .$params['ItemList']->param_prefix.'show_future' ),
 238                      ' '.T_('Reset all filters'), 3, 4 ).'<p>';
 239              }
 240          }
 241  
 242          echo $this->disp_params['block_body_end'];
 243  
 244          echo $this->disp_params['block_end'];
 245  
 246          return true;
 247      }
