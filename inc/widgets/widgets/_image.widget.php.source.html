 139      /**
 140       * Display the widget!
 141       *
 142       * @param array MUST contain at least the basic display params
 143       */
 144  	function display( $params )
 145      {
 146          global $Blog;
 147  
 148          switch( $this->disp_params['image_source'] )
 149          {
 150              case 'skin':
 151                  global $skins_url, $skins_path;
 152                  $skin_folder = $Blog->get_skin_folder();
 153                  $image_url = $skins_url.$skin_folder.'/';
 154                  $image_path = $skins_path.$skin_folder.'/';
 155                  break;
 156  
 157              case 'shared':
 158                  global $media_url, $media_path;
 159                  $image_url = $media_url.'shared/';
 160                  $image_path = $media_path.'shared/';
 161                  break;
 162  
 163              case 'coll':
 164              default:
 165                  $image_url = $Blog->get_media_url();
 166                  $image_path = $Blog->get_media_dir();
 167                  break;
 168          }
 169  
 170          if( $this->disp_params['check_file'] && ! file_exists( $image_path.$this->disp_params['image_file'] ) )
 171          { // Logo file doesn't exist, Exit here because of widget setting requires this
 172              return true;
 173          }
 174  
 175          $this->init_display( $params );
 176  
 177          // Collection logo:
 178          echo $this->disp_params['block_start'];
 179  
 180          $image_attrs = '';
 181          if( ! empty( $this->disp_params['width'] ) )
 182          { // Image width
 183              $image_attrs .= ' width="'.intval( $this->disp_params['width'] ).'"';
 184          }
 185          if( ! empty( $this->disp_params['height'] ) )
 186          { // Image height
 187              $image_attrs .= ' height="'.intval( $this->disp_params['height'] ).'"';
 188          }
 189  
 190          echo '<a href="'.$Blog->get( 'url' ).'">'
 191                              .'<img src="'.$image_url.$this->disp_params['image_file'].'" alt=""'.$image_attrs.' />'
 192                              .'</a>';
 193  
 194          echo $this->disp_params['block_end'];
 195  
 196          return true;
 197      }
