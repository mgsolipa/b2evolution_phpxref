 344        * Get param value.
 345        * @param string
 346        * @param boolean default false, set to true only if it is called from a widget::get_param_definition() function to avoid infinite loop
 347        * @return mixed
 348       */
 349  	function get_param( $parname, $check_infinite_loop = false )
 350      {
 351          $this->load_param_array();
 352          if( isset( $this->param_array[$parname] ) )
 353          {    // We have a value for this param:
 354              return $this->param_array[$parname];
 355          }
 356  
 357          // Try default values:
 358          // Note we set 'infinite_loop' param to avoid calling the get_param() from the get_param_definitions() function recursively
 359          $params = $this->get_param_definitions( $check_infinite_loop ? array( 'infinite_loop' => true ) : NULL );
 360          if( isset( $params[$parname]['defaultvalue'] ) )
 361          {    // We ahve a default value:
 362              return $params[$parname]['defaultvalue'] ;
 363          }
 364  
 365          return NULL;
 366      }

 379          $params = $this->get_param_definitions( NULL );

 383              // Make sure param_array is loaded before set the param value

 387              return $this->set_param( 'params', 'string', serialize($this->param_array), $make_null );
 388          }
 389  
 390          switch( $parname )
 391          {
 392              default:
 393                  return $this->set_param( $parname, 'string', $parvalue, $make_null );
 394          }

 549      /**
 550       * Wraps display in a cacheable block.
 551       *
 552       * @param array MUST contain at least the basic display params
 553       * @param array of extra keys to be used for cache keying
 554       */
 555  	function display_with_cache( $params, $keys = array() )
 556      {
 557          global $Blog, $Timer, $debug, $admin_url, $Session;
 558  
 559          $this->init_display( $params );
 560  
 561          // Display the debug conatainers when $debug = 2 OR when it is turned on from evo menu under "Blog" -> "Show/Hide containers"
 562          $display_containers = $Session->get( 'display_containers_'.$Blog->ID ) == 1 || $debug == 2;
 563  
 564          if( ! $Blog->get_setting('cache_enabled_widgets')
 565              || ! $this->disp_params['allow_blockcache']
 566              || $this->get_cache_status() == 'disallowed' )
 567          { // NO CACHING - We do NOT want caching for this collection or for this specific widget:
 568  
 569              if( $display_containers )
 570              { // DEBUG:
 571                  echo '<div class="dev-blocks dev-blocks--widget"><div class="dev-blocks-name" title="'.
 572                              ( $Blog->get_setting('cache_enabled_widgets') ? 'Widget params have BlockCache turned off' : 'Collection params have BlockCache turned off' ).'">'
 573                              .'<span class="dev-blocks-action"><a href="'.$admin_url.'?ctrl=widgets&amp;action=edit&amp;wi_ID='.$this->ID.'">Edit</a></span>'
 574                              .'Widget: <b>'.$this->get_name().'</b> - Cache OFF <i class="fa fa-info">?</i></div>'."\n";
 575              }
 576  
 577              $this->display( $params );
 578  
 579              if( $display_containers )
 580              { // DEBUG:
 581                  echo "</div>\n";
 582              }
 583          }
 584          else
 585          { // Instantiate BlockCache:
 586              $Timer->resume( 'BlockCache' );
 587              // Extend cache keys:
 588              $keys += $this->get_cache_keys();
 589  
 590              $this->BlockCache = new BlockCache( 'widget', $keys );
 591  
 592              $content = $this->BlockCache->check();
 593  
 594              $Timer->pause( 'BlockCache' );
 595  
 596              if( $content !== false )
 597              { // cache hit, let's display:
 598  
 599                  if( $display_containers )
 600                  { // DEBUG:
 601                      echo '<div class="dev-blocks dev-blocks--widget dev-blocks--widget--incache"><div class="dev-blocks-name" title="Cache key = '.$this->BlockCache->serialized_keys.'">'
 602                                  .'<span class="dev-blocks-action"><a href="'.$admin_url.'?ctrl=widgets&amp;action=edit&amp;wi_ID='.$this->ID.'">Edit</a></span>'
 603                                  .'Widget: <b>'.$this->get_name().'</b> - FROM cache <i class="fa fa-info">?</i></div>'."\n";
 604                  }
 605  
 606                  echo $content;
 607  
 608                  if( $display_containers )
 609                  { // DEBUG:
 610                      echo "</div>\n";
 611                  }
 612  
 613              }
 614              else
 615              { // Cache miss, we have to generate:
 616  
 617                  if( $display_containers )
 618                  { // DEBUG:
 619                      echo '<div class="dev-blocks dev-blocks--widget dev-blocks--widget--notincache"><div class="dev-blocks-name" title="Cache key = '.$this->BlockCache->serialized_keys.'">'
 620                                  .'<span class="dev-blocks-action"><a href="'.$admin_url.'?ctrl=widgets&amp;action=edit&amp;wi_ID='.$this->ID.'">Edit</a></span>'
 621                                  .'Widget: <b>'.$this->get_name().'</b> - NOT in cache <i class="fa fa-info">?</i></div>'."\n";
 622                  }
 623  
 624                  $this->BlockCache->start_collect();
 625  
 626                  $this->display( $params );
 627  
 628                  // Save collected cached data if needed:
 629                  $this->BlockCache->end_collect();
 630  
 631                  if( $display_containers )
 632                  { // DEBUG:
 633                      echo "</div>\n";
 634                  }
 635  
 636              }
 637          }
 638      }

 657      /**
 658       * Get cache status
 659       *
 660       * @param boolean TRUE to check if blog allows a caching for widgets
 661       * @return string 'enabled', 'disabled', 'disallowed', 'denied'
 662       */
 663  	function get_cache_status( $check_blog_restriction  = false )
 664      {
 665          $default_widget_params = $this->get_param_definitions( array() );
 666          if( ! empty( $default_widget_params )
 667              && isset( $default_widget_params['allow_blockcache'] )
 668              && isset( $default_widget_params['allow_blockcache']['disabled'] )
 669              && ( $default_widget_params['allow_blockcache']['disabled'] == 'disabled' ) )
 670          { // Widget cache is NOT allowed by widget config
 671              return 'disallowed';
 672          }
 673          else
 674          { // Check current cache status if it is allowed
 675              if( $check_blog_restriction )
 676              { // Check blog restriction for widget caching
 677                  $widget_Blog = & $this->get_Blog();
 678                  if( ! $widget_Blog->get_setting( 'cache_enabled_widgets' ) )
 679                  { // Widget/block cache is not allowed by blog setting
 680                      return 'denied';
 681                  }
 682              }
 683  
 684              if( $this->get_param( 'allow_blockcache' ) )
 685              { // Enabled
 686                  return 'enabled';
 687              }
 688              else
 689              { // Disabled
 690                  return 'disabled';
 691              }
 692          }
 693      }
