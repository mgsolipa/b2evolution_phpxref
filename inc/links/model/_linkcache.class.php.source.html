  86      /**
  87       * Add a dataobject to the cache
  88       *
  89       * @param object Link
  90       * @param string Type of cached object ( '' - empty value to cache depend on link owner type, 'file' - to cache file )
  91       * @return boolean TRUE on success
  92       */
  93  	function add( & $Obj, $cache_type = '' )
  94      {
  95          if( ( !isset($Obj->ID) ) || ( $Obj->ID == 0 ) )
  96          { // The object is not cachable, becuase it has no valid ID
  97              return false;
  98          }
  99  
 100          // If the object wasn't already cached and is valid:
 101          $this->cache[$Obj->ID] = & $Obj;
 102          if( $cache_type == 'file' )
 103          { // Cache by File ID
 104              $this->cache_file[$Obj->file_ID][$Obj->ID] = & $Obj;
 105              return true;
 106          }
 107  
 108          // Cache by Item, Comment or User, we need to get the owner
 109          $LinkOwner = & $Obj->get_LinkOwner();
 110          if( $LinkOwner == NULL )
 111          { // LinkOwner is not valid
 112              return false;
 113          }
 114  
 115          $link_object_ID = $LinkOwner->get_ID();
 116          switch( $LinkOwner->type )
 117          {
 118              case 'item': // cache indexed by Item ID
 119                  $this->cache_item[$link_object_ID][$Obj->ID] = & $Obj;
 120                  break;
 121  
 122              case 'comment': // cache indexed by Comment ID
 123                  $this->cache_comment[$link_object_ID][$Obj->ID] = & $Obj;
 124                  break;
 125  
 126              case 'user': // cache indexed by User ID
 127                  $this->cache_user[$link_object_ID][$Obj->ID] = & $Obj;
 128                  break;
 129  
 130              default:
 131                  return false;
 132          }
 133          return true;
 134      }

 137      /**
 138       * Remove object from the cache
 139       *
 140       * @param object Link
 141       * @param string Type of cached object ( '' - empty value to cache depend on link owner type, 'file' - to cache file )
 142       * @return boolean TRUE on success
 143       */
 144  	function remove( & $Obj, $cache_type = '' )
 145      {
 146          if( ( !isset($Obj->ID) ) || ( $Obj->ID == 0 ) )
 147          { // The object is not cachable, so it can't be in the cache
 148              return false;
 149          }
 150  
 151          $this->remove_by_ID( $Obj->ID );
 152  
 153          if( ( $cache_type == 'file' ) && ( isset( $this->cache_file[$Obj->file_ID][$Obj->ID] ) ) )
 154          { // Cache by File ID
 155              unset( $this->cache_file[$Obj->file_ID][$Obj->ID] );
 156              return true;
 157          }
 158  
 159          // Cache by Item, Comment or User, we need to get the owner
 160          $LinkOwner = & $Obj->get_LinkOwner();
 161          if( $LinkOwner == NULL )
 162          { // LinkOwner is not valid
 163              return false;
 164          }
 165  
 166          $link_object_ID = $LinkOwner->get_ID();
 167          switch( $LinkOwner->type )
 168          {
 169              case 'item':
 170                  if( isset( $this->cache_item[$link_object_ID][$Obj->ID] ) )
 171                  {
 172                      unset( $this->cache_item[$link_object_ID][$Obj->ID] );
 173                      return true;
 174                  }
 175                  break;
 176  
 177              case 'comment':
 178                  if( isset( $this->cache_comment[$link_object_ID][$Obj->ID] ) )
 179                  {
 180                      unset( $this->cache_comment[$link_object_ID][$Obj->ID] );
 181                      return true;
 182                  }
 183                  break;
 184  
 185              case 'user':
 186                  if( isset( $this->cache_user[$link_object_ID][$Obj->ID] ) )
 187                  {
 188                      unset( $this->cache_user[$link_object_ID][$Obj->ID] );
 189                      return true;
 190                  }
 191                  break;
 192  
 193              case 'file':
 194              default: // Invalid cache type
 195                  break;
 196          }
 197  
 198          // Object was not in the cache
 199          return false;
 200      }

 294      /**
 295       * Load links for a given Item
 296       *
 297       * Optimization: If the Item happens to be in the current MainList, Links for the whole MainList will be cached.
 298       *
 299       * @todo cache Link targets before letting the Link constructor handle it
 300       *
 301       * @param integer item ID to load links for
 302       */
 303  	function load_by_item_ID( $item_ID )
 304      {
 305          global $DB, $Debuglog, $MainList, $ItemList;
 306  
 307          if( isset( $this->loaded_cache_item[$item_ID] ) )
 308          {
 309              $Debuglog->add( "Already loaded <strong>$this->objtype(Item #$item_ID)</strong> into cache", 'dataobjects' );
 310              return false;
 311          }
 312          // Check if this Item is part of the MainList
 313          if( $MainList || $ItemList )
 314          {
 315              $prefetch_IDs = array();
 316              if( $MainList )
 317              {
 318                  $prefetch_IDs = array_merge($prefetch_IDs, $MainList->get_page_ID_array());
 319              }
 320              if( $ItemList )
 321              {
 322                  $prefetch_IDs = array_merge($prefetch_IDs, $ItemList->get_page_ID_array());
 323              }
 324              $Debuglog->add( "Loading <strong>$this->objtype(Item #$item_ID)</strong> into cache as part of MainList/ItemList...");
 325              $this->load_by_item_list( $prefetch_IDs );
 326          }
 327          else
 328          {    // NO, load Links for this single Item:
 329  
 330              // Remember this special load:
 331              $this->cache_item[$item_ID] = array();
 332              $this->loaded_cache_item[$item_ID] = true;
 333  
 334              $Debuglog->add( "Loading <strong>$this->objtype(Item #$item_ID)</strong> into cache", 'dataobjects' );
 335  
 336              $SQL = new SQL( 'Get the links by item ID' );
 337              $SQL->SELECT( '*' );
 338              $SQL->FROM( 'T_links' );
 339              $SQL->WHERE( 'link_itm_ID  = '.$DB->quote( $item_ID ) );
 340              $SQL->ORDER_BY( 'link_ltype_ID, link_file_ID' );
 341              $SQL->append( 'FOR UPDATE' ); // fp: we specify FOR UPDATE because we need to lock all changes to the link_order column. 
 342              // fp: Note: FOR UPDATE won't do anything if we're not in a transaction (which is good)
 343  
 344              $this->load_type_by_sql( $SQL, 'item' );
 345          }
 346  
 347          return true;
 348      }

 487      /**
 488       * Load required Files into the FileCache before Link class constructor will be called.
 489       * It's imporatnt to load all Files with one query instead of loading the one by one
 490       * 
 491       * private function
 492       * 
 493       * @param array link rows
 494       */
 495  	function load_linked_files( & $link_rows )
 496      {
 497          if( empty( $link_rows ) )
 498          { // There are nothing to load
 499              return;
 500          }
 501  
 502          // Collect required file Ids
 503          $link_file_ids = array();
 504          foreach( $link_rows as $row )
 505          {
 506              $link_file_ids[] = $row->link_file_ID;
 507          }
 508  
 509          if( !empty( $link_file_ids ) )
 510          { // Load required Files into FileCache
 511              $FileCache = & get_FileCache();
 512              $FileCache->load_where( 'file_ID IN ( '.implode( ',', $link_file_ids ).' )' );
 513          }
 514      }

 517      /**
 518       * Clear the cache **extensively**
 519       *
 520       * @param boolean Keep copy of cache in case we try to re instantiate previous object
 521       * @param string What to clear: 'all', 'user', 'item', 'comment', 'file'
 522       * @param integer ID of the clearing object
 523       */
 524  	function clear( $keep_shadow = false, $object = 'all', $object_ID = 0 )
 525      {
 526          parent::clear( $keep_shadow );
 527  
 528          switch( $object )
 529          {
 530              case 'all':
 531                  // Clear all cached objects
 532                  $this->cache_item = array();
 533                  $this->loaded_cache_item = array();
 534                  $this->cache_comment = array();
 535                  $this->loaded_cache_comment = array();
 536                  $this->cache_user = array();
 537                  $this->loaded_cache_user = array();
 538                  $this->cache_file = array();
 539                  $this->loaded_cache_file = array();
 540                  break;
 541  
 542              case 'item':
 543              case 'comment':
 544              case 'user':
 545              case 'file':
 546                  // Clear only the selected type of objects
 547                  if( empty( $object_ID ) )
 548                  { // Clear all cached objects of this type
 549                      $this->{'cache_'.$object} = array();
 550                      $this->{'loaded_cache_'.$object} = array();
 551                  }
 552                  else
 553                  { // Clear a cache only one object
 554                      unset( $this->{'cache_'.$object}[ $object_ID ] );
 555                      unset( $this->{'loaded_cache_'.$object}[ $object_ID ] );
 556                  }
 557                  break;
 558          }
 559      }
