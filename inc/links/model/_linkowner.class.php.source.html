 257      /**
 258       * Get list of attached files
 259       *
 260       * INNER JOIN on files ensures we only get back file links
 261       *
 262       * @param integer Limit max result
 263       * @param string Restrict to files/images linked to a specific position.
 264       *               Position can be 'teaser'|'aftermore'|'inline'
 265       *               Use comma as separator
 266       * @param string File type: 'image', 'audio', 'other'; NULL - to select all
 267       * @return DataObjectList2 on success or NULL if no linked files found
 268       */
 269  	function get_attachment_FileList( $limit = 1000, $position = NULL, $file_type = NULL )
 270      {
 271          if( ! isset($GLOBALS['files_Module']) )
 272          {
 273              return NULL;
 274          }
 275  
 276          global $DB;
 277  
 278          load_class( '_core/model/dataobjects/_dataobjectlist2.class.php', 'DataObjectList2' );
 279  
 280          $FileCache = & get_FileCache();
 281  
 282          $FileList = new DataObjectList2( $FileCache ); // IN FUNC
 283  
 284          $SQL = new SQL();
 285          $SQL->SELECT( 'file_ID, file_type, file_title, file_root_type, file_root_ID, file_path, file_alt, file_desc, file_path_hash, link_ID' );
 286          $SQL->FROM( 'T_links INNER JOIN T_files ON link_file_ID = file_ID' );
 287          $SQL->WHERE( $this->get_where_condition() );
 288          if( !empty($position) )
 289          {
 290              $position = explode( ',', $position );
 291              $SQL->WHERE_and( 'link_position IN ( '.$DB->quote( $position ).' )' );
 292          }
 293          $SQL->ORDER_BY( 'link_order' );
 294          $SQL->LIMIT( $limit );
 295  
 296          if( ! is_null( $file_type ) )
 297          { // Restrict the Links by File type
 298              $SQL->WHERE_and( 'file_type = '.$DB->quote( $file_type ).' OR file_type IS NULL' );
 299          }
 300  
 301          $FileList->sql = $SQL->get();
 302  
 303          $FileList->query( false, false, false, 'get_attachment_FileList' );
 304  
 305          if( $FileList->result_num_rows == 0 )
 306          {    // Nothing found
 307              $FileList = NULL;
 308          }
 309  
 310          return $FileList;
 311      }

 326      {
 327          if( ! isset($GLOBALS['files_Module']) )

 329              return NULL;

 332          $params = array_merge( array(
 333                  'sql_select_add' => '', // Additional fields for SELECT clause
 334                  'sql_order_by'   => 'link_order', // ORDER BY clause
 335              ), $params );

 337          global $DB;

 339          load_class( '_core/model/dataobjects/_dataobjectlist2.class.php', 'DataObjectList2' );
 340  
 341          $LinkCache = & get_LinkCache();

 343          $LinkList = new DataObjectList2( $LinkCache ); // IN FUNC

 345          $SQL = new SQL();
 346          $SQL->SELECT( 'l.*' );
 347          $SQL->SELECT_add( $params['sql_select_add'] );
 348          $SQL->FROM( 'T_links AS l' );
 349          $SQL->WHERE( $this->get_where_condition() );
 350          if( !empty($position) )
 351          {
 352              $position = explode( ',', $position );
 353              $SQL->WHERE_and( 'link_position IN ( '.$DB->quote( $position ).' )' );
 354          }
 355          $SQL->ORDER_BY( $params['sql_order_by'] );
 356          $SQL->LIMIT( $limit );
 357  
 358          if( ! is_null( $file_type ) )
 359          { // Restrict the Links by File type
 360              $SQL->FROM_add( 'INNER JOIN T_files ON link_file_ID = file_ID' );
 361              $SQL->WHERE_and( 'file_type = '.$DB->quote( $file_type ).' OR file_type IS NULL' );
 362          }
 363  
 364          $LinkList->sql = $SQL->get();

 366          $LinkList->query( false, false, false, 'get_attachment_LinkList' );
 367
