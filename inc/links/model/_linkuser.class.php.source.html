  95       * Add new link to owner User
  96       *
  97       * @param integer file ID
  98       * @param integer link position ( 'teaser', 'aftermore' )
  99       * @param int order of the link
 100       * @return integer|boolean Link ID on success, false otherwise
 101       */
 102  	function add_link( $file_ID, $position = NULL, $order = 1 )
 103      {
 104          global $current_User;
 105  
 106          if( is_null( $position ) )
 107          { // Use default link position
 108              $position = $this->get_default_position( $file_ID );
 109          }
 110  
 111          $edited_Link = new Link();
 112          $edited_Link->set( 'usr_ID', $this->User->ID );
 113          $edited_Link->set( 'file_ID', $file_ID );
 114          $edited_Link->set( 'position', $position );
 115          $edited_Link->set( 'order', $order );
 116  
 117          if( empty( $current_User ) )
 118          { // Current User not is set because probably we are creating links from upgrade script. Set the owner as creator and last editor.
 119              $edited_Link->set( 'creator_user_ID', $this->User->ID );
 120              $edited_Link->set( 'lastedit_user_ID', $this->User->ID );
 121          }
 122          if( $edited_Link->dbinsert() )
 123          {
 124              if( ! is_null( $this->Links ) )
 125              { // If user Links were already loaded update its content 
 126                  $this->Links[$edited_Link->ID] = & $edited_Link;
 127              }
 128              $FileCache = & get_FileCache();
 129              $File = $FileCache->get_by_ID( $file_ID, false, false );
 130              $file_name = empty( $File ) ? '' : $File->get_name();
 131              syslog_insert( sprintf( 'File %s was linked to %s with ID=%s', '<b>'.$file_name.'</b>', $this->type, $this->link_Object->ID ), 'info', 'file', $file_ID );
 132  
 133              return $edited_Link->ID;
 134          }
 135  
 136          return false;
 137      }
