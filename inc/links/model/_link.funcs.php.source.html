  22  /**
  23   * Get a link owner object from link_type and object ID
  24   *
  25   * @param string link type ( item, comment, ... )
  26   * @param integer the corresponding object ID
  27   */
  28  function get_link_owner( $link_type, $object_ID )
  29  {
  30      switch( $link_type )
  31      {
  32          case 'item':
  33              // create LinkItem object
  34              $ItemCache = & get_ItemCache();
  35              $Item = $ItemCache->get_by_ID( $object_ID, false );
  36              $LinkOwner = new LinkItem( $Item );
  37              break;
  38  
  39          case 'comment':
  40              // create LinkComment object
  41              $CommentCache = & get_CommentCache();
  42              $Comment = $CommentCache->get_by_ID( $object_ID, false );
  43              $LinkOwner = new LinkComment( $Comment );
  44              break;
  45  
  46          case 'user':
  47              // create LinkUser object
  48              $UserCache = & get_UserCache();
  49              $User = $UserCache->get_by_ID( $object_ID, false );
  50              $LinkOwner = new LinkUser( $User );
  51              break;
  52  
  53          default:
  54              $LinkOwner = NULL;
  55      }
  56      return $LinkOwner;
  57  }

 450  function get_file_links( $file_ID, $params = array() )
 451  {
 452      global $DB, $current_User, $baseurl, $admin_url;
 453  
 454      $params = array_merge( array(
 455              'separator'       => '<br />',
 456              'post_prefix'     => T_('Post').' - ',
 457              'comment_prefix'  => T_('Comment on').' - ',
 458              'user_prefix'     => T_('Profile picture').' - ',
 459              'current_link_ID' => 0,
 460              'current_before'  => '<b>',
 461              'current_after'   => '</b>',
 462          ), $params );
 463  
 464      // Create result array
 465      $attached_to = array();
 466  
 467      // Get all links with posts and comments
 468      $links_SQL = new SQL();
 469      $links_SQL->SELECT( 'link_ID, link_itm_ID, link_cmt_ID, link_usr_ID' );
 470      $links_SQL->FROM( 'T_links' );
 471      $links_SQL->WHERE( 'link_file_ID = '.$DB->quote( $file_ID ) );
 472      $links = $DB->get_results( $links_SQL->get() );
 473  
 474      if( !empty( $links ) )
 475      { // File is linked with some posts or comments
 476          $ItemCache = & get_ItemCache();
 477          $CommentCache = & get_CommentCache();
 478          $UserCache = & get_UserCache();
 479          $LinkCache = & get_LinkCache();
 480          foreach( $links as $link )
 481          {
 482              $link_object_ID = 0;
 483              $r = '';
 484              if( $params['current_link_ID'] == $link->link_ID )
 485              {
 486                  $r .= $params['current_before'];
 487              }
 488              if( !empty( $link->link_itm_ID ) )
 489              { // File is linked to a post
 490                  if( $Item = & $ItemCache->get_by_ID( $link->link_itm_ID, false ) )
 491                  {
 492                      $Blog = $Item->get_Blog();
 493                      if( $current_User->check_perm( 'item_post!CURSTATUS', 'view', false, $Item ) )
 494                      { // Current user can edit the linked post
 495                          $r .= $params['post_prefix'].'<a href="'.url_add_param( $admin_url, 'ctrl=items&amp;blog='.$Blog->ID.'&amp;p='.$link->link_itm_ID ).'">'.$Item->get( 'title' ).'</a>';
 496                      }
 497                      else
 498                      { // No access to edit the linked post
 499                          $r .= $params['post_prefix'].$Item->get( 'title' );
 500                      }
 501                      $link_object_ID = $link->link_itm_ID;
 502                  }
 503              }
 504              elseif( !empty( $link->link_cmt_ID ) )
 505              { // File is linked to a comment
 506                  if( $Comment = & $CommentCache->get_by_ID( $link->link_cmt_ID, false ) )
 507                  {
 508                      $Item = $Comment->get_Item();
 509                      if( $current_User->check_perm( 'comment!CURSTATUS', 'moderate', false, $Comment ) )
 510                      { // Current user can edit the linked Comment
 511                          $r .= $params['comment_prefix'].'<a href="'.url_add_param( $admin_url, 'ctrl=comments&amp;action=edit&amp;comment_ID='.$link->link_cmt_ID ).'">'.$Item->get( 'title' ).'</a>';
 512                      }
 513                      else
 514                      { // No access to edit the linked Comment
 515                          $r .= $params['comment_prefix'].$Item->get( 'title' );
 516                      }
 517                      $link_object_ID = $link->link_cmt_ID;
 518                  }
 519              }
 520              elseif( !empty( $link->link_usr_ID ) )
 521              { // File is linked to user
 522                  if( $User = & $UserCache->get_by_ID( $link->link_usr_ID, false ) )
 523                  {
 524                      if( $current_User->ID != $User->ID && !$current_User->check_perm( 'users', 'view' ) )
 525                      { // No permission to view other users in admin form
 526                          $r .= $params['user_prefix'].'<a href="'.url_add_param( $baseurl, 'disp=user&amp;user_ID='.$User->ID ).'">'.$User->login.'</a>';
 527                      }
 528                      else
 529                      { // Build a link to display a user in admin form
 530                          $r .= $params['user_prefix'].'<a href="?ctrl=user&amp;user_tab=profile&amp;user_ID='.$User->ID.'">'.$User->login.'</a>';
 531                      }
 532                      $link_object_ID = $link->link_usr_ID;
 533                  }
 534              }

 537              { // Action icon to unlink file from object

 542                          $admin_url.'?ctrl=links&amp;link_ID='.$link->link_ID.'&amp;link_type=item&amp;link_object_ID='.$link->link_usr_ID.'&amp;action=unlink&amp;redirect_to='.rawurlencode( regenerate_url( 'blog', '', '', '&' ) ).'&amp;'.url_crumb( 'link' ),
 543                          NULL, NULL, NULL,
 544                          array( 'onclick' => 'return confirm(\''.TS_('Are you sure want to unlink this file?').'\');' ) );
 545                  }
 546              }
 547  
 548              if( $params['current_link_ID'] == $link->link_ID )
 549              {
 550                  $r .= $params['current_after'];
 551              }
 552              if( !empty( $r ) )
 553              {
 554                  $attached_to[] = $r;
 555              }
 556          }
 557      }
 558  
 559      return implode( $params['separator'], $attached_to );

 563  /**
 564   * Save a vote for the link of file by user
 565   *
 566   * @param string Link ID
 567   * @param integer User ID
 568   * @param string Action of the voting ( 'like', 'noopinion', 'dontlike', 'inappropriate', 'spam' )
 569   * @param integer 1 = checked, 0 = unchecked (for checkboxes: 'Inappropriate' & 'Spam' )
 570   */
 571  function link_vote( $link_ID, $user_ID, $vote_action, $checked = 1 )
 572  {
 573      global $DB;
 574  
 575      // Set modified field name and value
 576      switch( $vote_action )
 577      {
 578          case 'like':
 579              $field_name = 'lvot_like';
 580              $field_value = '1';
 581              break;
 582  
 583          case 'noopinion':
 584              $field_name = 'lvot_like';
 585              $field_value = '0';
 586              break;
 587  
 588          case 'dontlike':
 589              $field_name = 'lvot_like';
 590              $field_value = '-1';
 591              break;
 592  
 593          case 'inappropriate':
 594              $field_name = 'lvot_inappropriate';
 595              $field_value = $checked;
 596              break;
 597  
 598          case 'spam':
 599              $field_name = 'lvot_spam';
 600              $field_value = $checked;
 601              break;
 602  
 603          default:
 604              // invalid vote action
 605              return;
 606      }
 607  
 608      $DB->begin();
 609  
 610      $SQL = new SQL();
 611      $SQL->SELECT( 'lvot_link_ID' );
 612      $SQL->FROM( 'T_links__vote' );
 613      $SQL->WHERE( 'lvot_link_ID = '.$DB->quote( $link_ID ) );
 614      $SQL->WHERE_and( 'lvot_user_ID = '.$DB->quote( $user_ID ) );
 615      $vote = $DB->get_row( $SQL->get() );
 616  
 617      // Save a voting results in DB
 618      if( empty( $vote ) )
 619      { // User replace into to avoid duplicate key conflict in case when user clicks two times fast one after the other
 620          $result = $DB->query( 'REPLACE INTO T_links__vote ( lvot_link_ID, lvot_user_ID, '.$field_name.' )
 621                          VALUES ( '.$DB->quote( $link_ID ).', '.$DB->quote( $user_ID ).', '.$DB->quote( $field_value ).' )' );
 622      }
 623      else
 624      { // Update existing record, because user already has a vote for this file
 625          $result = $DB->query( 'UPDATE T_links__vote
 626                      SET '.$field_name.' = '.$DB->quote( $field_value ).'
 627                      WHERE lvot_link_ID = '.$DB->quote( $link_ID ).'
 628                          AND lvot_user_ID = '.$DB->quote( $user_ID ) );
 629      }
 630  
 631      if( $result )
 632      {
 633          $DB->commit();
 634      }
 635      else
 636      {
 637          $DB->rollback();
 638      }
 639  }
