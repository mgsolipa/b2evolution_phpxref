  95      /**
  96       * Get default position for a new link
  97       *
  98       * @param integer File ID
  99       * @return string Position
 100       */
 101  	function get_default_position( $file_ID )
 102      {
 103          $FileCache = & get_FileCache();
 104          $File = & $FileCache->get_by_ID( $file_ID, false, false );
 105          if( empty( $File ) )
 106          { // If file is broken then get simple default position as "aftermore"
 107              return 'aftermore';
 108          }
 109  
 110          if( $File->is_image() )
 111          { // If file is image then get position depending on order
 112              $this->load_Links();
 113  
 114              if( $this->Links )
 115              { // There's only one file attached yet, the second becomes "aftermore"
 116                  return 'aftermore';
 117              }
 118              else
 119              { // No attachment yet
 120                  return 'teaser';
 121              }
 122          }
 123          else
 124          { // If file is not image then always use "aftermore"
 125              return 'aftermore';
 126          }
 127      }

 141      /**
 142       * Add new link to owner Item
 143       *
 144       * @param integer file ID
 145       * @param integer link position ( 'teaser', 'teaserperm', 'teaserlink', 'aftermore', 'inline', 'fallback' )
 146       * @param int order of the link
 147       * @param boolean true to update owner last touched timestamp after link was created, false otherwise
 148       * @return integer|boolean Link ID on success, false otherwise
 149       */
 150  	function add_link( $file_ID, $position = NULL, $order = 1, $update_owner = true )
 151      {
 152          if( is_null( $position ) )
 153          { // Use default link position
 154              $position = $this->get_default_position( $file_ID );
 155          }
 156  
 157          $edited_Link = new Link();
 158          $edited_Link->set( 'itm_ID', $this->Item->ID );
 159          $edited_Link->set( 'file_ID', $file_ID );
 160          $edited_Link->set( 'position', $position );
 161          $edited_Link->set( 'order', $order );
 162          if( $edited_Link->dbinsert() )
 163          {
 164              // New link was added to the item, invalidate blog's media BlockCache
 165              BlockCache::invalidate_key( 'media_coll_ID', $this->Item->get_blog_ID() );
 166  
 167              $FileCache = & get_FileCache();
 168              $File = $FileCache->get_by_ID( $file_ID, false, false );
 169              $file_name = empty( $File ) ? '' : $File->get_name();
 170              syslog_insert( sprintf( 'File %s was linked to %s with ID=%s', '<b>'.$file_name.'</b>', $this->type, $this->link_Object->ID ), 'info', 'file', $file_ID );
 171  
 172              if( $update_owner )
 173              { // Update last touched date of the Item
 174                  $this->update_last_touched_date();
 175              }
 176  
 177              // Reset the Links
 178              $this->Links = NULL;
 179              $this->load_Links();
 180  
 181              return $edited_Link->ID;
 182          }
 183  
 184          return false;
 185      }

 253       * @param integer Link ID
 254       */
 255  	function after_unlink_action( $link_ID = 0 )
 256      {
 257          if( empty( $this->Item ) )
 258          { // No existing Item, Exit here
 259              return;
 260          }
 261  
 262          if( ! empty( $link_ID ) )
 263          { // Find inline image placeholders if link ID is defined
 264              preg_match_all( '/\[(image|file|inline):'.$link_ID.':?[^\]]*\]/i', $this->Item->content, $inline_images );
 265              if( ! empty( $inline_images[0] ) )
 266              { // There are inline image placeholders in the post content
 267                  $this->Item->set( 'content', str_replace( $inline_images[0], '', $this->Item->content ) );
 268                  $this->Item->dbupdate();
 269                  return;
 270              }
 271          }
 272  
 273          // Update last touched date of the Item
 274          $this->update_last_touched_date();
