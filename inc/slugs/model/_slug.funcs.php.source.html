  20  /*
  21   * Generetae a tiny url (Slug)
  22   * The first character is always a lowercase letter
  23   * The second character is always a capital letter
  24   * The third caharacter is always a number
  25   * at most three letter (uppercase or lowercase) can be after each other -> every fourth character is a number
  26   * at most three number can be after each other -> every fourth character is a letter
  27   * 
  28   * @param string|NULL use this param from getnext_url function, when the last generated url already exists
  29   * @return string the newly generated tinyurl
  30   */
  31  function generate_tinyurl( $last_url = NULL )
  32  {
  33      global $Settings;
  34  
  35      /* Ascii code table for the used characters
  36      $zero = 48;
  37      $nine = 57;
  38      $lower_a = 97;
  39      $lower_z = 122;
  40      $upper_a = 65;
  41      $upper_z = 90;
  42      */
  43  
  44      // get the last used tinyurl
  45      if( ! $last_url && $Settings != null )
  46      { // last_url is not set
  47          $last_url = $Settings->get( 'tinyurl' );
  48      }
  49  
  50      if( ! $last_url )
  51      { // no tinyurl is set, used the default value for the first one
  52          $last_url = 'aA0';
  53          return $last_url;
  54      }
  55  
  56      $index = strlen($last_url);
  57      //find the last character what can be changed
  58      $newurl_ending = '';
  59      while( $index )
  60      {
  61          $char = substr( $last_url, $index - 1, 1 );
  62          $char_value = ord( $char );
  63          if( $index <= 3 )
  64          { // we have to change one of the first three character
  65              if( $char_value >= 65 )
  66              { // the character is a letter
  67                  if( $char_value >= 97 )
  68                  { // the character is a lowercase letter
  69                      if( $char_value < 122 )
  70                      { // there is still unused lowercase letter
  71                          $char_value++;
  72                          $newurl_ending = chr( $char_value ).$newurl_ending;
  73                          break;
  74                          // exit from the loop,the change was done
  75                      }
  76                      // there is no more variation , shift and try the next character
  77                      $newurl_ending = chr( 97 ).$newurl_ending;
  78                      $index--;
  79                      continue;
  80                  }
  81                  elseif( $char_value < 90 )
  82                  {
  83                      $char_value++;
  84                      $newurl_ending = chr( $char_value ).$newurl_ending;
  85                      break;
  86                  }
  87                  $newurl_ending = chr( 65 ).$newurl_ending;
  88                  $index--;
  89                  continue;
  90              }
  91              elseif( $char_value >= 48 )
  92              {
  93                  if( $char_value < 57 )
  94                  {
  95                      $char_value++;
  96                      $newurl_ending = chr( $char_value ).$newurl_ending;
  97                      break;
  98                  }
  99                  $newurl_ending = chr( 48 ).$newurl_ending;
 100                  $index--;
 101                  continue;
 102              }
 103              else
 104              {
 105                  debug_die( 'Wrong tinyurl in database!' );
 106              }
 107          }
 108          else
 109          { // we change a character after the third position
 110              if( $char_value >= 48 && $char_value <= 57 )
 111              {
 112                  if( $char_value != 57 )
 113                  {
 114                      $char_value++;
 115                  }
 116                  else
 117                  {
 118                      if( ( $index - 3) % 4 == 0 )
 119                      { // every fourth character must be a number
 120                          $newurl_ending = chr( 48 ).$newurl_ending;
 121                          $index--;
 122                          continue;
 123                      }
 124                      else
 125                      {
 126                          $char_value = 65;
 127                      }
 128                  }
 129                  $newurl_ending = chr( $char_value ).$newurl_ending;
 130                  break;
 131              }
 132              if( $char_value >= 65 && $char_value <= 90 )
 133              {
 134                  if( $char_value != 90 )
 135                  {
 136                      $char_value++;
 137                  }
 138                  else
 139                  {
 140                      $char_value = 97;
 141                  }
 142                  $newurl_ending = chr( $char_value ).$newurl_ending;
 143                  break;
 144              }
 145              if( $char_value >= 97 && $char_value < 122 )
 146              {
 147                  $char_value++;
 148                  $newurl_ending = chr( $char_value ).$newurl_ending;
 149                  break;
 150              }
 151              if( $char_value == 122 )
 152              { // no more variation we need to shift to another position
 153                  if( ( $index - 3 ) % 4 == 3 )
 154                  { // every fourth character must be a letter
 155                      $newurl_ending = chr( 65 ).$newurl_ending;
 156                  }
 157                  else
 158                  {
 159                      $newurl_ending = chr( 48 ).$newurl_ending;
 160                  }
 161                  $index--;
 162                  continue;
 163              }
 164              else
 165              {
 166                  debug_die( 'Wrong tinyurl in database!' );
 167              }
 168          }
 169      }
 170  
 171      if( $index )
 172      { // index is not null we change the last_url ending part
 173          $new_url = substr_replace( $last_url, $newurl_ending, $index - 1 );
 174          return $new_url;
 175      }
 176  
 177      // we need to add one more character, because we don't have any other unused variation
 178      $index = strlen($last_url);
 179      if( ( $index > 4 ) && ( ( $index - 2 ) % 4 == 3 ) )
 180      {
 181          $new_url = $newurl_ending.chr( 65 );
 182      }
 183      else
 184      {
 185          $new_url = $newurl_ending.chr( 48 );
 186      }
 187      return $new_url;
 188  }

 195  function getnext_tinyurl()
 196  {
 197      global $DB;
 198  
 199      $url = NULL;
 200      do
 201      { // create a new tinyurl, until it is unique
 202          $url = generate_tinyurl( $url );
 203          $query = 'SELECT slug_ID
 204                      FROM T_slug
 205                      WHERE slug_title = '.$DB->quote( $url );
 206      }
 207      while( $DB->get_var( $query ) != NULL );
 208      return $url;
