 156      {
 157          $counter_name .= '_count';
 158          if( isset( $this->$counter_name ) )
 159          {
 160              $this->set( $counter_name, $this->get( $counter_name ) + 1 );
 161              if( $counter_name != 'sent_count' )
 162              {    // Update last error date when we increase an error counter
 163                  global $servertimenow;
 164                  $this->set( 'last_error_ts', date( 'Y-m-d H:i:s', $servertimenow ) );

 178          if( $this->ID == 0 )
 179          {    // New record is creating, we can assign any status without level restrictions
 180              $this->set( 'status', $new_status );

 183          {    // The record is updating, we should update only status which has hight level then old status
 184  
 185              // Get statuses which we can update depending on levels of the statuses
 186              $available_statuses = emadr_get_statuses_less_level( $new_status );
 187  
 188              if( in_array( $this->get( 'status' ), $available_statuses ) )
 189              {    // Check if we can update this status
