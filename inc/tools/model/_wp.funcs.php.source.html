 526          $category_default = $new_Chapter->ID;
 527      }
 528  
 529      /* Import tags */
 530      $tags = array();
 531      if( isset( $xml_data['tags'] ) && count( $xml_data['tags'] ) > 0 )
 532      {
 533          echo T_('Importing the tags... ');
 534          evo_flush();
 535  
 536          // Get existing tags
 537          $SQL = new SQL();
 538          $SQL->SELECT( 'tag_name, tag_ID' );
 539          $SQL->FROM( 'T_items__tag' );
 540          $tags = $DB->get_assoc( $SQL->get() );
 541  
 542          $tags_count = 0;
 543          foreach( $xml_data['tags'] as $tag )
 544          {
 545              if( empty( $tags[ (string) $tag['tag_name'] ] ) )
 546              {    // Insert new tag into DB if tag doesn't exist with current name
 547                  mysqli_query( $DB->dbhandle, 'INSERT INTO '.$tableprefix.'items__tag ( tag_name )
 548                      VALUES ( '.$DB->quote( $tag['tag_name'] ).' )' );
 549                  $tag_ID = mysqli_insert_id( $DB->dbhandle );
 550                  // Save new tag
 551                  $tags[ $tag['tag_name'] ] = (string) $tag_ID;
 552                  $tags_count++;
 553              }
 554          }
 555          echo sprintf( T_('%d records'), $tags_count ).'<br />';
 556      }
 557  
 558  
 559      /* Import posts */
 560      $posts = array();
 561      $comments = array();
 562      if( isset( $xml_data['posts'] ) && count( $xml_data['posts'] ) > 0 )
 563      {
 564          load_class( 'items/model/_item.class.php', 'Item' );
 565  
 566          // Set status's links between WP and b2evo names
 567          $post_statuses = array(
 568              // WP statuses => Their analogs in b2evolution
 569              'publish'    => 'published',
 570              'pending'    => 'review',
 571              'draft'      => 'draft',
 572              'trash'      => 'deprecated',
 573              // These statuses don't exist in WP, but we handle them if they will appear once

 575              'deprecated' => 'deprecated',
 576              'protected'  => 'protected',
 577              'private'    => 'private',
 578              'review'     => 'review',
 579              'redirected' => 'redirected'
 580              // All other unknown statuses will be converted to 'review'
 581          );

 583          // Get post types
 584          $SQL = new SQL();
 585          $SQL->SELECT( 'LOWER( ityp_name ), ityp_ID' );
 586          $SQL->FROM( 'T_items__type' );
 587          $post_types = $DB->get_assoc( $SQL->get() );
 588  
 589          echo T_('Importing the posts... ');
 590          evo_flush();
 591  
 592          foreach( $xml_data['posts'] as $post )
 593          {
 594              $author_ID = isset( $authors[ (string) $post['post_author'] ] ) ? $authors[ (string) $post['post_author'] ] : 1;
 595              $last_edit_user_ID = isset( $authors[ (string) $post['post_lastedit_user'] ] ) ? $authors[ (string) $post['post_lastedit_user'] ] : $author_ID;
 596  
 597              $post_main_cat_ID = $category_default;
 598              $post_extra_cat_IDs = array();
 599              $post_tags = array();
 600              if( !empty( $post['terms'] ) )
 601              { // Set categories and tags
 602                  foreach( $post['terms'] as $term )
 603                  {
 604                      switch( $term['domain'] )
 605                      {
 606                          case 'category':
 607                              if( isset( $categories[ (string) $term['slug'] ] ) )
 608                              {
 609                                  if( $post_main_cat_ID == $category_default )
 610                                  { // Set main category
 611                                      $post_main_cat_ID = $categories[ (string) $term['slug'] ];
 612                                  }
 613                                  // Set extra categories
 614                                  $post_extra_cat_IDs[] = $categories[ (string) $term['slug'] ];
 615                              }
 616                              break;
 617  
 618                          case 'post_tag':
 619                              if( isset( $tags[ (string) $term['slug'] ] ) )
 620                              { // Set tag
 621                                  $post_tags[] = $term['slug'];
 622                              }
 623                              break;
 624                      }
 625                  }
 626              }
 627  
 628              // Set post type ID
 629              $post_type_ID = isset( $post_types[ strtolower( $post['post_type'] ) ] ) ? $post_types[ strtolower( $post['post_type'] ) ] : '1';
 630  
 631              // Get regional IDs by their names
 632              $item_regions = wp_get_regional_data( $post['post_country'], $post['post_region'], $post['post_subregion'], $post['post_city'] );
 633  
 634              $Item = new Item();
 635              $Item->set( 'main_cat_ID', $post_main_cat_ID );

 650              if( empty( $post['post_excerpt'] ) && !empty( $post['post_content'] ) )
 651              {    // Generate excerpt
 652                  $Item->set( 'excerpt', excerpt( $post['post_content'] ) );
 653                  $Item->set( 'excerpt_autogenerated', '1' );
 654              }
 655              $Item->set( 'extra_cat_IDs', $post_extra_cat_IDs );
 656              $Item->set( 'dateset', $post['post_date_mode'] == 'set' ? 1 : 0 );
 657              if( isset( $authors[ (string) $post['post_assigned_user'] ] ) )
 658              {
 659                  $Item->set( 'assigned_user', $authors[ (string) $post['post_assigned_user'] ] );
 660              }
 661              $Item->set( 'datedeadline', $post['post_datedeadline'] );
 662              $Item->set( 'locale', $post['post_locale'] );
 663              $Item->set( 'excerpt_autogenerated', $post['post_excerpt_autogenerated'] );

 665              $Item->set( 'notifications_status', empty( $post['post_notifications_status'] ) ? 'noreq' : $post['post_notifications_status'] );
 666              $Item->set( 'renderers', array( $post['post_renderers'] ) );
 667              $Item->set( 'priority', $post['post_priority'] );
 668              $Item->set( 'featured', $post['post_featured'] );
 669              $Item->set( 'order', $post['post_order'] );
 670              if( !empty( $item_regions['country'] ) )
 671              {    // Country
 672                  $Item->set( 'ctry_ID', $item_regions['country'] );
 673                  if( !empty( $item_regions['region'] ) )
 674                  {    // Region
 675                      $Item->set( 'rgn_ID', $item_regions['region'] );
 676                      if( !empty( $item_regions['subregion'] ) )
 677                      {    // Subregion
 678                          $Item->set( 'subrg_ID', $item_regions['subregion'] );
 679                      }
 680                      if( !empty( $item_regions['city'] ) )
 681                      {    // City
 682                          $Item->set( 'city_ID', $item_regions['city'] );

 684                  }
 685              }
 686  
 687              if( count( $post_tags ) > 0 )
 688              {
 689                  $Item->tags = $post_tags;
 690              }
 691  
 692              $Item->dbinsert();
 693              $posts[ $post['post_id'] ] = $Item->ID;
 694  
 695              if( ! empty( $files ) && ! empty( $post['links'] ) )
 696              { // Link the files to the Item if it has them
 697                  foreach( $post['links'] as $link )
 698                  {
 699                      if( isset( $files[ $link['link_file_ID'] ] ) )

 703                          $File->link_to_Object( $LinkOwner, $link['link_order'], $link['link_position'] );
 704                      }
 705                  }
 706              }
 707  
 708              if( !empty( $post['comments'] ) )
 709              { // Set comments
 710                  $comments[ $Item->ID ] = $post['comments'];
 711              }
 712          }
 713  
 714          foreach( $xml_data['posts'] as $post )
 715          {    // Set post parents
 716              if( !empty( $post['post_parent'] ) && isset( $posts[ (string) $post['post_parent'] ] ) )
 717              {
 718                  mysqli_query($DB->dbhandle, 'UPDATE '.$tableprefix.'items__item
 719                            SET post_parent_ID = '.$DB->quote( $posts[ (string) $post['post_parent'] ] ).'
 720                          WHERE post_ID = '.$DB->quote( $posts[ (string) $post['post_id'] ] ) );

 722          }
 723  
 724          echo sprintf( T_('%d records'), count( $xml_data['posts'] ) ).'<br />';
 725      }
 726  
 727  
 728      /* Import comments */
 729      if( !empty( $comments ) )
 730      {
 731          echo T_('Importing the comments... ');
 732          evo_flush();
 733  
 734          $comments_count = 0;
 735          $comments_IDs = array();
 736          foreach( $comments as $post_ID => $comments )
 737          {
 738              if( empty( $comments ) )
 739              {    // Skip if no comments
 740                  continue;
 741              }
 742  
 743              foreach( $comments as $comment )
 744              {
 745                  $comment_author_user_ID = 0;
 746                  if( !empty( $comment['comment_user_id'] ) && isset( $authors_IDs[ (string) $comment['comment_user_id'] ] ) )
 747                  {    // Author ID
 748                      $comment_author_user_ID = $authors_IDs[ (string) $comment['comment_user_id'] ];
 749                  }
 750  
 751                  $comment_parent_ID = 0;
 752                  if( !empty( $comment['comment_parent'] ) && isset( $comments_IDs[ (string) $comment['comment_parent'] ] ) )
 753                  {    // Parent comment ID
 754                      $comment_parent_ID = $comments_IDs[ (string) $comment['comment_parent'] ];
 755                  }
 756  
 757                  unset( $comment_IP_country );
 758                  if( !empty( $comment['comment_IP_country'] ) )
 759                  {    // Get country ID by code
 760                      $CountryCache = & get_CountryCache();
 761                      if( $Country = & $CountryCache->get_by_name( $comment['comment_IP_country'], false ) )
 762                      {
 763                          $comment_IP_country = $Country->ID;
 764                      }
 765                  }
 766  
 767                  $Comment = new Comment();
 768                  $Comment->set( 'item_ID', $post_ID );
 769                  if( !empty( $comment_parent_ID ) )
 770                  {
 771                      $Comment->set( 'in_reply_to_cmt_ID', $comment_parent_ID );
 772                  }
 773                  $Comment->set( 'date', $comment['comment_date'] );
 774                  if( !empty( $comment_author_user_ID ) )
 775                  {
 776                      $Comment->set( 'author_user_ID', $comment_author_user_ID );
 777                  }
 778                  $Comment->set( 'author', utf8_substr( $comment['comment_author'], 0, 100 ) );
 779                  $Comment->set( 'author_IP', $comment['comment_author_IP'] );
 780                  $Comment->set( 'author_email', $comment['comment_author_email'] );
 781                  $Comment->set( 'content', $comment['comment_content'] );
 782                  if( empty( $comment['comment_status'] ) )
 783                  {    // If comment status is empty (the export of wordpress doesn't provide this field)
 784                      $Comment->set( 'status', $comment['comment_approved'] == '1' ? 'published' : 'draft' );
 785                  }
 786                  else
 787                  {    // Set status when we have predefined value
 788                      $Comment->set( 'status', $comment['comment_status'] );
 789                  }
 790                  if( !empty( $comment_IP_country ) )
 791                  {    // Country
 792                      $Comment->set( 'IP_ctry_ID', $comment_IP_country );
 793                  }
 794                  $Comment->set( 'rating', $comment['comment_rating'] );
 795                  $Comment->set( 'featured', $comment['comment_featured'] );
 796                  $Comment->set( 'nofollow', $comment['comment_nofollow'] );
 797                  $Comment->set( 'helpful_addvotes', $comment['comment_helpful_addvotes'] );
 798                  $Comment->set( 'helpful_countvotes', $comment['comment_helpful_countvotes'] );
 799                  $Comment->set( 'spam_addvotes', $comment['comment_spam_addvotes'] );
 800                  $Comment->set( 'spam_countvotes', $comment['comment_spam_countvotes'] );
 801                  $Comment->set( 'karma', $comment['comment_karma'] );
 802                  $Comment->set( 'spam_karma', $comment['comment_spam_karma'] );
 803                  $Comment->set( 'allow_msgform', $comment['comment_allow_msgform'] );
 804                  $Comment->set( 'notif_status', empty( $comment['comment_notif_status'] ) ? 'noreq' : $comment['comment_notif_status'] );
 805                  $Comment->dbinsert();
 806  
 807                  $comments_IDs[ $comment['comment_id'] ] = $Comment->ID;
 808                  $comments_count++;
 809              }
 810          }
 811  
 812          echo sprintf( T_('%d records'), $comments_count ).'<br />';
 813      }
 814  
 815      echo '<p>'.T_('Import complete.').'</p>';
 816  
 817      $DB->commit();
 818  }

 858          $xml_import_convert_to_latin = true;
 859      }
 860      else
 861      { // Don't convert, it is correct encoding
 862          $xml_import_convert_to_latin = false;
 863      }
 864  
 865      $namespaces = $xml->getDocNamespaces();
 866      if( !isset( $namespaces['wp'] ) )
 867      {
 868          $namespaces['wp'] = 'http://wordpress.org/export/1.1/';

 923              'file_path'      => (string) $t->file_path,

 995          $post['status']         = (string) $wp->status;
 996          $post['post_parent']    = (int) $wp->post_parent;
 997          $post['menu_order']     = (int) $wp->menu_order;
 998          $post['post_type']      = (string) $wp->post_type;
 999          $post['post_password']  = (string) $wp->post_password;
1000          $post['is_sticky']      = (int) $wp->is_sticky;
1001          $post['post_date_mode']     = (string) $evo->post_date_mode;
1002          $post['post_lastedit_user'] = (string) $evo->post_lastedit_user;
1003          $post['post_assigned_user'] = (string) $evo->post_assigned_user;
1004          $post['post_datedeadline']  = (string) $evo->post_datedeadline;
