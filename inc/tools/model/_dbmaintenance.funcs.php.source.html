 146  /**
 147   * Optimize DB tables (MyISAM & InnoDB)
 148   *
 149   * @param boolean Display messages
 150   * @param boolean TRUE - to make optimize query for each table separately
 151   * @return array Results of the mysql command 'OPTIMIZE'
 152   */
 153  function dbm_optimize_tables( $display_messages = true, $separate_tables = true )
 154  {
 155      global $tableprefix, $DB, $Timer, $dbm_tables_count;
 156  
 157      $tables = $DB->get_results( 'SHOW TABLE STATUS FROM `'.$DB->dbname.'` LIKE \''.$tableprefix.'%\'');
 158  
 159      $myisam_tables = array();
 160      $innodb_tables = array();
 161      foreach( $tables as $table )
 162      {
 163          // Before MySQL 4.1.2, the "Engine" field was labeled as "Type".
 164          if( ( ( isset( $table->Engine ) && $table->Engine == 'MyISAM' )
 165                  || ( isset( $table->Type ) && $table->Type == 'MyISAM' ) )
 166              && $table->Data_free )
 167          {    // Optimization needed for MyISAM table
 168              $myisam_tables[] = $table->Name;
 169          }
 170          else if( ( ( isset( $table->Engine ) && $table->Engine == 'InnoDB' )
 171                  || ( isset( $table->Type ) && $table->Type == 'InnoDB' ) )
 172              && $table->Data_free )
 173          {    // Optimization needed for InnoDB table
 174              $innodb_tables[] = $table->Name;
 175          }
 176      }
 177      $dbm_tables_count = count( $myisam_tables ) + count( $innodb_tables );
 178  
 179      // Optimize MyISAM tables
 180      $myisam_results = dbm_optimize_tables_process( $display_messages, $separate_tables, $myisam_tables, 'MyISAM' );
 181      $results = $myisam_results;
 182  
 183      // Optimize InnoDB tables
 184      $innodb_results = dbm_optimize_tables_process( $display_messages, $separate_tables, $innodb_tables, 'InnoDB' );
 185      $results = array_merge( $results, $innodb_results );
 186  
 187      return $results;
 188  }

 191  /**
 192   * Optimize process DB tables (MyISAM & InnoDB)
 193   *
 194   * @param boolean Display messages
 195   * @param boolean TRUE - to make optimize query for each table separately
 196   * @param array Tables
 197   * @param string Table type: 'MyISAM' or 'InnoDB'
 198   * @return array Results of the mysql command 'OPTIMIZE'
 199   */
 200  function dbm_optimize_tables_process( $display_messages = true, $separate_tables = true, $tables = array(), $table_type = '' )
 201  {
 202      global $DB;
 203  
 204      load_class( '_core/model/_timer.class.php', 'Timer' );
 205      $Timer = new Timer('dbm_optimize');
 206  
 207      $results = array();
 208  
 209      if( $display_messages )
 210      { // Display messages
 211          echo '<b>'.sprintf( T_('Optimize %s tables...'), $table_type ).'</b><br />';
 212          evo_flush();
 213      }
 214      $timer_name = 'optimize_'.strtolower( $table_type );
 215  
 216      $Timer->start( $timer_name );
 217  
 218      if( !empty( $tables ) )
 219      { // Optimize tables
 220          if( $separate_tables )
 221          { // Optimize each table in separate query
 222              foreach( $tables as $table )
 223              {
 224                  $Timer->start( $timer_name.'_table' );
 225                  $table_results = $DB->get_results( 'OPTIMIZE NO_WRITE_TO_BINLOG TABLE '.$table );
 226                  $Timer->stop( $timer_name.'_table' );
 227                  if( $display_messages )
 228                  { // Display messages
 229                      dbm_display_result_messages( $table_results, 'optimize' );
 230                      echo '<b>'.sprintf( T_('Time: %s seconds'), $Timer->get_duration( $timer_name.'_table' ) ).'</b><br /><br />';
 231                  }
 232                  evo_flush();
 233                  $results = array_merge( $results, $table_results );
 234              }
 235          }
 236          else
 237          { // Optimize all table by one query, Used for cron job
 238              $results = $DB->get_results( 'OPTIMIZE NO_WRITE_TO_BINLOG TABLE '.implode( ', ', $tables ) );
 239          }
 240      }
 241  
 242      $Timer->stop( $timer_name );
 243  
 244      if( $display_messages )
 245      { // Display messages
 246          if( !$separate_tables || empty( $tables ) )
 247          { // Display full report log for case when the tables were optimized by one query
 248              dbm_display_result_messages( $results, 'optimize' );
 249          }
 250          echo '<b>'.sprintf( T_('Full execution time: %s seconds'), $Timer->get_duration( $timer_name ) ).'</b><br /><br />';
 251          evo_flush();
 252      }
 253  
 254      return $results;
 255  }

 258  /**
 259   * Print on screen the messages of 'OPTIMIZE', 'CHECK' OR 'ANALYZE' commands
 260   *
 261   * @param array Results of the mysql commands 'OPTIMIZE', 'CHECK' OR 'ANALYZE'
 262   * @param string mysql command type: optimize, check, analyze
 263   */
 264  function dbm_display_result_messages( $results, $command_type )
 265  {
 266      switch( $command_type )
 267      {
 268          case 'optimize':
 269              $params = array(
 270                      'message_ok'    => T_('Database table %s is optimized.'),
 271                      'message_error' => T_('Database table %s could not be optimized. Message: %s'),
 272                      'message_done'  => T_('All database tables are already optimized.'),
 273                  );
 274              break;
 275  
 276          case 'check':
 277              $params = array(
 278                      'message_ok'    => T_('Database table %s is checked.'),
 279                      'message_error' => T_('Database table %s could not be checked. Message: %s'),
 280                      'message_done'  => T_('All database tables are already checked.'),
 281                  );
 282              break;
 283  
 284          case 'analyze':
 285              $params = array(
 286                      'message_ok'    => T_('Database table %s is analyzed.'),
 287                      'message_error' => T_('Database table %s could not be analyzed. Message: %s'),
 288                      'message_done'  => T_('All database tables are already analyzed.'),
 289                  );
 290              break;
 291  
 292          default:
 293              // Don't support unknown commants, Exit here
 294              return;
 295      }
 296  
 297      if( count( $results ) > 0 )
 298      {
 299          foreach( $results as $result )
 300          {
 301              if( $result->Msg_type == 'status' && $result->Msg_text == 'OK' )
 302              { // OK
 303                  echo sprintf( $params['message_ok'], '<b>'.$result->Table.'</b>' ).'<br />';
 304              }
 305              elseif( $result->Msg_type == 'note' && $result->Msg_text == 'Table does not support optimize, doing recreate + analyze instead' )
 306              { // This warning is comming for every innodb table, but that is normal, Display for info
 307                  echo sprintf( T_('Database table %s does not support optimize, doing recreate + analyze instead'), '<b>'.$result->Table.'</b>' ).'<br />';
 308              }
 309              else
 310              { // Some errors
 311                  $message_class = $result->Msg_type == 'status' ? 'orange' : 'red';
 312                  echo '<span class="'.$message_class.'">'.sprintf( $params['message_error'], '<b>'.$result->Table.'</b>', '"'.$result->Msg_text.'"' ).'</span><br />';
 313              }
 314          }
 315      }
 316      else
 317      { // No tables found to optimize, probably all tables already were optimized
 318          echo $params['message_done'].'<br />';
 319      }
 320  }

 326  function dbm_delete_broken_posts()
 327  {
 328      global $DB, $Messages, $current_User;
 329  
 330      $num_deleted = 0;
 331  
 332      echo T_('Removing broken posts that have no matching category... ');
 333      evo_flush();
 334  
 335      // Delete the posts only by these IDs
 336      $post_IDs = trim( param( 'posts', '/^[\d,]+$/', true ), ',' );
 337      $post_IDs = preg_replace( '/(,){2,}/', ',', $post_IDs );
 338  
 339      if( ! empty( $post_IDs ) )
 340      {
 341          // select broken items
 342          $SQL = new SQL();
 343          $SQL->SELECT( '*' );
 344          $SQL->FROM( 'T_items__item' );
 345          $SQL->WHERE( 'post_main_cat_ID NOT IN ( SELECT cat_ID FROM T_categories )' );
 346          $SQL->WHERE_and( 'post_ID IN ( '.$post_IDs.' )' );
 347          $broken_items = $DB->get_results( $SQL->get(), OBJECT, 'Find broken posts' );
 348  
 349          foreach( $broken_items as $r => $row )
 350          { // delete broken items
 351              $broken_Item = new Item( $row );
 352              if( $broken_Item->dbdelete() )
 353              { // Post was deleted successfully
 354                  $num_deleted++;
 355              }
 356              else
 357              { // Post was NOT deleted
 358                  echo '<p class="red">'.sprintf( T_('Cannot delete post with ID %s'), $broken_Item->ID ).'</p>';
 359              }
 360              if( $r % 100 == 0 )
 361              { // Display a log dot after each 100 processed posts
 362                  echo '. ';
 363                  evo_flush();
 364              }
 365          }
 366      }
 367  
 368      echo '<p>'.sprintf( T_('Deleted %d posts.'), $num_deleted ).'</p>';
 369  }

 447  /**
 448   * Find and delete orphan File objects with no matching file on disk
 449   */
 450  function dbm_delete_orphan_files()
 451  {
 452      global $DB, $admin_url;
 453  
 454      $FileCache = & get_FileCache();
 455      $FileCache->clear();
 456  
 457      echo T_('Deleting of the orphan File objects from the database...');
 458      evo_flush();
 459  
 460      $files_SQL = new SQL();
 461      $files_SQL->SELECT( '*' );
 462      $files_SQL->FROM( 'T_files' );
 463      $files_SQL->ORDER_BY( 'file_ID' );
 464  
 465      $count_files_valid = 0;
 466      $count_files_invalid = 0;
 467      $count_files_deleted = 0;
 468  
 469      $page_size = 100;
 470      $current_page = 0;
 471      // Search the files by page to save memory
 472      $files_SQL->LIMIT( '0, '.$page_size );
 473      while( $loaded_Files = $FileCache->load_by_sql( $files_SQL ) )
 474      { // Check all loaded files
 475          foreach( $loaded_Files as $File )
 476          {
 477              if( is_null( $File ) )
 478              { // The File object couldn't be created because the db entry is invalid
 479                  $count_files_invalid++;
 480                  continue;
 481              }
 482              if( $File->exists() )
 483              { // File exists on the disk
 484                  $count_files_valid++;
 485              }
 486              else
 487              { // File doesn't exist on the disk, Remove it from DB
 488                  $File->dbdelete();
 489                  $count_files_deleted++;
 490              }
 491          }
 492  
 493          echo ' .';
 494          evo_flush();
 495  
 496          // Clear cache after each page to save memory
 497          $FileCache->clear();
 498  
 499          $current_page++;
 500          $files_SQL->LIMIT( ( $current_page * $page_size ).', '.$page_size );
 501      }
 502  
 503      echo 'OK<p>';
 504      echo sprintf( T_('Number of deleted orphan File objects: %d.'), $count_files_deleted ).'<br />';
 505      echo sprintf( T_('Number of valid File objects in the database: %d.'), $count_files_valid ).'</p>';
 506  
 507      if( $count_files_invalid )
 508      { // There are invalid files in the database
 509          // Display warning to show that the 'Remove orphan file roots' tool should be also called
 510          $remove_orphan_file_roots = 'href="'.$admin_url.'ctrl=tools&amp;action=delete_orphan_file_roots&amp;'.url_crumb('tools').'"';
 511          $invalid_files_note = ( $count_files_invalid == 1 ) ? T_('An invalid File object was found in the database.') : sprintf( T_('%d invalid File objects were found in the database.'), $count_files_invalid );
 512          echo '<p class="warning">'.$invalid_files_note."<br/>"
 513              .sprintf( T_('It is strongly recommended to also execute the &lt;<a %s>Remove orphan file roots</a>&gt; tool to remove invalid files from the database and from the disk as well!'), $remove_orphan_file_roots )
 514              .'</p>';
 515      }
 516  }

 520   * Remove orphan file roots ( with no matching Blog or User entry in the database ) recursively with all of the content
 521   */
 522  function dbm_delete_orphan_file_roots()
 523  {
 524      global $DB, $media_path;
 525  
 526      echo T_('Removing orphan file roots recursively with all of their content... ');
 527      evo_flush();
 528  
 529      // Store all directories that must be deleted
 530      $delete_dirs = array();
 531  
 532      /* BLOGS */
 533  
 534      // Get the media diretories of all existing blogs
 535      $BlogCache = & get_BlogCache();
 536      $BlogCache->load_all();
 537      $blog_dirs = array();
 538      foreach( $BlogCache->cache as $Blog )
 539      {
 540          $blog_dirs[] = $Blog->get_media_dir();
 541      }
 542      $BlogCache->clear();
 543  
 544      $media_path_blogs = $media_path.'blogs/';
 545  
 546      if( ( $media_dir_blogs = @opendir( $media_path_blogs ) ) === false )
 547      { // Could not open blogs media dir
 548          echo '<p class="red">'.sprintf( T_('Cannot open blogs media directory %s'), '<b>'.$media_path_blogs.'</b>' ).'</p>';
 549      }
 550      else
 551      {
 552          // Find the blog dirs that must be deleted
 553          while( ( $folder = readdir( $media_dir_blogs ) ) !== false )
 554          {
 555              if( $folder == '.' || $folder == '..' || ! is_dir( $media_path_blogs.$folder ) )
 556              { // Skip files
 557                  continue;
 558              }
 559              if( ! in_array( $media_path_blogs.$folder.'/', $blog_dirs ) )
 560              { // This dir must be deleted because it is not media dir of the existing blogs
 561                  $delete_dirs[] = $media_path_blogs.$folder.'/';
 562              }
 563          }
 564  
 565          closedir( $media_dir_blogs );
 566      }
 567  
 568      /* USERS */
 569      echo '. ';
 570      evo_flush();
 571  
 572      // Get logins of all existing users
 573      $SQL = new SQL();
 574      $SQL->SELECT( 'user_login' );
 575      $SQL->FROM( 'T_users' );
 576      $user_logins = $DB->get_col( $SQL->get() );
 577  
 578      $media_path_users = $media_path.'users/';
 579  
 580      if( ( $media_dir_users = @opendir( $media_path_users ) ) === false )
 581      { // Could not open users media dir
 582          echo '<p class="red">'.sprintf( T_('Cannot open users media directory %s'), '<b>'.$media_path_users.'</b>' ).'</p>';
 583      }
 584      else
 585      {
 586          // Find the user dirs that must be deleted
 587          while( ( $folder = readdir( $media_dir_users ) ) !== false )
 588          {
 589              if( $folder == '.' || $folder == '..' || ! is_dir( $media_path_users.$folder ) )
 590              { // Skip files
 591                  continue;
 592              }
 593              if( ! in_array( $folder, $user_logins ) )
 594              { // This dir must be deleted because it is not media dir of the existing users
 595                  $delete_dirs[] = $media_path_users.$folder.'/';
 596              }
 597          }
 598  
 599          closedir( $media_dir_users );
 600      }
 601  
 602      /* DELETE broken  file roots */
 603      echo '. ';
 604      evo_flush();
 605  
 606      foreach( $delete_dirs as $delete_dir )
 607      {
 608          if( rmdir_r( $delete_dir ) )
 609          { // Success deleting
 610              echo '<p class="green">'.sprintf( T_('Invalid file root %s was found and removed with all of its content.'), '<b>'.$delete_dir.'</b>' ).'</p>';
 611          }
 612          else
 613          { // Failed deleting
 614              echo '<p class="red">'.sprintf( T_('Cannot delete directory %s. Please check the permissions or delete it manually.'), '<b>'.$delete_dir.'</b>' ).'</p>';
 615          }
 616      }
 617  
 618      /* DELETE orphan DB file records of the blogs and the users */
 619      echo '. ';
 620      evo_flush();
 621  
 622      $count_files_deleted = $DB->query( 'DELETE f, l, lv FROM T_files AS f
 623               LEFT JOIN T_links AS l ON l.link_file_ID = f.file_ID
 624               LEFT JOIN T_links__vote AS lv ON l.link_ID = lv.lvot_link_ID
 625          WHERE ( file_root_type = "collection"
 626                  AND file_root_ID NOT IN ( SELECT blog_ID FROM T_blogs ) )
 627             OR ( file_root_type = "user"
 628                  AND file_root_ID NOT IN ( SELECT user_ID FROM T_users ) )' );
 629  
 630      echo 'OK.<p>';
 631      echo sprintf( T_('%d File roots have been removed from the disk.'), count( $delete_dirs ) ).'<br />';
 632      echo sprintf( T_('%d File objects have been deleted from DB.'), intval( $count_files_deleted ) ).'</p>';
 633  }

 637   * Recreate all item slugs (change title-[0-9] canonical slugs to a slug generated from current title). Old slugs will still work, but redirect to the new one.
 638   */
 639  function dbm_recreate_itemslugs()
 640  {
 641      global $Messages;
 642  
 643      $ItemCache = get_ItemCache();
 644      $ItemCache->load_where( '( post_title != "" ) AND ( post_urltitle = "title" OR post_urltitle LIKE "title-%" )');
 645      $items = $ItemCache->get_ID_array();
 646      $count_slugs = 0;
 647  
 648      foreach( $items as $item_ID )
 649      {
 650          $Item = $ItemCache->get_by_ID($item_ID);
 651  
 652          $prev_urltitle = $Item->get( 'urltitle' );
 653          $item_title = $Item->get( 'title' );
 654  
 655          // check if post title is not empty and urltitle was auto generated ( equals title or title-[0-9]+ )
 656          // Note: urltitle will be auto generated on this form (title-[0-9]+), if post title wass empty and, urltitle was not set
 657          // Note: Even if a post title was set to 'title' on purpose it's possible, that this tool will change the post urltitle
 658          if( ( ! empty( $item_title ) ) && ( ( $prev_urltitle == 'title' ) || ( preg_match( '#^title-[0-9]+$#', $prev_urltitle ) ) ) )
 659          {
 660              // set urltitle empty, so the item update function will regenerate the item slug
 661              $Item->set( 'urltitle', '' );
 662              $result = $Item->dbupdate(/* do not autotrack modification */ false, /* update slug */ true, /* do not update excerpt */ false);
 663              if( ( $result ) && ( $prev_urltitle != $Item->get( 'urltitle' ) ) )
 664              { // update was successful, and item urltitle was changed
 665                  $count_slugs++;
 666              }
 667          }
 668      }
 669      $Messages->add( sprintf( 'Created %d new URL slugs.', $count_slugs ), 'success' );
 670  }

 693  /**
 694   * Check DB tables
 695   *
 696   * @param boolean Display messages
 697   * @param boolean TRUE - to make optimize query for each table separately
 698   * @return array Results of the mysql command 'CHECK'
 699   */
 700  function dbm_check_tables( $display_messages = true, $separate_tables = true )
 701  {
 702      global $DB, $tableprefix, $dbm_tables_count;
 703  
 704      load_class( '_core/model/_timer.class.php', 'Timer' );
 705      $Timer = new Timer('dbm_check');
 706  
 707      // Get all table names from DB
 708      $tables = $DB->get_results( 'SHOW TABLE STATUS FROM `'.$DB->dbname.'` LIKE \''.$tableprefix.'%\'');
 709      $tables_names = array();
 710      foreach( $tables as $table )
 711      {
 712          $tables_names[] = $table->Name;
 713      }
 714      $dbm_tables_count = count( $tables_names );
 715  
 716      if( $display_messages )
 717      { // Display messages
 718          echo '<b>'.T_('Check tables...').'</b><br />';
 719          evo_flush();
 720      }
 721  
 722      $Timer->start( 'check_tables' );
 723      if( $separate_tables )
 724      { // Check each table in separate query
 725          $check_results = array();
 726          foreach( $tables_names as $table )
 727          {
 728              $Timer->start( 'check_one_table' );
 729              $table_results = $DB->get_results( 'CHECK TABLE '.$table.' FAST' );
 730              $Timer->stop( 'check_one_table' );
 731              if( $display_messages )
 732              { // Display messages
 733                  dbm_display_result_messages( $table_results, 'check' );
 734                  echo '<b>'.sprintf( T_('Time: %s seconds'), $Timer->get_duration( 'check_one_table' ) ).'</b><br /><br />';
 735              }
 736              evo_flush();
 737              $check_results = array_merge( $check_results, $table_results );
 738          }
 739      }
 740      else
 741      { // Check all table by one query, Used for cron job
 742          $check_results = $DB->get_results( 'CHECK TABLE '.implode( ', ', $tables_names ).' FAST' );
 743      }
 744      $Timer->stop( 'check_tables' );
 745  
 746      if( $display_messages )
 747      { // Display messages
 748          if( !$separate_tables )
 749          { // Display full report log for case when the tables were checked by one query
 750              dbm_display_result_messages( $check_results, 'check' );
 751          }
 752          echo '<b>'.sprintf( T_('Full execution time: %s seconds'), $Timer->get_duration( 'check_tables' ) ).'</b><br />';
 753      }
 754  
 755      return $check_results;
 756  }
