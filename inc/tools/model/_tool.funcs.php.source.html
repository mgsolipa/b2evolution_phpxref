 288  function tool_create_sample_messages( $num_loops, $num_messages, $num_words, $max_users )
 289  {
 290      global $Messages, $DB;
 291  
 292      echo T_('Creating of the sample messages...');
 293      evo_flush();
 294  
 295      /**
 296       * Disable log queries because it increases the memory and stops the process with error "Allowed memory size of X bytes exhausted..."
 297       */
 298      $DB->log_queries = false;
 299  
 300      // Get all users
 301      $SQL = new SQL();
 302      $SQL->SELECT( 'user_ID' );
 303      $SQL->FROM( 'T_users' );
 304      $users = $DB->get_col( $SQL->get() );
 305  
 306      if( count( $users ) < 2 )
 307      {    // No users
 308          $Messages->add( T_('At least two users must exist in DB to create the messages'), 'error' );
 309          $action = 'show_create_messages';
 310          break;
 311      }
 312  
 313      $count_threads = 0;
 314      $count_messages = 0;
 315      for( $l = 0; $l < $num_loops; $l++ )
 316      {
 317          $user_links = array();
 318          foreach( $users as $from_user_ID )
 319          {
 320              foreach( $users as $to_user_ID )
 321              {
 322                  if( $from_user_ID == $to_user_ID || isset( $user_links[ (string) $from_user_ID.'-'.$to_user_ID ] ) )
 323                  {
 324                      continue;
 325                  }
 326  
 327                  $user_links[ $from_user_ID.'-'.$to_user_ID ] = 1;
 328                  // Insert thread
 329                  $DB->query( 'INSERT INTO T_messaging__thread ( thrd_title, thrd_datemodified )
 330                      VALUES ( '.$DB->quote( generate_random_key( 16 ) ).', '.$DB->quote( date( 'Y-m-d H:i:s' ) ).' )' );
 331  
 332                  $thread_ID = $DB->insert_id;
 333                  $count_threads++;
 334  
 335                  for( $m = 0; $m < $num_messages; $m++ )
 336                  {
 337                      $msg_text = '';
 338                      for( $w = 0; $w < $num_words; $w++ )
 339                      {
 340                          $msg_text .= generate_random_key( 8 ).' ';
 341                      }
 342                      $message_user_ID = $m % 2 == 0 ? $from_user_ID : $to_user_ID;
 343                      // Insert message
 344                      $DB->query( 'INSERT INTO T_messaging__message ( msg_author_user_ID , msg_datetime, msg_thread_ID, msg_text, msg_renderers )
 345                          VALUES ( '.$DB->quote( $message_user_ID ).', '.$DB->quote( date( 'Y-m-d H:i:s' ) ).', '.$DB->quote( $thread_ID ).', '.$DB->quote( $msg_text ).', \'default\' )' );
 346                      $count_messages++;
 347  
 348                      if( $count_messages % 100 == 0 )
 349                      { // Display a process of creating by one dot for 100 users
 350                          echo ' .';
 351                          evo_flush();
 352                      }
 353                  }
 354  
 355                  // Insert link for thread & user
 356                  $DB->query( 'INSERT INTO T_messaging__threadstatus ( tsta_thread_ID , tsta_user_ID, tsta_first_unread_msg_ID )
 357                      VALUES ( '.$DB->quote( $thread_ID ).', '.$DB->quote( $from_user_ID ).', NULL ),
 358                                   ( '.$DB->quote( $thread_ID ).', '.$DB->quote( $to_user_ID ).', NULL )' );
 359              }
 360          }
 361  
 362          /** Create one conversation between all users ( limit by $max_users ) **/
 363  
 364          // Insert thread
 365          $DB->query( 'INSERT INTO T_messaging__thread ( thrd_title, thrd_datemodified )
 366              VALUES ( '.$DB->quote( generate_random_key( 16 ) ).', '.$DB->quote( date( 'Y-m-d H:i:s' ) ).' )' );
 367  
 368          $thread_ID = $DB->insert_id;
 369          $count_threads++;
 370  
 371          $user_number = 0;
 372          for( $m = 0; $m < $num_messages; $m++ )
 373          {
 374              $msg_text = '';
 375              for( $w = 0; $w < $num_words; $w++ )
 376              {
 377                  $msg_text .= generate_random_key( 8 ).' ';
 378              }
 379              // Insert message
 380              $DB->query( 'INSERT INTO T_messaging__message ( msg_author_user_ID , msg_datetime, msg_thread_ID, msg_text, msg_renderers )
 381                  VALUES ( '.$DB->quote( $users[ $user_number ] ).', '.$DB->quote( date( 'Y-m-d H:i:s' ) ).', '.$DB->quote( $thread_ID ).', '.$DB->quote( $msg_text ).', \'default\' )' );
 382              $count_messages++;
 383              $user_number++;
 384              if( $user_number == count( $users ) || $user_number == $max_users - 1 )
 385              {    // Reset user number to start of the list
 386                  $user_number = 0;
 387              }
 388          }
 389  
 390          // Insert the links between thread & users
 391          $threadstatuses = array();
 392          foreach( $users as $u => $user_ID )
 393          {
 394              $threadstatuses[] = '( '.$DB->quote( $thread_ID ).', '.$DB->quote( $user_ID ).', NULL )';
 395              if( $u == $max_users - 1 )
 396              {    // limit by max users in one thread
 397                  break;
 398              }
 399          }
 400          $DB->query( 'INSERT INTO T_messaging__threadstatus ( tsta_thread_ID , tsta_user_ID, tsta_first_unread_msg_ID )
 401              VALUES '.implode( ', ', $threadstatuses ) );
 402      }
 403  
 404      echo ' OK.';
 405  
 406      $Messages->add( sprintf( T_('%d threads and %d messages have been created.'), $count_threads, $count_messages ), 'success' );
 407  }
