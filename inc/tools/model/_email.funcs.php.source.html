 185  /**
 186   * Get result info of email log
 187   *
 188   * @param string Result ( 'ok', 'error', 'blocked' )
 189   * @param boolean Params
 190   * @return string Result info
 191   */
 192  function emlog_result_info( $result, $params = array() )
 193  {
 194      $params = array_merge( array(
 195              'display_icon'  => true,    // Display icon
 196              'display_text'  => true,    // Display text
 197              'link_blocked'  => false, // TRUE - to display 'Blocked' as link to go to page with Blocked email adresses with filter by email
 198              'email'         => '',    // Email address to filter
 199          ), $params );
 200  
 201      $result_info = '';
 202  
 203      switch( $result )
 204      {
 205          case 'ok':
 206              if( $params['display_icon'] )
 207              {
 208                  $result_info .= get_icon( 'bullet_green', 'imgtag', array( 'alt' => T_('Ok') ) );
 209              }
 210              if( $params['display_text'] )
 211              {
 212                  $result_info .= ' '.T_('Ok');
 213              }
 214              break;
 215  
 216          case 'error':
 217              if( $params['display_icon'] )
 218              {
 219                  $result_info .= get_icon( 'bullet_red', 'imgtag', array( 'alt' => T_('Error') ) );
 220              }
 221              if( $params['display_text'] )
 222              {
 223                  $result_info .= ' '.T_('Error');
 224              }
 225              break;
 226  
 227          case 'blocked':
 228              if( $params['display_icon'] )
 229              {
 230                  $result_info .= get_icon( 'bullet_black', 'imgtag', array( 'alt' => T_('Blocked') ) );
 231              }
 232              if( $params['display_text'] )
 233              {
 234                  if( $params['link_blocked'] && !empty( $params['email'] ) )
 235                  {    // Create a link for email address
 236                      global $admin_url;
 237                      $result_info .= ' <a href="'.$admin_url.'?ctrl=email&amp;tab=blocked&amp;email='.$params['email'].'">'.T_('Blocked').'</a>';
 238                  }
 239                  else
 240                  {    // Display only text
 241                      $result_info .= ' '.T_('Blocked');
 242                  }
 243              }
 244              break;
 245      }
 246  
 247      return $result_info;
 248  }

 252   * Add a mail log
 253   *
 254   * @param integer User ID
 255   * @param string To (email address)
 256   * @param string Subject
 257   * @param string Message
 258   * @param string Headers
 259   * @param string Result type ( 'ok', 'error', 'blocked' )
 260   */
 261  function mail_log( $user_ID, $to, $subject, $message, $headers, $result )
 262  {
 263      global $DB, $servertimenow;
 264  
 265      /**
 266       * @var integer|NULL This global var stores ID of the last inserted mail log
 267       */
 268      global $mail_log_insert_ID;
 269      $mail_log_insert_ID = NULL;
 270  
 271      if( empty( $user_ID ) )
 272      {
 273          $user_ID = NULL;
 274      }
 275  
 276      $to = utf8_strtolower( $to );
 277  
 278      // Insert mail log
 279      $DB->query( 'INSERT INTO T_email__log
 280          ( emlog_timestamp, emlog_user_ID, emlog_to, emlog_result, emlog_subject, emlog_message, emlog_headers )
 281          VALUES
 282          ( '.$DB->quote( date2mysql( $servertimenow ) ).',
 283            '.$DB->quote( $user_ID ).',
 284            '.$DB->quote( $to ).',
 285            '.$DB->quote( $result ).',
 286            '.$DB->quote( utf8_substr( $subject, 0, 255 ) ).',
 287            '.$DB->quote( $message ).',
 288            '.$DB->quote( $headers ).' )' );
 289  
 290      // Store ID of new inserted mail log
 291      $mail_log_insert_ID = $DB->insert_id;
 292  
 293      if( $result == 'ok' )
 294      { // Save a report about sending of this message in the table T_email__address
 295          // The mail sending is susccess. Update last sent date and increase a counter
 296          $DB->query( 'INSERT INTO T_email__address ( emadr_address, emadr_sent_count, emadr_sent_last_returnerror, emadr_last_sent_ts )
 297              VALUES( '.$DB->quote( $to ).', 1, 1, '.$DB->quote( date2mysql( $servertimenow ) ).' )
 298              ON DUPLICATE KEY UPDATE
 299                  emadr_sent_count = emadr_sent_count + 1,
 300                  emadr_sent_last_returnerror = emadr_sent_last_returnerror + 1,
 301                  emadr_last_sent_ts = '.$DB->quote( date( 'Y-m-d H:i:s', $servertimenow ) ) );
 302      }

 306  /**
 307   * Load the blocked emails from DB in cache
 308   *
 309   * @param array User IDs
 310   * @param array Blocked statuses to know what emails are blocked to send
 311   *     'unknown'     - Unknown
 312   *     'warning'     - Warning
 313   *     'suspicious1' - Suspicious 1
 314   *     'suspicious2' - Suspicious 2
 315   *     'suspicious3' - Suspicious 3
 316   *     'prmerror'    - Permament error
 317   *     'spammer'     - Spammer
 318   */
 319  function load_blocked_emails( $user_IDs, $blocked_statuses = array() )
 320  {
 321      global $DB, $cache_mail_is_blocked_status;
 322  
 323      if( empty( $user_IDs ) )
 324      { // No users, Exit here
 325          return;
 326      }
 327  
 328      if( !isset( $cache_mail_is_blocked_status ) )
 329      { // Init array first time
 330          $cache_mail_is_blocked_status = array();
 331      }
 332  
 333      $status_filter_name = implode( '_', $blocked_statuses );
 334      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ] ) )
 335      { // Init subarray for each filter by statuses
 336          $cache_mail_is_blocked_status[ $status_filter_name ] = array();
 337      }
 338  
 339      $SQL = new SQL();
 340      $SQL->SELECT( 'user_email, emadr_ID' );
 341      $SQL->FROM( 'T_users' );
 342      $SQL->FROM_add( 'LEFT JOIN T_email__address
 343           ON user_email = emadr_address
 344          AND '.get_mail_blocked_condition( true, $blocked_statuses ) );
 345      $SQL->WHERE( 'user_ID IN ( '.$DB->quote( $user_IDs ).' )' );
 346      $blocked_emails = $DB->get_assoc( $SQL->get() );
 347  
 348      foreach( $blocked_emails as $email => $email_blocked_ID )
 349      { // The blocked email has TRUE value; Trust emails - FALSE
 350          $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] = (boolean) $email_blocked_ID;
 351      }
 352  }

 355  /**
 356   * Check if the email address is blocked
 357   *
 358   * @param string Email address
 359   * @param array Blocked statuses to know what emails are blocked to send
 360   *     'unknown'     - Unknown
 361   *     'warning'     - Warning
 362   *     'suspicious1' - Suspicious 1
 363   *     'suspicious2' - Suspicious 2
 364   *     'suspicious3' - Suspicious 3
 365   *     'prmerror'    - Permament error
 366   *     'spammer'     - Spammer
 367   * @return boolean TRUE
 368   */
 369  function mail_is_blocked( $email, $blocked_statuses = array() )
 370  {
 371      global $cache_mail_is_blocked_status;
 372  
 373      if( !isset( $cache_mail_is_blocked_status ) )
 374      { // Init array first time
 375          $cache_mail_is_blocked_status = array();
 376      }
 377  
 378      $status_filter_name = implode( '_', $blocked_statuses );
 379      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ] ) )
 380      { // Init subarray for each filter by statuses
 381          $cache_mail_is_blocked_status[ $status_filter_name ] = array();
 382      }
 383  
 384      if( !isset( $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] ) )
 385      { // If we check status of this email first time - get it from DB and store in cache
 386          global $DB;
 387          $SQL = new SQL();
 388          $SQL->SELECT( 'emadr_ID' );
 389          $SQL->FROM( 'T_email__address' );
 390          $SQL->WHERE( 'emadr_address = '.$DB->quote( utf8_strtolower( $email ) ) );
 391          $SQL->WHERE_and( get_mail_blocked_condition( true, $blocked_statuses ) );
 392          $cache_mail_is_blocked_status[ $status_filter_name ][ $email ] = (boolean) $DB->get_var( $SQL->get() );
 393      }
 394  
 395      // Get email block status from cache variable
 396      return $cache_mail_is_blocked_status[ $status_filter_name ][ $email ];
 397  }

 414  function get_mail_blocked_condition( $is_blocked = true, $blocked_statuses = array() )
 415  {
 416      global $DB;
 417

 477  /**
 478   * Parse message of mail if content is in multipart or HTML format
 479   *
 480   * @param string Headers
 481   * @param string Message
 482   * @return array|boolean Mail data or FALSE when message has only Plain Text content
 483   */
 484  function mail_log_parse_message( $headers, $message )
 485  {
 486      preg_match( '/Content-Type: ([^;]+);/i', $headers, $header_matches );
 487  
 488      if( empty( $header_matches[1] ) )
 489      { // Incorrect headers, Exit here
 490          return false;
 491      }
 492  
 493      $data = array();
 494  
 495      if( $header_matches[1] == 'text/html' )
 496      { // Message has one content in HTML format
 497          $data['html'] = array(
 498                  'type'       => $header_matches[0],
 499                  'content'    => mail_log_parse_html_data( 'content', $message ),
 500                  'head_style' => mail_log_parse_html_data( 'head_style', $message ),
 501                  'body_style' => mail_log_parse_html_data( 'body_style', $message ),
 502                  'body_class' => mail_log_parse_html_data( 'body_class', $message ),
 503              );
 504  
 505          return $data;
 506      }
 507      elseif( $header_matches[1] != 'multipart/mixed' )
 508      { // Message content is not multipart
 509          return false;
 510      }
 511  
 512      preg_match( '/Content-Type: multipart\/alternative; boundary="([^;]+)"/i', $message, $boundary_matches );
 513  
 514      if( empty( $boundary_matches ) || empty( $boundary_matches[1] ) )
 515      { // No found boundary delimiter of message contents
 516          return false;
 517      }
 518  
 519      $boundary_delimiter = '--'.$boundary_matches[1];
 520  
 521      $contents = explode( $boundary_delimiter, $message );
 522      unset( $contents[0] );
 523      unset( $contents[3] );
 524  
 525      foreach( $contents as $content )
 526      {
 527          preg_match( '/(Content-Type: ([^;]+);(.+)){1}\n([\s\S\n]+)/i', $content, $type_matches );
 528  
 529          switch( $type_matches[2] )
 530          {
 531              case 'text/html':
 532                  // Get data of Plain Text content
 533                  $data['html'] = array(
 534                          'type'       => $type_matches[1],
 535                          'content'    => mail_log_parse_html_data( 'content', $type_matches[4] ),
 536                          'head_style' => mail_log_parse_html_data( 'head_style', $type_matches[4] ),
 537                          'body_style' => mail_log_parse_html_data( 'body_style', $type_matches[4] ),
 538                          'body_class' => mail_log_parse_html_data( 'body_class', $type_matches[4] ),
 539                      );
 540                  break;
 541  
 542              case 'text/plain':
 543                  // Get data of HTML content
 544                  $data['text'] = array(
 545                          'type'    => $type_matches[1],
 546                          'content' => $type_matches[4],
 547                      );
 548                  break;
 549          }
 550      }
 551  
 552      return $data;
 553  }

 556  /**
 557   * Extract the parts from html body message of mail
 558   *
 559   * @param string Data type: 'content', 'head_style', 'body_style', 'body_class'
 560   * @param string Message
 561   * @return string
 562   */
 563  function mail_log_parse_html_data( $type, $message )
 564  {
 565      switch( $type )
 566      {
 567          case 'content':
 568              // Get <body> content of html email message
 569              return preg_replace( '#.+<body[^>]*>(.+)</body>.+#is', '$1', $message );
 570              break;
 571  
 572          case 'head_style':
 573              // Get <style> content of html email message
 574              return preg_replace( '#.+<style[^>]*>(.+)</style>.+#is', '$1', $message );
 575              break;
 576  
 577          case 'body_style':
 578          case 'body_class':
 579              // Get class|style of <body> content of html email message
 580              $regexp_attr = str_replace( 'body_', '', $type );
 581              preg_match( '#.+<body[^>]*('.$regexp_attr.'="([^"]+)")[^>]*>.+#is', $message, $body_attrs_match );
 582              return empty( $body_attrs_match[2] ) ? '' : $body_attrs_match[2];
 583              break;
 584      }
 585  }

 588  /**
 589   * Check if SMTP Swift Mailer is available on this system
 590   *
 591   * @return boolean|string TRUE on success, Error message about why we cannot use SMTP
 592   */
 593  function check_smtp_mailer()
 594  {
 595      global $Settings;
 596  
 597      if( ! $Settings->get( 'smtp_enabled' ) )
 598      { // Swift Mailer is not enabled
 599          return T_( 'SMTP gateway is not enabled.' );
 600      }
 601  
 602      if( version_compare( phpversion(), '5.2', '<' ) )
 603      { // Swift Mailer works only with PHP 5.2 and higher
 604          return T_( 'PHP version must be 5.2 or higher to enable SMTP gateway.' );
 605      }
 606  
 607      if( ! function_exists( 'proc_open' ) )
 608      { // Swift Mailer requires proc_* functions
 609          return T_( 'The proc_* PHP functions are not available on this server.' );
 610      }
 611  
 612      $smtp_server_host = $Settings->get( 'smtp_server_host' );
 613      $smtp_server_port = $Settings->get( 'smtp_server_port' );
 614      if( empty( $smtp_server_host ) || empty( $smtp_server_port ) )
 615      { // These settings must be defined
 616          return T_( 'SMTP Host and Port Number must be defined to enable SMTP gateway.' );
 617      }
 618  
 619      $smtp_server_security = $Settings->get( 'smtp_server_security' );
 620      if( $smtp_server_security == 'ssl' || $smtp_server_security == 'tls' )
 621      { // Check if enabled encryption method is enabled in this system
 622          $available_transports = stream_get_transports();
 623          $method_is_available = false;
 624          foreach( $available_transports as $available_transport )
 625          {
 626              if( preg_match( '#^'.$smtp_server_security.'#i', $available_transport ) )
 627              { // Check if first symbols are match, because transport can be "ssl", "sslv2" or "sslv3"
 628                  $method_is_available = true;
 629                  break;
 630              }
 631          }
 632          if( ! $method_is_available )
 633          { // Stop the checking here because encryption method is not available
 634              return sprintf( T_( 'Encryption Method %s must be available on this system in order to enable SMTP gateway.' ), '<b>'.strtoupper( $smtp_server_security ).'</b>' );
 635          }
 636      }
 637  
 638      // SMTP can be used in this system
 639      return true;

 762              if( ! $result && $Settings->get( 'force_email_sending' ) )

 773  /**
 774   * Send mail by SMTP Swift Mailer
 775   *
 776   * @param string Receiver, or receivers of the mail
 777   * @param string Subject of the email
 778   * @param string|array Message OR Array: 'charset', 'full', 'html', 'text'
 779   * @param array Email headers
 780   * @param string Additional flags as command line options
 781   * @return boolean TRUE on success
 782   */
 783  function evo_mail_smtp( $to, $subject, $message, $headers = array(), $additional_parameters = '' )
 784  {
 785      $headers_string = get_mail_headers( $headers );
 786  
 787      // Check if we can use SMTP Swift Mailer
 788      if( check_smtp_mailer() === true && ( $Swift_Mailer = & get_Swift_Mailer() ) )
 789      { // Use Swift Mailer to send emails using SMTP
 790  
 791          // Get charset from content type part
 792          $charset = ( isset( $headers['Content-Type'] ) && preg_match( '#charset=(.+)$#i', $headers['Content-Type'], $charset ) ) ? $charset[1] : NULL;
 793  
 794          // Create a Swift_Message object
 795          $Swift_Message = Swift_Message::newInstance();
 796          // Subject:
 797          $Swift_Message->setSubject( $subject );
 798          // To:
 799          if( empty( $message['to_email'] ) )
 800          { // Use only email address
 801              $Swift_Message->setTo( $to );
 802          }
 803          else
 804          { // Use address with name
 805              $Swift_Message->setTo( $message['to_email'], $message['to_name'] );
 806          }
 807          // Body:
 808          if( isset( $headers['Content-Type'] ) && preg_match( '#^[^;]+#i', $headers['Content-Type'], $content_type ) )
 809          {
 810              switch( $content_type[0] )
 811              {
 812                  case 'multipart/mixed':
 813                      // MULTIPLE:
 814                      if( is_array( $message ) )
 815                      { // Body is multiple
 816                          $Swift_Message->setBody( $message['html'], 'multipart/alternative', $message['charset'] );
 817                          $Swift_Message->addPart( $message['html'], 'text/html', $message['charset'] );
 818                          $Swift_Message->addPart( $message['text'], 'text/plain', $message['charset'] );
 819                          break;
 820                      }
 821                      else
 822                      { // Unknown case, Send email with text/plain format
 823                          $content_type[0] = 'text/plain';
 824                      }
 825  
 826                  case 'text/html':
 827                      // HTML:
 828                  case 'text/plain':
 829                      // TEXT:
 830                      $Swift_Message->setBody( $message, $content_type, $charset );
 831                      break;
 832  
 833                  default:
 834                      // Unknown content type
 835                      $Swift_Message->setBody( $message, null, $charset );
 836                      break;
 837              }
 838          }
 839          else
 840          { // Unknown content type
 841              $Swift_Message->setBody( $message, null, $charset );
 842          }
 843          // From:
 844          if( ! empty( $message['from_email'] ) )
 845          { // Use address with name
 846              $Swift_Message->setFrom( $message['from_email'], $message['from_name'] );
 847          }
 848          elseif( ! empty( $headers['From'] ) )
 849          { // Use only email address
 850              $Swift_Message->setFrom( $headers['From'] );
 851          }
 852          if( ! empty( $headers['Reply-To'] ) )
 853          { // Reply-To:
 854              $Swift_Message->setReplyTo( $headers['Reply-To'] );
 855          }
 856          if( ! empty( $headers['Return-Path'] ) )
 857          { // Return-Path:
 858              $Swift_Message->setReturnPath( $headers['Return-Path'] );
 859          }
 860          if( ! empty( $headers['Date'] ) )
 861          { // Date:
 862              $Swift_Message->setDate( $headers['Date'] );
 863          }
 864  
 865          // Send the message
 866          return $Swift_Mailer->send( $Swift_Message );
 867      }
 868  
 869      // No email was sent
 870      return false;
 871  }
