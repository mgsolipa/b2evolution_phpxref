  75  /**
  76   * Check if a directory is ready for operation, i-e writable by PHP.
  77   *
  78   * @return integer result code, 0 means 'ok'
  79   */
  80  function system_check_dir( $directory = 'media', $relative_path = NULL )
  81  {
  82      global $media_path, $cache_path;
  83  
  84      switch( $directory )
  85      {
  86          case 'cache':
  87              $path = $cache_path;
  88              break;
  89  
  90          case 'media':
  91              $path = $media_path;
  92              break;
  93  
  94          default:
  95              return 1;
  96      }
  97  
  98      if( $relative_path != NULL )
  99      {
 100          $path .= $relative_path;
 101      }
 102  
 103      if( ! is_dir( $path ) )
 104      {
 105          return 2;
 106      }
 107      elseif( ! is_readable( $path ) )
 108      {
 109          return 3;
 110      }
 111      elseif( ! is_writable( $path ) )
 112      {
 113          return 4;
 114      }
 115      else
 116      {
 117          $tempfile_path = $path.'temp.tmp';
 118          if( !@touch( $tempfile_path ) || !@unlink( $tempfile_path ) )
 119          {
 120              return 5;
 121          }
 122      }
 123  
 124      if( $directory == 'cache' && $relative_path != NULL )
 125      { // Create .htaccess file with deny rules
 126          if( ! create_htaccess_deny( $cache_path ) )
 127          {
 128              return 6;
 129          }
 130      }
 131  
 132      return 0;
 133  }

 158  /**
 159   * Create _cache/ and  /_cache/plugins/ folders
 160   *
 161   * @return boolean false if cache/ folder not exists or has limited editing permission, true otherwise
 162   */
 163  function system_create_cache_folder( $verbose = false )
 164  {
 165      global $cache_path, $basepath;
 166  
 167      if( $cache_path == $basepath.'_cache/' )
 168      { // Cache path is the default one
 169          if( ! is_dir( $cache_path ) && is_dir( $basepath.'cache/' ) )
 170          { // The default cache folder doesn't exists yet, but the old default does exist
 171              if( @rename( $basepath.'cache/', $cache_path ) )
 172              { // Successful rename
 173                  if( $verbose )
 174                  { // Display successefully renamed message
 175                      echo '<strong>'.sprintf( T_('Your page cache folder was successfully renamed from "%s" to "%s".'), $basepath.'cache/', $cache_path ).'</strong>';
 176                  }
 177                  return true;
 178              }
 179              if( $verbose )
 180              { // Display error message
 181                  echo '<span class="error text-danger"><evo:error>'.sprintf( T_('You should rename your "%s" folder to "%s" for page caching to continue working without interruption.'), $basepath.'cache/', $cache_path ).'</evo:error></span>';
 182              }
 183              return false;
 184          }
 185      }
 186      // create /_cache folder
 187      mkdir_r( $cache_path );
 188      // check /cache folder
 189      if( system_check_dir( 'cache' ) > 0 )
 190      {
 191          if( $verbose )
 192          { // The cache folder is not readable/writable or doesn't exist
 193              echo '<span class="error text-danger"><evo:error>'.T_('The /cache folder could not be created/written to. b2evolution will still work but without caching, which will make it operate slower than optimal.').'</evo:error></span>';
 194          }
 195          return false;
 196      }
 197  
 198      // create /cache/plugins/ folder
 199      mkdir_r( $cache_path.'plugins/' );
 200      return true;
 201  }

 233  {
 234      global $Settings;

 236  
 237      $Blog = NULL;

 276  /**
 277   * Check and repair cache folders.
 278   */
 279  function system_check_caches( $repair = true )
 280  {
 281      global $DB;
 282  
 283      // Check cache/ folder
 284      $result = system_check_dir( 'cache' );
 285      if( $result > 0 )
 286      { // error with cache/ folder
 287          $failed = true;
 288          if( $repair && ( $result == 2 ) )
 289          { // if cache folder not exists, and should repair, then try to create it
 290              $failed = ( $failed && !system_create_cache_folder() );
 291          }
 292          if( $failed )
 293          { // could/should not repair
 294              list( $status, $message ) = system_get_result( $result, T_( 'Cache folder error' ).': ' );
 295              return array( $message );
 296          }
 297      }
 298  
 299      $error_messages = array();
 300      if( ( $result = system_check_blog_cache( NULL, $repair ) ) !== false )
 301      { // general cache folder should exists
 302          list( $status, $message ) = $result;
 303          if( $status != 'ok' )
 304          {
 305              $error_messages[] = T_( 'General cache folder error' ).': '.$message;
 306          }
 307      }
 308  
 309      $cache_enabled_blogs = system_get_blog_IDs( true );
 310      $BlogCache = & get_BlogCache();
 311      foreach( $cache_enabled_blogs as $blog_ID )
 312      { // blog's cache folder should exists
 313          if( ( $result = system_check_blog_cache( $blog_ID, $repair ) ) !== false )
 314          {
 315              list( $status, $message ) = $result;
 316              if( $status != 'ok' )
 317              {
 318                  $Blog = $BlogCache->get_by_ID( $blog_ID );
 319                  $error_messages[] = sprintf( T_( '&laquo;%s&raquo; page cache folder' ),  $Blog->get( 'shortname' ) ).': '.$message;
 320              }
 321          }
 322      }
 323  
 324      return $error_messages;
 325  }

 335      global $cache_path, $Settings, $Plugins, $DB;

 385  function system_check_db_utf8()
 386  {
 387      global $DB;
 388  
 389      $save_show_errors = $DB->show_errors;
 390      $save_halt_on_error = $DB->halt_on_error;
 391      $last_error = $DB->last_error;
 392      $error = $DB->error;
 393      // Blatantly ignore any error generated by mysqli::set_charset..

 395      $DB->halt_on_error = false;

 397      {
 398          $ok = false;
 399      }
 400      else
 401      {
 402          $ok = true;
 403      }
 404      $DB->show_errors = $save_show_errors;
 405      $DB->halt_on_error = $save_halt_on_error;
 406      $DB->last_error = $last_error;
 407      $DB->error = $error;
 408  
 409      return $ok;

 416  function system_check_process_user()
 417  {
 418      $process_uid = NULL;
 419      $process_user = NULL;
 420      if( function_exists('posix_geteuid') )
 421      {
 422          $process_uid = posix_geteuid();
 423  
 424          if( function_exists('posix_getpwuid')
 425              && ($process_user = posix_getpwuid($process_uid)) )
 426          {
 427              $process_user = $process_user['name'];
 428          }
 429  
 430          $running_as = sprintf( '%s (uid %s)',
 431              ($process_user ? $process_user : '?'), (!is_null($process_uid) ? $process_uid : '?') );
 432      }
 433      else
 434      {
 435          $running_as = '('.T_('Unknown').')';
 436      }
 437  
 438      return array( $process_uid, $process_user, $running_as );
 439  }

 524  /**
 525   * Get how much bytes php ini value takes
 526   *
 527   * @param string PHP ini value,
 528   *    Examples:
 529   *         912 - 912 bytes
 530   *          4K - 4 Kilobytes
 531   *         13M - 13 Megabytes
 532   *          8G - 8 Gigabytes
 533   * @return integer Bytes
 534   */
 535  function get_php_bytes_size( $php_ini_value )
 536  {
 537      if( (string) intval( $php_ini_value ) === (string) $php_ini_value )
 538      { // Bytes
 539          return $php_ini_value;
 540      }
 541      elseif( strpos( $php_ini_value, 'K' ) !== false )
 542      { // Kilobytes
 543          return intval( $php_ini_value ) * 1024;
 544      }
 545      elseif( strpos( $php_ini_value, 'M' ) !== false  )
 546      { // Megabytes
 547          return intval( $php_ini_value ) * 1024 * 1024;
 548      }
 549      elseif( strpos( $php_ini_value, 'G' ) !== false  )
 550      { // Gigabytes
 551          return intval( $php_ini_value ) * 1024 * 1024 * 1024;
 552      }
 553  
 554      // Unknown format
 555      return $php_ini_value;
 556  }

 560   * Check if some of the tables have different charset than what we expect
 561   *
 562   * @return boolean TRUE if the update is required
 563   */
 564  function system_check_charset_update()
 565  {
 566      global $DB, $evo_charset, $db_config, $tableprefix;
 567  
 568      $expected_connection_charset = DB::php_to_mysql_charmap( $evo_charset );
 569  
 570      $curr_db_charset = $DB->get_var( 'SELECT default_character_set_name
 571          FROM information_schema.SCHEMATA
 572          WHERE schema_name = '.$DB->quote( $db_config['name'] ) );
 573  
 574      $require_charset_update = ( $curr_db_charset != $expected_connection_charset ) ||
 575          $DB->get_var( 'SELECT COUNT( T.table_name )
 576              FROM information_schema.`TABLES` T,
 577                  information_schema.`COLLATION_CHARACTER_SET_APPLICABILITY` CCSA
 578              WHERE CCSA.collation_name = T.table_collation
 579                  AND T.table_schema = '.$DB->quote( $db_config['name'] ).'
 580                  AND T.table_name LIKE "'.$tableprefix.'%"
 581                  AND CCSA.character_set_name != '.$DB->quote( $expected_connection_charset ) );
 582  
 583      return $require_charset_update;
