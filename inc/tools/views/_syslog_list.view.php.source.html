 177  /**
 178   * Get a link to object of system log
 179   *
 180   * @param string Object type
 181   * @param integer Object ID
 182   * @return string
 183   */
 184  function syslog_object_link( $object_type, $object_ID )
 185  {
 186      global $current_User, $admin_url;
 187  
 188      $link = '';
 189  
 190      if( empty( $object_ID ) )
 191      { // Invalid object ID
 192          return 'Empty object ID';
 193      }
 194  
 195      switch( $object_type )
 196      {
 197          case 'comment':
 198              // Link to comment
 199              $CommentCache = & get_CommentCache();
 200              if( ( $Comment = & $CommentCache->get_by_ID( $object_ID, false, false ) ) !== false )
 201              {
 202                  if( $current_User->check_perm( 'comment!CURSTATUS', 'edit', false, $Comment ) )
 203                  { // Current user has permission to edit this comment
 204                      $Item = & $Comment->get_Item();
 205                      $link = '<a href="'.$admin_url.'?ctrl=comments&action=edit&comment_ID='.$Comment->ID.'">'.$Item->title.' #'.$Comment->ID.'</a>';
 206                  }
 207              }
 208              else
 209              { // Comment was deleted or ID is incorrect
 210                  $link = 'No comment';
 211              }
 212              break;
 213  
 214          case 'item':
 215              // Link to item
 216              $ItemCache = & get_ItemCache();
 217              if( ( $Item = & $ItemCache->get_by_ID( $object_ID, false, false ) ) !== false )
 218              {
 219                  if( $current_User->check_perm( 'item_post!CURSTATUS', 'edit', false, $Item ) )
 220                  { // Current user has permission to edit this item
 221                      $link = '<a href="'.$Item->get_edit_url().'">'.$Item->title.'</a>';
 222                  }
 223              }
 224              else
 225              { // Item was deleted or ID is incorrect
 226                  $link = 'No item';
 227              }
 228              break;
 229  
 230          case 'user':
 231              // Link to user
 232              if( $current_User->check_perm( 'users', 'view' ) )
 233              { // Current user has permission to view users
 234                  $UserCache = get_UserCache();
 235                  if( ( $User = & $UserCache->get_by_ID( $object_ID, false, false ) ) !== false )
 236                  { // User exists
 237                      $link = $User->get_identity_link();
 238                  }
 239                  else
 240                  { // User was deleted or ID is incorrect
 241                      $link = 'No user';
 242                  }
 243              }
 244              break;
 245  
 246          case 'file':
 247              // Link to file
 248              $FileCache = & get_FileCache();
 249              if( ( $File = & $FileCache->get_by_ID( $object_ID, false, false ) ) !== false )
 250              { // File exists
 251                  $link = $File->is_dir() ? '' : $File->get_view_link();
 252                  $link .= ' '.$File->get_target_icon();
 253              }
 254              else
 255              { // User was deleted or ID is incorrect
 256                  $link = 'No file';
 257              }
 258              break;
 259      }
 260  
 261      return $link;
 262  }
