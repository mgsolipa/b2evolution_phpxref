  85  {
  86      global $Settings, $Plugins, $force_upload_forbiddenext;
  87  
  88      // CHECK LOGIN:
  89      /**
  90       * @var User
  91       */
  92      if( ! $current_User = & xmlrpcs_login( $m, 1, 2 ) )
  93      {    // Login failed, return (last) error:
  94          return xmlrpcs_resperror();
  95      }
  96  
  97      // GET BLOG:
  98      /**
  99       * @var Blog
 100       */
 101      if( ! $Blog = & xmlrpcs_get_Blog( $m, 0 ) )
 102      {    // Login failed, return (last) error:
 103          return xmlrpcs_resperror();
 104      }
 105  
 106      // CHECK PERMISSION:
 107      if( ! $current_User->check_perm( 'files', 'add', false, $Blog->ID ) )
 108      {    // Permission denied
 109          return xmlrpcs_resperror( 3 );    // User error 3
 110      }
 111      logIO( 'Permission granted.' );
 112  
 113      if( ! $Settings->get('upload_enabled') )
 114      {
 115          return xmlrpcs_resperror( 2, 'Object upload not allowed' );
 116      }
 117  
 118      $xcontent = $m->getParam(3);
 119      // Get the main data - and decode it properly for the image - sorry, binary object
 120      logIO( 'Decoding content...' );
 121      $contentstruct = xmlrpc_decode_recurse($xcontent);
 122      $data = $contentstruct['bits'];
 123  
 124      $file_mimetype = isset($contentstruct['type']) ? $contentstruct['type'] : '(none)';
 125      logIO( 'Received MIME type: '.$file_mimetype );
 126  
 127      $overwrite = false;
 128      if( isset($contentstruct['overwrite']) )
 129      {
 130          $overwrite = (bool) $contentstruct['overwrite'];
 131      }
 132      logIO( 'Overwrite if exists: '.($overwrite ? 'yes' : 'no') );
 133  
 134      load_funcs('files/model/_file.funcs.php');
 135  
 136      $filesize = evo_bytes( $data );
 137      if( ( $maxfilesize = $Settings->get('upload_maxkb') * 1024 ) && $filesize > $maxfilesize )
 138      {
 139          return xmlrpcs_resperror( 4, sprintf( T_('The file is too large: %s but the maximum allowed is %s.'),
 140                      bytesreadable($filesize, false), bytesreadable($maxfilesize, false) ) );
 141      }
 142      logIO( 'File size is OK: '.bytesreadable($filesize, false) );
 143  
 144      $FileRootCache = & get_FileRootCache();
 145      $fm_FileRoot = & $FileRootCache->get_by_type_and_ID( 'collection', $Blog->ID, true );
 146      if( ! $fm_FileRoot )
 147      {    // fileRoot not found:
 148          return xmlrpcs_resperror( 14, 'File root not found' );
 149      }
 150  
 151      $rf_filepath = $contentstruct['name'];
 152      logIO( 'Received filepath: '.$rf_filepath );
 153  
 154      // Split into path + name:
 155      $filepath_parts = explode( '/', $rf_filepath );
 156      $filename = array_pop( $filepath_parts );
 157      logIO( 'Original file name: '.$filename );
 158  
 159      // Validate and sanitize filename
 160      if( $error_filename = process_filename( $filename, true ) )
 161      {
 162          return xmlrpcs_resperror( 5, $error_filename );
 163      }
 164      logIO( 'Sanitized file name: '.$filename );
 165  
 166      // Check valid path parts:
 167      $rds_subpath = '';
 168      foreach( $filepath_parts as $filepath_part )
 169      {
 170          if( empty($filepath_part) || $filepath_part == '.' )
 171          {    // self ref not useful
 172              continue;
 173          }
 174  
 175          if( $error = validate_dirname($filepath_part) )
 176          {    // invalid relative path:
 177              logIO( $error );
 178              syslog_insert( sprintf( 'Invalid name is detected for folder %s', '<b>'.$filepath_part.'</b>' ), 'warning', 'file' );
 179              return xmlrpcs_resperror( 6, $error );
 180          }

 183      }
 184      logIO( 'Subpath: '.$rds_subpath );
 185  
 186      // Create temporary file and insert contents into it.
 187      $tmpfile_name = tempnam(sys_get_temp_dir(), 'fmupload');
 188      if( $tmpfile_name )
 189      {
 190          if( save_to_file( $data, $tmpfile_name, 'wb' ) )
 191          {
 192              $image_info = @getimagesize($tmpfile_name);
 193          }

 195          {
 196              return xmlrpcs_resperror( 13, 'Error while writing to temp file.' );
 197          }
 198      }

 202          if( $image_info['mime'] != $file_mimetype )
 203          {    // Invalid file type

 206              $correct_Filetype = $FiletypeCache->get_by_mimetype( $image_info['mime'], false, false );
 207  
 208              $file_mimetype = $image_info['mime'];
 209  
 210              // Check if file type is known by us, and if it is allowed for upload.
 211              // If we don't know this file type or if it isn't allowed we don't change the extension! The current extension is allowed for sure.
 212              if( $correct_Filetype && $correct_Filetype->is_allowed() )
 213              {    // A FileType with the given mime type exists in database and it is an allowed file type for current User
 214                  // The "correct" extension is a plausible one, proceed...
 215                  $correct_extension = array_shift($correct_Filetype->get_extensions());
 216                  $path_info = pathinfo($filename);
 217                  $current_extension = $path_info['extension'];
 218  
 219                  // change file extension to the correct extension, but only if the correct extension is not restricted, this is an extra security check!
 220                  if( strtolower($current_extension) != strtolower($correct_extension) && ( !in_array( $correct_extension, $force_upload_forbiddenext ) ) )
 221                  {    // change the file extension to the correct extension
 222                      $old_filename = $filename;
 223                      $filename = $path_info['filename'].'.'.$correct_extension;
 224                  }
 225              }
 226          }
 227      }

 230      $FileCache = & get_FileCache();

 232  
 233      if( $newFile->exists() )
 234      {
 235          if( $overwrite && $newFile->unlink() )
 236          {    // OK, file deleted
 237              // Delete thumb caches from old location:
 238              logIO( 'Old file deleted' );
 239              $newFile->rm_cache();
 240          }
 241          else
 242          {
 243              return xmlrpcs_resperror( 8, sprintf( T_('The file &laquo;%s&raquo; already exists.'), $filename ) );
 244          }
 245      }
 246  
 247      // Trigger plugin event
 248      if( $Plugins->trigger_event_first_false( 'AfterFileUpload', array(
 249                  'File' => & $newFile,
 250                  'name' => & $filename,
 251                  'type' => & $file_mimetype,
 252                  'tmp_name' => & $tmpfile_name,
 253                  'size' => & $filesize,
 254              ) ) )
 255      {
 256          // Plugin returned 'false'.
 257          // Abort upload for this file:
 258          @unlink($tmpfile_name);
 259          return xmlrpcs_resperror( 16, 'File upload aborted by a plugin.' );
 260      }
 261  
 262      if( ! mkdir_r( $newFile->get_dir() ) )
 263      {    // Dir didn't already exist and could not be created
 264          return xmlrpcs_resperror( 9, 'Error creating sub directories: '.$newFile->get_rdfs_rel_path() );
 265      }
 266  
 267      if( ! @rename( $tmpfile_name, $newFile->get_full_path() ) )
 268      {
 269          return xmlrpcs_resperror( 13, 'Error while writing to file.' );
 270      }
 271  
 272      // chmod the file
 273      $newFile->chmod();
 274  
 275      // Initializes file properties (type, size, perms...)
 276      $newFile->load_properties();
 277  
 278      // Load meta data AND MAKE SURE IT IS CREATED IN DB:
 279      $newFile->meta == 'unknown';
 280      $newFile->load_meta( true );
 281  
 282      // Resize and rotate
 283      logIO( 'Running file post-processing (resize and rotate)...' );
 284      prepare_uploaded_files( array($newFile) );
 285      logIO( 'Done' );
 286  
 287      $url = $newFile->get_url();
 288      logIO( 'URL of new file: '.$url );
 289  
 290      $struct = new xmlrpcval(array(
 291              'file' => new xmlrpcval($filename, 'string'),
 292              'url' => new xmlrpcval($url, 'string'),
 293              'type' => new xmlrpcval($file_mimetype, 'string'),
 294          ), 'struct');
 295  
 296      logIO( 'OK.' );
 297      return new xmlrpcresp($struct);
 298  }

1707  function xmlrpcs_check_cats( & $maincat, & $Blog, & $extracats )
1708  {

1721          if( ( $key = array_search( $maincat, $extracats ) ) !== false )

1731      {    // We cannot use a maincat of another blog than the current one:
1732          $xmlrpcs_errcode = $xmlrpcerruser + 11;

1755      {
1756          logIO( '$maincat was not found in $extracats array - adding.' );
