  38  /**
  39   * mt.setPostCategories : set cats for a post
  40   *
  41   * @param xmlrpcmsg XML-RPC Message
  42   *                    0 postid (string): Unique identifier of the post to edit
  43   *                    1 username (string): Login for a Blogger user who is member of the blog.
  44   *                    2 password (string): Password for said username.
  45   */
  46  function mt_setPostCategories($m)
  47  {
  48      // CHECK LOGIN:
  49      /**
  50       * @var User
  51       */
  52      if( ! $current_User = & xmlrpcs_login( $m, 1, 2 ) )
  53      {    // Login failed, return (last) error:
  54          return xmlrpcs_resperror();
  55      }
  56  
  57      // GET POST:
  58      /**
  59       * @var Item
  60       */
  61      if( ! $edited_Item = & xmlrpcs_get_Item( $m, 0 ) )
  62      {    // Failed, return (last) error:
  63          return xmlrpcs_resperror();
  64      }
  65  
  66      if( ! $current_User->check_perm( 'item_post!CURSTATUS', 'edit', false, $edited_Item ) )
  67      {    // Permission denied
  68          return xmlrpcs_resperror( 3 );
  69      }
  70  
  71      $xcontent = $m->getParam(3); // This is now an array of structs
  72      $contentstruct = xmlrpc_decode_recurse($xcontent);
  73      logIO('Decoded xcontent');
  74  
  75      $categories = array();
  76      $category = NULL;
  77      foreach( $contentstruct as $catstruct )
  78      {
  79          logIO( 'Category ID: '.$catstruct['categoryId'] ) ;
  80          if( !empty($catstruct['isPrimary']) )
  81          {
  82              $category = $catstruct['categoryId'];
  83              logIO('got primary category and there should only be one... '.$category);
  84          }
  85          $categories[] = $catstruct['categoryId'];
  86      }
  87  
  88      if( empty( $categories ) )
  89      {
  90          return xmlrpcs_resperror( 4, 'No categories specified.' );
  91      }
  92      else if( empty( $category ) )
  93      {    // Use first one as default:
  94          $category = $categories[0];
  95      }
  96  
  97      // Check if category exists and can be used:
  98      $Blog = & $edited_Item->get_Blog();
  99      if( ! xmlrpcs_check_cats( $category, $Blog, $categories ) )
 100      {    // Error:
 101          return xmlrpcs_resperror();
 102      }
 103  
 104      // CHECK PERMISSION: (we need perm on all categories, especially if they are in different blogs)
 105      if( ! $current_User->check_perm( 'cats_post!'.$edited_Item->status, 'edit', false, $categories ) )
 106      {    // Permission denied
 107          return xmlrpcs_resperror( 3 );    // User error 3
 108      }
 109      logIO( 'Permission granted.' );
 110  
 111      logIO( 'Main Cat: '.$category.' - Other: '.implode(',',$categories) );
 112  
 113      // UPDATE POST CATEGORIES IN DB:
 114      $edited_Item->set( 'main_cat_ID', $category );
 115      $edited_Item->set( 'extra_cat_IDs', $categories );
 116  
 117      if( $edited_Item->dbupdate() === false )
 118      {
 119          logIO( 'Update failed.' );
 120          return xmlrpcs_resperror( 99, 'Update failed.' );
 121      }
 122  
 123      logIO( 'OK.' );
 124      return new xmlrpcresp( new xmlrpcval( 1, 'boolean' ) );
 125  }

 139  function mt_getPostCategories($m)
 140  {
 141      global $DB;
 142  
 143      // CHECK LOGIN:
 144      /**
 145       * @var User
 146       */
 147      if( ! $current_User = & xmlrpcs_login( $m, 1, 2 ) )
 148      {    // Login failed, return (last) error:
 149          return xmlrpcs_resperror();
 150      }
 151  
 152      // GET POST:
 153      /**
 154       * @var Item
 155       */
 156      if( ! $edited_Item = & xmlrpcs_get_Item( $m, 0 ) )
 157      {    // Failed, return (last) error:
 158          return xmlrpcs_resperror();
 159      }
 160  
 161      // CHECK PERMISSION (user needs to be able to view the post):
 162      if( ! xmlrpcs_can_view_item( $edited_Item, $current_User ) )
 163      {    // Permission denied
 164          return xmlrpcs_resperror( 3 );    // User error 3
 165      }
 166  
 167      // Tblue> TODO: We could save one DB query by using our own custom query
 168      //              instead of postcats_get_byID().
 169      $categories = postcats_get_byID( $edited_Item->ID ); // Secondary categories
 170      $iSize = count($categories); // The number of objects ie categories
 171      logIO('mt_getPostCategories  no. of categories... '.$iSize);// works
 172      $struct = array();
 173      for( $i=0; $i<$iSize; $i++)
 174      {
 175          logIO('mt_getPostCategories categories  ...'.$categories[$i]);
 176          // In database cat_ID and cat_name from tablecategories
 177          $sql = 'SELECT * FROM T_categories WHERE  cat_ID = '.$categories[$i];
 178          logIO('mt_getPostCategories  sql...'.$sql);
 179          $rows = $DB->get_results( $sql );
 180          foreach( $rows as $row )
 181          {
 182              $Categoryname =  $row->cat_name;
 183              logIO('mt_getPostCategories Categoryname  ...'.$Categoryname);
 184          }
 185  
 186          // Is this the primary cat?
 187          $isPrimary = ($categories[$i] == $edited_Item->main_cat_ID) ? 1 : 0;
 188  
 189          $struct[$i] = new xmlrpcval(array('categoryId' => new xmlrpcval($categories[$i]),    // Look up name from ID separately
 190                                          'categoryName' => new xmlrpcval($Categoryname),
 191                                          'isPrimary' => new xmlrpcval($isPrimary, 'boolean'),
 192                                          ),'struct');
 193      }
 194  
 195       logIO( 'OK.' );
 196      return new xmlrpcresp(new xmlrpcval($struct, 'array') );
 197  }

 231  function mt_publishPost($m)
 232  {
 233      global $localtimenow, $DB;
 234  
 235      // CHECK LOGIN:
 236      /**
 237       * @var User
 238       */
 239      if( ! $current_User = & xmlrpcs_login( $m, 1, 2 ) )
 240      {    // Login failed, return (last) error:
 241          return xmlrpcs_resperror();
 242      }
 243      logIO( 'mt_publishPost: Login OK' );
 244  
 245      // GET POST:
 246      /**
 247       * @var Item
 248       */
 249      if( ! $edited_Item = & xmlrpcs_get_Item( $m, 0 ) )
 250      {    // Failed, return (last) error:
 251          return xmlrpcs_resperror();
 252      }
 253  
 254      if( ! $current_User->check_perm( 'item_post!published', 'edit', false, $edited_Item )
 255          /*|| ! $current_User->check_perm( 'edit_timestamp' )*/ )
 256      {
 257          return xmlrpcs_resperror( 3 ); // Permission denied
 258      }
 259      logIO('mt_publishPost: Permission granted');
 260  
 261      logIO( 'mt_publishPost: Old post status: '.$edited_Item->status );
 262      $edited_Item->set( 'status', 'published' );
 263      //$edited_Item->set( 'datestart', date('Y-m-d H:i:s', $localtimenow) );
 264  
 265      if( $edited_Item->dbupdate() === false )
 266      {    // Could not update item...
 267          return xmlrpcs_resperror( 99, 'Database error: '.$DB->last_error ); // DB error
 268      }
 269      logIO('mt_publishPost: Item published.');
 270  
 271      // Execute or schedule notifications & pings:
 272      logIO( 'mt_publishPost: Handling notifications...' );
 273      $edited_Item->handle_post_processing( false, false );
 274  
 275      logIO( 'mt_publishPost: OK.' );
 276  
 277      return new xmlrpcresp( new xmlrpcval( 1, 'boolean' ) );
 278  }
