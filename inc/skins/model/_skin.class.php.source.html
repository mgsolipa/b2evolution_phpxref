 200       * Display a container
 201       *
 202       * @todo fp> if it doesn't get any skin specific, move it outta here! :P
 203       * fp> Do we need Skin objects in the frontoffice at all? -- Do we want to include the dispatcher into the Skin object? WARNING: globals

 209  	function container( $sco_name, $params = array() )
 210      {
 211          /**
 212           * Blog currently displayed
 213           * @var Blog
 214           */
 215          global $Blog;
 216          global $admin_url, $rsc_url;
 217          global $Timer, $Session;
 218  
 219          $timer_name = 'skin_container('.$sco_name.')';
 220          $Timer->start( $timer_name );
 221  
 222          $display_containers = $Session->get( 'display_containers_'.$Blog->ID ) == 1;
 223  
 224          if( $display_containers )
 225          { // Wrap container in visible container:
 226              echo '<div class="dev-blocks dev-blocks--container">';
 227              echo '<div class="dev-blocks-name"><span class="dev-blocks-action"><a href="'
 228                          .$admin_url.'?ctrl=widgets&amp;blog='.$Blog->ID.'">Edit</a></span>Container: <b>'.$sco_name.'</b></div>';
 229          }
 230  
 231          /**
 232           * @var EnabledWidgetCache
 233           */
 234          $EnabledWidgetCache = & get_EnabledWidgetCache();
 235          $Widget_array = & $EnabledWidgetCache->get_by_coll_container( $Blog->ID, $sco_name );
 236  
 237          if( ! empty( $Widget_array ) )
 238          {
 239              foreach( $Widget_array as $w => $ComponentWidget )
 240              { // Let the Widget display itself (with contextual params):
 241                  if( $w == 0 )
 242                  { // Use special params for first widget in the current container
 243                      $orig_params = $params;
 244                      if( isset( $params['block_first_title_start'] ) )
 245                      {
 246                          $params['block_title_start'] = $params['block_first_title_start'];
 247                      }
 248                      if( isset( $params['block_first_title_end'] ) )
 249                      {
 250                          $params['block_title_end'] = $params['block_first_title_end'];
 251                      }
 252                  }
 253                  $widget_timer_name = 'Widget->display('.$ComponentWidget->code.')';
 254                  $Timer->start( $widget_timer_name );
 255                  $ComponentWidget->display_with_cache( $params, array(
 256                          // 'sco_name' => $sco_name, // fp> not sure we need that for now
 257                      ) );
 258                  if( $w == 0 )
 259                  { // Restore the params for next widgets after first
 260                      $params = $orig_params;
 261                      unset( $orig_params );
 262                  }
 263                  $Timer->pause( $widget_timer_name );
 264              }
 265          }
 266  
 267          if( $display_containers )
 268          { // End of visible container:
 269              //echo get_icon( 'pixel', 'imgtag', array( 'class' => 'clear' ) );
 270              echo '</div>';
 271          }
 272  
 273          $Timer->pause( $timer_name );
 274      }

 277      /**
 278       * Discover containers included in skin files only in the given folder
 279       *
 280       * @param string Folder name
 281       * @param array Exclude the files
 282       * @return array Files that were prepared
 283       */
 284  	function discover_containers_by_folder( $folder, $exclude_files = array() )
 285      {
 286          global $skins_path, $Messages;
 287  
 288          if( ! $dir = @opendir( $skins_path.$folder ) )
 289          { // Skin directory not found!
 290              $Messages->add( T_('Cannot open skin directory.'), 'error' ); // No trans
 291              return false;
 292          }
 293  
 294          // Store the file names to return
 295          $files = array();
 296  
 297          // Go through all files in the skin directory:
 298          while( ( $file = readdir( $dir ) ) !== false )
 299          {
 300              if( in_array( $file, $exclude_files ) )
 301              { // Skip this file
 302                  continue;
 303              }
 304  
 305              $rf_main_subpath = trim( $folder.'/'.$file, '/' );
 306              $af_main_path = $skins_path.$rf_main_subpath;
 307  
 308              if( !is_file( $af_main_path ) || ! preg_match( '~\.php$~', $file ) )
 309              { // Not a php template file, go to next:
 310                  continue;
 311              }
 312  
 313              if( ! is_readable( $af_main_path ) )
 314              { // Cannot open PHP file:
 315                  $Messages->add( sprintf( T_('Cannot read skin file &laquo;%s&raquo;!'), $rf_main_subpath ), 'error' );
 316                  continue;
 317              }
 318  
 319              $file_contents = @file_get_contents( $af_main_path );
 320              if( ! is_string( $file_contents ) )
 321              { // Cannot get contents:
 322                  $Messages->add( sprintf( T_('Cannot read skin file &laquo;%s&raquo;!'), $rf_main_subpath ), 'error' );
 323                  continue;
 324              }
 325  
 326              $files[] = $files;
 327  
 328              // DETECT if the file contains containers:
 329              // if( ! preg_match_all( '~ \$Skin->container\( .*? (\' (.+?) \' )|(" (.+?) ") ~xmi', $file_contents, $matches ) )
 330              if( ! preg_match_all( '~ (\$Skin->|skin_)container\( .*? ((\' (.+?) \')|(" (.+?) ")) ~xmi', $file_contents, $matches ) )
 331              { // No containers in this file, go to next:
 332                  continue;
 333              }
 334  
 335              // Merge matches from the two regexp parts (due to regexp "|" )
 336              $container_list = array_merge( $matches[4], $matches[6] );
 337  
 338              $c = 0;
 339              foreach( $container_list as $container )
 340              {
 341                  if( empty( $container ) )
 342                  { // regexp empty match -- NOT a container:
 343                      continue;
 344                  }
 345  
 346                  // We have one more container:
 347                  $c++;
 348  
 349                  if( in_array( $container, $this->container_list ) )
 350                  { // we already have that one
 351                      continue;
 352                  }
 353  
 354                  $this->container_list[] = $container;
 355              }
 356  
 357              if( $c )
 358              {
 359                  $Messages->add( sprintf( T_('%d containers have been found in skin template &laquo;%s&raquo;.'), $c, $rf_main_subpath ), 'success' );
 360              }
 361          }
 362  
 363          return $files;
 364      }

 367      /**
 368       * Discover containers included in skin files
 369       */
 370  	function discover_containers()
 371      {
 372          global $Messages;
 373  
 374          $this->container_list = array();
 375  
 376          // Find the containers in the current skin folder
 377          $skin_files = $this->discover_containers_by_folder( $this->folder );
 378  
 379          // Find the containers in the root skins folder with excluding the files that are contained in the skin folder
 380          $this->discover_containers_by_folder( '', $skin_files );
 381  
 382          if( empty( $this->container_list ) )
 383          {
 384              $Messages->add( T_('No containers found in this skin!'), 'error' );
 385              return false;
 386          }
 387  
 388          return true;
 389      }

 460      /**
 461       * Save containers
 462       *
 463       * to be called by dbinsert / dbupdate
 464       */
 465  	function db_save_containers()
 466      {
 467          global $DB;
 468  
 469          // Get a list of all currently empty containers:
 470          $sql = 'SELECT sco_name
 471                            FROM T_skins__container LEFT JOIN T_widget ON ( sco_name = wi_sco_name )
 472                           WHERE sco_skin_ID = '.$this->ID.'
 473                           GROUP BY sco_name
 474                          HAVING COUNT(wi_ID) = 0';
 475          $empty_containers_list = $DB->get_col( $sql, 0, 'Get empty containers' );
 476          //pre_dump( $empty_containers_list );
 477  
 478          // Look for containers in skin file:
 479          $this->discover_containers();
 480  
 481          // Delete empty containers:
 482          foreach( $empty_containers_list as $empty_container )
 483          {
 484              if( !in_array( $empty_container, $this->container_list ) )
 485              {    // This container has been removed from the skin + it's empty, so delete it from DB:
 486                  $DB->query( 'DELETE FROM T_skins__container
 487                                      WHERE sco_name = '.$DB->quote($empty_container) );
 488              }
 489          }
 490  
 491          // Make sure new containers are added:
 492          if( ! empty( $this->container_list ) )
 493          {
 494              $values = array();
 495              foreach( $this->container_list as $container_name )
 496              {
 497                  $values [] = '( '.$this->ID.', '.$DB->quote($container_name).' )';
 498              }
 499  
 500              $DB->query( 'REPLACE INTO T_skins__container( sco_skin_ID, sco_name )
 501                                          VALUES '.implode( ',', $values ), 'Insert containers' );
 502          }
 503      }

 661      /**
 662       * Get a skin specific param value from current Blog
 663       *
 664       * @param string Setting name
 665       * @return string|array|NULL
 666       */
 667  	function get_setting( $parname )
 668      {
 669          /**
 670           * @var Blog
 671           */
 672          global $Blog;
 673  
 674          // Name of the setting in the blog settings:
 675          $blog_setting_name = 'skin'.$this->ID.'_'.$parname;
 676  
 677          $value = $Blog->get_setting( $blog_setting_name );
 678  
 679          if( ! is_null( $value ) )
 680          {    // We have a value for this param:
 681              return $value;
 682          }
 683  
 684          // Try default values:
 685          $params = $this->get_param_definitions( NULL );
 686          if( isset( $params[ $parname ]['defaultvalue'] ) )
 687          { // We have a default value:
 688              return $params[ $parname ]['defaultvalue'] ;
 689          }
 690          elseif( isset( $params[ $parname ]['type'] ) &&
 691                  $params[ $parname ]['type'] == 'checklist' &&
 692                  ! empty( $params[ $parname ]['options'] ) )
 693          { // Get default values for checkbox list:
 694              $options = array();
 695              foreach( $params[ $parname ]['options'] as $option )
 696              {
 697                  if( isset( $option[2] ) )
 698                  { // Set default value only if it is defined by skin:
 699                      $options[ $option[0] ] = $option[2];
 700                  }
 701              }
 702              return $options;

1131      /**
1132       * Translate a given string, in the Skin's context.
1133       *
1134       * This means, that the translation is obtained from the Skin's
1135       * "locales" folder.
1136       *
1137       * It uses the global/regular {@link T_()} function as a fallback.
1138       *
1139       * @param string The string (english), that should be translated
1140       * @param string Requested locale ({@link $current_locale} gets used by default)
1141       * @return string The translated string.
1142       *
1143       * @uses T_()
1144       * @since 3.2.0 (after beta)
1145       */
1146      function T_( $string, $req_locale = '' )
1147      {
1148          global $skins_path;
1149  
1150          if( ( $return = T_( $string, $req_locale, array(
1151                                  'ext_transarray' => & $this->_trans,
1152                                  'alt_basedir'    => $skins_path.$this->folder,
1153                              ) ) ) == $string )
1154          {    // This skin did not provide a translation - fallback to global T_():
1155              return T_( $string, $req_locale );
1156          }
1157  
1158          return $return;
1159      }

1164       *
1165       * This is to be used mainly for Javascript strings.
1166       *
1167       * @param string String to translate
1168       * @param string Locale to use
1169       * @return string The translated and escaped string.
1170       *
1171       * @uses Skin::T_()
1172       * @since 3.2.0 (after beta)
1173       */
1174  	function TS_( $string, $req_locale = '' )
