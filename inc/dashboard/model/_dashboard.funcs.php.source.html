  16   /**
  17   * Get updates from b2evolution.net
  18   *
  19   * @param boolean useful when trying to upgrade to a release that has just been published (in the last 12 hours)
  20   * @return NULL|boolean True if there have been updates, false on error,
  21   *                      NULL if the user has turned off updates.
  22   */
  23  function b2evonet_get_updates( $force_short_delay = false )
  24  {
  25      global $allow_evo_stats; // Possible values: true, false, 'anonymous'
  26      global $DB, $debug, $evonetsrv_host, $evonetsrv_port, $evonetsrv_uri, $servertimenow, $evo_charset;
  27      global $Messages, $Settings, $baseurl, $instance_name, $app_name, $app_version, $app_date;
  28      global $Debuglog;
  29      global $Timer;
  30      global $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password;
  31  
  32      if( ! isset( $allow_evo_stats ) )
  33      {    // Set default value:
  34          $allow_evo_stats = true; // allow (non-anonymous) stats
  35      }
  36      if( $allow_evo_stats === false )
  37      { // Get outta here:
  38          return NULL;
  39      }
  40  
  41      if( $debug == 2 )
  42      {
  43          $update_every = 8;
  44          $attempt_every = 3;
  45      }
  46      elseif( $force_short_delay )
  47      {
  48          $update_every = 180; // 3 minutes
  49          $attempt_every = 60; // 1 minute
  50      }
  51      else
  52      {
  53          $update_every = 3600*12; // 12 hours
  54          $attempt_every = 3600*4; // 4 hours
  55      }
  56  
  57      // Note: do not put $baseurl in here since it would cause too frequently updates, when you have the same install with different $baseurls.
  58      //           Everytime this method gets called on another baseurl, there's a new check for updates!
  59      $version_id = $instance_name.' '.$app_name.' '.$app_version.' '.$app_date;
  60      // This is the last version we checked against the server:
  61      $last_version_checked =  $Settings->get( 'evonet_last_version_checked' );
  62  
  63      $servertime_last_update = $Settings->get( 'evonet_last_update' );
  64      $servertime_last_attempt = $Settings->get( 'evonet_last_attempt' );
  65  
  66      if( $last_version_checked == $version_id )
  67      {    // Current version has already been checked, don't check too often:
  68  
  69          if( $servertime_last_update > $servertimenow - $update_every )
  70          {    // The previous update was less than 12 hours ago, skip this
  71              // echo 'recent update';
  72              return false;
  73          }
  74  
  75          if( $servertime_last_attempt > $servertimenow - $attempt_every)
  76          {    // The previous update attempt was less than 4 hours ago, skip this
  77              // This is so all b2evo's don't go crazy if the server ever is down
  78              // echo 'recent attempt';
  79              return false;
  80          }
  81      }
  82  
  83      $Timer->resume('evonet: check for updates');
  84      $Debuglog->add( sprintf('Getting updates from %s.', $evonetsrv_host), 'evonet' );
  85      if( $debug )
  86      {
  87          $Messages->add( sprintf(T_('Getting updates from %s.'), $evonetsrv_host), 'note' );
  88      }
  89      $Settings->set( 'evonet_last_attempt', $servertimenow );
  90      $Settings->dbupdate();
  91  
  92      // Construct XML-RPC client:
  93      load_funcs('xmlrpc/model/_xmlrpc.funcs.php');
  94      $client = new xmlrpc_client( $evonetsrv_uri, $evonetsrv_host, $evonetsrv_port );
  95      if( $debug > 1 )
  96      {
  97          $client->debug = 1;
  98      }
  99  
 100      // Set proxy for outgoing connections:
 101      if( !empty($outgoing_proxy_hostname) )
 102      {
 103          $client->setProxy( $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password );
 104      }
 105  
 106      // Run system checks:
 107      load_funcs( 'tools/model/_system.funcs.php' );
 108  
 109      // Get system stats to display:
 110      $system_stats = get_system_stats();
 111  
 112      // Construct XML-RPC message:
 113      $message = new xmlrpcmsg(
 114                                  'b2evo.getupdates',                           // Function to be called
 115                                  array(
 116                                      new xmlrpcval( ( $allow_evo_stats === 'anonymous' ? md5( $baseurl ) : $baseurl ), 'string'),    // Unique identifier part 1
 117                                      new xmlrpcval( $instance_name, 'string'),        // Unique identifier part 2
 118                                      new xmlrpcval( $app_name, 'string'),            // Version number
 119                                      new xmlrpcval( $app_version, 'string'),          // Version number
 120                                      new xmlrpcval( $app_date, 'string'),            // Version number
 121                                      new xmlrpcval( array(
 122                                              'this_update' => new xmlrpcval( $servertimenow, 'string' ),
 123                                              'last_update' => new xmlrpcval( $servertime_last_update, 'string' ),
 124                                              'mediadir_status' => new xmlrpcval( $system_stats['mediadir_status'], 'int' ), // If error, then the host is potentially borked
 125                                              'install_removed' => new xmlrpcval( ($system_stats['install_removed'] == 'ok') ? 1 : 0, 'int' ), // How many people do go through this extra measure?
 126                                              'evo_charset' => new xmlrpcval( $system_stats['evo_charset'], 'string' ),            // Do people actually use UTF8?
 127                                              'evo_blog_count' => new xmlrpcval( $system_stats['evo_blog_count'], 'int'),   // How many users do use multiblogging?
 128                                              'cachedir_status' => new xmlrpcval( $system_stats['cachedir_status'], 'int'),
 129                                              'cachedir_size' => new xmlrpcval( $system_stats['cachedir_size'], 'int'),
 130                                              'general_pagecache_enabled' => new xmlrpcval( $system_stats['general_pagecache_enabled'] ? 1 : 0, 'int' ),
 131                                              'blog_pagecaches_enabled' => new xmlrpcval( $system_stats['blog_pagecaches_enabled'], 'int' ),
 132                                              'db_version' => new xmlrpcval( $system_stats['db_version'], 'string'),    // If a version >95% we make it the new default.
 133                                              'db_utf8' => new xmlrpcval( $system_stats['db_utf8'] ? 1 : 0, 'int' ),    // if support >95%, we'll make it the default
 134                                              // How many "low security" hosts still active?; we'd like to standardize security best practices... on suphp?
 135                                              'php_uid' => new xmlrpcval( $system_stats['php_uid'], 'int' ),
 136                                              'php_uname' => new xmlrpcval( $system_stats['php_uname'], 'string' ),    // Potential unsecure hosts will use names like 'nobody', 'www-data'
 137                                              'php_gid' => new xmlrpcval( $system_stats['php_gid'], 'int' ),
 138                                              'php_gname' => new xmlrpcval( $system_stats['php_gname'], 'string' ),    // Potential unsecure hosts will use names like 'nobody', 'www-data'
 139                                              'php_version' => new xmlrpcval( $system_stats['php_version'], 'string' ),            // Target minimum version: PHP 5.2
 140                                              'php_reg_globals' => new xmlrpcval( $system_stats['php_reg_globals'] ? 1 : 0, 'int' ), // if <5% we may actually refuse to run future version on this
 141                                              'php_allow_url_include' => new xmlrpcval( $system_stats['php_allow_url_include'] ? 1 : 0, 'int' ),
 142                                              'php_allow_url_fopen' => new xmlrpcval( $system_stats['php_allow_url_fopen'] ? 1 : 0, 'int' ),
 143                                              // TODO php_magic quotes
 144                                              'php_upload_max' => new xmlrpcval( $system_stats['php_upload_max'], 'int' ),
 145                                              'php_post_max' => new xmlrpcval( $system_stats['php_post_max'], 'int' ),
 146                                              'php_memory' => new xmlrpcval( $system_stats['php_memory'], 'int'), // how much room does b2evo have to move on a typical server?
 147                                              'php_mbstring' => new xmlrpcval( $system_stats['php_mbstring'] ? 1 : 0, 'int' ),
 148                                              'php_xml' => new xmlrpcval( $system_stats['php_xml'] ? 1 : 0, 'int' ),
 149                                              'php_imap' => new xmlrpcval( $system_stats['php_imap'] ? 1 : 0, 'int' ),    // Does it make sense to rely on IMAP to handle undelivered emails (for user registrations/antispam)
 150                                              'php_opcode_cache' => new xmlrpcval( $system_stats['php_opcode_cache'], 'string' ), // How many use one? Which is the most popular?
 151                                              'gd_version' => new xmlrpcval( $system_stats['gd_version'], 'string' ),
 152                                              // TODO: add missing system stats
 153                                          ), 'struct' ),
 154                                  )
 155                              );
 156  
 157      $result = $client->send($message);
 158  
 159      if( $ret = xmlrpc_logresult( $result, $Messages, false ) )
 160      { // Response is not an error, let's process it:
 161          $response = $result->value();
 162          if( $response->kindOf() == 'struct' )
 163          { // Decode struct:
 164              $response = xmlrpc_decode_recurse($response);
 165  
 166              /**
 167               * @var AbstractSettings
 168               */
 169              global $global_Cache;
 170  
 171              foreach( $response as $key=>$data )
 172              {
 173                  $global_Cache->set( $key, serialize($data) );
 174              }
 175  
 176              $global_Cache->delete( 'evonet_updates' );    // Cleanup
 177  
 178              $global_Cache->dbupdate();
 179  
 180              $Settings->set( 'evonet_last_update', $servertimenow );
 181              $Settings->set( 'evonet_last_version_checked', $version_id );
 182              $Settings->dbupdate();
 183  
 184              $Debuglog->add( 'Updates saved', 'evonet' );
 185  
 186              $Timer->pause('evonet: check for updates');
 187              return true;
 188          }
 189          else
 190          {
 191              $Debuglog->add( 'Invalid updates received', 'evonet' );
 192              $Messages->add( T_('Invalid updates received'), 'error' );
 193          }
 194      }
 195  
 196      $Timer->pause('evonet: check for updates');
 197      return false;
 198  }

 207  function get_comments_awaiting_moderation_number( $blog_ID )
 208  {
 209      global $DB;

 212      $Blog = & $BlogCache->get_by_ID( $blog_ID, false, false );

 215  
 216      $sql = 'SELECT COUNT(DISTINCT(comment_ID))
 217                  FROM T_comments
 218                      INNER JOIN T_items__item ON comment_item_ID = post_ID ';

 226      $sql .= ' AND '.statuses_where_clause();
 227

 380   *

 382   * @param string SQL WHERE

 386  function get_table_count( $table_name, $sql_where = '', $sql_from = '' )
 387  {
 388      global $DB;
 389  
 390      $SQL = new SQL();
 391      $SQL->SELECT( 'COUNT( * )' );
 392      $SQL->FROM( $table_name );
 393      if( !empty( $sql_from ) )
 394      { // Additional sql for FROM clause
 395          $SQL->FROM_add( $sql_from );
 396      }
 397      if( !empty( $sql_where ) )
 398      { // Additional sql for WHERE clause
 399          $SQL->WHERE( $sql_where );
 400      }
 401  
 402      return intval( $DB->get_var( $SQL->get() ) );

 520  /**
 521   * Display charts
 522   *
 523   * @param array Chart data
 524   */
 525  function display_charts( $chart_data )
 526  {
 527      if( empty( $chart_data ) )
 528      { // No data
 529          return;
 530      }
 531  
 532      echo '<div class="charts">';
 533      foreach( $chart_data as $chart_item )
 534      {
 535          if( $chart_item['type'] == 'number' )
 536          { // Calculate a percent with log10 where max value is 100000
 537              $chart_percent = empty( $chart_item['value'] ) ? 0 : log10_percent( $chart_item['value'] );
 538              // Set a color for value, from green(0%) to red(100%)
 539              $chart_color = get_color_by_percent( '#61bd4f', '#f2d600', '#ffab4a', $chart_percent );
 540          }
 541          else
 542          { // Calculate a real percent
 543              $chart_percent = empty( $chart_item['100%'] ) ? 0 : floor( intval( $chart_item['value'] ) / $chart_item['100%'] ) * 100;
 544              $chart_item['value'] = $chart_percent.'%';
 545              $chart_color = '#00F';
 546          }
 547          if( $chart_item['value'] > 0 && $chart_percent == 0 )
 548          { // Display a little chart for not null values
 549              $chart_percent = 0.01;

 564   * Convert color #FFFFFF to array( 'R' => 255, 'G' => 255, 'B' => 255 )

 566   * @param string Color in hex format
 567   * @return array Color in RGB format
 568   */
 569  function color_hex2rgb( $color )
 570  {
 571      $color = str_replace( '#', '', $color );
 572      $color = str_split( $color, strlen( $color ) / 3 );
 573      foreach( $color as $c => $hex )
 574      {
 575          $color[ $c ] = strlen( $hex ) == 1 ? $hex.$hex : $hex;
 576      }
 577      return array(
 578              'R' => hexdec( $color[0] ),
 579              'G' => hexdec( $color[1] ),
 580              'B' => hexdec( $color[2] ),

 586   * Get color by percent between three colors
 587   *
 588   * @param string Start Color in hex format
 589   * @param string Middle Color in hex format
 590   * @param string End Color in hex format
 591   * @param integer Percent
 592   * @return string Result Color in hex format
 593   */
 594  function get_color_by_percent( $color_from, $color_middle, $color_to, $percent )
 595  {
 596      if( $percent < 50 )
 597      { // First 50 percents
 598          $color_to = $color_middle;
 599          $percent *= 2;
 600      }
 601      else
 602      { // Last 50 percents
 603          $color_from = $color_middle;
 604          $percent = ( $percent - 50 ) * 2;
 605      }
 606      $color_from = color_hex2rgb( $color_from );
 607      $color_to = color_hex2rgb( $color_to );
 608  
 609      $new_color = '#';
 610      foreach( $color_from as $rgb_key => $rgb_color )
 611      {
 612          $rgb_percent = $color_from[ $rgb_key ] + round( ( $color_to[ $rgb_key ] - $color_from[ $rgb_key ] ) * ( $percent / 100 ) );
 613          $rgb_percent = $rgb_percent > 255 ? 255 : ( $rgb_percent < 0 ? 0 : $rgb_percent );
 614  
 615          $new_color .= str_pad( dechex( $rgb_percent ), 2, 0, STR_PAD_LEFT );
 616      }
