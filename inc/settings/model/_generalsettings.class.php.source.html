 304      /**
 305       * Constructor.
 306       *
 307       * This loads the general settings and checks db_version.
 308       *
 309       * It will also turn off error-reporting/halting of the {@link $DB DB object}
 310       * temporarily to present a more decent error message if tables do not exist yet.
 311       *
 312       * Because the {@link $DB DB object} itself creates a connection when it gets
 313       * created "Error selecting database" occurs before we can check for it here.
 314       */
 315  	function GeneralSettings()
 316      {
 317          global $new_db_version, $DB, $demo_mode, $instance_name, $basehost;
 318  
 319          $save_DB_show_errors = $DB->show_errors;
 320          $save_DB_halt_on_error = $DB->halt_on_error;
 321          $DB->halt_on_error = false;
 322          $DB->show_errors = false;
 323  
 324          // Init through the abstract constructor. This should be the first DB connection.
 325          parent::AbstractSettings( 'T_settings', array( 'set_name' ), 'set_value', 0 );
 326  
 327          // check DB version:
 328          if( $this->get( 'db_version' ) != $new_db_version )
 329          { // Database is not up to date:
 330              if( $DB->last_error )
 331              {
 332                  $error_title = 'The database is not installed yet!';
 333                  $error_message = '<p>MySQL error:</p>'.$DB->last_error;
 334              }
 335              else
 336              {
 337                  $error_title = 'Database schema is not up to date!';
 338                  $error_message = '<p>Database schema is not up to date!</p>'
 339                      .'<p>You have schema version &laquo;'.(integer)$this->get( 'db_version' ).'&raquo;, '
 340                      .'but we would need &laquo;'.(integer)$new_db_version.'&raquo;.</p>';
 341              }
 342              global $adminskins_path;
 343              require $adminskins_path.'conf_error.main.php'; // error & exit
 344          }
 345  
 346          $DB->halt_on_error = $save_DB_halt_on_error;
 347          $DB->show_errors = $save_DB_show_errors;
 348  
 349  
 350          if( $demo_mode )
 351          { // Demo mode requires to allow multiple concurrent sessions:
 352              $this->_defaults['multiple_sessions'] = 'always';
 353          }
 354  
 355          // set those defaults which needs some global variables
 356          $this->_defaults['notification_sender_email'] = $instance_name.'-noreply@'.preg_replace( '/^www\./i', '', $basehost );
 357          $this->_defaults['notification_return_path'] = $instance_name.'-return@'.preg_replace( '/^www\./i', '', $basehost );
 358      }

 361      /**
 362       * Get a 32-byte string that can be used as salt for public keys.
 363       *
 364       * @return string
 365       */
 366  	function get_public_key_salt()
 367      {
 368          $public_key_salt = $this->get( 'public_key_salt' );
 369          if( empty($public_key_salt) )
 370          {
 371              $public_key_salt = generate_random_key(32);
 372              $this->set( 'public_key_salt', $public_key_salt );
 373              $this->dbupdate();
 374          }
 375          return $public_key_salt;
 376      }

 384       * @return mixed Value of parameter
 385       */
 386  	function get( $parname, $real_value = false )
 387      {
 388          if( $real_value )
 389          {
 390              return parent::get( $parname );
 391          }
 392  
 393          switch($parname)
 394          {
 395              case 'allow_avatars':

 397                  break;
 398  
 399              case 'upload_enabled':
 400                  return ( parent::get( $parname ) && isset($GLOBALS['files_Module']) );
 401                  break;
 402  
 403              default:
 404                  return parent::get( $parname );
 405          }
