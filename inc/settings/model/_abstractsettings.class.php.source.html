 141      /**
 142       * Loads the settings. Not meant to be called directly, but gets called
 143       * when needed.
 144       *
 145       * @access protected
 146       * @param string First column key
 147       * @param string Second column key
 148       * @param string Third column key
 149       * @return boolean always true
 150       */
 151  	function _load( $arg1 = NULL, $arg2 = NULL, $arg3 = NULL )
 152      {
 153          if( $this->all_loaded )
 154          { // already all loaded
 155              return true;
 156          }
 157          global $DB;
 158  
 159          /**
 160           * The where clause - gets filled when {@link $cache_by_col_keys} is used.
 161           */
 162          $whereList = array();
 163  
 164          if( $this->cache_by_col_keys && isset($arg1) )
 165          { // The number of column keys to cache by is > 0
 166              $testCache = $this->cache;
 167              $args = array( $arg1, $arg2, $arg3 );
 168  
 169              for( $i = 0; $i < $this->cache_by_col_keys; $i++ )
 170              {
 171                  $whereList[] = $this->col_key_names[$i]." = '".$args[$i]."'";
 172  
 173                  if( ! is_array( $testCache )
 174                          || is_null($args[$i])
 175                          || ! isset( $testCache[$args[$i]] )
 176                          || ! ($testCache = & $testCache[$args[$i]]) )
 177                  {
 178                      break;
 179                  }
 180              }
 181  
 182              if( $i == $this->cache_by_col_keys )
 183              { // already loaded!
 184                  return true;
 185              }
 186          }
 187          else
 188          { // We cache everything at once!
 189              $this->all_loaded = true;
 190          }
 191  
 192  
 193          $result = $DB->get_results( '
 194              SELECT '.implode( ', ', $this->col_key_names ).', '.$this->col_value_name.'
 195              FROM '.$this->db_table_name.(
 196                  isset( $whereList[0] )
 197                  ? ' WHERE '.implode( ' AND ', $whereList )
 198                  : '' ) );
 199  
 200          switch( $this->count_col_key_names )
 201          {
 202              case 1:
 203                  if( ! $result )
 204                  { // Remember that we've tried it
 205                      $this->cache[ $arg1 ] = NULL;
 206                  }
 207                  else foreach( $result as $loop_row )
 208                  {
 209                      $this->cache[$loop_row->{$this->col_key_names[0]}] = new stdClass();
 210  
 211                      $this->cache[$loop_row->{$this->col_key_names[0]}]->value = $loop_row->{$this->col_value_name};
 212                      $this->cache[$loop_row->{$this->col_key_names[0]}]->dbUptodate = true;
 213                      $this->cache[$loop_row->{$this->col_key_names[0]}]->dbRemove = false;
 214                  }
 215                  break;
 216  
 217              case 2:
 218                  if( ! $result )
 219                  { // Remember that we've tried it
 220                      $this->cache[ $arg1 ][ $arg2 ] = NULL;
 221                  }
 222                  else foreach( $result as $loop_row )
 223                  {
 224                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}] = new stdClass();
 225  
 226                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}]->value = $loop_row->{$this->col_value_name};
 227                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}]->dbUptodate = true;
 228                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}]->dbRemove = false;
 229                  }
 230                  break;
 231  
 232              case 3:
 233                  if( ! $result )
 234                  { // Remember that we've tried it
 235                      $this->cache[ $arg1 ][ $arg2 ][ $arg3 ] = NULL;
 236                  }
 237                  else foreach( $result as $loop_row )
 238                  {
 239                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}][$loop_row->{$this->col_key_names[2]}] = new stdClass();
 240  
 241                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}][$loop_row->{$this->col_key_names[2]}]->value = $loop_row->{$this->col_value_name};
 242                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}][$loop_row->{$this->col_key_names[2]}]->dbUptodate = true;
 243                      $this->cache[$loop_row->{$this->col_key_names[0]}][$loop_row->{$this->col_key_names[1]}][$loop_row->{$this->col_key_names[2]}]->dbRemove = false;
 244                  }
 245                  break;
 246          }
 247  
 248          return true;

 252      /**
 253       * Get a setting from the DB settings table.
 254       *
 255       * @uses get_default()
 256       * @param string First column key
 257       * @param string Second column key
 258       * @param string Third column key
 259       * @return string|false|NULL value as string on success; NULL if not found; false in case of error
 260       */
 261  	function get( $col_key1, $col_key2 = NULL, $col_key3 = NULL )
 262      {
 263          global $debug;
 264  
 265          if( $debug )
 266          {
 267              global $Debuglog, $Timer;
 268              $this_class = get_class($this);
 269              $Timer->resume('abstractsettings_'.$this_class.'_get', false );
 270          }
 271  
 272          switch( $this->count_col_key_names )
 273          {
 274              case 1:
 275                  $this->_load( $col_key1 );
 276                  if( !empty($this->cache[ $col_key1 ]->unserialized) )
 277                  {    // The value has been unserialized before:
 278                      $r = $this->cache[ $col_key1 ]->value;
 279                  }
 280                  elseif( isset($this->cache[ $col_key1 ]->value) )
 281                  {    // First attempt to access the value, we need to unserialize it:
 282                      // Try to unserialize setting (once) - this is as fast as checking an array of values that should get unserialized
 283                      if( ($r = @unserialize($this->cache[ $col_key1 ]->value)) !== false )
 284                      {
 285                          $this->cache[ $col_key1 ] = new stdClass();
 286                          $this->cache[ $col_key1 ]->value = $r;
 287                      }
 288                      else
 289                      {
 290                          $r = $this->cache[ $col_key1 ]->value;
 291                      }
 292                      $this->cache[ $col_key1 ]->unserialized = true;
 293                  }
 294                  else
 295                  {    // The value is not in the cache, we use the default:
 296                      $r = $this->get_default( $col_key1 );
 297                      $this->cache[ $col_key1 ] = new stdClass();
 298                      $this->cache[ $col_key1 ]->value = $r; // remember in cache
 299                      $this->cache[ $col_key1 ]->dbUptodate = true;
 300                      $this->cache[ $col_key1 ]->unserialized = true;
 301                      $from_default = true; // for debug
 302                  }
 303                  break;
 304  
 305              case 2:
 306                  $this->_load( $col_key1, $col_key2 );
 307  
 308                  if( isset($this->cache[ $col_key1 ][ $col_key2 ]->unserialized) )
 309                  {
 310                      $r = $this->cache[ $col_key1 ][ $col_key2 ]->value;
 311                  }
 312                  elseif( isset($this->cache[ $col_key1 ][ $col_key2 ]->value) )
 313                  {
 314                      // Try to unserialize setting (once) - this is as fast as checking an array of values that should get unserialized
 315                      if( ($r = @unserialize($this->cache[ $col_key1 ][ $col_key2 ]->value)) !== false )
 316                      {
 317                          $this->cache[ $col_key1 ][ $col_key2 ] = new stdClass();
 318                          $this->cache[ $col_key1 ][ $col_key2 ]->value = $r;
 319                      }
 320                      else
 321                      {
 322                          $r = $this->cache[ $col_key1 ][ $col_key2 ]->value;
 323                      }
 324                      $this->cache[ $col_key1 ][ $col_key2 ]->unserialized = true;
 325                  }
 326                  else
 327                  {
 328                      $r = $this->get_default( $col_key2 );
 329                      $this->cache[ $col_key1 ][ $col_key2 ] = new stdClass();
 330                      $this->cache[ $col_key1 ][ $col_key2 ]->value = $r; // remember in cache
 331                      $this->cache[ $col_key1 ][ $col_key2 ]->dbUptodate = true;
 332                      $this->cache[ $col_key1 ][ $col_key2 ]->unserialized = true;
 333                      $from_default = true; // for debug
 334                  }
 335                  break;
 336  
 337              case 3:
 338                  $this->_load( $col_key1, $col_key2, $col_key3 );
 339  
 340                  if( isset($this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->unserialized) )
 341                  {
 342                      $r = $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value;
 343                  }
 344                  elseif( isset($this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value) )
 345                  {
 346                      // Try to unserialize setting (once) - this is as fast as checking an array of values that should get unserialized
 347                      if( ($r = @unserialize($this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value)) !== false )
 348                      {
 349                          $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ] = new stdClass();
 350                          $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value = $r;
 351                      }
 352                      else
 353                      {
 354                          $r = $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value;
 355                      }
 356                      $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->unserialized = true;
 357                  }
 358                  else
 359                  {
 360                      $r = $this->get_default( $col_key3 );
 361                      $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ] = new stdClass();
 362                      $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->value = $r; // remember in cache
 363                      $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->dbUptodate = true;
 364                      $this->cache[ $col_key1 ][ $col_key2 ][ $col_key3 ]->unserialized = true;
 365                      $from_default = true; // for debug
 366                  }
 367                  break;
 368          }
 369  
 370          if( $debug )
 371          {
 372              $Debuglog->add( $this_class.'::get( '.$col_key1.'/'.$col_key2.'/'.$col_key3.' ): '
 373                  .( isset($from_default) ? '[DEFAULT]: ' : '' )
 374                  .var_export( $r, true ), 'settings' );
 375              $Timer->pause('abstractsettings_'.$this_class.'_get', false );
 376          }
 377  
 378          return $r;
 379      }

 427      /**
 428       * Temporarily sets a setting ({@link dbupdate()} writes it to DB).
 429       *
 430       * @param string $args,... the values for the {@link $col_key_names column keys}
 431       *                         and {@link $col_value_name column value}. Must match order and count!
 432       * @return boolean true, if the value has been set, false if it has not changed.
 433       */
 434  	function set()
 435      {
 436          global $Debuglog;
 437  
 438          $args = func_get_args();
 439          $value = array_pop($args);
 440  
 441          call_user_func_array( array(&$this, '_load'), $args );
 442  
 443          $debugMsg = get_class($this).'::set( '.implode(', ', $args ).' ): ';
 444  
 445          switch( $this->count_col_key_names )
 446          {
 447              case 1:
 448                  $atCache = & $this->cache[ $args[0] ];
 449                  break;
 450  
 451              case 2:
 452                  $atCache = & $this->cache[ $args[0] ][ $args[1] ];
 453                  break;
 454  
 455              case 3:
 456                  $atCache = & $this->cache[ $args[0] ][ $args[1] ][ $args[2] ];
 457                  break;
 458  
 459              default:
 460                  return false;
 461          }
 462  
 463          if( ! is_object($atCache) )
 464          {    // PHP 5.4 fix for "Warning: Creating default object from empty value"
 465              $atCache = new stdClass();
 466          }
 467  
 468          $atCache->dbRemove = false;
 469  
 470          if( isset($atCache->value) )
 471          {
 472              if( ( is_array( $value ) && serialize(  $value ) === $atCache->value ) ||
 473                  ( ! is_array( $value ) && $atCache->value === (string)$value ) )
 474              { // already set
 475                  $Debuglog->add( $debugMsg.' Already set to the same value.', 'settings' );
 476                  return false;
 477              }
 478          }
 479  
 480          $atCache->value = $value;
 481          $atCache->dbUptodate = false;
 482          $atCache->unserialized = false; // We haven't tried to unserialize the value yet
 483  
 484          $Debuglog->add( $debugMsg.' SET!', 'settings' );
 485  
 486          return true;
 487      }

 575      /**
 576       * Commit changed settings to DB.
 577       *
 578       * @return boolean true, if settings have been updated; false otherwise
 579       */
 580  	function dbupdate()
 581      {
 582          if( empty($this->cache) )
 583          {
 584              return false;
 585          }
 586  
 587          global $DB;
 588  
 589          $query_insert = array();
 590          $query_where_delete = array();
 591  
 592          switch( $this->count_col_key_names )
 593          {
 594              case 1:
 595                  foreach( $this->cache as $key => $value )
 596                  {
 597                      if( $value === NULL )
 598                      { // Remembered as not existing
 599                          continue;
 600                      }
 601                      if( ! empty($value->dbRemove) )
 602                      {
 603                          $query_where_delete[] = "{$this->col_key_names[0]} = '$key'";
 604                          unset( $this->cache[$key] );
 605                      }
 606                      elseif( isset($value->dbUptodate) && !$value->dbUptodate )
 607                      {
 608                          $value = $value->value;
 609                          if( is_array( $value ) || is_object( $value ) )
 610                          {
 611                              $value = serialize($value);
 612                          }
 613                          $query_insert[] = "('$key', '".$DB->escape( $value )."')";
 614                          $this->cache[$key]->dbUptodate = true;
 615                      }
 616                  }
 617                  break;
 618  
 619              case 2:
 620                  foreach( $this->cache as $key => $value )
 621                  {
 622                      foreach( $value as $key2 => $value2 )
 623                      {
 624                          if( $value2 === NULL )
 625                          { // Remembered as not existing
 626                              continue;
 627                          }
 628                          if( ! empty($value2->dbRemove) )
 629                          {
 630                              $query_where_delete[] = "{$this->col_key_names[0]} = '$key' AND {$this->col_key_names[1]} = '$key2'";
 631                              unset( $this->cache[$key][$key2] );
 632                          }
 633                          elseif( isset($value2->dbUptodate) && !$value2->dbUptodate )
 634                          {
 635                              $value2 = $value2->value;
 636                              if( is_array( $value2 ) || is_object( $value2 ) )
 637                              {
 638                                  $value2 = serialize($value2);
 639                              }
 640                              $query_insert[] = "('$key', '$key2', '".$DB->escape( $value2 )."')";
 641                              $this->cache[$key][$key2]->dbUptodate = true;
 642                          }
 643                      }
 644                  }
 645                  break;
 646  
 647              case 3:
 648                  foreach( $this->cache as $key => $value )
 649                  {
 650                      foreach( $value as $key2 => $value2 )
 651                      {
 652                          foreach( $value2 as $key3 => $value3 )
 653                          {
 654                              if( $value3 === NULL )

 718       *
 719       * If the setting was not set before (and there's no default given that gets returned), $default gets used.
 720       *
 721       * @param string Request param name
 722       * @param string setting name. Make sure this is unique!
 723       * @param string Force value type to one of:
 724       * - integer
 725       * - float
 726       * - string (strips (HTML-)Tags, trims whitespace)
 727       * - array
 728       * - object
 729       * - null
 730       * - html (does nothing)
 731       * - '' (does nothing)
 732       * - '/^...$/' check regexp pattern match (string)
 733       * - boolean (will force type to boolean, but you can't use 'true' as a default since it has special meaning. There is no real reason to pass booleans on a URL though. Passing 0 and 1 as integers seems to be best practice).
 734       * Value type will be forced only if resulting value (probably from default then) is !== NULL
 735       * @param mixed Default value or TRUE
 736       * @param boolean Do we need to memorize this to regenerate the URL for this page?
 737       * @param boolean Override if variable already set
 738       * @return NULL|mixed NULL, if neither a param was given nor knows about it.
 739       */
 740  	function param_Request( $param_name, $set_name, $type = '', $default = '', $memorize = false, $override = false ) // we do not force setting it..
 741      {
 742          $value = param( $param_name, $type, NULL, $memorize, $override, false ); // we pass NULL here, to see if it got set at all
 743  
 744          if( $value !== false )
 745          { // we got a value
 746              $this->set( $set_name, $value );
 747              $this->dbupdate();
 748          }
 749          else
 750          { // get the value from user settings
 751              $value = $this->get($set_name);
 752  
 753              if( is_null($value) )
 754              { // it's not saved yet and there's not default defined ($_defaults)
 755                  $value = $default;
 756              }
 757          }
 758  
 759          set_param( $param_name, $value );
 760          return get_param($param_name);
