  18  /**
  19   * Get regional option list from given array
  20   * 
  21   * @param array rows
  22   * @param integer ID of selected row
  23   * @param array field params
  24   * @return string html tags <option>
  25   */
  26  function get_regional_option_list( $rows, $selected = 0, $params = array() )
  27  {
  28      $params = array_merge( array(
  29              'allow_none' => true,
  30              'none_option_value' => '',
  31              'none_option_text' => T_('All'),
  32              'group_preferred' => true,
  33              'group_name_preferred' => T_('Frequent'),
  34              'group_name_other' => T_('Other'),
  35          ), $params );
  36  
  37      $r = '';
  38  
  39      if( $params['allow_none'] )
  40      {    // we set current value of a country if it is sent to function
  41          $r .= '<option value="'.$params['none_option_value'].'"';
  42          if( empty($default) ) $r .= ' selected="selected"';
  43          $r .= '>'.format_to_output( $params['none_option_text'] ).'</option>'."\n";
  44      }
  45  
  46      $pref_rows = array(); //preferred countries.
  47      $rest_rows = array(); // not preffered countries (the rest)
  48  
  49      foreach( $rows as $row_Obj )
  50      {
  51          if( $params['group_preferred'] && $row_Obj->preferred == 1 )
  52          {    // if the country is preferred we add it to selected array.
  53              $pref_rows[] = $row_Obj;
  54          }
  55          $rest_rows[] = $row_Obj;
  56      }
  57  
  58      if( count( $pref_rows ) )
  59      {    // if we don't have preferred countries in this case we don't have to show optgroup
  60          // in option list
  61          $r .= '<optgroup label="'.$params['group_name_preferred'].'">';
  62          foreach( $pref_rows as $row_Obj )
  63          {
  64              $r .=  '<option value="'.$row_Obj->ID.'"';
  65              if( $row_Obj->ID == $selected ) $r .= ' selected="selected"';
  66              $r .= '>';
  67              $r .= format_to_output( $row_Obj->name, 'htmlbody' );
  68              $r .=  '</option>'."\n";
  69          }
  70          $r .= '</optgroup>';
  71  
  72          if( count( $rest_rows ) )
  73          { // if we don't have rest countries we do not create optgroup for them
  74              $r .= '<optgroup label="'.$params['group_name_other'].'">';
  75              foreach( $rest_rows as $row_Obj )
  76              {
  77                  $r .=  '<option value="'.$row_Obj->ID.'"';
  78                  if( $row_Obj->ID == $selected ) $r .= ' selected="selected"';
  79                  $r .= '>';
  80                  $r .= format_to_output( $row_Obj->name, 'htmlbody' );
  81                  $r .=  '</option>'."\n";
  82              }
  83              $r .= '</optgroup>';
  84          }
  85      }
  86      else
  87      {    // if we have only rest countries we get here
  88          foreach( $rest_rows as $row_Obj )
  89          {
  90              $r .=  '<option value="'.$row_Obj->ID.'"';
  91              if( $row_Obj->ID == $selected ) $r .= ' selected="selected"';
  92              $r .= '>';
  93              $r .= format_to_output( $row_Obj->name, 'htmlbody' );
  94              $r .=  '</option>'."\n";
  95          }
  96      }
  97  
  98      return $r;
  99  }

 165  function get_cities_option_list( $country_ID, $region_ID = 0, $subregion_ID = 0, $city_ID = 0, $params = array() )
 166  {
 167      $params = array_merge( array(
 168              'group_name_preferred' => T_('Frequent cities'),
 169              'group_name_other' => T_('Other cities'),
 170          ), $params );
 171  
 172      global $DB;
 173  
 174      $get_cities = true;
 175      $sql_where = array();
 176  
 177      // Get city by country
 178      $sql_where[] = 'city_ctry_ID = "'.$DB->escape( $country_ID ).'"';
 179  
 180      if( $region_ID > 0 )
 181      {    // Get cities by region
 182          $sql_where[] = 'city_rgn_ID = "'.$DB->escape( $region_ID ).'"';
 183      }
 184      else
 185      {    // Select all cities from current country
 186          $regions_count = $DB->get_var( '
 187              SELECT COUNT( rgn_ID )
 188                FROM T_regional__region
 189               WHERE rgn_ctry_ID = "'.$DB->escape( $country_ID ).'"
 190                 AND rgn_enabled = 1' );
 191          if( $regions_count == 0 )
 192          {    // If regions don't exist for current country we show all cities of the country
 193              $sql_where[] = 'city_rgn_ID IS NULL';
 194          }
 195          else
 196          {    // Don't get cities without selected region
 197              $get_cities = false;
 198          }
 199      }
 200  
 201      if( $region_ID > 0 )
 202      {    // Select sub-regions only when region_ID is defined
 203          if( $subregion_ID > 0 )
 204          {    // Get cities by sub-region
 205              $sql_where[] = 'city_subrg_ID = "'.$DB->escape( $subregion_ID ).'"';
 206          }
 207          else
 208          {    // Select all cities from current region
 209              $subregions_count = $DB->get_var( '
 210                  SELECT COUNT( subrg_ID )
 211                    FROM T_regional__subregion
 212                   WHERE subrg_rgn_ID = "'.$DB->escape( $region_ID ).'"
 213                   AND subrg_enabled = 1' );
 214              if( $subregions_count == 0 )
 215              {    // If sub-regions don't exist for current region we show all cities of the region
 216                  $sql_where[] = 'city_subrg_ID IS NULL';
 217              }
 218              else
 219              {    // Don't get cities without selected sub-region
 220                  $get_cities = false;
 221              }
 222          }
 223      }
 224  
 225      if( $get_cities )
 226      {    // Get cities from DB
 227          $sql_where[] = 'city_enabled = 1';
 228          $cities = $DB->get_results( '
 229              SELECT city_ID as ID, CONCAT( city_name, " (", city_postcode, ")" ) as name, city_preferred as preferred
 230                FROM T_regional__city
 231               WHERE '.implode( ' AND ', $sql_where ).'
 232               ORDER BY city_name' );
 233      }
 234      else
 235      {
 236          $cities = array();
 237      }
 238  
 239      return get_regional_option_list( $cities, $city_ID, $params );
 240  }

 243  /**
 244   * Import cities from CSV file
 245   * 
 246   * @param integer country ID
 247   * @param string 
 248   * @return array (
 249   *   'inserted' => Count of inserted cities,
 250   *   'updated'  => Count of updated cities );
 251   */
 252  function import_cities( $country_ID, $file_name )
 253  {
 254      global $DB;
 255  
 256      // Begin transaction
 257      $DB->begin();
 258  
 259      // Get all sub-regions of the current country
 260      $subregions_data = $DB->get_results( '
 261          SELECT subrg_ID, subrg_code, rgn_ID, rgn_ctry_ID
 262            FROM T_regional__subregion
 263            LEFT JOIN T_regional__region ON subrg_rgn_ID = rgn_ID
 264           WHERE rgn_ctry_ID = '.$DB->quote( $country_ID ) );
 265  
 266      $subregions = array();
 267      foreach( $subregions_data as $subregion )
 268      {
 269          $subregions[$subregion->subrg_code] = $subregion;
 270      }
 271      unset( $subregions_data );
 272  
 273  
 274      // Open file
 275      $file_handle = fopen( $file_name, 'r' );
 276  
 277      $c = 0;
 278      $cities_insert_values = array();
 279      $cities_update_values = array();
 280      while( $data = fgetcsv( $file_handle, 1024, ";" ) )
 281      {
 282          if( $c == 0 )
 283          {    // Skip first row with titles
 284              $c++;
 285              continue;
 286          }
 287  
 288          $postcode = trim( $data[0], " \xA0" ); // \xA0 - ASCII Non-breaking space
 289          $name = trim( $data[1], " \xA0" );
 290          $subregion_code = '';
 291          if( isset( $data[2] ) )
 292          {    // Optional field
 293              $subregion_code = trim( $data[2], " \xA0" );
 294          }
 295  
 296          if( empty( $postcode ) && empty( $name ) )
 297          {    // Skip empty row
 298              continue;
 299          }
 300  
 301          /*if( empty( $subregion_code ) )
 302          {    // If field subregion_code is NOT defined, we get it from city postcode ( 2 first letters )
 303              $subregion_code = substr( $postcode, 0, 2 );
 304          }*/
 305  
 306          $city = array(
 307              'ctry_ID'  => $country_ID,
 308              'postcode' => $DB->quote( $postcode ),
 309              'name'     => $DB->quote( $name ),
 310          );
 311  
 312          if( empty( $subregion_code ) || ! isset( $subregions[$subregion_code] ) )
 313          {    // Subregion is not defined and not found in DB
 314              $city['rgn_ID'] = 'NULL';
 315              $city['subrg_ID'] = 'NULL';
 316          }
 317          else
 318          {    // Set region ID & subregion ID for current city
 319              $city['rgn_ID'] = $subregions[$subregion_code]->rgn_ID;
 320              $city['subrg_ID'] = $subregions[$subregion_code]->subrg_ID;
 321          }
 322  
 323          // Get city from DB with current country, postcode & name
 324          $existing_city = $DB->get_row( '
 325              SELECT city_ID, city_rgn_ID, city_subrg_ID
 326                FROM T_regional__city
 327               WHERE city_ctry_ID = '.$city['ctry_ID'].'
 328                 AND city_postcode = '.$city['postcode'].'
 329                 AND city_name = '.$city['name'] );
 330  
 331          if( !empty( $existing_city ) )
 332          {    // City already exist with such country, postcode & name
 333              if( $city['subrg_ID'] != $existing_city->city_subrg_ID )
 334              {    // Set new sub-region & region for current city
 335                  $cities_update_values[] = 'city_rgn_ID = '.$city['rgn_ID'].', city_subrg_ID = '.$city['subrg_ID'].'
 336                      WHERE city_ID = '.$existing_city->city_ID;
 337              }
 338          }
 339          else
 340          {    // Insert a new city
 341              $cities_insert_values[] = '( '.implode( ', ', $city ).' )';
 342          }
 343  
 344          $c++;
 345      }
 346  
 347      // Close file pointer
 348      fclose( $file_handle );
 349  
 350      $count_insert_cities = count( $cities_insert_values );
 351      $count_update_cities = count( $cities_update_values );
 352  
 353      if( $count_insert_cities > 0 )
 354      {    // New cities are exist to import
 355          // Split an insert data to avoid big sql queries
 356          $cities_insert_values = array_chunk( $cities_insert_values, 1000 );
 357  
 358          foreach( $cities_insert_values as $insert_values )
 359          {
 360              $insert_values = implode( ', ', $insert_values );
 361  
 362              // Insert new cities into DB
 363              $DB->query( '
 364                  INSERT INTO T_regional__city
 365                  ( city_ctry_ID, city_postcode, city_name, city_rgn_ID, city_subrg_ID )
 366                  VALUES '.$insert_values );
 367          }
 368      }
 369  
 370      if( $count_update_cities > 0 )
 371      {    // Cities to update
 372          foreach( $cities_update_values as $update_sql )
 373          {    // Update a existing city
 374              $DB->query( '
 375                  UPDATE T_regional__city
 376                  SET '.$update_sql );
 377          }
 378      }
 379  
 380      // Commit transaction
 381      $DB->commit();
 382  
 383      return array(
 384          'inserted' => $count_insert_cities,
 385          'updated'  => $count_update_cities );
 386  }

 626   * @param boolean TRUE - don't count cities from all region (when $region_ID = 0)
 627   * @return boolean TRUE - cities exist
 628   */
 629  function cities_exist( $country_ID, $region_ID, $subregion_ID, $exact_existance = false )
 630  {
 631      global $DB;
 632  
 633      $sql_where = array();
 634      $sql_where[] = 'city_enabled = 1';
 635  
 636      if( $country_ID > 0 )
 637      {    // Restrict cities with selected country
 638          $sql_where[] = 'city_ctry_ID = '.$DB->quote( $country_ID );
 639      }
 640      else if( $exact_existance )
 641      {    // Don't count cities with not selected country
 642          return false;
 643      }
 644  
 645      if( $region_ID > 0 )
 646      {    // Restrict cities with selected region
 647          $sql_where[] = 'city_rgn_ID = '.$DB->quote( $region_ID );
 648      }
 649      else if( $exact_existance )
 650      {    // Restrict cities with not defined region
 651          $sql_where[] = 'city_rgn_ID IS NULL';
 652      }
 653  
 654      if( $subregion_ID > 0 )
 655      {    // Restrict cities with selected subregion
 656          $sql_where[] = 'city_subrg_ID = '.$DB->quote( $subregion_ID );
 657      }
 658      else if( $exact_existance )
 659      {    // Restrict cities with not defined subregion
 660          $sql_where[] = 'city_subrg_ID IS NULL';
 661      }
 662  
 663      $cities_count = $DB->get_var( '
 664          SELECT COUNT(city_ID)
 665            FROM T_regional__city
 666           WHERE '.implode( ' AND ', $sql_where ) );
 667  
 668      return $cities_count > 0;
