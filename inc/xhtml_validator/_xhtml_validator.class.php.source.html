 161       * check(-)

 163  	function check($xhtml)
 164      {
 165          // Convert encoding:
 166          // TODO: use convert_encoding()
 167          if( empty($this->xml_parser_encoding) || $this->encoding != $this->xml_parser_encoding )
 168          { // we need to convert encoding:
 169              if( function_exists( 'mb_convert_encoding' ) )
 170              { // we can convert encoding to UTF-8
 171                  $this->encoding = 'UTF-8';
 172  
 173                  // Convert XHTML:
 174                  $xhtml = mb_convert_encoding( $xhtml, 'UTF-8' );
 175              }
 176              elseif( ($this->encoding == 'ISO-8859-1' || empty($this->encoding)) && function_exists('utf8_encode') )
 177              {
 178                  $this->encoding = 'UTF-8';
 179  
 180                  $xhtml = utf8_encode( $xhtml );
 181              }
 182          }
 183  
 184          // Open comments or '<![CDATA[' are dangerous
 185          $xhtml = str_replace('<!', '', $xhtml);
 186  
 187          // Convert isolated & chars
 188          $xhtml = preg_replace( '#(\s)&(\s)#', '\\1&amp;\\2', $xhtml );
 189  
 190          $xhtml_head = '<?xml version="1.0"';
 191          if( ! empty($this->encoding) )
 192          {
 193              $xhtml_head .= ' encoding="'.$this->encoding.'"';
 194          }
 195  
 196          $xhtml_head .= '?><!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd"';
 197  
 198          // Include entities:
 199          $xhtml_head .= '[';
 200          // Include latin1 entities (http://www.w3.org/TR/xhtml1/DTD/xhtml-lat1.ent):
 201          $xhtml_head .= file_get_contents( dirname(__FILE__).'/_xhtml-lat1.ent' );
 202          // Include symbol entities (http://www.w3.org/TR/xhtml1/DTD/xhtml-symbol.ent):
 203          $xhtml_head .= file_get_contents( dirname(__FILE__).'/_xhtml-symbol.ent' );
 204          // Include special entities (http://www.w3.org/TR/xhtml1/DTD/xhtml-special.ent):
 205          $xhtml_head .= file_get_contents( dirname(__FILE__).'/_xhtml-special.ent' );
 206          $xhtml_head .= ']>';
 207  
 208          $xhtml = $xhtml_head.'<body>'.$xhtml.'</body>';
 209          unset($xhtml_head);
 210  
 211          if( !xml_parse($this->parser, $xhtml) )
 212          {
 213              $xml_error_code = xml_get_error_code( $this->parser );
 214              $xml_error_string = xml_error_string( $xml_error_code );
 215              switch( $xml_error_code )
 216              {
 217                  case XML_ERROR_TAG_MISMATCH:
 218                      $xml_error_string .= ': <code>'.$this->stack[count($this->stack)-1].'</code>';
 219                      break;
 220              }
 221              $pos = xml_get_current_byte_index($this->parser);
 222              $xml_error_string .= ' near <code>'.htmlspecialchars( utf8_substr( $xhtml, $this->last_checked_pos, $pos-$this->last_checked_pos+20 ) ).'</code>';
 223  
 224              $this->html_error( T_('Parser error: ').$xml_error_string );
 225          }
 226  
 227          return $this->isOK();
 228      }

 232       * tag_open(-)
 233       *
 234       * Called when the parser finds an opening tag

 236  	function tag_open($parser, $tag, $attrs)
 237      {
 238          global $debug;
 239  
 240          // echo "processing tag: $tag <br />\n";
 241          $this->last_checked_pos = xml_get_current_byte_index($this->parser);
 242  
 243          if ($tag == 'body')
 244          {
 245              if( count($this->stack) > 0 )
 246                  $this->html_error( T_('Tag <code>body</code> can only be used once!') );
 247              $this->stack[] = $tag;
 248              return;
 249          }
 250          $previous = $this->stack[count($this->stack)-1];
 251  
 252          // If previous tag is illegal, no point in running tests
 253          if (!in_array($previous, array_keys($this->tags))) {
 254              $this->stack[] = $tag;
 255              return;
 256          }
 257          // Is tag a legal tag?
 258          if (!in_array($tag, array_keys($this->tags))) {
 259              $this->html_error( T_('Illegal tag'). ": <code>$tag</code>" );
 260              $this->stack[] = $tag;
 261              return;
 262          }
 263          // Is tag allowed in the current context?
 264          if (!in_array($tag, explode(' ', $this->tags[$previous]))) {
 265              if ($previous == 'body') {
 266                  $this->html_error(    sprintf( T_('Tag &lt;%s&gt; must occur inside another tag'), '<code>'.$tag.'</code>' ) );
 267              } else {
 268                  $this->html_error(    sprintf( T_('Tag &lt;%s&gt; is not allowed within tag &lt;%s&gt;'), '<code>'.$tag.'</code>', '<code>'.$previous.'</code>') );
 269              }
 270          }
 271          // Are tag attributes valid?
 272          foreach( $attrs as $attr => $value )
 273          {
 274              if (!isset($this->tagattrs[$tag]) || !in_array($attr, explode(' ', $this->tagattrs[$tag])))
 275              {
 276                  $this->html_error( sprintf( T_('Tag &lt;%s&gt; may not have attribute %s="..."'), '<code>'.$tag.'</code>', '<code>'.$attr.'</code>' ) );
 277              }
 278  
 279              if (in_array($attr, $this->uri_attrs))
 280              { // This attribute must be checked for URIs
 281                  $matches = array();
 282                  $value = trim($value);
 283                  if( $error = validate_url( $value, $this->context, false ) ) //Note: We do not check for spam here, should be done on whole message in check_html_sanity()
 284                  {
 285                      $this->html_error( T_('Found invalid URL: ').$error );
 286                  }
 287              }
 288  
 289          }
 290          // Set previous, used for checking nesting context rules
 291          $this->stack[] = $tag;
 292      }

 298      {
 299          $this->last_checked_pos = xml_get_current_byte_index($this->parser);
 300  
 301          // Simply check that the 'previous' tag allows CDATA
 302          $previous = $this->stack[count($this->stack)-1];
 303          // If previous tag is illegal, no point in running test
 304          if (!in_array($previous, array_keys($this->tags))) {
 305              return;
 306          }

 308              if (!in_array('#PCDATA', explode(' ', $this->tags[$previous]))) {
 309                  $this->html_error(    sprintf( T_('Tag &lt;%s&gt; may not contain raw character data'), '<code>'.$previous.'</code>' ) );
 310              }
 311          }
