 148  	function get_default_group_permissions( $grp_ID )
 149      {
 150          switch( $grp_ID )
 151          {
 152              case 1: // Administrators group ID equals 1
 153                  global $test_install_all_features;
 154                  $perm_messaging = $test_install_all_features ? 'abuse' : 'delete';
 155                  $max_new_threads = ''; // empty = no limit
 156                  break;
 157              case 2: // Moderators group equals 2
 158              case 3: // Editors group ID equals 3
 159                  $perm_messaging = 'write'; // Messaging permissions
 160                  $max_new_threads = '10'; // Maximum number of new threads per day
 161                  break;
 162              case 4: // Normal users group ID equals 4
 163                  $perm_messaging = 'write';
 164                  $max_new_threads = '5';
 165                  break;
 166              case 5:        // Misbehaving/Suspect users (group ID 5) have permission by default:
 167                  $perm_messaging = 'write';
 168                  $max_new_threads = '1';
 169                  break;
 170              case 6:  // Spammers/restricted Users
 171              default: // Other groups
 172                  $perm_messaging = 'reply';
 173                  $max_new_threads = '1';
 174                  break;
 175          }
 176  
 177          // We can return as many default permissions as we want:
 178          // e.g. array ( permission_name => permission_value, ... , ... )
 179          return $permissions = array( 'perm_messaging' => $perm_messaging, 'max_new_threads' => $max_new_threads );
 180      }

 249      /**
 250       * Check a permission for the group. ( see 'group_func' in get_available_group_permissions() function )
 251       *
 252       * @param string Requested permission level
 253       * @param string Permission value
 254       * @param mixed Permission target (blog ID, array of cat IDs...)
 255       * @return boolean True on success (permission is granted), false if permission is not granted
 256       */
 257  	function check_messaging_group_perm( $permlevel, $permvalue, $permtarget )
 258      {
 259          $perm = false;
 260          switch ( $permvalue )
 261          {
 262              case 'abuse':
 263                  // Abuse Management & Read, Send & Delete any messages (able to send messages even for blocked contacts)
 264                  if( $permlevel == 'abuse' )
 265                  {
 266                      $perm = true;
 267                      break;
 268                  }
 269              case 'delete':
 270                  // Read, Send & Delete any messages (able to send messages even for blocked contacts)
 271                  if( $permlevel == 'delete' )
 272                  { // User can ask for delete perm...
 273                      $perm = true;
 274                      break;
 275                  }
 276              case 'write':
 277                  // Read & Send messages to anyone (except for blocked contacts)
 278                  if( $permlevel == 'write' )
 279                  {
 280                      $perm = true;
 281                      break;
 282                  }
 283              case 'reply':
 284                  //  Read & Send messages to people in contacts list only (except for blocked contacts)
 285                  if( $permlevel == 'reply' )
 286                  {
 287                      $perm = true;
 288                      break;
 289                  }
 290          }
 291  
 292          return $perm;
 293      }

 436      /**
 437       * Handle messaging module htsrv actions
 438       */
 439  	function handle_htsrv_action()
 440      {
 441          global $current_User, $Blog, $Session, $Messages, $samedomain_htsrv_url;
 442  
 443          // Init objects we want to work on.
 444          $action = param_action( true, true );
 445          $disp = param( 'disp', '/^[a-z0-9\-_]+$/', 'threads' );
 446  
 447          // Check that this action request is not a CSRF hacked request:
 448          $Session->assert_received_crumb( 'messaging_'.$disp );
 449  
 450          // Load classes
 451          load_class( 'messaging/model/_thread.class.php', 'Thread' );
 452          load_class( 'messaging/model/_message.class.php', 'Message' );
 453  
 454          if( !is_logged_in() )
 455          { // user must be logged in
 456              debug_die( 'User must be logged in to proceed with messaging updates!' );
 457          }
 458  
 459          // Check permission:
 460          $current_User->check_perm( 'perm_messaging', 'reply', true );
 461  
 462          // set where to redirect
 463          $redirect_to = param( 'redirect_to', 'url', NULL );
 464          if( empty( $redirect_to ) )
 465          {
 466              if( isset( $Blog ) )
 467              {
 468                  $redirect_to = url_add_param( $Blog->gen_baseurl(), 'disp='.$disp );
 469              }
 470              else
 471              {
 472                  $redirect_to = url_add_param( $baseurl, 'disp='.$disp );
 473              }
 474          }
 475  
 476          if( ( $disp != 'contacts' ) && ( $thrd_ID = param( 'thrd_ID', 'integer', '', true ) ) )
 477          { // Load thread from cache:
 478              $ThreadCache = & get_ThreadCache();
 479              if( ( $edited_Thread = & $ThreadCache->get_by_ID( $thrd_ID, false ) ) === false )
 480              { // Thread doesn't exists with this ID
 481                  unset( $edited_Thread );
 482                  forget_param( 'thrd_ID' );
 483                  $Messages->add( T_('The requested thread does not exist any longer.'), 'error' );
 484                  $action = 'nil';
 485              }
 486          }
 487  
 488          switch( $disp )
 489          {
 490              // threads action
 491              case 'threads':
 492                  if( $action != 'create' && $action != 'preview' )
 493                  { // Make sure we got a thrd_ID:
 494                      param( 'thrd_ID', 'integer', true );
 495                  }
 496  
 497                  switch( $action )
 498                  {
 499                      case 'create': // create thread
 500                      case 'preview': // preview message
 501                          // Stop a request from the blocked IP addresses or Domains
 502                          antispam_block_request();
 503  
 504                          // check if create new thread is allowed
 505                          if( check_create_thread_limit() )
 506                          { // max new threads limit reached, don't allow to create new thread
 507                              debug_die( 'Invalid request, new conversation limit already reached!' );
 508                          }
 509  
 510                          $creating_success = create_new_thread();
 511                          if( !$creating_success || $action == 'preview' )
 512                          { // unsuccessful new thread creation OR preview mode
 513                              global $edited_Thread, $edited_Message, $thrd_recipients, $thrd_recipients_array;
 514  
 515                              $redirect_to .= '&action=new';
 516                              // save new message and thread params into the Session to not lose the content
 517                              $unsaved_message_params = array();
 518                              $unsaved_message_params[ 'action' ] = $action;
 519                              $unsaved_message_params[ 'subject' ] = $edited_Thread->title;
 520                              $unsaved_message_params[ 'message' ] = $edited_Message->text;
 521                              $unsaved_message_params[ 'message_original' ] = $edited_Message->original_text;
 522                              $unsaved_message_params[ 'renderers' ] = $edited_Message->get_renderers_validated();
 523                              $unsaved_message_params[ 'thrdtype' ] = param( 'thrdtype', 'string', 'individual' );  // alternative: discussion
 524                              $unsaved_message_params[ 'thrd_recipients' ] = $thrd_recipients;
 525                              $unsaved_message_params[ 'thrd_recipients_array' ] = $thrd_recipients_array;
 526                              $unsaved_message_params[ 'creating_success' ] = $creating_success;
 527                              save_message_params_to_session( $unsaved_message_params );
 528                          }
 529                          break;
 530  
 531                      case 'delete': // delete thread
 532                          // Check permission:
 533                          $current_User->check_perm( 'perm_messaging', 'delete', true );
 534  
 535                          $confirmed = param( 'confirmed', 'integer', 0 );
 536                          if( $confirmed )
 537                          {
 538                              $msg = sprintf( T_('Thread &laquo;%s&raquo; deleted.'), $edited_Thread->dget('title') );
 539                              $edited_Thread->dbdelete();
 540                              unset( $edited_Thread );
 541                              forget_param( 'thrd_ID' );
 542                              $Messages->add( $msg, 'success' );
 543                          }
 544                          else
 545                          {
 546                              $delete_url = $samedomain_htsrv_url.'action.php?mname=messaging&thrd_ID='.$edited_Thread->ID.'&action=delete&confirmed=1&redirect_to='.$redirect_to.'&'.url_crumb( 'messaging_threads' );
 547                              $ok_button = '<a href="'.$delete_url.'" class="btn btn-danger">'.T_( 'I am sure!' ).'</a>';
 548                              $cancel_button = '<a href="'.$redirect_to.'" class="btn btn-default">CANCEL</a>';
 549                              $msg = sprintf( T_( 'You are about to delete all messages in the conversation &laquo;%s&raquo;.' ), $edited_Thread->dget('title') );
 550                              $msg .= '<br />'.T_( 'This CANNOT be undone!').'<br />'.T_( 'Are you sure?' ).'<br /><br />'.$ok_button."\t".$cancel_button;
 551                              $Messages->add( $msg, 'error' );
 552                          }
 553                          break;
 554  
 555                      case 'leave': // user wants to leave the thread
 556                          leave_thread( $edited_Thread->ID, $current_User->ID, false );
 557  
 558                          $Messages->add( sprintf( T_( 'You have successfuly left the &laquo;%s&raquo; conversation!' ), $edited_Thread->get( 'title' ) ), 'success' );
 559                          break;
 560  
 561                      case 'close': // close the thread
 562                      case 'close_and_block': // close the thread and block contact
 563                          leave_thread( $edited_Thread->ID, $current_User->ID, true );
 564  
 565                          // user has closed this conversation because there was only one other user involved
 566                          $Messages->add( sprintf( T_( 'You have successfuly closed the &laquo;%s&raquo; conversation!' ), $edited_Thread->get( 'title' ) ), 'success' );
 567                          if( $action == 'close_and_block' )
 568                          { // user also wants to block contact with the other user involved in this thread
 569                              $block_user_ID = param( 'block_ID', 'integer', true );
 570                              $UserCache = & get_UserCache();
 571                              $blocked_User = $UserCache->get_by_ID( $block_user_ID );
 572  
 573                              set_contact_blocked( $block_user_ID, true );
 574                              $Messages->add( sprintf( T_( '&laquo;%s&raquo; was blocked.' ), $blocked_User->get( 'login' ) ), 'success' );
 575                          }
 576                          break;
 577                  }
 578                  break; // break from threads action switch
 579  
 580              // contacts action
 581              case 'contacts':
 582                  $user_ID = param( 'user_ID', 'string', true );
 583  
 584                  if( ( $action != 'block' ) && ( $action != 'unblock' ) )
 585                  { // only block or unblock is valid
 586                      debug_die( "Invalid action param" );
 587                  }
 588                  if( set_contact_blocked( $user_ID, ( ( $action == 'block' ) ? 1 : 0 ) ) )
 589                  {
 590                      if( $action == 'block' )
 591                      {
 592                          $Messages->add( T_('You have blocked this user from contacting you.'), 'success' );
 593                      }
 594                      else
 595                      {
 596                          $Messages->add( T_('You have unblocked this user so he can contact you again.'), 'success' );
 597                      }
 598                  }
 599                  $redirect_to = str_replace( '&amp;', '&', $redirect_to );
 600                  break;
 601  
 602              // messages action
 603              case 'messages':
 604                  switch( $action )
 605                  {
 606                      case 'create': // create new message
 607                          // Stop a request from the blocked IP addresses or Domains
 608                          antispam_block_request();
 609  
 610                          create_new_message( $thrd_ID );
 611                          break;
 612  
 613                      case 'preview': // create new message
 614                          // Stop a request from the blocked IP addresses or Domains
 615                          antispam_block_request();
 616  
 617                          global $edited_Message;
 618  
 619                          $creating_success = create_new_message( $thrd_ID, 'preview' );
 620  
 621                          // save new message and thread params into the Session to not lose the content
 622                          $unsaved_message_params = array();
