  44   * Generate category line when it has children
  45   *
  46   * @param object Chapter we want to display
  47   * @param integer Level of the category in the recursive tree
  48   * @return string HTML
  49   */
  50  function cat_line( $Chapter, $level )
  51  {
  52      global $line_class, $permission_to_edit, $current_User, $Settings;
  53      global $ChapterCache, $current_default_cat_ID;
  54      global $number_of_posts_in_cat;
  55  
  56      global $Session;
  57      $result_fadeout = $Session->get( 'fadeout_array' );
  58  
  59      $line_class = $line_class == 'even' ? 'odd' : 'even';
  60  
  61      // ID
  62      $r = '<tr id="tr-'.$Chapter->ID.'"class="'.$line_class.
  63                      ' chapter_parent_'.( $Chapter->parent_ID ? $Chapter->parent_ID : '0' ).
  64                      // Fadeout?
  65                      ( isset($result_fadeout) && in_array( $Chapter->ID, $result_fadeout ) ? ' fadeout-ffff00': '' ).'">
  66                      <td class="firstcol shrinkwrap">'.
  67                          $Chapter->ID.'
  68                  </td>';
  69  
  70      // Default
  71      if( $current_default_cat_ID == $Chapter->ID )
  72      {
  73          $makedef_icon = get_icon( 'enabled', 'imgtag', array( 'title' => format_to_output( T_( 'This is the default category' ), 'htmlattr' ) ) );
  74      }
  75      else
  76      {
  77          $makedef_url = regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action=make_default&amp;'.url_crumb('element') );
  78          $makedef_title = format_to_output( T_('Click to make this the default category'), 'htmlattr' );
  79          $makedef_icon = '<a href="'.$makedef_url.'" title="'.$makedef_title.'">'.get_icon( 'disabled', 'imgtag', array( 'title' => $makedef_title ) ).'</a>';
  80      }
  81      $r .= '<td class="center">'.$makedef_icon.'</td>';
  82  
  83      // Name
  84      if( $permission_to_edit )
  85      {    // We have permission permission to edit:
  86          $edit_url = regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action=edit' );
  87          $r .= '<td>
  88                          <strong style="padding-left: '.($level).'em;"><a href="'.$edit_url.'" title="'.T_('Edit...').'">'.$Chapter->dget('name').'</a></strong>
  89                       </td>';
  90      }
  91      else
  92      {
  93          $r .= '<td>
  94                           <strong style="padding-left: '.($level).'em;">'.$Chapter->dget('name').'</strong>
  95                       </td>';
  96      }
  97  
  98      // URL "slug"
  99      $edit_url = regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action=edit' );
 100      $r .= '<td><a href="'.htmlspecialchars($Chapter->get_permanent_url()).'">'.$Chapter->dget('urlname').'</a></td>';
 101  
 102      // Order
 103      if( $Chapter->get_parent_subcat_ordering() == 'manual' )
 104      { // Manual ordering
 105          $r .= '<td class="center cat_order_edit" rel="'.$Chapter->ID.'">'
 106                          .'<a href="#">'.( $Chapter->get( 'order' ) === NULL ? '-' : $Chapter->dget( 'order' ) ).'</a>'
 107                      .'</td>';
 108      }
 109      else
 110      { // Alphabetic ordering
 111          $r .= '<td class="center">'.T_('Alphabetic').'</td>';
 112      }
 113  
 114      if( $permission_to_edit )
 115      {    // We have permission permission to edit, so display these columns:
 116  
 117          if( $Chapter->meta )
 118          {
 119              $makemeta_icon = 'enabled';
 120              $makemeta_title = format_to_output( T_('Click to revert this from meta category'), 'htmlattr' );
 121              $action = 'unset_meta';
 122          }
 123          else
 124          {
 125              $makemeta_icon = 'disabled';
 126              $makemeta_title = format_to_output( T_('Click to make this as meta category'), 'htmlattr' );
 127              $action = 'set_meta';
 128          }
 129          // Meta
 130          $makemeta_url = regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action='.$action.'&amp;'.url_crumb('element') );
 131          $r .= '<td class="center"><a href="'.$makemeta_url.'" title="'.$makemeta_title.'">'.get_icon( $makemeta_icon, 'imgtag', array( 'title' => $makemeta_title ) ).'</a></td>';
 132  
 133          // Lock
 134          if( $Chapter->lock )
 135          {
 136              $makelock_icon = 'file_not_allowed';
 137              $makelock_title = format_to_output( T_('Unlock category'), 'htmlattr' );
 138              $action = 'unlock';
 139          }
 140          else
 141          {
 142              $makelock_icon = 'file_allowed';
 143              $makelock_title = format_to_output( T_('Lock category'), 'htmlattr' );
 144              $action = 'lock';
 145          }
 146          $makelock_url = regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action='.$action.'&amp;'.url_crumb('element') );
 147          $r .= '<td class="center"><a href="'.$makelock_url.'" title="'.$makelock_title.'">'.get_icon( $makelock_icon, 'imgtag', array( 'title' => $makelock_title ) ).'</a></td>';
 148      }
 149  
 150      // Posts
 151      if( isset($number_of_posts_in_cat[$Chapter->ID]) )
 152      {
 153          $r .= '<td class="center">'.(int)$number_of_posts_in_cat[$Chapter->ID].'</td>';
 154      }
 155      else
 156      {    // no posts in this category
 157          $r .= '<td class="center"> - </td>';
 158      }
 159  
 160      // Actions
 161      $r .= '<td class="lastcol shrinkwrap">';
 162      if( $permission_to_edit )
 163      {    // We have permission permission to edit, so display action column:
 164          $r .= action_icon( T_('Edit...'), 'edit', $edit_url );
 165          if( $Settings->get('allow_moving_chapters') )
 166          { // If moving cats between blogs is allowed:
 167              $r .= action_icon( T_('Move to a different blog...'), 'file_move', regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action=move' ), T_('Move') );
 168          }
 169          $r .= action_icon( T_('New...'), 'new', regenerate_url( 'action,cat_ID,cat_parent_ID', 'cat_parent_ID='.$Chapter->ID.'&amp;action=new' ) )
 170                      .action_icon( T_('Delete...'), 'delete', regenerate_url( 'action,cat_ID', 'cat_ID='.$Chapter->ID.'&amp;action=delete&amp;'.url_crumb('element') ) );
 171      }
 172      $r .= '</td>';
 173      $r .=    '</tr>';
 174  
 175      return $r;
 176  }
