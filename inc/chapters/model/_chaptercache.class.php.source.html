 118       * Get an object from cache by ID
 119       *
 120       * Load the cache if necessary (all at once if allowed).
 121       *
 122       * @param integer ID of object to load
 123       * @param boolean true if function should die on error
 124       * @param boolean true if function should die on empty/null
 125        * @param integer|NULL NULL for all subsets

 128      function & get_by_ID( $req_ID, $halt_on_error = true, $halt_on_empty = true, $subset_ID = NULL )
 129      {
 130          global $DB, $Debuglog;
 131  
 132          if( empty($req_ID) )
 133          {
 134              if($halt_on_empty)
 135              {
 136                  debug_die( "Requested Chapter from $this->dbtablename without ID!" );
 137              }
 138              $r = NULL;
 139              return $r;
 140          }
 141  
 142          if( !empty( $this->cache[ $req_ID ] ) )
 143          { // Already in cache
 144              $Debuglog->add( "Accessing Chapter($req_ID) from cache", 'dataobjects' );
 145              return $this->cache[ $req_ID ];
 146          }
 147          elseif( !$this->all_loaded )
 148          { // Not in cache, but not everything is loaded yet
 149              if( $this->load_all || is_null($subset_ID) )
 150              { // It's ok to just load everything:
 151                  $this->load_all();
 152              }
 153              else
 154              { // Load just the requested object:
 155                  $Debuglog->add( "Loading <strong>$this->objtype($req_ID)</strong> into cache", 'dataobjects' );
 156                  // Note: $req_ID MUST be an unsigned integer. This is how DataObject works.
 157                  $sql = "SELECT *
 158                            FROM T_categories
 159                           WHERE cat_ID = $req_ID
 160                             AND cat_blog_ID = ".$subset_ID;
 161  
 162                  if( $row = $DB->get_row( $sql, OBJECT, 0, 'ChapterCache::get_by_ID()' ) )
 163                  {
 164                      if( ! $this->instantiate( $row ) )
 165                      {
 166                          $Debuglog->add( 'Could not add() object to cache!', 'dataobjects' );
 167                      }
 168                  }
 169                  else
 170                  {
 171                      $Debuglog->add( 'Could not get DataObject by ID. Query: '.$sql, 'dataobjects' );
 172                  }
 173              }
 174          }
 175  
 176          if( empty( $this->cache[ $req_ID ] ) )
 177          { // Requested object does not exist
 178              // $Debuglog->add( 'failure', 'dataobjects' );
 179              if( $halt_on_error )
 180              {
 181                  debug_die( "Requested $this->objtype does not exist!" );
 182              }
 183              $r = false;
 184              return $r;
 185          }
 186  
 187          return $this->cache[ $req_ID ];
 188      }

 205              $chapters = $this->subset_cache[ $blog_ID ];
 206              if( isset( $chapters[ $chapter_ID ] ) )

 208                  $Chapter = $chapters[ $chapter_ID ];
 209                  while( $Chapter->get( 'parent_ID' ) > 0 )

 211                      $chapter_path[] = $Chapter->get( 'parent_ID' );

 480      {
 481          global $Debuglog;
 482          global $Timer;
 483  
 484          if( $this->revealed_all_children )
 485          {    // All children have already been revealed: (can happen even if we require a subset *now*)
 486              if( $sort_root_cats )
 487              {
 488                  $this->sort_root_cats( $subset_ID );
 489              }
 490              return;
 491          }
 492  
 493          if( empty($this->subset_property) )
 494          {    // We are not handling a subset but everything instead:
 495  
 496              echo 'DEBUG - PARTIAL IMPLEMENTATION - Revealing all children -- this is not yet handling all edge cases that the subset version can handle';
 497  
 498              // Make sure everything has been loaded:
 499              $this->load_all();
 500  
 501              // Reveal children:
 502              if( !empty( $this->cache ) )
 503              {    // There are loaded categories, so loop on all loaded categories to set their children list if it has:
 504                  foreach( $this->cache as $cat_ID => $Chapter )
 505                  {
 506                      // echo $Chapter->name;
 507                      if( ! is_null( $Chapter->parent_ID ) )
 508                      {    // This category has a parent, so add it to its parent children list:
 509                          $this->cache[$Chapter->parent_ID]->add_child_category( $this->cache[$cat_ID] );
 510                      }
 511                      else
 512                      {    // This category has no parent, so add it to the parent categories list
 513                          $this->root_cats[] = & $this->cache[$cat_ID];
 514                      }
 515                  }
 516              }
 517  
 518              if( $sort_root_cats )
 519              { // Sort root categories as it was requested
 520                  $this->sort_root_cats( NULL );
 521              }
 522  
 523              $this->revealed_all_children = true;
 524          }
 525          else
 526          {    // We are handling a subset (for example Chapers of a specific blog_ID only):
 527  
 528              if( is_null( $subset_ID ) )
 529              {    // No specific subset requested, we are going to reveal all subsets in a row:
 530  
 531                  // Make sure everything has been loaded:
 532                  $this->load_all();

 604              }

 610      /**
 611       * Support function for reveal_children()
 612       *
 613       * @param integer
 614       * @param integer
 615       * @return true if root parent
 616       */
 617  	function check_path_to_root( $subset_ID, $cat_ID, $path_array = array() )
 618      {
 619          global $Debuglog;
 620          $padding = str_pad('',count($path_array),'*');
 621          $Debuglog->add( $padding.'Checking path to root for cat ID='.$cat_ID.' with path:'.implode(',',$path_array), 'CategoryCache' );
 622          $Chapter = & $this->subset_cache[$subset_ID][$cat_ID];
 623          if( is_null( $Chapter->parent_ID ) )
 624          {    // Found root parent
 625              $Debuglog->add( $padding.'*OK', 'CategoryCache' );
 626              return true;
 627          }
 628          // This is not the last parent...
 629          // Record path...
 630          $path_array[] = $Chapter->ID;
 631  
 632          if( in_array( $Chapter->parent_ID, $path_array ) )
 633          {    // We are about to enter an infinite loop
 634              $Debuglog->add( $padding.'*LOOP! -> assign to root', 'CategoryCache' );
 635              $Chapter->parent_ID = NULL;
 636              return false;
 637          }
 638          elseif( ! isset($this->cache[$Chapter->parent_ID] ) )
 639          {
 640              $Debuglog->add( $padding.'*Missing parent ID('.$Chapter->parent_ID.')! -> assign to root', 'CategoryCache' );
 641              $Chapter->parent_ID = NULL;
 642              return false;
 643          }
 644  
 645          // Recurse!
 646          return $this->check_path_to_root( $subset_ID, $Chapter->parent_ID, $path_array );
 647      }

 653       *
 654       * @param Object Item

 659  	function compare_item_with_chapter( $Item, $Chapter, $order_type )
 660      {
 661          if( $order_type == 'alpha' )
 662          {
 663              return strcmp( $Item->title, $Chapter->name );
 664          }

 672              return -1;
 673          }
 674

 691                  $this->sorted_flags['root_cats'] = true;
 692              }
 693              return;
 694          }

 696          if( $subset_ID == NULL )
 697          {

 699              {
 700                  if( $this->sorted_flags[$key] )
 701                  { // This subset was already sorted
 702                      continue;
 703                  }
 704  
 705                  usort( $this->subset_root_cats[$key], array( 'Chapter','compare_chapters' ) );
 706                  $this->sorted_flags[$key] = true;
 707              }
 708              return;
 709          }
 710  
 711          if( empty($this->sorted_flags[$subset_ID]) )
 712          { // This subset was not sorted yet
 713              usort( $this->subset_root_cats[$subset_ID], array( 'Chapter','compare_chapters' ) );
 714              $this->sorted_flags[$subset_ID] = true;
 715          }
 716      }

 719      /**
 720       * Return recursive display of loaded categories
 721       *
 722       * @param array callback funtions (to format the display)
 723       * @param integer|NULL NULL for all subsets
 724       * @param array categories list to display
 725       * @param integer depth of categories list
 726       * @param integer Max depth of categories list, 0/empty - is infinite
 727       * @param array Other params regarding the recursive display:
 728       *   boolean sorted, sort categories or not - default value is false
 729       *   string chapter_path, array of chapter ids from root category to the selected category
 730       *   boolean expand_all, set true to expand all categories and not only the one in the chapter path
 731       *   etc.
 732       *
 733       * @return string recursive list of all loaded categories
 734       */
 735  	function recurse( $callbacks, $subset_ID = NULL, $cat_array = NULL, $level = 0, $max_level = 0, $params = array() )
 736      {
 737          $params = array_merge( array(
 738                  'sorted' => false,
 739                  'chapter_path' => array(),
 740                  'expand_all' => true,
 741                  'posts_are_loaded' => false,
 742                  'subset_ID' => $subset_ID,
 743                  'max_level' => $max_level
 744              ), $params );
 745          $sorted = $params['sorted'];
 746          $display_posts = ! empty( $callbacks['posts'] );
 747  
 748          // Make sure children have been (loaded and) revealed for specific subset:
 749          $this->reveal_children( $subset_ID, $sorted );
 750  
 751          if( $display_posts && !$params['posts_are_loaded'] && ( $params['expand_all'] || !empty( $params['chapter_path'] ) ) )
 752          { // Posts needs to be loaded
 753              $ItemCache = & get_ItemCache();
 754              $ItemCache->load_by_categories( $params['chapter_path'], $subset_ID );
 755              $params['posts_are_loaded'] = true;
 756          }
 757  
 758          if( is_null( $cat_array ) )
 759          {    // Get all parent categories:
 760              if( is_null( $subset_ID ) )
 761              {
 762                  $cat_array = $this->root_cats;
 763              }
 764              elseif( isset( $this->subset_root_cats[$subset_ID] ) )
 765              { // We have root cats for the requested subset:
 766                  $cat_array = $this->subset_root_cats[$subset_ID];
 767              }
 768              else
 769              {
 770                  $cat_array = array();
 771              }
 772          }
 773  
 774          $r = '';
 775  
 776          // Go through all categories at this level:
 777          foreach( $cat_array as $cat )
 778          {
 779              // Check if category is expended
 780              $params['is_selected'] = in_array( $cat->ID, $params['chapter_path'] );
 781              $params['is_opened'] = $params['expand_all'] || $params['is_selected'];
 782  
 783              // Display a category:
 784              if( is_array( $callbacks['line'] ) )
 785              { // object callback:
 786                  $r .= $callbacks['line'][0]->{$callbacks['line'][1]}( $cat, $level, $params ); // <li> Category  - or - <tr><td>Category</td></tr> ...
 787              }
 788              else
 789              {
 790                  $r .= $callbacks['line']( $cat, $level, $params ); // <li> Category  - or - <tr><td>Category</td></tr> ...
 791              }
 792  
 793              if( ( empty( $max_level ) || $max_level > $level + 1 ) && $params['is_opened'] )
 794              { // Iterate through sub categories recursively
 795                  $params['level'] = $level;
 796                  $r .= $this->iterate_through_category_children( $cat, $callbacks, true, $params );
 797              }
 798          }
 799  
 800          if( ! empty( $r ) )
 801          { // We have something to display at this level, wrap in in before/after:
 802              $r_before = '';
 803              if( is_array( $callbacks['before_level'] ) )
 804              { // object callback:
 805                  $r_before .= $callbacks['before_level'][0]->{$callbacks['before_level'][1]}( $level ); // <ul>
 806              }
 807              elseif( isset( $callbacks['before_level'] ) )
 808              {
 809                  $r_before .= $callbacks['before_level']( $level ); // <ul>
 810              }
 811  
 812              $r = $r_before.$r;
 813  
 814              if( is_array( $callbacks['after_level'] ) )
 815              { // object callback:
 816                  $r .= $callbacks['after_level'][0]->{$callbacks['after_level'][1]}( $level ); // </ul>
 817              }
 818              elseif( isset( $callbacks['after_level'] ) )
 819              {
 820                  $r .= $callbacks['after_level']( $level ); // </ul>
 821              }
 822          }
 823  
 824          return $r;
 825      }

 828      /**
 829       * Iterate through the given Chapter sub cats and items
 830       *
 831       * @param Object Chapter
 832       * @param array  Callback functions to display a Chapter and to display an Item
 833       * @param boolean Set true to iterate sub categories recursively, false otherwise
 834       * @param array Any additional params ( e.g. 'sorted', 'level', 'list_subs_start', etc. )
 835       * @return string the concatenated callbacks result
 836       */

 838      {
 839          $r = "";
 840          $cat_items = array();
 841          $has_sub_cats = ! empty( $Chapter->children );

 843                  'sorted'    => false,
 844                  'level'     => 0,
 845                  'max_level' => 0,
 846                  'subset_ID' => $Chapter->blog_ID

 867              $cat_order = $Chapter->get_subcat_ordering();

 869              $params['chapter_ID'] = $Chapter->ID;

 873              {

 875              }
 876  
 877              while( $has_more_children || $has_more_items )
 878              {
 879                  $current_sub_Chapter = $has_more_children ? $Chapter->children[$chapter_children_ids[$cat_index]] : NULL;
 880                  $current_Item = $has_more_items ? $cat_items[$item_index] : NULL;
 881                  if( $current_Item != NULL && ( $current_sub_Chapter == NULL || ( $this->compare_item_with_chapter( $current_Item, $current_sub_Chapter, $cat_order ) <= 0 ) ) )

 891                              foreach( $subcats_to_display as $sub_Chapter )

 964       * Return recursive select options list of all loaded categories
 965       *
 966       * @param integer selected category in the select input
 967       * @param integer NULL for all subsets
 968       * @param boolean Include the root element?
 969       * @param array GenercCategory objects to display (will recurse from those starting points)
 970       * @param integer depth of categories list
 971       * @param array IDs of categories to exclude (their children will be ignored to)
 972       *
 973       * @return string select options list of all loaded categories
 974       */
 975  	function recurse_select( $selected = NULL, $subset_ID = NULL, $include_root = false, $Cat_array = NULL,
 976                              $level = 0, $exclude_array = array() )
 977      {
 978          // pre_dump( $exclude_array );
 979  
 980          // Make sure children have been revealed for specific subset:
 981          $this->reveal_children( $subset_ID );
 982  
 983          if( is_null( $Cat_array ) )
 984          {    // Get all parent categorie:
 985              $Cat_array = $this->root_cats;
 986          }
 987

 989  
 990          if( $include_root )
 991          {
 992              $r .= '<option value="">'.T_('Root').'</option>';
 993              $level++;
 994          }
 995  
 996          foreach( $Cat_array as $Chapter )
 997          {
 998              if( in_array( $Chapter->ID, $exclude_array ) )
 999              {    // We want to exclude that cat.
1000                  continue;
1001              }
1002  
1003              // Set category indentation in the select:
1004              $indent = '';
1005              for($i = 0; $i < $level; $i++)
1006              {
1007                  $indent .='&nbsp;&nbsp;';
1008              }
1009              // Set category option:
1010              $r .= '<option value="'.$Chapter->ID.'" ';
1011              if( $Chapter->ID == $selected ) $r .= ' selected="selected"';
1012              $r .= ' >'.$indent.$Chapter->name.'</option>';
1013  
1014              if( !empty( $Chapter->children ) )
1015              {    // Add children categories:
1016                  $r .= $this->recurse_select( $selected, $subset_ID, false, $Chapter->children, $level+1, $exclude_array );
1017              }
1018          }

1020          return $r;
