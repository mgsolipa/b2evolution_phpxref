 161      /**
 162       * Compare two Chapters based on the parent/blog sort category setting
 163       *
 164       * @param Chapter A
 165       * @param Chapter B
 166       * @return number -1 if A < B, 1 if A > B, 0 if A == B
 167       */
 168  	static function compare_chapters( $a_Chapter, $b_Chapter )
 169      {
 170          if( $a_Chapter == NULL || $b_Chapter == NULL ) {
 171              debug_die('Invalid category objects received to compare.');
 172          }
 173  
 174          if( $a_Chapter->ID == $b_Chapter->ID )
 175          { // The two chapters are the same
 176              return 0;
 177          }
 178  
 179          if( $a_Chapter->blog_ID != $b_Chapter->blog_ID )
 180          { // Sort based on the ordering between different blogs
 181              $a_Chapter->load_Blog();
 182              $b_Chapter->load_Blog();
 183              return Blog::compare_blogs($a_Chapter->Blog, $b_Chapter->Blog);
 184          }
 185  
 186          $ChapterCache= & get_ChapterCache();
 187          if( $a_Chapter->parent_ID != $b_Chapter->parent_ID )
 188          { // Two chapters from the same blog, but different parrents
 189              // Compare those parents of these chapters which have a common parent Chapter or they are both root Chapters.
 190              $path_to_root_a = array_reverse( $ChapterCache->get_chapter_path( $a_Chapter->blog_ID, $a_Chapter->ID ) );
 191              $path_to_root_b = array_reverse( $ChapterCache->get_chapter_path( $b_Chapter->blog_ID, $b_Chapter->ID ) );
 192              $index = 0;
 193              while( isset( $path_to_root_a[$index] ) && isset( $path_to_root_b[$index] ) )
 194              {
 195                  if( $path_to_root_a[$index] != $path_to_root_b[$index] )
 196                  { // The first different parent on the same level was found, compare parent objects
 197                      $parent_a_Chapter = $ChapterCache->get_by_ID( $path_to_root_a[$index] );
 198                      $parent_b_Chapter = $ChapterCache->get_by_ID( $path_to_root_b[$index] );
 199                      return self::compare_chapters( $parent_a_Chapter, $parent_b_Chapter );
 200                  }
 201                  $index++;
 202              }
 203  
 204              // One of the chapters is a parent of the other, the parent is considered greater than the other
 205              return isset( $path_to_root_a[$index] ) ? 1 : -1;
 206          }
 207  
 208          if( empty( $a_Chapter->parent_ID ) )
 209          {
 210              $a_Chapter->load_Blog();
 211              $cat_order = $a_Chapter->Blog->get_setting('category_ordering');
 212          }
 213          else
 214          {
 215              $parent_Chapter = $ChapterCache->get_by_ID( $a_Chapter->parent_ID );
 216              $cat_order = $parent_Chapter->get_subcat_ordering();
 217          }
 218  
 219          switch( $cat_order )
 220          {
 221              case 'alpha':
 222                  $result = strcmp( $a_Chapter->name, $b_Chapter->name );
 223                  break;
 224  
 225              case 'manual':
 226                  if( $a_Chapter->order === NULL )
 227                  { // NULL values are greater than any number
 228                      $result = ( $b_Chapter->order !== NULL ) ? 1 : 0;
 229                      break;
 230                  }
 231  
 232                  if( $b_Chapter->order === NULL )
 233                  { // NULL values are greater than any number, so a is lower than b
 234                      $result = -1;
 235                      break;
 236                  }
 237  
 238                  $result = ( $a_Chapter->order > $b_Chapter->order ) ? 1 : ( ( $a_Chapter->order < $b_Chapter->order ) ? -1 : 0 );
 239                  break;
 240  
 241              default:
 242                  debug_die("Invalid category ordering value!");
 243          }
 244  
 245          if( $result == 0 )
 246          { // In case if the order fields are equal order by ID
 247              $result = $a_Chapter->ID > $b_Chapter->ID ? 1 : -1;
 248          }
 249  
 250          return $result;
 251      }

 291  	function load_from_request()
 292      {
 293          param_string_not_empty( 'cat_name', T_('Please enter a name.') );
 294          $this->set_from_Request( 'name' );
 295  
 296          if( param( 'cat_parent_ID', 'integer', -1 ) !== -1 )
 297          {    // Set parent ID:
 298              $this->set_from_Request( 'parent_ID' );
 299          }
 300  
 301          // Check url name
 302          param( 'cat_urlname', 'string' );
 303          $this->set_from_Request( 'urlname' );
 304  
 305          // Check description
 306          param( 'cat_description', 'string' );
 307          $this->set_from_Request( 'description' );
 308  
 309          $cat_Blog = & $this->get_Blog();
 310          if( $cat_Blog && $cat_Blog->get_setting('category_ordering') == 'manual' )
 311          { // Manual ordering
 312              param( 'cat_order', 'integer', NULL );
 313              $this->set_from_Request( 'order', 'cat_order', true );
 314          }
 315  
 316          // Sort sub-categories
 317          param( 'cat_subcat_ordering', 'string' );
 318          $this->set_from_Request( 'subcat_ordering' );
 319  
 320          // Meta category
 321          $cat_meta = param( 'cat_meta', 'integer', 0 );
 322          if( $this->has_posts() && $cat_meta )
 323          {    // Display error message if we want make the meta category from category with posts
 324              global $Messages;
 325              $Messages->add( sprintf( T_('The category &laquo;%s&raquo; cannot be set as meta category. You must remove the posts it contains first.'), $this->dget('name') ) );
 326          }
 327          else
 328          {    // Save the category as 'Meta' only if it has no posts
 329              $this->set_from_Request( 'meta' );
 330          }
 331  
 332          // Locked category
 333          param( 'cat_lock', 'integer', 0 );
 334          $this->set_from_Request( 'lock' );
 335  
 336          return ! param_errors_detected();
 337      }

 379      /**
 380       * Generate the URL to access the category.
 381       *
 382       * @param string|NULL 'param_num', 'subchap', 'chapters'
 383       * @param string|NULL url to use
 384       * @param integer category page to link to, default:1
 385       * @param integer|NULL number of posts per page (used for param_num only)
 386       * @param string glue between url params
 387       */
 388  	function get_permanent_url( $link_type = NULL, $blogurl = NULL, $paged = 1, $chapter_posts_per_page = NULL, $glue = '&amp;' )
 389      {
 390          global $DB, $cacheweekly, $Settings;
 391  
 392          if( empty( $link_type ) )
 393          {    // Use default from settings:
 394              $this->get_Blog();
 395              $link_type = $this->Blog->get_setting( 'chapter_links' );
 396          }
 397  
 398          if( empty( $blogurl ) )
 399          {
 400              $this->get_Blog();
 401              $blogurl = $this->Blog->gen_blogurl();
 402          }
 403  
 404          switch( $link_type )
 405          {
 406              case 'param_num':
 407                  $r = url_add_param( $blogurl, 'cat='.$this->ID, $glue );
 408                  if( empty($chapter_posts_per_page) )
 409                  {    // Use default from Blog
 410                      $this->get_Blog();
 411                      $chapter_posts_per_page = $this->Blog->get_setting( 'chapter_posts_per_page' );
 412                  }
 413                  if( !empty($chapter_posts_per_page) && $chapter_posts_per_page != $this->Blog->get_setting( 'posts_per_page' ) )
 414                  {    // We want a specific post per page count:
 415                      $r = url_add_param( $r, 'posts='.$chapter_posts_per_page, $glue );
 416                  }
 417                  break;
 418  
 419              case 'subchap':
 420                  $this->get_Blog();
 421                  $category_prefix = $this->Blog->get_setting('category_prefix');
 422                  if( !empty( $category_prefix ) )
 423                  {
 424                      $r = url_add_tail( $blogurl, '/'.$category_prefix.'/'.$this->urlname.'/' );
 425                  }
 426                  else
 427                  {
 428                      $r = url_add_tail( $blogurl, '/'.$this->urlname.'/' );
 429                  }
 430                  break;
 431  
 432              case 'chapters':
 433              default:
 434                  $this->get_Blog();
 435                  $category_prefix = $this->Blog->get_setting('category_prefix');
 436                  if( !empty( $category_prefix ) )
 437                  {
 438                      $r = url_add_tail( $blogurl, '/'.$category_prefix.'/'.$this->get_url_path() );
 439                  }
 440                  else
 441                  {
 442                      $r = url_add_tail( $blogurl, '/'.$this->get_url_path() );
 443                  }
 444                  break;
 445          }
 446  
 447          if( $paged > 1 )
 448          {
 449              $r = url_add_param( $r, 'paged='.$paged, $glue );
 450          }
 451  
 452          return $r;
 453      }

 565       * Update the DB based on previously recorded changes
 566       *
 567       * @return boolean true on success
 568       */
 569  	function dbupdate()
 570      {
 571          global $DB;
 572  
 573          // Start transaction because of urltitle validation
 574          $DB->begin( 'SERIALIZABLE' );
 575  
 576          // validate url title / slug
 577          if( empty($this->urlname) || isset($this->dbchanges['cat_urlname']) )
 578          { // Url title has changed or is empty
 579              $this->set( 'urlname', urltitle_validate( $this->urlname, $this->name, $this->ID, false, $this->dbprefix.'urlname', $this->dbIDname, $this->dbtablename) );
 580          }
 581  
 582          if( count( $this->dbchanges ) > 0 && !isset( $this->dbchanges['last_touched_ts'] ) )
 583          { // Update last_touched_ts field only if it wasn't updated yet and the datemodified will be updated for sure.
 584              global $localtimenow;
 585              $this->set_param( 'last_touched_ts', 'date', date( 'Y-m-d H:i:s', $localtimenow ) );
 586          }
 587  
 588          if( parent::dbupdate() === false )
 589          { // The update was unsuccessful
 590              $DB->rollback();
 591              return false;
 592          }
 593  
 594          // The chapter was updated successful
 595          $DB->commit();
 596          return true;

 687      /**
 688       * Get URL to edit a chapter if user has edit rights.
 689       *
 690       * @param array Params:
 691       *  - 'redirect_page': redirect to page: 'front', 'manual', 'list'
 692       *  - 'glue' : Glue string between url params
 693       * @return string|FALSE URL
 694       */
 695  	function get_edit_url( $params = array() )
 696      {
 697          $params = array_merge( array(
 698                  'redirect_page' => '',
 699                  'glue'          => '&amp;',
 700              ), $params );
 701  
 702          if( ! is_logged_in( false ) )
 703          { // User is not logged in
 704              return false;
 705          }
 706  
 707          if( ! $this->ID )
 708          { // New chapter
 709              return false;
 710          }
 711  
 712          global $current_User;
 713  
 714          if( ! $current_User->check_perm( 'admin', 'restricted' ) ||
 715              ! $current_User->check_perm( 'blog_cats', '', false, $this->blog_ID ) )
 716          { // User has no right to edit this chapter
 717              return false;
 718          }
