 151      {
 152          case 'parent':
 153              // Redirect to parent chapter on front-office:
 154              if( $parent_Chapter = & $Chapter->get_parent_Chapter() )
 155              {    // If 
 156                  $redirect_url = $parent_Chapter->get_permanent_url( NULL, NULL, 1, NULL, '&' );
 157                  break;
 158              }
 159              // else redirect to permanent url of current chapter:
 160  
 161          case 'front':
 162              // Redirect to front-office
 163              $redirect_url = $Chapter->get_permanent_url( NULL, NULL, 1, NULL, '&' );
 164              break;
 165  
 166          case 'manual':
 167              // Redirect to manual pages
 168              $redirect_url = $admin_url.'?ctrl=items&blog='.$blog.'&tab=manual';
 169              if( !empty( $parent_ID ) )
 170              { // Open parent category to display new created category
 171                  $redirect_url .= '&cat_ID='.$parent_ID;
 172              }
 173              break;
 174  
 175          default: // 'list'
 176              // Redirect to chapters list
 177              $redirect_url = $admin_url.'?ctrl=chapters&blog='.$blog;
 178              break;
 179      }
