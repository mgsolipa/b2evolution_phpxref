  70  function display_pluggable_permissions( &$Form, $perm_block )
  71  {
  72      global $edited_Group;
  73  
  74      $GroupSettings = & $edited_Group->get_GroupSettings();
  75      foreach( $GroupSettings->permission_modules as $perm_name => $module_name )
  76      {
  77          $Module = & $GLOBALS[$module_name.'_Module'];
  78          if( method_exists( $Module, 'get_available_group_permissions' ) )
  79          {
  80              $permissions = $Module->get_available_group_permissions( $edited_Group->ID );
  81              if( array_key_exists( $perm_name, $permissions ) )
  82              {
  83                  $perm = $permissions[$perm_name];
  84                  if( $perm['perm_block'] == $perm_block )
  85                  {
  86                      if( ! isset( $perm['perm_type'] ) )
  87                      {
  88                          $perm['perm_type'] = 'radiobox';
  89                      }
  90  
  91                      switch( $perm['perm_type'] )
  92                      {
  93                          case 'checkbox':
  94                              $Form->checkbox_input( 'edited_grp_'.$perm_name, $GroupSettings->permission_values[$perm_name] == 'allowed', $perm['label'], array( 'input_suffix' => ' '.$perm['note'], 'value' => 'allowed' ) );
  95                          break;
  96  
  97                          case 'radiobox':
  98                              if( ! isset( $perm['field_lines'] ) )
  99                              {
 100                                  $perm['field_lines'] = true;
 101                              }
 102                              if( ! isset( $perm['field_note'] ) )
 103                              {
 104                                  $perm['field_note'] = '';
 105                              }
 106                              $Form->radio( 'edited_grp_'.$perm_name, $GroupSettings->permission_values[$perm_name], $perm['options'], $perm['label'], $perm['field_lines'], $perm['field_note'] );
 107                          break;
 108  
 109                          case 'info':
 110                              $Form->info( $perm['label'], $perm['info'] );
 111                          break;
 112  
 113                          case 'text_input':
 114                              $Form->text_input( 'edited_grp_'.$perm_name, $GroupSettings->permission_values[$perm_name], 5, $perm['label'], $perm['note'], array( 'maxlength' => $perm['maxlength'] ) );
 115                          break;
 116                      }
 117                  }
 118              }
