 190  function init_jcrop_tool( image_obj )
 191  {
 192      target_cropped_image_width = image_obj.width();
 193      target_cropped_image_height = image_obj.height();
 194  
 195      var min_width = Math.floor( <?php echo intval( $Settings->get( 'min_picture_size' ) ); ?> * Math.min( target_cropped_image_width, target_cropped_image_height ) / <?php echo $file_size; ?> );
 196      var min_height = min_width;
 197  
 198      // Set default selected crop area
 199      if( target_cropped_image_width == target_cropped_image_height )
 200      { // Square image
 201          if( target_cropped_image_width > min_width )
 202          {

 208              var x1 = 0;

 212          if( target_cropped_image_height > min_height )
 213          {
 214              var y1 = target_cropped_image_height * 0.05;
 215              var y2 = target_cropped_image_height * 0.95;
 216          }
 217          else
 218          {
 219              var y1 = 0;
 220              var y2 = target_cropped_image_height;
 221              min_height = y2;
 222          }
 223      }
 224      else if( target_cropped_image_width > target_cropped_image_height )
 225      { // Horizontal image
 226          var x1 = target_cropped_image_width / 2 - target_cropped_image_height / 2
 227          var y1 = 0;
 228          var x2 = target_cropped_image_width / 2 + target_cropped_image_height / 2;
 229          var y2 = target_cropped_image_height;
 230      }
 231      else if( target_cropped_image_width < target_cropped_image_height )
 232      { // Vertical image
 233          var x1 = 0;
 234          var y1 = target_cropped_image_height / 2 - target_cropped_image_width / 2;

 261  // Update thumbnails on change a crop area
 262  function show_preview_cropped_image( coords )
 263  {
 264      var percent_width = Math.ceil( coords.w / target_cropped_image_width * 10000 ) / 100;
 265      var percent_height = Math.ceil( coords.h / target_cropped_image_height * 10000 ) / 100;
 266      var percent_top = Math.ceil( coords.x / target_cropped_image_width * 10000 ) / 100;
 267      var percent_left = Math.ceil( coords.y / target_cropped_image_height * 10000 ) / 100;
 268      jQuery( 'input[name=image_crop_data]' ).val( percent_top + ':' + percent_left + ':' + percent_width + ':' + percent_height );
 269  
 270      var top = coords.y;
 271      var left = coords.x;
 272      if( coords.w > coords.h )
 273      { // Center a cropping area of horizontal image
 274          left += ( coords.w / 2 ) - ( coords.h / 2 );
 275      }
 276      else
 277      {
 278          var top_shift = ( coords.h - coords.w ) * 0.15;
 279          if( top + top_shift + coords.w < top + coords.h )
 280          { // top - 15%
 281              top += top_shift;
 282          }
 283      }
