 127  	function load_from_Request()
 128      {
 129          global $Messages, $demo_mode;
 130  
 131          // Edited Group Name
 132          param( 'edited_grp_name', 'string' );
 133          param_check_not_empty( 'edited_grp_name', T_('You must provide a group name!') );
 134          $this->set_from_Request('name', 'edited_grp_name', true);
 135  
 136          // Edited Group Level
 137          param_integer_range( 'edited_grp_level', 0, 10, T_('Group level must be between %d and %d.') );
 138          $this->set_from_Request( 'level', 'edited_grp_level', true );
 139  
 140          // Edited Group Permission Blogs
 141          param( 'edited_grp_perm_blogs', 'string', true );
 142          $this->set_from_Request( 'perm_blogs', 'edited_grp_perm_blogs', true );
 143  
 144          $apply_antispam = ( param( 'apply_antispam', 'integer', 0 ) ? 0 : 1 );
 145          $perm_xhtmlvalidation = param( 'perm_xhtmlvalidation', 'string', true );
 146          $perm_xhtmlvalidation_xmlrpc = param( 'perm_xhtmlvalidation_xmlrpc', 'string', true );
 147          $prevent_css_tweaks = ( param( 'prevent_css_tweaks', 'integer', 0 ) ? 0 : 1 );
 148          $prevent_iframes = ( param( 'prevent_iframes', 'integer', 0 ) ? 0 : 1 );
 149          $prevent_javascript = ( param( 'prevent_javascript', 'integer', 0 ) ? 0 : 1 );
 150          $prevent_objects = ( param( 'prevent_objects', 'integer', 0 ) ? 0 : 1 );
 151  
 152          if( $demo_mode && ( $apply_antispam || ( $perm_xhtmlvalidation != 'always' ) && ( $perm_xhtmlvalidation_xmlrpc != 'always' )
 153               || $prevent_css_tweaks || $prevent_iframes || $prevent_javascript || $prevent_objects ) )
 154          { // Demo mode restriction: Do not allow to change these settings in demo mode, because it may lead to security problem!
 155              $Messages->add( 'Validation settings and security filters are not editable in demo mode!', 'error' );
 156          }
 157          else
 158          {
 159              // Apply Antispam
 160              $this->set( 'perm_bypass_antispam', $apply_antispam );
 161  
 162              // XHTML Validation
 163              $this->set( 'perm_xhtmlvalidation', $perm_xhtmlvalidation );
 164  
 165              // XHTML Validation XMLRPC
 166              $this->set( 'perm_xhtmlvalidation_xmlrpc', $perm_xhtmlvalidation_xmlrpc );
 167  
 168              // CSS Tweaks
 169              $this->set( 'perm_xhtml_css_tweaks', $prevent_css_tweaks );
 170  
 171              // Iframes
 172              $this->set( 'perm_xhtml_iframes', $prevent_iframes );
 173  
 174              // Javascript
 175              $this->set( 'perm_xhtml_javascript', $prevent_javascript );
 176  
 177              // Objects
 178              $this->set( 'perm_xhtml_objects', $prevent_objects );
 179          }
 180  
 181          // Stats
 182          $this->set( 'perm_stats', param( 'edited_grp_perm_stats', 'string', true ) );
 183  
 184          // Load pluggable group permissions from request
 185          $GroupSettings = & $this->get_GroupSettings();
 186          foreach( $GroupSettings->permission_values as $name => $value )
 187          {
 188              // We need to handle checkboxes and radioboxes separately , because when a checkbox isn't checked the checkbox variable is not sent
 189              if( $name == 'perm_createblog' || $name == 'perm_getblog' || $name == 'perm_templates'
 190                  || $name == 'cross_country_allow_profiles' || $name == 'cross_country_allow_contact' )
 191              { // These permissions are represented by checkboxes, all other pluggable group permissions are represented by radiobox.
 192                  $value = param( 'edited_grp_'.$name, 'string', 'denied' );
 193              }
 194              elseif( ( $name == 'perm_admin' || $name == 'perm_users' ) && ( $this->ID == 1 ) )
 195              { // Admin group has always admin perm, it can not be set or changed.
 196                  continue;
 197              }
 198              else
 199              {
 200                  $value = param( 'edited_grp_'.$name, 'string', '' );
 201              }
 202              if( ( $value != '') || ( $name == 'max_new_threads'/*allow empty*/ ) )
 203              { // if radio is not set, then doesn't change the settings
 204                  $GroupSettings->set( $name, $value, $this->ID );
 205              }
 206          }
 207  
 208          return !param_errors_detected();
 209      }

 249      /**
 250       * Check a permission for this group.
 251       *
 252       * @param string Permission name:
 253       *                - templates
 254       *                - stats
 255       *                - spamblacklist
 256       *                - options
 257       *                - users
 258       *                - blogs
 259       *                - admin (levels "visible", "hidden")
 260       *                - messaging
 261       * @param string Requested permission level
 262       * @param mixed Permission target (blog ID, array of cat IDs...)
 263       * @return boolean True on success (permission is granted), false if permission is not granted
 264       */
 265  	function check_perm( $permname, $permlevel = 'any', $perm_target = NULL )
 266      {
 267          global $Debuglog;
 268  
 269          $perm = false; // Default is false!
 270  
 271          // echo "<br>Checking group perm $permname:$permlevel against $permvalue";
 272          if( isset($this->{'perm_'.$permname}) )
 273          {
 274              $permvalue = $this->{'perm_'.$permname};
 275          }
 276          else
 277          { // Object's perm-property not set!
 278              $Debuglog->add( 'Group permission perm_'.$permname.' not defined!', 'perms' );
 279  
 280              $permvalue = false; // This will result in $perm == false always. We go on for the $Debuglog..
 281          }
 282  
 283          $pluggable_perms = array( 'admin', 'shared_root', 'import_root', 'spamblacklist', 'slugs', 'templates', 'options', 'emails', 'files', 'users' );
 284          if( in_array( $permname, $pluggable_perms ) )
 285          {
 286              $permname = 'perm_'.$permname;
 287          }
 288          // echo "<br>Checking group perm $permname:$permlevel against $permvalue";
 289  
 290          // Check group permission:
 291          switch( $permname )
 292          {
 293              case 'blogs':
 294                  switch( $permvalue )
 295                  { // Depending on current group permission:
 296  
 297                      case 'editall':
 298                          // All permissions granted
 299                          $perm = true;
 300                          break;
 301  
 302                      case 'viewall':
 303                          // User can only ask for view perm
 304                          if(( $permlevel == 'view' ) || ( $permlevel == 'any' ))
 305                          { // Permission granted
 306                              $perm = true;
 307                              break;
 308                          }
 309                  }
 310  
 311                  if( ! $perm && ( $permlevel == 'create' ) && $this->check_perm( 'perm_createblog', 'allowed' ) )
 312                  { // User is allowed to create a blog (for himself)
 313                      $perm = true;
 314                  }
 315                  break;
 316  
 317              case 'stats':
 318                  if( ! $this->check_perm( 'admin', 'restricted' ) )
 319                  {
 320                      $perm = false;
 321                      break;
 322                  }
 323                  switch( $permvalue )
 324                  { // Depending on current group permission:
 325  
 326                      case 'edit':
 327                          // All permissions granted
 328                          $perm = true;
 329                          break;
 330  
 331                      case 'view':
 332                          // User can ask for view perm...
 333                          if( $permlevel == 'view' )
 334                          {
 335                              $perm = true;
 336                              break;
 337                          }
 338                          // ... or for any lower priority perm... (no break)
 339  
 340                      case 'user':
 341                          // This is for stats. User perm can grant permissions in the User class
 342                          // Here it will just allow to list
 343                      case 'list':
 344                          // User can only ask for list perm
 345                          if( $permlevel == 'list' )
 346                          {
 347                              $perm = true;
 348                              break;
 349                          }
 350                  }
 351                  break;
 352  
 353              case 'perm_files':
 354                  if( ! $this->check_perm( 'admin', 'restricted' ) )
 355                  {
 356                      $perm = false;
 357                      break;
 358                  }
 359                  // no break, perm_files is pluggable permission
 360  
 361              default:
 362  
 363                  // Check pluggable permissions using group permission check function
 364                  $perm = Module::check_perm( $permname, $permlevel, $perm_target, 'group_func', $this );
 365                  if( $perm === NULL )
 366                  {    // Even if group permisson check function doesn't exist we should return false value
 367                      $perm = false;
 368                  }
 369  
 370                  break;
 371          }
 372  
 373          $target_ID = $perm_target;
 374          if( is_object($perm_target) ) $target_ID = $perm_target->ID;
 375  
 376          $Debuglog->add( "Group perm $permname:$permlevel:$target_ID => ".($perm?'granted':'DENIED'), 'perms' );
 377  
 378          return $perm;
 379      }

 382      /**
 383       * Check permission for this group on a specified blog
 384       *
 385       * This is not for direct use, please call {@link User::check_perm()} instead
 386       * user is checked for privileges first, group lookup only performed on a false result
 387       *
 388       * @see User::check_perm()
 389       * @param string Permission name can be any from the blog advanced perm names. A few possible permname:
 390       *                  - blog_ismember
 391       *                  - blog_can_be_assignee
 392       *                  - blog_del_post
 393       *                  - blog_edit_ts
 394       *                  - blog_post_statuses
 395       *                  - blog_edit
 396       *                  - blog_comment_statuses
 397       *                  - blog_edit_cmt
 398       *                  - blog_cats
 399       *                  - blog_properties
 400       * @param string Permission level
 401       * @param integer Permission target blog ID
 402       * @param Item post that we want to edit
 403       * @param User for who we would like to check this permission
 404       * @return boolean 0 if permission denied
 405       */
 406  	function check_perm_bloggroups( $permname, $permlevel, $perm_target_blog, $perm_target = NULL, $User = NULL )
 407      {
 408          if( !isset( $this->blog_post_statuses[$perm_target_blog] ) )
 409          {
 410              $this->blog_post_statuses[$perm_target_blog] = array();
 411              if( ! load_blog_advanced_perms( $this->blog_post_statuses[$perm_target_blog], $perm_target_blog, $this->ID, 'bloggroup' ) )
 412              { // Could not load blog advanced user perms
 413                  return false;
 414              }
 415          }
 416  
 417          $blog_perms = $this->blog_post_statuses[$perm_target_blog];
 418          if( empty( $User ) )
 419          { // User is not set
 420              $user_ID = NULL;
 421          }
 422          else
 423          { // User is set, advanced user perms must be loaded
 424              $user_ID = $User->ID;
 425              if( isset( $User->blog_post_statuses[$perm_target_blog] ) )
 426              { // Merge user advanced perms with group advanced perms
 427                  $edit_perms = array( 'no' => 0, 'own' => 1, 'anon' => 2, 'lt' => 3, 'le' => 4, 'all' => 5 );
 428                  foreach( $User->blog_post_statuses[$perm_target_blog] as $key => $value )
 429                  { // For each collection advanced permission use the higher perm value between user and group perms
 430                      if( ( $key == 'blog_edit' ) || ( $key == 'blog_edit_cmt' ) )
 431                      {
 432                          if( $edit_perms[$value] > $edit_perms[$blog_perms[$key]] )
 433                          { // Use collection user edit permission because it is greater than the collection group perm
 434                              $blog_perms[$key] = $value;
 435                          }
 436                      }
 437                      elseif( isset( $blog_perms[$key] ) )
 438                      { // Check user and group perm as well
 439                          $blog_perms[$key] = (int) $value | (int) $blog_perms[$key];
 440                      }
 441                  }
 442              }
 443          }
 444          return check_blog_advanced_perm( $blog_perms, $user_ID, $permname, $permlevel, $perm_target );
 445      }
