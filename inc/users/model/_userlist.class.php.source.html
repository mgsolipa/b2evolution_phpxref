 428  	function query_init()
 429      {
 430          global $current_User;
 431  
 432          if( empty( $this->filters ) )
 433          {    // Filters have not been set before, we'll use the default filterset:
 434              // If there is a preset filter, we need to activate its specific defaults:
 435              $this->filters['filter_preset'] = param( 'filter_preset', 'string', $this->default_filters['filter_preset'], true );
 436              $this->activate_preset_filters();
 437  
 438              // Use the default filters:
 439              $this->set_filters( $this->default_filters );
 440          }
 441  
 442          // GENERATE THE QUERY:
 443  
 444          // The SQL Query object:
 445          // If group == -1 we shouldn't group list by user group
 446          $this->query_params['grouped'] = ( $this->filters['group'] != -1 );
 447          $this->UserQuery = new UserQuery( $this->Cache->dbtablename, $this->Cache->dbprefix, $this->Cache->dbIDname, $this->query_params );
 448  
 449          if( isset( $this->query_params['keywords_fields'] ) )
 450          { // Change keywords_fields from query params
 451              $this->UserQuery->keywords_fields = $this->query_params['keywords_fields'];
 452          }
 453  
 454          if( isset( $this->query_params['where_status_closed'] ) )
 455          { // Limit by closed users
 456              $this->UserQuery->where_status( 'closed', $this->query_params['where_status_closed'] );
 457          }
 458  
 459          // If browse users from different countries is restricted, then always filter to the current User country
 460          if( has_cross_country_restriction( 'users', 'list' ) )
 461          { // Browse users from different countries is restricted, filter to current user country
 462              $ctry_filter = $current_User->ctry_ID;
 463              // if country filtering was changed the qurey must be refreshed
 464              $this->refresh_query = ( $this->refresh_query || ( $ctry_filter != $this->filters['country'] ) );
 465          }
 466          else
 467          { // Browse users from different countries is allowed
 468              $ctry_filter = $this->filters['country'];
 469          }
 470  
 471          /*
 472           * filtering stuff:
 473           */
 474          $this->UserQuery->where_members( $this->filters['membersonly'] );
 475          $this->UserQuery->where_keywords( $this->filters['keywords'] );
 476          $this->UserQuery->where_gender( $this->filters['gender'] );
 477          $this->UserQuery->where_status( $this->filters['status_activated'] );
 478          $this->UserQuery->where_status( $this->filters['account_status'], true, true );
 479          $this->UserQuery->where_reported( $this->filters['reported'] );
 480          $this->UserQuery->where_custom_sender( $this->filters['custom_sender_email'], $this->filters['custom_sender_name'] );
 481          $this->UserQuery->where_group( $this->filters['group'] );
 482          $this->UserQuery->where_location( 'ctry', $ctry_filter );
 483          $this->UserQuery->where_location( 'rgn', $this->filters['region'] );
 484          $this->UserQuery->where_location( 'subrg', $this->filters['subregion'] );
 485          $this->UserQuery->where_location( 'city', $this->filters['city'] );
 486          $this->UserQuery->where_age_group( $this->filters['age_min'], $this->filters['age_max'] );
 487          $this->UserQuery->where_userfields( $this->filters['userfields'] );
 488          $this->UserQuery->where_level( $this->filters['level_min'], $this->filters['level_max'] );
 489          if( isset( $this->filters['org'] ) || isset( $this->query_params['where_org_ID'] ) )
 490          { // Filter by organization ID
 491              $org_ID = isset( $this->query_params['where_org_ID'] ) ? $this->query_params['where_org_ID'] : $this->filters['org'];
 492              $this->UserQuery->where_organization( $org_ID );
 493          }
 494          if( ! is_logged_in() )
 495          { // Restrict users by group level for anonymous users
 496              global $Settings;
 497              $this->UserQuery->where_group_level( $Settings->get('allow_anonymous_user_level_min'), $Settings->get('allow_anonymous_user_level_max') );
 498          }
 499  
 500          if( $this->get_order_field_list() != '' )
 501          {
 502              $this->UserQuery->order_by( str_replace( '*', $this->get_order_field_list(), $this->UserQuery->order_by ) );
 503          }
 504          else
 505          {
 506              $this->UserQuery->order_by( $this->get_order_field_list() );
 507          }
 508      }

 514  	function query()
 515      {
 516          global $DB, $Session, $localtimenow;
 517  
 518          if( !is_null( $this->rows ) )
 519          { // Query has already executed:
 520              return;
 521          }
 522  
 523          // INIT THE QUERY:
 524          $this->query_init();
 525  
 526  
 527          // We are going to proceed in two steps (we simulate a subquery)
 528          // 1) we get the IDs we need
 529          // 2) we get all the other fields matching these IDs
 530          // This is more efficient than manipulating all fields at once.
 531  
 532          // *** STEP 1 ***
 533          $user_IDs = isset( $this->filters['users'] ) ? $this->filters['users'] : array();
 534          if( $this->refresh_query || // Some filters are changed
 535              $localtimenow - $Session->get( $this->filterset_name.'_refresh_time' ) > 7200 ) // Time has passed ( 2 hours )
 536          {    // We should create new list of user IDs
 537              global $Timer;
 538              $Timer->start( 'Users_IDs', false );
 539  
 540              $step1_SQL = new SQL();
 541              $step1_SQL->SELECT( 'T_users.user_ID, IF( user_avatar_file_ID IS NOT NULL, 1, 0 ) as has_picture, COUNT( DISTINCT blog_ID ) AS nb_blogs' );
 542              if( !empty( $this->filters['reported'] ) && $this->filters['reported'] )
 543              {    // Filter is set to 'Reported users'
 544                  $step1_SQL->SELECT_add( ', COUNT( DISTINCT urep_reporter_ID ) AS user_rep' );
 545              }
 546              $step1_SQL->FROM( $this->UserQuery->get_from( '' ) );
 547              $step1_SQL->WHERE( $this->UserQuery->get_where( '' ) );
 548              $step1_SQL->GROUP_BY( $this->UserQuery->get_group_by( '' ) );
 549              $step1_SQL->ORDER_BY( $this->UserQuery->get_order_by( '' ) );
 550              $step1_SQL->LIMIT( 0 );
 551  
 552              // Get list of the IDs we need:
 553              $user_IDs = $DB->get_col( $step1_SQL->get(), 0, 'UserList::Query() Step 1: Get ID list' );
 554              // Update filter with user IDs
 555              $this->filters['users'] = $user_IDs;
 556              $this->save_filterset();
 557  
 558              $Timer->stop( 'Users_IDs' );
 559          }
 560  
 561          // GET TOTAL ROW COUNT:
 562          parent::count_total_rows( count( $user_IDs ) );
 563  
 564          // Pagination, Get user IDs from array for current page
 565          $user_IDs_paged = array_slice( $user_IDs, ( $this->page - 1 ) * $this->limit, $this->limit );
 566  
 567          // *** STEP 2 ***
 568          $step2_SQL = $this->UserQuery;
 569

 736       *
 737       * @param integer the currently selected user ID ( Note: it must be set only if we would like to skip some users from the list )
 738       * @param string prev | next  (relative to the current sort order)
 739       */
 740      function & get_prevnext_User( $direction = 'next', $selected_user_ID = NULL )
 741      {
 742          $users_list = $this->filters['users'];
 743  
 744          if( count( $users_list ) < 2 )
 745          {    // Short users list
 746              $r = NULL;
 747              return $r;
 748          }
 749  
 750          // ID of selected user
 751          if( $selected_user_ID === NULL )
 752          { // get currently selected user ID from param
 753              $selected_user_ID = get_param( 'user_ID' );
 754          }
 755  
 756          $user_key = array_search( $selected_user_ID, $users_list );
 757          if( is_int( $user_key ) )
 758          {    // Selected user is located in the list
 759              $prevnext_key = $direction == 'next' ? $user_key + 1 : $user_key - 1;
 760              if( isset( $users_list[$prevnext_key] ) )
 761              {    // Prev/next user is located in the list
 762                  $prevnext_ID = $users_list[$prevnext_key];
 763              }
 764          }
 765  
 766          if( empty( $prevnext_ID ) )
 767          {    // No prev/next user
 768              $r = NULL;
 769              return $r;
 770          }
 771  
 772          $UserCache = & get_UserCache();
 773          $User = & $UserCache->get_by_ID( $prevnext_ID, false, false );

 779      /**
 780       * Set an order of a list (Use this function after when all $this->cols are already defined)
 781       *
 782       * @param string Field name
 783       * @param string Order direction (A|D)
 784       * @param boolean Save the filters from Session
 785       */
 786  	function set_order( $order_field, $direction = 'D', $save_filters = false )
 787      {
 788          global $Session;
 789  
 790          if( empty( $this->cols ) )
 791          { // The columns are not defined yet, Exit here
 792              return;
 793          }
 794  
 795          // Force filter param to reset the previous filters
 796          set_param( 'filter', 'new' );
 797          $this->refresh_query = true;
 798  
 799          foreach( $this->cols as $col_num => $col )
 800          {    // Find a column number
 801              if( $col['order'] == $order_field )
 802              {
 803                  break;
 804              }
 805          }
 806  
 807          if( $save_filters )
 808          { // Get the filters from Session
 809              $this->filters = $Session->get( $this->filterset_name );
 810              if( ! is_array( $this->filters ) )
 811              {
 812                  $this->filters = array();
 813              }
 814              $this->filters = array_merge( $this->default_filters, $this->filters );
 815          }
 816          else
 817          { // Reset the filters
 818              $this->filters = array();
 819          }
 820  
 821          // Rewrite a previous order to new value
 822          $this->filters['order'] = str_repeat( '-', $col_num ).$direction;
 823          $this->order = $this->filters['order'];
 824  
 825          // Save a new order
 826          $Session->set( $this->filterset_name, $this->filters );
 827          $this->save_filterset();
 828      }
