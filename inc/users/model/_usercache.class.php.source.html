  75      /**
  76       * Get a user object by login.
  77       *
  78       * Does not halt on error.
  79       *
  80       * @param string user login
  81       * @param boolean force to run db query to get user by the given login.
  82       *        !IMPORTANT! Set this to false only if it's sure that this user was already loaded if it exists on DB!
  83       *
  84       * @return false|User Reference to the user object or false if not found
  85       */
  86      function & get_by_login( $login, $force_db_check = true )
  87      {
  88          // Make sure we have a lowercase login:
  89          // We want all logins to be lowercase to guarantee uniqueness regardless of the database case handling for UNIQUE indexes.
  90          $login = utf8_strtolower( $login );
  91  
  92          if( !( $force_db_check || isset( $this->cache_login[$login] ) ) )
  93          { // force db check is false and this login is not set in the cache it means that user with the given login doesn't exists
  94              $this->cache_login[$login] = false;
  95          }
  96  
  97          if( !isset( $this->cache_login[$login] ) )
  98          {
  99              global $DB;
 100  
 101              if( $row = $DB->get_row( "
 102                      SELECT *
 103                        FROM T_users
 104                       WHERE user_login = '".$DB->escape($login)."'", 0, 0, 'Get User login' ) )
 105              {
 106                  $this->add( new User( $row ) );
 107              }
 108              else
 109              {
 110                  $this->cache_login[$login] = false;
 111              }
 112          }
 113  
 114          return $this->cache_login[$login];
 115      }

 119       * Get a user object by login, only if password matches.
 120       *
 121       * @param string Login
 122       * @param string Password
 123       * @param boolean Password is MD5()'ed
 124       * @return false|User

 126      function & get_by_loginAndPwd( $login, $pass, $pass_is_md5 = true )
 127      {
 128          if( !($User =& $this->get_by_login( $login )) )
 129          {
 130              return false;

 147      /**
 148       * Get a user object by email and password
 149       * If multiple accounts match, give priority to:
 150       *  -accounts that are activated over non activated accounts
 151       *  -accounts that were used more recently than others
 152       *
 153       * @param string email address
 154       * @param string password
 155       * @param array hashed password - If this is set, it means we need to check the hasshed password instead of the md5 password
 156       * @param string current session password salt
 157       * @return false|array false if user with this email not exists, array( $User, $exists_more ) pair otherwise
 158       */
 159  	function get_by_emailAndPwd( $email, $pass, $pwd_hashed = NULL, $pwd_salt = NULL )
 160      {
 161          global $DB;
 162  
 163          // Get all users with matching email address
 164          $result = $DB->get_results('SELECT * FROM T_users
 165                      WHERE user_email = '.$DB->quote( utf8_strtolower( $email ) ).'
 166                      ORDER BY user_lastseen_ts DESC, user_status ASC');
 167  
 168          if( empty( $result ) )
 169          { // user was not found with the given email address
 170              return false;
 171          }
 172  
 173          // check if exists more user with the same email address
 174          $exists_more = ( count( $result ) > 1 );
 175          $index = -1;
 176          $first_matched_index = false;
 177          // iterate through the result list
 178          foreach( $result as $row )
 179          {
 180              $index++;
 181              if( empty( $pwd_hashed ) )
 182              {
 183                  if( $row->user_pass != md5( $row->user_salt.$pass ) )
 184                  { // password doesn't match
 185                      continue;
 186                  }
 187              }
 188              else
 189              {
 190                  $pwd_matched = false;
 191                  foreach( $pwd_hashed as $encrypted_password )
 192                  {
 193                      $pwd_matched = ( sha1($row->user_pass.$pwd_salt) == $encrypted_password );
 194                      if( $pwd_matched )
 195                      { // The corresponding user was found
 196                          break;
 197                      }
 198                  }
 199                  if( ! $pwd_matched )
 200                  { // If the user with matched login credentials was not fount continue with the next row
 201                      continue;
 202                  }
 203              }
 204              // a user with matched password was found
 205              $first_matched_index = $index;
 206              if( ( $row->user_status == 'activated' ) || ( $row->user_status == 'autoactivated' ) )
 207              { // an activated user was found, break from the iteration
 208                  $User = new User( $row );
 209                  break;
 210              }
 211              if( ( !isset( $first_notclosed_User ) ) && ( $row->user_status != 'closed' ) )
 212              {
 213                  $first_notclosed_User = new User( $row );
 214              }
 215          }
 216  
 217          if( !isset( $User ) )
 218          { // There is no activated user with the given email and password
 219              if( isset( $first_notclosed_User ) )
 220              { // Get first not closed user with the given email and password
 221                  $User = $first_notclosed_User;
 222              }
 223              elseif( $first_matched_index !== false )
 224              { // There is only closed user with the given email and password
 225                  $User = new User( $result[$first_matched_index] );
 226              }
 227              else
 228              { // No matched user was found
 229                  return false;
 230              }
 231          }
 232  
 233          // Add user to the cache and return result
 234          $this->add( $User );
 235          return array( & $User, $exists_more );
 236      }

 258      /**
 259       * Load members of a given blog
 260       *
 261       * @todo make a UNION query when we upgrade to MySQL 4
 262       * @param integer Blog ID to load members for
 263       * @param integer Limit, 0 - for unlimit
 264       * @param boolean TRUE - to load only members, FALSE - to load the users which can be assigned to items if it is allowed by blog's settings
 265       */
 266  	function load_blogmembers( $blog_ID, $limit = 0, $load_members = true )
 267      {
 268          global $DB, $Debuglog;
 269  
 270          $load_assignees = false;
 271          if( ! $load_members && ! empty( $blog_ID ) )
 272          {
 273              $BlogCache = & get_BlogCache();
 274              if( $Blog = $BlogCache->get_by_ID( $blog_ID, false, false ) )
 275              { // Load assignees only when advanced perms are available and workflow is used
 276                  $load_assignees = $Blog->get( 'advanced_perms' ) && $Blog->get_setting( 'use_workflow' );
 277              }
 278          }
 279          if( $load_assignees )
 280          { // Load users which can be assigneed to items of the blog
 281              $cache_name = 'blogassignees';
 282              $db_field = 'can_be_assignee';
 283          }
 284          else
 285          { // Load members of the blog
 286              $cache_name = 'blogmembers';
 287              $db_field = 'ismember';
 288          }
 289  
 290          if( isset( $this->alreadyCached[ $cache_name ] ) && isset( $this->alreadyCached[ $cache_name ][ $blog_ID ] ) )
 291          {
 292              $Debuglog->add( "Already loaded <strong>$this->objtype(Blog #$blog_ID members)</strong> into cache", 'dataobjects' );
 293              return false;
 294          }
 295  
 296          $this->none_option_text = NT_('No user');
 297  
 298          // Clear previous users to get only the members of this blog
 299          $this->clear();
 300  
 301          // Remember this special load:
 302          $this->alreadyCached[ $cache_name ][ $blog_ID ] = true;
 303  
 304          $Debuglog->add( "Loading <strong>$this->objtype(Blog #$blog_ID members)</strong> into cache", 'dataobjects' );
 305  
 306          // Get users which are members of the blog:
 307          $user_perms_SQL = new SQL();
 308          $user_perms_SQL->SELECT( 'T_users.*' );
 309          $user_perms_SQL->FROM( 'T_users' );
 310          $user_perms_SQL->FROM_add( 'INNER JOIN T_coll_user_perms ON user_ID = bloguser_user_ID' );
 311          $user_perms_SQL->WHERE( 'bloguser_blog_ID = '.$DB->quote( $blog_ID ) );
 312          $user_perms_SQL->WHERE_and( 'bloguser_'.$db_field.' <> 0' );
 313  
 314          // Get users which groups are members of the blog:
 315          $group_perms_SQL = new SQL();
 316          $group_perms_SQL->SELECT( 'T_users.*' );
 317          $group_perms_SQL->FROM( 'T_users' );
 318          $group_perms_SQL->FROM_add( 'INNER JOIN T_coll_group_perms ON user_grp_ID = bloggroup_group_ID' );
 319          $group_perms_SQL->WHERE( 'bloggroup_blog_ID = '.$DB->quote( $blog_ID ) );
 320          $group_perms_SQL->WHERE_and( 'bloggroup_'.$db_field.' <> 0' );
 321  
 322          // Union two sql queries to execute one query and save an order as one list
 323          $users_sql = '( '.$user_perms_SQL->get().' )'
 324              .' UNION '
 325              .'( '.$group_perms_SQL->get().' )'
 326              .' ORDER BY user_login';
 327          if( $limit > 0 )
 328          { // Limit the users
 329              $users_sql .= ' LIMIT '.$limit;
 330          }
 331  
 332          $users = $DB->get_results( $users_sql );
 333          foreach( $users as $row )
 334          {
 335              if( !isset($this->cache[$row->user_ID]) )
 336              { // Save reinstatiating User if it's already been added
 337                  $this->add( new User( $row ) );
 338              }
 339          }
 340  
 341          return true;
 342      }
