 148      /**
 149       * Get a setting from the DB user settings table
 150       *
 151       * @param string name of setting
 152       * @param integer User ID (by default $current_User->ID will be used)
 153       */
 154  	function get( $setting, $user_ID = NULL )
 155      {
 156          global $Settings;
 157  
 158          if( ! isset($user_ID) )
 159          {
 160              global $current_User;
 161  
 162              if( ! isset($current_User) )
 163              { // no current/logged in user:
 164                  $result = $this->get_default($setting);
 165                  if( $result == NULL )
 166                  {
 167                      $result = $Settings->get( 'def_'.$setting );
 168                  }
 169                  return $result;
 170              }
 171  
 172              $user_ID = $current_User->ID;
 173          }
 174  
 175          $result = parent::get( $user_ID, $setting );
 176          if( $result == NULL )
 177          {
 178              $result = $Settings->get( 'def_'.$setting );
 179          }
 180          return $result;
 181      }

 233      /**
 234       * Get a param from Request and save it to UserSettings, or default to previously saved user setting.
 235       *
 236       * If the user setting was not set before (and there's no default given that gets returned), $default gets used.
 237       *
 238       * @todo Move this to _abstractsettings.class.php - the other Settings object can also make use of it!
 239       *
 240       * @param string Request param name
 241       * @param string User setting name. Make sure this is unique!
 242       * @param string Force value type to one of:
 243       * - integer
 244       * - float
 245       * - string (strips (HTML-)Tags, trims whitespace)
 246       * - array
 247       * - object
 248       * - null
 249       * - html (does nothing)
 250       * - '' (does nothing)
 251       * - '/^...$/' check regexp pattern match (string)
 252       * - boolean (will force type to boolean, but you can't use 'true' as a default since it has special meaning. There is no real reason to pass booleans on a URL though. Passing 0 and 1 as integers seems to be best practice).
 253       * Value type will be forced only if resulting value (probably from default then) is !== NULL
 254       * @param mixed Default value or TRUE if user input required
 255       * @param boolean Do we need to memorize this to regenerate the URL for this page?
 256       * @param boolean Override if variable already set
 257       * @return NULL|mixed NULL, if neither a param was given nor {@link $UserSettings} knows about it.
 258       */
 259  	function param_Request( $param_name, $uset_name, $type = '', $default = '', $memorize = false, $override = false ) // we do not force setting it..
 260      {
 261          $value = param( $param_name, $type, NULL, $memorize, $override, false ); // we pass NULL here, to see if it got set at all
 262  
 263          if( $value !== false )
 264          { // we got a value
 265              $this->set( $uset_name, $value );
 266              $this->dbupdate();
 267          }
 268          else
 269          { // get the value from user settings
 270              $value = $this->get($uset_name);
 271  
 272              if( is_null($value) )
 273              { // it's not saved yet and there's not default defined ($_defaults)
 274                  $value = $default;
 275              }
 276              if( $memorize )
 277              { // Memorize param
 278                  memorize_param( $param_name, $type, $default, $value );
 279              }
 280          }
 281  
 282          set_param( $param_name, $value );
 283          return get_param($param_name);
 284      }

 293  	function reset_to_defaults( $user_ID, $db_save = true )
 294      {
 295          // Remove all UserSettings where a default or configurable default exists:
 296          foreach( $this->_defaults as $k => $v )
 297          {
 298              $this->delete( $k, $user_ID );
 299          }
 300          foreach( $this->_configurable_defaults as $k => $v )
 301          {
 302              $this->delete( $k, $user_ID );
 303          }
 304      }

 307      /** Get user setting per collection
 308       *
 309       * @param string name of setting
 310       * @param integer Collection ID (by default global $Blog->ID will be used)
 311       * @param integer User ID (by default $current_User->ID will be used)
 312       */
 313  	function get_collection_setting( $setting, $coll_ID = NULL, $user_ID = NULL )
 314      {
 315          if( $coll_ID === NULL )
 316          { // Use current blog ID by default
 317              global $Blog;
 318  
 319              if( ! empty( $Blog ) )
 320              {
 321                  $coll_ID = $Blog->ID;
 322              }
 323          }
 324  
 325          if( $coll_ID === NULL )
 326          { // Collection is not detected
 327              return NULL;
 328          }
 329  
 330          // Try to get user-collection setting from DB
 331          $value = $this->get( $setting.'_'.$coll_ID, $user_ID );
 332  
 333          if( $value === NULL )
 334          { // The user-collection setting is not defined in DB
 335              // Try to get a default value for this setting:
 336              if( isset( $this->_defaults[ $setting ] ) )
 337              { // Default value is defined
 338                  $value = $this->_defaults[ $setting ];
 339              }
 340              else
 341              { // No default value
 342                  $value = NULL;
 343              }
 344          }
 345  
 346          return $value;
 347      }
