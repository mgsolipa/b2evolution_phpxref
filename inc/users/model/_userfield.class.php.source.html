 172      /**
 173       * Load data from Request form fields.
 174       *
 175       * @return boolean true if loaded data seems valid.
 176       */
 177  	function load_from_Request()
 178      {
 179          // Group
 180          $old_group_ID = $this->ufgp_ID; // Save old group ID to know if it was changed
 181          param_string_not_empty( 'ufdf_ufgp_ID', T_('Please select a group.') );
 182          $this->set_from_Request( 'ufgp_ID' );
 183  
 184          // Type
 185          param_string_not_empty( 'ufdf_type', T_('Please enter a type.') );
 186          $this->set_from_Request( 'type' );
 187  
 188          // Code
 189          $code = param( 'ufdf_code', 'string' );
 190          param_check_not_empty( 'ufdf_code', T_('Please provide a code to uniquely identify this field.') );
 191          param_check_regexp( 'ufdf_code', '#^[a-z0-9_]{1,20}$#', T_('The field code must contain only lowercase letters, digits or the "_" sign. 20 characters max.') );
 192          $this->set_from_Request( 'code' );
 193  
 194          // Name
 195          param_string_not_empty( 'ufdf_name', T_('Please enter a name.') );
 196          $this->set_from_Request( 'name' );
 197  
 198          // Icon name
 199          param( 'ufdf_icon_name', 'string' );
 200          $this->set_from_Request( 'icon_name', 'ufdf_icon_name', true );
 201  
 202          // Options
 203          if( param( 'ufdf_type', 'string' ) == 'list' )
 204          { // Save 'Options' only for Field type == 'Option list'
 205              $ufdf_options = param( 'ufdf_options', 'text' );
 206              if( count( explode( "\n", $ufdf_options ) ) < 2 )
 207              { // We don't want save an option list with one item
 208                  param_error( 'ufdf_options', T_('Please enter at least 2 options on 2 different lines.') );
 209              }
 210              elseif( utf8_strlen( $ufdf_options ) > 255 )
 211              { // This may not happen in normal circumstances because the textarea max length is set to 255 chars
 212                  // This extra check is for the case if js is not enabled or someone would try to directly edit the html
 213                  param_error( 'ufdf_options', T_('"Options" field content can not be longer than 255 symbols.') );
 214              }
 215              $this->set( 'options', $ufdf_options );
 216          }
 217  
 218          // Required
 219          param_string_not_empty( 'ufdf_required', 'Please select Hidden, Optional, Recommended or Required.' );
 220          $this->set_from_Request( 'required' );
 221  
 222          // Duplicated
 223          param_string_not_empty( 'ufdf_duplicated', 'Please select Forbidden, Allowed or List style.' );
 224          $this->set_from_Request( 'duplicated' );
 225  
 226          // Order
 227          if( $old_group_ID != $this->ufgp_ID )
 228          { // Group is changing, set order as last
 229              $this->set( 'order', $this->get_last_order( $this->ufgp_ID ) );
 230          }
 231  
 232          // Suggest
 233          if( param( 'ufdf_type', 'string' ) == 'word' )
 234          { // Save 'Suggest values' only for Field type == 'Single word'
 235              param( 'ufdf_suggest', 'integer', 0 );
 236              $this->set_from_Request( 'suggest' );
 237          }
 238  
 239          // Bubbletip
 240          param( 'ufdf_bubbletip', 'text', '' );
 241          $this->set_from_Request( 'bubbletip', NULL, true );
 242  
 243          if( ! param_errors_detected() )
 244          { // Field code must be unique, Check it only when no errors on the form
 245              if( $field_ID = $this->dbexists( 'ufdf_code', $this->get( 'code' ) ) )
 246              { // We have a duplicate entry:
 247                  param_error( 'ufdf_code',
 248                      sprintf( T_('Another user field already uses this code. Do you want to <a %s>edit the existing user field</a>?'),
 249                          'href="?ctrl=userfields&amp;action=edit&amp;ufdf_ID='.$field_ID.'"' ) );
 250              }
 251          }
 252  
 253          return ! param_errors_detected();
 254      }
