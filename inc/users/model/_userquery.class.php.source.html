 141          $usergroups_SQL->FROM( 'T_users' );
 142          $usergroups_SQL->FROM_add( 'INNER JOIN T_groups ON grp_ID = user_grp_ID' );

 158       * Restrict with keywords
 159       *
 160       * @param string Keyword search string
 161       */
 162  	function where_keywords( $keywords )
 163      {
 164          global $DB;
 165  
 166          if( empty( $keywords ) )
 167          {
 168              return;
 169          }
 170  
 171          $search = array();
 172  
 173          $kw_array = explode( ' ', $keywords );
 174          foreach( $kw_array as $kw )
 175          {
 176              // Note: we use CONCAT_WS (Concat With Separator) because CONCAT returns NULL if any arg is NULL
 177              $search[] = 'CONCAT_WS( " ", '.$this->keywords_fields.' ) LIKE "%'.$DB->escape($kw).'%"';
 178          }
 179  
 180          if( count( $search ) > 0 )
 181          {
 182              $this->WHERE_and( implode( ' AND ', $search ) );
 183          }
 184      }

 214       *
 215       * @param string user status ( 'activated', 'deactivated', 'new', 'emailchanged', 'failedactivation', 'closed' )
 216       * @param boolean set true to include users only with the given status, or set false to exclude users with the given status
 217       * @param boolean set true to make exact comparing with selected status
 218       */
 219  	function where_status( $status, $include = true, $exactly = false )
 220      {
 221          global $DB;
 222  
 223          if( empty( $status ) )
 224          {
 225              return;
 226          }
 227  
 228          if( $status == 'activated' && !$exactly )
 229          { // Activated and Autoactivated users
 230              if( $include )
 231              {
 232                  $this->WHERE_and( 'user_status = '.$DB->quote( 'activated' ).' OR user_status = '.$DB->quote( 'autoactivated' ) );
 233              }
 234              else
 235              {
 236                  $this->WHERE_and( 'user_status <> '.$DB->quote( 'activated' ).' AND user_status <> '.$DB->quote( 'autoactivated' ) );
 237              }
 238          }
 239          else
 240          { // Other status check
 241              // init compare, which depends if we want to include or exclude users with the given status
 242              $compare = $include ? ' = ' : ' <> ';
 243              $this->WHERE_and( 'user_status'.$compare.$DB->quote( $status ) );
 244          }
 245  
 246          return;

 365  	function where_userfields( $userfields )
 366      {
 367          global $DB;
 368  
 369          if( empty( $userfields ) )
 370          {
 371              return;
 372          }
 373  
 374          $criteria_where_clauses = array();
 375          foreach( $userfields as $field )
 376          {
 377              $type = (int)$field['type'];
 378              $value = trim( strip_tags( $field['value'] ) );
 379              if( $type > 0 && $value != '' )
 380              {    // Filter by Specific criteria
 381                  $words = explode( ' ', $value );
 382                  if( count( $words ) > 0 )
 383                  {
 384                      foreach( $words as $word )
 385                      {
 386                          $criteria_where_clauses[] = 'uf_ufdf_ID = "'.$DB->escape($type).'" AND uf_varchar LIKE "%'.$DB->escape($word).'%"';
 387                      }
 388                  }
 389              }
 390          }
 391  
 392          if( count( $criteria_where_clauses ) > 0 )
 393          {    // Some creteria is defined
 394              $this->FROM_add( ' LEFT JOIN T_users__fields ON uf_user_ID = user_ID' );
 395              $this->WHERE_and( ' ( ( '.implode( ' ) OR ( ', $criteria_where_clauses ).' ) ) ' );
 396          }
 397      }
