  38   *
  39   * Insert a new abuse string into DB
  40   */
  41  function antispam_create( $abuse_string, $aspm_source = 'local' )
  42  {
  43      global $DB;
  44  
  45      // Cut the crap if the string is empty:
  46      $abuse_string = trim( $abuse_string );
  47      if( empty( $abuse_string ) )
  48      {
  49          return false;
  50      }
  51  
  52      // Check if the string already is in the blacklist:
  53      if( antispam_check($abuse_string) )
  54      {
  55          return false;
  56      }
  57  
  58      // Insert new string into DB:
  59      $sql = "INSERT INTO T_antispam( aspm_string, aspm_source )
  60                      VALUES( '".$DB->escape($abuse_string)."', '$aspm_source' )";
  61      $DB->query( $sql );
  62  
  63      return true;
  64  }

  98  /**
  99   * Check if a string contains abusive substrings
 100   *
 101   * Note: Letting the database do the LIKE %% match is a little faster than doing in it PHP,
 102   * not to mention the incredibly long overhead of preloading the list into PHP
 103   *
 104   * @todo dh> IMHO this method is too generic used! It gets used for:
 105   *           - comment author name
 106   *           - comment/message author email
 107   *           - comment content
 108   *           - message (email) content
 109   *           - validate_url
 110   *           ..and validates all this against the antispam blacklist!
 111   *           We should rather differentiate here more and make it pluggable!
 112   *
 113   * @return string blacklisted keyword found or false if no spam detected
 114   */
 115  function antispam_check( $haystack )
 116  {
 117      global $DB, $Debuglog, $Timer;
 118  
 119      // TODO: 'SELECT COUNT(*) FROM T_antispam WHERE aspm_string LIKE "%'.$url.'%" ?
 120  
 121      $Timer->resume( 'antispam_url' ); // resuming to get the total number..
 122      $block = $DB->get_var(
 123          'SELECT aspm_string
 124             FROM  T_antispam
 125            WHERE '.$DB->quote($haystack).' LIKE CONCAT("%",aspm_string,"%")
 126            LIMIT 0, 1', 0, 0, 'Check URL against antispam blacklist' );
 127      if( $block )
 128      {
 129              $Debuglog->add( 'Spam block: '.$block );
 130              return $block;    // SPAM detected!
 131      }
 132      $Timer->pause( 'antispam_url' );
 133  
 134      return false;    // no problem.
 135  }

 146  function antispam_report_abuse( $abuse_string )
 147  {
 148      global $debug, $antispamsrv_host, $antispamsrv_port, $antispamsrv_uri, $antispam_test_for_real;
 149      global $baseurl, $Messages, $Settings;
 150      global $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password;
 151  
 152      if( ! $Settings->get('antispam_report_to_central') )
 153      {
 154          $Messages->add( 'Reporting is disabled.', 'error' );  // NO TRANS necessary
 155          return false;
 156      }
 157  
 158      if( preg_match( '#^http://localhost[/:]#', $baseurl) && ( $antispamsrv_host != 'localhost' ) && empty( $antispam_test_for_real )  )
 159      { // Local install can only report to local test server
 160          $Messages->add( T_('Reporting abuse to b2evolution aborted (Running on localhost).'), 'error' );
 161          return false;
 162      }
 163  
 164      // Construct XML-RPC client:
 165      load_funcs( 'xmlrpc/model/_xmlrpc.funcs.php' );
 166      $client = new xmlrpc_client( $antispamsrv_uri, $antispamsrv_host, $antispamsrv_port );
 167      // yura: I commented this because xmlrpc_client prints the debug info on screen and it breaks header_redirect()
 168      // $client->debug = $debug;
 169  
 170      // Set proxy for outgoing connections:
 171      if( ! empty( $outgoing_proxy_hostname ) )
 172      {
 173          $client->setProxy( $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password );
 174      }
 175  
 176      // Construct XML-RPC message:
 177      $message = new xmlrpcmsg(
 178                                  'b2evo.reportabuse',                        // Function to be called
 179                                  array(
 180                                      new xmlrpcval(0,'int'),                   // Reserved
 181                                      new xmlrpcval('annonymous','string'),     // Reserved
 182                                      new xmlrpcval('nopassrequired','string'), // Reserved
 183                                      new xmlrpcval($abuse_string,'string'),    // The abusive string to report
 184                                      new xmlrpcval($baseurl,'string'),         // The base URL of this b2evo
 185                                  )
 186                              );
 187      $result = $client->send( $message );
 188      if( $ret = xmlrpc_logresult( $result, $Messages, false ) )
 189      { // Remote operation successful:
 190          antispam_update_source( $abuse_string, 'reported' );
 191  
 192          $Messages->add( sprintf( T_('Reported abuse to %s.'), $antispamsrv_host ), 'success' );
 193      }
 194      else
 195      {
 196          $Messages->add( T_('Failed to report abuse to b2evolution.net.'), 'error' );
 197      }
 198  
 199      return $ret;
 200  }

 203  /**
 204   * Request abuse list from central blacklist.
 205   *
 206   * @return boolean true = success, false = error
 207   */
 208  function antispam_poll_abuse()
 209  {
 210      global $Messages, $Settings, $baseurl, $debug, $antispamsrv_host, $antispamsrv_port, $antispamsrv_uri;
 211      global $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password;
 212  
 213      // Construct XML-RPC client:
 214      load_funcs('xmlrpc/model/_xmlrpc.funcs.php');
 215      $client = new xmlrpc_client( $antispamsrv_uri, $antispamsrv_host, $antispamsrv_port );
 216      // yura: I commented this because xmlrpc_client prints the debug info on screen and it breaks header_redirect()
 217      // $client->debug = $debug;
 218  
 219      // Set proxy for outgoing connections:
 220      if( ! empty( $outgoing_proxy_hostname ) )
 221      {
 222          $client->setProxy( $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password );
 223      }
 224  
 225      // Get datetime from last update, because we only want newer stuff...
 226      $last_update = $Settings->get( 'antispam_last_update' );
 227      // Encode it in the XML-RPC format
 228      $Messages->add( T_('Latest update timestamp').': '.$last_update, 'note' );
 229      $startat = mysql2date( 'Ymd\TH:i:s', $last_update );
 230      //$startat = iso8601_encode( mktime(substr($m,11,2),substr($m,14,2),substr($m,17,2),substr($m,5,2),substr($m,8,2),substr($m,0,4)) );
 231  
 232      // Construct XML-RPC message:
 233      $message = new xmlrpcmsg(
 234                                  'b2evo.pollabuse',                            // Function to be called
 235                                  array(
 236                                      new xmlrpcval(0,'int'),                     // Reserved
 237                                      new xmlrpcval('annonymous','string'),       // Reserved
 238                                      new xmlrpcval('nopassrequired','string'),   // Reserved
 239                                      new xmlrpcval($startat,'dateTime.iso8601'), // Datetime to start at
 240                                      new xmlrpcval(0,'int')                      // Reserved
 241                                  )
 242                              );
 243  
 244      $Messages->add( sprintf( T_('Requesting abuse list from %s...'), $antispamsrv_host ), 'note' );
 245  
 246      $result = $client->send( $message );
 247  
 248      if( $ret = xmlrpc_logresult( $result, $Messages, false ) )
 249      { // Response is not an error, let's process it:
 250          $response = $result->value();
 251          if( $response->kindOf() == 'struct' )
 252          { // Decode struct:
 253              $response = xmlrpc_decode_recurse( $response );
 254              if( !isset( $response['strings'] ) || !isset( $response['lasttimestamp'] ) )
 255              {
 256                  $Messages->add( T_('Incomplete reponse.'), 'error' );
 257                  $ret = false;
 258              }
 259              else
 260              { // Start registering strings:
 261                  $value = $response['strings'];
 262                  if( count( $value ) == 0 )
 263                  {
 264                      $Messages->add( T_('No new blacklisted strings are available.'), 'note' );
 265                  }
 266                  else
 267                  { // We got an array of strings:
 268                      $Messages->add( T_('Adding strings to local blacklist:'), 'note' );
 269                      foreach( $value as $banned_string )
 270                      {
 271                          if( antispam_create( $banned_string, 'central' ) )
 272                          { // Creation successed
 273                              $Messages->add( T_('Adding:').' &laquo;'.$banned_string.'&raquo;: '
 274                                  .T_('OK.'), 'note' );
 275                          }
 276                          else
 277                          { // Was already handled
 278                              $Messages->add( T_('Adding:').' &laquo;'.$banned_string.'&raquo;: '
 279                                  .T_('Not necessary! (Already handled)'), 'note' );
 280                              antispam_update_source( $banned_string, 'central' );
 281                          }
 282                      }
 283                      // Store latest timestamp:
 284                      $endedat = date('Y-m-d H:i:s', iso8601_decode( $response['lasttimestamp'] ) );
 285                      $Messages->add( T_('New latest update timestamp').': '.$endedat, 'note' );
 286  
 287                      $Settings->set( 'antispam_last_update', $endedat );
 288                      $Settings->dbupdate();
 289                  }
 290                  $Messages->add( T_('Done.'), 'success' );
 291              }
 292          }
 293          else
 294          {
 295              $Messages->add( T_('Invalid response.'), 'error' );
 296              $ret = false;
 297          }
 298      }
 299  
 300      return $ret ;
 301  }

 304  /**
 305   * Get the base domain that could be blacklisted from an URL.
 306   *
 307   * We want to concentrate on the main domain and we want to prefix it with either . or // in order not
 308   * to blacklist too large.
 309   *
 310   * {@internal This function gets tested in _misc.funcs.simpletest.php}}
 311   *
 312   * @param string URL or domain
 313   * @return string|false the pattern to match this domain in the blacklist; false if we could not extract the base domain
 314   */
 315  function get_ban_domain( $url )
 316  {
 317      // echo '<p>'.$url;
 318  
 319      // Remove http:// part + everything after the last path element ( '/' alone is ignored on purpose )
 320      $domain = preg_replace( '~^ ([a-z]+://)? ([^/#\?]+) (/ ([^/]*/)+ )? .* ~xi', '\\2\\3', $url );
 321  
 322      // echo '<br>'.$domain;
 323  
 324      if( preg_match( '~^[0-9.]+$~', $domain ) )
 325      {    // All numeric = IP address, don't try to cut it any further
 326          return '//'.$domain;
 327      }
 328  
 329      // Remove any www*. prefix:
 330      $base_domain = preg_replace( '~^(www \w* \. )~xi', '', $domain );
 331  
 332      if( empty($base_domain) )
 333      {
 334          return false;
 335      }
 336  
 337      if( utf8_strlen( $base_domain ) < utf8_strlen( $domain ) )
 338      {    // The guy is spamming with subdomains (or www):
 339          return '.'.$base_domain;
 340      }
 341  
 342      // The guy is spamming with the base domain:
 343      return '//'.$base_domain;
 344  }

 350   * Creates an sql command part, which is a condition, that restrict to show comments from those blogs,

 383      }
 384

 462   * @param integer IP start of range

 537  /**
 538   * Block request by IP address
 539   */
 540  function antispam_block_by_ip()
 541  {
 542      global $DB;
 543  
 544      // Detect request IP adresses
 545      $request_ip_list = get_ip_list();
 546  
 547      if( empty( $request_ip_list ) )
 548      { // Could not get any IP address, so can't check anything
 549          return;
 550      }
 551  
 552      $condition = '';
 553      foreach( $request_ip_list as $ip_address )
 554      { // create condition for each detected IP address
 555          $numeric_ip_address = ip2int( $ip_address );
 556          $condition .= ' OR ( aipr_IPv4start <= '.$DB->quote( $numeric_ip_address ).' AND aipr_IPv4end >= '.$DB->quote( $numeric_ip_address ).' )';
 557      }
 558      $condition = '( '.substr( $condition, 4 ).' )';
 559  
 560      $SQL = new SQL();
 561      $SQL->SELECT( 'aipr_ID' );
 562      $SQL->FROM( 'T_antispam__iprange' );
 563      $SQL->WHERE( $condition );
 564      $SQL->WHERE_and( 'aipr_status = \'blocked\'' );
 565      $SQL->LIMIT( 1 );
 566      $ip_range_ID = $DB->get_var( $SQL->get() );
 567  
 568      if( !is_null( $ip_range_ID ) )
 569      { // The request from this IP address must be blocked
 570          $DB->query( 'UPDATE T_antispam__iprange
 571              SET aipr_block_count = aipr_block_count + 1
 572              WHERE aipr_ID = '.$DB->quote( $ip_range_ID ) );
 573  
 574          $debug_message = sprintf( 'A request with ( %s ) ip addresses was blocked because of a blocked IP range ID#%s.', implode( ', ', $request_ip_list ), $ip_range_ID );
 575          exit_blocked_request( 'IP', $debug_message ); // WILL exit();
 576      }
 577  }

 616   * @param boolean TRUE to check if user is in trust group
 617   * @return boolean TRUE if current user can be moved

 619  function antispam_suspect_check( $user_ID = NULL, $check_trust_group = true )
 620  {
 621      global $Settings;
 622  
 623      $suspicious_group_ID = $Settings->get('antispam_suspicious_group');
 624  
 625      if( empty( $suspicious_group_ID ) )
 626      { // We don't need to move users to suspicious group
 627          return false;
 628      }
 629  
 630      if( is_null( $user_ID ) )
 631      { // current User
 632          global $current_User;
 633          $User = $current_User;
 634      }
 635      else
 636      { // Get User by ID
 637          $UserCache = & get_UserCache();
 638          $User = $UserCache->get_by_ID( $user_ID, false, false );
 639      }
 640  
 641      if( empty( $User ) )
 642      { // User must be logged in for this action
 643          return false;
 644      }
 645  
 646      if( $User->grp_ID == $suspicious_group_ID )
 647      { // Current User already is in suspicious group
 648          return false;
 649      }
 650  
 651      if( $check_trust_group )
 652      { // Check if user is in trust group
 653          $antispam_trust_groups = $Settings->get('antispam_trust_groups');
 654          if( !empty( $antispam_trust_groups ) )
 655          {
 656              $antispam_trust_groups = explode( ',', $antispam_trust_groups );
 657              if( in_array( $User->grp_ID, $antispam_trust_groups ) )
 658              { // Current User has group which cannot be moved to suspicious users
 659                  return false;
 660              }
 661          }
 662      }
 663  
 664      // We can move current user to suspect group
 665      return true;
 666  }
