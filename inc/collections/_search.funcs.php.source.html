  17  /**
  18   * Return a search score for the given text and search keywords
  19   *
  20   * @param string the text to score
  21   * @param string the search keywords to score by
  22   * @return integer the result score
  23   */
  24  function score_text( $text, $search_term, $words = array(), $quoted_terms = array(), $score_weight = 1 )
  25  {
  26      global $Debuglog;
  27  
  28      $score = 0.0;
  29      $scores_map = array();
  30  
  31      if( !empty( $search_term ) && strpos( $text, $search_term ) !== false )
  32      {
  33          // the max score what it may received in this methods
  34          $score = ( ( count( $words ) * 4 ) + ( count( $quoted_terms ) * 6 ) ) * $score_weight;
  35          $scores_map['whole_term'] = $score;
  36          return array( 'score' => $score, 'map' => $scores_map );
  37      }
  38  
  39      $all_found = true;
  40      foreach( $quoted_terms as $quoted_term )
  41      { // find quoted keywords
  42          $count = empty( $quoted_term ) ? 0 : substr_count( $text, $quoted_term );
  43          if( $count == 0 )
  44          {
  45              $all_found = false;
  46          }
  47          else
  48          {
  49              $score += ( ( $count > 1 ) ? 8 : 6 );
  50              $scores_map['quoted_term'][$quoted_term] = ( $count > 1 ) ? 8 : 6;
  51          }
  52      }
  53      if( $all_found && count( $quoted_terms ) > 0 )
  54      {
  55          $score += 4;
  56          $scores_map['quoted_term_all'] = 4;
  57      }
  58  
  59      $all_case_sensitive_match = true;
  60      $all_whole_word_match = true;
  61      foreach( $words as $word )
  62      {
  63          $count = empty( $word ) ? 0 : substr_count( $text, $word );
  64          if( $count === 0 )
  65          {
  66              $all_case_sensitive_match = false;
  67          }
  68          else
  69          {
  70              $scores_map['word_case_sensitive_match'][$word] = $score_weight;
  71              $score += $scores_map['word_case_sensitive_match'][$word];
  72          }
  73  
  74          if( $word_match_count = preg_match_all( '/\b'.$word.'\b/i', $text, $matches ) )
  75          { // Every word match gives one more score
  76              $scores_map['whole_word_match'][$word] = $score_weight;
  77              $score += $scores_map['whole_word_match'][$word];
  78          }
  79          else
  80          {
  81              $all_whole_word_match = false;
  82          }
  83  
  84          if( $any_match_count = preg_match_all( '/'.$word.'/i', $text, $matches ) )
  85          { // Every word match gives one more score
  86              $scores_map['word_case_insensitive_match'][$word] = $score_weight;
  87              $score += $scores_map['word_case_insensitive_match'][$word];
  88              if( $any_match_count > 1 )
  89              { // there are multiple occurrences
  90                  $scores_map['word_multiple_occurences'][$word] = score_multiple_occurences( $any_match_count, $score_weight );
  91                  $score += $scores_map['word_multiple_occurences'][$word];
  92              }
  93          }
  94      }
  95  
  96      if( $all_case_sensitive_match )
  97      {
  98          $score += 4;
  99          $scores_map['all_case_sensitive'] = 4;
 100      }
 101  
 102      if( $all_whole_word_match )
 103      {
 104          $score += 4;
 105          $scores_map['all_whole_words'] = 4;
 106      }
 107  
 108      return array( 'score' => $score, 'score_weight' => $score_weight, 'map' => $scores_map );
 109  }

 135  /**
 136   * Return a search score for the given date. Recent dates get higher scores.
 137   *
 138   * @param string the date to score
 139   * @return integer the result score
 140   */
 141  function score_date( $date )
 142  {
 143      global $localtimenow;
 144  
 145      $day_diff = floor( ($localtimenow - strtotime($date)) / (60 * 60 * 24) );
 146      if( $day_diff < 0 )
 147      {
 148          return 0;
 149      }
 150  
 151      if( $day_diff <= 7 )
 152      {
 153          return ( ( $day_diff < 5 ) ? ( 8 - $day_diff ) : 3 );
 154      }
 155  
 156      if( $day_diff < 15 )
 157      {
 158          return 2;
 159      }
 160  
 161      return ( ( $day_diff < 31 ) ? 1 : 0 );
 162  }

 165  /**
 166   * Count score of multiple occurrences
 167   * The score is sum( $score_weight / x ) where x goes from 2 to match count
 168   *
 169   * @param integer match count
 170   * @param integer score weight
 171   * @return float multiple occurrences score based on the received parameters
 172   */
 173  function score_multiple_occurences( $match_count, $score_weight )
 174  {
 175      $result = 0.0;
 176      for( $i = 2; $i <= $match_count; $i++ )
 177      {
 178          $result = $result + floatval($score_weight) / floatval($match_count);
 179      }
 180      return $result;
 181  }

 189  function get_percentage_from_result_map( $type, $scores_map, $quoted_parts, $keywords )
 190  {
 191      switch( $type )
 192      {
 193          case 'item':
 194              $searched_parts = array( 'title', 'content', 'tags' );
 195              break;
 196  
 197          case 'comment':
 198              $searched_parts = array( 'item_title', 'content' );
 199              break;
 200  
 201          case 'category':
 202              $searched_parts = array( 'name', 'description' );
 203              break;
 204  
 205          case 'tag':
 206              $searched_parts = array( 'name' );
 207              break;
 208  
 209          default:
 210              debug_die( 'Invalid search type received!' );
 211      }
 212  
 213      // Check whole term match
 214      foreach( $searched_parts as $searched_part )
 215      {
 216          if( isset( $scores_map[$searched_part]['map']['whole_term'] ) )
 217          { // The whole search term was found
 218              return 100;
 219          }
 220      }
 221  
 222      // Whole search term was not found, count percentage based on the matched parts
 223      $matched_quoted_parts = 0;
 224      foreach( $quoted_parts as $quoted_part )
 225      {
 226          foreach( $searched_parts as $searched_part )
 227          {
 228              if( isset( $scores_map[$searched_part]['map']['quoted_term'][$quoted_part] ) )
 229              {
 230                  $matched_quoted_parts++;
 231                  break; // go to the next quoted part
 232              }
 233          }
 234      }
 235  
 236      $matched_keywords = 0;
 237      foreach( $keywords as $keyword )
 238      {
 239          foreach( $searched_parts as $searched_part )
 240          {
 241              if( isset( $scores_map[$searched_part]['map']['word_case_insensitive_match'][$keyword] ) )
 242              {
 243                  $matched_keywords++;
 244                  break; // go to the next quoted part
 245              }
 246          }
 247      }
 248  
 249      // return round( ( $matched_keywords + ( 2 * $matched_quoted_parts ) ) * 100 / ( count( $keywords ) + ( 2 * count( $quoted_parts ) ) ) );
 250      return round( ( $matched_keywords + $matched_quoted_parts ) * 100 / ( count( $keywords ) + count( $quoted_parts ) ) );
 251  }

 313              + $scores_map['content']['score']
 314              + $scores_map['tags']['score']
 315              + ( isset( $scores_map['creator_login'] ) ? $scores_map['creator_login'] : 0 )
 316              + $scores_map['last_mod_date'];

 319              'type' => 'item',

 516  /**
 517   * Perform a scrored search 
 518   * This searches matching objects and gives a match-quality-score to each found object
 519   *
 520   * @param string the search keywords
 521   * @return array scored search result, each element is an array( type, ID, score )
 522   */
 523  function perform_scored_search( $search_keywords )
 524  {
 525      $keywords = trim( $search_keywords );
 526      if( empty( $keywords ) )
 527      {
 528          return array();
 529      }
 530  
 531      global $Blog, $DB, $posttypes_perms, $debug;
 532      global $scores_map, $score_prefix, $score_map_key, $Debuglog;
 533  
 534      // Get quoted parts parts of the search query
 535      $quoted_parts = array();
 536      if( preg_match_all( '/(["\'])(?:(?=(\\\\?))\\2.)*?\\1/', $search_keywords, $matches ) )
 537      { // There are quoted search terms
 538          $quoted_parts = $matches[0];
 539          for( $index = 0; $index < count( $quoted_parts ); $index++ )
 540          { // Remove douple quotes around the parts
 541              $quoted_part_length = utf8_strlen( $quoted_parts[$index] );
 542              if( $quoted_part_length > 2 )
 543              { // The quoted part is not an empty string
 544                  $quoted_parts[$index] = utf8_substr( $quoted_parts[$index], 1, $quoted_part_length - 2 );
 545              }
 546          }
 547          $quoted_parts = array_unique( $quoted_parts );
 548      };
 549  
 550      // Remove word separators:
 551      $keywords = preg_replace( '/, +/', '', $search_keywords );
 552      $keywords = str_replace( ',', ' ', $keywords );
 553      // Remove quotes:
 554      $keywords = str_replace( '"', ' ', $keywords );
 555      // Remove extra white space:
 556      $keywords = trim( $keywords );
 557      // Split keywords:
 558      $keywords = preg_split( '/\s+/', $keywords );
 559      // Remove duplicate keywords:
 560      $keywords = array_unique( $keywords );
 561  
 562      if( isset( $keywords[0] ) && empty( $keywords[0] ) )
 563      { // The first item is an empty string when the search keyword contains only irrelevant characters
 564          unset( $keywords[0] ); // remove empty word
 565      }
 566      if( empty( $keywords ) && empty( $quoted_parts ) )
 567      { // There is nothing to search for
 568          return array();
 569  // TODO: return NULL and display a specific error message like "Please enter some keywords to search."
 570      }
 571  
 572      // Perform search on Items:
 573      $item_search_result = search_and_score_items( $search_keywords, $keywords, $quoted_parts );
 574      if( $debug )
 575      {
 576          echo '<p class="text-muted">Just found '.count( $item_search_result ).' Items.</p>';
 577          evo_flush();
 578      }
 579  
 580      // Perform search on Comments:
 581      $comment_search_result = search_and_score_comments( $search_keywords, $keywords, $quoted_parts );
 582      $search_result = array_merge( $item_search_result, $comment_search_result );
 583      if( $debug )
 584      {
 585          echo '<p class="text-muted">Just found '.count( $comment_search_result ).' Comments.</p>';
 586          evo_flush();
 587      }
 588  
 589      // Perform search on Chapters:
 590      $cats_search_result = search_and_score_chapters( $search_keywords, $keywords, $quoted_parts );
 591      $search_result = array_merge( $search_result, $cats_search_result );
 592      if( $debug )
 593      {
 594          echo '<p class="text-muted">Just found '.count( $cats_search_result ).' Catageories.</p>';
 595          evo_flush();
 596      }
 597  
 598      // Perform search on Tags:
 599      $tags_search_result = search_and_score_tags( $search_keywords, $keywords, $quoted_parts );
 600      $search_result = array_merge( $search_result, $tags_search_result );
 601      if( $debug )
 602      {
 603          echo '<p class="text-muted">Just found '.count( $tags_search_result ).' Tags.</p>';
 604          evo_flush();
 605      }
 606  
 607      // Sort results by score:
 608      $score_result = array();
 609      foreach( $search_result as $result_item )
 610      {
 611          $score_result[] = $result_item['score'];
 612      }
 613      array_multisort( $score_result, SORT_DESC, $search_result );
 614  
 615      if( count( $search_result ) > 0 )
 616      {
 617          $first_result = $search_result[0];
 618          $max_percentage = get_percentage_from_result_map( $first_result['type'], $first_result['scores_map'], $quoted_parts, $keywords );
 619          $search_result[0]['percentage'] = $max_percentage;
 620          $search_result[0]['nr_of_items'] = count( $item_search_result );
 621          $search_result[0]['nr_of_comments'] = count( $comment_search_result );
 622          $search_result[0]['nr_of_cats'] = count( $cats_search_result );
 623          $search_result[0]['nr_of_tags'] = count( $tags_search_result );
 624      }
 625  
 626      return $search_result;
 627  }

 630  /*
 631   * Perform search (after having displayed the first part of the page) & display results.
 632   * The search results are cached in the session for faster page by page navigation.
 633   *
 634   * @param array Display Params
 635   */
 636  function search_result_block( $params = array() )
 637  {
 638      global $Blog, $Session, $debug;
 639  
 640      $search_keywords = param( 's', 'string', '', true );
 641  
 642      // Try to load existing search results from Session:
 643      $search_params = $Session->get( 'search_params' );
 644      $search_result = $Session->get( 'search_result' );
 645      $search_result_loaded = false;
 646      if( empty( $search_params ) 
 647          || ( $search_params['search_keywords'] != $search_keywords )    // We had saved search results but for a different search string
 648          || ( $search_params['search_blog'] != $Blog->ID )         // We had saved search results but for a different collection
 649          || ( $search_result === NULL ) )
 650      { // We need to perform a new search:
 651          if( $debug )
 652          {
 653              echo '<p class="text-muted">Starting a new search...</p>';
 654          }
 655  
 656          // Flush first part of the page before starting search, which can be long...
 657          evo_flush();
 658  
 659          $search_params = array( 
 660              'search_keywords' => $search_keywords, 
 661              'search_blog' => $Blog->ID 
 662          );
 663  
 664          // Perform new search:
 665          $search_result = perform_scored_search( $search_keywords );
 666          
 667          // Save results into session:
 668          $Session->set( 'search_params', $search_params );
 669          $Session->set( 'search_result', $search_result );
 670          $search_result_loaded = true;
 671      }
 672      else
 673      { // We found the desired saved search results in the Session:
 674          if( $debug )
 675          {    // Display counts
 676              echo '<div class="text-muted">';
 677              echo '<p>We found the desired saved search results in the Session:</p>';
 678              echo '<ul><li>'.sprintf( '%d posts', $search_result[0]['nr_of_items'] ).'</li>';
 679              echo '<li>'.sprintf( '%d comments', $search_result[0]['nr_of_comments'] ).'</li>';
 680              echo '<li>'.sprintf( '%d chapters', $search_result[0]['nr_of_cats'] ).'</li>';
 681              echo '<li>'.sprintf( '%d tags', $search_result[0]['nr_of_tags'] ).'</li></ul>';
 682              echo '</div>';
 683          }
 684  
 685          // Flush first part of the page before starting search, which can be long...
 686          evo_flush();
 687      }
 688  
 689  
 690      // Make sure we are not missing any display params:
 691      $params = array_merge( array(
 692              'no_match_message'      =>  '<p class="alert alert-info msg_nothing" style="margin: 2em 0">'.T_('Sorry, we could not find anything matching your request, please try to broaden your search.').'<p>',
 693              'title_suffix_post'     => ' ('.T_('Post').')',
 694              'title_suffix_comment'  => ' ('.T_('Comment').')',
 695              'title_suffix_category' => ' ('.T_('Category').')',
 696              'title_suffix_tag'      => ' ('.T_('Tag').')',
 697              'block_start'           => '',
 698              'block_end'             => '',
 699              'pagination'            => array(),
 700              'use_editor'            => false, // Use editor instead of author if it is allowed (only the posts have an editor)
 701              'author_format'         => 'avatar_name', // @see User::get_identity_link() // avatar_name | avatar_login | only_avatar | name | login | nickname | firstname | lastname | fullname | preferredname
 702              'date_format'           => locale_datefmt(),
 703          ), $params );
 704  
 705      $search_result = $Session->get( 'search_result' );
 706      if( empty( $search_result ) )
 707      {
 708          echo $params['no_match_message'];
 709          return;
 710      }
 711  
 712      // Prepare pagination:
 713      $result_count = count( $search_result );
 714      $result_per_page = $Blog->get_setting( 'search_per_page' );
 715      if( $result_count > $result_per_page )
 716      {    // We will have multiple search result pages:
 717          $current_page = param( 'page', 'integer', 1 );
 718          $total_pages = ceil($result_count / $result_per_page);
 719          if( $current_page > $total_pages )
 720          {
 721              $current_page = $total_pages;
 722          }
 723  
 724          $page_params = array_merge( array(
 725              'total'        => $result_count,
 726              'current_page' => $current_page,
 727              'total_pages'  => $total_pages,
 728              'list_span'    => 11, // Number of visible pages on navigation line
 729          ), $params['pagination'] );
 730          search_page_links( $page_params );
 731      }
 732      else
 733      {    // Only one page of results:
 734          $current_page = 1;
 735          $total_pages = 1;
 736      }
 737  
 738      // Set current page indexes:
 739      $from = ( ( $current_page -1 ) * $result_per_page );
 740      $to = ( $current_page < $total_pages ) ? ( $from + $result_per_page ) : ( $result_count );
 741  
 742  
 743      // Init caches
 744      $ItemCache = & get_ItemCache();
 745      $CommentCache = & get_CommentCache();
 746      $ChapterCache = & get_ChapterCache();
 747  
 748  
 749      if( !$search_result_loaded )
 750      { // Search result objects are not loaded into memory yet, load them
 751          // Group required object ids by type:
 752          $required_ids = array();
 753          for( $index = $from; $index < $to; $index++ )
 754          {
 755              $row = $search_result[ $index ];
 756              if( isset( $required_ids[ $row['type'] ] ) )
 757              {
 758                  $required_ids[ $row['type'] ][] = $row['ID'];
 759              }
 760              else
 761              {
 762                  $required_ids[ $row['type'] ] = array( $row['ID'] );
 763              }
 764          }
 765  
 766          // Load each required object into the corresponding cache:
 767          foreach( $required_ids as $type => $object_ids )
 768          {
 769              switch( $type )
 770              {
 771                  case 'item':
 772                      $ItemCache->load_list( $object_ids );
 773                      break;
 774  
 775                  case 'comment':
 776                      $CommentCache->load_list( $object_ids );
 777                      break;
 778  
 779                  case 'category':
 780                      $ChapterCache->load_list( $object_ids );
 781                      break;
 782  
 783                  // TODO: we'll probably load "tag" objects once we support tag-synonyms.
 784  
 785                  default: // Not handled search result type
 786                      break;
 787              }
 788          }
 789      }
 790  
 791      // ----------- Display ------------
 792  
 793      echo $params['block_start'];
 794  
 795      // Memorize best scores:
 796      $max_percentage = $search_result[0]['percentage'];
 797      $max_score = $search_result[0]['score'];
 798  
 799      // Display results for current page:
 800      for( $index = $from; $index < $to; $index++ )
 801      {
 802          $row = $search_result[ $index ];
 803          switch( $row['type'] )
 804          {
 805              case 'item':
 806                  // Prepare to display an Item:
 807  
 808                  $Item = $ItemCache->get_by_ID( $row['ID'], false );
 809  
 810                  if( empty( $Item ) )
 811                  { // This Item was deleted, since the search process was executed
 812                      continue 2; // skip from switch and skip to the next item in loop
 813                  }
 814                  
 815                  $display_params = array(
 816                      'title'   => $Item->get_title( array( 'link_type' => 'permalink' ) ).$params['title_suffix_post'],
 817                      'excerpt' => $Item->get_excerpt2(),
 818                      'chapter' => sprintf( T_('In %s'), $Item->get_chapter_links() ),
 819                  );
 820                  
 821                  if( $params['use_editor'] )
 822                  { // Get editor info to display:
 823                      $lastedit_User = & $Item->get_lastedit_User();
 824                      if( empty( $lastedit_User ) )
 825                      { // If editor is not defined yet then use author
 826                          $lastedit_User = & $Item->get_creator_User();
 827                      }
 828                      $display_params = array_merge( array(
 829                              'editor'        => $lastedit_User->get_identity_link( array( 'link_text' => $params['author_format'] ) ),
 830                              'lastedit_date' => mysql2date( $params['date_format'], empty( $Item->datemodified ) ? $Item->datecreated : $Item->datemodified ),
 831                          ), $display_params );
 832                  }
 833                  else
 834                  { // Get author info to display:
 835                      $creator_User = & $Item->get_creator_User();
 836                      $display_params = array_merge( array(
 837                              'author'        => $creator_User->get_identity_link( array( 'link_text' => $params['author_format'] ) ),
 838                              'creation_date' => mysql2date( $params['date_format'], $Item->datecreated ),
 839                              'lastedit_date' => mysql2date( $params['date_format'], $Item->datemodified ),
 840                          ), $display_params );
 841                  }
 842                  break;
 843  
 844              case 'comment':
 845                  // Prepare to display a Comment:
 846  
 847                  $Comment = $CommentCache->get_by_ID( $row['ID'], false );
 848  
 849                  if( empty( $Comment ) || ( $Comment->status == 'trash' ) )
 850                  { // This Comment was deleted, since the search process was executed
 851                      continue 2; // skip from switch and skip to the next item in loop
 852                  }
 853  
 854                  $display_params = array(
 855                      'title'   => $Comment->get_permanent_link( '#item#' ).$params['title_suffix_comment'],
 856                      'excerpt' => excerpt( $Comment->content ),
 857                      'author'  => $Comment->get_author( array(
 858                              'link_text'   => $params['author_format'],
 859                              'thumb_size'  => 'crop-top-15x15',
 860                              'thumb_class' => 'avatar_before_login'
 861                          ) ),
 862                      'creation_date' => mysql2date( $params['date_format'], $Comment->date )
 863                  );
 864                  break;
 865  
 866              case 'category':
 867                  // Prepare to display a Category:
 868  
 869                  $Chapter = $ChapterCache->get_by_ID( $row['ID'], false );
 870  
 871                  if( empty( $Chapter ) )
 872                  { // This Chapter was deleted, since the search process was executed
 873                      continue 2; // skip from switch and skip to the next item in loop
 874                  }
 875                  $display_params = array(
 876                      'title'   => '<a href="'.$Chapter->get_permanent_url().'">'.$Chapter->get_name().'</a>'.$params['title_suffix_category'],
 877                      'excerpt' => excerpt( $Chapter->get( 'description' ) ),
 878                  );
 879                  break;
 880  
 881              case 'tag':
 882                  // Prepare to display a Tag:
 883  
 884                  list( $tag_name, $post_count ) = explode( ':', $row['ID'] );
 885                  $display_params = array(
 886                      'title'   => '<a href="'.url_add_param( $Blog->gen_blogurl(), 'tag='.$tag_name ).'">'.$tag_name.'</a>'.$params['title_suffix_tag'],
 887                      'excerpt' => sprintf( T_('%d posts are tagged with \'%s\''), $post_count, $tag_name ),
 888                  );
 889                  break;
 890  
 891              default: 
 892                  // Other type of result is not implemented
 893  
 894                  // TODO: maybe find collections (especially in case of aggregation)? users? files?
 895  
 896                  continue 2;
 897          }
 898  
 899          // Common display params for all types:
 900          $display_params['score'] = $row['score'];
 901          $display_params['percentage'] = isset( $row['percentage'] ) ? $row['percentage'] : round( $row['score'] * $max_percentage / $max_score );
 902          $display_params['scores_map'] = $row['scores_map'];
 903          $display_params['type'] = $row['type'];
 904          $display_params['best_result'] = $index == 0;
 905          $display_params['max_score'] = sprintf( ( floor( $max_score ) != $max_score ) ? '%.2f' : '%d', $max_score );
 906          $display_params['max_percentage'] = $max_percentage;
 907  
 908          // Display one search result:
 909          display_search_result( array_merge( $params, $display_params ) );
 910      }
 911  
 912      echo $params['block_end'];
 913  
 914      // Display pagination:
 915      if( $result_count > $result_per_page )
 916      {
 917          search_page_links( $page_params );
 918      }
 919  }

 922  /**
 923   * Display search result page links
 924   *
 925   * @param array params
 926   */
 927  function search_page_links( $params = array() )
 928  {
 929      if( !isset( $params['total_pages'] ) )
 930      { // this param is required
 931          return;
 932      }
 933  
 934      $params = array_merge( array(
 935              'block_start'           => '<p class="center">',
 936              'block_end'             => '</p>',
 937              'page_current_template' => '<b>$page_num$</b>',
 938              'page_item_before'      => ' ',
 939              'page_item_after'       => '',
 940              'page_item_current_before' => ' ',
 941              'page_item_current_after'  => '',
 942              'prev_text'             => '&lt;&lt;',
 943              'next_text'             => '&gt;&gt;',
 944              'prev_class'            => '',
 945              'next_class'            => '',
 946          ), $params );
 947  
 948      $page_list_span = $params['list_span'];
 949      $total_pages = $params['total_pages'];
 950      $current_page = isset( $params['current_page'] ) ? $params['current_page'] : 1;
 951      $page_url = regenerate_url( 'page', '' );
 952  
 953      // Initialize a start of pages list:
 954      if( $current_page <= intval( $page_list_span / 2 ) )
 955      { // the current page number is small
 956          $page_list_start = 1;
 957      }
 958      elseif( $current_page > $total_pages - intval( $page_list_span / 2 ) )
 959      { // the current page number is big
 960          $page_list_start = max( 1, $total_pages - $page_list_span+1);
 961      }
 962      else
 963      { // the current page number can be centered
 964          $page_list_start = $current_page - intval( $page_list_span / 2 );
 965      }
 966  
 967      // Initialize an end of pages list:
 968      if( $current_page > $total_pages - intval( $page_list_span / 2 ) )
 969      { //the current page number is big
 970          $page_list_end = $total_pages;
 971      }
 972      else
 973      {
 974          $page_list_end = min( $total_pages, $page_list_start + $page_list_span - 1 );
 975      }
 976  
 977      echo $params['block_start'];
 978  
 979      if( $current_page > 1 )
 980      { // A link to previous page:
 981          echo $params['page_item_before'];
 982          $prev_attrs = empty( $params['prev_class'] ) ? '' : ' class="'.$params['prev_class'].'"';
 983          echo '<a href="'.url_add_param( $page_url, 'page='.( $current_page - 1 ) ).'" rel="prev"'.$prev_attrs.'>'.$params['prev_text'].'</a>';
 984          echo $params['page_item_after'];
 985      }
 986  
 987      if( $page_list_start > 1 )
 988      { // The pages list doesn't contain the first page
 989          // Display a link to first page:
 990          echo $params['page_item_before'];
 991          echo '<a href="'.url_add_param( $page_url, 'page=1' ).'">1</a>';
 992          echo $params['page_item_after'];
 993  
 994          if( $page_list_start > 2 )
 995          { // Display a link to previous pages range:
 996              $page_no = ceil( $page_list_start / 2 );
 997              echo $params['page_item_before'];
 998              echo '<a href="'.url_add_param( $page_url, 'page='.$page_no ).'">...</a>';
 999              echo $params['page_item_after'];
1000          }
1001      }
1002  
1003      $page_prev_i = $current_page - 1;
1004      $page_next_i = $current_page + 1;
1005      for( $i = $page_list_start; $i <= $page_list_end; $i++ )
1006      {
1007          
1008          if( $i == $current_page )
1009          { // Current page
1010              echo $params['page_item_current_before'];
1011              echo str_replace( '$page_num$', $i, $params['page_current_template'] );
1012              echo $params['page_item_current_after'];
1013          }
1014          else
1015          {
1016              $attr_rel = '';
1017              if( $page_prev_i == $i )
1018              { // Add attribute rel="prev" for previous page
1019                  $attr_rel = ' rel="prev"';
1020              }
1021              elseif( $page_next_i == $i )
1022              { // Add attribute rel="next" for next page
1023                  $attr_rel = ' rel="next"';
1024              }
1025              echo $params['page_item_before'];
1026              echo '<a href="'.url_add_param( $page_url, 'page='.$i ).'"'.$attr_rel.'>'.$i.'</a>';
1027              echo $params['page_item_after'];
1028          }
1029      }
1030  
1031      if( $page_list_end < $total_pages )
1032      { // The pages list doesn't contain the last page
1033          if( $page_list_end < $total_pages - 1 )
1034          { // Display a link to next pages range:
1035              $page_no = $page_list_end + floor( ( $total_pages - $page_list_end ) / 2 );
1036              echo $params['page_item_before'];
1037              echo '<a href="'.url_add_param( $page_url, 'page='.$page_no ).'">...</a>';
1038              echo $params['page_item_after'];
1039          }
1040  
1041          // Display a link to last page:
1042          echo $params['page_item_before'];
1043          echo '<a href="'.url_add_param( $page_url, 'page='.$total_pages ).'">'.$total_pages.'</a>';
1044          echo $params['page_item_after'];
1045      }
1046  
1047      if( $current_page < $total_pages )
1048      { // A link to next page:
1049          echo $params['page_item_before'];
1050          $next_attrs = empty( $params['next_class'] ) ? '' : ' class="'.$params['next_class'].'"';
1051          echo ' <a href="'.url_add_param( $page_url, 'page='.( $current_page + 1 ) ).'" rel="next"'.$next_attrs.'>'.$params['next_text'].'</a>';
1052          echo $params['page_item_after'];
1053      }
1054  
1055      echo $params['block_end'];
1056  }

1138  /**
1139   * Display score map regarding a search result
1140   *
1141   * @param array detailed information about the received search scores
1142   */
1143  function display_score_map( $params )
1144  {
1145      echo '<ul class="search_score_map dimmed">';
1146      foreach( $params['scores_map'] as $result_part => $score_map )
1147      {
1148  
1149          if( ! is_array( $score_map ) )
1150          {
1151              if( $score_map > 0 )
1152              { // Score received for this field
1153                  echo '<li>'.sprintf( 'Extra points for [%s]', $result_part ).'</li><ul>';
1154                  switch ( $result_part )
1155                  {
1156                      case 'last_mod_date':
1157                      case 'creation_date':
1158                          echo '<li>'.sprintf( '%d points.', $score_map );
1159                          echo ' Rule: The number of points are calculated based on the days passed since creation or last modification';
1160                          echo '<ul><li>days_passed < 5 => ( 8 - days_passed )</li>';
1161                          echo '<li>5 <= days_passed < 8 => 3</li>';
1162                          echo '<li>when days_passed >= 8: ( days_passed < 15 ? 2 : ( days_passed < 30 ? 1 : 0 ) )</li>';
1163                          echo '</ul>';
1164                          break;
1165  
1166                      case 'post_count':
1167                          if( $params['type'] == 'category' )
1168                          {
1169                              echo '<li>'.sprintf( '%d points for the amount of posts in this category. Rule: number_of_posts > 30 ? 10 : intval( number_of_posts / 3 )', $score_map ).'</li>';
1170                              break;
1171                          }
1172                          elseif( $params['type'] == 'tag' )
1173                          {
1174                              echo '<li>'.sprintf( '%d posts in this tag. Total points = sum( points ) * number_of_posts.', $score_map ).'</li>';
1175                              break;
