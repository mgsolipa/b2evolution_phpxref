 411       * Get default module permissions
 412       *
 413       * @param integer Group ID
 414       * @return array
 415       */
 416  	function get_default_group_permissions( $grp_ID )
 417      {
 418          switch( $grp_ID )
 419          {
 420              case 1:        // Administrators (group ID 1) have permission by default:
 421                  $permapi = 'always'; // Can use APIs
 422                  $permcreateblog = 'allowed'; // Creating new blogs
 423                  $permgetblog = 'denied'; // Automatically add a new blog to the new users
 424                  break;
 425  
 426              case 2:        // Moderators (group ID 2) have permission by default:
 427                  $permapi = 'always';
 428                  $permcreateblog = 'allowed';
 429                  $permgetblog = 'denied';
 430                  break;
 431  
 432              case 3:        // Editors (group ID 3) have permission by default:
 433              case 4:     // Normal Users (group ID 4) have permission by default:
 434                  $permapi = 'always';
 435                  $permcreateblog = 'denied';
 436                  $permgetblog = 'denied';
 437                  break;
 438  
 439              default:
 440                  // Other groups have no permission by default
 441                  $permapi = 'never';
 442                  $permcreateblog = 'denied';
 443                  $permgetblog = 'denied';
 444                  break;
 445          }
 446  
 447          // We can return as many default permissions as we want:
 448          // e.g. array ( permission_name => permission_value, ... , ... )
 449          return $permissions = array( 'perm_api' => $permapi, 'perm_createblog' => $permcreateblog, 'perm_getblog' => $permgetblog );
 450      }

 514       *
 515       * @param string Requested permission level
 516       * @param string Permission value
 517       * @param mixed Permission target (blog ID, array of cat IDs...)
 518       * @return boolean True on success (permission is granted), false if permission is not granted

 520  	function check_api_group_perm( $permlevel, $permvalue, $permtarget )
 521      {
 522          $perm = false;
 523          switch ( $permvalue )
 524          {
 525              case 'always':
 526                  // Users can use APIs
 527                  if( $permlevel == 'always' )
 528                  {
 529                      $perm = true;
 530                      break;
 531                  }
 532  
 533              case 'never':
 534                  // Users can`t use APIs
 535                  if( $permlevel == 'never' )
 536                  {
 537                      $perm = false;
 538                      break;
 539                  }
 540          }
 541  
 542          return $perm;
 543      }

 972      /**
 973       * Handle collections module htsrv actions
 974       */
 975  	function handle_htsrv_action()
 976      {
 977          global $demo_mode, $current_User, $DB, $Session, $Messages;
 978          global $UserSettings, $samedomain_htsrv_url;
 979  
 980          if( !is_logged_in() )
 981          { // user must be logged in
 982              bad_request_die( $this->T_( 'You are not logged in.' ) );
 983          }
 984  
 985          // Init the objects we want to work on.
 986          $action = param_action( true );
 987  
 988          // Check that this action request is not a CSRF hacked request:
 989          $Session->assert_received_crumb( 'collections_'.$action );
 990  
 991          switch( $action )
 992          {
 993              case 'unlink':
 994                  // Unlink a file from a LinkOwner ( Item, Comment ) object, and delete that file if it's not linked to any other object
 995  
 996                  $link_ID = param( 'link_ID', 'integer', true );
 997                  $redirect_to = param( 'redirect_to', 'url', '' );
 998                  $LinkCache = & get_LinkCache();
 999                  $edited_Link = & $LinkCache->get_by_ID( $link_ID, false );
1000  
1001                  if( !$edited_Link )
1002                  { // the edited Link object doesn't exists
1003                      $Messages->add( sprintf( T_('Requested &laquo;%s&raquo; object does not exist any longer.'), T_('Link') ), 'error' );
1004                      header_redirect();
1005                  }
1006  
1007                  // We have a link, get the LinkOwner it is attached to:
1008                  $LinkOwner = & $edited_Link->get_LinkOwner();
1009                  $linked_File = & $edited_Link->get_File();
1010  
1011                  // Load the blog we're in:
1012                  $Blog = & $LinkOwner->get_Blog();
1013                  set_working_blog( $Blog->ID );
1014  
1015                  // Check permission:
1016                  $LinkOwner->check_perm( 'edit', true );
1017  
1018                  $confirmed = param( 'confirmed', 'integer', 0 );
1019                  if( $confirmed )
1020                  { // Unlink File from Item:
1021                      $deleted_link_ID = $edited_Link->ID;
1022                      $edited_Link->dbdelete();
1023                      unset($edited_Link);
1024  
1025                      $LinkOwner->after_unlink_action( $deleted_link_ID );
1026  
1027                      $Messages->add( $LinkOwner->translate( 'Link has been deleted from $xxx$.' ), 'success' );
1028  
1029                      if( $current_User->check_perm( 'files', 'edit' ) )
1030                      { // current User has permission to edit/delete files
1031                          $file_name = $linked_File->get_name();
1032                          // Get number of objects where this file is attahced to
1033                          // TODO: attila>this must be handled with a different function
1034                          $file_links = get_file_links( $linked_File->ID, array( 'separator' => '<br />' ) );
1035                          $links_count = ( strlen( $file_links ) > 0 ) ? substr_count( $file_links, '<br />' ) + 1 : 0;
1036                          if( $links_count > 0 )
1037                          { // File is linked to other objects
1038                              $Messages->add( sprintf( T_('File %s is still linked to %d other objects'), $file_name, $links_count ), 'note' );
1039                          }
1040                          else
1041                          { // File is not linked to other objects
1042                              if( $linked_File->unlink() )
1043                              { // File removed successful ( removed from db and from storage device also )
1044                                  $Messages->add( sprintf( T_('File %s has been deleted.'), $file_name ), 'success' );
1045                              }
1046                              else
1047                              { // Could not completly remove the file
1048                                  $Messages->add( sprintf( T_('File %s could not be deleted.'), $file_name ), 'error' );
1049                              }
1050                          }
1051                      }
1052                  }
1053                  else
1054                  { // Display confirm unlink/delete message
1055                      $delete_url = $samedomain_htsrv_url.'action.php?mname=collections&action=unlink&link_ID='.$edited_Link->ID.'&confirmed=1&crumb_collections_unlink='.get_crumb( 'collections_unlink' );
1056                      $ok_button = '<a href="'.$delete_url.'" class="btn btn-danger">'.T_('I am sure!').'</a>';
1057                      $cancel_button = '<a href="'.$redirect_to.'" class="btn btn-default">'.T_('CANCEL').'</a>';
1058                      $msg = sprintf( T_( 'You are about to unlink and delete the attached file from %s path.' ), $linked_File->get_root_and_rel_path() );
1059                      $msg .= '<br />'.T_( 'This CANNOT be undone!').'<br />'.T_( 'Are you sure?' ).'<br /><br />'.$ok_button."\t".$cancel_button;
1060                      $Messages->add( $msg, 'error' );
1061                  }
1062                  header_redirect( $redirect_to );
1063                  break;
1064  
1065              case 'isubs_update':
1066                  // Subscribe/Unsubscribe user on the selected item
1067  
1068                  if( $demo_mode && ( $current_User->ID <= 3 ) )
1069                  { // don't allow default users profile change on demo mode
1070                      bad_request_die( 'Demo mode: you can\'t edit the admin and demo users profile!<br />[<a href="javascript:history.go(-1)">'
1071                                  . T_('Back to profile') . '</a>]' );
1072                  }
1073  
1074                  // Get params
1075                  $item_ID = param( 'p', 'integer', true );
1076                  $notify = param( 'notify', 'integer', 0 );
1077  
1078                  if( ( $notify < 0 ) || ( $notify > 1 ) )
1079                  { // Invalid notify param. It should be 0 for unsubscribe and 1 for subscribe.
1080                      $Messages->add( 'Invalid params!', 'error' );
1081                  }
1082  
1083                  if( ! is_email( $current_User->get( 'email' ) ) )
1084                  { // user doesn't have a valid email address
1085                      $Messages->add( T_( 'Your email address is invalid. Please set your email address first.' ), 'error' );
1086                  }
1087  
1088                  if( $Messages->has_errors() )
1089                  { // errors detected
1090                      header_redirect();
1091                      // already exited here
1092                  }
1093  
1094                  if( set_user_isubscription( $current_User->ID, $item_ID, $notify ) )
1095                  {
1096                      if( $notify == 0 )
1097                      {
1098                          $Messages->add( T_( 'You have successfully unsubscribed.' ), 'success' );
1099                      }
1100                      else
1101                      {
1102                          $Messages->add( T_( 'You have successfully subscribed to notifications.' ), 'success' );
1103                      }
1104                  }
1105                  else
1106                  {
1107                      $Messages->add( T_( 'Could not subscribe to notifications.' ), 'error' );
1108                  }
1109  
1110                  header_redirect();
1111                  break; // already exited here
1112          }
1113      }
