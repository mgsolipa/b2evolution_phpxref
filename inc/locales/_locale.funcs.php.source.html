 300  /**
 301   * Activate a locale.
 302   *
 303   * @todo dh> this should make sure, that e.g. "charset" is set for the locale in {@link $locales}. See http://forums.b2evolution.net/viewtopic.php?p=43980#43980
 304   *
 305   * @param string locale to activate
 306   * @param boolean True on success/change, false on failure (if already set or not existant)
 307   */
 308  function locale_activate( $locale )
 309  {
 310      global $locales, $current_locale, $current_charset;
 311  
 312      if( $locale == $current_locale
 313              || empty( $locale )
 314              || ! isset( $locales[$locale] ) )
 315      {
 316          return false;
 317      }
 318  
 319      // Memorize new locale:
 320      $current_locale = $locale;
 321      // Memorize new charset:
 322      $current_charset = $locales[ $locale ][ 'charset' ];
 323      return true;
 324  }

 333  {
 334      global $locales, $default_locale;

 339          { // found first matching locale
 340              return $locale;
 341          }
 342      }
 343

 345      if( $fallback_to_default )
 346          return $default_locale;

 521  /**
 522   * [callback function] Outputs an <option> set with default locale selected.
 523   * Optionally returns an array with a locale key and name if there's only one enabled locale.
 524   *
 525   * @param string default value
 526   * @param boolean echo output?
 527   * @param boolean Return array (locale key + name) if there's only one enabled locale?
 528   * @return string|array The options string or an array (locale key + name) if there's only one enabled locale and $array_if_onelocale == true.
 529   */
 530  function locale_options( $default = '', $disp = true, $array_if_onelocale = false )
 531  {
 532      global $locales, $default_locale;
 533  
 534      if( empty( $default ) ) $default = $default_locale;
 535  
 536      $r = '';
 537      $enabled_count = 0;
 538      $enabled_lastkey = '';
 539  
 540      foreach( $locales as $this_localekey => $this_locale )
 541      {
 542          if( $this_locale['enabled'] || $this_localekey == $default )
 543          {
 544              $r .= '<option value="'. $this_localekey. '"';
 545              if( $this_localekey == $default )
 546                  $r .= ' selected="selected"';
 547              $r .= '>'. T_($this_locale['name']). '</option>';
 548  
 549              ++$enabled_count;
 550              $enabled_lastkey = $this_localekey;
 551          }
 552      }
 553  
 554      if( $disp )
 555      {    // the result must be displayed
 556          echo $r;
 557      }
 558  
 559      if ( $array_if_onelocale && $enabled_count == 1 )
 560      {    // We've only one enabled locale:
 561          return array( $enabled_lastkey, $locales[$enabled_lastkey]['name'] );
 562      }
 563      else
 564      {    // Return the string.
 565          return $r;
 566      }
 567  }

 582  /**
 583   * Detect language from HTTP_ACCEPT_LANGUAGE
 584   *
 585   * HTTP_ACCEPT_LANGUAGE is sorted by prio and then the best match is used
 586   * (either full locale ("en-US") or best fitting locale for a short one ("en").
 587   *
 588   * This gets tested in {@link test_locale_from_httpaccept()}.
 589   *
 590   * @author Rewritten by blueyed in Revision 1.42
 591   *
 592   * @return string Locale made out of HTTP_ACCEPT_LANGUAGE or $default_locale, if no match
 593   */
 594  function locale_from_httpaccept()
 595  {
 596      global $locales, $default_locale;
 597      if( isset($_SERVER['HTTP_ACCEPT_LANGUAGE']) )
 598      {
 599          $accept = strtolower( $_SERVER['HTTP_ACCEPT_LANGUAGE'] );
 600  
 601          // Create list of accepted locales.
 602          if( ! preg_match_all('/([a-z]{1,8}(?:-[a-z]{1,8})?)\s*(?:;\s*q\s*=\s*(1|0(?:\.[0-9]+)?))?/i', $accept, $accept_list) )
 603          {
 604              return $default_locale;
 605          }
 606          // Create list of enabled locales.
 607          $enabled_locales = array();
 608          foreach($locales as $k => $v)
 609          {
 610              if( empty($v['enabled']) )
 611              {
 612                  continue;
 613              }
 614              $enabled_locales[strtolower($k)] = $k;
 615          }
 616          if( empty($enabled_locales) )
 617          {
 618              return $default_locale;
 619          }
 620          // Build mapping of short code to long code(s)
 621          $short_locales = array();
 622          foreach($enabled_locales as $v)
 623          {
 624              if( $pos = strpos($v, '-') )
 625              {
 626                  $short = substr($v, 0, $pos);
 627                  $short_locales[$short][] = $v;
 628              }
 629          }
 630          // Create "locale" => "prio" list
 631          $accept_list = array_combine($accept_list[1], $accept_list[2]);
 632          $maxq = count($accept_list)+1;
 633          foreach( $accept_list as $k => $v )
 634          {
 635              if( $v === '' )
 636              { // should be kept in order
 637                  $accept_list[$k] = $maxq--;
 638              }
 639              elseif( $v == 0 )
 640              { // not acceptable (RFC 2616)
 641                  unset($accept_list[$k]);
 642              }
 643          }
 644          arsort($accept_list);
 645          $accept_list = array_values(array_keys($accept_list));
 646  
 647          // Go through the list of accepted locales and find best match.
 648          for( $i = 0, $n = count($accept_list); $i<$n; $i++ )
 649          {
 650              $test = $accept_list[$i];
 651  
 652              if( isset($enabled_locales[$test]) )
 653              { // the accepted locale is enabled and a full match
 654                  return $enabled_locales[$test];
 655              }
 656              if( isset($short_locales[$test]) )
 657              { // this is a short locale: find the best/first match in accepted locales
 658                  $first = NULL;
 659                  foreach($short_locales[$test] as $v)
 660                  {
 661                      $pos = array_search(strtolower($v), $accept_list);
 662                      if( $pos !== false && ( ! isset($first) || $pos < $first ) )
 663                      {
 664                          $first = $pos;
 665                      }
 666                  }
 667                  if( isset($first) )
 668                  { // found exact match, via short locale.
 669                      return $enabled_locales[$accept_list[$first]];
 670                  }
 671                  if( isset($enabled_locales[$test.'-'.$test]) )
 672                  { // test for e.g. "de-DE" when only "de" is accepted
 673                      return $enabled_locales[$test.'-'.$test];
 674                  }
 675                  // Fallback: use first enabled locale matching the short accepted
 676                  return $short_locales[$test][0];
 677              }
 678          }
 679      }
 680      return $default_locale;
 681  }

 701  function locale_overwritefromDB()

 703      global $DB, $locales, $default_locale, $Settings, $Debuglog, $Messages;
 704  
 705      // Load all locales from disk to get 'charset' insrtad of DB
 706      locales_load_available_defs();
 707  
 708      $usedprios = array();  // remember which priorities are used already.
 709      $priocounter = 0;
 710      $query = 'SELECT
 711                          loc_locale, loc_datefmt, loc_timefmt, loc_shorttimefmt, loc_startofweek,
 712                          loc_name, loc_messages, loc_priority, loc_transliteration_map, loc_enabled
 713                          FROM T_locales ORDER BY loc_priority';
 714  
 715      $ctrl = isset( $_GET['ctrl'] ) ? $_GET['ctrl'] : '';
 716      $priocounter_max = 0;
 717      $wrong_locales = array();
 718      $wrong_default_locale = '';
 719      foreach( $DB->get_results( $query, ARRAY_A ) as $row )
 720      { // Loop through loaded locales:
 721  
 722          if( $row['loc_priority'] == $priocounter )
 723          { // priority conflict (the same)
 724              $priocounter++;
 725          }
 726          else
 727          {
 728              $priocounter = $row['loc_priority'];
 729          }
 730          $priocounter_max = $priocounter;
 731  
 732          //remember that we used this
 733          $usedprios[] = $priocounter;
 734  
 735          $transliteration_map = '';
 736          // Try to unserialize the value
 737          if( ( $r = @unserialize( @base64_decode( $row[ 'loc_transliteration_map' ] ) ) ) !== false )
 738          {
 739              $transliteration_map = $r;
 740          }
 741  
 742          if( isset( $locales[ $row['loc_locale'] ], $locales[ $row['loc_locale'] ]['charset'] ) )
 743          { // Use charset from locale file if such file exists on disk
 744              $loc_charset = $locales[ $row['loc_locale'] ]['charset'];
 745          }
 746          else
 747          { // Use charset from DB
 748              $loc_charset = 'utf-8';
 749          }
 750  
 751          if( ( $ctrl == 'locales' || $ctrl == 'regional' ) && $row['loc_enabled'] && $loc_charset != 'utf-8' )
 752          { // Check wrong non utf-8 locales in order to deactivate them
 753              $Messages->add( sprintf( T_('The locale %s has been deactivated because it\'s not UTF-8'), '<b>'.$row['loc_locale'].'</b>' ) );
 754              if( $Settings->get( 'default_locale' ) == $row['loc_locale'] && $default_locale != $row['loc_locale'] )
 755              { // Change main locale to default
 756                  $wrong_default_locale = $row['loc_locale'];
 757                  $Messages->add( sprintf( T_('The main locale was no longer valid and has been set to %s.'), '<b>'.$default_locale.'</b>' ) );
 758              }
 759              $wrong_locales[] = $row['loc_locale'];
 760              $row['loc_enabled'] = 0;
 761          }
 762  
 763          $locales[ $row['loc_locale'] ] = array(
 764                  'charset'      => $loc_charset,
 765                  'datefmt'      => $row['loc_datefmt'],
 766                  'timefmt'      => $row['loc_timefmt'],
 767                  'shorttimefmt' => $row['loc_shorttimefmt'],
 768                  'startofweek'  => $row['loc_startofweek'],
 769                  'name'         => $row['loc_name'],
 770                  'messages'     => $row['loc_messages'],
 771                  'transliteration_map' => $transliteration_map,
 772                  'priority'     => $priocounter,
 773                  'enabled'      => $row['loc_enabled'],
 774                  'fromdb'       => 1
 775              );
 776      }
 777  
 778      if( count( $wrong_locales ) )
 779      { // Deactivate locales that have wrong charset
 780          $DB->query( 'UPDATE T_locales
 781                SET loc_enabled = 0
 782              WHERE loc_locale IN( '.$DB->quote( $wrong_locales ).' )' );
 783          if( ! empty( $wrong_default_locale ) )
 784          { // If main locale is wring we sgould updated it to default and use this for all users and blogs that used this wrong locale
 785              $Settings->set( 'default_locale', $default_locale );
 786              $Settings->dbupdate();
 787              $DB->query( 'UPDATE T_users
 788                    SET user_locale = '.$DB->quote( $default_locale ).'
 789                  WHERE user_locale = '.$DB->quote( $wrong_default_locale ) );
 790              $DB->query( 'UPDATE T_blogs
 791                    SET blog_locale = '.$DB->quote( $default_locale ).'
 792                  WHERE blog_locale = '.$DB->quote( $wrong_default_locale ) );
 793          }
 794      }
 795  
 796      foreach( $locales as $l_ley => $locale )
 797      { // Change priority of non-db locales to put them at the end of list
 798          if( empty( $locale['fromdb'] ) && $locale['priority'] < $priocounter_max )
 799          {
 800              $locales[ $l_ley ]['priority'] = $priocounter_max + $locale['priority'];
 801          }
 802      }
 803  
 804      // set default priorities, if nothing was set in DB.
 805      // Missing "priority gaps" will get filled here.
 806      if( $DB->num_rows != count($locales) )
 807      { // we have locales from conf file that need a priority
 808          $priocounter = 1;
 809          foreach( $locales as $lkey => $lval )
 810          { // Loop through memory locales:
 811              if( !isset($lval['priority']) )
 812              { // Found one that has no assigned priority
 813                  while( in_array( $priocounter, $usedprios ) )
 814                  {
 815                      $priocounter++;
 816                  }
 817                  // Priocounter has max value
 818                  $locales[$lkey]['priority'] = $priocounter;
 819                  $usedprios[] = $priocounter;
 820              }
 821          }
 822      }
 823  
 824      // sort by priority
 825      uasort( $locales, 'locale_priosort' );
 826  
 827      // overwrite default_locale from DB settings - if enabled.
 828      // Checks also if previous $default_locale is enabled. Defaults to en-EU, even if not enabled.
 829      $locale_fromdb = $Settings->get('default_locale');
 830  
 831      if( $locale_fromdb )
 832      {
 833          if( !isset( $locales[$locale_fromdb] ) )
 834          {
 835              $Debuglog->add( 'Default locale ['.$locale_fromdb.'] from general settings is not available.', 'locale' );
 836              return false;
 837          }
 838          else
 839          {
 840              $default_locale = $locale_fromdb;
 841              return $default_locale;
 842          }
 843      }
 844  }

 847  /**
 848   * Write $locales array to DB table
 849   */
 850  function locale_updateDB()
 851  {
 852      global $locales, $DB, $Settings, $Messages, $action;
 853      global $saved_params;
 854  
 855      $templocales = $locales;
 856      $disabled_locales = array();
 857      $saved_params = array();
 858  
 859      $current_default_locale = $Settings->get( 'default_locale' );
 860  
 861      $lnr = 0;
 862      // Loop through list of all HTTP POSTed params:
 863      foreach( $_POST as $pkey => $pval )
 864      {
 865          if( ! preg_match( '/loc_(\d+)_(.*)/', $pkey, $matches ) )
 866          {
 867              continue;
 868          }
 869  
 870          // This is a locale related parameter, get it now:
 871          $pval = param( $pkey, 'string', '' );
 872          $saved_params[ $pkey ] = $pval;
 873  
 874          $lfield = $matches[2];
 875  
 876          if( $matches[1] != $lnr )
 877          { // we have a new locale
 878              $lnr = $matches[1];
 879              $plocale = $pval;
 880  
 881              if( $templocales[ $plocale ]['enabled'] )
 882              { // Collect previously enabled locales
 883                  $disabled_locales[ $plocale ] = $plocale;
 884              }
 885              // checkboxes default to 0
 886              $templocales[ $plocale ]['enabled'] = 0;
 887          }
 888          elseif( $lnr != 0 )  // be sure to have catched a locale before
 889          {
 890              if( $lfield == 'startofweek' && ( $lfield < 0 || $lfield > 6 ) )
 891              { // startofweek must be between 0 and 6
 892                  continue;
 893              }
 894              if( $lfield == 'enabled' )
 895              { // This locale is enabled, remove from the list of the disabled locales
 896                  unset( $disabled_locales[ $plocale ] );
 897              }
 898              elseif( $plocale == $current_default_locale && isset( $disabled_locales[ $plocale ] ) && ! isset( $_POST['loc_'.$lnr.'_enabled'] ) )
 899              { // The default locale must be always enabled
 900                  unset( $disabled_locales[ $plocale ] );
 901                  // Force the default locale to be enabled
 902                  $templocales[ $plocale ]['enabled'] = 1;
 903                  // Inform user about this action
 904                  $Messages->add( sprintf( T_('The locale %s was re-enabled automatically because it\'s currently the default locale. The default locale must be a valid and enabled locale.'), '<b>'.$current_default_locale.'</b>' ), 'warning' );
 905              }
 906              $templocales[ $plocale ][ $lfield ] = $pval;
 907          }
 908      }
 909  
 910      $locales = $templocales;
 911  
 912      // Check the usage of the disabled locales
 913      $disabled_locales = ( count( $disabled_locales ) > 0 ) ? $DB->quote( $disabled_locales ) : false;
 914      if( $disabled_locales && ( $action != 'confirm_update' ) )
 915      { // Some locales were disabled, create warning message if required
 916          $user_disabled_locales = $DB->get_assoc( '
 917              SELECT user_locale, count( user_ID ) as number_of_users
 918              FROM T_users
 919              WHERE user_locale IN ( '.$disabled_locales.' )
 920              GROUP BY user_locale
 921              HAVING number_of_users > 0
 922              ORDER BY user_locale'
 923          );
 924          $coll_disabled_locales = $DB->get_assoc( '
 925              SELECT blog_locale, count( blog_ID ) as number_of_blogs
 926              FROM T_blogs
 927              WHERE blog_locale IN ( '.$disabled_locales.' )
 928              GROUP BY blog_locale
 929              HAVING number_of_blogs > 0
 930              ORDER BY blog_locale'
 931          );
 932  
 933          global $warning_message;
 934          $warning_message = array();
 935          $users_message = T_('%d users with invalid locale %s');
 936          $colls_message = T_('%d collections with invalid locale %s');
 937          foreach( $user_disabled_locales as $disabled_locale => $numbe_of_users )
 938          { // Disabled locale is used by users, add warning
 939              $warning_message[] = sprintf( $users_message, $numbe_of_users, $disabled_locale );
 940              if( isset( $coll_disabled_locales[$disabled_locale] ) )
 941              { // Disabled locale is used by blogs, add warning
 942                  $warning_message[] = sprintf( $colls_message, $coll_disabled_locales[$disabled_locale], $disabled_locale );
 943                  unset( $coll_disabled_locales[$disabled_locale] );
 944              }
 945          }
 946          foreach( $coll_disabled_locales as $disabled_locale => $numbe_of_colls )
 947          { // Disabled locale is used by blogs, add warning
 948              $warning_message[] = sprintf( $colls_message, $numbe_of_colls, $disabled_locale );
 949          }
 950  
 951          if( !empty( $warning_message ) )
 952          { // There are disabled locales which are used, create the final warning message
 953              $warning_message = T_('You have disabled some locales. This results in:')."\n"
 954                  .'<ul><li>'.implode( '</li><li>', $warning_message ).'</li></ul>'
 955                  .sprintf( T_('These will be assigned the default locale %s'), $current_default_locale );
 956              return false;
 957          }
 958      }
 959  
 960      $query = "REPLACE INTO T_locales ( loc_locale, loc_datefmt, loc_timefmt, loc_shorttimefmt, loc_startofweek, loc_name, loc_messages, loc_priority, loc_transliteration_map, loc_enabled ) VALUES ";
 961      foreach( $locales as $localekey => $lval )
 962      {
 963          if( empty($lval['messages']) )
 964          { // if not explicit messages file is given we'll translate the locale
 965              $lval['messages'] = strtr($localekey, '-', '_');
 966          }
 967  
 968          $transliteration_map = '';
 969          if( !empty($lval['transliteration_map']) )
 970          {
 971              if( is_string($lval['transliteration_map']) )
 972              {    // The value is already serialized and encoded
 973                  $transliteration_map = $lval['transliteration_map'];
 974              }
 975              else
 976              {    // Encode the value
 977                  $transliteration_map = base64_encode( serialize($lval['transliteration_map']) );
 978              }
 979          }
 980  
 981          $query .= '(
 982              '.$DB->quote($localekey).',
 983              '.$DB->quote($lval['datefmt']).',
 984              '.$DB->quote($lval['timefmt']).',
 985              '.$DB->quote($lval['shorttimefmt']).',
 986              '.$DB->quote($lval['startofweek']).',
 987              '.$DB->quote($lval['name']).',
 988              '.$DB->quote($lval['messages']).',
 989              '.$DB->quote($lval['priority']).',
 990              '.$DB->quote($transliteration_map).',
 991              '.$DB->quote($lval['enabled']).'
 992          ), ';
 993      }
 994      $query = substr($query, 0, -2);
 995  
 996      $DB->begin();
 997  
 998      $result = true;
 999      if( $action == 'confirm_update' )
1000      { // Update users and blogs locale to the default if the prevously used locale was disabled
1001          $users_update = $DB->query( 'UPDATE T_users
1002              SET user_locale = '.$DB->quote( $current_default_locale ).'
1003              WHERE  user_locale IN ( '.$disabled_locales.' )'
1004          );
1005          $blogs_update = $DB->query( 'UPDATE T_blogs
1006              SET blog_locale = '.$DB->quote( $current_default_locale ).'
1007              WHERE  blog_locale IN ( '.$disabled_locales.' )'
1008          );
1009          $result = ( $users_update !== false ) && ( $blogs_update !== false );
1010      }
1011  
1012      if( $result && ( $DB->query($query) !== false ) )
1013      { // Commit after successful update
1014          $DB->commit();
1015          return true;
1016      }
1017  
1018      // Some error occured, rollback the transaction and add error message
1019      $DB->rollback();
1020      $Messages->add( T_('Unexpected error occured, regional settings could not be updated.') );
1021      return false;
1022  }

1027   *
1028   * @todo Implement iconv and PHP mapping tables

1030   * @see can_convert_charsets()
1031   * @param string String to convert
1032   * @param string Target charset (TO)
1033   * @param string Source charset (FROM)

1036  function convert_charset( $string, $dest_charset, $src_charset )
1037  {

1056  
1057      if( isset($GLOBALS['Timer']) )
1058      {
1059          $GLOBALS['Timer']->pause('convert_charset', false );
1060      }
1061      return $string;

1065  /**
1066   * Can we convert from charset A to charset B?
1067   * @param string Target charset (TO)
1068   * @param string Source charset (FROM)
1069   * @return boolean
1070   */
1071  function can_convert_charsets( $dest_charset, $src_charset )
1072  {
1073      if( empty($dest_charset) || empty($src_charset) )
1074      {
1075          return false;
1076      }
1077  
1078      if( function_exists('mb_internal_encoding') )
1079      { // mb_string extension:
1080          $orig = mb_internal_encoding();
1081  
1082          $r = false;
1083          if( @mb_internal_encoding($dest_charset) && @mb_internal_encoding($src_charset) )
1084          { // we can set both encodings, so we should be able to convert:
1085              $r = true;
1086          }
1087  
1088          mb_internal_encoding($orig);
1089          return $r;
1090      }
1091  
1092      return false;
1093  }

1125  /**
1126   * Init charset handling between Input/Output ($io_charset) and the internal
1127   * handling ($evo_charset).
1128   *
1129   * Check and possibly adjust {@link $evo_charset}.
1130   *
1131   * @staticvar boolean Used to only start mb_output_handler once
1132   * @param string I/O (input/output) charset to use
1133   * @return boolean true, if encoding has been changed
1134   */
1135  function init_charsets( $req_io_charset )
1136  {
1137      static $mb_output_handler_started;
1138      global $io_charset, $evo_charset, $Debuglog, $DB;
1139      global $force_io_charset_if_accepted;
1140  
1141      if( $req_io_charset == $io_charset )
1142      { // no conversation/init needed
1143          return false;
1144      }
1145  
1146      // check, if we want to force a specific charset (e.g. 'utf-8'):
1147      if( ! empty($force_io_charset_if_accepted) )
1148      { // we want to force a specific charset:
1149          if( ! isset($_SERVER['HTTP_ACCEPT_CHARSET']) // all allowed
1150              || preg_match( '~\b(\*|'.preg_quote( $force_io_charset_if_accepted, '~' ).')\b~', $_SERVER['HTTP_ACCEPT_CHARSET'] ) )
1151          {
1152              $req_io_charset = $force_io_charset_if_accepted; // pretend that the first one has been requested
1153          }
1154      }
1155  
1156      if( $req_io_charset == $io_charset )
1157      { // no conversation/init needed
1158          return false;
1159      }
1160  
1161      $io_charset = $req_io_charset;
1162  
1163      if( empty($evo_charset) )
1164      { // empty evo_charset follows I/O charset:
1165          // TODO: $evo_charset will not follow, if it has followed before.. (because not empty anymore)
1166          $Debuglog->add( '$evo_charset follows $io_charset ('.$io_charset.').', array('locale') );
1167          $evo_charset = $io_charset;
1168      }
1169      elseif( $evo_charset != $io_charset )
1170      { // we have to convert for I/O
1171          // TODO: dh> $io_charset has to forcefully follow $evo_charset, if we cannot convert, e.g. utf-8/iso-8859-1
1172          if( ! function_exists('mb_convert_encoding') )
1173          {
1174              $Debuglog->add( '$evo_charset differs from $io_charset, but mbstrings is not available - cannot convert I/O to internal charset!', array('errors','locale') );
1175              $evo_charset = $io_charset; // we cannot convert I/O to internal charset
1176          }
1177          else
1178          { // check if the encodings are supported:
1179              // NOTE: mb_internal_encoding() is the best way to find out if the encoding is supported
1180              $old_mb_internal_encoding = mb_internal_encoding();
1181              if( ! @mb_internal_encoding($io_charset) )
1182              {
1183                  $Debuglog->add( 'Cannot I/O convert because I/O charset ['.$io_charset.'] is not supported by mbstring!', array('error','locale') );
1184                  $evo_charset = $io_charset;
1185                  mb_internal_encoding($old_mb_internal_encoding);
1186              }
1187              elseif( ! @mb_internal_encoding($evo_charset) )
1188              {
1189                  $Debuglog->add( 'Cannot I/O convert because $evo_charset='.$evo_charset.' is not supported by mbstring!', array('error','locale') );
1190                  $evo_charset = $io_charset;
1191                  mb_internal_encoding($old_mb_internal_encoding);
1192              }
1193              else
1194              { // we can convert between I/O
1195                  mb_http_output( $io_charset );
1196                  if( ! $mb_output_handler_started )
1197                  {
1198                      ob_start( 'mb_output_handler' ); // NOTE: this will send a Content-Type header by itself for "text/..."
1199                      $mb_output_handler_started = true;
1200                      $Debuglog->add( 'Started mb_output_handler.', 'locale' );
1201                  }
1202              }
1203          }
1204      }
1205  
1206      // Make sure the DB send us text in the same charset as $evo_charset (override whatever may have been set before)
1207      if( isset($DB) ) // not available in /install/index.php
1208      {    // Set encoding for MySQL connection:
1209          $DB->set_connection_charset( $evo_charset );
1210      }
1211  
1212      $Debuglog->add( 'evo_charset: '.$evo_charset, 'locale' );
1213      $Debuglog->add( 'io_charset: '.$io_charset, 'locale' );
1214  
1215      return true;
1216  }

1219  /**
1220   * Load available locale definitions
1221   */
1222  function locales_load_available_defs()
1223  {
1224      global $locales_path, $locales, $locales_loaded_form_disk;
1225  
1226      if( ! empty( $locales_loaded_form_disk ) )
1227      { // All locales have been already loaded
1228          return;
1229      }
1230  
1231      // This is where language packs will store their locale defintions:
1232      $locale_defs = array();
1233  
1234      // Get all locale folder names:
1235      $filename_params = array(
1236              'inc_files' => false,
1237              'recurse'   => false,
1238              'basename'  => true,
1239          );
1240      $locale_folders = get_filenames( $locales_path, $filename_params );
1241      // Go through all locale folders:
1242      foreach( $locale_folders as $locale_folder )
1243      {
1244          //pre_dump( $locale_folder );
1245          $ad_locale_folder = $locales_path.'/'.$locale_folder;
1246          // Get files in folder:
1247          $filename_params = array(
1248                  'inc_dirs'    => false,
1249                  'recurse'    => false,
1250                  'basename'    => true,
1251              );
1252          $locale_def_files = get_filenames( $ad_locale_folder, $filename_params );
1253          // Go through files in locale folder:
1254          foreach( $locale_def_files as $locale_def_file )
1255          {    // Check if it's a definition file:
1256              // pre_dump( $locale_def_file );
1257              if( preg_match( '~[a-z0-9\-.]\.locale\.php$~i', $locale_def_file ) )
1258              {    // We found a definition file:
1259                  // pre_dump( $locale_def_file );
1260                  include $ad_locale_folder.'/'.$locale_def_file;
1261              }
1262          }
1263      }
1264  
1265      // Copy any new locale definitions over to $locales:
1266      foreach( $locale_defs as $locale_code => $locale_def )
1267      {
1268          if( !isset( $locales[$locale_code] ) )
1269          {    // New locale, add to main array:
1270              $locales[$locale_code] = $locale_def;
1271              // ... but mark as not enabled!
1272              $locales[$locale_code]['enabled'] = 0;
1273          }
1274      }
1275  
1276  
1277      // Assign priorities:
1278  
1279      // Find highest used priority:
1280      $max_prio = 0;
1281      foreach( $locales as $locale )

1302  /**
1303   * Get a number of the messages in the .PO & .POT file
1304   *
1305   * @param string File name (messages.po)
1306   * @param boolean TRUE - to calc a percent of translated messages
1307   * @return integer Number of the messages
1308   */
1309  function locale_file_po_info( $po_file_name, $calc_percent_done = false )
1310  {
1311      $all = 0;
1312      $fuzzy = 0;
1313      $this_fuzzy = false;
1314      $untranslated = 0;
1315      $translated = 0;
1316      $status = '-';
1317      $matches = array();
1318  
1319      if( file_exists( $po_file_name ) )
1320      {
1321          $lines = file( $po_file_name );
1322          $lines[] = '';    // Adds a blank line at the end in order to ensure complete handling of the file
1323  
1324          foreach( $lines as $line )
1325          {
1326              if( trim( $line ) == '' )
1327              {    // Blank line, go back to base status:
1328                  if( $status == 't' )
1329                  {    // ** End of a translation ** :
1330                      if( $msgstr == '' )
1331                      {
1332                          $untranslated++;
1333                          // echo 'untranslated: ', $msgid, '<br />';
1334                      }
1335                      else
1336                      {
1337                          $translated++;
1338                      }
1339                      if( $msgid == '' && $this_fuzzy )
1340                      {    // It's OK if first line is fuzzy
1341                          $fuzzy--;
1342                      }
1343                      $msgid = '';
1344                      $msgstr = '';
1345                      $this_fuzzy = false;
1346                  }
1347                  $status = '-';
1348              }
1349              elseif( ( $status == '-' ) && preg_match( '#^msgid "(.*)"#', $line, $matches ) )
1350              {    // Encountered an original text
1351                  $status = 'o';
1352                  $msgid = $matches[1];
1353                  // echo 'original: "', $msgid, '"<br />';
1354                  $all++;
1355              }
1356              elseif( ( $status == 'o' ) && preg_match( '#^msgstr "(.*)"#', $line, $matches ) )
1357              {    // Encountered a translated text
1358                  $status = 't';
1359                  $msgstr = $matches[1];
1360                  // echo 'translated: "', $msgstr, '"<br />';
1361              }
1362              elseif( preg_match( '#^"(.*)"#', $line, $matches ) )
1363              {    // Encountered a followup line
1364                  if( $status == 'o' )
1365                      $msgid .= $matches[1];
1366                  elseif( $status == 't' )
1367                      $msgstr .= $matches[1];
1368              }
1369              elseif( strpos( $line,'#, fuzzy' ) === 0 )
1370              {
1371                  $this_fuzzy = true;
1372                  $fuzzy++;
1373              }
1374          }
1375      }
1376  
1377      $info = array(
1378              'all'          => $all,
1379              'fuzzy'        => $fuzzy,
1380              'translated'   => $translated,
1381              'untranslated' => $untranslated
1382          );
1383  
1384      if( $calc_percent_done )
1385      {
1386          $info['percent'] = locale_file_po_percent_done( $info );
1387      }
1388  
1389      return $info;
1390  }
