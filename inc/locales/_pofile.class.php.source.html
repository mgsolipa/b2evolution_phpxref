  35          if( in_array($msgid, array('trans_locale', 'trans_charset', 'trans_available')) )

  42          // replace newlines, tabs, carriage returns and double quotes:
  43          $msgid = str_replace( array( "\n", "\t", "\r", '"' ),

  97      /**
  98       * Read a .po file
  99       *
 100       * @param boolean Log source info to {@link $Messages}?
 101       * @return array with msgids => array( 'trans' => msgstr )
 102       */
 103  	function read( $log_source_info = true )
 104      {
 105          $lines = file( $this->filename );
 106          $lines[] = '';    // Adds a blank line at the end in order to ensure complete handling of the file
 107          $all = 0;
 108          $fuzzy = 0;
 109          $untranslated = 0;
 110          $translated = 0;
 111          $status = '-';
 112          $matches = array();
 113          $sources = array();
 114          $loc_vars = array();
 115          $this->msgids = array();
 116          $is_fuzzy = false;
 117          foreach ($lines as $line)
 118          {
 119              // echo 'LINE:', $line, '<br />';
 120              if(trim($line) == '' )
 121              {    // Blank line, go back to base status:
 122                  if( $status == 't' )
 123                  {    // ** End of a translation **:
 124                      if( $msgstr == '' )
 125                      {
 126                          $untranslated++;
 127                          // echo 'untranslated: ', $msgid, '<br />';
 128                      }
 129                      elseif( $is_fuzzy )
 130                      {
 131                          $fuzzy++;
 132                          // echo 'fuzzy: ', $msgid, "<br />\n";
 133                      }
 134                      else
 135                      {
 136                          $translated++;
 137  
 138                          // Inspect where the string is used
 139                          $sources = array_unique( $sources );
 140                          // echo '<p>sources: ', implode( ', ', $sources ), '</p>';
 141                          foreach( $sources as $source )
 142                          {
 143                              if( !isset( $loc_vars[$source]  ) ) $loc_vars[$source] = 1;
 144                              else $loc_vars[$source] ++;
 145                          }
 146  
 147                          // Save the string
 148                          $this->msgids[$msgid]['trans'] = $msgstr;
 149                      }
 150                  }
 151                  $status = '-';
 152                  $msgid = '';
 153                  $msgstr = '';
 154                  $sources = array();
 155                  $is_fuzzy = false;
 156              }
 157              elseif( ($status=='-') && preg_match( '#^msgid "(.*)"#', $line, $matches))
 158              {    // Encountered an original text
 159                  $status = 'o';
 160                  $msgid = $matches[1];
 161                  // echo 'original: "', $msgid, '"<br />';
 162                  $all++;
 163              }
 164              elseif( ($status=='o') && preg_match( '#^msgstr "(.*)"#', $line, $matches))
 165              {    // Encountered a translated text
 166                  $status = 't';
 167                  $msgstr = $matches[1];
 168                  // echo 'translated: "', $msgstr, '"<br />';
 169              }
 170              elseif( preg_match( '#^"(.*)"#', $line, $matches))
 171              {    // Encountered a followup line
 172                  if ($status=='o')
 173                      $msgid .= $matches[1];
 174                  elseif ($status=='t')
 175                      $msgstr .= $matches[1];
 176              }
 177              elseif( ($status=='-') && preg_match( '@^#:(.*)@', $line, $matches))
 178              {    // Encountered a source code location comment
 179                  // echo $matches[0],'<br />';
 180                  $sourcefiles = preg_replace( '@\\\\@', '/', $matches[1] );
 181                  // $c = preg_match_all( '@ ../../../([^:]*):@', $sourcefiles, $matches);
 182                  $c = preg_match_all( '@ ../../../([^/:]*/?)@', $sourcefiles, $matches);
 183                  for( $i = 0; $i < $c; $i++ )
 184                  {
 185                      $sources[] = $matches[1][$i];
 186                  }
 187                  // echo '<br />';
 188              }
 189              elseif(strpos($line,'#, fuzzy') === 0)
 190              {
 191                  $is_fuzzy = true;
 192              }
 193          }
 194  
 195          if( $loc_vars && $log_source_info )
 196          {
 197              global $Messages;
 198              ksort( $loc_vars );
 199  
 200              $list_counts = '';
 201              foreach( $loc_vars as $source => $c )
 202              {
 203                  $list_counts .= "\n<li>$source = $c</li>";
 204              }
 205              $Messages->add( 'Sources and number of strings: <ul>'.$list_counts.'</ul>', 'note' );
 206          }
 207  
 208          return $this->msgids;
 209      }

 212      /**
 213       * Write POFile::$msgids into $file_path.
 214       *
 215       * @return true|string True on success, string with error on failure
 216       */
 217  	function write_evo_trans($file_path, $locale)
 218      {
 219          $fp = fopen( $file_path, 'w+' );
 220  
 221          if( ! $fp )
 222          {
 223              return "Could not open $file_path for writing!";
 224          }
 225  
 226          fwrite( $fp, "<?php\n" );
 227          fwrite( $fp, "/*\n" );
 228          fwrite( $fp, " * Global lang file\n" );
 229          fwrite( $fp, " * This file was generated automatically from messages.po\n" );
 230          fwrite( $fp, " */\n" );
 231          fwrite( $fp, "if( !defined('EVO_MAIN_INIT') ) die( 'Please, do not access this page directly.' );" );
 232          fwrite( $fp, "\n\n" );
 233  
 234  
 235          fwrite( $fp, '$trans[\''.$locale."'] = array(\n" );
 236  
 237          // Write meta/format info:
 238          $charset = 'utf-8'; // default
 239          if( isset($this->msgids['']) )
 240          {
 241              if( preg_match( '~\\\nContent-Type: text/plain; charset=(.*?);?\\\n~', $this->msgids['']['trans'], $match ) )
 242              {
 243                  $charset = strtolower($match[1]);
 244              }
 245          }
 246          fwrite( $fp, "'__meta__' => array('format_version'=>1, 'charset'=>'$charset'),\n" );
 247  
 248          foreach( $this->msgids as $msgid => $msginfo )
 249          {
 250              $msgstr = $msginfo['trans'];
 251  
 252              fwrite( $fp, POFile::quote($msgid).' => '.POFile::quote($msgstr).",\n" );
 253          }
 254          fwrite( $fp, "\n);\n?>" );
 255          fclose( $fp );
 256  
 257          return true;
 258      }

 261      /**
 262       * Quote a msgid/msgstr, preferrable with single quotes.
 263       *
 264       * Single quotes are preferred, as PHP just handles them as strings and
 265       * does no extra parsing.
 266       * Double quotes are used, if there's \n, \r or \t in the string.
 267       *
 268       * @param string
 269       * @return string Quoted string (either using double or single quotes (preferred))
 270       */
 271  	function quote($s)
 272      {
 273          if( preg_match('~\\\\[nrt]~', $s) ) // \r, \n or \t in there
 274          {
 275              // NOTE: no need to escape '"', as its escaped in .po files already
 276              return '"'.str_replace( '$', '\$', $s ).'"';
 277          }
 278          else
 279          {
 280              return "'".str_replace( array("'", '\"'), array("\'", '"'), $s )."'";
 281          }
 282      }
