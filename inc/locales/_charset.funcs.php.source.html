  20  /**
  21   * Use iconv() to transliterate non-ASCII chars in a string encoded with $evo_charset.
  22   *
  23   * This function will figure out a usable LC_CTYPE setting and revert it to the original value
  24   * after calling iconv().
  25   *
  26   * @author Tilman BLUMENBACH - tblue246
  27   * @todo Tblue> Try more locales.
  28   *
  29   * @param string The string to transliterate.
  30   * @param NULL|string The post locale. NULL to not try switching to it.
  31   * @return string|boolean The transliterated ASCII string on success or false on failure.
  32   */
  33  function evo_iconv_transliterate( $str, $post_locale = NULL )
  34  {
  35      global $evo_charset, $current_locale, $default_locale;
  36  
  37      if( ! function_exists( 'iconv' ) )
  38      {
  39          return false;
  40      }
  41  
  42      // iconv() needs a proper LC_CTYPE to work.
  43      // See http://www.php.net/manual/en/function.iconv.php#94481
  44      $orig_lc_ctype  = setlocale( LC_CTYPE, 0 );
  45      $lc_evo_charset = strtolower( str_replace( '-', '', $evo_charset ) );
  46  
  47      $locales_to_try = array(
  48          str_replace( '-', '_', $current_locale ).'.'.$lc_evo_charset, // Try to use current b2evo locale
  49          str_replace( '-', '_', $default_locale ).'.'.$lc_evo_charset, // Fallback to default b2evo locale
  50      );
  51      if( $post_locale !== NULL )
  52      {    // Try to switch to the post locale:
  53          array_unshift( $locales_to_try, str_replace( '-', '_', $post_locale ).'.'.$lc_evo_charset );
  54      }
  55  
  56      if( setlocale( LC_CTYPE, $locales_to_try ) === false )
  57      {    // The last thing we try is to use the system locale with our charset.
  58          if( ( $pos = strrpos( $orig_lc_ctype, '.' ) ) !== false )
  59          {    // Remove existing charset string:
  60              $syslocale = substr( $orig_lc_ctype, 0, $pos );
  61          }
  62          else
  63          {
  64              $syslocale = $orig_lc_ctype;
  65          }
  66  
  67          if( setlocale( LC_CTYPE, $syslocale.'.'.$lc_evo_charset ) === false )
  68          {    // We could not set a usable locale, giving up...
  69              return false;
  70          }
  71      }
  72  
  73      //pre_dump( setlocale( LC_CTYPE, 0 ) );
  74  
  75      // Transliterate the string:
  76      $newstr = iconv( $evo_charset, 'ASCII//TRANSLIT', $str );
  77  
  78      // Restore the original locale:
  79      setlocale( LC_CTYPE, $orig_lc_ctype );
  80  
  81      return $newstr;
  82  }

  93  function replace_special_chars( $str, $post_locale = NULL )
  94  {
  95      global $evo_charset, $default_locale, $current_locale, $locales;
  96  
  97      // Decode entities to be able to transliterate the associated chars:
  98      // Tblue> TODO: Check if this could have side effects.
  99      $str = html_entity_decode( $str, ENT_NOQUOTES, $evo_charset );
 100  
 101      $our_locale = $post_locale;
 102      if( $our_locale === NULL )
 103      {    // post locale is not set, try to guess current locale
 104          if( !empty($default_locale) )
 105          {
 106              $our_locale = $default_locale;
 107          }
 108          if( !empty($current_locale) )
 109          {    // Override with current locale if available
 110              $our_locale = $current_locale;
 111          }
 112      }
 113      if( $our_locale !== NULL && isset($locales[$our_locale]) && !empty($locales[$our_locale]['transliteration_map']) )
 114      {    // Use locale 'transliteration_map' if present
 115          if( ! array_key_exists( '', $locales[$our_locale]['transliteration_map'] ) )
 116          {    // Make sure there's no empty string key, otherwise strtr() returns false
 117              if( $tmp_str = strtr( $str, $locales[$our_locale]['transliteration_map'] ) );
 118              {    // Use newly transliterated string
 119                  $str = $tmp_str;
 120              }
 121          }
 122      }
 123  
 124      if( ( $newstr = evo_iconv_transliterate( $str, $post_locale ) ) !== false )
 125      {    // iconv allows us to get nice URL titles by transliterating non-ASCII chars.
 126          // Tblue> htmlentities() does not know anything about ASCII?! ISO-8859-1 will work too, though.
 127          $newstr_charset = 'ISO-8859-1';
 128      }
 129      // TODO: sam2kb> convert this to 'transliteration_map'
 130      else if( can_convert_charsets('UTF-8', $evo_charset) && can_convert_charsets('UTF-8', 'ISO-8859-1') /* source */ )
 131      {    // Fallback to the limited old method: Transliterate only a few known chars.
 132          $newstr = convert_charset( $str, 'UTF-8', $evo_charset );
 133          $newstr_charset = 'UTF-8';
 134  
 135          $search = array( 'Ä', 'ä', 'Ö', 'ö', 'Ü', 'ü', 'ß', 'à', 'ç', 'è', 'é', 'ì', 'ò', 'ô', 'ù' ); // iso-8859-1
 136          $replace = array( 'Ae', 'ae', 'Oe', 'oe', 'Ue', 'ue', 'ss', 'a', 'c', 'e', 'e', 'i', 'o', 'o', 'u' );
 137  
 138          foreach( $search as $k => $v )
 139          { // convert $search to UTF-8
 140              $search[$k] = convert_charset( $v, 'UTF-8', 'ISO-8859-1' );
 141          }
 142  
 143          $newstr = str_replace( $search, $replace, $newstr );
 144      }
 145      else
 146      {
 147          // Replace HTML entities only.
 148          $newstr = $str;
 149          $newstr_charset = $evo_charset;
 150      }
 151  
 152      // Replace HTML entities
 153      $newstr = htmlentities( $newstr, ENT_NOQUOTES, $newstr_charset );
 154  
 155      // Handle special entities (e.g., use "-" instead of "a" for "&"):
 156      $newstr = str_replace(
 157          array( '&amp;', '&laquo;', '&raquo;' ),
 158          '-',
 159          $newstr );
 160  
 161  
 162      // Keep only one char in entities!
 163      $newstr = preg_replace( '/&(.).+?;/', '$1', $newstr );
 164      // Replace non acceptable chars
 165      $newstr = preg_replace( '/[^A-Za-z0-9_]+/', '-', $newstr );
 166      // Remove '-' at start and end:
 167      $newstr = preg_replace( '/^-+/', '', $newstr );
 168      $newstr = preg_replace( '/-+$/', '', $newstr );
 169  
 170      //pre_dump( $str, $newstr );
 171  
 172      return $newstr;
 173  }
