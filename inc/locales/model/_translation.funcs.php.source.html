  22  function translation_update_table_pot()
  23  {
  24      global $DB, $locales_path;
  25  
  26      // Reset all previous strings
  27      $DB->query( 'UPDATE T_i18n_original_string SET iost_inpotfile = 0' );
  28  
  29      $status = '-';
  30  
  31      $lines = file( $locales_path.'messages.pot' );
  32      $lines[] = '';    // Adds a blank line at the end in order to ensure complete handling of the file
  33  
  34      foreach( $lines as $line )
  35      {
  36          if( trim( $line ) == '' )
  37          {    // Blank line, go back to base status:
  38              if( $status == 't' && !empty( $msgid ) )
  39              {    // ** End of an original text ** :
  40                  translation_update_table_pot_row( trim( $msgid ) );
  41              }
  42              $msgid = '';
  43              $msgstr = '';
  44              $status = '-';
  45          }
  46          elseif( ( $status == '-' ) && preg_match( '#^msgid "(.*)"#', $line, $matches ) )
  47          {    // Encountered an original text
  48              $status = 'o';
  49              $msgid = $matches[1];
  50          }
  51          elseif( ( $status == 'o' ) && preg_match( '#^msgstr "(.*)"#', $line, $matches ) )
  52          {    // Encountered a translated text
  53              $status = 't';
  54              $msgstr = $matches[1];
  55          }
  56          elseif( preg_match( '#^"(.*)"#', $line, $matches ) )
  57          {    // Encountered a followup line
  58              if( $status == 'o' )
  59                  $msgid .= $matches[1];
  60              elseif( $status == 't' )
  61                  $msgstr .= $matches[1];
  62          }
  63      }
  64

 262   * Generate .POT file
 263   */
 264  function translation_generate_pot_file()
 265  {
 266      global $DB, $locales_path;
 267  
 268      $pot_file_name = $locales_path.'messages.pot';
 269  
 270      $pot_content = array();
 271      $pot_content[] = '# b2evolution - Language file';
 272      $pot_content[] = '# Copyright (C) '.date( 'Y' ).' Francois PLANQUE';
 273      $pot_content[] = '# This file is distributed under the same license as the b2evolution package.';
 274      $pot_content[] = '';
 275  
 276      global $basepath;
 277      $translation_strings = array();
 278      translation_scandir( $basepath, $translation_strings );
 279  
 280      foreach( $translation_strings as $string => $files )
 281      { // Format the translation strings to write in .POT file
 282          if( isset( $files['trans'] ) )
 283          { // Text of TRANS info
 284              if( is_array( $files['trans'] ) )
 285              { // Multiline TRANS info
 286                  foreach( $files['trans'] as $ft => $files_trans )
 287                  {
 288                      $pot_content[] = '#. '.( $ft == 0 ? 'TRANS: ' : '' ).$files_trans;
 289                  }
 290              }
 291              else
 292              { // Single TRANS info
 293                  $pot_content[] = '#. TRANS: '.$files['trans'];
 294              }
 295              unset( $files['trans'] );
 296          }
 297          foreach( $files as $file )
 298          { // File name and line number where string exists
 299              $pot_content[] = '#: '.$file[1].':'.$file[0];
 300          }
 301          if( strpos( $string, '%' ) !== false )
 302          { // Char '%' is detected in the string
 303              if( preg_match( '/%(s|\d*d)/', $string ) )
 304              { // The string contains a mask like %s or %d
 305                  $pot_content[] = '#, php-format';
 306              }
 307              else
 308              { // The string contains a simple char '%'
 309                  $pot_content[] = '#, no-php-format';
 310              }
 311          }
 312          $pot_content[] = 'msgid "'.$string.'"';
 313          $pot_content[] = 'msgstr ""';
 314          $pot_content[] = '';
 315      }
 316  
 317      // Write to .POT file
 318      $ok = (bool) save_to_file( implode( "\n", $pot_content ), $pot_file_name, 'w+' );
 319  
 320      if( ! $ok )
 321      { // Inform user about no permission to write POT file
 322          global $Messages;
 323          $Messages->add( sprintf( T_('The file %s cannot be written to disk. Please check the filesystem permissions.'), '<b>'.$pot_file_name.'</b>' ), 'error' );
 324      }
 325  
 326      return $ok;
 327  }

 336  function translation_scandir( $path, & $translation_strings )
 337  {
 338      $files = scandir( $path );
 339      foreach( $files as $file )
 340      {
 341          if( is_file( $path.$file ) && preg_match( '/\.php$/i', $path.$file ) )
 342          {    // PHP file; Find all translation strings in current file
 343              translation_find_T_strings( $path.$file, $translation_strings );
 344          }
 345          elseif( $file != '.' && $file != '..' && is_dir( $path.$file ) )
 346          {    // Directory; Scan each directory recursively to find all PHP files
 347              translation_scandir( $path.$file.'/', $translation_strings );
 348          }
 349      }
 350  }

 353  /**
 354   * Get substring from the selected position with shift
 355   *
 356   * @param string Source string
 357   * @param integer Current index
 358   * @param integer Shift
 359   * @return string
 360   */
 361  function translation_get_chars( $line_string, $index, $shift )
 362  {
 363      $char = '';
 364  
 365      if( $shift > 0 )
 366      { // Get next chars
 367          if( strlen( $line_string ) > $index + $shift )
 368          {
 369              $char = substr( $line_string, $index + 1, $shift );
 370          }
 371      }
 372      elseif( $shift < 0 )
 373      { // Get previous chars
 374          if( $index + $shift >= 0 )
 375          {
 376              $char = substr( $line_string, $index + $shift, 0 - $shift );
 377          }
 378      }
 379      else
 380      { // Current char
 381          $char = $line_string[ $index ];
 382      }
 383  
 384      return $char;
 385  }

 387  /**
 388   * Find the translation strings in the file
 389   *
 390   * @param string File path
 391   * @param array Translation strings (by reference)
 392   */
 393  function translation_find_T_strings( $file, & $translation_strings )
 394  {
 395      global $basepath;
 396  
 397      $line_is_multiple = false;
 398  
 399      // Split file content with lines in order to know line number of each string
 400      $file_lines = explode( "\n", file_get_contents( $file ) );
 401  
 402      $T_line_number = 0;
 403      $T_string_is_opened = false;
 404      $T_string_text = '';
 405      $T_string_quote_sign = "'";
 406      $TRANS_is_opened = false;
 407      $TRANS_text = '';
 408  
 409      $prev_line_number = -1;
 410      foreach( $file_lines as $line_number => $line_string )
 411      {
 412          if( ! $T_string_is_opened && strpos( $line_string, 'T_' ) === false && strpos( $line_string, 'TS_' ) === false  &&
 413              ! $TRANS_is_opened && strpos( $line_string, 'TRANS:' ) === false )
 414          { // This line doesn't contain any T_ string AND TRANS info, Skip it
 415              continue;
 416          }
 417  
 418          for( $l = 0; $l <= strlen( $line_string ); $l++ )
 419          {
 420              $char = $line_string[ $l ];
 421  
 422              /************ T_ strings ************/
 423              $char_prev3 = translation_get_chars( $line_string, $l, -3 );
 424              $char_prev2 = translation_get_chars( $line_string, $l, -2 );
 425  
 426              if( ! $T_string_is_opened &&
 427                  ( $char_prev2 == 'T_' || $char_prev3 == 'TS_' || $char_prev3 == 'NT_' ) )
 428              { // T_ string is detected
 429                  $char_next1 = translation_get_chars( $line_string, $l, 1 );
 430                  $char_next2 = translation_get_chars( $line_string, $l, 2 );
 431  
 432                  if( $char == '(' &&
 433                      ( $char_next1 == '"' || $char_next1 == "'" ||
 434                        $char_next2 == ' "' || $char_next2 == " '" ) )
 435                  { // OPEN T_ string
 436                      $T_string_quote_sign = ( $char_next1 == '"' || $char_next2 == ' "' ) ? '"' : "'";
 437                      $l += ( $char_next1 == '"' || $char_next1 == "'" ) ? 1 : 2;
 438                      $T_line_number = $line_number + 1;
 439                      $T_string_is_opened = true;
 440                      $TRANS_is_opened = false;
 441                      continue;
 442                  }
 443              }
 444  
 445              $char_prev1 = translation_get_chars( $line_string, $l, -1 );
 446              if( $T_string_is_opened && $char_prev1 != '\\' && $char == $T_string_quote_sign )
 447              { // CLOSE T_ string
 448  
 449                  // Save T_ string to array
 450                  $T_string_text = str_replace( array( '"', "\'", "\r\n", "\n", '\n', "\t" ), array( '\"', "'", '\n', '\n', '\n"'."\n".'"', '\t' ), $T_string_text );
 451                  if( strpos( $T_string_text, "\n" ) !== false )
 452                  { // Add empty new line before multiline string
 453                      $T_string_text = '"'."\n".'"'.$T_string_text;
 454                  }
 455                  if( !isset( $translation_strings[ $T_string_text ] ) )
 456                  { // Set array for each string in order to store the file paths where this string is found
 457                      $translation_strings[ $T_string_text ] = array();
 458                  }
 459                  if( ! empty( $TRANS_text ) && ! isset( $translation_strings[ $T_string_text ]['trans'] ) )
 460                  { // Text of TRANS info
 461                      if( strpos( $TRANS_text, "\n" ) !== false )
 462                      {
 463                          $TRANS_text = explode( "\n", $TRANS_text );
 464                          foreach( $TRANS_text as $t_i => $TRANS_text_value )
 465                          {
 466                              $TRANS_text[ $t_i ] = trim( $TRANS_text_value, '     ' );
 467                          }
 468                      }
 469                      $translation_strings[ $T_string_text ]['trans'] = $TRANS_text;
 470                  }
 471                  $translation_strings[ $T_string_text ][] = array(
 472                          $T_line_number, // Line number
 473                          str_replace( $basepath, '../../../', $file ), // String
 474                      );
 475  
 476                  // Reset vars for next T_ string
 477                  $T_line_number = 0;
 478                  $T_string_text = '';
 479                  $T_string_is_opened = false;
 480                  $TRANS_text = '';
 481              }
 482  
 483              if( $T_string_is_opened )
 484              { // Store chars of the current opened T_ string
 485                  if( $prev_line_number != $line_number )
 486                  { // String is multiline
 487                      if( $T_string_text != '' )
 488                      {
 489                          $T_string_text .= "\n";
 490                      }
 491                      $prev_line_number = $line_number;
 492                  }
 493                  $T_string_text .= $line_string[ $l ];
 494              }
 495  
 496              /************ TRANS info ************/
 497              if( $T_string_is_opened )
 498              { // T_ is opened skip this
 499                  continue;
 500              }
 501  
 502              // Try to find TRANS
 503              $char_prev5 = translation_get_chars( $line_string, $l, -5 );
 504              if( $char == ':' && $char_prev5 == 'TRANS' && ( strpos( $line_string, '//' ) !== false || strpos( $line_string, '/*' ) !== false ) )
 505              { // TRANS info is opened
 506                  $TRANS_is_opened = true;
 507                  $prev_line_number = $line_number;
 508                  $l++;
 509                  continue;
 510              }
 511  
 512              $char_next2 = translation_get_chars( $line_string, $l, 2 );
 513              if( $TRANS_is_opened &&
 514                  ( $char_next2 == '*/' ||
 515                    ( $prev_line_number != $line_number &&
 516                      ( strpos( $line_string, 'T_' ) !== false || strpos( $line_string, 'TS_' ) !== false ) ) ) )
 517              { // CLOSE TRANS info
 518                  $TRANS_is_opened = false;
 519              }
 520  
 521              if( $TRANS_is_opened )
 522              { // Store chars of the current opened TRANS info
 523                  if( $prev_line_number != $line_number )
 524                  {
 525                      $TRANS_text .= "\n";
 526                      $prev_line_number = $line_number;
 527                  }
 528                  $TRANS_text .= $line_string[ $l ];
 529              }
 530          }
 531  
 532          $prev_line_number = $line_number;
 533      }
 534  }
