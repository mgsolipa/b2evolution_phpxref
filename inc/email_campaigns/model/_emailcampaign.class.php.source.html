  87  	function add_users()
  88      {
  89          global $DB;
  90  
  91          // Get user IDs from current filterset of users list
  92          $new_users_IDs = get_filterset_user_IDs();
  93  
  94          if( count( $new_users_IDs ) )
  95          { // Users are found in the filterset
  96  
  97              // Get all active users which accept newsletter email
  98              $new_users_SQL = get_newsletter_users_sql( $new_users_IDs );
  99              $new_users = $DB->get_col( $new_users_SQL->get() );
 100  
 101              // Remove the users which didn't accept email before
 102              $DB->query( 'DELETE FROM T_email__campaign_send
 103                  WHERE csnd_camp_ID = '.$DB->quote( $this->ID ).'
 104                   AND csnd_emlog_ID IS NULL' );
 105  
 106              // Get users which already accept newsletter email
 107              $old_users = $this->get_users( 'accept' );
 108  
 109              // Exclude old users from new users (To store value of csnd_emlog_ID)
 110              $new_users = array_diff( $new_users, $old_users );
 111  
 112              if( count( $new_users ) )
 113              { // Insert new users for this campaign
 114                  $insert_SQL = 'INSERT INTO T_email__campaign_send ( csnd_camp_ID, csnd_user_ID ) VALUES';
 115                  foreach( $new_users as $user_ID )
 116                  {
 117                      $insert_SQL .= "\n".'( '.$DB->quote( $this->ID ).', '.$DB->quote( $user_ID ).' ),';
 118                  }
 119                  $DB->query( substr( $insert_SQL, 0, -1 ) );
 120              }
 121          }
 122      }

 125      /**
 126       * Get user IDs of this campaign
 127       *
 128       * @param string Type of users:
 129       *   'all'    - All users which assigned to this campaign
 130       *   'accept' - Users which already receive email newsletter
 131       *   'wait'   - Users which still didn't receive email by some reason (Probably their newsletter limit was full)
 132       * @return array user IDs
 133       */
 134  	function get_users( $type = 'all' )
 135      {
 136          global $DB;
 137  
 138          if( !is_null( $this->users ) )
 139          { // Get users from cache
 140              return $this->users[ $type ];
 141          }
 142  
 143          // Get users from DB
 144          $users_SQL = new SQL();
 145          $users_SQL->SELECT( 'csnd_user_ID, csnd_emlog_ID' );
 146          $users_SQL->FROM( 'T_email__campaign_send' );
 147          $users_SQL->WHERE( 'csnd_camp_ID = '.$DB->quote( $this->ID ) );
 148          $users = $DB->get_assoc( $users_SQL->get() );
 149  
 150          $this->users['all'] = array();
 151          $this->users['accept'] = array();
 152          $this->users['wait'] = array();
 153  
 154          foreach( $users as $user_ID => $emlog_ID )
 155          {
 156              $this->users['all'][] = $user_ID;
 157              if( $emlog_ID > 0 )
 158              { // This user already accepted newsletter email
 159                  $this->users['accept'][] = $user_ID;
 160              }
 161              else
 162              { // This user didn't still accept email
 163                  $this->users['wait'][] = $user_ID;
 164              }
 165          }
 166  
 167          return $this->users[ $type ];
 168      }

 285      {
 286          $newsletter_params = array(

 288                  'message_html'     => $this->get( 'email_html' ),
 289                  'message_text'     => $this->get( 'email_text' ),
 290              );
 291  
 292          if( $mode == 'test' )
 293          { // Send a test newsletter
 294              global $current_User;
 295  
 296              $newsletter_params['boundary'] = 'b2evo-'.md5( rand() );
 297              $headers = array( 'Content-Type' => 'multipart/mixed; boundary="'.$newsletter_params['boundary'].'"' );
 298  
 299              $UserCache = & get_UserCache();
 300              if( $test_User = & $UserCache->get_by_ID( $user_ID, false, false ) )
 301              { // Send a test email only when test user exists
 302                  $message = mail_template( 'newsletter', 'auto', $newsletter_params, $test_User );
 303                  return send_mail( $email_address, NULL, $this->get( 'email_title' ), $message, NULL, NULL, $headers );
 304              }
 305              else
 306              { // No test user found
 307                  return false;
 308              }
 309          }
 310          else
 311          { // Send a newsletter to real user
 312              return send_mail_to_User( $user_ID, $this->get( 'email_title' ), 'newsletter', $newsletter_params, false, array(), $email_address );
 313          }

 317      /**
 318       * Send email newsletter for all users of this campaign
 319       *
 320       * @param boolean
 321       */
 322  	function send_all_emails( $display_messages = true )
 323      {
 324          global $DB, $localtimenow, $mail_log_insert_ID;
 325  
 326          // Send emails only for users which still don't accept emails
 327          $user_IDs = $this->get_users( 'wait' );
 328  
 329          if( empty( $user_IDs ) )
 330          { // No users, Exit here
 331              return;
 332          }
 333  
 334          $DB->begin();
 335  
 336          // Update date of sending
 337          $this->set( 'sent_ts', date( 'Y-m-d H:i:s', $localtimenow ) );
 338          $this->dbupdate();
 339  
 340          if( $display_messages )
 341          { // We need in this cache when display the messages
 342              $UserCache = & get_UserCache();
 343          }
 344  
 345          foreach( $user_IDs as $user_ID )
 346          {
 347              $result = $this->send_email( $user_ID );
 348  
 349              if( $result )
 350              { // Email newsletter was sent for user successfully
 351                  if( !empty( $mail_log_insert_ID ) )
 352                  { // ID of last inserted mail log is defined in function mail_log()
 353                      $DB->query( 'UPDATE T_email__campaign_send
 354                              SET csnd_emlog_ID = '.$DB->quote( $mail_log_insert_ID ).'
 355                          WHERE csnd_camp_ID = '.$DB->quote( $this->ID ).'
 356                              AND csnd_user_ID = '.$DB->quote( $user_ID ) );
 357  
 358                      // Update arrays where we store which users accepted email and who waiting it now
 359                      $this->users['accept'][] = $user_ID;
 360                      if( ( $wait_user_ID_key = array_search( $user_ID, $this->users['wait'] ) ) !== false )
 361                      {
 362                          unset( $this->users['wait'][ $wait_user_ID_key ] );
 363                      }
 364                  }
 365              }
 366  
 367              if( $display_messages )
 368              { // Print the messages
 369                  $User = & $UserCache->get_by_ID( $user_ID, false, false );
 370  
 371                  if( $result === true )
 372                  { // Success
 373                      echo sprintf( T_('Email was sent to user: %s'), $User->get_identity_link() ).'<br />';
 374                  }
 375                  else
 376                  { // Failed, Email was NOT sent
 377                      if( ! check_allow_new_email( 'newsletter_limit', 'last_newsletter', $user_ID ) )
 378                      { // Newsletter email is limited today for this user
 379                          echo '<span class="orange">'.sprintf( T_('User %s has already received max # of newsletters today.'), $User->get_identity_link() ).'</span><br />';
 380                      }
 381                      else
 382                      { // Another error
 383                          echo '<span class="red">'.sprintf( T_('Email was not sent to user: %s'), $User->get_identity_link() ).'</span><br />';
 384                      }
 385                  }
 386  
 387                  evo_flush();
 388              }
 389          }
 390  
 391          $DB->commit();
 392      }
