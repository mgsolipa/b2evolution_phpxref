  18  /**
  19   * Get number of users for newsletter from UserList filterset
  20   *
  21   * @return array Numbers of users:
  22   *     'all' - Currently selected recipients (Accounts which accept newsletter emails)
  23   *     'active' - Already received (Accounts which have already been sent this newsletter)
  24   *     'newsletter' - Ready to send (Accounts which have not been sent this newsletter yet)
  25   */
  26  function get_newsletter_users_numbers()
  27  {
  28      $numbers = array(
  29              'all'        => 0,
  30              'active'     => 0,
  31              'newsletter' => 0,
  32          );
  33  
  34      $users_IDs = get_filterset_user_IDs();
  35  
  36      if( count( $users_IDs ) )
  37      { // Found users in the filterset
  38          global $DB;
  39  
  40          $numbers['all'] = count( $users_IDs );
  41  
  42          // Get number of all active users
  43          $SQL = new SQL();
  44          $SQL->SELECT( 'COUNT( * )' );
  45          $SQL->FROM( 'T_users' );
  46          $SQL->WHERE( 'user_ID IN ( '.implode( ', ', $users_IDs ).' )' );
  47          $SQL->WHERE_and( 'user_status IN ( \'activated\', \'autoactivated\' )' );
  48          $numbers['active'] = $DB->get_var( $SQL->get() );
  49  
  50          // Get number of all active users which accept newsletter email
  51          $SQL = get_newsletter_users_sql( $users_IDs );
  52          $SQL->SELECT( 'COUNT( * )' );
  53          $numbers['newsletter'] = $DB->get_var( $SQL->get() );
  54      }
  55  
  56      return $numbers;
  57  }

  60  /**
  61   * Get SQL for active users which accept newsletter email
  62   *
  63   * @param array users IDs
  64   * @return object SQL
  65   */
  66  function get_newsletter_users_sql( $users_IDs )
  67  {
  68      global $Settings;
  69  
  70      $SQL = new SQL();
  71      $SQL->SELECT( 'u.user_ID' );
  72      $SQL->FROM( 'T_users u' );
  73      $SQL->FROM_add( 'LEFT OUTER JOIN T_users__usersettings us ON u.user_ID = us.uset_user_ID' );
  74      $SQL->FROM_add( 'AND us.uset_name = \'newsletter_news\'' );
  75      $SQL->WHERE( 'u.user_ID IN ( '.implode( ', ', $users_IDs ).' )' );
  76      $SQL->WHERE_and( 'u.user_status IN ( \'activated\', \'autoactivated\' )' );
  77      if( $Settings->get( 'def_newsletter_news' ) )
  78      {    // If General setting "newsletter_news" = 1 we also should include all users without defined user's setting "newsletter_news"
  79          $SQL->WHERE_and( '( us.uset_value = 1 OR us.uset_value IS NULL )' );
  80      }
  81      else
  82      {    // If General setting "newsletter_news" = 0 we take only users which accept newsletter email
  83          $SQL->WHERE_and( 'us.uset_value = 1' );
  84      }
  85  
  86      return $SQL;
  87  }

 165  /**
 166   * Get URL for current/next tab of edit campaign view
 167   *
 168   * @param string Current tab: 'info', 'html', 'text', 'send'
 169   * @param integer Campaign ID
 170   * @param string Type of tab: 'current', 'next'
 171   * @param string Glue
 172   * @return string URL
 173   */
 174  function get_campaign_tab_url( $current_tab, $campaign_ID, $type = 'current', $glue = '&' )
 175  {
 176      $modes = get_campaign_edit_modes( $campaign_ID, $glue );
 177  
 178      switch( $type )
 179      {
 180          case 'current':
 181              // Get URL of current tab
 182              if( !empty( $modes[ $current_tab ] ) )
 183              {
 184                  return $modes[ $current_tab ]['href'];
 185              }
 186              break;
 187  
 188          case 'next':
 189          default:
 190              // Get URL of next tab
 191              $this_tab = false;
 192              foreach( $modes as $tab_name => $tab_info )
 193              {
 194                  if( $this_tab )
 195                  { // We find URL for next tab
 196                      return $tab_info['href'];
 197                  }
 198                  if( $tab_name == $current_tab )
 199                  { // The next tab will be what we find
 200                      $this_tab = true;
 201                  }
 202              }
 203          break;
 204      }
 205  
 206      return '';
 207  }
