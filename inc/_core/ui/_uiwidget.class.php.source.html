 119      /**
 120       * Replaces $vars$ with appropriate values.
 121       *
 122       * You can give an alternative string to display, if the substituted variable
 123       * is empty, like:
 124       * <code>$vars "Display if empty"$</code>
 125       *
 126       * @param string template
 127       * @param array optional params that are put into {@link $this->params}
 128       *              to be accessible by derived replace_callback() methods
 129       * @return string The substituted string
 130       */
 131  	function replace_vars( $template, $params = NULL )
 132      {
 133          if( !is_null( $params ) )
 134          {
 135              $this->params = $params;
 136          }
 137  
 138          return preg_replace_callback(
 139              '~\$([a-z_]+)(?:\s+"([^"]*)")?\$~', # pattern
 140              array( $this, 'replace_callback_wrapper' ), # callback
 141              $template );
 142      }

 145      /**
 146       * This is an additional wrapper to {@link replace_vars()} that allows to react
 147       * on the return value of it.
 148       *
 149       * Used by replace_callback()
 150       *
 151       * @param array {@link preg_match() preg match}
 152       * @return string
 153       */
 154  	function replace_callback_wrapper( $match )
 155      {
 156          // Replace the variable with its content (which will be computed on the fly)
 157          $r = $this->replace_callback( $match );
 158  
 159          if( empty($r) )
 160          {    // Empty result
 161              if( !empty($match[2]) )
 162              {
 163                  return $match[2]; // "display if empty"
 164              }
 165  
 166              // return $match[1];
 167          }
 168          return $r;
 169      }

 172      /**
 173       * Callback function used to replace only necessary values in template.
 174       *
 175       * This gets used by {@link replace_vars()} to replace $vars$.
 176       *
 177       * @param array {@link preg_match() preg match}. Index 1 is the template variable.
 178       * @return string to be substituted
 179       */
 180  	function replace_callback( $matches )
 181      {
 182          //echo $matches[1];
 183          switch( $matches[1] )
 184          {
 185              case 'global_icons' :
 186                  // Icons for the whole result set:
 187                  return $this->gen_global_icons();
 188  
 189              case 'title':
 190                  // Results title:
 191                  // Espace $title$ strings from the title to avoid infinite loop replacing
 192                  $escaped_title = str_replace( '$title$', '&#36;title&#36;', $this->title );
 193                  // Replace vars on the title
 194                  $result = $this->replace_vars( $escaped_title );
 195                  // Replace back the $title$ strings and return the result
 196                  return str_replace( '&#36;title&#36;', '$title$', $result );
 197  
 198              case 'no_results':
 199                  // No Results text:
 200                  return $this->no_results_text;
 201  
 202              case 'top_block':
 203                  // Top block:
 204                  return $this->top_block;
 205  
 206              case 'prefix' :
 207                  //prefix
 208                  return $this->param_prefix;
 209  
 210              default:
 211                  return '[Unknown:'.$matches[1].']';
 212          }
 213      }

 216      /**
 217       * Generate img tags for registered icons, through {@link global_icon()}.
 218       *
 219       * This is used by the default callback to replace '$global_icons$'.
 220       */
 221  	function gen_global_icons()
 222      {
 223          global $AdminUI;
 224  
 225          $icons = array();
 226  
 227          foreach( $this->global_icons as $icon_params )
 228          {
 229              if( isset( $this->params, $this->params['global_icons_class'] ) )
 230              { // Append a link class from global params
 231                  if( strpos( $icon_params['link_attribs']['class'], 'btn-' ) !== false )
 232                  {
 233                      $global_icons_class = str_replace( 'btn-default', '', $this->params['global_icons_class'] );
 234                  }
 235                  else
 236                  {
 237                      $global_icons_class = $this->params['global_icons_class'];
 238                  }
 239                  $icon_params['link_attribs']['class'] = ( empty( $icon_params['link_attribs']['class'] ) ? '' : $icon_params['link_attribs']['class'].' ' ).$global_icons_class;
 240              }
 241              if( $icon_params['group'] != NULL )
 242              { // It is a groupped button
 243                  if( ! isset( $icons[ $icon_params['group'] ] ) )
 244                  { // Initialize an array for grouped icons:
 245                      $icons[ $icon_params['group'] ] = array();
 246                  }
 247                  $icons[ $icon_params['group'] ][] = $icon_params;
 248              }
 249              else
 250              { // Separated icon
 251                  $before = $icon_params['link_attribs']['before'];
 252                  $after = $icon_params['link_attribs']['after'];
 253                  unset( $icon_params['link_attribs']['before'] );
 254                  unset( $icon_params['link_attribs']['after'] );
 255                  $icons[] = $before.action_icon( $icon_params['title'], $icon_params['icon'], $icon_params['url'], $icon_params['word'],
 256                              $icon_params['icon_weight'], $icon_params['word_weight'], $icon_params['link_attribs'] ).$after;
 257              }
 258          }
 259  
 260          $r = '';
 261          foreach( $icons as $icon )
 262          {
 263              if( is_array( $icon ) && count( $icon ) )
 264              { // Grouped icons
 265                  $first_icon = $icon[0];
 266                  $r .= '<div class="btn-group dropdown">';
 267                  $r .= '<a href="'.$first_icon['url'].'" class="'.$first_icon['link_attribs']['class'].'" title="'.$first_icon['title'].'">'.get_icon( $first_icon['icon'] ).' '.$first_icon['word'].'</a>';
 268                  $r .= '<button type="button" class="btn btn-sm btn-default dropdown-toggle" data-toggle="dropdown" aria-expanded="false">'
 269                              .' <span class="caret"></span></button>';
 270                  $r .= '<ul class="dropdown-menu dropdown-menu-right" role="menu">';
 271                  foreach( $icon as $grouped_icon )
 272                  {
 273                      $r .= '<li role="presentation"><a href="'.$grouped_icon['url'].'" role="menuitem" tabindex="-1" title="'.$grouped_icon['title'].'">'.get_icon( $grouped_icon['icon'] ).' '.$grouped_icon['word'].'</a></li>';
 274                  }
 275                  $r .= '</ul>';
 276                  $r .= '</div>';
 277              }
 278              else
 279              { // Separated icon
 280                  $r .= $icon;
 281              }
 282          }
 283  
 284          return $r;
 285      }

1127       *
1128       * This is one of the key functions to look at when you want to use the Results class.
1129       * - #var#
1130       */
1131  	function parse_class_content( $content )
1132      {
1133          // Make variable substitution for RAWS:
1134          while (preg_match('!\# (\w+) \#!ix', $content, $matchesarray))
1135          { // Replace all matches to the content of the current row's cell. That means that several variables can be inserted to the class.
1136              if (! empty($this->rows[$this->current_idx]->$matchesarray[1]))
1137              {
1138                  $content = str_replace($matchesarray[0],$this->rows[$this->current_idx]->$matchesarray[1] , $content);
1139              }
1140              else
1141              {
1142                  $content = str_replace($matchesarray[0], 'NULL' , $content);
1143              }
1144          }
1145  
1146          while (preg_match('#% (.+?) %#ix', $content, $matchesarray))
1147          {
1148               eval('$result = '.$matchesarray[1].';');
1149               $content = str_replace($matchesarray[0],$result, $content);
