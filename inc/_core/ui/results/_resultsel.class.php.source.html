 218  /**
 219   * Display a checkbox allowing to add the item to a selection
 220   *
 221   * Only one checkbox will be displayed for each ID.
 222   * IDs which are already in the selection will be pre-checked.
 223   *
 224   * @deprecated should go into ResultSel class
 225   *
 226   * @param integer item ID
 227   * @param string item name / prefix for form values
 228   * @return string the correct input tag
 229   */
 230  function selection_checkbox( $item_ID, $param_prefix )
 231  {
 232      global $current_User;
 233      // List of checkboxes to pre-check:
 234      global $cols_check;
 235      // List of already displayed checkboxes (can be used outside to get a list of checkboxes which have been displayed)
 236      global $item_ID_array;
 237  
 238      if( in_array( $item_ID, $item_ID_array ) )
 239      {    // We have already displayed a checkbox for this ID
 240          return '&nbsp;';    // nbsp is for IE...
 241      }
 242  
 243      $item_ID_array[] = $item_ID; //construction of the ID list
 244  
 245      $r = '';
 246  
 247      if( $current_User->check_perm( 'selections', 'edit' ) )
 248      {    // User is allowed to edit
 249          $r .= '<span name="surround_check" class="checkbox_surround_init"><input type="checkbox" class="checkbox" name="'.$param_prefix.'items[]" value="'.$item_ID.'"';
 250          if( in_array( $item_ID, $cols_check ) )
 251          {    // already in selection:
 252              $r .= ' checked="checked" ';
 253          }
 254          $r .= ' /></span>';
 255      }
 256      else
 257      {    // User CANNOT edit:
 258          if( in_array( $item_ID, $cols_check ) )
 259          {    // already in selection:
 260              $r .= '*';
 261          }
 262          else
 263          {
 264              $r .= '&nbsp;';
 265          }
 266      }
 267  
 268      return $r;
 269  }

 272  /**
 273   * Creates the select tag in the search menu and fills it with the appropriate option tags
 274   *
 275   * @param string the selection category prefix(the one used during the construction of the result object)
 276   * @param string the name of the table containing the selections
 277   * @param string the name field  in the selection table
 278   * @param string the id field in the selection table
 279   * @param integer the current selection id
 280   */
 281  function selection_select_tag(
 282                                                              $category_prefix,
 283                                                              $selections_table,
 284                                                              $selections_table_name,
 285                                                              $selections_table_ID,
 286                                                              $selection_ID
 287                                                              )
 288  {
 289      global $DB, $selection_name, $current_User;
 290  
 291      $r = T_('Selection');
 292      $r .= ' <select name="selection_'.$category_prefix.'ID"
 293                      onchange="selform = get_form( this );
 294                          selform.elements[\''.$category_prefix.'previous_sel_ID\'].value=-1;
 295                          selform.submit()" >'."\n";
 296      // in the onchange attribute, option_db is set to -1 to avoid updating the database
 297  
 298      if( $current_User->check_perm( 'selections', 'edit' ) )
 299      {    // User is allowed to edit
 300          $r .= '<option value="0">'.T_('New selection')."</option>\n";
 301      }
 302      else
 303      {    // User CANNOT edit:
 304          $r .= '<option value="0">'.T_('None')."</option>\n";
 305      }
 306  
 307      $sql = 'SELECT * FROM '.$selections_table.' ORDER BY '.$selections_table_name;
 308      $rows = $DB->get_results( $sql );
 309      if( !empty( $rows ) )
 310      {
 311          $selection_name = '';
 312          foreach( $rows as $row )
 313          { // construction of the option tags
 314              if( $row->$selections_table_ID == $selection_ID )
 315              { // option selected by default
 316                  $selected = ' selected="selected" ';
 317                  $selection_name = $row->$selections_table_name;
 318              }
 319              else
 320              {
 321                  $selected = '';
 322              }
 323              $r .= '<option value="'.$row->$selections_table_ID.'" '.$selected.' >'.$row->$selections_table_name."</option>\n";
 324          }
 325      }
 326      $r .= "</select>\n\n";
 327  
 328      echo $r;
 329  
 330      return $selection_name;
 331  }

 343  function handle_selection_actions( $selection_ID, $prefix, $prefix_sel )
 344  {
 345      // previous selection ID, need it to check for updating sel (no javascript possibility)
 346      $previous_sel_ID = param( $prefix.'previous_sel_ID', 'integer', 0, true );
 347  
 348      if( $selection_ID == $previous_sel_ID ) // the databse has to be updated
 349      {    // The selected ID is the same than the edited one in the previous page
 350          if( $selection_ID == 0 )
 351          { // A new selection must be created
 352              $action = 'create';
 353          }
 354          elseif( $selection_ID >0 )
 355          { // An existing selection is being updated
 356              $action = 'update';
 357          }
 358          // Get the selection name
 359          $selection_name = param( 'selection_'.$prefix.'name', 'string', '', true );
 360  
 361          // Do the slection action
 362          selection_action( $action, $selection_ID, $selection_name, $prefix, $prefix_sel );
 363      }
 364  }

 367  /**
 368   * Manages the various database changes to make on selections
 369   *
 370   * @param string the action currently effectuated
 371   * @param integer the current selection id
 372   * @param string the current selection name
 373   * @param string prefix (cont_, etab_, firm_, ..)
 374   * @param string selection prefix (cocs_, etes_, fifs_, ..)
 375   */
 376  function selection_action( $action, $selection_ID, $selection_name, $prefix, $prefix_sel )
 377  { // the form has been submitted to act on the database and not only to change the display
 378  
 379      global $DB, $Messages, $confirm, $item_ID_list, $current_User;
 380  
 381      $items = param( $prefix.'items', 'array:string', array(), false );    // do NOT memorize // ?????????????
 382      param( 'item_ID_list', 'string', '', false );
 383  
 384      $current_User->check_perm( 'selections', 'edit', true );
 385  
 386  
 387      // Set global vars, selection_.prefix.ID, selection_.prefix_name
 388      $selection_prefix_ID = 'selection_'.$prefix.'ID';
 389      $selection_prefix_name = 'selection_'.$prefix.'name';
 390      global $$selection_prefix_ID, $$selection_prefix_name;
 391  
 392      // Moche ms bon...
 393      $selections_table = 'T_'.substr($prefix,0,strlen($prefix)-1).'selections';
 394      $sel_table = 'T_'.$prefix.substr($prefix,0,1).'sel';
 395      $selections_table_name = substr($prefix,0,1).'sel_name';
 396      $selections_table_id = substr($prefix,0,1).'sel_ID';
 397      $sel_table_selection = $prefix_sel.$selections_table_id;
 398      $sel_table_item = $prefix_sel.$prefix.'ID';
 399  
 400  
 401      switch( $action )
 402      {
 403  
 404          // creation of a new selection
 405          case 'create':
 406  
 407              if( empty($selection_name) )
 408              {    // No name provided:
 409                  $Messages->add( T_('Cannot create a selection with an empty name'), 'error' );
 410                  // Abord creation!
 411                  break;
 412              }
 413  
 414              $sql_selections = "INSERT INTO $selections_table ( $selections_table_name )
 415                                                              VALUES( ".$DB->quote($selection_name)." ) ";         // construction of the query
 416              $DB->query( $sql_selections ); // insertion of a new selection in the database
 417  
 418              $selection_ID = mysqli_insert_id($DB->dbhandle); // id generated by the last sql query
 419  
 420              if( !empty( $items ) )
 421              { // nothing must be inserted if no items are selected
 422                  $sql_sel = 'INSERT INTO '.$sel_table.'( '.$sel_table_item.', '.$sel_table_selection .')  VALUES ';
 423                  $sel_array = array();
 424  
 425                  $i = 0;
 426                  foreach( $items as $item )
 427                  { // construction of the sql query depending on selected values in the result table
 428                      $sel_array[$i++] = ' ('.$item.','.$selection_ID.' ) ';
 429                  }
 430                  $sql_sel .= implode( $sel_array, ',' );
 431                  $DB->query( $sql_sel ); // insertion of the relation between selections and items in the database
 432              }
 433  
 434              // Set $selection_.$prefix.ID var
 435              $sel_prefix_ID = 'selection_'.$prefix.'ID';
 436              $$sel_prefix_ID = $selection_ID;
 437              // Set $selection_.$prefix.name var
 438              $sel_prefix_name = 'selection_'.$prefix.'name';
 439              $$sel_prefix_name = $selection_name;
 440  
 441              $Messages->add( T_('Selection created.'), 'success' );
 442  
 443              break;
 444  
 445  
 446          case 'edit':
 447          case 'update':
 448              // update of an existing selection
 449              $DB->begin();
 450  
 451               if( empty($selection_name) )
 452              {    // No name provided:
 453                  $Messages->add( T_('Please provide a selection name.'), 'error' );
 454              }
 455              else
 456              {    // Update name:
 457                  $sql_selections = "UPDATE $selections_table
 458                                                              SET $selections_table_name = ".$DB->quote($selection_name)."
 459                                                          WHERE $selections_table_id = $selection_ID"; // construction of the update query
 460                  $DB->query( $sql_selections );
 461              }
 462  
 463              if( preg_match( '#[0-9,]+#', $item_ID_list ) )
 464              { // check the format of the item list to avoid sql injection
 465                  $sql_delete = 'DELETE FROM '.$sel_table.' WHERE '.$sel_table_selection.' = '.$selection_ID
 466                              .' AND '.$sel_table_item.' IN ('.$item_ID_list.')'; // deletion of the former db entries
 467                  $DB->query( $sql_delete );
 468  
 469                  $Messages->add( T_('Obsolete selection entries deleted.'), 'success' );
 470              }
 471  
 472              if( !empty( $items ) )
 473              { // there have been some items selected in the result table: they must be inserted into the database
 474                  $sql_sel = 'INSERT INTO '.$sel_table.'( '.$sel_table_item.', '.$sel_table_selection .')  VALUES ';
 475                  $sel_array = array();
 476  
 477                  foreach( $items as $item )
 478                  { // construction of the sql query depending on selected values in the result table
 479                      $sel_array[] = ' ( '.$item.', '.$selection_ID.' ) ';
 480                  }
 481                  $sql_sel .= implode( $sel_array, ',' );
 482                  $DB->query( $sql_sel ); // insertion of the relation between selections and items in the database
 483  
 484                  $Messages->add( T_('New selections entries inserted.'), 'success' );
 485              }
 486  
 487              $DB->commit();
 488              break;
 489  
 490  
 491          case 'copy':
 492              // creation of a new selection with the same name
 493              $sql_selections = 'INSERT INTO '.$selections_table.'('.$selections_table_name.
 494                                                  ') VALUES( "'.$selection_name.'" )';
 495              $DB->query( $sql_selections );
 496              $Messages->add( T_('Selection copied.'), 'success' );
 497  
 498              $new_selection_ID = mysqli_insert_id($DB->dbhandle);// gets the new selection id
 499  
 500              // creation of the links between the new selection and the selected items
 501              $sql_sel = 'INSERT INTO '.$sel_table.'( '.$sel_table_item.', '.$sel_table_selection.' ) '
 502                                   .'SELECT '.$sel_table_item.', '.$new_selection_ID.' FROM '.$sel_table.' WHERE '
 503                                                  .$sel_table_selection.'='.$selection_ID;
 504              $DB->query( $sql_sel );
 505              $Messages->add( T_('Selection links copied.'), 'success' );
 506  
 507              $selection_ID = $new_selection_ID;
 508  
 509              break;
 510  
 511  
 512          case 'delete':
 513              // deletion of the selection
 514              if( !$confirm )
 515              { // ask for confirmation before deleting
 516                  ?>
 517                  <div class="panelinfo">
 518                      <h3><?php printf( T_('Delete selection &laquo;%s&raquo;?'), $selection_name )?></h3>
 519  
 520                      <p><?php echo T_('Warning').': '.T_('Cascading deletes!') ?></p>
 521  
 522                      <p><?php echo T_('THIS CANNOT BE UNDONE!') ?></p>
 523  
 524                      <p>
 525  
 526                  <?php
 527                      $Form = new Form( regenerate_url('', '', '', '&'), 'form_confirm', 'post', '' );
 528  
 529                      $action = '';
 530  
 531                      $Form->begin_form( 'inline' );
 532                      $Form->hidden( 'action', 'delete' );
 533                      $Form->hidden( 'selection_ID', $selection_ID );
 534                      $Form->hidden( 'selection_name', $selection_name );
 535                      $Form->hidden( 'confirm', 1 );
 536                      $Form->button( array( 'submit', '', T_('I am sure!'), 'DeleteButton btn-danger' ) );
 537                      $Form->end_form();
 538  
 539                      unset( $Form );
 540  
 541                      $Form = new Form( regenerate_url('', '', '', '&'), 'form_cancel', 'post', '' );
 542  
 543                      $Form->begin_form( 'inline' );
 544                      $Form->button( array( 'submit', '', T_('CANCEL'), 'CancelButton' ) );
 545                      $Form->end_form();
 546                  ?>
 547  
 548                  </p>
 549  
 550                  </div>
 551                  <?php
 552              }
 553              else
 554              { // the deletion has been confirmed
 555                  $sql_sel = 'DELETE FROM '.$sel_table.' WHERE '.$sel_table_selection.'='.$selection_ID;
 556                  $DB->query( $sql_sel );// deletion of the links between the selection and the selected items
 557                  $Messages->add( T_('Selection attachments deleted.'), 'success' );
 558  
 559                  $sql_selections = 'DELETE FROM '.$selections_table.' WHERE '.$selections_table_id.'='.$selection_ID;
 560                  $DB->query( $sql_selections );// deletion of the selection
 561                  $Messages->add( T_('Selection deleted.'), 'success' );
 562              }
 563  
 564              $selection_ID = -1;
 565  
 566              break;
 567  
 568          default:
 569              break;
 570      }
 571  
 572      return $selection_ID;
 573  
 574  }
