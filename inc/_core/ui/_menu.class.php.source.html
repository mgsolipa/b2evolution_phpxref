  38      /**
  39       * Add menu entries to the list of entries for a given path.
  40       *
  41       * @param NULL|string|array The path to add the entries to. See {@link get_node_by_path()}.
  42       * @param array Menu entries to add (key (string) => entry (array)).
  43       *   An entry can have the following keys:
  44       *     'text': Text/Caption for this entry.
  45       *     'href': The link for this entry.
  46       *     'entries': array of sub-entries
  47       *     DEPRECATED 'style': CSS style for this entry.
  48       *     DEPRECATED 'onclick': onclick property for this entry.
  49       *     DEPRECATED 'name': name attribute of the link/entry.
  50       * @param string Node name after which we should insert the menu entries, NULL - to insert to the end
  51       */
  52  	function add_menu_entries( $path, $new_entries, $after_node = NULL )
  53      {
  54          // Get a reference to the node in the menu list.
  55          $node = & $this->get_node_by_path( $path, true );
  56  
  57          /*
  58          if( !is_array($node) )
  59          {
  60              debug_die( 'add_menu_entries() with non-existing path!' );
  61          }
  62          */
  63  
  64          $new_entires_are_inserted = false;
  65  
  66          if( !is_null( $after_node ) )
  67          {    // We should insert new entries after specific node
  68              $new_node = $node;
  69              $new_node['entries'] = array();
  70              if( isset( $node['entries'] ) )
  71              {
  72                  foreach( $node['entries'] as $node_key => $node_entry )
  73                  {
  74                      $new_node['entries'][ $node_key ] = $node_entry;
  75                      if( $node_key == $after_node )
  76                      {    // Insert new entires here after specific node
  77                          foreach( $new_entries as $l_key => $l_new_entry )
  78                          {
  79                              $new_node['entries'][$l_key] = $l_new_entry;
  80                          }
  81                          $new_entires_are_inserted = true;
  82                      }
  83                  }
  84              }
  85              $node = $new_node;
  86          }
  87  
  88          if( !$new_entires_are_inserted )
  89          {    // Insert new entries to the end if they are still not inserted after specific node
  90              foreach( $new_entries as $l_key => $l_new_entry )
  91              {
  92                  $node['entries'][$l_key] = $l_new_entry;
  93              }
  94          }
  95      }

  97      /**
  98       * Insert new menu entries right after the menu entry passed as path
  99       *
 100       * @param NULL|string|array The path. See {@link get_node_by_path()}.
 101       * @param new entries
 102       * @param position after which to insert new entries. If not given, new entries is inserted after Path
 103       * @returns boolean Whether inserting was successfull.
 104       */
 105  	function insert_menu_entries_after( $path, $new_entries, $index = false )
 106      {
 107          $menu_item = '';
 108          if( $index === false )
 109          { // get menu item after which to insert new entries
 110  
 111              if( is_array( $path ) )
 112              {
 113                  $menu_item = array_pop( $path );
 114              }
 115              elseif( is_string( $path ) )
 116              {
 117                  $menu_item = $path;
 118                  $path = NULL;
 119              }
 120          }
 121          $menu = & $this->get_node_by_path( $path );
 122          if( $menu === false )
 123          { // no such path
 124              return false;
 125          }
 126          $entries = & $menu['entries'];
 127          if( $menu_item )
 128          { // find index of menu itemafter which to insert new entries
 129              $keys = array_keys( $entries );
 130  
 131              if( !empty( $keys ) ) $index = array_search( $menu_item, $keys );
 132          }
 133  
 134          if( ( $index === false ) || ($index === NULL) )
 135          {
 136              return false;
 137          }
 138  
 139          // make new menu entries
 140          $menu['entries'] = array_merge(
 141              array_slice( $entries, 0, $index + 1 ),
 142              $new_entries,
 143              array_slice( $entries, $index )
 144          );
 145  
 146          return true;
 147      }

 149      /**
 150       * Get the reference of a node from the menu entries using a path.
 151       *
 152       * @param array|string|NULL The path. NULL means root, string means child of root,
 153       *                          array means path below root. (eg <code>array('options', 'general')</code>).
 154       * @param boolean Should the node be created if it does not exist already?
 155       * @return array|false The node as array or false, if the path does not exist (and we do not $createIfNotExisting).
 156       */
 157      function & get_node_by_path( $path, $createIfNotExisting = false )
 158      {
 159          if( is_null($path) )
 160          { // root element
 161              $path = array();
 162          }
 163          elseif( ! is_array($path) )
 164          {
 165              $path = array($path);
 166          }
 167  
 168          $node = & $this->_menus;
 169          foreach( $path as $lStep )
 170          {
 171              if( ! isset($node['entries'][$lStep]) )
 172              {
 173                  if( $createIfNotExisting )
 174                  {
 175                      $node['entries'][$lStep] = array();
 176                  }
 177                  else
 178                  {
 179                      $r = false;
 180                      return $r;
 181                  }
 182              }
 183              $node = & $node['entries'][$lStep];
 184          }
 185  
 186          return $node;
 187      }

 223      /**
 224       * Get the HTML for the menu entries of a specific path.
 225       *
 226       * @param NULL|string|array The path. See {@link get_node_by_path()}.
 227       * @param string Template name, see {@link get_template()}.
 228       * @param integer Level
 229       * @param boolean TRUE - to get template for empty menu, used to hide menus
 230       * @return string The HTML for the menu.
 231       */
 232  	function get_html_menu( $path = NULL, $template = 'main', $level = 0, $force_empty = false )
 233      {
 234          global $current_User;
 235  
 236          $r = '';
 237  
 238          if( is_null($path) )
 239          {
 240              $path = array();
 241          }
 242          elseif( ! is_array( $path ) )
 243          {
 244              $path = array( $path );
 245          }
 246  
 247          $templateForLevel = $this->get_template( $template, $level );
 248  
 249          if( $force_empty || !( $menuEntries = $this->get_menu_entries($path) ) )
 250          {    // No menu entries at this level
 251              if( isset($templateForLevel['empty']) )
 252              {
 253                  $r .= $templateForLevel['empty'];
 254              }
 255          }
 256          else
 257          {    // There are entries to display:
 258              $r .= $templateForLevel['before'];
 259  
 260              $selected = $this->get_selected($path);
 261  
 262              foreach( $menuEntries as $loop_key => $loop_details )
 263              {
 264                  if( empty( $loop_details ) )
 265                  { // Empty placeholder, skip it. Might happen if the files module is disabled for example, then we had a file placeholder
 266                      // in the blog menu that will never be used. So don't display it...
 267                      continue;
 268                  }
 269  
 270                  if( !empty( $loop_details['separator'] ) )
 271                  { // Separator
 272                      $r .= $templateForLevel['separator'];
 273                      continue;
 274                  }
 275  
 276  
 277                  // Menu entry
 278                  if( isset( $loop_details['href'] ) )
 279                  {
 280                      $href = $loop_details['href'];
 281                  }
 282                  elseif( ! empty( $loop_details['href_eval'] ) )
 283                  { // Useful for passing dynamic context vars (fp>> I AM using it)
 284                      $href = eval( $loop_details['href_eval'] );
 285                  }
 286                  else
 287                  {
 288                      $href = NULL;
 289                  }
 290  
 291                  $anchor = '<a';
 292  
 293                  if( !empty($href) )
 294                  {
 295                      $anchor .= ' href="'.$href.'"';
 296                  }
 297                  if( isset($loop_details['target']) )
 298                  {
 299                      $anchor .= ' target="'.$loop_details['target'].'"';
 300                  }
 301                  if( isset($loop_details['style']) )
 302                  {
 303                      $anchor .= ' style="'.$loop_details['style'].'"';
 304                  }
 305                  if( isset($loop_details['onclick']) )
 306                  {
 307                      $anchor .= ' onclick="'.$loop_details['onclick'].'"';
 308                  }
 309                  if( isset($loop_details['name']) )
 310                  {
 311                      $anchor .= ' name="'.$loop_details['name'].'"';
 312                  }
 313                  if( isset($loop_details['title']) )
 314                  {
 315                      $anchor .= ' title="'.$loop_details['title'].'"';
 316                  }
 317  
 318                  // CLASS
 319                  $class = '';
 320                  if( !empty( $loop_details['class'] ) )
 321                  { // disabled
 322                      $class .= ' '.$loop_details['class'];
 323                  }
 324                  if( !empty( $loop_details['disabled'] ) )
 325                  { // disabled
 326                      $class .= ' '.$templateForLevel['disabled_class'];
 327                  }
 328                  if( ! empty( $class ) )
 329                  { // disabled
 330                      $anchor .= ' class="'.trim($class).'"';
 331                  }
 332  
 333                  $anchor .= '>'.(isset($loop_details['text']) ? format_to_output( $loop_details['text'], 'htmlbody' ) : '?');
 334                  $anchor_end = '</a>';
 335  
 336                  if( $loop_key == $selected )
 337                  { // Highlight selected entry
 338                      if( isset( $templateForLevel['_props']['recurse'] )
 339                              && $templateForLevel['_props']['recurse'] != 'no'
 340                              && ( $recursePath = array_merge( $path, array($loop_key) ) )
 341                              && $this->get_menu_entries($recursePath) )
 342                      {
 343                          $r .= isset($templateForLevel['beforeEachSelWithSub']) ? $templateForLevel['beforeEachSelWithSub'] : $templateForLevel['beforeEachSel'];
 344                          $r .= $anchor;
 345  
 346                          if( $templateForLevel['_props']['recurse'] != 'no' && // Recurse:
 347                                ( ! isset( $templateForLevel['_props']['recurse_level'] ) ||
 348                                  ( isset( $templateForLevel['_props']['recurse_level'] ) &&
 349                                    $templateForLevel['_props']['recurse_level'] > $level + 1 ) ) )
 350                          { // Display submenus if this level is not limited by param 'recurse_level'
 351                              $r .= $templateForLevel['arrow_level_'.( $level + 1 )].'</a>';
 352                              $r .= $this->get_html_menu( $recursePath, $template, $level+1 );
 353                          }
 354                          else
 355                          { // End anchor without sub menus
 356                              $r .= $anchor_end;
 357                          }
 358  
 359                          $r .= isset($templateForLevel['afterEachSelWithSub']) ? $templateForLevel['afterEachSelWithSub'] : $templateForLevel['afterEachSel'];
 360                      }
 361                      else
 362                      {
 363                          if( isset( $loop_details['order'] ) && $loop_details['order'] == 'group_last' &&
 364                              isset( $templateForLevel['beforeEachSelGrpLast'], $templateForLevel['afterEachSelGrpLast'] ) )
 365                          { // This selected menu item is last in a group
 366                              $r .= $templateForLevel['beforeEachSelGrpLast'];
 367                              $r .= $anchor.$anchor_end;
 368                              $r .= $templateForLevel['afterEachSelGrpLast'];
 369                          }
 370                          else
 371                          { // Normal selected menu item
 372                              $r .= $templateForLevel['beforeEachSel'];
 373                              $r .= $anchor.$anchor_end;
 374                              $r .= $templateForLevel['afterEachSel'];
 375                          }
 376                      }
 377                  }
 378                  else
 379                  {    // Not selected entry
 380                      if( isset( $templateForLevel['_props']['recurse'] )
 381                              && $templateForLevel['_props']['recurse'] == 'always'
 382                              && ( $recursePath = array_merge( $path, array($loop_key) ) )
 383                              && $this->get_menu_entries($recursePath) )
 384                      {
 385                          $r .= isset($templateForLevel['beforeEachWithSub']) ? $templateForLevel['beforeEachWithSub'] : $templateForLevel['beforeEachSel'];
 386                          $r .= $anchor.$templateForLevel['arrow_level_'.( $level + 1 )].'</a>';
 387                          // recurse:
 388                          $r .= $this->get_html_menu( $recursePath, $template, $level+1 );
 389                          $r .= isset($templateForLevel['afterEachWithSub']) ? $templateForLevel['afterEachWithSub'] : $templateForLevel['afterEachSel'];
 390                      }
 391                      else
 392                      {
 393                          if( isset( $loop_details['order'] ) && $loop_details['order'] == 'group_last' &&
 394                              isset( $templateForLevel['beforeEachGrpLast'], $templateForLevel['afterEachGrpLast'] ) )
 395                          { // This menu item is last in a group
 396                              $r .= $templateForLevel['beforeEachGrpLast'];
 397                              $r .= $anchor.$anchor_end;
 398                              $r .= $templateForLevel['afterEachGrpLast'];
 399                          }
 400                          else
 401                          { // Normal menu item
 402                              $r .= $templateForLevel['beforeEach'];
 403                              $r .= $anchor.$anchor_end;
 404                              $r .= $templateForLevel['afterEach'];
 405                          }
 406                      }
 407                  }
 408  
 409                  // Additional attribures for each menu entry
 410                  $entry_attrs = '';
 411                  $entry_class = '';
 412                  if( ! empty( $loop_details['entry_class'] ) )
 413                  { // Css class for entry
 414                      if( strpos( $r, '$entry_class$' ) === false )
 415                      {
 416                          $entry_attrs .= ' class="'.$loop_details['entry_class'].'"';
 417                      }
 418                      $entry_class .= ' '.$loop_details['entry_class'];
 419                  }
 420                  $r = str_replace( array( '$entry_attrs$', '$entry_class$' ), array( $entry_attrs, $entry_class ), $r );
 421              }
 422              $r .= $templateForLevel['after'];
 423          }
 424  
 425          return $r;
 426      }

 430      /**
 431       * Get a template by name.
 432       *
 433       * This is a method (and not a member array) to allow dynamic generation and T_()
 434       *
 435       * @param string Name of the template ('main', 'sub')
 436       * @return array Associative array which defines layout and optionally properties.
 437       */
 438  	function get_template( $name, $level = 0 )
 439      {
 440          switch( $name )
 441          {
 442              case 'evobar-menu-right':
 443                  $arrow_level_2 = '<span class="evobar-icon-left fa fa-caret-left"></span>';
 444              case 'evobar-menu-left':
 445                  return array(
 446                      'before'         => '<ul class="evobar-menu '.$name.'">',
 447                      'after'          => '</ul>',
 448                      'beforeEach'     => '<li$entry_attrs$>',
 449                      'afterEach'      => '</li>',
 450                      'beforeEachSel'  => '<li class="current$entry_class$"$entry_attrs$>',
 451                      'afterEachSel'   => '</li>',
 452                      'separator'      => '<li class="separator"><hr /></li>',
 453                      'arrow_level_1'  => '<span class="evobar-icon-down fa fa-caret-down"></span>',
 454                      'arrow_level_2'  => isset( $arrow_level_2 ) ? $arrow_level_2 : '<span class="evobar-icon-right fa fa-caret-right"></span>',
 455                      'disabled_class' => 'disabled',
 456                      '_props' => array(
 457                          'recurse' => 'always',  // options are: 'no' 'always' or 'intoselected'
 458                      ),
 459                  );
 460                  break;
 461  
 462              default:
 463                  debug_die( 'Unknown $name for Menu::get_template(): '.var_export($name, true) );
 464          }
 465      }
