  23   * @deprecated Deprecated by (@link Form::text_input())
  24   *
  25   * @param string the name of the input field
  26   * @param string initial value
  27   * @param integer size of the input field
  28   * @param string label displayed in front of the field
  29   * @param string note displayed with field
  30   * @param integer max length of the value (if 0 field_size will be used!)
  31   * @param string the CSS class to use
  32   * @param string input type (only 'text' or 'password' makes sense)
  33   * @param boolean display (default) or return

  36  function form_text( $field_name, $field_value, $field_size, $field_label, $field_note = '',
  37                                          $field_maxlength = 0 , $field_class = '', $inputtype = 'text', $output = true  )
  38  {
  39      if( $field_maxlength == 0 )
  40          $field_maxlength = $field_size;
  41  
  42      // question: is it necessary to enclose each field in a fieldset.
  43      // fplanque> YES, for CSS
  44      // shouldn't there be a fieldset for a set of field (i.e. all fields
  45      // in the form)?
  46      // fplanque>> Create a new 'simple' layout if this is what you want
  47  
  48      $r = "<fieldset>\n"
  49              .'<div class="label"><label for="'.$field_name.'">'.$field_label.":</label></div>\n"
  50              .'<div class="input"><input type="'.$inputtype.'" name="'.$field_name
  51              .'" id="'.$field_name.'" size="'.$field_size.'" maxlength="'.$field_maxlength
  52              .'" value="'.format_to_output($field_value, 'formvalue').'"';
  53      if( !empty($field_class) )
  54      {
  55          $r .= ' class="'.$field_class.'"';
  56      }
  57      $r .= " />\n";
  58  
  59      if( !empty( $field_note ) )
  60      {
  61          $r .= '<span class="notes">'.$field_note.'</span>';
  62      }
  63  
  64      $r .= "</div>\n</fieldset>\n\n";
  65  
  66      if( $output )
  67      {
  68          echo $r;
  69          return true;
  70      }
  71      else
  72      {
  73          return $r;

 216  /**
 217   * Builds a form header and puts GET params of $action into hidden form inputs
 218   *
 219   * @deprecated Deprecated by (@link Form::begin_form())
 220   * waltercruz> still used by inc/widgets/widgets/_coll_search_form.widget.php
 221   */
 222  function form_formstart( $action, $class = '', $name = '', $method = 'get', $id = '' )
 223  {
 224      if( $method == 'get' )
 225      {
 226          $action = explode( '?', $action );
 227          if( isset($action[1]) )
 228          { // we have GET params in $action
 229              $getparams = preg_split( '/&amp;|&/i', $action[1], -1, PREG_SPLIT_NO_EMPTY );
 230          }
 231          $action = $action[0];
 232      }
 233  
 234      echo '<form action="'.$action.'" method="'.$method.'"';
 235  
 236      // this is not xhtml strict, see: http://forums.b2evolution.net//viewtopic.php?t=8475
 237      // if( !empty($name) ) echo ' name="'.$name.'"';
 238      if( !empty($id) ) echo ' id="'.$id.'"';
 239      if( !empty($class) ) echo ' class="'.$class.'"';
 240  
 241      echo '>';
 242  
 243      if( isset($getparams) )
 244      { // These need to be wrapped in a div to validate xhtml strict
 245          echo '<div>';
 246          foreach( $getparams as $param)
 247          {
 248              $param = explode( '=', $param );
 249              if( isset($param[1]) )
 250              {
 251                  echo '<input type="hidden" name="'.$param[0].'" value="'.$param[1].'" />';
 252              }
 253          }
 254          // close the div
 255          echo '</div>';
 256      }
 257  }

 261   * Builds a textarea field.
 262   *
 263   * @deprecated Deprecated by (@link Form::textarea_input())
 264   *
 265   * @param string the name of the input field
 266   * @param string initial value
 267   * @param integer rows of the textarea field
 268   * @param string label displayed in front of the field
 269   * @param array params
 270   */
 271  function form_textarea( $field_name, $field_value, $field_rows, $field_label, $field_params = array() )
 272  {
 273  
 274      $textarea_rows = '';
 275      if( !empty( $field_rows ) )
 276      {
 277          $textarea_rows = ' rows="'.$field_rows.'"';
 278      }
 279  
 280      $textarea_cols = '';
 281      if( !empty( $field_params['cols'] ) )
 282      {
 283          $textarea_cols = ' cols="'.$field_params['cols'].'"';
 284      }
 285  
 286      $textarea_class = '';
 287      if( !empty( $field_params['class'] ) )
 288      {
 289          $textarea_class = ' class="'.$field_params['class'].'"';
 290      }
 291  
 292      $r = "<fieldset>\n"
 293              .'<div class="label"><label for="'.$field_name.'">'.$field_label.":</label></div>\n"
 294              .'<div class="input"><textarea name="'.$field_name.'" id="'.$field_name.'"'.$textarea_rows.$textarea_cols.$textarea_class.'>'
 295              .format_to_output($field_value, 'formvalue')
 296              .'</textarea>'."\n";
 297  
 298      if( !empty( $field_params['note'] ) )
 299      {
 300          $r .= '<span class="notes">'.$field_params['note'].'</span>';
 301      }
 302  
 303      $r .= "</div>\n</fieldset>\n\n";
 304  
 305      echo $r;
 306  }
