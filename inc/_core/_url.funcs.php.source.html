  21  /**
  22   * Check the validity of a given URL
  23   *
  24   * Checks allowed URI schemes and URL ban list.
  25   * URL can be empty.
  26   *
  27   * Note: We have a problem when trying to "antispam" a keyword which is already blacklisted
  28   * If that keyword appears in the URL... then the next page has a bad referer! :/
  29   *
  30   * {@internal This function gets tested in misc.funcs.simpletest.php.}}
  31   *
  32   * @param string Url to validate
  33   * @param string Context ("posting", "commenting", "download_src", "http-https")
  34   * @param boolean also do an antispam check on the url
  35   * @return mixed false (which means OK) or error message
  36   */
  37  function validate_url( $url, $context = 'posting', $antispam_check = true )
  38  {
  39      global $Debuglog, $debug;
  40  
  41      if( empty($url) )
  42      { // Empty URL, no problem
  43          return false;
  44      }
  45  
  46      // Do not give verbose info for comments, unless debug is enabled.
  47      $verbose = $debug || $context != 'commenting';
  48  
  49      $allowed_uri_schemes = get_allowed_uri_schemes( $context );
  50  
  51      // Validate URL structure
  52      if( $url[0] == '$' )
  53      { // This is a 'special replace code' URL (used in footers)
  54          if( ! preg_match( '~\$([a-z_]+)\$~', $url ) )
  55          {
  56              return T_('Invalid URL $code$ format');
  57          }
  58      }
  59      elseif( preg_match( '~^\w+:~', $url ) )
  60      { // there's a scheme and therefor an absolute URL:
  61          if( substr($url, 0, 7) == 'mailto:' )
  62          { // mailto:link
  63              if( ! in_array( 'mailto', $allowed_uri_schemes ) )
  64              { // Scheme not allowed
  65                  $scheme = 'mailto:';
  66                  $Debuglog->add( 'URI scheme &laquo;'.$scheme.'&raquo; not allowed!', 'error' );
  67                  return $verbose
  68                      ? sprintf( T_('URI scheme "%s" not allowed.'), htmlspecialchars($scheme) )
  69                      : T_('URI scheme not allowed.');
  70              }
  71  
  72              preg_match( '~^(mailto):(.*?)(\?.*)?$~', $url, $match );
  73              if( ! $match )
  74              {
  75                  return $verbose
  76                      ? sprintf( T_('Invalid email link: %s.'), htmlspecialchars($url) )
  77                      : T_('Invalid email link.');
  78              }
  79              elseif( ! is_email($match[2]) )
  80              {
  81                  return $verbose
  82                      ? sprintf( T_('Supplied email address (%s) is invalid.'), htmlspecialchars($match[2]) )
  83                      : T_('Invalid email address.');
  84              }
  85          }
  86          elseif( substr($url, 0, 6) == 'clsid:' )
  87          { // clsid:link
  88              if( ! in_array( 'clsid', $allowed_uri_schemes ) )
  89              { // Scheme not allowed
  90                  $scheme = 'clsid:';
  91                  $Debuglog->add( 'URI scheme &laquo;'.$scheme.'&raquo; not allowed!', 'error' );
  92                  return $verbose
  93                      ? sprintf( T_('URI scheme "%s" not allowed.'), htmlspecialchars($scheme) )
  94                      : T_('URI scheme not allowed.');
  95              }
  96  
  97              if( ! preg_match( '~^(clsid):([a-fA-F0-9\-]+)$~', $url, $match) )
  98              {
  99                  return T_('Invalid class ID format');
 100              }
 101          }
 102          elseif( substr($url, 0, 11) == 'javascript:' )
 103          { // javascript:
 104              // Basically there could be anything here
 105              if( ! in_array( 'javascript', $allowed_uri_schemes ) )
 106              { // Scheme not allowed
 107                  $scheme = 'javascript:';
 108                  $Debuglog->add( 'URI scheme &laquo;'.$scheme.'&raquo; not allowed!', 'error' );
 109                  return $verbose
 110                      ? sprintf( T_('URI scheme "%s" not allowed.'), htmlspecialchars($scheme) )
 111                      : T_('URI scheme not allowed.');
 112              }
 113  
 114              preg_match( '~^(javascript):~', $url, $match );
 115          }
 116          else
 117          {
 118              // convert URL to IDN:
 119              $url = idna_encode($url);
 120  
 121              if( ! preg_match('~^           # start
 122                  ([a-z][a-z0-9+.\-]*)             # scheme
 123                  ://                              # authorize absolute URLs only ( // not present in clsid: -- problem? ; mailto: handled above)
 124                  (\w+(:\w+)?@)?                   # username or username and password (optional)
 125                  ( localhost |
 126                          [a-z0-9]([a-z0-9\-])*            # Don t allow anything too funky like entities
 127                          \.                               # require at least 1 dot
 128                          [a-z0-9]([a-z0-9.\-])+           # Don t allow anything too funky like entities
 129                  )
 130                  (:[0-9]+)?                       # optional port specification
 131                  .*                               # allow anything in the path (including spaces - used in FileManager - but no newlines).
 132                  $~ix', $url, $match) )
 133              { // Cannot validate URL structure
 134                  $Debuglog->add( 'URL &laquo;'.$url.'&raquo; does not match url pattern!', 'error' );
 135                  return $verbose
 136                      ? sprintf( T_('Invalid URL format (%s).'), htmlspecialchars($url) )
 137                      : T_('Invalid URL format.');
 138              }
 139  
 140              $scheme = strtolower($match[1]);
 141              if( ! in_array( $scheme, $allowed_uri_schemes ) )
 142              { // Scheme not allowed
 143                  $Debuglog->add( 'URI scheme &laquo;'.$scheme.'&raquo; not allowed!', 'error' );
 144                  return $verbose
 145                      ? sprintf( T_('URI scheme "%s" not allowed.'), htmlspecialchars($scheme) )
 146                      : T_('URI scheme not allowed.');
 147              }
 148          }
 149      }
 150      else
 151      { // URL is relative..
 152          if( $context == 'commenting' || $context == 'download_src' || $context == 'http-https' )
 153          { // We do not allow relative URLs in comments and download urls
 154              return $verbose ? sprintf( T_('URL "%s" must be absolute.'), htmlspecialchars($url) ) : T_('URL must be absolute.');
 155          }
 156  
 157          $char = substr( $url, 0, 1 );
 158          if( $char != '/' && $char != '#' )
 159          { // must start with a slash or hash (for HTML anchors to the same page)
 160              return $verbose
 161                  ? sprintf( T_('URL "%s" must be a full path starting with "/" or an anchor starting with "#".'), htmlspecialchars($url) )
 162                  : T_('URL must be a full path starting with "/" or an anchor starting with "#".');
 163          }
 164      }
 165  
 166      if( $antispam_check )
 167      { // Search for blocked keywords:
 168          if( $block = antispam_check($url) )
 169          {
 170              return $verbose
 171                  ? sprintf( T_('URL "%s" not allowed: blacklisted word "%s".'), htmlspecialchars($url), $block )
 172                  : T_('URL not allowed');
 173          }
 174      }
 175  
 176      return false; // OK
 177  }

 180  /**
 181   * Get allowed URI schemes for a given context.
 182   * @param string Context ("posting", "commenting", "download_src", "http-https")
 183   * @return array
 184   */
 185  function get_allowed_uri_schemes( $context = 'posting' )
 186  {
 187      /**
 188       * @var User
 189       */
 190      global $current_User;
 191  
 192      $schemes = array(
 193              'http',
 194              'https'
 195          );
 196  
 197      if( $context == 'http-https' )
 198      { // for context == 'http-https' we accepts only http, https.
 199          return $schemes;
 200      }
 201  
 202      $schemes[] = 'ftp';
 203  
 204      if( $context == 'download_src' )
 205      { // for context == 'download_src' we also accepts ftp.
 206          return $schemes;
 207      }
 208  
 209      $schemes = array_merge( $schemes, array(
 210              'gopher',
 211              'nntp',
 212              'news',
 213              'mailto',
 214              'irc',
 215              'aim',
 216              'icq'
 217          ) );
 218  
 219      if( $context == 'commenting' )
 220      {
 221          return $schemes;
 222      }
 223  
 224      // for context == 'posting' we MAY allow additional "DANGEROUS" schemes:
 225      if( !empty( $current_User ) )
 226      { // Add additional permissions the current User may have:
 227  
 228          $Group = & $current_User->get_Group();
 229  
 230          if( $Group->perm_xhtml_javascript )
 231          {
 232              $schemes[] = 'javascript';
 233          }
 234  
 235          if( $Group->perm_xhtml_objects )
 236          {
 237              $schemes[] = 'clsid';
 238          }
 239  
 240      }
 241  
 242      return $schemes;
 243  }

 246  /**
 247   * Get the last HTTP status code received by the HTTP/HTTPS wrapper of PHP.
 248   *
 249   * @param array The $http_response_header array (by reference).
 250   * @return integer|boolean False if no HTTP status header could be found,
 251   *                         the HTTP status code otherwise.
 252   */
 253  function _http_wrapper_last_status( & $headers )
 254  {
 255      for( $i = count( $headers ) - 1; $i >= 0; --$i )
 256      {
 257          if( preg_match( '|^HTTP/\d+\.\d+ (\d+)|', $headers[$i], $matches ) )
 258          {
 259              return $matches[1];
 260          }
 261      }
 262  
 263      return false;
 264  }

 267  /**
 268   * Fetch remote page
 269   *
 270   * Attempt to retrieve a remote page using a HTTP GET request, first with
 271   * cURL, then fsockopen, then fopen.
 272   *
 273   * cURL gets skipped, if $max_size_kb is requested, since there appears to be no
 274   * method to control this.
 275   * {@internal (CURLOPT_READFUNCTION maybe? But it has not been called for me.. seems
 276   *            to affect sending, not fetching?!)}}
 277   *
 278   * @todo dh> Should we try remaining methods, if the previous one(s) failed?
 279   * @todo Tblue> Also allow HTTP POST.
 280   *
 281   * @param string URL
 282   * @param array Info (by reference)
 283   *        'error': holds error message, if any
 284   *        'status': HTTP status (e.g. 200 or 404)
 285   *        'used_method': Used method ("curl", "fopen", "fsockopen" or null if no method
 286   *                       is available)
 287   * @param integer Timeout (default: 15 seconds)
 288   * @param integer Maximum size in kB
 289   * @return string|false The remote page as a string; false in case of error
 290   */
 291  function fetch_remote_page( $url, & $info, $timeout = NULL, $max_size_kb = NULL )
 292  {
 293      global $outgoing_proxy_hostname, $outgoing_proxy_port, $outgoing_proxy_username, $outgoing_proxy_password;
 294  
 295      $info = array(
 296          'error' => '',
 297          'status' => NULL,
 298          'mimetype' => NULL,
 299          'used_method' => NULL,
 300      );
 301  
 302      if( ! isset($timeout) )
 303          $timeout = 15;
 304  
 305      if( extension_loaded('curl') && ! $max_size_kb ) // dh> I could not find an option to support "maximum size" for curl (to abort during download => memory limit).
 306      {    // CURL:
 307          $info['used_method'] = 'curl';
 308  
 309          $ch = curl_init();
 310          curl_setopt( $ch, CURLOPT_URL, $url );
 311          curl_setopt( $ch, CURLOPT_RETURNTRANSFER, true );
 312          curl_setopt( $ch, CURLOPT_CONNECTTIMEOUT, $timeout );
 313          curl_setopt( $ch, CURLOPT_TIMEOUT, $timeout );
 314          
 315          // Set proxy:
 316          if( !empty($outgoing_proxy_hostname) )
 317          {
 318              curl_setopt( $ch, CURLOPT_PROXY, $outgoing_proxy_hostname );
 319              curl_setopt( $ch, CURLOPT_PROXYPORT, $outgoing_proxy_port );
 320              curl_setopt( $ch, CURLOPT_PROXYUSERPWD, $outgoing_proxy_username.':'.$outgoing_proxy_password );
 321          }
 322  
 323          @curl_setopt( $ch, CURLOPT_FOLLOWLOCATION, true ); // made silent due to possible errors with safe_mode/open_basedir(?)
 324          curl_setopt( $ch, CURLOPT_MAXREDIRS, 3 );
 325          $r = curl_exec( $ch );
 326  
 327          $info['mimetype'] = curl_getinfo( $ch, CURLINFO_CONTENT_TYPE );
 328          $info['status'] = curl_getinfo( $ch, CURLINFO_HTTP_CODE );
 329          $info['error'] = curl_error( $ch );
 330          if( ( $errno = curl_errno( $ch ) ) )
 331          {
 332              $info['error'] .= ' (#'.$errno.')';
 333          }
 334          curl_close( $ch );
 335  
 336          return $r;
 337      }
 338  
 339      if( function_exists( 'fsockopen' ) ) // may have been disabled
 340      {    // FSOCKOPEN:
 341          $info['used_method'] = 'fsockopen';
 342  
 343          if ( ( $url_parsed = @parse_url( $url ) ) === false
 344               || ! isset( $url_parsed['host'] ) )
 345          {
 346              $info['error'] = NT_( 'Could not parse URL' );
 347              return false;
 348          }
 349  
 350          $host = $url_parsed['host'];
 351          $port = empty( $url_parsed['port'] ) ? 80 : $url_parsed['port'];
 352          $path = empty( $url_parsed['path'] ) ? '/' : $url_parsed['path'];
 353          if( ! empty( $url_parsed['query'] ) )
 354          {
 355              $path .= '?'.$url_parsed['query'];
 356          }
 357  
 358          $out = 'GET '.$path.' HTTP/1.1'."\r\n";
 359          $out .= 'Host: '.$host;
 360          if( ! empty( $url_parsed['port'] ) )
 361          {    // we don't want to add :80 if not specified. remote end may not resolve it. (e-g b2evo multiblog does not)
 362              $out .= ':'.$port;
 363          }
 364          $out .= "\r\n".'Connection: Close'."\r\n\r\n";
 365  
 366          $fp = @fsockopen( $host, $port, $errno, $errstr, $timeout );
 367          if( ! $fp )
 368          {
 369              $info['error'] = $errstr.' (#'.$errno.')';
 370              return false;
 371          }
 372  
 373          // Send request:
 374          fwrite( $fp, $out );
 375  
 376          // Set timeout for data:
 377          if( function_exists( 'stream_set_timeout' ) )
 378          {
 379              stream_set_timeout( $fp, $timeout ); // PHP 4.3.0
 380          }
 381          else
 382          {
 383              socket_set_timeout( $fp, $timeout ); // PHP 4
 384          }
 385  
 386          // Read response:
 387          $r = '';
 388          // First line:
 389          $s = fgets( $fp );
 390          if( ! preg_match( '~^HTTP/\d+\.\d+ (\d+)~', $s, $match ) )
 391          {
 392              $info['error'] = NT_( 'Invalid response.' );
 393              fclose( $fp );
 394              return false;
 395          }
 396  
 397          while( ! feof( $fp ) )
 398          {
 399              $r .= fgets( $fp );
 400              if( $max_size_kb && evo_bytes($r) >= $max_size_kb*1024 )
 401              {
 402                  $info['error'] = NT_( sprintf( 'Maximum size of %d kB reached.', $max_size_kb ) );
 403                  return false;
 404              }
 405          }
 406          fclose($fp);
 407  
 408          if ( ( $pos = strpos( $r, "\r\n\r\n" ) ) === false )
 409          {
 410              $info['error'] = NT_( 'Could not locate end of headers' );
 411              return false;
 412          }
 413  
 414          // Remember headers to extract info at the end
 415          $headers = explode("\r\n", substr($r, 0, $pos));
 416  
 417          $info['status'] = $match[1];
 418          $r = substr( $r, $pos + 4 );
 419      }
 420      elseif( ini_get( 'allow_url_fopen' ) )
 421      {    // URL FOPEN:
 422          $info['used_method'] = 'fopen';
 423  
 424          $fp = @fopen( $url, 'r' );
 425          if( ! $fp )
 426          {
 427              if( isset( $http_response_header )
 428                  && ( $code = _http_wrapper_last_status( $http_response_header ) ) !== false )
 429              {    // fopen() returned false because it got a bad HTTP code:
 430                  $info['error'] = NT_( 'Invalid response' );
 431                  $info['status'] = $code;
 432                  return '';
 433              }
 434  
 435              $info['error'] = NT_( 'fopen() failed' );
 436              return false;
 437          }
 438          // Check just to be sure:
 439          else if ( ! isset( $http_response_header )
 440                    || ( $code = _http_wrapper_last_status( $http_response_header ) ) === false )
 441          {
 442              $info['error'] = NT_( 'Invalid response' );
 443              return false;
 444          }
 445          else
 446          {
 447              // Used to get info at the end
 448              $headers = $http_response_header;
 449  
 450              // Retrieve contents
 451              $r = '';
 452              while( ! feof( $fp ) )
 453              {
 454                  $r .= fgets( $fp );
 455                  if( $max_size_kb && evo_bytes($r) >= $max_size_kb*1024 )
 456                  {
 457                      $info['error'] = NT_( sprintf( 'Maximum size of %d kB reached.', $max_size_kb ) );
 458                      return false;
 459                  }
 460              }
 461  
 462              $info['status'] = $code;
 463          }
 464          fclose( $fp );
 465      }
 466  
 467      // Extract mimetype info from the headers (for fsockopen/fopen)
 468      if( isset($r) )
 469      {
 470          foreach($headers as $header)
 471          {
 472              $header = strtolower($header);
 473              if( substr($header, 0, 13) == 'content-type:' )
 474              {
 475                  $info['mimetype'] = trim(substr($header, 13));
 476                  break; // only looking for mimetype
 477              }
 478          }
 479  
 480          return $r;
 481      }
 482  
 483      // All failed:
 484      $info['error'] = NT_( 'No method available to access URL!' );
 485      return false;
 486  }

 489  /**
 490   * Get $url with the same protocol (http/https) as $other_url.
 491   *
 492   * @param string URL
 493   * @param string other URL (defaults to {@link $ReqHost})
 494   * @return string
 495   */
 496  function url_same_protocol( $url, $other_url = NULL )
 497  {
 498      if( is_null($other_url) )
 499      {
 500          global $ReqHost;
 501  
 502          $other_url = $ReqHost;
 503      }
 504  
 505      // change protocol of $url to same of admin ('https' <=> 'http')
 506      if( substr( $url, 0, 7 ) == 'http://' )
 507      {
 508          if( substr( $other_url, 0, 8 ) == 'https://' )
 509          {
 510              $url = 'https://'.substr( $url, 7 );
 511          }
 512      }
 513      elseif( substr( $url, 0, 8 ) == 'https://' )
 514      {
 515          if( substr( $other_url, 0, 7 ) == 'http://' )
 516          {
 517              $url = 'http://'.substr( $url, 8 );
 518          }
 519      }
 520  
 521      return $url;
 522  }

 525  /**
 526   * Add param(s) at the end of an URL, using either "?" or "&amp;" depending on existing url
 527   *
 528   * @param string existing url
 529   * @param string|array Params to add (string as-is) or array, which gets urlencoded.
 530   * @param string delimiter to use for more params
 531   */
 532  function url_add_param( $url, $param, $glue = '&amp;' )
 533  {
 534      if( empty( $param ) )
 535      {
 536          return $url;
 537      }
 538  
 539      if( ( $anchor_pos = strpos( $url, '#' ) ) !== false )
 540      { // There's an "#anchor" in the URL
 541          $anchor = substr( $url, $anchor_pos );
 542          $url = substr( $url, 0, $anchor_pos );
 543      }
 544      else
 545      { // URL without "#anchor"
 546          $anchor = '';
 547      }
 548  
 549      // Handle array use case
 550      if( is_array( $param ) )
 551      { // list of key => value pairs
 552          $param_list = array();
 553          foreach( $param as $k => $v )
 554          {
 555              $param_list[] = get_param_urlencoded( $k, $v, $glue );
 556          }
 557          $param = implode( $glue, $param_list );
 558      }
 559  
 560      if( strpos( $url, '?' ) !== false )
 561      { // There are already params in the URL
 562          $r = $url;
 563          if( substr( $url, -1 ) != '?' && substr( $param, 0, 1 ) != '#' )
 564          { // the "?" is not the last char AND "#" is not first char of param
 565              $r .= $glue;
 566          }
 567          return $r.$param.$anchor;
 568      }
 569  
 570      // These are the first params
 571      return $url.'?'.$param.$anchor;
 572  }

 575  /**
 576   * Add a tail (starting with "/") at the end of an URL before any params (starting with "?")
 577   *
 578   * @param string existing url
 579   * @param string tail to add
 580   */
 581  function url_add_tail( $url, $tail )
 582  {
 583      $parts = explode( '?', $url );
 584      if( substr($parts[0], -1) == '/' )
 585      {
 586          $parts[0] = substr($parts[0], 0, -1);
 587      }
 588      if( isset($parts[1]) )
 589      {
 590          return $parts[0].$tail.'?'.$parts[1];
 591      }
 592  
 593      return $parts[0].$tail;
 594  }

 622  /**
 623   * Try to make $url relative to $target_url, if scheme, host, user and pass matches.
 624   *
 625   * This is useful for redirect_to params, to keep them short and avoid mod_security
 626   * rejecting the request as "Not Acceptable" (whole URL as param).
 627   *
 628   * @param string URL to handle
 629   * @param string URL where we want to make $url relative to
 630   * @return string
 631   */
 632  function url_rel_to_same_host( $url, $target_url )
 633  {
 634      // Prepend fake scheme to URLs starting with "//" (relative to current protocol), since
 635      // parse_url fails to handle them correctly otherwise (recognizes them as path-only)
 636      $mangled_url = substr($url, 0, 2) == '//' ? 'noprotocolscheme:'.$url : $url;
 637  
 638      if( substr($target_url, 0, 2) == '//' )
 639          $target_url = 'noprotocolscheme:'.$target_url;
 640  
 641  
 642      $parsed_url = @parse_url( $mangled_url );
 643      if( ! $parsed_url )
 644      { // invalid url
 645          return $url;
 646      }
 647      if( empty($parsed_url['scheme']) || empty($parsed_url['host']) )
 648      { // no protocol or host information
 649          return $url;
 650      }
 651  
 652      $target_url = @parse_url( $target_url );
 653      if( ! $target_url )
 654      { // invalid url
 655          return $url;
 656      }
 657      if( ! empty($target_url['scheme']) && $target_url['scheme'] != $parsed_url['scheme']
 658          && $parsed_url['scheme'] != 'noprotocolscheme' )
 659      { // scheme/protocol is different
 660          return $url;
 661      }
 662      if( ! empty($target_url['host']) )
 663      {
 664          if( empty($target_url['scheme']) || $target_url['host'] != $parsed_url['host'] )
 665          { // target has no scheme (but a host) or hosts differ
 666              return $url;
 667          }
 668  
 669          if( @$target_url['port'] != @$parsed_url['port'] )
 670              return $url;
 671          if( @$target_url['user'] != @$parsed_url['user'] )
 672              return $url;
 673          if( @$target_url['pass'] != @$parsed_url['pass'] )
 674              return $url;
 675      }
 676  
 677      // We can make the URL relative:
 678      $r = '';
 679      if( isset($parsed_url['path']) && strlen($parsed_url['path']) )
 680          $r .= $parsed_url['path'];
 681  
 682      if( isset($parsed_url['query']) && strlen($parsed_url['query']) )
 683          $r .= '?'.$parsed_url['query'];
 684  
 685      if( isset($parsed_url['fragment']) && strlen($parsed_url['fragment']) )
 686          $r .= '#'.$parsed_url['fragment'];
 687  
 688      return $r;
 689  }

 700  {
 701      load_funcs('_ext/_url_rel2abs.php');

 704      {    // URL is already absolute
 705          return $url;

 714              $base = $base_tag_set;

 720                  $SkinCache = & get_SkinCache();
 721                  if( ($Skin = $SkinCache->get_by_ID( $Blog->get_skin_ID(), false )) !== false )
 722                  {
 723                      $base = $Blog->get_local_skins_url().$Skin->folder.'/';
 724                  }
 725                  else
 726                  { // Skin not set:
 727                      $base = $Blog->gen_baseurl();
 728                  }
 729              }
 730              else
 731              {    // We are displaying a general page that is not specific to a blog:
 732                  $base = $ReqHost;
 733              }
 734          }
 735      }
 736  
 737      if( ($absurl = url_to_absolute($url, $base)) === false )
 738      {    // Return relative URL in case of error
 739          $absurl = $url;
 740      }
 741      return $absurl;
