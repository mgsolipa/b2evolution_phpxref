  48  	function BlockCache( $type, $keys )
  49      {
  50          global $instance_name;
  51  
  52          $this->type = $type;

  54          // Make sure keys are always in the same order:
  55          ksort( $keys );
  56          $this->keys = $keys;

  69      /**
  70       * Invalidate a special key
  71       *
  72       * All we do is store the timestamp of teh invalidation
  73       *
  74       * @see http://b2evolution.net/man/widget-caching
  75       */
  76  	static function invalidate_key( $key, $val )
  77      {
  78          global $Debuglog, $servertimenow, $instance_name;
  79  
  80          $lastchanged_key_name = $instance_name.'+last_changed+'.$key.'='.$val;
  81  
  82          // Invalidate using the real time (seconds may have elapsed since $sertimenow)
  83          // Add 1 second because of the granularity that's down to the second
  84          // Worst case scenario: content will be collected/cahced several times for a whole second (as well as the first request after the end of that second)
  85          BlockCache::cacheproviderstore( $lastchanged_key_name, time()+1 );
  86  
  87          $Debuglog->add( 'Invalidated: '.$lastchanged_key_name.' @ '.(time()+1), 'blockcache' );
  88      }

  91      /**
  92       * Check if cache contents are available, otherwise start collecting output to be cached
  93       *
  94       * Basically we get all the invalidation dates we need, then we get the
  95       * data and then we check if some invalidation occured after the data was cached.
  96       * If an invalidation date is missing we consider the cache to be
  97       * obsolete but we generate a new invalidation date for next time we try to retrieve.
  98       *
  99       * @return true if we found and have echoed content from the cache
 100       */
 101  	function check()
 102      {
 103          global $Debuglog, $servertimenow, $instance_name;
 104  
 105          $missing_date = false;
 106          $most_recent_invalidation_ts = 0;
 107          $most_recent_invaliating_key = '';
 108          foreach( $this->keys as $key => $val )
 109          {
 110              $lastchanged_key_name = $instance_name.'+last_changed+'.$key.'='.$val;
 111              $last_changed_ts = $this->cacheproviderretrieve( $lastchanged_key_name, $success );
 112              if( ! $success )
 113              {    // We have lost the key! Recreate and keep going for other keys:
 114                  $Debuglog->add( 'Missing: '.$lastchanged_key_name, 'blockcache' );
 115                  $missing_date = true;
 116                  $this->cacheproviderstore( $lastchanged_key_name, $servertimenow );
 117                  continue;
 118              }
 119  
 120              if( $last_changed_ts > $most_recent_invalidation_ts )
 121              {    // This is the new most recent invalidation date.
 122                  $most_recent_invalidation_ts = $last_changed_ts;
 123                  $most_recent_invaliating_key = $lastchanged_key_name;
 124              }
 125          }
 126  
 127          if( !$missing_date && ($content = $this->retrieve( $most_recent_invalidation_ts, $most_recent_invaliating_key )) !== false )
 128          { // cache was not invalidated yet and we could retrieve:
 129              return $content;
 130          }
 131  
 132          // Caller should call BlockCache->start_collect() right after this
 133          return false;
 134      }

 137      /**
 138       * Retrieve and output cache
 139       *
 140       * @param integer oldest acceptable timestamp
 141       * @return boolean true if we could retrieve
 142       */
 143  	function retrieve( $oldest_acceptable_ts = NULL, $most_recent_invaliating_key = '' )
 144      {
 145          global $Debuglog;
 146          global $servertimenow;
 147  
 148          // return false;
 149  
 150          $content = $this->cacheproviderretrieve( $this->serialized_keys, $success );
 151  
 152          if( ! $success )
 153          {
 154              return false;
 155          }
 156  
 157          if( !is_null($oldest_acceptable_ts) )
 158          { // We want to do timestamp checking:
 159  
 160  
 161              //if( ! preg_match( '/^([0-9]+) (.*)$/ms', $content, $matches ) )
 162              if( ! $pos = strpos( $content, ' ' ) )
 163              {    // Could not find timestamp
 164                  $Debuglog->add( 'MISSING TIMESTAMP on retrieval of: '.$this->serialized_keys, 'blockcache' );
 165                  return false;
 166              }
 167  
 168              // if( $matches[1] < $oldest_acceptable_ts )
 169              if( ($cache_ts = substr( $content, 0, $pos )) < $oldest_acceptable_ts )
 170              {    // Timestamp too old (there has been an invalidation in between)
 171                  $Debuglog->add( 'Retrieved INVALIDATED cached content: '.$this->serialized_keys
 172                      .' (invalidated by '.$most_recent_invaliating_key.' - '.$cache_ts.' < '.$oldest_acceptable_ts.')', 'blockcache' );
 173                  return false;
 174              }
 175  
 176              // OK, we have content that is still valid:
 177              // $content = $matches[2];
 178              $content = substr( $content, $pos+1 );
 179          }
 180  
 181          $Debuglog->add( 'Retrieved: '.$this->serialized_keys, 'blockcache' );
 182  
 183          return $content;
 184      }

 190  	function output_handler( $buffer )

 192          $this->cached_page_content .= $buffer;

 197      /**
 198       * This should be called right after ::check() in order to start collecting a new block for the cache.
 199       */
 200  	function start_collect()
 201      {
 202          global $Debuglog;
 203  
 204          $this->is_collecting = true;
 205  
 206          $Debuglog->add( 'Collecting: '.$this->serialized_keys, 'blockcache' );
 207  
 208          ob_start( array( & $this, 'output_handler'), $this->output_chunk_size );
 209  
 210      }

 213      /**
 214       * We are going to output personal data and we want to abort collecting the data for the cache.
 215       */
 216  	function abort_collect()
 217      {
 218          global $Debuglog;
 219  
 220          if( ! $this->is_collecting )
 221          {    // We are not collecting anyway
 222              return;
 223          }
 224  
 225           $Debuglog->add( 'Aborting cache data collection...', 'blockcache' );
 226  
 227          ob_end_flush();
 228  
 229          // We are no longer collecting...
 230          $this->is_collecting = false;
 231      }

 234      /**
 235       * End collecting output to be cached
 236       *
 237       * We just concatenate all the individual keys to have a single one
 238       * Then we store with the current timestamp
 239       */
 240  	function end_collect()
 241      {
 242          global $Debuglog, $servertimenow;
 243  
 244          if( ! $this->is_collecting )
 245          {    // We are not collecting
 246              return;
 247          }
 248  
 249          ob_end_flush();
 250  
 251          // We use servertimenow because we may have used data that was loaded at the very start of this page
 252          $this->cacheproviderstore( $this->serialized_keys, $servertimenow.' '.$this->cached_page_content );
 253      }

 259       * @todo dh> This method should get removed from here, it's not limited to BlockCache.

 262       * @param int Time to live in seconds (default: 86400; 0 means "as long as possible")

 264  	static function cacheproviderstore( $key, $payload, $ttl = 86400 )

 266          return set_to_mem_cache($key, $payload, $ttl);
