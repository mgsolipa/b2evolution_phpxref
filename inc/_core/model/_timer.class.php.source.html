  23   * This is a simple class to allow timing/profiling of code portions.
  24   */
  25  class Timer
  26  {
  27      /**
  28       * Remember times.
  29       *
  30       * We store for each category (primary key) the state, start/resume time and the total passed time.
  31       *
  32       * @access protected

  38       * @access protected
  39       * @var integer Level of internal indentation, used to indent Debuglog messages.
  40       */
  41      var $indent = 0;

  68       * Start a timer.
  69       */

  72          $this->reset( $category );
  73          $this->resume( $category, $log );

  77      /**
  78       * Stops a timer category. It may me resumed later on, see {@link resume()}. This is an alias for {@link pause()}.
  79       *
  80       * @return boolean false, if the timer had not been started.
  81       */
  82  	function stop( $category )
  83      {
  84          global $Debuglog;
  85  
  86          if( ! $this->pause( $category ) )
  87              return false;
  88  
  89          $Debuglog->add( str_repeat('&nbsp;', $this->indent*4).$category.' stopped at '.$this->get_duration( $category, 3 ), 'timer' );
  90  
  91          return true;
  92      }

  95      /**
  96       * Pauses a timer category. It may me resumed later on, see {@link resume()}.
  97       *
  98       * NOTE: The timer needs to be started, either through the {@link Timer() Constructor} or the {@link start()} method.
  99       *
 100       * @return boolean false, if the timer had not been started.
 101       */
 102  	function pause( $category, $log = true )
 103      {
 104          global $Debuglog;
 105          $since_resume = $this->get_current_microtime() - $this->_times[$category]['resumed'];
 106          if( $log )
 107          {
 108              $this->indent--;
 109              if( $this->indent < 0 ) $this->indent = 0;
 110              $Debuglog->add( str_repeat('&nbsp;', $this->indent*4).$category.' paused at '.$this->get_duration( $category, 3 ).' (<strong>+'.number_format($since_resume, 4).'</strong>)', 'timer' );
 111          }
 112          if( $this->get_state($category) != 'running' )
 113          { // Timer is not running!
 114              $Debuglog->add("Warning: tried to pause already paused '$category'.", 'timer');
 115              return false;
 116          }
 117  
 118          $this->_times[$category]['total'] += $since_resume;
 119          $this->_times[$category]['state'] = 'paused';
 120  
 121          return true;
 122      }

 125      /**
 126       * Resumes the timer on a category.
 127       */
 128  	function resume( $category, $log = true )
 129      {
 130          global $Debuglog;
 131  
 132          if( !isset($this->_times[$category]['total']) )
 133          {
 134              $this->start( $category, $log );
 135              return;
 136          }
 137  
 138          $this->_times[$category]['resumed'] = $this->get_current_microtime();
 139          $this->_times[$category]['count']++;
 140  
 141          $this->_times[$category]['state'] = 'running';
 142  
 143          if( $log )
 144          {
 145              $Debuglog->add( str_repeat('&nbsp;', $this->indent*4).$category.' resumed at '.$this->get_duration( $category, 3 ), 'timer' );
 146              $this->indent++;
 147          }
 148      }

 191      /**
 192       * Get the time that was spent in the given category in seconds with microsecond precision
 193       *
 194       * @return float (seconds.microseconds)
 195       */
 196  	function get_microtime( $category )
 197      {
 198          switch( $this->get_state($category) )
 199          {
 200              case 'running':
 201                  // The timer is running, we need to return the additional time since the last resume.
 202                  return $this->_times[$category]['total']
 203                      + $this->get_current_microtime() - $this->_times[$category]['resumed'];
 204  
 205              case 'paused':
 206                  return $this->_times[$category]['total'];
 207  
 208              default:
 209                  return (float)0;
 210          }
 211      }

 214      /**
 215       * Get the state a category timer is in.
 216       *
 217       * @return string 'unknown', 'not initialised', 'running', 'paused'
 218       */
 219  	function get_state( $category )
 220      {
 221          if( !isset($this->_times[$category]) )
 222          {
 223              return 'unknown';
 224          }
 225  
 226          if( !isset($this->_times[$category]['state']) )
 227          {
 228              return 'not initialised';
 229          }
 230  
 231          return $this->_times[$category]['state'];
 232      }

 246      /**
 247       * Get the current time with microsecond precision
 248       *
 249       * @return float (seconds.microseconds)
 250       */
 251  	function get_current_microtime()
 252      {
 253          list($usec, $sec) = explode(' ', microtime());
 254          return ((float)$usec + (float)$sec);
 255      }
