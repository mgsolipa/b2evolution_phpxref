 115       * Add a message.
 116       *
 117       * @param string the message
 118       * @param string the message type, it can have this values: 'success', 'warning', 'error', 'note'

 120  	function add( $text, $type = 'error' )
 121      {
 122          $this->messages_text[$this->count] = $text;
 123          $this->messages_type[$this->count] = $type;
 124          $this->count++;

 127              $this->has_errors = ( $type == 'error' );

 137  	function add_messages( $p_Messages )
 138      {
 139          $this->count = $this->count + $p_Messages->count;
 140          for( $i = 0; $i < $p_Messages->count; $i++ )
 141          {
 142              $this->messages_text[] = $p_Messages->messages_text[$i];
 143              $this->messages_type[] = $p_Messages->messages_type[$i];

 183      /**
 184       * Display messages of the object.
 185       *
 186       * - You can output/get the messages
 187       * - Head/Foot will be displayed on top/bottom of the messages.
 188       * - You can suppress the outer div or set a css class for it (defaults to
 189       *   'log_container').
 190       *
 191       * @todo Make this simple!
 192       * start by getting rid of the $category selection and the special cases for 'all'. If you don't want to display ALL messages,
 193       * then you should not log them in the same Log object and you should instantiate separate logs instead.
 194       *
 195       * @param string|NULL Header/title
 196       * @param string|NULL Footer
 197       * @param boolean to display or return (default: display)
 198       * @param mixed the outer div, may be false
 199       * @return boolean false, if no messages; else true (and outputs if $display)
 200       */
 201  	function display( $head = NULL, $foot = NULL, $display = true, $outerdivclass = 'log_container' )
 202      {
 203          if( $this->count == 0 ) {
 204              return false;
 205          }
 206  
 207          $disp = '';
 208  
 209          if( isset( $this->params['class_outerdiv'] ) && $outerdivclass == 'log_container' )
 210          { // Use default class from object params instead of default function param
 211              $outerdivclass = $this->params['class_outerdiv'];
 212          }
 213  
 214          if( $outerdivclass )
 215          {
 216              $disp .= "\n<div class=\"$outerdivclass\">";
 217          }
 218  
 219          if( !empty( $head ) )
 220          {
 221              $disp .= '<h3>'.$head.'</h3>';
 222          }
 223  
 224          $disp .= '<ul>';
 225          for( $i = 0; $i < $this->count; $i++ )
 226          {
 227              $class = isset( $this->params['class_'.$this->messages_type[$i]] ) ?
 228                      $this->params['class_'.$this->messages_type[$i]] :
 229                      $this->params['class_note'];
 230              $disp .= "<li>\t<div class=\"{$class}\"".'>'
 231                      .$this->params['before_'.$this->messages_type[$i]]
 232                      .$this->params['before_message']
 233                      .$this->messages_text[$i]
 234                      .$this->params['after_'.$this->messages_type[$i]]
 235                  ."</div></li>\n";
 236          }
 237          $disp .= '</ul>';
 238  
 239          if( !empty( $foot ) )
 240          {
 241              $disp .= "\n<p>".$foot."</p>";
 242          }
 243  
 244          if( $outerdivclass )
 245          {
 246              $disp .= "</div>\n";
 247          }
 248  
 249          if( $display )
 250          {
 251              echo $disp;
 252              return true;
 253          }
 254          return $disp;
 255      }

 258      /**
 259       * Concatenates messages of a given category to a string
 260       *
 261       * @param string prefix of the string
 262       * @param string suffic of the string
 263       * @param string the glue
 264       * @param string result format
 265       * @return string the messages, imploded. Tags stripped.
 266       */
 267  	function get_string( $head = '', $foot = '', $implodeBy = ', ', $format = 'striptags' )
 268      {
 269          if( !$this->count )
 270          {
 271              return false;
 272          }
 273  
 274          $r = '';
 275          if( '' != $head )
 276          {
 277              $r .= $head.' ';
 278          }
 279          $r .= implode( $implodeBy, $this->messages_text );
 280          if( '' != $foot )
 281          {
 282              $r .= ' '.$foot;
 283          }
 284  
 285          switch( $format )
 286          {
 287              case 'xmlrpc':
 288                  $r = strip_tags( $r );    // get rid of <code>
 289                  $r = str_replace( '&lt;', '<', $r );
 290                  $r = str_replace( '&gt;', '>', $r );
 291                  $r = str_replace( '&quot;', '"', $r );
 292                  break;
 293  
 294              case 'striptags':
 295                  $r = strip_tags( $r );
 296                  break;
 297          }
 298  
 299          return $r;
 300      }
