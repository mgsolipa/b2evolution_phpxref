 234      /**
 235       * Check module permission
 236       *
 237       * @param string Permission name
 238       * @param string Requested permission level
 239       * @param mixed Permission target (blog ID, array of cat IDs...)
 240       * @param string function name
 241       * @param object user's Group - can't be NULL
 242       * @return boolean True on success (permission is granted), false if permission is not granted
 243       *                 NULL if permission not implemented.
 244       */
 245  	static function check_perm( $permname, $permlevel, $permtarget, $function, $Group )
 246      {
 247          if( empty( $Group ) )
 248          { // group must be set
 249              return NULL;
 250          }
 251  
 252          $GroupSettings = & $Group->get_GroupSettings();
 253  
 254          if( array_key_exists( $permname, $GroupSettings->permission_modules ) )
 255          {    // Requested permission found in the group settings
 256              $Module = & $GLOBALS[$GroupSettings->permission_modules[$permname].'_Module'];
 257              if( method_exists( $Module, 'get_available_group_permissions' ) )
 258              {    // Function to get available permission exists
 259                  $permissions = $Module->get_available_group_permissions();
 260                  if( array_key_exists( $permname, $permissions ) )
 261                  {    // Requested permission found in available permisssion list
 262                      $permission = $permissions[$permname];
 263                      if( array_key_exists( $function, $permission ) )
 264                      {    // Function to check permission exists
 265                          $function = $permission[$function];
 266                          if( method_exists( $Module, $function ) )
 267                          {    // We can call check permission function
 268                              return $Module->{$function}( $permlevel, $GroupSettings->get( $permname, $Group->ID ), $permtarget );
 269                          }
 270                      }
 271                  }
 272              }
 273          }
 274  
 275          // Required parameters of check permission function not found
 276          return NULL;
 277      }

 317      /**
 318       * Check Minimum PHP version required for the module
 319       *
 320       * @param string module name/id
 321       */
 322  	function check_required_php_version( $module )
 323      {
 324          global $app_name, $app_version, $required_php_version;
 325  
 326          $php_version = phpversion();
 327          if( version_compare( $php_version, $required_php_version[ $module ], '<' ) )
 328          {
 329              $error_message = sprintf( 'You cannot use %1s module of %2$s %3$s on this server because it requires PHP version %4$s or higher. You are running version %5$s.',
 330                                  $module, $app_name, $app_version, $required_php_version[ $module ], $php_version );
 331  
 332              die('<h1>Insufficient Requirements</h1><p>'.$error_message.'</p>');
 333          }
