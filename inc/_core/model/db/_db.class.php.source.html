 493       *
 494       * @param string|array|null

 497  	function quote($str)
 498      {
 499          if( is_null( $str ) )
 500          {
 501              return 'NULL';
 502          }
 503          elseif( is_array( $str ) )
 504          {
 505              $r = '';
 506              foreach( $str as $elt )
 507              {
 508                  $r .= $this->quote($elt).',';
 509              }
 510              return substr( $r, 0, -1 );  // remove last ,
 511          }
 512          else
 513          {
 514              return "'".$this->escape($str)."'";
 515          }
 516      }

 749      /**
 750       * Basic Query
 751       *
 752       * @param string SQL query
 753       * @param string title for debugging
 754       * @return mixed # of rows affected or false if error
 755       */
 756  	function query( $query, $title = '' )
 757      {
 758          global $Timer;
 759  
 760          // initialise return
 761          $return_val = 0;
 762  
 763          // Flush cached values..
 764          $this->flush();
 765  
 766          // Replace aliases:
 767          if( ! empty($this->dbaliases) )
 768          {
 769              // TODO: this should only replace the table name part(s), not the whole query!
 770              // blueyed> I've changed it to replace in table name parts for UPDATE, INSERT and REPLACE, because
 771              //          it corrupted serialized data..
 772              //          IMHO, a cleaner solution would be to use {T_xxx} in the queries and replace it here. In object properties (e.g. DataObject::$dbtablename), only "T_xxx" would get used and surrounded by "{..}" in the queries it creates.
 773  
 774              if( preg_match( '~^\s*(UPDATE\s+)(.*?)(\sSET\s.*)$~is', $query, $match ) )
 775              { // replace only between UPDATE and SET, but check subqueries:
 776                  $subquery_result = '';
 777                  while( preg_match( '~^(.*SELECT.*FROM\s+)(.*?)(\s.*)$~is', $match[3], $subquery_match ) )
 778                  { // replace in subquery
 779                      $match[3] = $subquery_match[1];
 780                      $subquery_result = preg_replace( $this->dbaliases, $this->dbreplaces, $subquery_match[2] ).$subquery_match[3].$subquery_result;
 781                  }
 782                  $match[3] = $match[3].$subquery_result;
 783                  if( preg_match( '~^(.*SELECT.*JOIN\s+)(.*?)(\s.*)$~is', $match[3], $subquery_match ) )
 784                  { // replace in whole subquery, there can be any number of JOIN:
 785                      $match[3] = preg_replace( $this->dbaliases, $this->dbreplaces, $match[3] );
 786                  }
 787                  $query = $match[1].preg_replace( $this->dbaliases, $this->dbreplaces, $match[2] ).$match[3];
 788              }
 789              elseif( preg_match( '~^\s*(INSERT|REPLACE\s+)(.*?)(\s(VALUES|SET)\s.*)$~is', $query, $match ) )
 790              { // replace only between INSERT|REPLACE and VALUES|SET:
 791                  $query = $match[1].preg_replace( $this->dbaliases, $this->dbreplaces, $match[2] ).$match[3];
 792              }
 793              else
 794              { // replace in whole query:
 795                  $query = preg_replace( $this->dbaliases, $this->dbreplaces, $query );
 796  
 797                  if( ! empty($this->table_options) && preg_match( '#^ \s* create \s* table \s #ix', $query) )
 798                  { // Query is a table creation, we add table options:
 799                      $query = preg_replace( '~;\s*$~', '', $query ); // remove any ";" at the end
 800                      $query .= ' '.$this->table_options;
 801                  }
 802              }
 803          }
 804          elseif( ! empty($this->table_options) )
 805          { // No aliases, but table_options:
 806              if( preg_match( '#^ \s* create \s* table \s #ix', $query) )
 807              { // Query is a table creation, we add table options:
 808                  $query = preg_replace( '~;\s*$~', '', $query ); // remove any ";" at the end
 809                  $query .= $this->table_options;
 810              }
 811          }
 812          // echo '<p>'.$query.'</p>';
 813  
 814          // Keep track of the last query for debug..
 815          $this->last_query = $query;
 816  
 817          // Perform the query via std mysqli_query function..
 818          $this->num_queries++;
 819  
 820          if( $this->log_queries )
 821          {    // We want to log queries:
 822              $this->queries[ $this->num_queries - 1 ] = array(
 823                  'title' => $title,
 824                  'sql' => $query,
 825                  'rows' => -1,
 826                  'time' => 'unknown',
 827                  'results' => 'unknown' );
 828          }
 829  
 830          if( is_object($Timer) )
 831          {
 832              // Resume global query timer
 833              $Timer->resume( 'SQL QUERIES' , false );
 834              // Start a timer for this particular query:
 835              $Timer->start( 'sql_query', false );
 836  
 837              // Run query:
 838              $this->result = @mysqli_query( $this->dbhandle, $query );
 839  
 840              if( $this->log_queries )
 841              {    // We want to log queries:
 842                  // Get duration for last query:
 843                  $this->queries[ $this->num_queries - 1 ]['time'] = $Timer->get_duration( 'sql_query', 10 );
 844              }
 845  
 846              // Pause global query timer:
 847              $Timer->pause( 'SQL QUERIES' , false );
 848          }
 849          else
 850          {
 851              // Run query:
 852              $this->result = @mysqli_query( $this->dbhandle, $query );
 853          }
 854  
 855          // If there is an error then take note of it..
 856          if( is_object( $this->dbhandle ) && $this->dbhandle->errno != 0 )
 857          {
 858              if( is_object( $this->result ) )
 859              {
 860                  mysqli_free_result($this->result);
 861              }
 862              $last_errno = $this->dbhandle->errno;
 863              if( $this->use_transactions && ( $this->transaction_isolation_level == 'SERIALIZABLE' ) && ( 1213 == $last_errno ) )
 864              { // deadlock exception occured, transaction must be rolled back
 865                  $this->rollback_nested_transaction = true;
 866                  return false;
 867              }
 868              $this->print_error( '', '', $title );
 869              return false;
 870          }
 871  
 872          if( preg_match( '#^\s*(INSERT|DELETE|UPDATE|REPLACE)\s#i', $query, $match ) )
 873          { // Query was an insert, delete, update, replace:
 874  
 875              $this->rows_affected = mysqli_affected_rows($this->dbhandle);
 876              if( $this->log_queries )
 877              {    // We want to log queries:
 878                  $this->queries[ $this->num_queries - 1 ]['rows'] = $this->rows_affected;
 879              }
 880  
 881              // Take note of the insert_id, for INSERT and REPLACE:
 882              $match[1] = strtoupper($match[1]);
 883              if( $match[1] == 'INSERT' || $match[1] == 'REPLACE' )
 884              {
 885                  $this->insert_id = mysqli_insert_id($this->dbhandle);
 886              }
 887  
 888              // Return number of rows affected
 889              $return_val = $this->rows_affected;
 890          }
 891          else
 892          { // Query was a select, alter, etc...:
 893              if( is_object($this->result) )
 894              { // It's not a resource for CREATE or DROP for example and can even trigger a fatal error (see http://forums.b2evolution.net//viewtopic.php?t=9529)
 895                  $this->num_rows = mysqli_num_rows($this->result);
 896              }
 897  
 898              if( $this->log_queries )
 899              {    // We want to log queries:
 900                  $this->queries[ $this->num_queries - 1 ]['rows'] = $this->num_rows;
 901              }
 902  
 903              // Return number of rows selected
 904              $return_val = $this->num_rows;
 905          }
 906          if( $this->log_queries )
 907          {    // We want to log queries:
 908              if( $this->debug_dump_function_trace_for_queries )
 909              {
 910                  $this->queries[ $this->num_queries - 1 ]['function_trace'] = debug_get_backtrace( $this->debug_dump_function_trace_for_queries, array( array( 'class' => 'DB' ) ), 1 ); // including first stack entry from class DB
 911              }
 912  
 913              if( $this->debug_dump_rows && $this->num_rows )
 914              {
 915                  $this->queries[ $this->num_queries - 1 ]['results'] = $this->debug_get_rows_table( $this->debug_dump_rows );
 916              }
 917  
 918              // Profile queries
 919              if( $this->debug_profile_queries )
 920              {
 921                  // save values:
 922                  $saved_last_result = $this->result;
 923                  $saved_num_rows = $this->num_rows;
 924  
 925                  $this->num_rows = 0;
 926  
 927                  $this->result = @mysqli_query( $this->dbhandle, 'SHOW PROFILE' );
 928                  $this->num_rows = mysqli_num_rows($this->result);
 929  
 930                  if( $this->num_rows )
 931                  {
 932                      $this->queries[$this->num_queries-1]['profile'] = $this->debug_get_rows_table( 100, true );
 933  
 934                      // Get time information from PROFILING table (which corresponds to "SHOW PROFILE")
 935                      $this->result = mysqli_query( $this->dbhandle, 'SELECT FORMAT(SUM(DURATION), 6) AS DURATION FROM INFORMATION_SCHEMA.PROFILING GROUP BY QUERY_ID ORDER BY QUERY_ID DESC LIMIT 1' );
 936                      $this->queries[$this->num_queries-1]['time_profile'] = array_shift(mysqli_fetch_row($this->result));
 937                  }
 938  
 939                  // Free "PROFILE" result resource:
 940                  mysqli_free_result($this->result);
 941  
 942  
 943                  // Restore:
 944                  $this->result = $saved_last_result;
 945                  $this->num_rows = $saved_num_rows;
 946              }
 947          }
 948          return $return_val;
 949      }

1217      /**
1218       * Format a SQL query
1219       *
1220       * @param string SQL
1221       * @param boolean Format with/for HTML?
1222       */
1223  	static function format_query( $sql, $html = true, $maxlen = NULL )
1224      {
1225          $sql = trim( str_replace("\t", '  ', $sql ) );
1226          if( $maxlen )
1227          {
1228              $sql = strmaxlen($sql, $maxlen, '...');
1229          }
1230  
1231          $new = '';
1232          $word = '';
1233          $in_comment = false;
1234          $in_literal = false;
1235          for( $i = 0, $n = strlen($sql); $i < $n; $i++ )
1236          {
1237              $c = $sql[$i];
1238              if( $in_comment )
1239              {
1240                  if( $in_comment === '/*' && substr($sql, $i, 2) == '*/' )
1241                      $in_comment = false;
1242                  elseif( $c == "\n" )
1243                      $in_comment = false;
1244              }
1245              elseif( $in_literal )
1246              {
1247                  if( $c == $in_literal )
1248                      $in_literal = false;
1249              }
1250              elseif( $c == '#' || ($c == '-' && substr($sql, $i, 3) == '-- ') )
1251              {
1252                  $in_comment = true;
1253              }
1254              elseif( ctype_space($c) )
1255              {
1256                  $uword = strtoupper($word);
1257                  if( in_array($uword, array('SELECT', 'FROM', 'WHERE', 'GROUP', 'ORDER', 'LIMIT', 'VALUES', 'AND', 'OR', 'LEFT', 'RIGHT', 'INNER')) )
1258                  {
1259                      $new = rtrim($new)."\n".str_pad($word, 6, ' ', STR_PAD_LEFT).' ';
1260                      # Remove any trailing whitespace after keywords
1261                      while( ctype_space($sql[$i+1]) ) {
1262                          ++$i;
1263                      }
1264                  }
1265                  else
1266                  {
1267                      $new .= $word.$c;
1268                  }
1269                  $word = '';
1270                  continue;
1271              }
1272              $word .= $c;
1273          }
1274          $sql = trim($new.$word);
1275  
1276          if( $html )
1277          { // poor man's indent
1278              $sql = preg_replace_callback("~^(\s+)~m", create_function('$m', 'return str_replace(" ", "&nbsp;", $m[1]);'), $sql);
1279              $sql = nl2br($sql);
1280          }
1281          return $sql;
1282      }
