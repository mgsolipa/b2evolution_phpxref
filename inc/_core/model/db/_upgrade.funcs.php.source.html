  21  /**
  22   * Get the delta query to adjust the current database according to a given (list of)
  23   * "CREATE TABLE"-, "CREATE DATABASE"-, "INSERT"- or "UPDATE"-statement(s).
  24   *
  25   * It's not recommended to use INSERT or UPDATE statements with this function, as they
  26   * are just handled "as-is".
  27   *
  28   * NOTE:
  29   *   - You should use single quotes (') to give string type values (this is in fact
  30   *     required for ENUM and SET fields).
  31   *   - KEYs for AUTO_INCREMENT fields should be defined in column_definition, otherwise
  32   *     we had to detect the key type from the INDEX query and add it to the ALTER/ADD query.
  33   *   - If a column changes from "NULL" to "NOT NULL" we generate an extra UPDATE query
  34   *     to prevent "Data truncated for column 'X' at row Y" errors.
  35   *
  36   * The following query types are generated/marked and can be excluded:
  37   *  - 'create_table'
  38   *  - 'create_database'
  39   *  - 'insert'
  40   *  - 'update'
  41   *  - 'drop_column'
  42   *  - 'change_column'
  43   *  - 'change_default'
  44   *  - 'add_column'
  45   *  - 'add_index'
  46   *  - 'drop_index'
  47   *  - 'alter_engine'
  48   * NOTE: it may be needed to merge an 'add_index' or 'drop_index' type query into an
  49   *       'add_column'/'change_column' query (adding "AUTO_INCREMENT" for example)!
  50   *
  51   * NOTE: collations and charset changes are ignored. It seems quite difficult to support this,
  52   *       and it seems to be best to handle this "manually".
  53   *
  54   * @author Originally taken from Wordpress, heavily enhanced and modified by blueyed
  55   *
  56   * @todo Handle COMMENT for tables?!
  57   *
  58   * @see http://dev.mysql.com/doc/refman/4.1/en/create-table.html
  59   *
  60   * @param array The list of queries for which the DB should be adjusted
  61   * @param array Exclude query types (see list above).
  62   * @param boolean Execute generated queries?  TODO: get this outta here!!!! (sooooo bloated!)
  63   * @return array The generated queries.
  64   *        table_name => array of arrays (queries with keys 'queries' (array), 'note' (string) and 'type' (string))
  65   *        There's usually just a single query in "queries", but in some cases additional queries
  66   *        are needed (e.g., 'UPDATE' before we can change "NULL" setting).
  67   */
  68  function db_delta( $queries, $exclude_types = array(), $execute = false )
  69  {
  70      global $Debuglog, $DB, $debug;
  71  
  72      if( ! is_array($queries) )
  73      {
  74          $queries = array( $queries );
  75      }
  76  
  77      if( ! is_array($exclude_types) )
  78      {
  79          $exclude_types = empty($exclude_types) ? array() : array($exclude_types);
  80      }
  81  
  82      /**
  83       * Generated query items, indexed by table name.
  84       */
  85      $items = array();
  86  
  87  
  88      // Split the queries into $items, by their type:
  89      foreach( $queries as $qry )
  90      {
  91          // Remove any comments from the SQL:
  92          $qry = remove_comments_from_query( $qry );
  93  
  94          if( preg_match( '|^(\s*CREATE TABLE\s+)(IF NOT EXISTS\s+)?([^\s(]+)(.*)$|is', $qry, $match) )
  95          {
  96              $tablename = db_delta_remove_quotes(preg_replace( $DB->dbaliases, $DB->dbreplaces, $match[3] ));
  97              $qry = $match[1].( empty($match[2]) ? '' : $match[2] ).$tablename.$match[4];
  98  
  99              $items[strtolower($tablename)][] = array(
 100                  'queries' => array($qry),
 101                  'note' => sprintf( 'Created table &laquo;<strong>%s</strong>&raquo;', $tablename ),
 102                  'type' => 'create_table' );
 103          }
 104          elseif( preg_match( '|^\s*CREATE DATABASE\s([\S]+)|i', $qry, $match) )
 105          { // add to the beginning
 106              array_unshift( $items, array(
 107                  'queries' => array($qry),
 108                  'note' => sprintf( 'Created database &laquo;<strong>%s</strong>&raquo;', $match[1] ),
 109                  'type' => 'create_database' ) );
 110          }
 111          elseif( preg_match( '|^(\s*INSERT INTO\s+)([\S]+)(.*)$|is', $qry, $match) )
 112          {
 113              $tablename = db_delta_remove_quotes(preg_replace( $DB->dbaliases, $DB->dbreplaces, $match[2] ));
 114              $items[strtolower($tablename)][] = array(
 115                  'queries' => array($match[1].$tablename.$match[3]),
 116                  'note' => '',
 117                  'type' => 'insert' );
 118          }
 119          elseif( preg_match( '|^(\s*UPDATE\s+)([\S]+)(.*)$|is', $qry, $match) )
 120          {
 121              $tablename = db_delta_remove_quotes(preg_replace( $DB->dbaliases, $DB->dbreplaces, $match[2] ));
 122              $items[strtolower($tablename)][] = array(
 123                  'queries' => array($match[1].$tablename.$match[3]),
 124                  'note' => '',
 125                  'type' => 'update' );
 126          }
 127          else
 128          {
 129              $Debuglog->add( 'db_delta: Unrecognized query type: '.$qry, 'note' );
 130          }
 131      }
 132  
 133      /**
 134       * @global array Available tables in the current database
 135       */
 136      $tables = $DB->get_col('SHOW TABLES');
 137  
 138      // Loop through existing tables and check which tables and fields exist
 139      foreach($tables as $table)
 140      { // For every table in the database
 141          $table_lowered = strtolower($table);  // table names are treated case insensitive
 142  
 143          if( ! isset( $items[$table_lowered] ) )
 144          { // This table exists in the database, but not in the creation queries.
 145              continue;
 146          }

 148          /**
 149           * @global array Hold the indices we want to create/have, with meta data keys.
 150           */
 151          $indices = array();
 152

 155           */
 156          $obsolete_indices = array();

 159           * @global array Fields of the existing primary key (if any)
 160           */
 161          $existing_primary_fields = array();

 164           * @global array Fields of existing keys (including PRIMARY), lowercased (if any)

 166          $existing_key_fields = array();

 169           * @global array Column field names of PRIMARY KEY, lowercased (if any)
 170           */
 171          $primary_key_fields = array();
 172

 174           * @global array Column field names of FOREIGN KEY, lowercased (if any)
 175           */
 176          $foreign_key_fields = array();
 177

 179           * @global array of col_names that have KEYs (including PRIMARY; lowercased). We use this for AUTO_INCREMENT magic.
 180           */
 181          $fields_with_keys = array();

 183          /**
 184           * @global string Holds the fielddef of an obsolete ("drop_column") AUTO_INCREMENT field. We must alter this with a PK "ADD COLUMN" query.
 185           */
 186          $obsolete_autoincrement = NULL;

 190           * @global array List of fields (and definition from query)
 191           *   <code>fieldname (lowercase) => array(
 192           *         'field' => "column_definition",
 193           *         'where' => "[FIRST|AFTER xxx]" )
 194           *   </code>

 196          $wanted_fields = array();

 199           * @global boolean Do we have any variable-length fields? (see http://dev.mysql.com/doc/refman/4.1/en/silent-column-changes.html)

 207          //echo "<hr/><pre>\n".print_r(strtolower($table), true).":\n".print_r($items, true)."</pre><hr/>";

 209          // ALTER ENGINE, if different (and given in query):
 210          if( ( $wanted_engine = get_engine_from_query( $items[$table_lowered][0]['queries'][0] ) ) !== false )
 211          {
 212              $current_engine = $DB->get_row( '
 213                  SHOW TABLE STATUS LIKE '.$DB->quote($table) );

 215  
 216              if( strtolower($current_engine) != strtolower($wanted_engine) )

 219                              'queries' => array('ALTER TABLE '.$table.' ENGINE='.$wanted_engine),

 222              }
 223          }

 226          foreach( $flds as $create_definition )
 227          { // For every field line specified in the query
 228              // Extract the field name
 229              preg_match( '|^([^\s(]+)|', trim($create_definition), $match );
 230              $fieldname = db_delta_remove_quotes($match[1]);
 231              $fieldname_lowered = strtolower($fieldname);
 232  
 233              $create_definition = trim($create_definition, ", \r\n\t");
 234  
 235              if( in_array( $fieldname_lowered, array( '', 'primary', 'foreign', 'index', 'fulltext', 'unique', 'key' ) ) )
 236              { // INDEX (but not in column_definition - those get handled later)
 237                  $add_index = array(
 238                      'create_definition' => $create_definition,
 239                  );

 241                  if( !  preg_match( '~^(PRIMARY(?:\s+KEY)|(?:FULLTEXT|UNIQUE)(?:\s+(?:INDEX|KEY))?|KEY|INDEX) (?:\s+()     (\w+)      )? (\s+USING\s+\w+)? \s* \((.*)\)$~ix', $create_definition, $match )

 244                  { // invalid type, should not happen
 245                      debug_die( 'Invalid type in $indices: '.$create_definition );
 246                      // TODO: add test: Invalid type in $indices: KEY "coord" ("lon","lat")
 247                  }
 248  
 249                  if( $fieldname_lowered == 'foreign' )
 250                  { // Remember FOREIGN KEY fields, but they don't have to be indexed
 251                      $reference_table_name = db_delta_remove_quotes(preg_replace( $DB->dbaliases, $DB->dbreplaces, $match[4] ));
 252                      $foreign_key_fields[] = array( 'fk_fields' => $match[2], 'reference_table' => $reference_table_name, 'reference_columns' => $match[5], 'fk_definition' => $match[6], 'create' => true );
 253                      continue;
 254                  }
 255  
 256                  $add_index['keyword'] = $match[1];
 257                  $add_index['name'] = strtoupper($match[3]);
 258                  $add_index['type'] = $match[4]; // "USING [type_name]"
 259                  $add_index['col_names'] = explode( ',', $match[5] );
 260                  foreach( $add_index['col_names'] as $k => $v )
 261                  {
 262                      $add_index['col_names'][$k] = strtolower(db_delta_remove_quotes(trim($v)));
 263                  }
 264  
 265                  if( $fieldname_lowered == 'primary' )
 266                  { // Remember PRIMARY KEY fields to be indexed (used for NULL check)
 267                      $primary_key_fields = $add_index['col_names'];
 268                      $add_index['is_PK'] = true;
 269                  }
 270                  else
 271                  {
 272                      $add_index['is_PK'] = false;
 273                  }
 274                  $fields_with_keys = array_unique( array_merge( $fields_with_keys, $add_index['col_names'] ) );
 275  
 276                  $indices[] = $add_index;
 277              }
 278              else
 279              { // "normal" field, add it to the field array
 280                  $wanted_fields[ strtolower($fieldname_lowered) ] = array(
 281                          'field' => $create_definition,
 282                          'where' => ( empty($prev_fld) ? 'FIRST' : 'AFTER '.$prev_fld ),
 283                      );
 284                  $prev_fld = $fieldname;
 285  
 286                  if( preg_match( '~^\S+\s+(VARCHAR|TEXT|BLOB)~i', $create_definition ) )
 287                  {
 288                      $has_variable_length_field = true;
 289                  }
 290              }
 291          }

 294          // INDEX STUFF:
 295  
 296          /**
 297           * @global array Holds the existing indices (with array's key UPPERcased)
 298           */
 299          $existing_indices = array();
 300  
 301          // Fetch the table index structure from the database
 302          $tableindices = $DB->get_results( 'SHOW INDEX FROM '.$table );
 303  
 304          if( ! empty($tableindices) )
 305          {
 306              // For every index in the table
 307              foreach( $tableindices as $tableindex )
 308              {
 309                  // Add the index to the index data array
 310                  $keyname = strtoupper($tableindex->Key_name);
 311  
 312                  $existing_indices[$keyname]['name'] = $tableindex->Key_name; // original case
 313                  $existing_indices[$keyname]['columns'][] = array('fieldname' => $tableindex->Column_name, 'subpart' => $tableindex->Sub_part);
 314                  $existing_indices[$keyname]['unique'] = ($tableindex->Non_unique == 0) ? true : false;
 315              }
 316              unset($tableindices);
 317  
 318              // Let's see which indices are present already for the table:
 319              // TODO: dh> use meta data available now in $indices, instead of building a regular expression!?
 320              $obsolete_indices = $existing_indices; // will get unset as found
 321          }
 322  
 323  
 324          // Pre-run KEYs defined in "column_definition" (e.g. used for AUTO_INCREMENT handling)
 325          foreach( $wanted_fields as $fieldname_lowered => $field_info )
 326          {
 327              $parse = $field_info['field'];
 328  
 329              if( preg_match( '~ \b UNIQUE (?:\s+ KEY)? \b ~ix ', $parse, $match ) )
 330              { // This has an "inline" UNIQUE index:
 331                  if( ! is_in_quote( $parse, ' '.$match[0] ) )
 332                  { // isn't between quotation marks, so it must be a primary key.
 333                      $indices[] = array(
 334                              'name' => $fieldname_lowered,
 335                              'is_PK' => false,
 336                              'create_definition' => NULL, // "inline"
 337                              'col_names' => array($fieldname_lowered),
 338                              'keyword' => NULL,
 339                              #'type' => $match[3], // "USING [type_name]"
 340                          );
 341  
 342                      unset( $obsolete_indices[strtoupper($fieldname_lowered)] );
 343                      $parse = str_replace( $match[0], '', $parse );
 344                      $fields_with_keys[] = $fieldname_lowered;
 345                  }
 346              }
 347  
 348              if( preg_match( '~ \b (PRIMARY\s+)? KEY \b ~ix', $parse, $match ) )
 349              { // inline PK:
 350                  // Check if this key is between quotation marks
 351                  if( ! is_in_quote( $parse, ' '.$match[0] ) )
 352                  { // it isn't between quotation marks, so it must be a primary key.
 353                      $indices[] = array(
 354                              'name' => 'PRIMARY',
 355                              'is_PK' => true,
 356                              'create_definition' => NULL, // "inline"
 357                              'col_names' => array($fieldname_lowered),
 358                              'keyword' => NULL,
 359                              #'type' => $match[3], // "USING [type_name]"
 360                          );
 361                      $fields_with_keys[] = $fieldname_lowered;
 362                      $primary_key_fields = array($fieldname_lowered);
 363                      unset( $obsolete_indices['PRIMARY'] );
 364                  }
 365              }
 366          }
 367          $fields_with_keys = array_unique($fields_with_keys);
 368  
 369  
 370          foreach( $existing_indices as $index_name => $index_data )
 371          {
 372              // Build a create string to compare to the query
 373              $index_pattern = '^';
 374              if( $index_name == 'PRIMARY' )
 375              {
 376                  $index_pattern .= 'PRIMARY(\s+KEY)?';
 377                  // optional primary key name:
 378                  $index_pattern .= '(\s+[`"]?\w+[`"]?)?';
 379              }
 380              elseif( $index_data['unique'] )
 381              {
 382                  $index_pattern .= 'UNIQUE(\s+(?:INDEX|KEY))?';
 383              }
 384              else
 385              {
 386                  $index_pattern .= '(INDEX|(?:FULLTEXT\s+)?KEY)';
 387              }
 388              if( $index_name != 'PRIMARY' )
 389              {
 390                  $index_pattern .= '(\s+[`"]?'.$index_name.'[`"]?)?'; // optionally in backticks (and index name is optionally itself)
 391              }
 392  
 393              $index_columns = '';
 394              // For each column in the index
 395              foreach( $index_data['columns'] as $column_data )
 396              {
 397                  if( $index_columns != '' )
 398                  {
 399                      $index_columns .= '\s*,\s*';
 400                  }
 401                  // Add the field to the column list string
 402                  $index_columns .= '[`"]?'.$column_data['fieldname'].'[`"]?'; // optionally in backticks
 403                  if( ! empty($column_data['subpart']) )
 404                  {
 405                      $index_columns .= '\s*\(\s*'.$column_data['subpart'].'\s*\)\s*';
 406                  }
 407              }
 408  
 409              // Sort index definitions with names to the beginning:
 410              /*
 411              usort( $indices, create_function( '$a, $b', '
 412                  if( preg_match( "~^\w+\s+[^(]~", $a["create_definition"] )
 413                  {
 414  
 415                  }' ) );
 416              */
 417  
 418  
 419              $used_auto_keys = array();
 420              foreach( $indices as $k => $index )
 421              {
 422                  $pattern = $index_pattern;
 423                  if( ! preg_match( '~^\w+\s+[^(]~', $index['create_definition'], $match ) )
 424                  { // no key name given, make the name part optional, if it's the default one:
 425                      // (Default key name seems to be the first column, eventually with "_\d+"-suffix)
 426                      $auto_key = db_delta_remove_quotes(strtoupper($index['col_names'][0]));
 427                      if( isset($used_auto_keys[$auto_key]) )
 428                      {
 429                          $used_auto_keys[$auto_key]++;
 430                          $auto_key .= '_'.$used_auto_keys[$auto_key];
 431                      }
 432                      $used_auto_keys[$auto_key] = 1;
 433  
 434                      if( $auto_key == $index_name )
 435                      { // the auto-generated keyname is the same as the one we have, so make it optional in the pattern:
 436                          $pattern .= '?';
 437                      }
 438                  }
 439                  // Add the column list to the index create string
 440                  $pattern .= '\s*\(\s*'.$index_columns.'\s*\)';
 441  
 442                  #pre_dump( '~'.$pattern.'~i', trim($index['create_definition']) );
 443                  if( preg_match( '~'.$pattern.'~i', trim($index['create_definition']) ) )
 444                  { // This index already exists: remove the index from our indices to create
 445                      unset($indices[$k]);
 446                      unset($obsolete_indices[$index_name]);
 447                      break;
 448                  }
 449              }
 450              if( isset($obsolete_indices[$index_name]) )
 451              {
 452                  #echo "<pre style=\"border:1px solid #ccc;margin-top:5px;\">{$table}:<br/><b>Did not find index:</b>".$index_name.'/'.$index_pattern."<br/>".print_r($indices, true)."</pre>\n";
 453              }
 454          }
 455  
 456          // Set $existing_primary_fields and $existing_key_fields
 457          foreach( $existing_indices as $l_key_name => $l_key_info )
 458          {
 459              $l_key_fields = array();
 460              foreach( $l_key_info['columns'] as $l_col )
 461              {
 462                  $l_key_fields[] = strtolower($l_col['fieldname']);
 463              }
 464              if( $l_key_name == 'PRIMARY' )
 465              { // Remember _existing_ PRIMARY KEYs
 466                  $existing_primary_fields = $l_key_fields;
 467              }
 468  
 469              $existing_key_fields = array_merge( $existing_key_fields, $l_key_fields );
 470          }
 471          $existing_key_fields = array_unique($existing_key_fields);
 472          #pre_dump( 'existing_primary_fields', $existing_primary_fields );
 473          #pre_dump( 'existing_key_fields', $existing_key_fields );
 474

 476          // Fetch the table column structure from the database
 477          $tablefields = $DB->get_results( 'SHOW FULL COLUMNS FROM '.$table );
 478  
 479  
 480          // If "drop_column" is not excluded we have to check if all existing cols would get dropped,
 481          // to prevent "You can't delete all columns with ALTER TABLE; use DROP TABLE instead(Errno=1090)"
 482          if( ! in_array('drop_column', $exclude_types) )
 483          {
 484              $at_least_one_col_stays = false;
 485              foreach($tablefields as $tablefield)
 486              {
 487                  $fieldname_lowered = strtolower($tablefield->Field);
 488  
 489                  if( isset($wanted_fields[ $fieldname_lowered ]) )
 490                  {
 491                      $at_least_one_col_stays = true;
 492                  }
 493              }
 494  
 495              if( ! $at_least_one_col_stays )
 496              { // all columns get dropped: so we need to DROP TABLE and then use the original CREATE TABLE
 497                  array_unshift($items[$table_lowered], array(
 498                      'queries' => array('DROP TABLE '.$table),
 499                      'note' => 'Dropped <strong>'.$table.'.</strong>',
 500                      'type' => 'drop_column' ));
 501                  continue; // next $table
 502              }
 503          }
 504  
 505          // For every field in the existing table
 506          foreach($tablefields as $tablefield)
 507          {
 508              $fieldname_lowered = strtolower($tablefield->Field);
 509  
 510              if( ! isset($wanted_fields[ $fieldname_lowered ]) )
 511              { // This field exists in the table, but not in the creation queries
 512  
 513                  if( in_array('drop_column', $exclude_types) )
 514                  {
 515                      if( preg_match('~\bAUTO_INCREMENT\b~i', $tablefield->Extra) )
 516                      { // must be modified with a ADD COLUMN which drops a PK
 517                          $obsolete_autoincrement = $tablefield;
 518                      }
 519                  }
 520                  else
 521                  {
 522                      $items[$table_lowered][] = array(
 523                          'queries' => array('ALTER TABLE '.$table.' DROP COLUMN '.$tablefield->Field),
 524                          'note' => 'Dropped '.$table.'.<strong>'.$tablefield->Field.'</strong>',
 525                          'type' => 'drop_column' );
 526  
 527                      // Unset in key indices:
 528                      if( ($k = array_search($fieldname_lowered, $existing_key_fields)) !== false )
 529                      {
 530                          unset($existing_key_fields[$k]);
 531                      }
 532                      if( ($k = array_search($fieldname_lowered, $existing_primary_fields)) !== false )
 533                      {
 534                          unset($existing_primary_fields[$k]);
 535                      }
 536                  }
 537  
 538                  continue;
 539              }

 541              $column_definition = trim( $wanted_fields[$fieldname_lowered]['field'] );

 544              $fieldtype = '';
 545

 547  
 548              // Get the field type from the query
 549              if( preg_match( '~^'.$pattern_field.'\s+ (TINYINT|SMALLINT|MEDIUMINT|INTEGER|INT|BIGINT|REAL|DOUBLE|FLOAT|DECIMAL|DEC|NUMERIC) ( \s* \([\d\s,]+\) )? (\s+ UNSIGNED)? (\s+ ZEROFILL)? (.*)$~ix', $column_definition, $match ) )
 550              {
 551                  $fieldtype = strtoupper($match[1]);
 552  
 553                  if( $fieldtype == 'INTEGER' )
 554                  { // synonym
 555                      $fieldtype = 'INT';
 556                  }
 557                  elseif( $fieldtype == 'DEC' )
 558                  { // synonym
 559                      $fieldtype = 'DECIMAL';
 560                  }
 561  
 562                  if( isset($match[2]) )
 563                  { // append optional "length" param (trimmed)
 564                      $fieldtype .= preg_replace( '~\s+~', '', $match[2] );
 565                  }
 566                  if( ! empty($match[3]) )
 567                  { // "unsigned"
 568                      $fieldtype .= ' '.trim($match[3]);
 569                  }
 570                  if( ! empty($match[4]) )
 571                  { // "zerofill"
 572                      $fieldtype .= ' '.trim($match[4]);
 573                  }
 574  
 575                  $field_to_parse = $match[5];
 576  
 577                  // The length param is optional:
 578                  if( substr($fieldtype, 0, 7) == 'DECIMAL' )
 579                      $matches_pattern = '~^'.preg_quote($tablefield->Type, '~').'$~i';
 580                  else
 581                      $matches_pattern = '~^'.preg_replace( '~\((\d+)\)~', '(\(\d+\))?', $tablefield->Type ).'$~i';
 582                  $type_matches = preg_match( $matches_pattern, $fieldtype );
 583              }
 584              elseif( preg_match( '~^'.$pattern_field.'\s+(DATETIME|DATE|TIMESTAMP|TIME|YEAR|TINYBLOB|BLOB|MEDIUMBLOB|LONGBLOB|TINYTEXT|TEXT|MEDIUMTEXT|LONGTEXT) ( \s+ BINARY )? (.*)$~ix', $column_definition, $match ) )
 585              {
 586                  $fieldtype = strtoupper($match[1]);
 587                  if( isset($match[2]) )
 588                  { // "binary"
 589                      $fieldtype .= trim($match[2]);
 590                  }
 591                  $field_to_parse = $match[3];
 592  
 593                  // There's a bug with a "NOT NULL" field reported as "NULL", work around it (http://bugs.mysql.com/bug.php?id=20910):
 594                  if( $fieldtype == 'TIMESTAMP' )
 595                  {
 596                      $ct_sql = $DB->get_var( 'SHOW CREATE TABLE '.$table, 1, 0 );
 597                      if( preg_match( '~^\s*`'.$tablefield->Field.'`\s+TIMESTAMP\s+(NOT )?NULL~im', $ct_sql, $match ) )
 598                      {
 599                          $tablefield->Null = empty($match[1]) ? 'YES' : 'NO';
 600                      }
 601                  }
 602              }
 603              elseif( preg_match( '~^'.$pattern_field.'\s+ (CHAR|VARCHAR|BINARY|VARBINARY) \s* \( ([\d\s]+) \) (\s+ (BINARY|ASCII|UNICODE) )? (.*)$~ix', $column_definition, $match ) )
 604              {
 605                  $len = trim($match[2]);
 606                  $fieldtype = strtoupper($match[1]).'('.$len.')';
 607  
 608                  if( ! empty($match[3]) )
 609                  { // "binary", "ascii", "unicode"
 610                      $fieldtype .= ' '.$match[3];
 611                  }
 612                  $field_to_parse = $match[5];
 613  
 614                  if( strtoupper($match[1]) == 'VARCHAR' )
 615                  {
 616                      if( $len < 4 )
 617                      { // VARCHAR shorter than 4 get converted to CHAR (but reported as VARCHAR in MySQL 5.0)
 618                          $type_matches = preg_match( '~^(VAR)?CHAR\('.$len.'\)'.( $match[3] ? ' '.$match[3] : '' ).'$~i', $tablefield->Type );
 619                      }
 620                  }
 621                  elseif( $has_variable_length_field && strtoupper($match[1]) == 'CHAR' )
 622                  { // CHARs in a row with variable length fields get silently converted to VARCHAR (but reported as CHAR in MySQL 5.0)
 623                      $type_matches = preg_match( '~^(VAR)?'.preg_quote( $fieldtype, '~' ).'$~i', $tablefield->Type );
 624                  }
 625              }
 626              elseif( preg_match( '~^'.$pattern_field.'\s+ (ENUM|SET) \s* \( (.*) \) (.*)$~ix', $column_definition, $match ) )
 627              {
 628                  $values = preg_split( '~\s*,\s*~', trim($match[2]), -1, PREG_SPLIT_NO_EMPTY ); // TODO: will fail for values containing ","..
 629                  $values = implode( ',', $values );
 630  
 631                  $fieldtype = strtoupper($match[1]).'('.$values.')';
 632                  $field_compare = strtolower($match[1]).'('.$values.')';
 633  
 634                  // compare case-sensitive
 635                  $type_matches = ( $field_compare == $tablefield->Type );
 636  
 637                  $field_to_parse = $match[3];
 638              }
 639              else
 640              {
 641                  if( $debug )
 642                  {
 643                      debug_die( 'db_delta(): Cannot find existing types field in column definition ('.$pattern_field.'/'.$column_definition.')' );
 644                  }
 645                  continue;
 646              }
 647  
 648  
 649              // DEFAULT
 650              $want_default = false;
 651              if( preg_match( '~^(.*?) \s DEFAULT \s+ (?: (?: (["\']) (.*?) \2 ) | (\w+) ) (\s .*)?$~ix', $field_to_parse, $match ) )
 652              {
 653                  if( isset($match[4]) && $match[4] !== '' )
 654                  {
 655                      $want_default = $match[4];
 656                      $want_default_set = $match[4];
 657                  }
 658                  else
 659                  {
 660                      $want_default = $match[3];
 661                      $want_default_set = $match[2].$match[3].$match[2];  // encapsulate in quotes again
 662                  }
 663  
 664                  $field_to_parse = $match[1].( isset($match[5]) ? $match[5] : '' );
 665              }
 666  
 667  
 668              // KEY
 669              $has_inline_primary_key = false;
 670              if( preg_match( '~^(.*) \b (?: (UNIQUE) (?:\s+ (?:INDEX|KEY))? | (?:PRIMARY \s+)? KEY ) \b (.*)$~ix', $field_to_parse, $match ) )
 671              { // fields got added to primary_key_fields and fields_with_keys before
 672                  // Check if this key is between quotation marks
 673                  if( ! has_open_quote( $match[1] ) )
 674                  { // is not between quotation marks, so it must be a key.
 675                      $field_to_parse = $match[1].$match[3];
 676                      if( empty($match[2]) )
 677                      {
 678                          $has_inline_primary_key = true; // we need to DROP the PK if this column definition does not match
 679                      }
 680                  }
 681              }
 682  
 683  
 684              // AUTO_INCREMENT (with special index handling: AUTO_INCREMENT fields need to be PRIMARY or UNIQUE)
 685              $is_auto_increment = false;
 686              if( preg_match( '~(.*?) \b AUTO_INCREMENT \b (.*)$~ix', $field_to_parse, $match ) )
 687              {
 688                  $is_auto_increment = true;
 689                  $field_to_parse = $match[1].$match[2];
 690  
 691                  if( ! preg_match( '~\bAUTO_INCREMENT\b~i', $tablefield->Extra ) )
 692                  { // not AUTO_INCREMENT yet
 693                      $type_matches = false;
 694                  }
 695  
 696                  if( ! in_array( $fieldname_lowered, $fields_with_keys ) )
 697                  { // no KEY defined (but required for AUTO_INCREMENT fields)
 698                      debug_die('No KEY/INDEX defined for AUTO_INCREMENT column!');
 699                  }
 700  
 701                  if( in_array( $fieldname_lowered, $existing_key_fields ) )
 702                  {
 703                      if( ! empty( $primary_key_fields ) )
 704                      {
 705                          $column_definition .= ', DROP PRIMARY KEY';
 706                          unset( $obsolete_indices['PRIMARY'] );
 707                      }
 708                  }
 709                  else
 710                  { // a key for this AUTO_INCREMENT field does not exist yet, we search it in $indices
 711                      foreach( $indices as $k_index => $l_index )
 712                      { // go through the indexes we want to have
 713  
 714                          if( array_search( $fieldname_lowered, $l_index['col_names'] ) === false )
 715                          { // this is not an index for our column
 716                              continue;
 717                          }
 718  
 719                          // this index definition affects us, we have to add it to our ALTER statement..
 720  
 721                          // See if we need to drop it, before adding it:
 722                          if( $l_index['is_PK'] )
 723                          { // Part of a PRIMARY key..
 724                              if( ! empty( $existing_primary_fields ) )
 725                              { // and a PRIMARY key exists already
 726                                  $column_definition .= ', DROP PRIMARY KEY';
 727                                  unset( $obsolete_indices['PRIMARY'] );
 728                              }
 729                              $existing_primary_fields = array(); // we expect no existing primary key anymore
 730                              $primary_key_fields = $l_index['col_names']; // this becomes our primary key
 731                          }
 732                          elseif( isset( $existing_indices[$l_index['name']] ) )
 733                          { // this index already exists, drop it:
 734                              $column_definition .= ', DROP INDEX '.$existing_indices[$l_index['name']]; // original case
 735                              unset( $existing_indices[$l_index['name']] ); // we expect that it does not exist anymore
 736                              if( ! in_array( $fieldname_lowered, $fields_with_keys ) )
 737                              { // add te field to the list of keys we want/expect to have:
 738                                  $fields_with_keys[] = $fieldname_lowered;
 739                              }
 740                          }
 741  
 742                          // Merge the INDEX creation into our ALTER query:
 743                          $column_definition .= ', ADD '.$l_index['create_definition'];
 744                          unset( $indices[$k_index] );
 745                      }
 746                  }
 747              }
 748  
 749  
 750              // "[NOT] NULL" (requires $primary_key_fields to be finalized)
 751              if( preg_match( '~(.*?) \b (NOT\s+)? NULL \b (.*)$~ix', $field_to_parse, $match ) )
 752              { // if "NOT" not matched it's NULL
 753                  $want_null = empty($match[2]);
 754                  $field_to_parse = $match[1].$match[3];
 755              }
 756              else
 757              { // not specified: "NULL" is default
 758                  $want_null = true;
 759              }
 760  
 761              if( in_array($fieldname_lowered, $primary_key_fields) || $is_auto_increment )
 762              { // If part of PRIMARY KEY or AUTO_INCREMENT field "NULL" is implicit
 763                  $change_null = false; // implicit NULL
 764                  $want_null = 'IMPLICIT';
 765              }
 766              elseif( in_array($fieldname_lowered, $existing_primary_fields) && ! in_array($fieldname_lowered, $primary_key_fields) )
 767              { // the field was in PRIMARY KEY, but is no longer. It should get altered only if we want "NOT NULL"
 768                  $change_null = ( ! $want_null && $tablefield->Null == 'YES' );
 769                  #pre_dump( $want_null );
 770                  #$want_null = 'IMPLICIT2';
 771                  #pre_dump( $primary_key_fields );
 772              }
 773              else
 774              {
 775                  if( $tablefield->Null == 'YES' )
 776                  {
 777                      $change_null = ! $want_null;
 778                  }
 779                  else
 780                  { // I've seen '' and 'NO' for no..
 781                      $change_null = $want_null;
 782                  }
 783              }
 784  
 785  
 786              // COMMENT ( check if there is difference in field comment )
 787              if( preg_match( '~^(.*?) \s COMMENT \s+ (?: (?: (["\']) (.*?) \2 ) ) (\s .*)?$~ix', $field_to_parse, $match ) )
 788              {
 789                  if( isset($match[4]) && $match[4] !== '' )
 790                  {
 791                      $want_comment = $match[4];
 792                  }
 793                  else
 794                  {
 795                      $want_comment = $match[3];
 796                  }
 797  
 798                  $want_comment = stripslashes( $want_comment );
 799                  $type_matches = $tablefield->Comment == $want_comment;
 800  
 801                  $field_to_parse = $match[1].( isset($match[5]) ? $match[5] : '' );
 802              }
 803  
 804  
 805              // TODO: "COLLATE" and other attribute handling should happen here, based on $field_to_parse
 806  
 807  
 808              if( ! isset($type_matches) )
 809              { // not tried to match before
 810                  $type_matches = ( strtoupper($tablefield->Type) == $fieldtype );
 811              }
 812  
 813              #pre_dump( 'change_null ($change_null, $tablefield, $want_null)', $change_null, $tablefield, $want_null );
 814              #pre_dump( 'type_matches', $type_matches, strtolower($tablefield->Type), $fieldtype );
 815  
 816  
 817              // See what DEFAULT we would get or want
 818              $update_default = NULL;
 819              $update_default_set = NULL;
 820  
 821              if( $want_default !== false )
 822              {
 823                  $update_default = $want_default;
 824                  $update_default_set = $want_default_set;
 825              }
 826              else
 827              { // implicit default, see http://dev.mysql.com/doc/refman/4.1/en/data-type-defaults.html
 828                  if( preg_match( '~^(TINYINT|SMALLINT|MEDIUMINT|INTEGER|INT|BIGINT|REAL|DOUBLE|FLOAT|DECIMAL|DEC|NUMERIC)~', $fieldtype ) )
 829                  { // numeric
 830                      $update_default = '0';
 831                      $update_default_set = '0';
 832                  }
 833                  elseif( $fieldtype == 'TIMESTAMP' )
 834                  { // TODO: the default should be current date and time for the first field - but AFAICS we won't have NULL fields anyway
 835                  }
 836                  elseif( preg_match( '~^(DATETIME|DATE|TIME|YEAR)$~', $fieldtype ) )
 837                  {
 838                      $update_default = '0'; // short form for various special "zero" values
 839                      $update_default_set = '0';
 840                  }
 841                  elseif( substr($fieldtype, 0, 4) == 'ENUM' )
 842                  {
 843                      preg_match( '~["\']?.*?["\']?\s*[,)]~x', substr($fieldtype,5), $match );
 844                      $update_default_set = trim( $match[0], "\n\r\t\0\x0bB ()," ); // strip default whitespace, braces & comma
 845                      // first value (until "," or end) of $fieldtype_param:
 846                      $update_default = preg_replace( '~^(["\'])(.*)\1$~', '$2', $update_default_set ); // without quotes
 847                  }
 848                  else
 849                  {
 850                      $update_default_set = "''"; // empty string for string types
 851                      $update_default = '';
 852                  }
 853              }
 854  
 855  
 856              // Is actual field type different from the field type in query?
 857              if( ! $type_matches || $change_null )
 858              { // Change the whole column to $column_definition:
 859                  /*
 860                  echo '<h2>No_Match</h2>';
 861                  pre_dump( $type_matches, $change_null, $want_null );
 862                  pre_dump( $tablefield, $column_definition );
 863                  pre_dump( 'flds', $flds );
 864                  pre_dump( 'wanted_fields', $wanted_fields );
 865                  pre_dump( strtolower($tablefield->Type), $fieldtype, $column_definition );
 866                  */
 867  
 868                  $queries = array( 'ALTER TABLE '.$table );
 869  
 870                  // Handle inline PRIMARY KEY definition:
 871                  if( $has_inline_primary_key && ! empty($existing_primary_fields) ) // there's a PK that needs to get removed
 872                  { // the column is part of the PRIMARY KEY, which needs to get dropped before (we already handle that for AUTO_INCREMENT fields)
 873                      $queries[0] .= ' DROP PRIMARY KEY,';
 874                      $existing_primary_fields = array(); // we expect no existing primary key anymore
 875                      unset( $obsolete_indices['PRIMARY'] );
 876                  }
 877  
 878                  $queries[0] .= ' CHANGE COLUMN '.$tablefield->Field.' '.$column_definition;
 879  
 880                  // Handle changes from "NULL" to "NOT NULL"
 881                  if( $change_null && ! $want_null && isset($update_default_set) )
 882                  { // Prepend query to update NULL fields to default
 883                      array_unshift( $queries, 'UPDATE '.$table.' SET '.$tablefield->Field.' = '.$update_default_set.' WHERE '.$tablefield->Field.' IS NULL' );
 884  
 885                      if( substr( $tablefield->Type, 0, 5 ) == 'enum(' )
 886                      {
 887                          $existing_enum_field_values = preg_split( '~\s*,\s*~', substr( $tablefield->Type, 5, -1 ), -1, PREG_SPLIT_NO_EMPTY );
 888  
 889                          foreach( $existing_enum_field_values as $k => $v )
 890                          {
 891                              $existing_enum_field_values[$k] = preg_replace( '~^(["\'])(.*)\1$~', '$2', $v ); // strip quotes
 892                          }
 893  
 894                          if( ! in_array( $update_default, $existing_enum_field_values ) )
 895                          { // we cannot update straight to the new default, because it does not exist yet!
 896  
 897                              // Update the column first, without the NULL change
 898                              array_unshift( $queries, 'ALTER TABLE '.$table.' CHANGE COLUMN '.$tablefield->Field.' '.preg_replace( '~\sNOT\s+NULL~i', '', $column_definition ) );
 899                          }
 900                      }
 901                  }
 902  
 903                  // Add a query to change the column type
 904                  $items[$table_lowered][] = array(
 905                      'queries' => $queries,
 906                      'note' => 'Changed type of '.$table.'.<strong>'.$tablefield->Field.'</strong> from '.$tablefield->Type.' to '.$column_definition,
 907                      'type' => 'change_column' );
 908              }
 909              else
 910              { // perhaps alter or drop DEFAULT:
 911                  if( $want_default !== false )
 912                  { // DEFAULT given
 913                      $existing_default = $tablefield->Default === NULL ? 'NULL' : $tablefield->Default;
 914  
 915                      if( $existing_default != $want_default ) // DEFAULT is case-sensitive
 916                      { // Add a query to change the column's default value
 917                          $item_queries = array();
 918                          if( $existing_default == 'CURRENT_TIMESTAMP' )
 919                          {    // MySQL command "SET DEFAULT" does NOT upgrade a column with default value CURRENT_TIMESTAMP by some reason,
 920                              // So we must use the command "MODIFY" to update a column completely:
 921                              $item_queries[] = 'ALTER TABLE '.$table.' MODIFY '.$column_definition;
 922                          }
 923                          else
 924                          {    // Use "SET DEFAULT" for all other normal column types:
 925                              $item_queries[] = 'ALTER TABLE '.$table.' ALTER COLUMN '.$tablefield->Field.' SET DEFAULT '.$want_default_set;
 926                          }
 927                          $items[$table_lowered][] = array(
 928                              'queries' => $item_queries,
 929                              'note' => "Changed default value of {$table}.<strong>{$tablefield->Field}</strong> from $existing_default to $want_default_set",
 930                              'type' => 'change_default' );
 931                      }
 932                  }
 933                  elseif( ! empty($tablefield->Default) && $tablefield->Default != $update_default )
 934                  { // No DEFAULT given, but it exists one, so drop it (IF not a TIMESTAMP or DATETIME field)
 935                      if( $tablefield->Type != 'timestamp' && $tablefield->Type != 'datetime' )
 936                      {
 937                          $items[$table_lowered][] = array(
 938                              'queries' => array('ALTER TABLE '.$table.' ALTER COLUMN '.$tablefield->Field.' DROP DEFAULT'),
 939                              'note' => "Dropped default value of {$table}.<strong>{$tablefield->Field}</strong>",
 940                              'type' => 'change_default' ); // might be also 'drop_default'
 941                      }
 942                  }
 943              }
 944  
 945              // Remove the field from the array (so it's not added)
 946              unset($wanted_fields[$fieldname_lowered]);
 947          }
 948  
 949  
 950          foreach($wanted_fields as $fieldname_lowered => $fielddef)
 951          { // For every remaining field specified for the table
 952              $column_definition = $fielddef['field'].' '.$fielddef['where'];
 953  
 954              $is_auto_increment = false;
 955              // AUTO_INCREMENT (with special index handling: AUTO_INCREMENT fields need to be PRIMARY or UNIQUE)
 956              if( preg_match( '~(.*?) \b AUTO_INCREMENT \b (.*)$~ix', $fielddef['field'], $match ) )
 957              {
 958                  if( ! in_array( $fieldname_lowered, $fields_with_keys ) )
 959                  { // no KEY defined (but required for AUTO_INCREMENT fields)
 960                      debug_die('No KEY/INDEX defined for AUTO_INCREMENT column!');
 961                  }
 962                  $is_auto_increment = true;
 963  
 964  
 965                  foreach( $indices as $k_index => $l_index )
 966                  { // go through the indexes we want to have
 967  
 968                      if( array_search( $fieldname_lowered, $l_index['col_names'] ) === false )
 969                      { // this is not an index for our column
 970                          continue;
 971                      }
 972  
 973                      // this index definition affects us, we have to add it to our ALTER statement..
 974  
 975                      // See if we need to drop it, before adding it:
 976                      if( $l_index['is_PK'] )
 977                      { // Part of a PRIMARY key..
 978                          if( ! empty( $existing_primary_fields ) )
 979                          { // and a PRIMARY key exists already
 980                              $column_definition .= ', DROP PRIMARY KEY';
 981                              unset( $obsolete_indices['PRIMARY'] );
 982                          }
 983                          $existing_primary_fields = array(); // we expect no existing primary key anymore
 984                          $primary_key_fields = $l_index['col_names']; // this becomes our primary key
 985                      }
 986                      elseif( isset( $existing_indices[$l_index['name']] ) )
 987                      { // this index already exists, drop it:
 988                          $column_definition .= ', DROP INDEX '.$existing_indices[$l_index['name']]; // original case
 989                          unset( $existing_indices[$l_index['name']] ); // we expect that it does not exist anymore
 990                          if( ! in_array( $fieldname_lowered, $fields_with_keys ) )
 991                          { // add te field to the list of keys we want/expect to have:
 992                              $fields_with_keys[] = $fieldname_lowered;
 993                          }
 994                      }
 995  
 996                      // Merge the INDEX creation into our ALTER query:
 997                      $column_definition .= ', ADD '.$l_index['create_definition'];
 998                      unset( $indices[$k_index] );
 999                  }
1000              }
1001  
1002              // Push a query line into $items that adds the field to that table
1003              $query = 'ALTER TABLE '.$table.' ADD COLUMN '.$column_definition;
1004  
1005              // Handle inline PRIMARY KEY definition:
1006              if( preg_match( '~^(.*) \b (?: (UNIQUE) (?:\s+ (?:INDEX|KEY))? | (?:PRIMARY \s+)? KEY ) \b (.*)$~ix', $column_definition, $match ) // "has_inline_primary_key"
1007                      && count($existing_primary_fields)
1008                      && ! in_array($fieldname_lowered, $existing_primary_fields) )
1009              { // the column is part of the PRIMARY KEY, which needs to get dropped before (we already handle that for AUTO_INCREMENT fields)
1010                  $query .= ', DROP PRIMARY KEY';
1011                  $existing_primary_fields = array(); // we expect no existing primary key anymore
1012                  unset( $obsolete_indices['PRIMARY'] );
1013  
1014                  if( isset($obsolete_autoincrement) )
1015                  {
1016                      $query .= ', MODIFY COLUMN '.$obsolete_autoincrement->Field.' '.$obsolete_autoincrement->Type.' '.( $obsolete_autoincrement->Field == 'YES' ? 'NULL' : 'NOT NULL' );
1017                  }
1018              }
1019  
1020              $items[$table_lowered][] = array(
1021                  'queries' => array($query),
1022                  'note' => 'Added column '.$table.'.<strong>'.$fielddef['field'].'</strong>',
1023                  'type' => 'add_column' );
1024          }
1025  
1026  
1027          // Remove the original table creation query from processing
1028          array_shift( $items[$table_lowered] );
1029  
1030          // Add foreign key constraints
1031          $result = db_delta_foreign_keys( $foreign_key_fields, $table, false );
1032          foreach( $result as $foreign_key_update )
1033          { // loop through foreign key differences in this table
1034              if( $foreign_key_update['type'] == 'alter_engine' )
1035              { // this is an alter engine command, check if this command was already added during engine difference detection, and skip to the next if it was added
1036                  $skip = false;
1037                  foreach( $items[$table_lowered] as $itemlist )
1038                  {
1039                      if( ( $itemlist[ 'type' ] == 'alter_engine' ) && ( $itemlist['queries'][0] == $foreign_key_update['queries'][0] ) )
1040                      { // the same command was already added, don't add again
1041                          $skip = true;
1042                          break;
1043                      }
1044                  }
1045                  if( $skip )
1046                  { // skip is set, don't add the alter engine command again
1047                      continue;
1048                  }
1049              }
1050              // add FK updates
1051              $items[$table_lowered][] = $foreign_key_update;
1052          }
1053  
1054          // Add the remaining indices (which are not "inline" with a column definition and therefor already handled):
1055          $add_index_queries = array();
1056          foreach( $indices as $k => $index )
1057          {
1058              if( empty($index['create_definition']) )
1059              { // skip "inline"
1060                  continue;
1061              }
1062              $query = 'ALTER TABLE '.$table;
1063              if( $index['is_PK'] && $existing_primary_fields )
1064              {
1065                  $query .= ' DROP PRIMARY KEY,';
1066                  unset( $obsolete_indices['PRIMARY'] );
1067              }
1068  
1069              // Create a query that adds the index to the table
1070              $query = array(
1071                  'queries' => array($query.' ADD '.$index['create_definition']),
1072                  'note' => 'Added index <strong>'.$index['create_definition'].'</strong>',
1073                  'type' => 'add_index',
1074                  'name' => $index['name'] );
1075  
1076              // Check if the index creation has to get appended after any DROPs (required for indices with the same name)
1077              $append_after_drops = false;
1078              foreach( $obsolete_indices as $obsolete_index )
1079              {
1080                  if( strtolower($obsolete_index['name']) == strtolower($index['name']) )
1081                  {
1082                      $append_after_drops = true;
1083                      break;
1084                  }
1085              }
1086              if( $append_after_drops )
1087              { // do this after any DROPs (i.e. KEY name changes)
1088                  $add_index_queries[] = $query;
1089              }
1090              else
1091              { // this needs to get done before any other DROPs
1092                  // to prevent e.g. "Incorrect table definition; there can be only one auto column and it must be defined as a key(Errno=1075)"
1093                  $items[$table_lowered][] = $query;
1094              }
1095          }
1096  
1097          // Now add queries to drop any (maybe changed!) indices
1098          foreach( $obsolete_indices as $index_info )
1099          {
1100              // Push a query line into $items that drops the index from the table
1101              $items[$table_lowered][] = array(
1102                  'queries' => array("ALTER TABLE {$table} DROP ".( $index_info['name'] == 'PRIMARY' ? 'PRIMARY KEY' : 'INDEX '.$index_info['name'] )),
1103                  'note' => 'Dropped index <strong>'.$index_info['name'].'</strong>',
1104                  'type' => 'drop_index',
1105                  'name' => $index_info['name'] );
1106          }
1107  
1108          // Add queries to (re)create (maybe changed indices) to the end
1109          $items[$table_lowered] = array_merge($items[$table_lowered], $add_index_queries);
1110      }
1111  
1112  
1113      // Filter types we want to exclude:
1114      if( ! empty($exclude_types) )
1115      {
1116          foreach( $items as $table => $itemlist )
1117          {
1118              $removed_one = false;
1119              foreach( $itemlist as $k => $item )
1120              {
1121                  if( in_array($item['type'], $exclude_types) )
1122                  { // this type of update should be excluded
1123                      if( $item['type'] == 'drop_index' )
1124                      { // drop index command should not be excluded in case when we would like to update an index!
1125                          $skip = false;
1126                          foreach( $itemlist as $other_item )
1127                          { // check if there are an add_index command for the same table with the same index name
1128                              if( ( $other_item['type'] == 'add_index' ) && ( strcasecmp( $item['name'], $other_item['name'] ) === 0 ) )
1129                              { // add index with the same index name was found so we need to process this drop_index command to be able to add a new correct index with the same name
1130                                  $skip = true;
1131                                  break;
1132                              }
1133                          }
1134                          if( $skip )
1135                          { // skip excluding this item
1136                              continue;
1137                          }
1138                      }
1139                      unset( $items[$table][$k] );
1140                      $removed_one = true;
1141                  }
1142              }
1143              if( $removed_one )
1144              { // Re-order (0, 1, 2, ..)
1145                  $items[$table] = array_values($items[$table]);
1146              }
1147          }
1148      }
1149  
1150      // Unset empty table indices:
1151      foreach( $items as $table => $itemlist )
1152      {
1153          if( empty($itemlist) )
1154          {
1155              unset( $items[$table] );
1156              continue;
1157          }
1158  
1159          // Check if we have alter engine and drop foreign key queries for the same table. In this case the drop query must be processed before the alter engine query!
1160          $alter_engine_index = NULL;
1161          for( $i = 0; $i < count( $itemlist ); $i++ )
1162          {
1163              if( ( $itemlist[$i]['type'] == 'alter_engine' ) && ( $alter_engine_index == NULL ) )
1164              { // save alter engine query index
1165                  $alter_engine_index = $i;
1166              }
1167              elseif( ( $itemlist[$i]['type'] == 'drop_foreign_key' ) && ( $alter_engine_index !== NULL ) && ( $alter_engine_index < $i ) )
1168              { // switch engine update and drop foreign key queries, because in many cases we must drop the foreign key first to be able to chagne the table engine
1169                  $switch_item = $itemlist[$alter_engine_index];
1170                  $items[$table][$alter_engine_index] = $itemlist[$i];
1171                  $items[$table][$i] = $switch_item;
1172                  // save new alter engine index and the alter engine command in case of we have to drop multiple foreign keys
1173                  $alter_engine_index = $i;
1174                  $itemlist[$i] = $switch_item;
1175              }
1176          }
1177      }
1178  
1179      if( $execute )
1180      {
1181          foreach( $items as $table => $itemlist )
1182          {
1183              foreach( $itemlist as $item )
1184              {
1185                  foreach( $item['queries'] as $query )
1186                  {
1187                      #pre_dump( $query );
1188                      $DB->query( $query );
1189                  }
1190              }
1191          }
1192      }
1193  
1194      return $items;
1195  }

1222  /**
1223   * Get the delta queries between existing foreign key values and new foreign key values in the given table
1224   *
1225   * @param array foreign key lines from the install script Create Table commands
1226   * @param string the processed table name
1227   * @param boolean set to false to not process the required queries without asking the user consent
1228   * @param string leave this to NUL for delta, set to 'add' to add a new foreign key and set to 'drop' to drop a single foreign key.
1229   * @return array the delta queries if there are any or empty array if the required db queries have been processed or if there are no foreign key differences
1230   */
1231  function db_delta_foreign_keys( $foreign_key_fields, $table, $silent = true, $action = NULL )
1232  {
1233      global $DB;
1234  
1235      $result = array();
1236  
1237      // get create table sql from db
1238      $ct_sql = $DB->get_var( 'SHOW CREATE TABLE '.$table, 1, 0 );
1239  
1240      // Check related tables engine because the foreign key table and the reference table both must have InnoDB engine
1241      if( !empty( $foreign_key_fields ) && ( $action != 'drop' ) )
1242      { // we have foreign key contraints, and we don't want to drop that ( In case of drop FK the table engine doesn't matter )
1243          // which tables engine must be changed
1244          $modify_table_eninges = array( $table => 'InnoDB' );
1245          foreach( $foreign_key_fields as $foreign_key )
1246          { // check reference tables engine
1247              $modify_table_eninges[ $foreign_key['reference_table'] ] = 'InnoDB';
1248          }
1249          $engine_queries = db_delta_table_engines( $modify_table_eninges, $silent );
1250          if( !$silent && !empty( $engine_queries ) )
1251          {
1252              $result = array_merge( $engine_queries, $result );
1253          }
1254      }
1255  
1256      // get foreign key constraints from db
1257      $existing_foreign_key_fields = array();
1258      $db_fieldlines = get_fieldlines_from_query( $ct_sql );
1259      foreach( $db_fieldlines as $fieldname => $fieldline )
1260      { // loop through all field lines and get those lines where are foreign key definitions
1261          $fieldline = str_replace( array( '`', '"' ), '', $fieldline );
1262          if( preg_match( '~^(?:(CONSTRAINT)* \s* ([^ ]*)) \s* (FOREIGN\s+KEY) \s* \((.*)\) \s* (REFERENCES) \s* ([^( ]*) \s* \((.*)\) \s* (.*)$~ixs', $fieldline, $match ) )
1263          { // add existing foreign key fields, but set drop param to true only if we don't want to add/drop a single foreign key!
1264              $existing_foreign_key_fields[] = array( 'fk_symbol' => $match[2], 'fk_fields' => $match[4], 'reference_table' => $match[6], 'reference_columns' => $match[7], 'fk_definition' => $match[8], 'drop' => ( $action == NULL ) );
1265          }
1266      }
1267  
1268      foreach( $existing_foreign_key_fields as &$existing_foreign_key )
1269      { // loop through existing foreign key definitions
1270          foreach( $foreign_key_fields as &$foreign_key )
1271          { // loop through the install script foreign key definitions
1272              if( ( $action == NULL ) && ( !$existing_foreign_key['drop'] ) && ( !$foreign_key['create'] ) )
1273              { // if we already know that an old key should not be removed, and the recent key already exists skip this check
1274                  continue;
1275              }
1276              // check if the two foreign key constraint is the same or not
1277              $match_found = ( $existing_foreign_key['fk_fields'] == $foreign_key['fk_fields'] )
1278                  && ( $existing_foreign_key['reference_table'] == $foreign_key['reference_table'] )
1279                  && ( $existing_foreign_key['reference_columns'] == $foreign_key['reference_columns'] );
1280              $exact_match_found = ( $match_found && ( $existing_foreign_key['fk_definition'] == $foreign_key['fk_definition'] ) );
1281              if( ( !empty( $action ) ) && $match_found )
1282              { // action is not empty it means that we would like to add or drop a single foreign key
1283                  if( $action == 'drop' )
1284                  { // set existing foreign key to be droped
1285                      $existing_foreign_key['drop'] = true;
1286                      break;
1287                  }
1288                  elseif( $action = 'add' )
1289                  { // add a new foreign key
1290                      if( $exact_match_found )
1291                      { // Exact match found so the FK already exists with the same definition
1292                          return;
1293                      }
1294                      // foreign key exists but not with the given definition so we have to drop the old FK
1295                      $existing_foreign_key['drop'] = true;
1296                      break;
1297                  }
1298              }
1299              // if exact match found betwen existing and recent foreign keys then we should not drop the old one
1300              $existing_foreign_key['drop'] = $existing_foreign_key['drop'] && ( !$exact_match_found );
1301              // if recent foreign keys already exists then we doesn't have to create a new one
1302              $foreign_key['create'] = $foreign_key['create'] && ( !$exact_match_found );
1303          }
1304          if( ( $action !== NULL ) && ( $existing_foreign_key['drop'] ) )
1305          { // in case of add/drop a single foreign key, if we have found a match, then we don't have to look forward
1306              break;
1307          }
1308      }
1309      unset( $existing_foreign_key );
1310      unset( $foreign_key );
1311  
1312      foreach( $existing_foreign_key_fields as $existing_foreign_key )
1313      { // loop through existing foreign keys
1314          if( $existing_foreign_key['drop'] )
1315          { // this foreign key constraint should be removed, create the query
1316              $query = 'ALTER TABLE '.$table.' DROP FOREIGN KEY '.$existing_foreign_key['fk_symbol'];
1317              if( $silent )
1318              { // execute query in silent mode
1319                  $DB->query( $query );
1320              }
1321              else
1322              { // set query definition
1323                  $result[] = array(
1324                      'queries' => array( $query ),
1325                      'note' => 'Drop <strong>'.$existing_foreign_key['fk_symbol'].'</strong> foreign key constraint from <strong>'.$table.'</strong> table.',
1326                      'type' => 'drop_foreign_key' );
1327              }
1328          }
1329      }
1330      foreach( $foreign_key_fields as $foreign_key )
1331      { // loop through in up to date foreign keys
1332          if( $foreign_key['create'] )
1333          { // // this foreign key constraint is new, it must be created
1334              // Create delete query for orphan entries
1335              $delete_query = 'DELETE FROM '.$table.
1336                                  ' WHERE '.$foreign_key['fk_fields'].' NOT IN (
1337                                      SELECT DISTINCT('.$foreign_key['reference_columns'].') FROM '.$foreign_key['reference_table'].' )';
1338              $query = 'ALTER TABLE '.$table.' ADD FOREIGN KEY ('.$foreign_key['fk_fields'].') REFERENCES '.$foreign_key['reference_table'].' ('.$foreign_key['reference_columns'].') '.$foreign_key['fk_definition'];
1339              if( $silent )
1340              { // execute queries in silent mode
1341                  if( $DB->query( $delete_query ) !== false )
1342                  { // orphan child entries have been deleted, create foreign key
1343                      $DB->query( $query );
1344                  }
1345              }
1346              else
1347              { // set query definition
1348                  $result[] = array(
1349                      'queries' => array( $delete_query ),
1350                      'note' => 'Delete orphan <strong>'.$table.'</strong> entries.',
1351                      'type' => 'delete_orphan_entries' );
1352                  $result[] = array(
1353                      'queries' => array( $query ),
1354                      'note' => 'Add foreign key constraint on <strong>'.$table.'('.$foreign_key['fk_fields'].')'.'</strong> in reference to <strong>'.$foreign_key['reference_table'].'('.$foreign_key['reference_columns'].')'.'</strong>',
1355                      'type' => 'add_foreign_key' );
1356              }
1357          }
1358      }
1359  
1360      return $result;
1361  }

1531  /**
1532   * Remove any comments from the SQL query
1533   *
1534   * @param string query
1535   * @return string the same query without comments
1536   */
1537  function remove_comments_from_query( $query )
1538  {
1539      $n = strlen( $query );
1540      $in_string = false;
1541      for( $i = 0; $i < $n; $i++ )
1542      {
1543          if( $query[$i] == '\\' )
1544          { // backslash/escape; skip
1545              continue;
1546          }
1547          if( $query[$i] == '"' || $query[$i] == "'" )
1548          {
1549              if( ! $in_string )
1550              { // string begins:
1551                  $in_string = $query[$i];
1552              }
1553              elseif( $query[$i] === $in_string )
1554              {
1555                  $in_string = false;
1556              }
1557          }
1558          elseif( $in_string === false )
1559          { // not in string, check for comment start:
1560              if( $query[$i] == '#' || substr($query, $i, 3) == '-- ' )
1561              { // comment start
1562                  // search for newline
1563                  for( $j = $i+1; $j < $n; $j++ )
1564                  {
1565                      if( $query[$j] == "\n" || $query[$j] == "\r" )
1566                      {
1567                          break;
1568                      }
1569                  }
1570                  // remove comment
1571                  $query = substr($query, 0, $i).substr($query, $j);
1572                  $n = strlen($query);
1573                  continue;
1574              }
1575          }
1576      }
1577      // return query without comments
1578      return $query;
1579  }

1582  /**
1583   * Get all lines from a create table query
1584   *
1585   * @param string the query
1586   * @return array field lines
1587   */
1588  function get_fieldlines_from_query( $query )
1589  {
1590      // Get all of the field names in the query from between the parens
1591      preg_match( '|\((.*)\).*$|s', $query, $match ); // we have only one query here
1592      $qrylines = trim($match[1]);
1593  
1594      $flds = array();
1595      $in_parens = 0;
1596      $in_quote = false;
1597      $buffer = '';
1598      for( $i = 0; $i < strlen($qrylines); $i++ )
1599      {
1600          $c = $qrylines[$i];
1601  
1602          if( ( $c == ',' ) && ( ! $in_parens ) && ( ! $in_quote ) )
1603          { // split here:
1604              $line = trim($buffer);
1605              preg_match( '|^([^\s(]+)|', $line, $linematch );
1606              $fieldname = db_delta_remove_quotes($linematch[1]);
1607              if( isset( $flds[$fieldname] ) )
1608              {
1609                  $fieldname .= '_'.$i;
1610              }
1611              $flds[$fieldname] = $line;
1612              $buffer = '';
1613              continue;
1614          }
1615  
1616          if( $c == '(' )
1617          {
1618              $in_parens++;
1619          }
1620          elseif( $c == ')' )
1621          {
1622              $in_parens--;
1623          }
1624  
1625          if( ( ! $in_parens ) && ( $c == "'" ) && ( $qrylines[$i - 1] != '\\' ) )
1626          { // Text between quotation marks outside from parentheses, must be in a field COMMENT
1627              // Commas in field comments are not separating two fields, so don't split there.
1628              $in_quote = ! $in_quote;
1629          }
1630  
1631          $buffer .= $c;
1632      }
1633      if( strlen($buffer) )
1634      {
1635          $flds[] = trim($buffer);
1636      }
1637  
1638      return $flds;
1639  }

1701          return;
1702      }
1703      $table_alias = $DB->dbaliases[$table_index];
1704      $table_alias = substr( $table_alias, 3, strlen( $table_alias ) - 6 );
1705

1712      else
1713      { // There are columns with charset or collation definition
1714          // Skip columns with explicit definition but update all of the other *text, *char, set and enum columns since they all have charset definition and we will not call convert charset for the whole table
1715          $text_columns = $DB->get_results( 'SHOW FULL COLUMNS FROM `'.$table.'` WHERE type LIKE "%TEXT%" OR type LIKE "%CHAR%" OR type LIKE "SET%" OR type LIKE "ENUM%"' );

1720                  unset( $text_columns[$column_key] );
1721              }
1722          }
1723      }
1724

1726      // This way we avoid the automatic modification of e.g. mediumtext colums to longtext
1727      $col_definitions = array();
1728      foreach( $text_columns as $text_column )
1729      { // Update each text column character set explicitly to utf8. Those columns which should not be updated were already removed from this array.

1731              ' '.$text_column->Type.' COLLATE utf8_general_ci'.

1745          $DB->query( 'ALTER TABLE `'.$table.'`
1746              DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci,

1756              MODIFY '.implode( ',
