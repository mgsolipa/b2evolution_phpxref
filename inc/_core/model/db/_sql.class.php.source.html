 369      /**
 370       * create the filter whith the search field array
 371       *
 372       * @param string keywords separated by space
 373       * @param string operator( AND , OR , PHRASE ) for the filter
 374       */
 375  	function WHERE_keywords( $search, $search_kw_combine )
 376      {
 377          global $DB;
 378  
 379          // Concat the list of search fields ( concat(' ',field1,field2,field3...) )
 380          if (count( $this->search_field ) > 1)
 381          {
 382              $search_field = 'CONCAT_WS(\' \','.implode( ',', $this->search_field).')';
 383          }
 384          else
 385          {
 386              $search_field = $this->search_field[0];
 387          }
 388  
 389          switch( $search_kw_combine )
 390          {
 391              case 'AND':
 392              case 'OR':
 393                  // Create array of key words of the search string
 394                  $keyword_array = explode( ' ', $search );
 395                  $keyword_array = array_filter( $keyword_array, create_function( '$val', 'return !empty($val);' ) );
 396  
 397                  $twhere = array();
 398                  // Loop on all keywords
 399                  foreach($keyword_array as $keyword)
 400                  {
 401                      $twhere[] = '( '.$search_field.' LIKE \'%'.$DB->escape( $keyword ).'%\''.$this->WHERE_regexp( $keyword, $search_kw_combine ).' )';
 402                  }
 403                  $where = implode( ' '.$search_kw_combine.' ', $twhere);
 404                  break;
 405  
 406              case 'PHRASE':
 407                      $where = $search_field." LIKE '%".$DB->escape( $search )."%'".$this->WHERE_regexp( $search, $search_kw_combine );
 408                      break;
 409  
 410              case 'BEGINWITH':
 411                  $twhere = array();
 412                  foreach( $this->search_field as $field )
 413                  {
 414                      $twhere[] = $field." LIKE '".$DB->escape( $search )."%'";
 415                  }
 416                  $where = implode( ' OR ', $twhere ).$this->WHERE_regexp( $search, $search_kw_combine);
 417                  break;
 418  
 419          }
 420          $this->WHERE_and( $where );
 421      }

 423      /**
 424       * create the filter whith the search field regexp array
 425       *
 426       * @param string search
 427       * @param string operator( AND , OR , PHRASE ) for the filter
 428       *
 429       */
 430  	function WHERE_regexp( $search, $search_kw_combine )
 431      {
 432          $where = '';
 433  
 434          // Loop on all fields we have to use a replace regular expression on search:
 435          foreach( $this->search_field_regexp as $field=>$reg_exp )
 436          {
 437                  // Use reg exp replace on search
 438                  $search_reg_exp = preg_replace( $reg_exp, '', $search );
 439  
 440                  if( !empty( $search_reg_exp ) )
 441                  {    // The reg exp search is not empty, so we add it to the request with an 'OR' operator:
 442                      switch( $search_kw_combine )
 443                      {
 444                          case 'AND':
 445                          case 'OR':
 446                          case 'PHRASE':
 447                              $where .= ' OR '.$field.' LIKE \'%'.$DB->escape( $search_reg_exp ).'%\'';
 448                              break;
 449  
 450                          case 'BEGINWITH':
 451                              $where .= ' OR '.$field.' LIKE \''.$DB->escape( $search_reg_exp ).'%\'';
 452                              break;
 453                      }
 454                  }
 455          }
 456          return $where;
 457      }
