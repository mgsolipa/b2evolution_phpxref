 285      /**
 286       * Instantiate an object for requested row by field and cache it:
 287       *
 288       * @param string DB field name
 289       * @param string Value
 290       * @return object
 291       */
 292      function & get_by_field( $field_name, $field_value )
 293      {
 294          $obj_ID = 0;
 295          $null_Obj = NULL;
 296  
 297          foreach( $this->rows as $row )
 298          {    // Find object ID by field value
 299              if( $row->$field_name == $field_value )
 300              {
 301                  $obj_ID = $row->{$this->ID_col};
 302                  break;
 303              }
 304          }
 305  
 306          if( $obj_ID == 0 )
 307          {    // No object ID found, Exit here
 308              return $null_Obj;
 309          }
 310  
 311          $this->restart();
 312          while( $Obj = & $this->get_next() )
 313          {    // Find Object by ID
 314              if( $Obj->ID == $obj_ID )
 315              {
 316                  return $Obj;
 317              }
 318          }
 319  
 320          return $null_Obj;
 321      }

 374       * Move up the element order in database
 375       *
 376       * @param integer id element

 379  	function move_up( $id )
 380      {
 381          global $DB, $Messages, $result_fadeout;
 382  
 383          $DB->begin();
 384  
 385          if( ($obj = & $this->Cache->get_by_ID( $id )) === false )
 386          {
 387              $Messages->add( sprintf( T_('Requested &laquo;%s&raquo; object does not exist any longer.'), T_('Entry') ), 'error' );
 388              $DB->commit();
 389              return false;
 390          }
 391          $order = $obj->order;
 392  
 393          // Get the ID of the inferior element which his order is the nearest
 394          $rows = $DB->get_results( 'SELECT '.$this->Cache->dbIDname
 395                   .' FROM '.$this->Cache->dbtablename
 396                  .' WHERE '.$this->Cache->dbprefix.'order < '.$order
 397                  .' ORDER BY '.$this->Cache->dbprefix.'order DESC'
 398                  .' LIMIT 0,1' );
 399  
 400          if( count( $rows ) )
 401          {
 402              // instantiate the inferior element
 403              $obj_inf = & $this->Cache->get_by_ID( $rows[0]->{$this->Cache->dbIDname} );
 404  
 405              //  Update element order
 406              $obj->set( 'order', $obj_inf->order );
 407              $obj->dbupdate();
 408  
 409              // Update inferior element order
 410              $obj_inf->set( 'order', $order );
 411              $obj_inf->dbupdate();
 412  
 413              // EXPERIMENTAL FOR FADEOUT RESULT
 414              $result_fadeout[$this->Cache->dbIDname][] = $id;
 415              $result_fadeout[$this->Cache->dbIDname][] = $obj_inf->ID;
 416          }
 417          else
 418          {
 419              $Messages->add( T_('This element is already at the top.'), 'error' );
 420          }
 421          $DB->commit();
 422      }
