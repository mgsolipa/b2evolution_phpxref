 340       * Add a dataobject to the cache

 342  	function add( & $Obj )
 343      {
 344          global $Debuglog;
 345  
 346          if( ! is_object( $Obj ) || ! isset( $Obj->ID ) || is_null( $Obj->ID ) ) // value 0 is used by item preview
 347          {
 348              $Debuglog->add( 'No object to add!', 'dataobjects' );
 349              return false;
 350          }
 351  
 352          // fplanque: I don't want an extra (and expensive) comparison here. $this->cache[$Obj->ID] === $Obj.
 353          // If you need this you're probably misusing the cache.
 354          if( isset($this->cache[$Obj->ID]) )
 355          {
 356              $Debuglog->add( $this->objtype.': Object with ID '.$Obj->ID.' is already cached', 'dataobjects' );
 357              return false;
 358          }
 359  
 360          // If the object is valid and not already cached:
 361          // Add object to cache:
 362          $this->cache[$Obj->ID] = & $Obj;
 363          // Add a reference in the object list:
 364          $this->DataObject_array[] = & $Obj;
 365          // Add the ID to the list of IDs
 366          $this->ID_array[] = $Obj->ID;
 367  
 368          return true;
 369      }
