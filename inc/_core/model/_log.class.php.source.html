 109      /**
 110       * Add a message to the Log.
 111       *
 112       * @param string the message
 113       * @param string|array the category, default is to use the object's default category.
 114       *        Can also be an array of categories to add the same message to.
 115       * @param boolean Dump (output) this directly?
 116       */
 117  	function add( $message, $category = NULL )
 118      {
 119          if( $category === NULL )
 120          { // By default, we use the default category:
 121              $category = $this->defaultcategory;
 122          }
 123  
 124          if( is_string($category) && isset($GLOBALS['debug_'.$category]) && $GLOBALS['debug_'.$category] == false )
 125          {    // We don't want to debug this category
 126              return;
 127          }
 128  
 129          if( is_array($category) )
 130          {
 131              foreach( $category as $l_cat )
 132              {
 133                  $this->add( $message, $l_cat, false );
 134              }
 135          }
 136          else
 137          {
 138              $this->messages[$category][] = $message;
 139  
 140              if( empty($this->_count[$category]) )
 141              {
 142                  $this->_count[$category] = 0;
 143              }
 144              $this->_count[$category]++;
 145          }
 146      }

 166      /**
 167       * Get head/foot for a specific category, designed for internal use of {@link display()}
 168       *
 169       * @access private
 170       *
 171       * @param mixed head or foot (array [ category => head/foot, category => 'string', 'template',
 172       *              or string [for container only])
 173       * @param string the category (or container)
 174       * @param string template, where the head/foot gets used (%s)
 175       */
 176  	static function get_head_foot( $headfoot, $category, $template = NULL )
 177      {
 178          if( is_string($headfoot) && $category == 'container' )
 179          { // container head or foot
 180              $r = $headfoot;
 181          }
 182          elseif( is_array($headfoot) )
 183          { // head or foot for categories
 184              if( isset($headfoot[$category]) )
 185              {
 186                  $r = $headfoot[$category];
 187              }
 188              elseif( isset($headfoot['all']) && $category != 'container' )
 189              { // use 'all' info, except if for container
 190                  $r = $headfoot['all'];
 191              }
 192              else
 193              {
 194                  return false;
 195              }
 196  
 197              if( is_array($r) )
 198              {
 199                  if( isset($r['template']) )
 200                  {
 201                      $template = $r['template'];
 202                  }
 203                  $r = $r['string'];
 204              }
 205  
 206              // Replace '%s' with category:
 207              $r = str_replace( '%s', $category, $r );
 208          }
 209  
 210          if( empty($r) )
 211          {
 212              return false;
 213          }
 214  
 215          if( !empty($template) )
 216          {
 217              $r = sprintf( $template, $r );
 218          }
 219  
 220          return $r;
 221      }

 276      /**
 277       * Display messages of the Log object.
 278       *
 279       * - You can either output/get the messages of a category (string),
 280       *   all categories ('all') or category groups (array of strings) (defaults to 'all').
 281       * - Head/Foot will be displayed on top/bottom of the messages. You can pass
 282       *   an array as head/foot with the category as key and this will be displayed
 283       *   on top of the category's messages.
 284       * - You can choose from various styles for message groups ('ul', 'p', 'br')
 285       *   and set a css class for it (by default 'log_'.$category gets used).
 286       * - You can suppress the outer div or set a css class for it (defaults to
 287       *   'log_container').
 288       *
 289       * @todo Make this simple!
 290       * start by getting rid of the $category selection and the special cases for 'all'. If you don't want to display ALL messages,
 291       * then you should not log them in the same Log object and you should instantiate separate logs instead.
 292       *
 293       * @param string|NULL Header/title, might be array ( category => msg ),
 294       *                    'container' is then top. NULL for object's default ({@link Log::$head}.
 295       * @param string|NULL Footer, might be array ( category => msg ), 'container' is then bottom.
 296       *                    NULL for object's default ({@link Log::$foot}.
 297       * @param boolean to display or return (default: display)
 298       * @param mixed the category of messages to use (category, 'all', list of categories (array)
 299       *              or NULL for {@link $defaultcategory}).
 300       * @param string the CSS class of the messages div tag (default: 'log_'.$category)
 301       * @param string the style to use, 'ul', 'p', 'br', 'raw'
 302       *               (default: 'br' for single message, 'ul' for more)
 303       * @param mixed the outer div, may be false
 304       * @return boolean false, if no messages; else true (and outputs if $display)
 305       */
 306  	function display( $head = NULL, $foot = NULL, $display = true, $category = 'all', $cssclass = NULL, $style = NULL, $outerdivclass = 'log_container' )
 307      {
 308          if( is_null( $head ) )
 309          { // Use object default:
 310              $head = isset( $this->head ) ? $this->head : '';
 311          }
 312          if( is_null( $foot ) )
 313          { // Use object default:
 314              $foot = isset( $this->foot ) ? $this->foot : '';
 315          }
 316          if( is_null( $category ) )
 317          {
 318              $category = isset( $this, $this->defaultcategory ) ? $this->defaultcategory : 'error';
 319          }
 320  
 321          if( !$this->count( $category ) )
 322          { // no messages
 323              return false;
 324          }
 325          else
 326          {
 327              $messages = $this->get_messages( $category );
 328          }
 329  
 330          if( !is_array($cssclass) )
 331          {
 332              $cssclass = array( 'all' => array( 'class' => is_null($cssclass) ? NULL : $cssclass, 'divClass' => true ) );
 333          }
 334          elseif( !isset($cssclass['all']) )
 335          {
 336              $cssclass['all'] = array( 'class' => NULL, 'divClass' => true );
 337          }
 338  
 339  
 340          $disp = '';
 341  
 342          if( $outerdivclass )
 343          {
 344              $disp .= "\n<div class=\"$outerdivclass\">";
 345          }
 346  
 347          $disp .= Log::get_head_foot( $head, 'container', '<h2>%s</h2>' );
 348  
 349  
 350          foreach( $messages as $lcategory => $lmessages )
 351          {
 352              $lcssclass = isset($cssclass[$lcategory]) ? $cssclass[$lcategory] : $cssclass['all'];
 353              if( !isset($lcssclass['class']) || is_null($lcssclass['class']) )
 354              {
 355                  $lcssclass['class'] = 'log_'.$lcategory;
 356              }
 357              if( !isset($lcssclass['divClass']) || is_null($lcssclass['divClass']) || $lcssclass['divClass'] === true )
 358              {
 359                  $lcssclass['divClass'] = $lcssclass['class'];
 360              }
 361  
 362  
 363              $disp .= "\n";
 364              if( $lcssclass['divClass'] )
 365              {
 366                  $disp .= "\t<div class=\"{$lcssclass['divClass']}\">";
 367              }
 368  
 369              $disp .= Log::get_head_foot( $head, $lcategory, '<h3>%s</h3>' );
 370  
 371              if( $style == NULL )
 372              { // 'br' for a single message, 'ul' for more
 373                  $style = count($lmessages) == 1 ? 'br' : 'ul';
 374              }
 375  
 376              // implode messages
 377              if( $style == 'ul' )
 378              {
 379                  $disp .= "\t<ul".( $lcssclass['class'] ? " class=\"{$lcssclass['class']}\"" : '' ).'>'
 380                      .'<li>'.implode( "</li>\n<li>", $lmessages )."</li></ul>\n";
 381              }
 382              elseif( $style == 'p' )
 383              {
 384                  $disp .= "\t<p".( $lcssclass['class'] ? " class=\"{$lcssclass['class']}\"" : '' ).'>'
 385                              .implode( "</p>\n<p class=\"{$lcssclass['class']}\">", $lmessages )."</p>\n";
 386              }
 387              elseif( $style == 'raw' )
 388              {
 389                  $disp .= implode( "\n", $lmessages )."\n";
 390              }
 391              else
 392              {
 393                  $disp .= "\t".implode( "\n<br />\t", $lmessages );
 394              }
 395              $disp .= Log::get_head_foot( $foot, $lcategory, "\n<p>%s</p>" );
 396              if( $lcssclass['divClass'] )
 397              {
 398                  $disp .= "\t</div>\n";
 399              }
 400          }
 401  
 402          $disp .= Log::get_head_foot( $foot, 'container', "\n<p>%s</p>" );
 403  
 404          if( $outerdivclass )
 405          {
 406              $disp .= "</div>\n";
 407          }
 408  
 409          if( $display )
 410          {
 411              echo $disp;
 412              return true;
 413          }
 414  
 415          return $disp;
 416      }

 419      /**
 420       * Concatenates messages of a given category to a string
 421       *
 422       * @param string prefix of the string
 423       * @param string suffic of the string
 424       * @param string the category
 425       * @return string the messages, imploded. Tags stripped.
 426       */
 427  	function get_string( $head = '', $foot = '', $category = NULL, $implodeBy = ', ', $format = 'striptags' )
 428      {
 429          if( !$this->count( $category ) )
 430          {
 431              return false;
 432          }
 433  
 434          $r = '';
 435          if( '' != $head )
 436          {
 437              $r .= $head.' ';
 438          }
 439          $r .= implode( $implodeBy, $this->get_messages( $category, true ) );
 440          if( '' != $foot )
 441          {
 442              $r .= ' '.$foot;
 443          }
 444  
 445          switch( $format )
 446          {
 447              case 'xmlrpc':
 448                  $r = strip_tags( $r );    // get rid of <code>
 449                  $r = str_replace( '&lt;', '<', $r );
 450                  $r = str_replace( '&gt;', '>', $r );
 451                  $r = str_replace( '&quot;', '"', $r );
 452                  break;
 453  
 454              case 'striptags':
 455                  $r = strip_tags( $r );
 456                  break;
 457          }
 458  
 459          return $r;
 460      }

 463      /**
 464       * Counts messages of a given category
 465       *
 466       * @todo this seems a bit weird (not really relying on the cache ($_count) and unsetting 'all') -> write testcases to safely be able to change it.
 467       * @param string|array the category, NULL=default, 'all' = all
 468       * @return number of messages

 470  	function count( $category = NULL )
 471      {
 472          if( is_null($category) )
 473          {    // use default category:
 474              $category = $this->defaultcategory;
 475          }
 476  
 477          if( is_string($category) )
 478          {
 479              if( empty( $this->_count[$category] ) )
 480              {
 481                  $this->_count[$category] = count( $this->get_messages( $category, true ) );
 482              }
 483              if( $category != 'all' )
 484              {
 485                  unset($this->_count['all']);
 486              }
 487              return $this->_count[$category];
 488          }
 489  
 490          return count( $this->get_messages( $category, true ) );
 491      }

 494      /**
 495       * Returns array of messages of a single category or group of categories.
 496       *
 497       * If the category is an array, those categories will be used (where 'all' will
 498       * be translated with the not already processed categories).
 499       * <code>get_messages( array('error', 'note', 'all') )</code> would return
 500       * 'errors', 'notes' and the remaining messages, in that order.
 501       *
 502       * @param string|array the category, NULL=default, 'all' = all
 503       * @param boolean if true will use subarrays for each category
 504       * @return array the messages, one or two dimensions (depends on second param)
 505       */
 506  	function get_messages( $category = NULL, $singleDimension = false )
 507      {
 508          $messages = array();
 509  
 510          if( is_null($category) )
 511          {
 512              $category = $this->defaultcategory;
 513          }
 514  
 515          if( $category == 'all' )
 516          {
 517              $category = array_keys( $this->messages );
 518              sort($category);
 519          }
 520          elseif( !is_array($category) )
 521          {
 522              $category = array( $category );
 523          }
 524  
 525          $categoriesDone = array();
 526  
 527          while( $lcategory = array_shift( $category ) )
 528          {
 529              if( $lcategory == 'all' )
 530              { // Put those categories in queue, which have not been processed already
 531                  $category = array_merge( array_diff( array_keys( $this->messages ), $categoriesDone ), $category );
 532                  sort($category);
 533                  continue;
 534              }
 535              if( in_array( $lcategory, $categoriesDone ) )
 536              {
 537                  continue;
 538              }
 539              $categoriesDone[] = $lcategory;
 540  
 541  
 542              if( !isset($this->messages[$lcategory][0]) )
 543              { // no messages
 544                  continue;
 545              }
 546  
 547              if( $singleDimension )
 548              {
 549                  $messages = array_merge( $messages, $this->messages[$lcategory] );
 550              }
 551              else
 552              {
 553                  $messages[$lcategory] = $this->messages[$lcategory];
 554              }
 555          }
 556          return $messages;
 557      }
