 406  //Process result after publish/deprecate/delete action has been completed
 407  function processResult( result, modifiedlist )
 408  {
 409      var resultObject = jQuery('<div/>');
 410      resultObject.html(result);
 411  
 412      var removed_count = 0;
 413      // Remove those comments which were moderated during refresh
 414      for(var id in removed_ids)
 415      {
 416          var removed_comment = resultObject.find('#' + removed_ids[id]);
 417          if( removed_comment.length !== 0 )
 418          { // The removed element is in the refreshed content
 419              removed_comment.remove();
 420              var new_visible_comment = resultObject.find(".hidden_comment:first");
 421              new_visible_comment.removeClass( 'hidden_comment' );
 422              removed_count = removed_count + 1;
 423          }
 424      }
 425      // Clear removed_ids content, because it was already processed
 426      removed_ids.length = 0;
 427  
 428      // Update comments container content with up to date content
 429      jQuery('#comments_container').html(resultObject.html());
 430  
 431      var comments = jQuery('#comments_container [id^=comment_]');
 432      if( comments )
 433      { // Set displayed comments number ( even hidden comments are counted )
 434          displayed = comments.length;
 435      }
 436      else
 437      {
 438          displayed = 0;
 439      }
 440      for(var id in modifiedlist)
 441      {
 442          switch(modifiedlist[id])
 443          {
 444              case 'published':
 445                  jQuery('#' + id).css( 'backgroundColor', '#339900' );
 446                  break;
 447              case 'deprecated':
 448                  jQuery('#' + id).css( 'backgroundColor', '#656565' );
 449                  break;
 450              case 'deleted':
 451                  jQuery('#' + id).css( 'backgroundColor', '#fcc' );
 452                  break;
 453          }
 454      }
 455  
 456      var comments_number = jQuery('#new_badge').val();
 457      if(comments_number == '0')
 458      { // no comments, so remove the comment block
 459          var options = {};
 460          jQuery('#comments_block').effect('blind', options, 200);
 461          jQuery('#comments_block').remove();
 462      }
 463      else
 464      { // update comments awaiting moderation number
 465          var new_value = parseInt(comments_number) - removed_count;
 466          jQuery('#badge').text(new_value);
 467      }
 468  }
