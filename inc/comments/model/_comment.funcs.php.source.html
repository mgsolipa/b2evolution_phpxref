  22   * Generic comments/trackbacks/pingbacks counting
  23   *
  24   * @todo check this in a multiblog page...
  25   * @todo This should support visibility: at least in the default front office (_feedback.php), there should only the number of visible comments/trackbacks get used ({@link Item::feedback_link()}).
  26   *
  27   * @param integer
  28   * @param string what to count
  29   * @param mixed string or array to count comments with this/these status(es)
  30   * @param boolean set true to count expired comments, leave on false otherwise
  31   */
  32  function generic_ctp_number( $post_id, $mode = 'comments', $status = 'published', $count_expired = false, $filter_by_perm = true )
  33  {
  34      global $DB, $debug, $postdata, $cache_ctp_number, $preview, $servertimenow, $blog;
  35  
  36      if( $preview )
  37      { // we are in preview mode, no comments yet!
  38          return 0;
  39      }
  40  
  41      $show_statuses = ( is_admin_page() || ( ! $filter_by_perm ) ) ? get_visibility_statuses( 'keys', array( 'trash', 'redirected' ) ) : get_inskin_statuses( $blog, 'comment' );
  42      $filter_index = $filter_by_perm ? 0 : 1;
  43      if( !isset($cache_ctp_number) || !isset($cache_ctp_number[$filter_index][$post_id]) )
  44      { // we need a query to count comments
  45          $count_SQL = new SQL();
  46          $count_SQL->SELECT( 'comment_item_ID, comment_type, comment_status, COUNT(*) AS type_count' );
  47          $count_SQL->FROM( 'T_comments' );
  48          $count_SQL->GROUP_BY( 'comment_item_ID, comment_type, comment_status' );
  49          if( !empty( $blog ) )
  50          {
  51              $count_SQL->WHERE( statuses_where_clause( $show_statuses, 'comment_', $blog, 'blog_comment!', $filter_by_perm ) );
  52          }
  53  
  54          if( !$count_expired )
  55          {
  56              $count_SQL->FROM_add( 'LEFT JOIN T_items__item_settings as expiry_setting ON comment_item_ID = iset_item_ID AND iset_name = "comment_expiry_delay"' );
  57              $count_SQL->WHERE_and( 'expiry_setting.iset_value IS NULL OR expiry_setting.iset_value = "" OR TIMESTAMPDIFF(SECOND, comment_date, '.$DB->quote( date2mysql( $servertimenow ) ).') < expiry_setting.iset_value' );
  58          }
  59      }
  60  
  61      // init statuses count array
  62      $statuses_array = array( 'published' => 0, 'community' => 0, 'protected' => 0, 'private' => 0, 'review' => 0, 'draft' => 0, 'deprecated' => 0, 'trash' => 0, 'total' => 0 );
  63  
  64      /*
  65       * Make sure cache is loaded for current display list:
  66       */
  67      if( !isset($cache_ctp_number) || !isset($cache_ctp_number[$filter_index]) )
  68      {
  69          global $postIDlist, $postIDarray;
  70  
  71          // if( $debug ) echo "LOADING generic_ctp_number CACHE for posts: $postIDlist<br />";
  72  
  73          if( ! empty( $postIDlist ) )    // This can happen when displaying a featured post of something that's not in the MainList
  74          {
  75              foreach( $postIDarray as $tmp_post_id)
  76              {    // Initializes each post to nocount!
  77                  $cache_ctp_number[$filter_index][$tmp_post_id] = array(
  78                          'comments'   => $statuses_array,
  79                          'trackbacks' => $statuses_array,
  80                          'pingbacks'  => $statuses_array,
  81                          'feedbacks'  => $statuses_array,
  82                          'metas'      => $statuses_array
  83                      );
  84              }
  85  
  86              $countall_SQL = $count_SQL;
  87              $countall_SQL->WHERE_and( 'comment_item_ID IN ('.$postIDlist.')' );
  88  
  89              foreach( $DB->get_results( $countall_SQL->get() ) as $row )
  90              {
  91                  // detail by status, tyep and post:
  92                  $cache_ctp_number[$filter_index][$row->comment_item_ID][$row->comment_type.'s'][$row->comment_status] = $row->type_count;
  93  
  94                  // Total for type on post:
  95                  $cache_ctp_number[$filter_index][$row->comment_item_ID][$row->comment_type.'s']['total'] += $row->type_count;
  96  
  97                  if( $row->comment_type != 'meta' )
  98                  { // Exclude meta comments from feedbacks
  99                      // Total for status on post:
 100                      $cache_ctp_number[$filter_index][$row->comment_item_ID]['feedbacks'][$row->comment_status] += $row->type_count;
 101  
 102                      // Total for post:
 103                      $cache_ctp_number[$filter_index][$row->comment_item_ID]['feedbacks']['total'] += $row->type_count;
 104                  }
 105              }
 106          }
 107      }
 108      /*    else
 109      {
 110          echo "cache set";
 111      }*/
 112  
 113  
 114      if( !isset($cache_ctp_number[$filter_index][$post_id]) )
 115      { // this should be extremely rare...
 116          // echo "CACHE not set for $post_id";
 117  
 118          // Initializes post to nocount!
 119          $cache_ctp_number[$filter_index][intval($post_id)] = array(
 120                  'comments'   => $statuses_array,
 121                  'trackbacks' => $statuses_array,
 122                  'pingbacks'  => $statuses_array,
 123                  'feedbacks'  => $statuses_array,
 124                  'metas'      => $statuses_array
 125              );
 126  
 127          $count_SQL->WHERE_and( 'comment_item_ID = '.intval($post_id) );
 128  
 129          foreach( $DB->get_results( $count_SQL->get() ) as $row )
 130          {
 131              // detail by status, type and post:
 132              $cache_ctp_number[$filter_index][$row->comment_item_ID][$row->comment_type.'s'][$row->comment_status] = $row->type_count;
 133  
 134              // Total for type on post:
 135              $cache_ctp_number[$filter_index][$row->comment_item_ID][$row->comment_type.'s']['total'] += $row->type_count;
 136  
 137              if( $row->comment_type != 'meta' )
 138              { // Exclude meta comments from feedbacks
 139                  // Total for status on post:
 140                  $cache_ctp_number[$filter_index][$row->comment_item_ID]['feedbacks'][$row->comment_status] += $row->type_count;
 141  
 142                  // Total for post:
 143                  $cache_ctp_number[$filter_index][$row->comment_item_ID]['feedbacks']['total'] += $row->type_count;
 144              }
 145          }
 146      }
 147  
 148      if( ! in_array( $mode, array( 'comments', 'trackbacks', 'pingbacks', 'metas' ) ) )
 149      {
 150          $mode = 'feedbacks';
 151      }
 152  
 153      if( is_array( $status ) )
 154      { // $status is an array and probably contains more then one visibility status
 155          $result = 0;
 156          foreach( $status as $one_status )
 157          {
 158              if( isset( $cache_ctp_number[$filter_index][$post_id][$mode][$one_status] ) )
 159              {
 160                  $result = $result + $cache_ctp_number[$filter_index][$post_id][$mode][$one_status];
 161              }
 162          }
 163      }
 164      elseif( isset( $cache_ctp_number[$filter_index][$post_id][$mode][$status] ) )
 165      { // $status is a string with one visibility status
 166          $result = $cache_ctp_number[$filter_index][$post_id][$mode][$status];
 167      }
 168      else
 169      { // $status is not recognized return total feedback number
 170          $result = $cache_ctp_number[$filter_index][$post_id][$mode]['total'];
 171      }
 172  
 173      // pre_dump( $cache_ctp_number[$filter_index][$post_id] );
 174  
 175      return $result;
 176  }

 207  function comments_number( $zero='#', $one='#', $more='#', $post_ID = NULL )
 208  {
 209      if( $zero == '#' ) $zero = T_('Leave a comment');
 210      if( $one == '#' ) $one = T_('1 comment');
 211      if( $more == '#' ) $more = T_('%d comments');
 212  
 213      if( empty( $post_ID ) )
 214      {
 215          global $id;
 216          $post_ID = $id;
 217      }
 218  
 219      // attila> This function is called only from the backoffice ( _item_list_full.view.php ).
 220      // There we always have to show all comments.
 221      $number = generic_ctp_number( $post_ID, 'comments', 'total' );
 222      if ($number == 0)
 223      {
 224          $blah = $zero;
 225      }
 226      elseif ($number == 1)
 227      {
 228          $blah = $one;
 229      }
 230      elseif ($number  > 1)
 231      {
 232          $n = $number;
 233          $more = str_replace('%d', $n, $more);
 234          $blah = $more;
 235      }
 236      echo $blah;
 237  }

 404  function add_ban_icons( $content )
 405  {
 406      global $current_User;

 426  /**
 427   * Callback function to add a javascript ban action icon after each url in the given content
 428   *
 429   * @param string Comment content
 430   * @return string the content with a ban icon after each url if the user has spamblacklist permission, the incoming content otherwise
 431   */
 432  function add_ban_icons_callback( $content )
 433  {
 434      $atags = get_atags( $content );
 435      $imgtags = get_imgtags( $content );
 436      $urls = get_urls( $content );
 437      $result = '';
 438      $from = 0; // current processing position
 439      $length = 0; // current url or tag length
 440      $i = 0; // url counter
 441      $j = 0; // "a" tag counter
 442      $k = 0; // "img" tag counter
 443  
 444      // Remove the duplicated <a> tags from array
 445      $atags_fixed = array();
 446      foreach( $atags as $atag )
 447      {
 448          if( preg_match( '#href="([^"]+)"#', $atag, $matches ) && ! isset( $atags_fixed[ $matches[1] ] ) )
 449          {
 450              $atags_fixed[ $matches[1] ] = $atag;
 451          }
 452      }
 453      $atags = array();
 454      foreach( $atags_fixed as $atag )
 455      {
 456          $atags[] = $atag;
 457      }
 458  
 459      $used_urls = array();
 460      while( isset( $urls[$i] ) )
 461      { // there is unprocessed url
 462          $url = $urls[$i];
 463          if( validate_url( $url, 'posting', false ) )
 464          { // skip not valid urls
 465              $i++;
 466              continue;
 467          }
 468          if( in_array( $url, $used_urls ) )
 469          { // skip already passed url
 470              $i++;
 471              continue;
 472          }
 473          $used_urls[] = $url;
 474          while( isset( $imgtags[$k] ) && ( strpos( $content, $imgtags[$k] ) < $from ) )
 475          { // skip already passed img tags
 476              $k++;
 477          }
 478  
 479          $pos = strpos( $content, $url, $from );
 480          $length = utf8_strlen($url);
 481          $i++;
 482  
 483          // check img tags
 484          if( isset( $imgtags[$k] ) && ( strpos( $imgtags[$k], $url ) !== false )
 485              && ( $pos > strpos( $content, $imgtags[$k], $from ) ) )
 486          { // current url is inside the img tag, we need to skip this url.
 487              $result .= substr( $content, $from, $pos + $length - $from );
 488              $from = $pos + $length;
 489              $k++;
 490              continue;
 491          }
 492  
 493          // check a tags
 494          if( isset($atags[$j]) )
 495          { // there is unprocessed "a" tag
 496              $tag = $atags[$j];
 497              if( ( ( $urlpos = strpos( $tag, $url ) ) !== false ) && ( $pos > strpos( $content, $tag, $from ) ) )
 498              { // the url is inside the current tag, we have to add ban icon after the tag
 499                  $pos = strpos( $content, $tag, $from );
 500                  $length = strlen($tag);
 501                  while( isset($urls[$i]) && ( ( $urlpos = strpos( $tag, $urls[$i], $urlpos + 1 ) ) !== false ) )
 502                  { // skip all other urls from this tag
 503                      $i++;
 504                  }
 505                  $j++;
 506              }
 507          }
 508          // add processed part and ban icon to result and set current position
 509          $result .= substr( $content, $from, $pos + $length - $from );
 510          $from = $pos + $length;
 511          $result .= add_jsban( $url );
 512      }
 513  
 514      // add the end of the content to the result
 515      $result .= substr( $content, $from, strlen($content) - $from );
 516      return $result;
 517  }

 520  /**
 521   * Get opentrash link
 522   *
 523   * @param boolean check permission or not. Should be false only if it was already checked.
 524   * @param boolean show "Open recycle bin" link even if there is no comment with 'trash' status
 525   * @param array Additional params
 526   * @return Open recycle bin link if user has the corresponding 'blogs' - 'editall' permission, empty string otherwise
 527   */
 528  function get_opentrash_link( $check_perm = true, $force_show = false, $params = array() )
 529  {
 530      global $admin_url, $current_User, $DB, $blog;
 531  
 532      $params = array_merge( array(
 533              'before' => '<div id="recycle_bin" class="floatright">',
 534              'after'  => ' </div>',
 535              'class'  => 'action_icon btn btn-default btn-sm',
 536          ), $params );
 537  
 538      $show_recycle_bin = ( !$check_perm || $current_User->check_perm( 'blogs', 'editall' ) );
 539      if( $show_recycle_bin && ( !$force_show ) )
 540      { // get number of trash comments:
 541          $SQL = new SQL( 'Get number of trash comments' );
 542          $SQL->SELECT( 'COUNT( comment_ID )' );
 543          $SQL->FROM( 'T_comments' );
 544          $SQL->FROM_add( 'INNER JOIN T_items__item ON comment_item_ID = post_ID' );
 545          $SQL->FROM_add( 'INNER JOIN T_categories ON post_main_cat_ID = cat_ID' );
 546          $SQL->WHERE( 'comment_status = "trash"' );
 547          if( isset( $blog ) )
 548          {
 549              $SQL->WHERE_and( 'cat_blog_ID = '.$DB->quote( $blog ) );
 550          }
 551          $show_recycle_bin = ( $DB->get_var( $SQL->get() ) > 0 );
 552      }
 553  
 554      $result = $params['before'];
 555      if( $show_recycle_bin )
 556      { // show "Open recycle bin"
 557          global $CommentList;
 558          $comment_list_param_prefix = 'cmnt_fullview_';
 559          if( !empty( $CommentList->param_prefix ) )
 560          {
 561              $comment_list_param_prefix = $CommentList->param_prefix;
 562          }
 563          $result .= action_icon( T_('Open recycle bin'), 'recycle_full',
 564                          $admin_url.'?ctrl=comments&amp;blog='.$blog.'&amp;'.$comment_list_param_prefix.'show_statuses[]=trash', ' '.T_('Open recycle bin'), 5, 3, array( 'class' => $params['class'] ) );
 565      }
 566      return $result.$params['after'];
 567  }

 712   * Dispay the replies of a comment
 713   *
 714   * @param integer Comment ID
 715   * @param array Template params
 716   * @param integer Level
 717   */
 718  function display_comment_replies( $comment_ID, $params = array(), $level = 1 )
 719  {
 720      global $CommentReplies;
 721  
 722      $params = array_merge( array(
 723              'comment_template'    => '_item_comment.inc.php',
 724              'preview_block_start' => '',
 725              'preview_start'       => '<div class="bComment" id="comment_preview">',
 726              'preview_end'         => '</div>',
 727              'preview_block_end'   => '',
 728              'comment_start'       => '<div class="bComment">',
 729              'comment_end'         => '</div>',
 730              'comment_error_start' => '<div class="bComment" id="comment_error">',
 731              'comment_error_end'   => '</div>',
 732              'link_to'             => 'userurl>userpage', // 'userpage' or 'userurl' or 'userurl>userpage' or 'userpage>userurl'
 733              'author_link_text'    => 'name', // avatar_name | avatar_login | only_avatar | name | login | nickname | firstname | lastname | fullname | preferredname
 734          ), $params );
 735  
 736      if( isset( $CommentReplies[ $comment_ID ] ) )
 737      { // This comment has the replies
 738          foreach( $CommentReplies[ $comment_ID ] as $Comment )
 739          { // Loop through the replies:
 740              if( empty( $Comment->ID ) )
 741              { // Get html tag of the comment block of preview
 742                  $comment_start = $Comment->email_is_detected ? $params['comment_error_start'] : $params['preview_start'];
 743              }
 744              else
 745              { // Get html tag of the comment block of existing comment
 746                  $comment_start = $params['comment_start'];
 747              }
 748  
 749              // Set margin left for each sub level comment
 750              $attrs = ' style="margin-left:'.( 20 * $level ).'px"';
 751              if( strpos( $comment_start, 'class="' ) === false )
 752              { // Add a class attribute for the replied comment
 753                  $attrs .= ' class="replied_comment"';
 754              }
 755              else
 756              { // Add a class name for the replied comment
 757                  $comment_start = str_replace( 'class="', 'class="replied_comment ', $comment_start );
 758              }
 759              $comment_start = str_replace( '>', $attrs.'>', $comment_start );
 760  
 761              if( ! empty( $Comment->ID ) )
 762              { // Comment from DB
 763                  skin_include( $params['comment_template'], array_merge( $params, array(
 764                          'Comment'          => & $Comment,
 765                          'comment_start'    => $comment_start
 766                      ) ) );
 767              }
 768              else
 769              { // PREVIEW comment
 770                  skin_include( $params['comment_template'], array_merge( $params, array(
 771                          'Comment'              => & $Comment,
 772                          'comment_block_start'  => $Comment->email_is_detected ? '' : $params['preview_block_start'],
 773                          'comment_start'        => $comment_start,
 774                          'comment_end'          => $Comment->email_is_detected ? $params['comment_error_end'] : $params['preview_end'],
 775                          'comment_block_end'    => $Comment->email_is_detected ? '' : $params['preview_block_end'],
 776                          'author_link_text'     => $params['author_link_text'],
 777                      ) ) );
 778              }
 779  
 780              // Display the rest replies recursively
 781              display_comment_replies( $Comment->ID, $params, $level + 1 );
 782          }
 783      }
 784  }

 917  /**
 918   * Delete the comments
 919   *
 920   * @param string Type of deleting:
 921   *               'recycle' - to move into recycle bin
 922   *               'delete' - to delete permanently
 923   * @param string sql query to get deletable comment ids
 924   */
 925  function comment_mass_delete_process( $mass_type, $deletable_comments_query )
 926  {
 927      if( $mass_type != 'recycle' && $mass_type != 'delete' )
 928      { // Incorrect action
 929          return;
 930      }
 931  
 932      global $DB, $cache_comments_has_replies, $user_post_read_statuses, $cache_postcats;
 933  
 934      /**
 935       * Disable log queries because it increases the memory and stops the process with error "Allowed memory size of X bytes exhausted..."
 936       */
 937      $DB->log_queries = false;
 938  
 939      $Form = new Form();
 940  
 941      $Form->begin_form( 'fform' );
 942  
 943      $Form->begin_fieldset( T_('Mass deleting log') );
 944  
 945      echo T_('The comments are deleting...');
 946      evo_flush();
 947  
 948      $CommentCache = & get_CommentCache();
 949      $ItemCache = & get_ItemCache();
 950      $ChapterCache = & get_ChapterCache();
 951  
 952      // Get the comments by 1000 to avoid an exhausting of memory
 953      $deletable_comment_ids = $DB->get_col( $deletable_comments_query.' LIMIT 1000' );
 954      while( !empty( $deletable_comment_ids ) )
 955      {
 956          // Get the first slice of the deletable comment ids list
 957          $ids = array_splice( $deletable_comment_ids, 0, 100 );
 958          // Make sure the CommentCache is empty
 959          $CommentCache->clear();
 960          // Load deletable comment ids
 961          $CommentCache->load_list( $ids );
 962  
 963          while( ( $iterator_Comment = & $CommentCache->get_next() ) != NULL )
 964          { // Delete all comments from CommentCache
 965              $iterator_Comment->dbdelete( $mass_type == 'delete' );
 966          }
 967  
 968          // Display progress dot
 969          echo ' .';
 970          evo_flush();
 971  
 972          if( empty( $deletable_comment_ids ) )
 973          {
 974              // Clear all caches to save memory
 975              $ItemCache->clear();
 976              $ChapterCache->clear();
 977              $cache_comments_has_replies = array();
 978              $user_post_read_statuses = array();
 979              $cache_postcats = array();
 980  
 981              // Get new portion of deletable comments
 982              $deletable_comment_ids = $DB->get_col( $deletable_comments_query.' LIMIT 1000' );
 983          }
 984      }
 985  
 986      echo ' OK';
 987  
 988      $Form->end_form();
 989  
 990      // Clear a comment cache
 991      $CommentCache->clear();
 992  }

 995  /**
 996   * Move all child comments to new post by parent comment ID
 997   *
 998   * @param integer/array Parent comment IDs
 999   * @param integer Item ID
1000   * @param boolean TRUE to convert a comment to root/top comment of the post
1001   */
1002  function move_child_comments_to_item( $comment_IDs, $item_ID, $convert_to_root = true )
1003  {
1004      global $DB;
1005  
1006      $child_comments_SQL = new SQL();
1007      $child_comments_SQL->SELECT( 'comment_ID' );
1008      $child_comments_SQL->FROM( 'T_comments' );
1009      $child_comments_SQL->WHERE( 'comment_in_reply_to_cmt_ID IN ( '.$DB->quote( $comment_IDs ).' )' );
1010      $child_comments_IDs = $DB->get_col( $child_comments_SQL->get() );
1011

1013      { // No child comments, Exit here
1014          return;

1026          SET comment_item_ID = '.$DB->quote( $item_ID ).$update_sql.'
1027          WHERE comment_in_reply_to_cmt_ID IN ( '.$DB->quote( $comment_IDs ).' )' );

1462   * Get the edit actions for comment
1463   *
1464   * @param object Comment
1465   * @return string The edit actions
1466   */
1467  function comment_edit_actions( $Comment )
1468  {
1469      global $current_User, $admin_url;
1470  
1471      $r = '';
1472      if( !is_logged_in() )
1473      {
1474          return $r;
1475      }
1476  
1477      $user_has_edit_perm = $current_User->check_perm( 'comment!CURSTATUS', 'edit', false, $Comment );
1478      $user_has_delete_perm = $current_User->check_perm( 'comment!CURSTATUS', 'delete', false, $Comment );
1479  
1480      if( $user_has_edit_perm || $user_has_delete_perm )
1481      { // Display edit and delete button if current user has the rights:
1482          $redirect_to = rawurlencode( regenerate_url( 'comment_ID,action', 'filter=restore', '', '&' ) );
1483  
1484          if( $user_has_edit_perm )
1485          { // Display edit button only if current user can edit comment with current status
1486              $Comment->get_Item();
1487              $item_Blog = & $Comment->Item->get_Blog();
1488              if( $item_Blog->get_setting( 'in_skin_editing' ) && !is_admin_page() )
1489              {
1490                  $edit_url = url_add_param( $item_Blog->gen_blogurl(), 'disp=edit_comment&amp;c='.$Comment->ID );
1491              }
1492              else
1493              {
1494                  $edit_url = $admin_url.'?ctrl=comments&amp;comment_ID='.$Comment->ID.'&amp;action=edit&amp;redirect_to='.$redirect_to;
1495              }
1496              $r .= action_icon( TS_('Edit this comment...'), 'properties', $edit_url );
1497          }
1498  
1499          if( $user_has_delete_perm )
1500          { // Display delete/recycle button because current user has permission to delete/recycle this comment
1501              $params = array();
1502              if( $Comment->status == 'trash' )
1503              { // Comment is already in the recycle bin, display delete action and add js confirm
1504                  $title = T_('Delete this comment!');
1505                  $params['onclick'] = "return confirm('".TS_('You are about to delete this comment!\\nThis cannot be undone!')."')";
1506              }
1507              else
1508              { // Comment will be moved into the recycle bin
1509                  $title = T_('Recycle this comment!');
1510              }
1511  
1512              $r .=  action_icon( $title, 'recycle',
1513                  $admin_url.'?ctrl=comments&amp;comment_ID='.$Comment->ID.'&amp;action=delete&amp;'.url_crumb('comment')
1514                  .'&amp;redirect_to='.$redirect_to, NULL, NULL, NULL, $params );
1515          }
1516      }
1517  
1518      return $r;
1519  }
