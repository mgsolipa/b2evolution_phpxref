  25  function trackbacks( $post_trackbacks, $Item )
  26  {
  27      global $Messages;
  28  
  29      $excerpt = $Item->get_excerpt();
  30  
  31      $Messages->add( T_('Excerpt sent in trackbacks:').' '.$excerpt, 'note' );
  32      $trackback_urls = preg_split( '/( )+/', $post_trackbacks, 10 );        // fplanque: ;
  33      foreach($trackback_urls as $tb_url)
  34      { // trackback each url:
  35          $tb_url = trim($tb_url);
  36          if( empty( $tb_url ) ) continue;
  37          trackback( $tb_url, $Item->title, $excerpt, $Item->ID, $Item->get_permanent_url('', '', '&') );
  38      }
  39  }

  43   * Send Trackback to single URL
  44   *
  45   * trackback(-)
  46   *
  47   * @todo add autodiscovery
  48   */
  49  function trackback(
  50      $trackback_url,
  51      $title,
  52      $excerpt,
  53      $ID,
  54      $url ) // post ID
  55  {
  56      global $app_name, $app_version, $Blog, $Messages;
  57  
  58      $trackback_message = T_('Sending trackback to:').' '.htmlspecialchars($trackback_url).' ';
  59  
  60      $title = rawurlencode($title);
  61      $excerpt = rawurlencode($excerpt);
  62      $blog_name = rawurlencode($Blog->get( 'name' ));
  63      $url = rawurlencode($url);
  64      // dis is the trackback stuff to be sent:
  65      $query_string = 'title='.$title.'&url='.$url.'&blog_name='.$blog_name.'&excerpt='.$excerpt;
  66      // echo "url:$trackback_url<br>$sending:$query_string<br />";
  67  
  68      $result = '';
  69      if (strstr($trackback_url, '?'))
  70      {    // use a HTTP GET request
  71          $Messages->add( $trackback_message.'[get]', 'note' );
  72          $trackback_url = parse_url($trackback_url.'&'.$query_string);
  73  
  74          if (!empty($trackback_url['host']) && !empty($trackback_url['path'])
  75              && ($fp = @fsockopen($trackback_url['host'], 80, $foo, $foo, 20)) !== false)
  76          {
  77              $header  = 'GET '.$trackback_url['path'].'?'.$trackback_url['query']." HTTP/1.0\r\n";
  78              $header .= 'Host: '.$trackback_url['host']."\r\n";
  79              $header .= 'User-Agent: '.$app_name.'/'.$app_version."\r\n\r\n";
  80              fwrite($fp, $header, strlen($header));
  81  
  82              if(function_exists('stream_set_timeout'))
  83              {
  84                  stream_set_timeout($fp, 20); // PHP 4.3.0
  85              }
  86              else
  87              {
  88                  socket_set_timeout($fp, 20); // PHP 4
  89              }
  90  
  91              while (!feof($fp))
  92              {
  93                  $result .= fgets($fp);
  94              }
  95              fclose($fp);
  96  
  97              /* debug code
  98              $df = fopen('trackback.log', 'a');
  99              fwrite($df, "---------\nRequest:\n\n$header\n\nResponse:\n\n$result\n");
 100              fclose($df);
 101              */
 102          }
 103      }
 104      else
 105      {
 106          $Messages->add( $trackback_message.'[post]', 'note' );
 107          $trackback_url = parse_url($trackback_url);
 108          if( ! empty($trackback_url['host']) && ! empty($trackback_url['path']) )
 109          { // Only try trackback if we have host and path:
 110              $port = isset($trackback_url['port']) ? $trackback_url['port'] : 80;
 111              $http_request  = 'POST '.$trackback_url['path']." HTTP/1.0\r\n";
 112              $http_request .= 'Host: '.$trackback_url['host']."\r\n";
 113              $http_request .= 'Content-Type: application/x-www-form-urlencoded'."\r\n";
 114              $http_request .= 'Content-Length: '.strlen($query_string)."\r\n";
 115              $http_request .= "User-Agent: $app_name/$app_version\r\n";
 116              $http_request .= "\r\n";
 117              $http_request .= $query_string;
 118              evo_flush();
 119              if( $fs = @fsockopen($trackback_url['host'], $port, $errno, $errst, 20) ) // this timeout is just for setting up the socket
 120              {
 121                  // Set timeout for data:
 122                  if( function_exists('stream_set_timeout') )
 123                  {
 124                      stream_set_timeout( $fs, 20 ); // PHP 4.3.0
 125                  }
 126                  else
 127                  {
 128                      socket_set_timeout( $fs, 20 ); // PHP 4
 129                  }
 130                  fputs($fs, $http_request);
 131                  $result = '';
 132                  while(!feof($fs))
 133                  {
 134                      $result .= fgets($fs, 4096);
 135                  }
 136  
 137                  /* debug code
 138                  $debug_file = 'trackback.log';
 139                  $fp = fopen($debug_file, 'a');
 140                  fwrite($fp, "\n*****\nRequest:\n\n$http_request\n\nResponse:\n\n$result");
 141                  while(!@feof($fs)) {
 142                      fwrite($fp, @fgets($fs, 4096));
 143                  }
 144                  fwrite($fp, "\n\n");
 145                  fclose($fp);
 146                  */
 147  
 148                  fclose($fs);
 149              }
 150          }
 151      }
 152      // extract the error code and message, then make the error code readable
 153      if ( preg_match("/<error>[\r\n\t ]*(\d+)[\r\n\t ]*<\/error>/", $result, $error) )
 154      {
 155          preg_match("/<message>(.*?)<\/message>/", $result, $error_message);
 156  
 157          $message = isset($error_message[1]) ? $error_message[1] : '';
 158  
 159          switch ($error[1]) {
 160              case '0':
 161                  $result_message = '[' . T_('Succeeded') . '] ' . $message;
 162                  break;
 163              case '1':
 164                  $result_message = '[' . T_('Failed') . '] ' . $message;
 165                  break;
 166              default:
 167                  $result_message = '[' . T_('Unknown error') . ' (' . $error[1] . ')] ' . $message;
 168                  break;
 169          }
 170      }
 171      else
 172      {
 173          $result_message = T_('No valid trackback response. Maybe the given url is not a Trackback url.') . ' &quot;' . $result . '&quot;';
 174      }
 175      $Messages->add( T_('Response:').' '.strip_tags($result_message), 'note' );
 176      return $result;
 177  }

 197          $blah = $zero;
 198      } elseif ($number == 1) {
 199          $blah = $one;
 200      } elseif ($number  > 1) {
 201          $n = $number;
 202          $more = str_replace('%d', $n, $more);
