 207       * Constructor, not meant to be called directly. Use {@link FileCache::get_by_root_and_path()}
 208       * instead, which provides caching and checks that only one object for
 209       * a unique file exists (references).
 210       *
 211       * @param string Root type: 'user', 'group', 'collection' or 'absolute'
 212       * @param integer ID of the user, the group or the collection the file belongs to...
 213       * @param string Posix subpath for this file/folder, relative to the associated root (no trailing slash)
 214       * @param boolean check for meta data?
 215       * @return mixed false on failure, File object on success

 217  	function File( $root_type, $root_ID, $rdfp_rel_path, $load_meta = false )
 218      {
 219          global $Debuglog;
 220  
 221          $Debuglog->add( "new File( $root_type, $root_ID, $rdfp_rel_path, load_meta=$load_meta)", 'files' );
 222  
 223          // Call parent constructor
 224          parent::DataObject( 'T_files', 'file_', 'file_ID', '', '', '', '' );
 225  
 226          // Memorize filepath:
 227          $FileRootCache = & get_FileRootCache();
 228          $this->_FileRoot = & $FileRootCache->get_by_type_and_ID( $root_type, $root_ID );
 229  
 230          // If there's a valid file root, handle extra stuff. This should not get done when the FileRoot is invalid.
 231          if( $this->_FileRoot )
 232          {
 233              $this->_rdfp_rel_path = trim( str_replace( '\\', '/', $rdfp_rel_path ), '/' );
 234              $this->_adfp_full_path = $this->_FileRoot->ads_path.$this->_rdfp_rel_path;
 235              $this->_name = basename( $this->_adfp_full_path );
 236              $this->_dir = dirname( $this->_adfp_full_path ).'/';
 237              $this->_md5ID = md5( $this->_adfp_full_path );
 238  
 239              // Initializes file properties (type, size, perms...)
 240              $this->load_properties();
 241  
 242              if( $load_meta )
 243              { // Try to load DB meta info:
 244                  $this->load_meta();
 245              }
 246          }
 247      }

 292      /**
 293       * Attempt to load meta data.
 294       *
 295       * Will attempt only once and cache the result.
 296       *
 297       * @param boolean create meta data in DB if it doesn't exist yet? (generates a $File->ID)
 298       * @param object database row containing all fields needed to initialize meta data
 299       * @return boolean true if meta data has been loaded/initialized.
 300       */
 301  	function load_meta( $force_creation = false, $row = NULL )
 302      {
 303          global $DB, $Debuglog;
 304  
 305          if( $this->meta == 'unknown' )
 306          { // We haven't tried loading yet:
 307              if( is_null( $row ) )
 308              {    // No DB data has been provided:
 309                  $row = $DB->get_row( "
 310                      SELECT * FROM T_files
 311                       WHERE file_path_hash = ".$DB->quote( md5( $this->_FileRoot->type.$this->_FileRoot->in_type_ID.$this->_rdfp_rel_path, true ) ),
 312                      OBJECT, 0, 'Load file meta data' );
 313              }
 314  
 315              // We check that we got something AND that the CASE matches (because of case insensitive collations on MySQL)
 316              if( $row && $row->file_path == $this->_rdfp_rel_path )
 317              { // We found meta data
 318                  $Debuglog->add( "Loaded metadata for {$this->_FileRoot->ID}:{$this->_rdfp_rel_path}", 'files' );
 319                  $this->meta  = 'loaded';
 320                  $this->ID    = $row->file_ID;
 321                  $this->type  = $row->file_type;
 322                  $this->title = $row->file_title;
 323                  $this->alt   = $row->file_alt;
 324                  $this->desc  = $row->file_desc;
 325                  if( isset( $row->file_hash ) )
 326                  {
 327                      $this->hash  = $row->file_hash;
 328                  }
 329                  $this->path_hash = $row->file_path_hash;
 330                  if( isset( $row->file_can_be_main_profile ) )
 331                  {
 332                      $this->can_be_main_profile = $row->file_can_be_main_profile;
 333                  }
 334  
 335                  // Store this in the FileCache:
 336                  $FileCache = & get_FileCache();
 337                  $FileCache->add( $this );
 338              }
 339              else
 340              { // No meta data...
 341                  $Debuglog->add( sprintf('No metadata could be loaded for %d:%s', $this->_FileRoot ? $this->_FileRoot->ID : 'FALSE', $this->_rdfp_rel_path), 'files' );
 342                  $this->meta = 'notfound';
 343  
 344                  if( $force_creation )
 345                  {    // No meta data, we have to create it now!
 346                      $this->dbinsert();
 347                  }
 348              }
 349          }
 350  
 351          return ($this->meta == 'loaded');
 352      }

 355      /**
 356       * Create the file/folder on disk, if it does not exist yet.
 357       *
 358       * Also sets file permissions.
 359       * Also inserts meta data into DB (if file/folder was successfully created).
 360       *
 361       * @param string type ('dir'|'file')
 362       * @param string optional permissions (octal format), otherwise the default from {@link $Settings} gets used
 363       * @return boolean true if file/folder was created, false on failure
 364       */
 365  	function create( $type = 'file', $chmod = NULL )
 366      {
 367          if( $type == 'dir' )
 368          { // Create an empty directory:
 369              $success = @mkdir( $this->_adfp_full_path );
 370              $this->_is_dir = true; // used by chmod
 371              $syslog_message = $success ?
 372                      'Folder %s was created' :
 373                      sprintf( 'Folder %s could not be created', '<b>'.$this->_adfp_full_path.'</b>' );
 374          }
 375          else
 376          { // Create an empty file:
 377              $success = @touch( $this->_adfp_full_path );
 378              $this->_is_dir = false; // used by chmod
 379              $syslog_message = $success ?
 380                      'File %s was created' :
 381                      sprintf( 'File %s could not be created', '<b>'.$this->_adfp_full_path.'</b>' );
 382          }
 383          $this->chmod( $chmod ); // uses $Settings for NULL
 384  
 385          if( $success )
 386          {    // The file/folder has been successfully created:
 387  
 388              // Initializes file properties (type, size, perms...)
 389              $this->load_properties();
 390  
 391              // If there was meta data for this file in the DB:
 392              // (maybe the file had existed before?)
 393              // Let's recycle it! :
 394              if( ! $this->load_meta() )
 395              { // No meta data could be loaded, let's make sure localization info gets recorded:
 396                  $this->set( 'root_type', $this->_FileRoot->type );
 397                  $this->set( 'root_ID', $this->_FileRoot->in_type_ID );
 398                  $this->set( 'path', $this->_rdfp_rel_path );
 399              }
 400  
 401              // Record to DB:
 402              $this->dbsave();
 403          }
 404  
 405          syslog_insert( sprintf( $syslog_message, '<b>'.$this->get_name().'</b>' ), 'info', 'file', $this->ID );
 406  
 407          return $success;
 408      }

 834      /**
 835       * Get permissions
 836       *
 837       * Possible return formats are:
 838       *   - 'raw'=integer
 839       *   - 'lsl'=string like 'ls -l'
 840       *   - 'octal'=3 digits
 841       *
 842       * Default value:
 843       *   - 'r'/'r+w' for windows
 844       *   - 'octal' for other OS
 845       *
 846       * @param string type, see desc above.
 847       * @return mixed permissions
 848       */
 849  	function get_perms( $type = NULL )
 850      {
 851          if( ! isset($this->_perms) )
 852          {
 853              $this->_perms = @fileperms( $this->_adfp_full_path );
 854          }
 855          switch( $type )
 856          {
 857              case 'raw':
 858                  return $this->_perms;
 859  
 860              case 'lsl':
 861                  $sP = '';
 862  
 863                  if(($this->_perms & 0xC000) == 0xC000)     // Socket
 864                      $sP = 's';
 865                  elseif(($this->_perms & 0xA000) == 0xA000) // Symbolic Link
 866                      $sP = 'l';
 867                  elseif(($this->_perms & 0x8000) == 0x8000) // Regular
 868                      $sP = '&minus;';
 869                  elseif(($this->_perms & 0x6000) == 0x6000) // Block special
 870                      $sP = 'b';
 871                  elseif(($this->_perms & 0x4000) == 0x4000) // Directory
 872                      $sP = 'd';
 873                  elseif(($this->_perms & 0x2000) == 0x2000) // Character special
 874                      $sP = 'c';
 875                  elseif(($this->_perms & 0x1000) == 0x1000) // FIFO pipe
 876                      $sP = 'p';
 877                  else                                   // UNKNOWN
 878                      $sP = 'u';
 879  
 880                  // owner
 881                  $sP .= (($this->_perms & 0x0100) ? 'r' : '&minus;') .
 882                         (($this->_perms & 0x0080) ? 'w' : '&minus;') .
 883                         (($this->_perms & 0x0040) ? (($this->_perms & 0x0800) ? 's' : 'x' )
 884                                                   : (($this->_perms & 0x0800) ? 'S' : '&minus;'));
 885  
 886                  // group
 887                  $sP .= (($this->_perms & 0x0020) ? 'r' : '&minus;') .
 888                         (($this->_perms & 0x0010) ? 'w' : '&minus;') .
 889                         (($this->_perms & 0x0008) ? (($this->_perms & 0x0400) ? 's' : 'x' )
 890                                                   : (($this->_perms & 0x0400) ? 'S' : '&minus;'));
 891  
 892                  // world
 893                  $sP .= (($this->_perms & 0x0004) ? 'r' : '&minus;') .
 894                         (($this->_perms & 0x0002) ? 'w' : '&minus;') .
 895                         (($this->_perms & 0x0001) ? (($this->_perms & 0x0200) ? 't' : 'x' )
 896                                                   : (($this->_perms & 0x0200) ? 'T' : '&minus;'));
 897                  return $sP;
 898  
 899              case NULL:
 900                  if( is_windows() )
 901                  {
 902                      if( $this->_perms & 0x0080 )
 903                      {
 904                          return 'r+w';
 905                      }
 906                      else return 'r';
 907                  }
 908  
 909              case 'octal':
 910                  return substr( sprintf('%o', $this->_perms), -3 );
 911          }
 912  
 913          return false;
 914      }
