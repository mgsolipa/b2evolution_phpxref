  33   * Converts bytes to readable bytes/kb/mb/gb, like "12.45mb"
  34   *
  35   * @param integer bytes
  36   * @param boolean use HTML <abbr> tags
  37   * @param boolean Display full text of size type when $htmlabbr == false
  38   * @return string bytes made readable
  39   */
  40  function bytesreadable( $bytes, $htmlabbr = true, $display_size_type = true )
  41  {
  42      static $types = NULL;
  43  
  44      if( empty($bytes) )
  45      {
  46          return T_('Empty');
  47      }
  48  
  49      if( !isset($types) )
  50      { // generate once:
  51          $types = array(
  52              0 => array( 'abbr' => /* TRANS: Abbr. for "Bytes" */ T_('B.'), 'text' => T_('Bytes') ),
  53              1 => array( 'abbr' => /* TRANS: Abbr. for "Kilobytes" */ T_('KB'), 'text' => T_('Kilobytes') ),
  54              2 => array( 'abbr' => /* TRANS: Abbr. for Megabytes */ T_('MB'), 'text' => T_('Megabytes') ),
  55              3 => array( 'abbr' => /* TRANS: Abbr. for Gigabytes */ T_('GB'), 'text' => T_('Gigabytes') ),
  56              4 => array( 'abbr' => /* TRANS: Abbr. for Terabytes */ T_('TB'), 'text' => T_('Terabytes') )
  57          );
  58      }
  59  
  60      for( $i = 0; $bytes > 1024; $i++ )
  61      {
  62          $bytes /= 1024;
  63      }
  64  
  65      // Format to maximum of 1 digit after .
  66      $precision = max( 0, ( 1 -floor(log($bytes)/log(10))) );
  67      $r = sprintf( '%.'.$precision.'f', $bytes );
  68  
  69      $r .= $htmlabbr ? ( '&nbsp;<abbr title="'.$types[$i]['text'].'">' ) : ' ';
  70      $r .= $types[$i]['abbr'];
  71      $r .= $htmlabbr ? '</abbr>' : ( $display_size_type ? ' ('.$types[$i]['text'].')' : '' );
  72  
  73      // $r .= ' '.$precision;
  74  
  75      return $r;
  76  }

  79  /**
  80   * Get an array of all directories (and optionally files) of a given
  81   * directory, either flat (one-dimensional array) or multi-dimensional (then
  82   * dirs are the keys and hold subdirs/files).
  83   *
  84   * Note: there is no ending slash on dir names returned.
  85   *
  86   * @param string the path to start
  87   * @param array of params
  88   * @return false|array false if the first directory could not be accessed,
  89   *                     array of entries otherwise
  90   */
  91  function get_filenames( $path, $params = array() )
  92  {
  93      global $Settings;
  94  
  95      $params = array_merge( array(
  96              'inc_files'      => true,  // include files (not only directories)
  97              'inc_dirs'       => true,  // include directories (not the directory itself!)
  98              'flat'           => true,  // return a one-dimension-array
  99              'recurse'        => true,  // recurse into subdirectories
 100              'basename'       => false, // get the basename only
 101              'trailing_slash' => false, // add trailing slash
 102              'inc_hidden'     => true,  // inlcude hidden files, directories and content
 103              'inc_evocache'   => false, // exclude evocache directories and content
 104              'inc_temp'       => true,  // include temporary files and directories
 105          ), $params );
 106  
 107      $r = array();
 108  
 109      $path = trailing_slash( $path );
 110  
 111      if( $dir = @opendir( $path ) )
 112      {
 113          while( ( $file = readdir( $dir ) ) !== false )
 114          {
 115              if( $file == '.' || $file == '..' )
 116              { // Skip these reserved names
 117                  continue;
 118              }
 119  
 120              // asimo> Also check if $Settings is not empty because this function is called from the install srcipt too, where $Settings is not initialized yet
 121              if( ! $params['inc_evocache'] && ! empty( $Settings ) && $file == $Settings->get( 'evocache_foldername' ) )
 122              { // Do not load evocache directories
 123                  continue;
 124              }
 125  
 126              // Check for hidden status...
 127              if( ( ! $params['inc_hidden'] ) && ( substr( $file, 0, 1 ) == '.' ) )
 128              { // Do not load & show hidden files and folders (prefixed with .)
 129                  continue;
 130              }
 131  
 132              // Check for temp directories/files
 133              if( ! $params['inc_temp'] && $file == 'upload-tmp' )
 134              { // Do not load temporary files and directories
 135                  continue;
 136              }
 137  
 138              if( is_dir( $path.$file ) )
 139              {
 140                  if( $params['flat'] )
 141                  {
 142                      if( $params['inc_dirs'] )
 143                      {
 144                          $directory_name = $params['basename'] ? $file : $path.$file;
 145                          if( $params['trailing_slash'] )
 146                          {
 147                              $directory_name = trailing_slash( $directory_name );
 148                          }
 149  
 150                          $r[] = $directory_name;
 151                      }
 152                      if( $params['recurse'] )
 153                      {
 154                          $rSub = get_filenames( $path.$file, $params );
 155                          if( $rSub )
 156                          {
 157                              $r = array_merge( $r, $rSub );
 158                          }
 159                      }
 160                  }
 161                  else
 162                  {
 163                      $r[ $file ] = get_filenames( $path.$file, $params );
 164                  }
 165              }
 166              elseif( $params['inc_files'] )
 167              {
 168                  $r[] = $params['basename'] ? $file : $path.$file;
 169              }
 170          }
 171          closedir( $dir );
 172      }
 173      else
 174      {
 175          return false;
 176      }
 177  
 178      return $r;
 179  }

 406  /**
 407   * Returns canonicalized pathname of a directory + ending slash
 408   *
 409   * @param string absolute path to be reduced ending with slash
 410   * @return string absolute reduced path, slash terminated or NULL if the path could not get canonicalized.
 411   */
 412  function get_canonical_path( $ads_path )
 413  {
 414      // Remove windows backslashes:
 415      $ads_path = str_replace( '\\', '/', $ads_path );
 416  
 417      $is_absolute = is_absolute_pathname($ads_path);
 418  
 419      // Make sure there's a trailing slash
 420      $ads_path = trailing_slash($ads_path);
 421  
 422      while( strpos($ads_path, '//') !== false )
 423      {
 424          $ads_path = str_replace( '//', '/', $ads_path );
 425      }
 426      while( strpos($ads_path, '/./') !== false )
 427      {
 428          $ads_path = str_replace( '/./', '/', $ads_path );
 429      }
 430      $parts = explode('/', $ads_path);
 431      for( $i = 0; $i < count($parts); $i++ )
 432      {
 433          if( $parts[$i] != '..' )
 434          {
 435              continue;
 436          }
 437          if( $i <= 0 || $parts[$i-1] == '' || substr($parts[$i-1], -1) == ':' /* windows drive letter */ )
 438          {
 439              return NULL;
 440          }
 441          // Remove ".." and the part before it
 442          unset($parts[$i-1], $parts[$i]);
 443          // Respin array
 444          $parts = array_values($parts);
 445          $i = $i-2;
 446      }
 447      $ads_realpath = implode('/', $parts);
 448  
 449      // pre_dump( 'get_canonical_path()', $ads_path, $ads_realpath );
 450  
 451      if( strpos( $ads_realpath, '..' ) !== false )
 452      {    // Path malformed:
 453          return NULL;
 454      }
 455  
 456      if( $is_absolute && ! strlen($ads_realpath) )
 457      {
 458          return NULL;
 459      }
 460  
 461      return $ads_realpath;
 462  }

 489  /**
 490   * Process filename:
 491   *  - convert to lower case
 492   *  - replace consecutive dots with one dot
 493   *  - if force_validation is true, then replace every not valid character to '_'
 494   *  - check if file name is valid
 495   *
 496   * @param string file name (by reference) - this file name will be processed
 497   * @param boolean force validation ( replace not valid characters to '_' without warning )
 498   * @param boolean Crop long file name to $filename_max_length
 499   * @param object FileRoot
 500   * @param string Path in the FileRoot
 501   * @return error message if the file name is not valid, false otherwise
 502   */
 503  function process_filename( & $filename, $force_validation = false, $autocrop_long_file = false, $FileRoot = NULL, $path = '' )
 504  {
 505      if( empty( $filename ) )
 506      {
 507          return T_( 'Empty file name is not valid.' );
 508      }
 509  
 510      if( $autocrop_long_file && ! empty( $FileRoot ) )
 511      { // Crop long file names
 512          global $filename_max_length, $Messages;
 513  
 514          if( ! empty( $filename_max_length ) &&
 515                  $filename_max_length > 8 &&
 516                  strlen( $filename ) > $filename_max_length )
 517          { // Limit file name by $filename_max_length
 518              $new_file_ext_pos = strrpos( $filename, '.' );
 519              $new_file_name = substr( $filename, 0, $new_file_ext_pos );
 520              $new_file_name = trim( substr( $new_file_name, 0, $filename_max_length - 8 ) );
 521              $new_file_ext = substr( $filename, $new_file_ext_pos - strlen( $filename ) + 1 );
 522  
 523              $fnum = 1;
 524              $FileCache = & get_FileCache();
 525              do
 526              { // Make file name unique
 527                  $filename = $new_file_name.'-'.$fnum.'.'.$new_file_ext;
 528                  $newFile = & $FileCache->get_by_root_and_path( $FileRoot->type, $FileRoot->in_type_ID, trailing_slash( $path ).$filename, true );
 529                  $fnum++;
 530              }
 531              while( $newFile->exists() );
 532              $filename = $newFile->get( 'name' );
 533  
 534              $Messages->add( T_('The filename was too long. It has been shortened.'), 'warning' );
 535          }
 536      }
 537  
 538      if( $force_validation )
 539      { // replace every not valid characters
 540          $filename = preg_replace( '/[^a-z0-9\-_.]+/i', '_', $filename );
 541          // Make sure the filename length doesn't exceed the maximum allowed. Remove characters from the end of the filename ( before the extension ) if required.
 542          $extension_pos = strrpos( $filename, '.' );
 543          $filename = fix_filename_length( $filename, strrpos( $filename, '.', ( $extension_pos ? $extension_pos : strlen( $filename ) ) ) );
 544      }
 545  
 546      // check if the file name contains consecutive dots, and replace them with one dot without warning ( keep only one dot '.' instead of '...' )
 547      $filename = preg_replace( '/\.(\.)+/', '.', utf8_strtolower( $filename ) );
 548  
 549      if( $error_filename = validate_filename( $filename ) )
 550      { // invalid file name
 551          return $error_filename;
 552      }
 553  
 554      // on success
 555      return false;
 556  }

 580      }

 596      }

 601          $FiletypeCache = & get_FiletypeCache();

 611              }
 612          }
 613          else
 614          { // Filename hasn't an allowed extension
 615              return sprintf( T_('&laquo;%s&raquo; has an unrecognized extension.'), $filename );
 616          }
 617      }
 618      else
 619      { // Filename hasn't a valid extension
 620          return sprintf( T_('&laquo;%s&raquo; has not a valid extension.'), $filename );
 621      }

1013  /**
1014   * Copy directory recursively or one file
1015   *
1016   * @param string Source path
1017   * @param string Destination path
1018   * @param string Name of folder name for the copied folder, Use NULL to get a folder name from source
1019   * @param array What directories should be excluded
1020   * @return boolean TRUE on success, FALSE when no permission
1021   */
1022  function copy_r( $source, $dest, $new_folder_name = NULL, $exclude_dirs = array() )
1023  {
1024      $result = true;
1025  
1026      if( is_dir( $source ) )
1027      { // Copy directory recusively
1028          if( in_array( basename( $source ), $exclude_dirs ) )
1029          { // Don't copy this folder
1030              return true;
1031          }
1032          if( ! ( $dir_handle = @opendir( $source ) ) )
1033          { // Unable to open dir
1034              return false;
1035          }
1036          $source_folder = is_null( $new_folder_name ) ? basename( $source ) : $new_folder_name;
1037          if( ! mkdir_r( $dest.'/'.$source_folder ) )
1038          { // No rights to create a dir
1039              return false;
1040          }
1041          while( $file = readdir( $dir_handle ) )
1042          {
1043              if( $file != '.' && $file != '..' )
1044              {
1045                  if( is_dir( $source.'/'.$file ) )
1046                  { // Copy the files of subdirectory
1047                      $result = copy_r( $source.'/'.$file, $dest.'/'.$source_folder, NULL, $exclude_dirs ) && $result;
1048                  }
1049                  else
1050                  { // Copy one file of the directory
1051                      $result = @copy( $source.'/'.$file, $dest.'/'.$source_folder.'/'.$file ) && $result;
1052                  }
1053              }
1054          }
1055          closedir( $dir_handle );
1056      }
1057      else
1058      { // Copy a file and check destination folder for existing
1059          $dest_folder = preg_replace( '#(.+)/[^/]+$#', '$1', $dest );
1060          if( ! file_exists( $dest_folder ) )
1061          { // Create destination folder recursively if it doesn't exist
1062              if( ! mkdir_r( $dest_folder ) )
1063              { // Unable to create a destination folder
1064                  return false;
1065              }
1066          }
1067          // Copy a file
1068          $result = @copy( $source, $dest );
1069      }
1070  
1071      return $result;
1072  }

1308  /**
1309   * Process file uploads (this can process multiple file uploads at once)
1310   *
1311   * @param string FileRoot id string
1312   * @param string the upload dir relative path in the FileRoot
1313   * @param boolean Shall we create path dirs if they do not exist?
1314   * @param boolean Shall we check files add permission for current_User?
1315   * @param boolean upload quick mode
1316   * @param boolean show warnings if filename is not valid
1317   * @param integer minimum size for pictures in pixels (width and height)
1318   * @return mixed NULL if upload was impossible to complete for some reason (wrong fileroot ID, insufficient user permission, etc.)
1319   *                        array, which contains uploadedFiles, failedFiles, renamedFiles and renamedMessages
1320   */
1321  function process_upload( $root_ID, $path, $create_path_dirs = false, $check_perms = true, $upload_quickmode = true, $warn_invalid_filenames = true, $min_size = 0 )
1322  {

1324  
1325      if( empty($_FILES) )
1326      {    // We have NO uploaded files to process...
1327          return NULL;
1328      }

1330      /**
1331       * Remember failed files (and the error messages)
1332       * @var array
1333       */
1334      $failedFiles = array();
1335      /**
1336       * Remember uploaded files
1337       * @var array
1338       */
1339      $uploadedFiles = array();
1340      /**
1341       * Remember renamed files
1342       * @var array
1343       */
1344      $renamedFiles = array();
1345      /**
1346       * Remember renamed Messages
1347       * @var array
1348       */
1349      $renamedMessages = array();

1352      $fm_FileRoot = & $FileRootCache->get_by_ID($root_ID, true);

1354      { // fileRoot not found:
1355          return NULL;
1356      }

1358      if( $check_perms && ( !isset( $current_User ) || $current_User->check_perm( 'files', 'add', false, $fm_FileRoot ) ) )
1359      { // Permission check required but current User has no permission to upload:
1360          return NULL;
1361      }
1362  
1363      // Let's get into requested list dir...
1364      $non_canonical_list_path = $fm_FileRoot->ads_path.$path;
1365      // Dereference any /../ just to make sure, and CHECK if directory exists:
1366      $ads_list_path = get_canonical_path( $non_canonical_list_path );

1368      // check if the upload dir exists
1369      if( !is_dir( $ads_list_path ) )
1370      {
1371          if( $create_path_dirs )
1372          { // Create path
1373              mkdir_r( $ads_list_path );
1374          }
1375          else
1376          { // This case should not happen! If it happens then there is a bug in the code where this function was called!
1377              return NULL;
1378          }
1379      }
1380  
1381      // Get param arrays for all uploaded files:
1382      $uploadfile_title = param( 'uploadfile_title', 'array:string', array() );
1383      $uploadfile_alt = param( 'uploadfile_alt', 'array:string', array() );
1384      $uploadfile_desc = param( 'uploadfile_desc', 'array:string', array() );
1385      $uploadfile_name = param( 'uploadfile_name', 'array:string', array() );
1386  
1387      // LOOP THROUGH ALL UPLOADED FILES AND PROCCESS EACH ONE:
1388      foreach( $_FILES['uploadfile']['name'] as $lKey => $lName )
1389      {
1390          if( empty( $lName ) )
1391          { // No file name:
1392              if( $upload_quickmode
1393                   || !empty( $uploadfile_title[$lKey] )
1394                   || !empty( $uploadfile_alt[$lKey] )
1395                   || !empty( $uploadfile_desc[$lKey] )
1396                   || !empty( $uploadfile_name[$lKey] ) )
1397              { // User specified params but NO file! Warn the user:
1398                  $failedFiles[$lKey] = T_( 'Please select a local file to upload.' );
1399              }
1400              // Abort upload for this file:
1401              continue;
1402          }
1403  
1404          if( $Settings->get( 'upload_maxkb' )
1405                  && $_FILES['uploadfile']['size'][$lKey] > $Settings->get( 'upload_maxkb' )*1024 )
1406          { // File is larger than allowed in settings:
1407              $failedFiles[$lKey] = sprintf(
1408                      T_('The file is too large: %s but the maximum allowed is %s.'),
1409                      bytesreadable( $_FILES['uploadfile']['size'][$lKey] ),
1410                      bytesreadable($Settings->get( 'upload_maxkb' )*1024) );
1411              // Abort upload for this file:
1412              continue;
1413          }
1414  
1415          if( ( !( $_FILES['uploadfile']['error'][$lKey] ) ) && ( !empty( $min_size ) ) )
1416          { // If there is no error and a minimum size is required, check if the uploaded picture satisfies the "minimum size" criteria
1417              $image_sizes = imgsize( $_FILES['uploadfile']['tmp_name'][$lKey], 'widthheight' );
1418              if( $image_sizes[0] < $min_size || $image_sizes[1] < $min_size )
1419              {    // Abort upload for this file:
1420                  $failedFiles[$lKey] = sprintf(
1421                      T_( 'Your profile picture must have a minimum size of %dx%d pixels.' ),
1422                      $min_size,
1423                      $min_size );
1424                  continue;
1425              }
1426          }
1427  
1428          if( $_FILES['uploadfile']['error'][$lKey] )
1429          { // PHP itself has detected an error!:
1430              switch( $_FILES['uploadfile']['error'][$lKey] )
1431              {
1432                  case UPLOAD_ERR_FORM_SIZE:
1433                      // The uploaded file exceeds the MAX_FILE_SIZE directive that was specified in the html form.
1434                      // This can easily be edited by the user/hacker, so we do not use it.. file size gets checked for real just above.
1435                      break;
1436  
1437                  case UPLOAD_ERR_INI_SIZE:
1438                      // File is larger than allowed in php.ini:
1439                      $failedFiles[$lKey] = 'The file exceeds the upload_max_filesize directive in php.ini.'; // Configuration error, no translation
1440                      // Abort upload for this file:
1441                      continue;
1442  
1443                  case UPLOAD_ERR_PARTIAL:
1444                      $failedFiles[$lKey] = T_('The file was only partially uploaded.');
1445                      // Abort upload for this file:
1446                      continue;
1447  
1448                  case UPLOAD_ERR_NO_FILE:
1449                      // Is probably the same as empty($lName) before.
1450                      $failedFiles[$lKey] = T_('No file was uploaded.');
1451                      // Abort upload for this file:
1452                      continue;
1453  
1454                  case 6: // numerical value of UPLOAD_ERR_NO_TMP_DIR
1455                  # (min_php: 4.3.10, 5.0.3) case UPLOAD_ERR_NO_TMP_DIR:
1456                      // Missing a temporary folder.
1457                      $failedFiles[$lKey] = 'Temporary upload dir is missing! (upload_tmp_dir in php.ini)'; // Configuration error, no translation
1458                      // Abort upload for this file:
1459                      continue;
1460  
1461                  default:
1462                      $failedFiles[$lKey] = T_('An unknown error has occurred!').' Error code #'.$_FILES['uploadfile']['error'][$lKey];
1463                      // Abort upload for this file:
1464                      continue;
1465              }
1466          }
1467  
1468          if( ! isset($_FILES['uploadfile']['_evo_fetched_url'][$lKey]) // skip check for fetched URLs
1469              && ! is_uploaded_file( $_FILES['uploadfile']['tmp_name'][$lKey] ) )
1470          { // Ensure that a malicious user hasn't tried to trick the script into working on files upon which it should not be working.
1471              $failedFiles[$lKey] = T_('The file does not seem to be a valid upload! It may exceed the upload_max_filesize directive in php.ini.');
1472              // Abort upload for this file:
1473              continue;
1474          }
1475  
1476          // Use new name on server if specified:
1477          $newName = !empty( $uploadfile_name[ $lKey ] ) ? $uploadfile_name[ $lKey ] : $lName;
1478          // validate file name
1479          if( $error_filename = process_filename( $newName, !$warn_invalid_filenames, true, $fm_FileRoot, $path ) )
1480          { // Not a valid file name or not an allowed extension:
1481              $failedFiles[$lKey] = $error_filename;
1482              syslog_insert( sprintf( 'The uploaded file %s has an unrecognized extension', '<b>'.$newName.'</b>' ), 'warning', 'file' );
1483              // Abort upload for this file:
1484              continue;
1485          }
1486  
1487          // Check if the uploaded file type is an image, and if is an image then try to fix the file extension based on mime type
1488          // If the mime type is a known mime type and user has right to upload files with this kind of file type,
1489          // this part of code will check if the file extension is the same as admin defined for this file type, and will fix it if it isn't the same
1490          // Note: it will also change the jpeg extensions to jpg.
1491          $uploadfile_path = $_FILES['uploadfile']['tmp_name'][$lKey];
1492          // this image_info variable will be used again to get file thumb
1493          $image_info = getimagesize($uploadfile_path);
1494          if( $image_info )
1495          { // This is an image, validate mimetype vs. extension:
1496              $image_mimetype = $image_info['mime'];
1497              $FiletypeCache = & get_FiletypeCache();
1498              // Get correct file type based on mime type
1499              $correct_Filetype = $FiletypeCache->get_by_mimetype( $image_mimetype, false, false );
1500  
1501              // Check if file type is known by us, and if it is allowed for upload.
1502              // If we don't know this file type or if it isn't allowed we don't change the extension! The current extension is allowed for sure.
1503              if( $correct_Filetype && $correct_Filetype->is_allowed() )
1504              { // A FileType with the given mime type exists in database and it is an allowed file type for current User
1505                  // The "correct" extension is a plausible one, proceed...
1506                  $correct_extension = array_shift($correct_Filetype->get_extensions());
1507                  $path_info = pathinfo($newName);
1508                  $current_extension = $path_info['extension'];
1509  
1510                  // change file extension to the correct extension, but only if the correct extension is not restricted, this is an extra security check!
1511                  if( strtolower($current_extension) != strtolower($correct_extension) && ( !in_array( $correct_extension, $force_upload_forbiddenext ) ) )
1512                  { // change the file extension to the correct extension
1513                      $old_name = $newName;
1514                      $newName = $path_info['filename'].'.'.$correct_extension;
1515                      $Messages->add( sprintf(T_('The extension of the file &laquo;%s&raquo; has been corrected. The new filename is &laquo;%s&raquo;.'), $old_name, $newName), 'warning' );
1516                  }
1517              }
1518          }
1519  
1520          // Get File object for requested target location:
1521          $oldName = strtolower( $newName );
1522          list( $newFile, $oldFile_thumb ) = check_file_exists( $fm_FileRoot, $path, $newName, $image_info );
1523          $newName = $newFile->get( 'name' );
1524  
1525          // Trigger plugin event
1526          if( $Plugins->trigger_event_first_false( 'AfterFileUpload', array(
1527                      'File' => & $newFile,
1528                      'name' => & $_FILES['uploadfile']['name'][$lKey],
1529                      'type' => & $_FILES['uploadfile']['type'][$lKey],
1530                      'tmp_name' => & $_FILES['uploadfile']['tmp_name'][$lKey],
1531                      'size' => & $_FILES['uploadfile']['size'][$lKey],
1532                  ) ) )
1533          {
1534              // Plugin returned 'false'.
1535              // Abort upload for this file:
1536              continue;
1537          }
1538  
1539          // Attempt to move the uploaded file to the requested target location:
1540          if( isset($_FILES['uploadfile']['_evo_fetched_url'][$lKey]) )
1541          { // fetched remotely
1542              if( ! rename( $_FILES['uploadfile']['tmp_name'][$lKey], $newFile->get_full_path() ) )
1543              {
1544                  $failedFiles[$lKey] = T_('An unknown error occurred when moving the uploaded file on the server.');
1545                  // Abort upload for this file:
1546                  continue;
1547              }
1548          }
1549          elseif( ! move_uploaded_file( $_FILES['uploadfile']['tmp_name'][$lKey], $newFile->get_full_path() ) )
1550          {
1551              syslog_insert( sprintf( 'File %s could not be uploaded', '<b>'.$newFile->get_name().'</b>' ), 'warning', 'file', $newFile->ID );
1552              $failedFiles[$lKey] = T_('An unknown error occurred when moving the uploaded file on the server.');
1553              // Abort upload for this file:
1554              continue;
1555          }
1556  
1557          // change to default chmod settings
1558          if( $newFile->chmod( NULL ) === false )
1559          { // add a note, this is no error!
1560              $Messages->add( sprintf( T_('Could not change permissions of &laquo;%s&raquo; to default chmod setting.'), $newFile->dget('name') ), 'note' );
1561          }
1562  
1563          // Refreshes file properties (type, size, perms...)
1564          $newFile->load_properties();
1565  
1566          if( ! empty( $oldFile_thumb ) )
1567          { // The file name was changed!
1568              if( $image_info )
1569              {
1570                  $newFile_thumb = $newFile->get_preview_thumb( 'fulltype' );
1571              }
1572              else
1573              {
1574                  $newFile_thumb = $newFile->get_size_formatted();
1575              }
1576              //$newFile_size = bytesreadable ($_FILES['uploadfile']['size'][$lKey]);
1577              $renamedMessages[$lKey]['message'] = sprintf( T_('"%s was renamed to %s. Would you like to replace %s with the new version instead?'),
1578                                                           '&laquo;'.$oldName.'&raquo;', '&laquo;'.$newName.'&raquo;', '&laquo;'.$oldName.'&raquo;' );
1579              $renamedMessages[$lKey]['oldThumb'] = $oldFile_thumb;
1580              $renamedMessages[$lKey]['newThumb'] = $newFile_thumb;
1581              $renamedFiles[$lKey]['oldName'] = $oldName;
1582              $renamedFiles[$lKey]['newName'] = $newName;
1583          }
1584  
1585          // Store extra info about the file into File Object:
1586          if( isset( $uploadfile_title[$lKey] ) )
1587          { // If a title text has been passed... (does not happen in quick upload mode)
1588              $newFile->set( 'title', trim( strip_tags($uploadfile_title[$lKey])) );
1589          }
1590          if( isset( $uploadfile_alt[$lKey] ) )
1591          { // If an alt text has been passed... (does not happen in quick upload mode)
1592              $newFile->set( 'alt', trim( strip_tags($uploadfile_alt[$lKey])) );
1593          }
1594          if( isset( $uploadfile_desc[$lKey] ) )
1595          { // If a desc text has been passed... (does not happen in quick upload mode)
1596              $newFile->set( 'desc', trim( strip_tags($uploadfile_desc[$lKey])) );
1597          }
1598  
1599          // Store File object into DB:
1600          $newFile->dbsave();
1601          syslog_insert( sprintf( 'File %s was uploaded', '<b>'.$newFile->get_name().'</b>' ), 'info', 'file', $newFile->ID );
1602          $uploadedFiles[] = $newFile;
1603      }
1604  
1605      prepare_uploaded_files( $uploadedFiles );
1606  
1607      return array( 'uploadedFiles' => $uploadedFiles, 'failedFiles' => $failedFiles, 'renamedFiles' => $renamedFiles, 'renamedMessages' => $renamedMessages );
1608  }

1638   * Prepare image file (Resize, Rotate and etc.)
1639   *
1640   * @param object File
1641   * @param string mimetype
1642   */
1643  function prepare_uploaded_image( $File, $mimetype )
1644  {

1646  
1647      $thumb_width = $Settings->get( 'fm_resize_width' );
1648      $thumb_height = $Settings->get( 'fm_resize_height' );
1649      $thumb_quality = $Settings->get( 'fm_resize_quality' );
1650

1652      if( $Settings->get( 'fm_resize_enable' ) &&
1653          $thumb_width > 0 && $thumb_height > 0 )
1654      {    // Image resizing is enabled
1655          list( $image_width, $image_height ) = explode( 'x', $File->get_image_size() );
1656          if( $image_width > $thumb_width || $image_height > $thumb_height )

1663

1668          if( empty( $err ) )

1670              list( $err, $resized_imh ) = generate_thumb( $src_imh, 'fit', $thumb_width, $thumb_height );
1671          }

1694      }
1695

1697      {    // Save resized image ( and also rotated image if this operation was done )

1703  /**
1704   * Rotate the JPEG image if EXIF Orientation tag is defined
1705   *
1706   * @param string File name (with full path)
1707   * @param resource Image resource ( result of the function imagecreatefromjpeg() ) (by reference)
1708   * @param boolean TRUE - to save the rotated image in the end of this function
1709   */
1710  function exif_orientation( $file_name, & $imh/* = null*/, $save_image = false )
1711  {
1712      global $Settings;
1713  
1714      if( !$Settings->get( 'exif_orientation' ) )
1715      { // Autorotate is disabled
1716          return;
1717      }
1718  
1719      if( ! function_exists('exif_read_data') )
1720      { // Exif extension is not loaded
1721          return;
1722      }
1723  
1724      $image_info = array();
1725      getimagesize( $file_name, $image_info );
1726      if( ! isset( $image_info['APP1'] ) || ( strpos( $image_info['APP1'], 'Exif' ) !== 0 ) )
1727      { // This file format is not an 'Exchangeable image file format' so there are no Exif data to read
1728          return;
1729      }
1730  
1731      if( ( $exif_data = exif_read_data( $file_name ) ) === false )
1732      { // Could not read Exif data
1733          return;
1734      }
1735  
1736      if( !( isset( $exif_data['Orientation'] ) && in_array( $exif_data['Orientation'], array( 3, 6, 8 ) ) ) )
1737      { // Exif Orientation tag is not defined OR we don't interested in current value
1738          return;
1739      }
1740  
1741      load_funcs( 'files/model/_image.funcs.php' );
1742  
1743      if( is_null( $imh ) )
1744      { // Create image resource from file name
1745          $imh = imagecreatefromjpeg( $file_name );
1746      }
1747  
1748      if( !$imh )
1749      { // Image resource is incorrect
1750          return;
1751      }
1752  
1753      switch( $exif_data['Orientation'] )
1754      {
1755          case 3:    // Rotate for 180 degrees
1756              $imh = @imagerotate( $imh, 180, 0 );
1757              break;
1758  
1759          case 6:    // Rotate for 90 degrees to the right
1760              $imh = @imagerotate( $imh, 270, 0 );
1761              break;
1762  
1763          case 8:    // Rotate for 90 degrees to the left
1764              $imh = @imagerotate( $imh, 90, 0 );
1765              break;
1766      }
1767  
1768      if( !$imh )
1769      {    // Image resource is incorrect
1770          return;
1771      }
1772  
1773      if( $save_image )
1774      {    // Save rotated image
1775          save_image( $imh, $file_name, 'image/jpeg' );
1776      }
1777  }

1780  /**
1781   * Check if file exists in the target location with the given name. Used during file upload.
1782   *
1783   * @param FileRoot target file Root
1784   * @param string target path
1785   * @param string file name
1786   * @param array the new file image_info
1787   * @return array contains two elements
1788   *             first elements is a new File object
1789   *             second element is the existing file thumb, or empty string, if the file doesn't exists
1790   */
1791  function check_file_exists( $fm_FileRoot, $path, $newName, $image_info = NULL )
1792  {
1793      global $filename_max_length;
1794  
1795      // Get File object for requested target location:
1796      $FileCache = & get_FileCache();
1797      $newFile = & $FileCache->get_by_root_and_path( $fm_FileRoot->type, $fm_FileRoot->in_type_ID, trailing_slash($path).$newName, true );
1798  
1799      $num_ext = 0;
1800      $oldName = $newName;
1801  
1802      $oldFile_thumb = "";
1803      while( $newFile->exists() )
1804      { // The file already exists in the target location!
1805          $num_ext++;
1806          $ext_pos = strrpos( $newName, '.');
1807          if( $num_ext == 1 )
1808          {
1809              if( $image_info == NULL )
1810              {
1811                  $image_info = getimagesize( $newFile->get_full_path() );
1812              }
1813              $newName = substr_replace( $newName, '-'.$num_ext.'.', $ext_pos, 1 );
1814              if( $image_info )
1815              {
1816                  $oldFile_thumb = $newFile->get_preview_thumb( 'fulltype' );
1817              }
1818              else
1819              {
1820                  $oldFile_thumb = $newFile->get_size_formatted();
1821              }
1822          }
1823          else
1824          {
1825              $replace_length = strlen( '-'.($num_ext-1) );
1826              $newName = substr_replace( $newName, '-'.$num_ext, $ext_pos-$replace_length, $replace_length );
1827          }
1828          if( strlen( $newName ) > $filename_max_length )
1829          {
1830              $newName = fix_filename_length( $newName, strrpos( $newName, '-' ) );
1831              if( $error_filename = process_filename( $newName, true ) )
1832              { // The file name is still not valid
1833                  syslog_insert( sprintf( 'Invalid file name %s has found during file exists check', '<b>'.$newName.'</b>' ), 'warning', 'file' );
1834                  debug_die( 'Invalid file name has found during file exists check: '.$error_filename );
