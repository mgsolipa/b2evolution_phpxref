 304      /**
 305       * Update field 'file_type' in the table T_files
 306       */
 307  	function update_file_types()
 308      {
 309          if( $this->extensions == $this->old_extensions && $this->mimetype == $this->old_mimetype )
 310          { // The extensions and mimetype were not changed
 311              return;
 312          }
 313  
 314          global $DB;
 315  
 316          $old_extensions = explode( ' ', $this->old_extensions );
 317  
 318          if( preg_match( '#^audio/#', $this->old_mimetype ) && $this->mimetype != $this->old_mimetype )
 319          { // Previous mimetype was 'audio' but it is changed to other, We should update file type
 320              $deleted_extensions = $old_extensions;
 321          }
 322  
 323          if( preg_match( '#^audio/#', $this->mimetype ) )
 324          { // Current file type is Audio format
 325  
 326              $new_extensions = $this->get_extensions();
 327              if( empty( $new_extensions ) && empty( $old_extensions ) )
 328              { // No extensions
 329                  return;
 330              }
 331  
 332              $deleted_extensions = array_diff( $old_extensions, $new_extensions );
 333              $added_extensions = array_diff( $new_extensions, $old_extensions );
 334          }
 335  
 336          if( ! empty( $deleted_extensions ) )
 337          { // Delete file type from old extensions
 338              $DB->query( 'UPDATE T_files
 339                      SET file_type = NULL
 340                  WHERE file_path LIKE "%.'.implode( '" OR file_path LIKE "%.', $deleted_extensions ).'"' );
 341          }
 342  
 343          if( ! empty( $added_extensions ) )
 344          { // Set audio file type for new extensions
 345              $DB->query( 'UPDATE T_files
 346                      SET file_type = "audio"
 347                  WHERE file_path LIKE "%.'.implode( '" OR file_path LIKE "%.', $added_extensions ).'"' );
 348          }
 349      }
