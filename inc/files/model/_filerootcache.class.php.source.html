  21   * This class provides info about File Roots.
  22   *
  23   * These are root directories available for media file storage, under access permission.
  24   *
  25   * @package evocore
  26   */
  27  class FileRootCache
  28  {
  29      /**
  30       * Internal cache
  31       * @var array
  32       */
  33      var $cache = array();

  36      /**
  37       * Get an array of ALL available Fileroots (not just the cached ones).
  38       *
  39       * @todo fp> it would probably make sense to refactor this as the constructor for the file roots
  40       * and initialize the whole cache at construction time
  41       *
  42       * @param string Special file root ID (Used e.g. to view file root of the special user)
  43       * @return array of FileRoots (key being the FileRoot's ID)
  44       */
  45  	static function get_available_FileRoots( $special_root_ID = NULL )
  46      {
  47          global $current_User;
  48          global $collections_Module;
  49  
  50          $r = array();
  51  
  52          $FileRootCache = & get_FileRootCache();
  53  
  54          if( ! empty( $special_root_ID ) &&
  55              ( $special_FileRoot = & $FileRootCache->get_by_ID( $special_root_ID, true ) ) &&
  56              $current_User->check_perm( 'files', 'edit_allowed', false, $special_FileRoot ) )
  57          { // Try to add special file root if current user has an access
  58              $r[ $special_FileRoot->ID ] = & $special_FileRoot;
  59          }
  60  
  61          // The user's blog (if available) is the default/first one:
  62          $user_FileRoot = & $FileRootCache->get_by_type_and_ID( 'user', $current_User->ID, true );
  63          if( $user_FileRoot )
  64          { // We got a user media dir:
  65              $r[ $user_FileRoot->ID ] = & $user_FileRoot;
  66          }
  67  
  68          if( isset($collections_Module) )
  69          {    // Blog/collection media dirs:
  70              $BlogCache = & get_BlogCache();
  71              $bloglist = $BlogCache->load_user_blogs( 'blog_media_browse', $current_User->ID );
  72              foreach( $bloglist as $blog_ID )
  73              {
  74                  if( $Root = & $FileRootCache->get_by_type_and_ID( 'collection', $blog_ID, true ) )
  75                  {
  76                      $r[ $Root->ID ] = & $Root;
  77                  }
  78              }
  79          }
  80  
  81          // Shared root:
  82          $shared_FileRoot = & $FileRootCache->get_by_type_and_ID( 'shared', 0, true );
  83          if( $shared_FileRoot )
  84          { // We got a shared dir:
  85              $r[ $shared_FileRoot->ID ] = & $shared_FileRoot;
  86          }
  87  
  88          if( isset($collections_Module) )
  89          { // Skins root:
  90              $skins_FileRoot = & $FileRootCache->get_by_type_and_ID( 'skins', 0, false );
  91              if( $skins_FileRoot )
  92              { // We got a skins dir:
  93                  $r[ $skins_FileRoot->ID ] = & $skins_FileRoot;
  94              }
  95          }
  96  
  97          // Import root:
  98          $import_FileRoot = & $FileRootCache->get_by_type_and_ID( 'import', 0, true );
  99          if( $import_FileRoot )
 100          { // We got an import dir:
 101              $r[ $import_FileRoot->ID ] = & $import_FileRoot;
 102          }
 103  
 104          return $r;
 105      }

 108      /**
 109       * Get a FileRoot (cached) by ID.
 110       *
 111       * @uses FileRootCache::get_by_type_and_ID()
 112       * @param string ID of the FileRoot (e.g. 'user_X' or 'collection_X')
 113       * @param boolean Create the directory, if it does not exist yet?
 114       * @return FileRoot|false FileRoot on success, false on failure (ads_path is false).
 115       */
 116      function & get_by_ID( $id, $create = false )
 117      {
 118          $part = explode( '_', $id );
 119          $root_type = $part[0];
 120          $root_in_type_ID = $part[1];
 121  
 122          return $this->get_by_type_and_ID( $root_type, $root_in_type_ID, $create );
 123      }
 124  
 125  
 126      /**
 127       * Get a FileRoot (cached).
 128       *
 129       * @param string Root type: 'user', 'group', 'collection' or 'absolute'
 130       * @param integer ID of the user, the group or the collection the file belongs to...
 131       * @param boolean Create the directory, if it does not exist yet?
 132       * @return FileRoot|false FileRoot on success, false on failure (ads_path is false).
 133       */
 134      function & get_by_type_and_ID( $root_type, $root_in_type_ID, $create = false )
 135      {
 136          $root_ID = FileRoot::gen_ID( $root_type, $root_in_type_ID );
 137  
 138          if( ! isset( $this->cache[$root_ID] ) )
 139          {    // Not in Cache, let's instantiate:
 140              $Root = new FileRoot( $root_type, $root_in_type_ID, $create ); // COPY (func)
 141              if( empty($Root->ads_path) ) // false
 142              {
 143                  $Root = false;
 144              }
 145              $this->cache[$root_ID] = & $Root;
 146          }
 147  
 148          return $this->cache[$root_ID];
 149      }
 150

 152      /**
 153       * Get the absolute path (FileRoot::ads_path) to a given root (with ending slash).
 154       *
 155       * @deprecated since 1.9
 156       * @param boolean Create the directory, if it does not exist yet?
 157       * @return string|boolean
 158       */
 159  	function get_root_dir( $root_type, $root_in_type_ID, $create = false )
 160      {
 161          if( $tmp_FileRoot = & $this->get_by_type_and_ID( $root_type, $root_in_type_ID, $create ) )
 162          { // Return  path only when file root is detected
 163              return $tmp_FileRoot->ads_path;
 164          }
 165          else
 166          { // Impossible to get path by root type and ID
 167              return false;
 168          }
 169      }

 173       * Clear the cache
 174       */
 175  	function clear()
 176      {
 177          $this->cache = array();
 178      }
 179  }
