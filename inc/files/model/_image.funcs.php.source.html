  44   * Crop dimensions to fit into a constrained size, while preserving aspect ratio.
  45   *
  46   * @param integer source width
  47   * @param integer source height
  48   * @param integer constrained width
  49   * @param integer constrained height
  50   * @param string align: center, top
  51   * @return array ( x, y, width, height )
  52   */
  53  function crop_to_constraint( $src_width, $src_height, $max_width, $max_height, $align = 'center' )
  54  {
  55      $src_ratio = $src_width / $src_height;
  56      $max_ratio = $max_width / $max_height;
  57      if( $max_ratio <= $src_ratio )
  58      {
  59          $y = 0;
  60          $x = ($src_width - ($src_width * ($max_ratio/$src_ratio))) / 2;
  61          $src_width = $src_width * ($max_ratio/$src_ratio);
  62      }
  63      else
  64      {
  65          $x = 0;
  66          if( $align == 'top' )
  67          {    // top - 15%
  68              $y = ( $src_height - $src_width ) * 0.15;
  69          }
  70          else
  71          {    // center
  72              $y = ($src_height - $src_width/$max_ratio) / 2;
  73          }
  74          $src_height = $src_width/$max_ratio;
  75      }
  76  
  77      return array( $x, $y, $src_width, $src_height );
  78  }

  82   * Scale dimensions to fit into a constrained size, while preserving aspect ratio.
  83   *
  84   * @param integer source width
  85   * @param integer source height
  86   * @param integer constrained width (might be NULL/0 to use source width)
  87   * @param integer constrained height (might be NULL/0 to use source height)
  88   * @return array (width, height)
  89   */
  90  function scale_to_constraint( $src_width, $src_height, $max_width, $max_height )
  91  {
  92      if( $max_width == 0 )
  93          $max_width = $src_width;
  94      if( $max_height == 0 )
  95          $max_height = $src_height;
  96  
  97      $src_ratio = $src_width / $src_height;
  98      if( $max_width / $max_height <= $src_ratio )
  99      {
 100          $width = $max_width;
 101          $height = (int)round( $max_width / $src_ratio );
 102      }
 103      else
 104      {
 105          $width = (int)round( $max_height * $src_ratio );
 106          $height = $max_height;
 107      }
 108  
 109      return array( $width, $height );
 110  }

 225  function save_image( $imh, $path, $mimetype, $quality = 90, $chmod = NULL )
 226  {
 227      $err = NULL;
 228  
 229      switch( $mimetype )
 230      {
 231          case 'image/jpeg':
 232              if( imagesx( $imh ) > 32 || imagesy( $imh ) > 32 )
 233              { // Enable interlacing
 234                  imageinterlace( $imh, 1 );
 235              }
 236              $r = @imagejpeg( $imh, $path, $quality );
 237              break;
 238  
 239          case 'image/gif':
 240              $r = @imagegif( $imh, $path );
 241              break;
 242  
 243          case 'image/png':
 244              $r = @imagepng( $imh, $path );
 245              break;
 246  
 247          default:
 248              // Unrecognized mime type
 249              $err = '!Unsupported format '.$mimetype.' (save_image)';
 250              break;
 251      }
 252  
 253      // Catch any errors by image* functions:
 254      if( ! $r )
 255      {
 256          // TODO: dh> This might become a generic function, since it's useful. Something similar is used in DB, too.
 257          if( isset($php_errormsg) )
 258              $err = '!'.$php_errormsg;
 259          elseif( function_exists('error_get_last') ) // PHP 5.2
 260          {
 261              $err = error_get_last();
 262              $err = '!'.$err['message'];
 263          }
 264  
 265          if( ! isset($err) )
 266              $err = '!Unknown error in save_image().';
 267      }
 268  
 269      if( empty( $err ) )
 270      {
 271          // Make sure the file has the default permissions we want:
 272          if( $chmod === NULL )
 273          {
 274              global $Settings;
 275              $chmod = $Settings->get('fm_default_chmod_file');
 276          }
 277          chmod( $path, octdec( $chmod ) );
 278      }
 279  
 280      return $err;
 281  }

 321  /**
 322   * Check and fix thumbnail sizes depending on original image sizes
 323   * It is useful to fix thumbnail sizes for images less than requested thumbnail
 324   *
 325   * @param string Thumbnail type ('crop'|'crop-top'|'fit')
 326   * @param integer Thumbnail width (by reference)
 327   * @param integer Thumbnail height (by reference)
 328   * @param integer Original image width
 329   * @param integer Original image height
 330   * @return boolean TRUE if no need to resample
 331   */
 332  function check_thumbnail_sizes( $thumb_type, & $thumb_width, & $thumb_height, $src_width, $src_height )
 333  {
 334      if( empty( $src_width ) || empty( $src_height ) )
 335      { // We don't know how this case can happen but it has been reported, so division by zero:
 336          return true;
 337      }
 338  
 339      if( $src_width <= $thumb_width && $src_height <= $thumb_height )
 340      { // If original image sizes are less than thumbnail sizes
 341          if( $thumb_type == 'fit' )
 342          { // There is no need to resample, use original!
 343              return true;
 344          }
 345          else
 346          { // crop & crop-top
 347              $src_ratio = number_format( $src_width / $src_height, 4, '.', '' );
 348              $thumb_ratio = number_format( $thumb_width / $thumb_height, 4, '.', '' );
 349              if( $src_ratio == $thumb_ratio )
 350              { // If original image has the same ratio then no need to resample
 351                  return true;
 352              }
 353              else
 354              { // Set new thumb sizes depending on thumbnail ratio
 355                  if( $thumb_ratio == 1 )
 356                  { // Use min size for square size
 357                      $thumb_width = ( $src_width > $src_height ) ? $src_height : $src_width;
 358                      $thumb_height = $thumb_width;
 359                  }
 360                  elseif( $thumb_ratio > $src_ratio )
 361                  { // If thumbnail ratio > original image ratio
 362                      $thumb_width = $src_width;
 363                      $thumb_height = round( $src_width / $thumb_ratio );
 364                  }
 365                  else//if( $thumb_ratio < $src_ratio )
 366                  { // If thumbnail ratio < original image ratio
 367                      $thumb_width = round( $src_height * $thumb_ratio );
 368                      $thumb_height = $src_height;
 369                  }
 370              }
 371          }
 372      }
 373  
 374      return false;
 375  }

 378  /**
 379   * Generate a thumbnail
 380   *
 381   * @param resource Image resource
 382   * @param string Thumbnail type ('crop'|'crop-top'|'fit')
 383   * @param int Thumbnail width
 384   * @param int Thumbnail height
 385   * @param int Thumbnail percent of blur effect (0 - No blur, 1% - Max blur effect, 99% - Min blur effect)
 386   * @param integer Ratio size, can be 1, 2 and etc.
 387   * @return array short error code + dest image handler
 388   */
 389  function generate_thumb( $src_imh, $thumb_type, $thumb_width, $thumb_height, $thumb_percent_blur = 0, $size_x = 1 )
 390  {
 391      $src_width = imagesx( $src_imh );
 392      $src_height = imagesy( $src_imh );
 393  
 394      $size_x = intval( $size_x );
 395      if( $size_x > 1 )
 396      { // Use the expanded size
 397          $thumb_width = $thumb_width * $size_x;
 398          $thumb_height = $thumb_height * $size_x;
 399      }
 400  
 401      if( check_thumbnail_sizes( $thumb_type, $thumb_width, $thumb_height, $src_width, $src_height ) )
 402      { // There is no need to resample, use original!
 403          return array( NULL, $src_imh );
 404      }
 405  
 406      switch( $thumb_type )
 407      {
 408          case 'crop':
 409          case 'crop-top':
 410              $align = $thumb_type == 'crop-top' ? 'top' : 'center';
 411              list( $src_x, $src_y, $src_width, $src_height) = crop_to_constraint( $src_width, $src_height, $thumb_width, $thumb_height, $align );
 412              $dest_width = $thumb_width;
 413              $dest_height = $thumb_height;
 414              break;
 415  
 416          case 'fit':
 417          default:
 418              list( $dest_width, $dest_height ) = scale_to_constraint( $src_width, $src_height, $thumb_width, $thumb_height );
 419              $src_x = $src_y = 0;
 420      }
 421  
 422      // pre_dump( $src_x, $src_y, $dest_width, $dest_height, $src_width, $src_height );
 423  
 424      // Create a transparent image:
 425      $dest_imh = imagecreatetruecolor( $dest_width, $dest_height );
 426      imagealphablending($dest_imh, true);
 427      imagefill($dest_imh, 0, 0, imagecolortransparent($dest_imh, imagecolorallocatealpha($dest_imh, 0, 0, 0, 127)));
 428      imagesavealpha($dest_imh, true);
 429  
 430      if( ! imagecopyresampled( $dest_imh, $src_imh, 0, 0, $src_x, $src_y, $dest_width, $dest_height, $src_width, $src_height ) )
 431      {
 432          return array( '!GD-library internal error (resample)', $dest_imh );
 433      }
 434  
 435      if( $thumb_percent_blur > 0 )
 436      {    // Apply blur effect
 437          $dest_imh = pixelblur( $dest_imh, $dest_width, $dest_height, $thumb_percent_blur );
 438      }
 439  
 440      // TODO: imageinterlace();
 441  
 442      return array( NULL, $dest_imh );
 443  }

 446  /**
 447   * Apply blur effect
 448   *
 449   * @param resource Image resource
 450   * @param int Source width
 451   * @param int Source height
 452   * @param int Percent of blur effect (0 - No blur, 1% - Max blur effect, 99% - Min blur effect)
 453   * @return resource Image resource
 454   */
 455  function pixelblur( $image_source, $width_source, $height_source, $percent_blur )
 456  {
 457      if( $percent_blur < 1 && $percent_blur > 99 )
 458      {    // Don't spend a time for processing of blur effect with bad percent request
 459          return $image_source;
 460      }
 461  
 462      $width_resized = ceil( $width_source * $percent_blur / 100 );
 463      $height_resized = ceil( $height_source * $percent_blur / 100 );
 464  
 465      $image_resized = imagecreatetruecolor( $width_resized, $height_resized );
 466      // Reduce image size by given percent
 467      imagecopyresampled( $image_resized, $image_source, 0, 0, 0, 0, $width_resized, $height_resized, $width_source, $height_source );
 468      // Apply blur effect from GD library
 469    if( function_exists('imagefilter') )
 470    {
 471      imagefilter( $image_resized, IMG_FILTER_GAUSSIAN_BLUR );
 472    }
 473      // Expand image to the source size
 474      imagecopyresampled( $image_source, $image_resized, 0, 0, 0, 0, $width_source, $height_source, $width_resized, $height_resized );
 475  
 476      return $image_source;
 477  }

 592  /**
 593   * Provide imagerotate for undefined cases
 594   *
 595   * Rotate an image with a given angle
 596   * @param resource Image: An image resource, returned by one of the image creation functions, such as imagecreatetruecolor().
 597   * @param integer Angle: Rotation angle, in degrees. The rotation angle is interpreted as the number of degrees to rotate the image anticlockwise.
 598   * @param string Bgd_color: Specifies the color of the uncovered zone after the rotation
 599   * @param integer Ignore_transparent: If set and non-zero, transparent colors are ignored (otherwise kept).
 600   * @return resource Returns an image resource for the rotated image, or FALSE on failure.
 601   */
 602  if( !function_exists( 'imagerotate' ) )
 603  {
 604      /**
 605       * Imagerotate replacement. ignore_transparent is work for png images
 606       * Also, have some standard functions for 90, 180 and 270 degrees.
 607       */
 608  	function imagerotate( $srcImg, $angle, $bgcolor, $ignore_transparent = 0 )
 609      {
 610  		function rotateX( $x, $y, $theta )
 611          {
 612              return $x * cos( $theta ) - $y * sin( $theta );
 613          }
 614  		function rotateY( $x, $y, $theta )
 615          {
 616              return $x * sin( $theta ) + $y * cos( $theta );
 617          }
 618  
 619          $srcw = imagesx( $srcImg );
 620          $srch = imagesy( $srcImg );
 621  
 622          //Normalize angle
 623          $angle %= 360;
 624  
 625          if( $angle == 0 )
 626          {
 627              if( $ignore_transparent == 0 )
 628              {
 629                  imagesavealpha( $srcImg, true );
 630              }
 631              return $srcImg;
 632          }
 633  
 634          // Convert the angle to radians
 635          $theta = deg2rad( $angle );
 636  
 637          //Standart case of rotate
 638          if( ( abs( $angle ) == 90 ) || ( abs( $angle ) == 270) )
 639          {
 640              $width = $srch;
 641              $height = $srcw;
 642              if( ( $angle == 90 ) || ( $angle == -270 ) )
 643              {
 644                  $minX = 0;
 645                  $maxX = $width;
 646                  $minY = -$height+1;
