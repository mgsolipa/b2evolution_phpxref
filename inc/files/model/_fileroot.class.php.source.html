  20   * A FileRoot describes a directory available for media file storage, under access permission.

  22   * @package evocore
  23   */

  25  {

  60      /**
  61       * Constructor
  62       *
  63       * Will fail if non existent User or Blog is requested.
  64       * But specific access permissions on (threfore existence of) this User or Blog should have been tested before anyway.
  65       *
  66       * @param string Root type: 'user', 'group' or 'collection'
  67       * @param integer ID of the user, the group or the collection the file belongs to...
  68       * @param boolean Create the directory, if it does not exist yet?
  69       */
  70  	function FileRoot( $root_type, $root_in_type_ID, $create = true )
  71      {
  72          /**
  73           * @var User
  74           */
  75          global $current_User;
  76          global $Messages;
  77          global $Settings, $Debuglog;
  78          global $Blog;
  79  
  80          // Store type:
  81          $this->type = $root_type;
  82          // Store ID in type:
  83          $this->in_type_ID = $root_in_type_ID;
  84          // Generate unique ID:
  85          $this->ID = FileRoot::gen_ID( $root_type, $root_in_type_ID );
  86  
  87          switch( $root_type )
  88          {
  89              case 'user':
  90                  $UserCache = & get_UserCache();
  91                  if( ! $User = & $UserCache->get_by_ID( $root_in_type_ID, false, false ) )
  92                  {    // User not found
  93                      return false;
  94                  }
  95                  $this->name = $User->get( 'login' ); //.' ('. /* TRANS: short for "user" */ T_('u').')';
  96                  $this->ads_path = $User->get_media_dir( $create );
  97                  $this->ads_url = $User->get_media_url();
  98                  return;
  99  
 100              case 'collection':
 101                  $BlogCache = & get_BlogCache();
 102                  if( ! $Blog = & $BlogCache->get_by_ID( $root_in_type_ID, false, false ) )
 103                  {    // Blog not found
 104                      return false;
 105                  }
 106                  $this->name = $Blog->get( 'shortname' ); //.' ('. /* TRANS: short for "blog" */ T_('b').')';
 107                  $this->ads_path = $Blog->get_media_dir( $create );
 108                  $this->ads_url = $Blog->get_media_url();
 109                  return;
 110  
 111              case 'shared':
 112                  // fp> TODO: handle multiple shared directories
 113                  global $media_path, $media_url;
 114                  $rds_shared_subdir = 'shared/global/';
 115                  $ads_shared_dir = $media_path.$rds_shared_subdir;
 116  
 117                  if( ! $Settings->get( 'fm_enable_roots_shared' ) )
 118                  { // Shared dir is disabled:
 119                      $Debuglog->add( 'Attempt to access shared dir, but this feature is globally disabled', 'files' );
 120                  }
 121                  /* Try to create shared directory if it doesn't exist.
 122                   * Note: mkdir_r() already checks if the dir to create exists.
 123                   */
 124                  elseif( ! mkdir_r( $ads_shared_dir ) )
 125                  {
 126                      // Only display error on an admin page:
 127                      if( is_admin_page() )
 128                      {
 129                          $Messages->add( sprintf( T_('The directory &laquo;%s&raquo; could not be created.'), $rds_shared_subdir ).get_manual_link('directory_creation_error'), 'error' );
 130                      }
 131                  }
 132                  else
 133                  {
 134                      $this->name = T_('Shared');
 135                      $this->ads_path = $ads_shared_dir;
 136                      if( isset($Blog) )
 137                      {    // (for now) Let's make shared files appear as being part of the currently displayed blog:
 138                          $this->ads_url = $Blog->get_local_media_url().'shared/global/';
 139                      }
 140                      else
 141                      {
 142                          $this->ads_url = $media_url.'shared/global/';
 143                      }
 144                  }
 145                  return;
 146  
 147              case 'skins':
 148                  // fp> some stuff here should go out of here... but I don't know where to put it yet. I'll see after the Skin refactoring.
 149                  if( ! $Settings->get( 'fm_enable_roots_skins' ) )
 150                  { // Skins root is disabled:
 151                      $Debuglog->add( 'Attempt to access skins dir, but this feature is globally disabled', 'files' );
 152                  }
 153                  elseif( empty( $current_User ) || ( ! $current_User->check_perm( 'templates' ) ) )
 154                  { // No perm to access templates:
 155                      $Debuglog->add( 'Attempt to access skins dir, but no permission', 'files' );
 156                  }
 157                  else
 158                  {
 159                      global $skins_path, $skins_url;
 160                      $this->name = T_('Skins');
 161                      $this->ads_path = $skins_path;
 162                      if( isset($Blog) )
 163                      { // (for now) Let's make skin files appear as being part of the currently displayed blog:
 164                          $this->ads_url = $Blog->get_local_skins_url();
 165                      }
 166                      else
 167                      {
 168                          $this->ads_url = $skins_url;
 169                      }
 170                  }
 171                  return;
 172  
 173              case 'import':
 174                  // Import dir
 175                  global $media_path, $media_url;
 176                  $rds_import_subdir = 'import/';
 177                  $ads_import_dir = $media_path.$rds_import_subdir;
 178                  if( ! mkdir_r( $ads_import_dir ) )
 179                  {
 180                      if( is_admin_page() )
 181                      { // Only display error on an admin page:
 182                          $Messages->add( sprintf( T_('The directory &laquo;%s&raquo; could not be created.'), $rds_import_subdir ).get_manual_link('directory_creation_error'), 'error' );
 183                      }
 184                  }
 185                  else
 186                  {
 187                      $this->name = T_('Import');
 188                      $this->ads_path = $media_path.$rds_import_subdir;
 189                      $this->ads_url = $media_url.$rds_import_subdir;
 190                  }
 191                  return;
 192          }
 193  
 194          debug_die( "Invalid root type" );
 195      }

 199      {

 201          {
 202              case 'user':
 203                  return NT_('User roots');

 205              case 'collection':

 214       * Generate ID for File Root

 217       * @param integer Root ID

 220  	static function gen_ID( $root_type, $root_in_type_ID )
 221      {
 222          if( !is_number( $root_in_type_ID ) )
 223          {
 224              debug_die( "Invalid root type ID" );
 225          }
 226  
 227          switch( $root_type )
 228          {
 229              case 'user':
 230              case 'shared':
 231              case 'collection':
 232              case 'skins':
 233              case 'import':
 234                  return $root_type.'_'.$root_in_type_ID;
 235          }
 236  
 237          debug_die( "Invalid root type" );
 238      }
