  27  $SQL = new SQL();
  28  $SQL->SELECT( 'f1.*,
  29      SUM( IFNULL( lvot_like, 0 ) ) as total_like,
  30      SUM( IFNULL( lvot_inappropriate, 0 ) ) as total_inappropriate,
  31      SUM( IFNULL( lvot_spam, 0 ) ) as total_spam,
  32      ( SELECT COUNT( file_ID ) FROM T_files AS f2 WHERE f1.file_hash = f2.file_hash ) - 1 AS total_duplicates' );
  33  $SQL->FROM( 'T_links__vote' );
  34  $SQL->FROM_add( 'INNER JOIN T_links ON link_ID = lvot_link_ID' );
  35  $SQL->FROM_add( 'INNER JOIN T_files AS f1 ON link_file_ID = file_ID' );
  36  $SQL->GROUP_BY( 'link_file_ID' );
  37  $SQL->ORDER_BY( '*, total_spam DESC, total_inappropriate DESC' );
  38  
  39  // Set filters condition to SQL queries
  40  if( $min_inappropriate_votes <= 1 && $min_spam_votes <= 1 && ( !( $min_inappropriate_votes && $min_spam_votes ) ) )
  41  { // We must show all votes or where is at least one spam vote or where is at least one inappropriate ( one filter must be 0 and none of them > 1 )
  42      if( $min_inappropriate_votes )
  43      { // Min inappropriate filter is set to 1 but min spam is 0
  44          $sql_where = 'lvot_inappropriate = 1';
  45      }
  46      elseif( $min_spam_votes )
  47      { // Min spam filter is set to 1 but min inappropriate is 0
  48          $sql_where = 'lvot_spam = 1';
  49      }
  50      else
  51      { // We have to show all files which has any kind of spam vote
  52          $sql_where = '( lvot_inappropriate = 1 OR lvot_spam = 1 )';
  53      }
  54      // Set the main query where condition
  55      $SQL->WHERE_and( $sql_where );
  56      // Create count result query
  57      $count_SQL = new SQL();
  58      $count_SQL->SELECT( 'COUNT( DISTINCT( link_file_ID ) )' );
  59      $count_SQL->FROM( 'T_links__vote' );
  60      $count_SQL->FROM_add( 'INNER JOIN T_links ON link_ID = lvot_link_ID' );
  61      $count_SQL->WHERE_and( $sql_where );
  62      // count the number of filtered result
  63      $filtered_num_results = $DB->get_var( $count_SQL->get() );
  64  }
  65  else
  66  { // check to fit at least one of the minimum requirements
  67      // Set the main query having condition
  68      $SQL->HAVING( '( total_inappropriate >= '.$DB->quote( $min_inappropriate_votes ).' ) AND ( total_spam >= '.$DB->quote( $min_spam_votes ).' )' );
  69      // Create count result query
  70      $count_SQL = new SQL();
  71      $count_SQL->SELECT( 'link_file_ID' );
  72      $count_SQL->FROM( 'T_links__vote' );
  73      $count_SQL->FROM_add( 'INNER JOIN T_links ON link_ID = lvot_link_ID' );
  74      $count_SQL->GROUP_BY( 'link_file_ID' );
  75      $count_SQL->HAVING( '( SUM( lvot_inappropriate ) >= '.$DB->quote( $min_inappropriate_votes ).' ) AND ( SUM( lvot_spam ) >= '.$DB->quote( $min_spam_votes ).' )' );
  76      // count the number of filtered result ( we need subquery because we can't count all when we have group by )
  77      $filtered_num_results = $DB->get_var( "SELECT COUNT(*) FROM (". $count_SQL->get() ." )  AS TotalSelected " );
