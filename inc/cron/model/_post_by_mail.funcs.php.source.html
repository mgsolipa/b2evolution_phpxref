  39  /**
  40   * Connect to an IMAP or POP mail server
  41   *
  42   * @param boolean TRUE if script is executed by cron
  43   * @return resource $mbox
  44   */
  45  function pbm_connect( $cron = false )
  46  {
  47      if( ! extension_loaded( 'imap' ) )
  48      {    // Exit here if imap extension is not loaded:
  49          pbm_msg( '<b class="red">'.T_( 'IMAP extension is NOT loaded! ').'</b>', $cron );
  50          return false;
  51      }
  52  
  53      global $Settings;
  54  
  55      $host = $Settings->get('eblog_server_host').':'.$Settings->get('eblog_server_port');
  56      $mailserver = '{'.$host;
  57  
  58      pbm_msg( sprintf( T_('Connecting and authenticating to mail server %s'), '<b>'.$host.'</b>' ), $cron );
  59  
  60      switch( $Settings->get('eblog_encrypt') )
  61      {
  62          case 'ssl':
  63              $mailserver .= '/ssl';
  64              break;
  65  
  66          case 'tls':
  67              $mailserver .= '/tls';
  68              break;
  69  
  70          case 'none':
  71          default:
  72              $mailserver .= '/notls';
  73              break;
  74      }
  75  
  76      switch( $Settings->get('eblog_method') )
  77      {
  78          case 'pop3':
  79          case 'pop3a':
  80              $mailserver .= '/pop3';
  81              break;
  82  
  83          case 'imap':
  84          default:
  85              // imap needs no additional options
  86              break;
  87      }
  88  
  89      if( $Settings->get('eblog_novalidatecert') )
  90      {
  91          $mailserver .= '/novalidate-cert';
  92      }
  93  
  94      $mailserver .= '}INBOX';
  95  
  96      // Connect to mail server (one retry)
  97      $mbox = @imap_open( $mailserver, $Settings->get('eblog_username'), $Settings->get('eblog_password'), NULL, 1 );
  98  
  99      if( is_null(@get_resource_type($mbox)) )
 100      {    // Not a resource
 101          $error = imap_errors();
 102          if( is_array($error) )
 103          {
 104              $error = implode( "<br />\n", $error );
 105          }
 106  
 107          pbm_msg( sprintf( /* TRANS: %s is the error message */ T_('Connection failed: %s'), $error ), $cron );
 108          return false;
 109      }
 110      pbm_msg( '<b class="green">'.T_('Successfully connected!').'</b>', $cron );
 111  
 112      @imap_errors();
 113  
 114      return $mbox;
 115  }

 559          pbm_msg( T_('Message posting successful'), $cron );
 560  
 561          // Delete temporary directory
 562          rmdir_r( $tmpDirMIME );
 563  
 564          if( ! $test_mode_on && $Settings->get('eblog_delete_emails') )
 565          {
 566              pbm_msg( sprintf( T_('Marking message for deletion from inbox: %s'), $index ), $cron );
 567              imap_delete( $mbox, $index );
 568              ++$del_cntr;
 569          }
 570      }
 571  
 572      // Expunge messages marked for deletion
 573      imap_expunge($mbox);
 574  
 575      return true;

 578  /**
 579   * Process Header information like subject and date of a mail.
 580   *
 581   * @param array $header header as set by mime_parser_class::Analyze()
 582   * @param string message subject by reference
 583   * @param string message date by reference
 584   * @param boolean TRUE if script is executed by cron
 585   * @return bool true if valid subject prefix is detected
 586   */
 587  function pbm_process_header( $header, & $subject, & $post_date, $cron = false )
 588  {
 589      global $Settings;
 590  
 591      $subject = $header['Subject'];
 592      $ddate = $header['Date'];
 593  
 594      $prefix = $Settings->get( 'eblog_subject_prefix' );
 595      pbm_msg( T_('Subject').': '.$subject, $cron );
 596  
 597      if( utf8_substr($subject, 0, utf8_strlen($prefix)) !== $prefix )
 598      {
 599          pbm_msg( sprintf( T_('Subject prefix is not "%s", skip this email'), $prefix ), $cron );
 600          return false;
 601      }
 602  
 603      $subject = utf8_substr($subject, utf8_strlen($prefix));
 604  
 605      // Parse Date
 606      if( !preg_match('#^(.{3}, )?(\d{2}) (.{3}) (\d{4}) (\d{2}):(\d{2}):(\d{2})#', $ddate, $match) )
 607      {
 608          $ddate_U = @strtotime($ddate);
 609          if( empty($ddate_U) || strlen($ddate_U) < 2 )
 610          {
 611              pbm_msg( sprintf( T_('Could not parse date header "%s"'), $ddate ), $cron );
 612              return false;
 613          }
 614      }
 615  
 616      if( empty($ddate_U) )
 617      {
 618          $dmonths = array(
 619              'Jan' => 1,
 620              'Feb' => 2,
 621              'Mar' => 3,
 622              'Apr' => 4,
 623              'May' => 5,
 624              'Jun' => 6,
 625              'Jul' => 7,
 626              'Aug' => 8,
 627              'Sep' => 9,
 628              'Oct' => 10,
 629              'Nov' => 11,
 630              'Dec' => 12,
 631          );
 632  
 633          $ddate_H = $match[5];
 634          $ddate_i = $match[6];
 635          $ddate_s = $match[7];
 636  
 637          if( ! isset( $dmonths[$match[3]] ) )
 638          {
 639              pbm_msg( T_('Invalid month name in message date string.'), $cron );
 640              return false;
 641          }
 642          $ddate_m = $dmonths[$match[3]];
 643          $ddate_d = $match[2];
 644          $ddate_Y = $match[4];
 645  
 646          $ddate_U = mktime( $ddate_H, $ddate_i, $ddate_s, $ddate_m, $ddate_d, $ddate_Y );
 647      }
 648  
 649      $post_date = date( 'Y-m-d H:i:s', $ddate_U );
 650  
 651      return true;
 652  }

 655  /**
 656   * process attachments by saving into media directory and optionally creating image tag in post
 657   *
 658   * @param string message content that is optionally manipulated by adding image tags (by reference)
 659   * @param  array $mailAttachments array containing path to attachment files
 660   * @param  string $mediadir path to media directory of blog as seen by file system
 661   * @param  string $media_url url to media directory as seen by user
 662   * @param  bool $add_img_tags should img tags be added to the post (instead of linking through the file manager)
 663   * @param  string $type defines attachment type: 'attach' or 'related'
 664   * @param boolean TRUE if script is executed by cron
 665   */
 666  function pbm_process_attachments( & $content, $mailAttachments, $mediadir, $media_url, $add_img_tags = true, $type = 'attach', $cron = false )
 667  {
 668      global $Settings, $pbm_item_files, $filename_max_length;
 669  
 670      pbm_msg( '<h4>'.T_('Processing attachments').'</h4>', $cron );
 671  
 672      foreach( $mailAttachments as $attachment )
 673      {
 674          if( isset($attachment['FileName']) )
 675          {
 676              $filename = trim( utf8_strtolower($attachment['FileName']) );
 677          }
 678          else
 679          {    // Related attachments may not have file name, we'll generate one below
 680              $filename = '';
 681          }
 682  
 683          if( $filename == '' )
 684          {
 685              $filename = 'upload_'.uniqid().'.'.$attachment['SubType'];
 686              pbm_msg( sprintf( T_('Attachment without name. Using "%s".'), htmlspecialchars( $filename ) ), $cron );
 687          }
 688  
 689          // Check valid filename/extension: (includes check for locked filenames)
 690          if( $error_filename = process_filename( $filename, true ) )
 691          {
 692              pbm_msg( T_('Invalid filename').': '.$error_filename, $cron );
 693              syslog_insert( sprintf( 'The posted by mail file %s has an unrecognized extension', '<b>'.$filename.'</b>' ), 'warning', 'file' );
 694              continue;
 695          }
 696  
 697          // If file exists count up a number
 698          $cnt = 0;
 699          $prename = substr( $filename, 0, strrpos( $filename, '.' ) ).'-';
 700          $sufname = strrchr( $filename, '.' );
 701          $error_in_filename = false;
 702  
 703          while( file_exists( $mediadir.$filename ) )
 704          {
 705              $filename = $prename.$cnt.$sufname;
 706              if( strlen( $filename ) > $filename_max_length )
 707              { // This is a special case, when the filename is longer then the maximum allowed
 708                  // Cut as many characters as required before the counter on the file name
 709                  $filename = fix_filename_length( $filename, strlen( $prename ) - 1 );
 710                  if( $error_in_filename = process_filename( $filename, true ) )
 711                  { // The file name is not valid, this is an unexpected situation, because the file name was already validated before
 712                      pbm_msg( T_('Invalid filename').': '.$error_filename, $cron );
 713                      syslog_insert( sprintf( 'The posted by mail file %s has an unrecognized extension', '<b>'.$filename.'</b>' ), 'warning', 'file' );
 714                      break;
 715                  }
 716              }
 717              ++$cnt;
 718          }
 719          if( $error_in_filename )
 720          { // Don't create file with invalid file name
 721              continue;
 722          }
 723          pbm_msg( sprintf( T_('New file name is %s'), '<b>'.$filename.'</b>' ), $cron );
 724  
 725          $imginfo = NULL;
 726          if( ! $Settings->get('eblog_test_mode') )
 727          {
 728              pbm_msg( sprintf( T_('Saving file to: %s'), htmlspecialchars( $mediadir.$filename ) ), $cron );
 729              if( !copy( $attachment['DataFile'], $mediadir.$filename ) )
 730              {
 731                  pbm_msg( sprintf( T_('Unable to copy uploaded file to %s'), htmlspecialchars( $mediadir.$filename ) ), $cron );
 732                  continue;
 733              }
 734  
 735              // chmod uploaded file:
 736              $chmod = $Settings->get('fm_default_chmod_file');
 737              @chmod( $mediadir.$filename, octdec( $chmod ) );
 738  
 739              $imginfo = @getimagesize($mediadir.$filename);
 740              pbm_msg( sprintf( T_('Is this an image?: %s'), ( is_array( $imginfo ) ? 'yes' : 'no' ) ), $cron );
 741          }
 742  
 743          if( $type == 'attach' )
 744          {
 745              $content .= "\n";
 746              if( is_array($imginfo) && $add_img_tags )
 747              {
 748                  $content .= '<img src="'.$media_url.$filename.'" '.$imginfo[3].' />';
 749              }
 750              else
 751              {
 752                  pbm_msg( sprintf( T_('The file %s will be attached to the post later, after we save the post in the database.'), '<b>'.$filename.'</b>' ), $cron );
 753                  $pbm_item_files[] = $filename;
 754              }
 755              $content .= "\n";
 756          }
 757          elseif( !empty($attachment['ContentID']) )
 758          {    // Replace relative "cid:xxxxx" URIs with absolute URLs to media files
 759              $content = str_replace( 'cid:'.$attachment['ContentID'], $media_url.$filename, $content );
 760          }
 761      }
 762  }

 799   * Extract <auth> tag with login and password
 800   *
 801   * @param string Message body (by reference)
 802   * @return array login and password
 803   */
 804  function pbm_get_auth_tag( & $content )
 805  {
 806      if( preg_match( '~<(auth)>(.+?)</\\1>~is', $content, $match ) )
 807      {
 808          // tblue> splitting only into 2 parts allows colons in the user PW
 809          // Note: login and password cannot include '<' !

 821  /**
 822   * Prepare html message
 823   *
 824   * @param string Message
 825   * @param boolean TRUE if script is executed by cron
 826   * @return string Content
 827   */
 828  function pbm_prepare_html_message( $message, $cron = false )
 829  {
 830      pbm_msg( sprintf( T_('Message body (original): %s'), '<pre style="font-size:10px">'.htmlspecialchars( $message ).'</pre>' ), $cron );
 831  
 832      $marker = 0;
 833      if( preg_match( '~<body[^>]*>(.*?)</body>~is', $message, $result ) )
 834      {    // First see if we can get contents of <body> tag
 835          $content = $result[1];
 836          $marker = 1;
 837      }
 838      elseif( preg_match( '~<html[^>]*>(.*?)</html>~is', $message, $result ) )
 839      {    // <body> was not found, use <html> contents and delete <head> section from it
 840          $content = preg_replace( '~<head[^>]*>(.*?)</head>~is', '', $result[1] );
 841          $marker = 1;
 842      }
 843  
 844      if( empty($marker) )
 845      {    // None of the above methods worked, just use the original message body
 846          $content = $message;
 847      }
 848  
 849      // First fix different line-endings (dos, mac, unix), remove double newlines
 850      $content = str_replace( array("\r", "\n\n"), "\n", trim($content) );
 851  
 852      // Decode 'category', 'title' and 'auth' tags
 853      $content = preg_replace( '~&lt;(/)?(category|title|auth)&gt;~i', '<\\1\\2>', $content );
 854  
 855      if( ($auth = pbm_get_auth_tag($content)) === false )
 856      {    // No 'auth' tag provided, exit
 857          pbm_msg( sprintf( T_('&lt;auth&gt; tag not found! Please add username and password in message body in format %s.'),
 858                      '"&lt;auth&gt;username:password&lt;/auth&gt;"' ), $cron );
 859          return false;
 860      }
 861  
 862      // Balance tags
 863      $content = balance_tags($content);
 864  
 865      // Remove markup that cause validator errors
 866      $patterns = array(
 867          '~ moz-do-not-send="true"~',            // Thunderbird inline image with absolute "src"
 868          '~ class="moz-signature" cols="\d+"~',    // Thunderbird signature in HTML message
 869          '~ goomoji="[^"]+"~',                    // Gmail smilies
 870      );
 871      $content = preg_replace( $patterns, '', $content );
 872  
 873      pbm_msg( sprintf( T_('Message body (processed): %s'), '<pre style="font-size:10px">'.htmlspecialchars( $content ).'</pre>' ), $cron );
 874  
 875      return array( $auth, $content );
 876  }

 879  function pbm_validate_user_password( $user_login, $user_pass )
 880  {
 881      $UserCache = & get_UserCache();
 882      $User = & $UserCache->get_by_login( $user_login );
 883      if( ! $User )
 884      {
 885          return false;
 886      }
 887  
 888      // First check unhashed password
 889      if( ! $User->check_password( $user_pass, false ) )
 890      {
 891          if( preg_match( '~^[a-f0-9]{32}$~i', $user_pass ) )
 892          {    // This is a hashed password, see if it's valid
 893              // We check it here because some crazy user may use a real 32-chars password!
 894              if( $User->check_password( $user_pass, true ) )
 895              {    // Valid password
 896                  return $User;
 897              }
 898          }
 899          return false;
 900      }
 901      return $User;
 902  }
