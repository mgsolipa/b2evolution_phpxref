  50  /**
  51   * Connect to an IMAP or POP mail server
  52   *
  53   * @param boolean TRUE if script is executed by cron
  54   * @return resource $mbox
  55   */
  56  function dre_connect( $cron = false )
  57  {
  58      if( ! extension_loaded( 'imap' ) )
  59      {    // Exit here if imap extension is not loaded:
  60          dre_msg( '<b class="red">'.T_( 'IMAP extension is NOT loaded! ').'</b>', $cron );
  61          return false;
  62      }
  63  
  64      global $Settings;
  65  
  66      $host = $Settings->get('repath_server_host').':'.$Settings->get('repath_server_port');
  67      $mailserver = '{'.$host;
  68  
  69      dre_msg( sprintf( T_('Connecting and authenticating to mail server %s'), '<b>'.$host.'</b>' ), $cron );
  70  
  71      switch( $Settings->get('repath_encrypt') )
  72      {
  73          case 'ssl':
  74              $mailserver .= '/ssl';
  75              break;
  76  
  77          case 'tls':
  78              $mailserver .= '/tls';
  79              break;
  80  
  81          case 'none':
  82          default:
  83              $mailserver .= '/notls';
  84              break;
  85      }
  86  
  87      switch( $Settings->get('repath_method') )
  88      {
  89          case 'pop3':
  90          case 'pop3a':
  91              $mailserver .= '/pop3';
  92              break;
  93  
  94          case 'imap':
  95          default:
  96              // imap needs no additional options
  97              break;
  98      }
  99  
 100      if( $Settings->get('repath_novalidatecert') )
 101      {
 102          $mailserver .= '/novalidate-cert';
 103      }
 104  
 105      $mailserver .= '}INBOX';
 106  
 107      // Connect to mail server (one retry)
 108      $mbox = @imap_open( $mailserver, $Settings->get('repath_username'), $Settings->get('repath_password'), NULL, 1 );
 109  
 110      if( is_null(@get_resource_type($mbox)) )
 111      {    // Not a resource
 112          $error = imap_errors();
 113          if( is_array($error) )
 114          {
 115              $error = implode( "<br />\n", $error );
 116          }
 117  
 118          dre_msg( sprintf( /* TRANS: %s is the error message */ T_('Connection failed: %s'), $error ), $cron );
 119          return false;
 120      }
 121      dre_msg( '<b class="green">'.T_('Successfully connected!').'</b>', $cron );
 122  
 123      @imap_errors();
 124  
 125      return $mbox;
 126  }

 129  /**
 130   * Read messages from server and save returned emails into DB
 131   *
 132   * @param resource $mbox created by dre_connect() (by reference)
 133   * @param integer the number of messages to process
 134   * @param boolean TRUE if script is executed by cron
 135   * @return boolean true on success
 136   */
 137  function dre_process_messages( & $mbox, $limit, $cron = false )
 138  {
 139      global $Settings, $debug;
 140      global $dre_messages, $dre_emails, $email_cntr, $del_cntr, $is_cron_mode;
 141  
 142      // This may take a very long time if there are many messages; No execution time limit:
 143      set_max_execution_time(0);
 144  
 145      if( $Settings->get( 'repath_ignore_read' ) )
 146      {    // Read status info of all messages in order to know which have already been read:
 147          $msg_statuses = imap_fetch_overview( $mbox, '1:'.$limit );
 148      }
 149  
 150      $email_cntr = 0;
 151      $del_cntr = 0;
 152      for( $index = 1; $index <= $limit; $index++ )
 153      {    // Repeat for as many messages as allowed...
 154  
 155          dre_msg( '<hr /><h3>'.sprintf( T_('Processing message %s:'), '#'.$index ).'</h3>', $cron );
 156  
 157          if( $Settings->get( 'repath_ignore_read' ) )
 158          {    // Check if we can read this message or we should skip this:
 159              if( isset( $msg_statuses[ $index - 1 ] ) && $msg_statuses[ $index - 1 ]->seen == 1 )
 160              {    // Skip this message because it has already been read:
 161                  dre_msg( T_('Ignoring this message because it has aleady been read.'), $cron );
 162                  continue;
 163              }
 164              else
 165              {    // Mark this message as "Seen" in order to don't read it twice:
 166                  imap_setflag_full( $mbox, $index, '\\Seen' );
 167              }
 168          }
 169  
 170          $html_body = '';
 171          $strbody = '';
 172          $hasAttachment = false;
 173          $hasRelated = false;
 174  
 175          // Save email to a temporary file on hard drive, otherwise BIG attachments may take a lot of RAM:
 176          if( ! ($tmpMIME = tempnam( sys_get_temp_dir(), 'b2evoMail' )) )
 177          {
 178              dre_msg( T_('Could not create temporary file.'), $cron );
 179              continue;
 180          }
 181          // Save the whole body of a specific message from the mailbox:
 182          imap_savebody( $mbox, $tmpMIME, $index );
 183  
 184  // fp> TODO: soemwhere here we should skip messages that already have the "seen" flag. This should be optional but should be the default.
 185  // This will allow to keep the emails in the INBOX without reprocessing them but to easily try them again my marking them unread.
 186  
 187          // Create random temp directory for message parts:
 188          $tmpDirMIME = dre_tempdir( sys_get_temp_dir(), 'b2evo_' );
 189  
 190          // Instanciate mime_parser.php library:
 191          $mimeParser = new mime_parser_class();
 192          $mimeParser->mbox = 0;                        // Set to 0 for parsing a *single* RFC 2822 message
 193          $mimeParser->decode_headers = 1;            // Set to 1 if it is    necessary to decode message headers that may have non-ASCII    characters and use other character set encodings
 194          $mimeParser->ignore_syntax_errors = 1;    // ignore syntax errors in    malformed messages.
 195          $mimeParser->extract_addresses = 0;
 196  
 197          // Associative array to specify parameters for the messagedata parsing and decoding operation.
 198          $MIMEparameters = array(
 199                  'File' => $tmpMIME,            // Name of the file from which the message data will be read.
 200                  'SaveBody' => $tmpDirMIME,    // Save message body parts to a directory
 201                  'SkipBody' => 1,                // 1 means the information about the message body part structure is returned in $decodedMIME below but it does not return any body data.
 202              );
 203  
 204          // STEP 1: Parse and decode message data and retrieve its structure:
 205          if( !$mimeParser->Decode( $MIMEparameters, /* BY REF */ $decodedMIME ) )
 206          {    // error:
 207              dre_msg( sprintf( T_('MIME message decoding error: %s at position %d.'), $mimeParser->error, $mimeParser->error_position ), $cron );
 208              rmdir_r( $tmpDirMIME );
 209              unlink( $tmpMIME );
 210              continue;
 211          }
 212          else
 213          {    // the specified message data was parsed successfully:
 214              dre_msg( T_('MIME message decoding successful'), $cron );
 215  
 216              // STEP 2: Analyze (the first) parsed message to describe its contents:
 217              if( ! $mimeParser->Analyze( $decodedMIME[0], /* BY REF */ $parsedMIME ) )
 218              {    // error:
 219                  dre_msg( sprintf( T_('MIME message analyze error: %s'), $mimeParser->error ), $cron );
 220                  rmdir_r( $tmpDirMIME );
 221                  unlink( $tmpMIME );
 222                  continue;
 223              }
 224  
 225              // Get message $subject and $post_date from headers (by reference)
 226              if( ! dre_process_header( $parsedMIME, /* BY REF */ $subject, /* BY REF */ $post_date, $cron ) )
 227              {    // Couldn't process message headers:
 228                  rmdir_r( $tmpDirMIME );
 229                  unlink( $tmpMIME );
 230                  continue;
 231              }
 232  
 233              // TODO: handle type == "message" recursively
 234  // fp> where is type == "message" ???
 235  // yura> I don't find the type == 'message' in dump of $decodedMIME and $parsedMIME
 236  
 237              // sam2kb> For some reason imap_qprint() demages HTML text... needs more testing
 238              // yura> I replaced imap_qprint() with quoted_printable_decode() to avoid notices about invalid quoted-printable sequence
 239              // yura> imap_qprint() and quoted_printable_decode() do empty the message text, thus they were deleted.
 240  
 241              dre_msg( T_('Email Type').': '.$parsedMIME['Type'], $cron );
 242  
 243              if( $parsedMIME['Type'] == 'html' )
 244              {    // Mail is HTML:
 245                  if( $debug )
 246                  {    // Display this info only in debug mode:
 247                      dre_msg( sprintf( T_('HTML message part saved as %s'), $parsedMIME['DataFile'] ), $cron );
 248                  }
 249                  $html_body = file_get_contents( $parsedMIME['DataFile'] );
 250  
 251                  if( empty( $html_body ) )
 252                  {    // Try to get a body text from alternative parts if main html body is empty:
 253                      foreach( $parsedMIME['Alternative'] as $alternative )
 254                      {    // First try to get HTML alternative (when possible)
 255                          if( $alternative['Type'] == 'html' )
 256                          {    // HTML text
 257                              if( $debug )
 258                              {    // Display this info only in debug mode:
 259                                  dre_msg( sprintf( T_('HTML alternative message part saved as %s'), $alternative['DataFile'] ), $cron );
 260                              }
 261                              $strbody = file_get_contents( $alternative['DataFile'] );
 262                              break; // stop after first alternative
 263                          }
 264                          elseif( $alternative['Type'] == 'text' )
 265                          {    // Plain text
 266                              if( $debug )
 267                              {    // Display this info only in debug mode:
 268                                  dre_msg( sprintf( T_('Text alternative message part saved as %s'), $alternative['DataFile'] ), $cron );
 269                              }
 270                              $strbody = file_get_contents( $alternative['DataFile'] );
 271                              break; // stop after first alternative
 272                          }
 273                      }
 274                  }
 275              }
 276              elseif( $parsedMIME['Type'] == 'text' )
 277              {    // Mail is plain text:
 278                  if( $debug )
 279                  {    // Display this info only in debug mode:
 280                      dre_msg( sprintf( T_('Plain-text message part saved as %s'), $parsedMIME['DataFile'] ), $cron );
 281                  }
 282                  $strbody = file_get_contents( $parsedMIME['DataFile'] );
 283              }
 284              elseif( $parsedMIME['Type'] == 'delivery-status' )
 285              {    // Mail is delivery-status:
 286                  $strbody = $parsedMIME['Response'];
 287              }
 288  
 289  
 290              if( count($mimeParser->warnings) > 0 )
 291              { // Record potential warnings:
 292                  dre_msg( '<h4>'.sprintf( T_('%d warnings during decode:'), count( $mimeParser->warnings ) ).'</h4>', $cron );
 293                  foreach( $mimeParser->warnings as $k => $v )
 294                  {
 295                      dre_msg( sprintf( T_('Warning: %s at position %s'), $v, $k ), $cron );
 296                  }
 297              }
 298          }
 299          unlink( $tmpMIME );
 300  
 301  
 302          if( empty( $html_body ) )
 303          {    // Plain-text message
 304              dre_msg( sprintf( T_('Message type: %s'), 'TEXT' ), $cron );
 305  
 306              // Process body. First fix different line-endings (dos, mac, unix), remove double newlines
 307              $content = str_replace( array( "\r", "\n\n" ), "\n", trim( $strbody ) );
 308  
 309              dre_msg( sprintf( T_('Message body: %s'), '<pre style="font-size:10px">'.htmlspecialchars( $strbody ).'</pre>' ), $cron );
 310          }
 311          else
 312          {    // HTML message
 313              dre_msg( sprintf( T_('Message type: %s'), 'HTML' ), $cron );
 314              dre_msg( sprintf( T_('Message body (original): %s'), '<pre style="font-size:10px">'.htmlspecialchars( $html_body ).'</pre>', $cron ) );
 315  
 316              // Prepare html message body text:
 317              $content = dre_prepare_html_message( $html_body );
 318  
 319              dre_msg( sprintf( T_('Message body (processed): %s'), '<pre style="font-size:10px">'.htmlspecialchars( $content ).'</pre>', $cron ) );
 320          }
 321  
 322  
 323          dre_msg( '<b class="green">'.T_('MIME Decoding Successful').'</b>', $cron );
 324  
 325          $message_text = $content;
 326  
 327          // Remove content after terminators
 328          $content = dre_limit_by_terminators( $content );
 329  
 330          global $Messages;
 331          if( $Messages->has_errors() )
 332          {
 333              // Make it easier for user to find and correct the errors
 334              dre_msg( "\n".sprintf( T_('Processing message: %s'), $post_title ), $cron );
 335              dre_msg( $Messages->get_string( T_('Cannot post, please correct these errors:'), 'error' ), $cron );
 336  
 337              $Messages->clear();
 338              rmdir_r( $tmpDirMIME );
 339              continue;
 340          }
 341  
 342          global $dre_emails, $DB, $localtimenow;
 343  
 344          dre_msg( '<h4>'.T_('Saving the returned email in the database').'</h4>', $cron );
 345  
 346          // Get Headers from Decoded MIME Data:
 347          $email_headers = dre_get_headers( $decodedMIME );
 348  
 349          // Get data of the returned email:
 350          $email_data = dre_get_email_data( $content, $message_text, $email_headers );
 351  
 352          dre_msg( T_('Email Address').': '.$email_data['address'], $cron );
 353          dre_msg( T_('Error Type').': '.dre_decode_error_type( $email_data['errtype'] ), $cron );
 354          dre_msg( T_('Error Message').': '.$email_data['errormsg'], $cron );
 355  
 356          // Insert a returned email's data into DB
 357          if( dre_insert_returned_email( $email_data ) )
 358          {
 359              ++$email_cntr;
 360          }
 361  
 362          // Delete temporary directory:
 363          rmdir_r( $tmpDirMIME );
 364  
 365          // Mark message to be deleted:
 366          if( $Settings->get('repath_delete_emails') )
 367          {
 368              dre_msg( sprintf( T_('Marking message for deletion from inbox: %s'), $index ), $cron );
 369              imap_delete( $mbox, $index );
 370              ++$del_cntr;
 371          }
 372      }
 373  
 374      // Expunge messages marked for deletion
 375      imap_expunge( $mbox );
 376  
 377      return true;
 378  }

 437  /**
 438   * Process Header information like subject and date of a mail.
 439   *
 440   * @param array $header header as set by mime_parser_class::Analyze()
 441   * @param string message subject by reference
 442   * @param string message date by reference
 443   * @param boolean TRUE if script is executed by cron
 444   * @return bool true if valid subject prefix is detected
 445   */
 446  function dre_process_header( $header, & $subject, & $post_date, $cron = false )
 447  {
 448      global $Settings;
 449  
 450      $subject = $header['Subject'];
 451      $ddate = $header['Date'];
 452  
 453      dre_msg( T_('Subject').': '.$subject, $cron );
 454  
 455      // Check subject to match in titles to identify return path emails
 456      $subject_is_correct = false;
 457      $repath_subjects = explode( "\n", str_replace( array( '\r\n', '\n\n' ), '\n', $Settings->get( 'repath_subject' ) ) );
 458      foreach( $repath_subjects as $repath_subject )
 459      {
 460          if( strpos( $subject, $repath_subject ) !== false )
 461          {
 462              $subject_is_correct = true;
 463              break;
 464          }
 465      }
 466  
 467      if( !$subject_is_correct )
 468      { // Subject is not match to identify return email
 469          dre_msg( sprintf( T_('Subject prefix is not %s, skip this email'), '"'.implode( '", "', $repath_subjects ).'"' ), $cron );
 470          return false;
 471      }
 472  
 473      // Parse Date
 474      if( !preg_match( '#^(.{3}, )?(\d{2}) (.{3}) (\d{4}) (\d{2}):(\d{2}):(\d{2})#', $ddate, $match ) )
 475      {
 476          $ddate_U = @strtotime($ddate);
 477          if( empty($ddate_U) || strlen($ddate_U) < 2 )
 478          {
 479              dre_msg( sprintf( T_('Could not parse date header "%s"'), $ddate ), $cron );
 480              return false;
 481          }
 482      }
 483  
 484      if( empty($ddate_U) )
 485      {
 486          $dmonths = array(
 487              'Jan' => 1,
 488              'Feb' => 2,
 489              'Mar' => 3,
 490              'Apr' => 4,
 491              'May' => 5,
 492              'Jun' => 6,
 493              'Jul' => 7,
 494              'Aug' => 8,
 495              'Sep' => 9,
 496              'Oct' => 10,
 497              'Nov' => 11,
 498              'Dec' => 12,
 499          );
 500  
 501          $ddate_H = $match[5];
 502          $ddate_i = $match[6];
 503          $ddate_s = $match[7];
 504  
 505          if( ! isset( $dmonths[$match[3]] ) )
 506          {
 507              dre_msg( T_('Invalid month name in message date string.'), $cron );
 508              return false;
 509          }
 510          $ddate_m = $dmonths[$match[3]];
 511          $ddate_d = $match[2];
 512          $ddate_Y = $match[4];
 513  
 514          $ddate_U = mktime( $ddate_H, $ddate_i, $ddate_s, $ddate_m, $ddate_d, $ddate_Y );
 515      }
 516  
 517      $post_date = date( 'Y-m-d H:i:s', $ddate_U );
 518  
 519      return true;
 520  }

 523  /**
 524   * Extract emails from a message body
 525   *
 526   * @param string Message body
 527   * @param integer Max count emails
 528   * @param string Delimeter between emails
 529   * @return string Emails separated by delimeter
 530   */
 531  function dre_get_emails( $content, $max_count = 1, $delimeter = ', ' )
 532  {
 533      if( preg_match_all( '/([a-z0-9&\-_.]+?@[\w\-]+\.([\w\-\.]+\.)?[\w]+)/is', $content, $emails ) )
 534      {    // Get the returned emails
 535          global $Settings;
 536  
 537          $emails = array_unique( $emails[1] );
 538          foreach( $emails as $e => $email )
 539          {
 540              if( in_array( $email, array( $Settings->get( 'notification_sender_email' ), $Settings->get( 'notification_return_path' ) ) ) )
 541              {    // Exclude Sender email & Return path
 542                  unset( $emails[$e] );
 543              }
 544          }
 545          // Limit by max count emails
 546          $emails = array_slice( $emails, 0, $max_count );
 547  
 548          return implode( $delimeter, $emails );
 549      }
 550  
 551      return '';
 552  }

 555  /**
 556   * Prepare html message
 557   *
 558   * @param string Message
 559   * @param boolean TRUE if script is executed by cron
 560   * @return string Content
 561   */
 562  function dre_prepare_html_message( $message, $cron = false )
 563  {
 564      $marker = 0;
 565      if( preg_match( '~<body[^>]*>(.*?)</body>~is', $message, $result ) )
 566      {    // First see if we can get contents of <body> tag
 567          $content = $result[1];
 568          $marker = 1;
 569      }
 570      elseif( preg_match( '~<html[^>]*>(.*?)</html>~is', $message, $result ) )
 571      {    // <body> was not found, use <html> contents and delete <head> section from it
 572          $content = preg_replace( '~<head[^>]*>(.*?)</head>~is', '', $result[1] );
 573          $marker = 1;
 574      }
 575  
 576      if( empty($marker) )
 577      {    // None of the above methods worked, just use the original message body
 578          $content = $message;
 579      }
 580  
 581      // First fix different line-endings (dos, mac, unix), remove double newlines
 582      $content = str_replace( array( "\r", "\n\n" ), "\n", trim( $content ) );
 583  
 584      // Decode 'category', 'title' and 'auth' tags
 585      $content = preg_replace( '~&lt;(/)?(category|title|auth)&gt;~i', '<\\1\\2>', $content );
 586  
 587      // Balance tags
 588      $content = balance_tags( $content );
 589  
 590      // Remove markup that cause validator errors
 591      $patterns = array(
 592          '~ moz-do-not-send="true"~',            // Thunderbird inline image with absolute "src"
 593          '~ class="moz-signature" cols="\d+"~',    // Thunderbird signature in HTML message
 594          '~ goomoji="[^"]+"~',                    // Gmail smilies
 595      );

 626   *
 627   * @param mixed $message
 628   * @param mixed $emails
 629   * @return string Content
 630   */
 631  function dre_get_processing_content( $content, $emails )
 632  {
 633      $error_text = '';
 634  
 635      $emails = explode( ', ', $emails );
 636      if( count( $emails ) > 0 )
 637      {    // If emails exist
 638          // Get last email
 639          $email = $emails[ count( $emails ) - 1 ];
 640          if( !empty( $email ) )
 641          {    // Get error text after last email address
 642              $error_text = trim( utf8_substr( $content, utf8_strpos( $content, $email ) + utf8_strlen( $email ) ) );
 643          }
 644          if( empty( $error_text ) )
 645          {    // If error text is empty we should get all content before email OR full content if no email address in content
 646              $error_text = empty( $email ) ? $content : trim( utf8_substr( $content, 0, utf8_strpos( $content, $email ) ) );
 647          }
 648      }
 649      else
 650      {    // If no emails - get full content as error text
 651          $error_text = $content;
 652      }
 653  
 654      if( !empty( $error_text ) )
 655      {    // Replace all new line sumbols with space symbol
 656          $error_text = str_replace( array( "\r\n\r\n", "\r\n", "\n\n", "\n" ), " ", $error_text );
 657      }
 658  
 659      return $error_text;

 913  /**
 914   * Insert/Update the data of email address into DB
 915   *
 916   * @param array Data of returned email:
 917   *               'address'
 918   *               'errormsg'
 919   *               'message'
 920   *               'headers'
 921   *               'errtype'
 922   */
 923  function dre_save_email_address_data( $email_returned )
 924  {
 925      global $DB;
 926  
 927      if( empty( $email_returned['address'] ) )
 928      {    // No emails, Exit here
 929          return;
 930      }
 931  
 932      $EmailAddressCache = & get_EmailAddressCache();
 933      // Get an existing email address to update if it exist
 934      $EmailAddress = & $EmailAddressCache->get_by_name( $email_returned['address'], false );
 935      if( !$EmailAddress )
 936      {    // Insert new email address
 937          $EmailAddress = new EmailAddress();
 938          $EmailAddress->set( 'address', $email_returned['address'] );
 939      }
 940  
 941      switch( $email_returned['errtype'] )
 942      {    // Error type of the returned email:
 943          case 'P':    // Permanent error
 944              $EmailAddress->increase_counter( 'prmerror' );
 945              // Update only the adresses with NOT spammer statuses
 946              $EmailAddress->set_status( 'prmerror' );
 947              break;
 948  
 949          case 'T':    // Temporary error
 950              if( in_array( $EmailAddress->get( 'status' ), array( 'suspicious1', 'suspicious2', 'suspicious3' ) ) )
 951              { // If current status already is defined as 'suspicious1', 'suspicious2' or 'suspicious3'
 952                  if( $EmailAddress->get( 'sent_last_returnerror' ) <= 1 )
 953                  {
 954                      if( $EmailAddress->get( 'status' ) == 'suspicious1' )
 955                      {    // Increase status from suspicious1 to suspicious2
 956                          $EmailAddress->set( 'status', 'suspicious2' );
 957                      }
 958                      elseif( $EmailAddress->get( 'status' ) == 'suspicious2' )
 959                      {    // Increase status from suspicious2 to suspicious3
 960                          $EmailAddress->set( 'status', 'suspicious3' );
 961                      }
 962                  }
 963              }
 964              elseif( $EmailAddress->get( 'status' ) == 'redemption' )
 965              { // IF current status is 'redemption' we should set it as 'suspicious3'
 966                  $EmailAddress->set_status( 'suspicious3' );
 967              }
 968              else
 969              { // Update only the email addresses with level status less then Suspicious 1
 970                  $EmailAddress->set_status( 'suspicious1' );
 971              }
 972              $EmailAddress->increase_counter( 'tmperror' );
 973              break;
 974  
 975          case 'S':    // Spam suspicion
 976              $EmailAddress->increase_counter( 'spamerror' );
 977              // Update only the email addresses with 'unknown' status
 978              $EmailAddress->set_status( 'warning' );
 979              break;
 980  
 981          default:    // Other errors
 982              $EmailAddress->increase_counter( 'othererror' );
 983              // Update only the email addresses with 'unknown' status
 984              $EmailAddress->set_status( 'warning' );
 985              break;
 986      }
 987  
 988      // Insert/Update an email address
 989      $EmailAddress->dbsave();
 990  }
