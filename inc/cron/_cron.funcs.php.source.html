  18   * Log a message from cron.
  19   * @param string Message

  21   *        (if $quiet (number of "-q" params passed to cron_exec.php)

  24  function cron_log( $message, $level = 0 )
  25  {
  26      global $is_web, $quiet;

  28      if( $quiet > $level )
  29      {
  30          return;
  31      }
  32  
  33      if( $is_web )
  34      {
  35          echo '<p>'.$message.'</p>';
  36      }
  37      else

  40      }

  44  /**
  45   * Call a cron job.
  46   *
  47   * @param string Key of the job
  48   * @param string Params for the job:
  49   *               'ctsk_ID'   - task ID
  50   *               'ctsk_name' - task name
  51   * @return string Error message
  52   */
  53  function call_job( $job_key, $job_params = array() )
  54  {
  55      global $DB, $inc_path, $Plugins, $admin_url;
  56  
  57      global $result_message, $result_status, $timestop, $time_difference;
  58  
  59      $error_message = '';
  60      $result_message = NULL;
  61      $result_status = 'error';
  62  
  63      $job_ctrl = get_cron_jobs_config( 'ctrl', $job_key );
  64  
  65      if( preg_match( '~^plugin_(\d+)_(.*)$~', $job_ctrl, $match ) )
  66      { // Cron job provided by a plugin:
  67          if( ! is_object($Plugins) )
  68          {
  69              load_class( 'plugins/model/_plugins.class.php', 'Plugins' );
  70              $Plugins = new Plugins();
  71          }
  72  
  73          $Plugin = & $Plugins->get_by_ID( $match[1] );
  74          if( ! $Plugin )
  75          {
  76              $result_message = 'Plugin for controller ['.$job_ctrl.'] could not get instantiated.';
  77              cron_log( $result_message, 2 );
  78              return $result_message;
  79          }
  80  
  81          // CALL THE PLUGIN TO HANDLE THE JOB:
  82          $tmp_params = array( 'ctrl' => $match[2], 'params' => $job_params );
  83          $sub_r = $Plugins->call_method( $Plugin->ID, 'ExecCronJob', $tmp_params );
  84  
  85          $error_code = (int)$sub_r['code'];
  86          $result_message = $sub_r['message'];
  87      }
  88      else
  89      {
  90          $controller = $inc_path.$job_ctrl;
  91          if( ! is_file( $controller ) )
  92          {
  93              $result_message = 'Controller ['.$job_ctrl.'] does not exist.';
  94              cron_log( $result_message, 2 );
  95              return $result_message;
  96          }
  97  
  98          // INCLUDE THE JOB FILE AND RUN IT:
  99          $error_code = require $controller;
 100      }
 101  
 102      if( is_array( $result_message ) )
 103      {    // If result is array (we should store it as serialized data later)
 104          // array keys: 'message' - Result message
 105          //             'table_cols' - Columns names of the table to display on the Execution details of the cron job log
 106          //             'table_data' - Array
 107          $result_message_text = $result_message['message'];
 108      }
 109      else
 110      {    // Result is text string
 111          $result_message_text = $result_message;
 112      }
 113  
 114      if( $error_code != 1 )
 115      {    // We got an error
 116          $result_status = 'error';
 117          $result_message_text = '[Error code: '.$error_code.' ] '.$result_message_text;
 118          if( is_array( $result_message ) )
 119          { // If result is array
 120              $result_message['message'] = $result_message_text;
 121          }
 122          $cron_log_level = 2;
 123  
 124          $error_message = $result_message_text;
 125      }
 126      else
 127      {
 128          $result_status = 'finished';
 129          $cron_log_level = 1;
 130      }
 131  
 132      $timestop = time() + $time_difference;
 133      cron_log( 'Task finished at '.date( 'H:i:s', $timestop ).' with status: '.$result_status
 134          ."\nMessage: $result_message_text", $cron_log_level );
 135  
 136      return $error_message;
 137  }

 166  function cron_job_manual_link( $job_key )

 169  
 170      if( empty( $help_config ) )
 171      { // There was no 'help' topic defined for this job
 172          return '';

 175      if( $help_config == '#' )
 176      { // The 'help' topic is set to '#', use the default 'task-' + job_key
 177          return get_manual_link( 'task-'.$job_key );
 178      }
 179

 181      { // This cron job help topic is an url
 182          return action_icon( T_('Open relevant page in online manual'), 'manual', $help_config, T_('Manual'), 5, 1, array( 'target' => '_blank', 'style' => 'vertical-align:top' ) );

 189  /**
 190   * Get name of cron job
 191   *
 192   * @param string Job key
 193   * @param string Job name
 194   * @param string|array Job params
 195   * @return string Default value of job name of Name from DB
 196   */
 197  function cron_job_name( $job_key, $job_name = '', $job_params = '' )
 198  {
 199      if( empty( $job_name ) )
 200      { // Get default name by key
 201          $job_name = get_cron_jobs_config( 'name', $job_key );
 202      }
 203  
 204      $job_params = is_string( $job_params ) ? unserialize( $job_params ) : $job_params;
 205      if( ! empty( $job_params ) )
 206      { // Prepare job name with the specified params
 207          switch( $job_key )
 208          {
 209              case 'send-post-notifications':
 210                  // Add item title to job name
 211                  if( ! empty( $job_params['item_ID'] ) )
 212                  {
 213                      $ItemCache = & get_ItemCache();
 214                      if( $Item = $ItemCache->get_by_ID( $job_params['item_ID'], false, false ) )
 215                      {
 216                          $job_name = sprintf( $job_name, $Item->get( 'title' ) );
 217                      }
 218                  }
 219                  break;
 220  
 221              case 'send-comment-notifications':
 222                  // Add item title of the comment to job name
 223                  if( ! empty( $job_params['comment_ID'] ) )
 224                  {
 225                      $CommentCache = & get_CommentCache();
 226                      if( $Comment = & $CommentCache->get_by_ID( $job_params['comment_ID'], false, false ) )
 227                      {
 228                          if( $Item = $Comment->get_Item() )
 229                          {
 230                              $job_name = sprintf( $job_name, $Item->get( 'title' ) );
 231                          }
 232                      }
 233                  }
 234                  break;
 235          }
 236      }
 237  
 238      return $job_name;
 239  }

 243   * Detect timed out cron jobs and Send notifications
 244   *
 245   * @param array Task with error
 246   *             'name'
 247   *             'message'
 248   */
 249  function detect_timeout_cron_jobs( $error_task = NULL )
 250  {
 251      global $DB, $time_difference, $cron_timeout_delay, $admin_url;
 252  
 253      $SQL = new SQL( 'Find cron timeouts' );
 254      $SQL->SELECT( 'ctsk_ID, ctsk_name, ctsk_key' );
 255      $SQL->FROM( 'T_cron__log' );
 256      $SQL->FROM_add( 'INNER JOIN T_cron__task ON ctsk_ID = clog_ctsk_ID' );
 257      $SQL->WHERE( 'clog_status = "started"' );
 258      $SQL->WHERE_and( 'clog_realstart_datetime < '.$DB->quote( date2mysql( time() + $time_difference - $cron_timeout_delay ) ) );
 259      $SQL->GROUP_BY( 'ctsk_ID' );
 260      $timeout_tasks = $DB->get_results( $SQL->get(), OBJECT, $SQL->title );
 261  
 262      $tasks = array();
 263  
 264      if( count( $timeout_tasks ) > 0 )
 265      {
 266          $cron_jobs_names = get_cron_jobs_config( 'name' );
 267          foreach( $timeout_tasks as $timeout_task )
 268          {
 269              if( ! empty( $timeout_task->ctsk_name ) )
 270              { // Task name is defined in DB
 271                  $task_name = $timeout_task->ctsk_name;
 272              }
 273              else
 274              { // Try to get default task name by key:
 275                  $task_name = ( isset( $cron_jobs_names[ $timeout_task->ctsk_key ] ) ? $cron_jobs_names[ $timeout_task->ctsk_key ] : $timeout_task->ctsk_key );
 276              }
 277              $tasks[ $timeout_task->ctsk_ID ] = array(
 278                      'name'    => $task_name,
 279                      'message' => NT_('Cron job has timed out.'),    // Here is not a good place to translate! We don't know the language of the recipient here.
 280                  );
 281          }
 282  
 283          // Update timed out cron jobs:
 284          $DB->query( 'UPDATE T_cron__log
 285                SET clog_status = "timeout"
 286              WHERE clog_ctsk_ID IN ( '.$DB->quote( array_keys( $tasks ) ).' )', 'Mark timeouts in cron jobs.' );
 287      }
 288  
 289      if( !is_null( $error_task ) )
 290      { // Send notification with error task
 291          $tasks[ $error_task['ID'] ] = $error_task;
 292      }
 293  
 294      if( count( $tasks ) > 0 )
 295      { // Send notification email about timed out and error cron jobs to users with edit options permission
 296          $email_template_params = array(
 297                  'tasks' => $tasks,
 298              );
 299          send_admin_notification( NT_('Scheduled task error'), 'scheduled_task_error_report', $email_template_params );
 300      }
 301  }

 304  /**
 305   * Get config array with all available cron jobs
 306   *
 307   * @param string What param we should get from config: 'name', 'ctrl', 'params'
 308   * @param string Get one cron job by specified key
 309   * @return array|string Array of all cron jobs | Value of specified cron job
 310   */
 311  function get_cron_jobs_config( $get_param = '', $get_by_key = '' )
 312  {
 313      global $cron_jobs_config;
 314  
 315      if( isset( $cron_jobs_config ) )
 316      { // Get config from global vaiable to don't initialize this var twice
 317          if( !empty( $get_by_key ) )
 318          { // A specific job param(s) was requested
 319              if( empty( $get_param ) )
 320              { // Return all params
 321                  return $cron_jobs_config[ $get_by_key ];
 322              }
 323              // Return the requested job param if it is set or NULL otherwise
 324              return ( isset( $cron_jobs_config[ $get_by_key ][$get_param] ) ) ? $cron_jobs_config[ $get_by_key ][$get_param] : NULL;
 325          }
 326  
 327          if( empty( $get_param ) )
 328          { // No specific param or job key was requested, return the whole config list
 329              return $cron_jobs_config;
 330          }
 331  
 332          // Get a specific config param for all jobs
 333          $restricted_config = array();
 334          foreach( $cron_jobs_config as $job_key => $job_config )
 335          {
 336              $restricted_config[ $job_key ] = isset( $job_config[ $get_param ] ) ? $job_config[ $get_param ] : NULL;
 337          }
 338          return $restricted_config;
 339      }
 340  
 341      // This array will contain the modules and the plugins cron jobs data
 342      $cron_jobs_config = array();
 343  
 344      // Get additional jobs from different modules and Plugins:
 345      global $modules, $Plugins;
 346  
 347      foreach( $modules as $module )
 348      {
 349          $Module = & $GLOBALS[$module.'_Module'];
 350          // Note: cron jobs with the same key will ovverride previously defined cron jobs data
 351          $cron_jobs_config = array_merge( $cron_jobs_config, $Module->get_cron_jobs() );
 352      }
 353  
 354      if( !empty( $Plugins ) )
 355      {
 356          foreach( $Plugins->trigger_collect( 'GetCronJobs' ) as $plug_ID => $jobs )
 357          {
 358              if( ! is_array($jobs) )
 359              {
 360                  $Debuglog->add( sprintf('GetCronJobs() for plugin #%d did not return array. Ignoring its jobs.', $plug_ID), array('plugins', 'error') );
 361                  continue;
 362              }
 363              foreach( $jobs as $job )
 364              {
 365                  // Validate params from plugin:
 366                  if( ! isset($job['params']) )
 367                  {
 368                      $job['params'] = NULL;
 369                  }
 370                  if( ! is_array($job) || ! isset($job['ctrl'], $job['name']) )
 371                  {
 372                      $Debuglog->add( sprintf('GetCronJobs() for plugin #%d did return invalid job. Ignoring.', $plug_ID), array('plugins', 'error') );
 373                      continue;
 374                  }
 375                  if( isset($job['params']) && ! is_array($job['params']) )
 376                  {
 377                      $Debuglog->add( sprintf('GetCronJobs() for plugin #%d did return invalid job params (not an array). Ignoring.', $plug_ID), array('plugins', 'error') );
 378                      continue;
 379                  }
 380                  $job['ctrl'] = 'plugin_'.$plug_ID.'_'.$job['ctrl'];
 381  
 382                  add_cron_jobs_config( str_replace( '_', '-', $job['ctrl'] ), $job, true );
 383              }
 384          }
 385      }
 386  
 387      return get_cron_jobs_config( $get_param, $get_by_key );
 388  }

 411  /**
 412   * Build cron job names SELECT query from crong_jobs_config array
 413   *
 414   * @param string What fields to get, separated by comma
 415   * @return string SQL query
 416   */
 417  function cron_job_sql_query( $fields = 'key,name' )
 418  {
 419      global $DB;
 420  
 421      $cron_jobs_config = get_cron_jobs_config();
 422  
 423      // We need to set the collation explicitly if the current db connection charset is utf-8 in order to avoid "Illegal mix of collation" issue
 424      // Basically this is a hack which should be reviewed when the charset issues are fixed generally.
 425      // TODO: asimo>Review this temporary solution after the charset issues were fixed.
 426      $default_collation = ( $DB->connection_charset == 'utf8' ) ? ' COLLATE utf8_general_ci' : '';
 427  
 428      $name_query = '';
 429      if( !empty( $cron_jobs_config ) )
 430      {
 431          $fields = explode( ',', $fields );
 432  
 433          $name_query = 'SELECT task_'.implode( ', task_', $fields ).' FROM ('."\n";
 434          $first_task = true;
 435          foreach( $cron_jobs_config as $ctsk_key => $ctsk_data )
 436          {
 437              $field_values = '';
 438              $field_separator = '';
 439              foreach( $fields as $field )
 440              {
 441                  $field_values .= $field_separator.$DB->quote( $field == 'key' ? $ctsk_key : $ctsk_data[ $field ] ).$default_collation.' AS task_'.$field;
 442                  $field_separator = ', ';
 443              }
 444  
 445              if( $first_task )
 446              {
 447                  $name_query .= "\t".'SELECT '.$field_values."\n";
 448                  $first_task = false;
 449              }
 450              else
 451              {
 452                  $name_query .= "\t".'UNION SELECT '.$field_values."\n";
 453              }
 454          }
 455          $name_query .= ') AS inner_temp';
 456      }
 457  
 458      return $name_query;
 459  }
