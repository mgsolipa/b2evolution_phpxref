 110  function display_sess_length( $sess_ID, $sess_length )
 111  {
 112      $result = '';
 113      $second = $sess_length % 60;
 114      $sess_length = ( $sess_length - $second ) / 60;

 119  
 120      if( $day > 0 )
 121      {
 122          $result = sprintf( ( ( $day > 1 ) ? T_( '%d days' ) : T_( '%d day' ) ), $day ).' ';
 123      }
 124      if( $hour < 10 )

 130          $minute = '0'.$minute;

 137      $result .= $hour.':'.$minute.':'.$second;
 138      return stat_session_hits( $sess_ID, $result );
