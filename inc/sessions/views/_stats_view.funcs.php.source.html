 254  /**
 255   * Helper function for "Requested URI" column
 256   *
 257   * @param integer Blog ID
 258   * @param string Requested URI
 259   * @param integer Output string lenght
 260   * @param string Display
 261   * @param string Controller
 262   * @return string
 263   */
 264  function stats_format_req_URI( $hit_coll_ID, $hit_uri, $max_len = 40, $hit_disp = NULL, $hit_ctrl = NULL, $hit_action = NULL)
 265  {
 266      if( ! empty( $hit_coll_ID ) )
 267      {
 268          $BlogCache = & get_BlogCache();
 269          $tmp_Blog = & $BlogCache->get_by_ID( $hit_coll_ID );
 270          $full_url = $tmp_Blog->get_baseurl_root().$hit_uri;
 271      }
 272      else
 273      {
 274          $full_url = $hit_uri;
 275      }
 276  
 277      $int_search_uri = urldecode($hit_uri);
 278      if( ( utf8_strpos( $int_search_uri , '?s=' ) !== false )
 279       || ( utf8_strpos( $int_search_uri , '&s=' ) !== false ) )
 280      { // This is an internal search:
 281          preg_match( '~[?&]s=([^&#]*)~', $int_search_uri, $res );
 282          $hit_uri = sprintf( T_( 'Internal search: %s' ), $res[1] );
 283      }
 284      elseif( utf8_strlen($hit_uri) > $max_len )
 285      {
 286          $hit_uri = '...'.utf8_substr( $hit_uri, -$max_len );
 287      }
 288  
 289      if( $hit_disp != NULL || $hit_ctrl != NULL || $hit_action != NULL)
 290      {
 291          $hit_uri = '';
 292          if( $hit_disp != NULL )
 293          {
 294              $hit_uri .= '[disp=<a href="'.$full_url.'">'.$hit_disp.'</a>]';
 295          }
 296          if( $hit_ctrl != NULL )
 297          {
 298              $hit_uri .= ' [ctrl=<a href="'.$full_url.'">'.$hit_ctrl.'</a>]';
 299          }
 300          if( $hit_action != NULL )
 301          {
 302              $hit_uri .= ' [action=<a href="'.$full_url.'">'.$hit_action.'</a>]';
 303          }
 304  
 305          return $hit_uri;
 306      }
 307  
 308      return '<a href="'.$full_url.'">'.$hit_uri.'</a>';
 309  }

 518  /**
 519   * Get status code of IP range by IP address
 520   *
 521   * @param string IP address in format xxx.xxx.xxx.xxx
 522   * @return string Status value as it is stored in DB
 523   */
 524  function hit_iprange_status( $IP_address )
 525  {
 526      global $DB, $hit_iprange_statuses_cache;
 527  
 528      $IP_address = ip2int( $IP_address );
 529  
 530      if( ! is_array( $hit_iprange_statuses_cache ) )
 531      { // Initialize it only first time
 532          $SQL = new SQL();
 533          $SQL->SELECT( 'aipr_status AS status, aipr_IPv4start AS start, aipr_IPv4end AS end' );
 534          $SQL->FROM( 'T_antispam__iprange' );
 535          $hit_iprange_statuses_cache = $DB->get_results( $SQL->get() );
 536      }
 537  
 538      // Use this empty value for IPs without detected ranges in DB
 539      $ip_range_status = '';
 540  
 541      // Find status in the cache
 542      foreach( $hit_iprange_statuses_cache as $hit_iprange_status )
 543      {
 544          if( $IP_address >= $hit_iprange_status->start &&
 545              $IP_address <= $hit_iprange_status->end )
 546          { // IP is detected in this range
 547              $ip_range_status = $hit_iprange_status->status;
 548              break;
 549          }
 550      }
 551  
 552      return $ip_range_status;
 553  }

 556  /**
 557   * Get status title of IP range by IP address
 558   *
 559   * @param string IP address in format xxx.xxx.xxx.xxx
 560   * @return string Status title
 561   */
 562  function hit_iprange_status_title( $IP_address )
 563  {
 564      global $current_User, $admin_url;
 565  
 566      // Get status code of IP range by IP address
 567      $ip_range_status = hit_iprange_status( $IP_address );
 568  
 569      if( $ip_range_status === '' )
 570      { // No IP range for this IP address
 571          if( $current_User->check_perm( 'spamblacklist', 'edit' ) )
 572          { // Display a link to create new one if user has an access
 573              return '<a href="'.$admin_url.'?ctrl=antispam&amp;tab3=ipranges&amp;action=iprange_new&amp;ip='.$IP_address.'">'.T_('Create').'</a>';
 574          }
 575          else
 576          { // No access to create new IP range
 577              return '';
 578          }
 579      }
 580  
 581      if( $current_User->check_perm( 'spamblacklist', 'view' ) )
 582      { // Current user has access to view IP ranges
 583          global $blog;
 584          $blog_param = empty( $blog ) ? '' : '&amp;blog=1';
 585          return '<a href="'.$admin_url.'?ctrl=antispam&amp;tab=stats&amp;tab3=ipranges&amp;ip_address='.$IP_address.$blog_param.'">'.aipr_status_title( $ip_range_status ).'</a>';
 586      }
 587      else
 588      { // No view access, Display only the status
 589          return aipr_status_title( $ip_range_status );
 590      }
 591  }
