  66  function & get_GoalCache()
  67  {
  68      global $GoalCache;
  69  
  70      if( ! isset( $GoalCache ) )
  71      { // Cache doesn't exist yet:
  72          load_class( 'sessions/model/_goal.class.php', 'Goal' );
  73          $GoalCache = new DataObjectCache( 'Goal', false, 'T_track__goal', 'goal_', 'goal_ID', 'goal_key', 'goal_name' ); // COPY (FUNC)
  74      }
  75  
  76      return $GoalCache;
  77  }
