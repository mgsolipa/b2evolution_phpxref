 112      /**
 113       * Load data from Request form fields.
 114       *
 115       * @return boolean true if loaded data seems valid.
 116       */
 117  	function load_from_Request()
 118      {
 119          // Category
 120          param( 'goal_gcat_ID', 'integer', true );
 121          param_check_not_empty( 'goal_gcat_ID', T_('Please select a category.') );
 122          $this->set_from_Request( 'gcat_ID' );
 123  
 124          // Name
 125          $this->set_string_from_param( 'name', true );
 126  
 127          // Key
 128          $this->set_string_from_param( 'key', true );
 129  
 130          // Temporary Redirection URL:
 131          $this->set_string_from_param( 'temp_redir_url' );
 132  
 133          // Normal Redirection URL:
 134          param( 'goal_redir_url', 'string' );
 135          if( $this->get( 'temp_redir_url' ) != '' )
 136          { // Normal Redirection URL is required when Temporary Redirection URL is not empty
 137              param_check_not_empty( 'goal_redir_url', T_('Please enter Normal Redirection URL.') );
 138          }
 139          $this->set_from_Request( 'redir_url' );
 140  
 141          if( $this->get( 'temp_redir_url' ) != '' && $this->get( 'temp_redir_url' ) == $this->get( 'redir_url' ) )
 142          { // Compare normal and temp urls
 143              param_error( 'goal_temp_redir_url', T_( 'Temporary Redirection URL should not be equal to Normal Redirection URL' ) );
 144              param_error( 'goal_redir_url', NULL, '' );
 145          }
 146  
 147          // Temporary Start
 148          $temp_start_date = param_date( 'goal_temp_start_date', T_('Please enter a valid date.'), false );
 149          if( ! empty( $temp_start_date ) )
 150          {
 151              $temp_start_time = param( 'goal_temp_start_time', 'string' );
 152              $temp_start_time = empty( $temp_start_time ) ? '00:00:00' : param_time( 'goal_temp_start_time' );
 153              $this->set( 'temp_start_ts', form_date( $temp_start_date, $temp_start_time ) );
 154          }
 155          else
 156          {
 157              $this->set( 'temp_start_ts', NULL );
 158          }
 159  
 160          // Temporary End
 161          $temp_end_date = param_date( 'goal_temp_end_date', T_('Please enter a valid date.'), false );
 162          if( ! empty( $temp_end_date ) )
 163          {
 164              $temp_end_time = param( 'goal_temp_end_time', 'string' );
 165              $temp_end_time = empty( $temp_end_time ) ? '00:00:00' : param_time( 'goal_temp_end_time' );
 166              $this->set( 'temp_end_ts', form_date( $temp_end_date, $temp_end_time ) );
 167          }
 168          else
 169          {
 170              $this->set( 'temp_end_ts', NULL );
 171          }
 172  
 173          if( $this->get( 'temp_start_ts' ) !== NULL && $this->get( 'temp_end_ts' ) !== NULL &&
 174              strtotime( $this->get( 'temp_start_ts' ) ) >= strtotime( $this->get( 'temp_end_ts' ) ) )
 175          { // Compare Start and End dates
 176              param_error( 'goal_temp_start_date', NULL, '' );
 177              param_error( 'goal_temp_start_time', NULL, '' );
 178              param_error( 'goal_temp_end_date', NULL, '' );
 179              param_error( 'goal_temp_end_time', T_( 'Temporary Start Date/Time should not be greater than Temporary End Date/Time' ) );
 180          }
 181  
 182          // Default value:
 183          param( 'goal_default_value', 'string' );
 184          param_check_decimal( 'goal_default_value', T_('Default value must be a number.') );
 185          $this->set_from_Request( 'default_value', 'goal_default_value', true  );
 186  
 187          // Notes
 188          param( 'goal_notes', 'text' );
 189          $this->set_from_Request( 'notes', 'goal_notes' );
 190  
 191          return ! param_errors_detected();
 192      }

 241      /**
 242       * Get redirection URL that is active now
 243       *
 244       * @param array Params
 245       * @return string $this->redir_url or $this->temp_redir_url
 246       */
 247  	function get_active_url( $params = array() )
 248      {
 249          $params = array_merge( array(
 250              'before_temp' => '',
 251              'after_temp'  => '',
 252              ), $params );
 253  
 254          if( empty( $this->temp_redir_url ) )
 255          { // Use normal redirection URL when temporary redirection URL is empty
 256              return $this->redir_url;
 257          }
 258          else
 259          { // Check if we can use temporary redirection URL
 260              global $localtimenow;
 261  
 262              if( ( empty( $this->temp_start_ts ) || $this->temp_start_ts <= $localtimenow ) &&
 263                  ( empty( $this->temp_end_ts ) || $this->temp_end_ts >= $localtimenow ) )
 264              { // Use temporary redirection URL now
 265                  return $params['before_temp'].$this->temp_redir_url.$params['after_temp'];
 266              }
 267              else
 268              { // Temporary redirection URL is out date, Use normal URL now
 269                  return $this->redir_url;
 270              }
 271          }
 272      }

 275      /**
 276       * Record goal hit
 277       *
 278       * @param string Extra params, Use '#' to get params from the $_SERVER['QUERY_STRING']
 279       */
 280  	function record_hit( $extra_params = '#' )
 281      {
 282          global $DB, $Hit;
 283  
 284          if( $extra_params == '#' )
 285          { // Use standard extra params from query string
 286              if( isset( $_SERVER['QUERY_STRING'] ) )
 287              { // Set additional params in goal hit
 288                  $extra_params = '&'.$_SERVER['QUERY_STRING'].'&';
 289                  $extra_params = preg_replace( '/&key=[^&]+(&)?/i', '$1', $extra_params );
 290                  $extra_params = trim( $extra_params, '&' );
 291              }
 292              if( empty( $extra_params ) )
 293              { // No extra params
 294                  $extra_params = NULL;
 295              }
 296          }
 297  
 298          // We need to log the HIT now! Because we need the hit ID!
 299          $Hit->log();
 300  
 301          // Insert a goal hit:
 302          $DB->query( 'INSERT INTO T_track__goalhit ( ghit_goal_ID, ghit_hit_ID, ghit_params )
 303              VALUES ( '.$this->ID.', '.$Hit->ID.', '.$DB->quote( $extra_params ).' )',
 304              'Record goal hit' );
 305      }
