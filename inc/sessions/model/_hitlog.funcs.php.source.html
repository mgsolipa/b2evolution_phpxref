 260  function refererList(
 261      $howMany = 5,
 262      $visitURL = '',
 263      $disp_blog = 0,
 264      $disp_uri = 0,
 265      $type = "'referer'",        // was: 'referer' normal refer, 'invalid', 'badchar', 'blacklist', 'rss', 'robot', 'search'
 266                                                      // new: 'search', 'blacklist', 'referer', 'direct', ('spam' but spam is not logged)
 267      $groupby = '',     // dom_name
 268      $blog_ID = '',
 269      $get_total_hits = false, // Get total number of hits (needed for percentages)
 270      $get_user_agent = false ) // Get the user agent
 271  {
 272      global $DB, $res_stats, $stats_total_hits, $ReqURI;
 273  
 274      if( strpos( $type, "'" ) !== 0 )
 275      { // no quote at position 0
 276          $type = "'".$type."'";
 277      }
 278  
 279      //if no visitURL, will show links to current page.
 280      //if url given, will show links to that page.
 281      //if url="global" will show links to all pages
 282      if (!$visitURL)
 283      {
 284          $visitURL = $ReqURI;
 285      }
 286  
 287      if( $groupby == '' )
 288      { // No grouping:
 289          $sql = 'SELECT hit_ID, UNIX_TIMESTAMP(hit_datetime) AS hit_datetime, hit_referer, dom_name';
 290      }
 291      else
 292      { // group by
 293          if( $groupby == 'baseDomain' )
 294          { // compatibility HACK!
 295              $groupby = 'dom_name';
 296          }
 297          $sql = 'SELECT COUNT(*) AS totalHits, hit_referer, dom_name';
 298      }
 299      if( $disp_blog )
 300      {
 301          $sql .= ', hit_coll_ID';
 302      }
 303      if( $disp_uri )
 304      {
 305          $sql .= ', hit_uri';
 306      }
 307      if( $get_user_agent )
 308      {
 309          $sql .= ', agnt_signature';
 310      }
 311  
 312      $sql_from_where = "
 313                FROM T_hitlog LEFT JOIN T_basedomains ON dom_ID = hit_referer_dom_ID
 314               WHERE hit_referer_type IN (".$type.")
 315                 AND hit_agent_type = 'browser'";
 316      if( !empty($blog_ID) )
 317      {
 318          $sql_from_where .= " AND hit_coll_ID = '".$blog_ID."'";
 319      }
 320      if ( $visitURL != 'global' )
 321      {
 322          $sql_from_where .= " AND hit_uri = '".$DB->escape($visitURL, 0, 250)."'";
 323      }
 324  
 325      $sql .= $sql_from_where;
 326  
 327      if( $groupby == '' )
 328      { // No grouping:
 329          $sql .= ' ORDER BY hit_ID DESC';
 330      }
 331      else
 332      { // group by
 333          $sql .= " GROUP BY ".$groupby." ORDER BY totalHits DESC";
 334      }
 335      $sql .= ' LIMIT '.$howMany;
 336  
 337      $res_stats = $DB->get_results( $sql, ARRAY_A );
 338  
 339      if( $get_total_hits )
 340      { // we need to get total hits
 341          $sql = 'SELECT COUNT(*) '.$sql_from_where;
 342          $stats_total_hits = $DB->get_var( $sql );
 343      }
 344      else
 345      { // we're not getting total hits
 346          $stats_total_hits = 1;        // just in case some tries a percentage anyway (avoid div by 0)
 347      }
 348  
 349  }

 474  function stats_search_keywords( $keyphrase, $length = 45 )
 475  {
 476      global $evo_charset;
 477  
 478      if( empty( $keyphrase ) )
 479      {
 480          return '<span class="note">['.T_('n.a.').']</span>';
 481      }
 482  
 483      // Save original string
 484      $keyphrase_orig = $keyphrase;
 485  
 486      $keyphrase = strmaxlen($keyphrase, $length, '...', 'raw');
 487  
 488      // Convert keyword encoding, some charsets are supported only in PHP 4.3.2 and later.
 489      // This fixes encoding problem for Cyrillic keywords
 490      // See http://forums.b2evolution.net/viewtopic.php?t=17431
 491      $keyphrase = htmlentities( $keyphrase, ENT_COMPAT, $evo_charset );
 492  
 493      return '<span title="'.format_to_output( $keyphrase_orig, 'htmlattr' ).'">'.$keyphrase.'</span>';
 494  }

1037  function & get_Domain_by_url( $url )
1038  {

1040      $domain_name = url_part( $url, 'host' );
1041  
1042      $DomainCache = & get_DomainCache();

1044      {
1045          if( $Domain = & $DomainCache->get_by_name( $domain_name, false, false ) )

1047              return $Domain;
1048          }

1051              break;

1053          $domain_name = $matches[1];
1054      }
1055  
1056      $Domain = NULL;
1057      return $Domain;

1082  /**
1083   * Extract keyphrases from the hitlog
1084   *
1085   * @return mixed boolean true on success, string message if the process is already running and not allowed to run
1086   */
1087  function extract_keyphrase_from_hitlogs()
1088  {
1089      global $DB, $Messages;
1090  
1091      // Set lock name based on the database name, table name and process name
1092      $lock_name = $DB->dbname.'.T_track__keyphrase.extract_keyphrase';
1093  
1094      if( $DB->get_var( 'SELECT IS_FREE_LOCK( '.$DB->quote( $lock_name ).' )' ) === '0' )
1095      { // The "exctract_keyphrase" process is already running on a different request, do not start it again
1096          // Do not translate.
1097          return 'Keyphrase extraction is already in progress in a different process. This new request would duplicate the effort so it is aborted.';
1098      }
1099  
1100      // Important: If a two or more different simultanious process will arrive to this point at the same time, only one of them will acquire the lock!
1101      // The other processes have to wait until the one who acquired the lock will release it. After that the other process will get it one by one.
1102  
1103      // Get lock with a 20 seconds timeout
1104      $DB->get_var( 'SELECT GET_LOCK( '.$DB->quote( $lock_name ).', 20 )' );
1105  
1106      // Look for unextracted keyphrases:
1107      $sql = 'SELECT MIN(h.hit_ID) as min, MAX(h.hit_ID) as max
1108                  FROM T_hitlog as h
1109                  WHERE h.hit_keyphrase IS NOT NULL
1110                      AND h.hit_keyphrase_keyp_ID IS NULL';
1111      $ids = $DB->get_row( $sql, "ARRAY_A", NULL, ' Get max/min hits ids of unextracted keyphrases' );
1112  
1113      if ( ! empty ( $ids['min'] ) && ! empty ( $ids['max'] ) )
1114      { // Extract keyphrases if needed:
1115  
1116          $sql = 'INSERT INTO T_track__keyphrase(keyp_phrase, keyp_count_refered_searches)
1117                      SELECT h.hit_keyphrase, 1
1118                      FROM T_hitlog as h
1119                      WHERE
1120                          (h.hit_ID >= '.$ids['min'].' AND h.hit_ID <= '.$ids['max'].')
1121                          AND h.hit_keyphrase IS NOT NULL
1122                          AND h.hit_keyphrase_keyp_ID IS NULL
1123                          AND h.hit_referer_type = "search"
1124                  ON DUPLICATE KEY UPDATE
1125                  T_track__keyphrase.keyp_count_refered_searches = T_track__keyphrase.keyp_count_refered_searches + 1';
1126          $DB->query( $sql, ' Insert/Update external keyphrase' );
1127  
1128          $sql = 'INSERT INTO T_track__keyphrase(keyp_phrase, keyp_count_internal_searches)
1129                      SELECT h.hit_keyphrase, 1
1130                      FROM T_hitlog as h
1131                      WHERE
1132                          (h.hit_ID >= '.$ids['min'].' AND h.hit_ID <= '.$ids['max'].')
1133                          AND h.hit_keyphrase IS NOT NULL
1134                          AND h.hit_keyphrase_keyp_ID IS NULL
1135                          AND h.hit_referer_type != "search"
1136                  ON DUPLICATE KEY UPDATE
1137                  T_track__keyphrase.keyp_count_internal_searches = T_track__keyphrase.keyp_count_internal_searches + 1';
1138          $DB->query( $sql, 'Insert/Update  internal keyphrase' );
1139  
1140          $sql = 'UPDATE T_hitlog as h, T_track__keyphrase as k
1141                  SET h.hit_keyphrase_keyp_ID = k.keyp_ID
1142                  WHERE
1143                      h.hit_keyphrase = k.keyp_phrase
1144                      AND ( h.hit_ID >= '.$ids['min'].' )
1145                      AND ( h.hit_ID <= '.$ids['max'].' )
1146                      AND ( h.hit_keyphrase_keyp_ID IS NULL )';
1147          $DB->query( $sql, 'Update hitlogs keyphrase id' );
1148      }
1149  
1150      $DB->get_var( 'SELECT RELEASE_LOCK( '.$DB->quote( $lock_name ).' )' );
1151  
1152      return true;
1153  }
